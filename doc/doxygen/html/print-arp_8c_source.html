00042 /*
00043  * Address Resolution Protocol.
00044  *
00045  * See RFC 826 for protocol description.  ARP packets are variable
00046  * in size; the arphdr structure defines the fixed-length portion.
00047  * Protocol type values are the same as those for 10 Mb/s Ethernet.
00048  * It is followed by the variable-sized fields ar_sha, arp_spa,
00049  * arp_tha and arp_tpa in that order, according to the lengths
00050  * specified.  Field names used correspond to RFC 826.
00051  */
00052 struct  arp_pkthdr {
00053         u_short ar_hrd;         /* format of hardware address */
00054 #define ARPHRD_ETHER    1       /* ethernet hardware format */
00055 #define ARPHRD_IEEE802  6       /* token-ring hardware format */
00056 #define ARPHRD_ARCNET   7       /* arcnet hardware format */
00057 #define ARPHRD_FRELAY   15      /* frame relay hardware format */
00058 #define ARPHRD_ATM2225  19      /* ATM (RFC 2225) */
00059 #define ARPHRD_STRIP    23      /* Ricochet Starmode Radio hardware format */
00060 #define ARPHRD_IEEE1394 24      /* IEEE 1394 (FireWire) hardware format */
00061         u_short ar_pro;         /* format of protocol address */
00062         u_char  ar_hln;         /* length of hardware address */
00063         u_char  ar_pln;         /* length of protocol address */
00064         u_short ar_op;          /* one of: */
00065 #define ARPOP_REQUEST   1       /* request to resolve address */
00066 #define ARPOP_REPLY     2       /* response to previous request */
00067 #define ARPOP_REVREQUEST 3      /* request protocol address given hardware */
00068 #define ARPOP_REVREPLY  4       /* response giving protocol address */
00069 #define ARPOP_INVREQUEST 8      /* request to identify peer */
00070 #define ARPOP_INVREPLY  9       /* response identifying peer */
00071 #define ARPOP_NAK       10      /* NAK - only valif for ATM ARP */
00072 
00073 /*
00074  * The remaining fields are variable in size,
00075  * according to the sizes above.
00076  */
00077 #ifdef COMMENT_ONLY
00078     u_char  ar_sha[];   /* sender hardware address */
00079     u_char  ar_spa[];   /* sender protocol address */
00080     u_char  ar_tha[];   /* target hardware address */
00081     u_char  ar_tpa[];   /* target protocol address */
00082 #endif
00083 #define ar_sha(ap)  (((const u_char *)((ap)+1))+0)
00084 #define ar_spa(ap)  (((const u_char *)((ap)+1))+  (ap)->ar_hln)
00085 #define ar_tha(ap)  (((const u_char *)((ap)+1))+  (ap)->ar_hln+(ap)->ar_pln)
00086 #define ar_tpa(ap)  (((const u_char *)((ap)+1))+2*(ap)->ar_hln+(ap)->ar_pln)
00087 };
00088 
00089 #define ARP_HDRLEN  8
00090 
00091 #define HRD(ap) EXTRACT_16BITS(&(ap)->ar_hrd)
00092 #define HRD_LEN(ap) ((ap)->ar_hln)
00093 #define PROTO_LEN(ap) ((ap)->ar_pln)
00094 #define OP(ap)  EXTRACT_16BITS(&(ap)->ar_op)
00095 #define PRO(ap) EXTRACT_16BITS(&(ap)->ar_pro)
00096 #define SHA(ap) (ar_sha(ap))
00097 #define SPA(ap) (ar_spa(ap))
00098 #define THA(ap) (ar_tha(ap))
00099 #define TPA(ap) (ar_tpa(ap))

00124 /*
00125  * ATM Address Resolution Protocol.
00126  *
00127  * See RFC 2225 for protocol description.  ATMARP packets are similar
00128  * to ARP packets, except that there are no length fields for the
00129  * protocol address - instead, there are type/length fields for
00130  * the ATM number and subaddress - and the hardware addresses consist
00131  * of an ATM number and an ATM subaddress.
00132  */
00133 struct  atmarp_pkthdr {
00134         u_short aar_hrd;        /* format of hardware address */
00135         u_short aar_pro;        /* format of protocol address */
00136         u_char  aar_shtl;       /* length of source ATM number */
00137         u_char  aar_sstl;       /* length of source ATM subaddress */
00138 #define ATMARP_IS_E164  0x40    /* bit in type/length for E.164 format */
00139 #define ATMARP_LEN_MASK 0x3F    /* length of {sub}address in type/length */
00140         u_short aar_op;         /* same as regular ARP */
00141         u_char  aar_spln;       /* length of source protocol address */
00142         u_char  aar_thtl;       /* length of target ATM number */
00143         u_char  aar_tstl;       /* length of target ATM subaddress */
00144         u_char  aar_tpln;       /* length of target protocol address */
00145 /*
00146  * The remaining fields are variable in size,
00147  * according to the sizes above.
00148  */
00149 #ifdef COMMENT_ONLY
00150     u_char  aar_sha[];  /* source ATM number */
00151     u_char  aar_ssa[];  /* source ATM subaddress */
00152     u_char  aar_spa[];  /* sender protocol address */
00153     u_char  aar_tha[];  /* target ATM number */
00154     u_char  aar_tsa[];  /* target ATM subaddress */
00155     u_char  aar_tpa[];  /* target protocol address */
00156 #endif
00157 
00158 #define ATMHRD(ap)  EXTRACT_16BITS(&(ap)->aar_hrd)
00159 #define ATMSHRD_LEN(ap) ((ap)->aar_shtl & ATMARP_LEN_MASK)
00160 #define ATMSSLN(ap) ((ap)->aar_sstl & ATMARP_LEN_MASK)
00161 #define ATMSPROTO_LEN(ap) ((ap)->aar_spln)
00162 #define ATMOP(ap)   EXTRACT_16BITS(&(ap)->aar_op)
00163 #define ATMPRO(ap)  EXTRACT_16BITS(&(ap)->aar_pro)
00164 #define ATMTHRD_LEN(ap) ((ap)->aar_thtl & ATMARP_LEN_MASK)
00165 #define ATMTSLN(ap) ((ap)->aar_tstl & ATMARP_LEN_MASK)
00166 #define ATMTPROTO_LEN(ap) ((ap)->aar_tpln)
00167 #define aar_sha(ap) ((const u_char *)((ap)+1))
00168 #define aar_ssa(ap) (aar_sha(ap) + ATMSHRD_LEN(ap))
00169 #define aar_spa(ap) (aar_ssa(ap) + ATMSSLN(ap))
00170 #define aar_tha(ap) (aar_spa(ap) + ATMSPROTO_LEN(ap))
00171 #define aar_tsa(ap) (aar_tha(ap) + ATMTHRD_LEN(ap))
00172 #define aar_tpa(ap) (aar_tsa(ap) + ATMTSLN(ap))
00173 };

00178 #define ATMTHA(ap) (aar_tha(ap))
00179 #define ATMTSA(ap) (aar_tsa(ap))
00180 #define ATMTPA(ap) (aar_tpa(ap))

00199 static void
00200 atmarp_print(netdissect_options *ndo,
00201          const u_char *bp, u_int length, u_int caplen)
00202 {
00203     const struct atmarp_pkthdr *ap;
00204     u_short pro, hrd, op;
00205 
00206     ap = (const struct atmarp_pkthdr *)bp;
00207     ND_TCHECK(*ap);
00208 
00209     hrd = ATMHRD(ap);
00210     pro = ATMPRO(ap);
00211     op = ATMOP(ap);
00212 
00213     if (!ND_TTEST2(*aar_tpa(ap), ATMTPROTO_LEN(ap))) {
00214         ND_PRINT((ndo, "[|ARP]"));
00215         ND_DEFAULTPRINT((const u_char *)ap, length);
00216         return;
00217     }
00218 
00219         if (!ndo->ndo_eflag) {
00220             ND_PRINT((ndo, "ARP, "));
00221         }
00222 
00223     if ((pro != ETHERTYPE_IP && pro != ETHERTYPE_TRAIL) ||
00224         ATMSPROTO_LEN(ap) != 4 ||
00225             ATMTPROTO_LEN(ap) != 4 ||
00226             ndo->ndo_vflag) {
00227                 ND_PRINT((ndo, "%s, %s (len %u/%u)",
00228                           tok2str(arphrd_values, "Unknown Hardware (%u)", hrd),
00229                           tok2str(ethertype_values, "Unknown Protocol (0x%04x)", pro),
00230                           ATMSPROTO_LEN(ap),
00231                           ATMTPROTO_LEN(ap)));
00232 
00233                 /* don't know know about the address formats */
00234                 if (!ndo->ndo_vflag) {
00235                     goto out;
00236                 }
00237     }
00238 
00239         /* print operation */
00240         printf("%s%s ",
00241                ndo->ndo_vflag ? ", " : "", 
00242                tok2str(arpop_values, "Unknown (%u)", op));
00243 
00244     switch (op) {
00245 
00246     case ARPOP_REQUEST:
00247         ND_PRINT((ndo, "who-has %s", ipaddr_string(ATMTPA(ap))));
00248         if (ATMTHRD_LEN(ap) != 0) {
00249             ND_PRINT((ndo, " ("));
00250             atmarp_addr_print(ndo, ATMTHA(ap), ATMTHRD_LEN(ap),
00251                 ATMTSA(ap), ATMTSLN(ap));
00252             ND_PRINT((ndo, ")"));
00253         }
00254         ND_PRINT((ndo, "tell %s", ipaddr_string(ATMSPA(ap))));
00255         break;
00256 
00257     case ARPOP_REPLY:
00258         ND_PRINT((ndo, "%s is-at ", ipaddr_string(ATMSPA(ap))));
00259         atmarp_addr_print(ndo, ATMSHA(ap), ATMSHRD_LEN(ap), ATMSSA(ap),
00260                                   ATMSSLN(ap));
00261         break;
00262 
00263     case ARPOP_INVREQUEST:
00264         ND_PRINT((ndo, "who-is "));
00265         atmarp_addr_print(ndo, ATMTHA(ap), ATMTHRD_LEN(ap), ATMTSA(ap),
00266             ATMTSLN(ap));
00267         ND_PRINT((ndo, " tell "));
00268         atmarp_addr_print(ndo, ATMSHA(ap), ATMSHRD_LEN(ap), ATMSSA(ap),
00269             ATMSSLN(ap));
00270         break;
00271 
00272     case ARPOP_INVREPLY:
00273         atmarp_addr_print(ndo, ATMSHA(ap), ATMSHRD_LEN(ap), ATMSSA(ap),
00274             ATMSSLN(ap));
00275         ND_PRINT((ndo, "at %s", ipaddr_string(ATMSPA(ap))));
00276         break;
00277 
00278     case ARPOP_NAK:
00279         ND_PRINT((ndo, "for %s", ipaddr_string(ATMSPA(ap))));
00280         break;
00281 
00282     default:
00283         ND_DEFAULTPRINT((const u_char *)ap, caplen);
00284         return;
00285     }
00286 
00287  out:
00288         ND_PRINT((ndo, ", length %u", length));
00289         return;
00290 
00291 trunc:

00306     pro = PRO(ap);

00309         
00310         /* if its ATM then call the ATM ARP printer
00311            for Frame-relay ARP most of the fields
00312            are similar to Ethernet so overload the Ethernet Printer

00315         switch(hrd) {

00320             linkaddr = LINKADDR_FRELAY;

00327         ND_PRINT((ndo, "[|ARP]"));
00328         ND_DEFAULTPRINT((const u_char *)ap, length);
00329         return;
00330     }
00331 
00332         if (!ndo->ndo_eflag) {
00333             ND_PRINT((ndo, "ARP, "));
00334         }
00335 
00336         /* print hardware type/len and proto type/len */
00337         if ((pro != ETHERTYPE_IP && pro != ETHERTYPE_TRAIL) ||
00338         PROTO_LEN(ap) != 4 ||
00339             HRD_LEN(ap) == 0 ||
00340             ndo->ndo_vflag) {
00341             ND_PRINT((ndo, "%s (len %u), %s (len %u)",
00342                       tok2str(arphrd_values, "Unknown Hardware (%u)", hrd),
00343                       HRD_LEN(ap),

00352

00355                ndo->ndo_vflag ? ", " : "",

00357 
00358     switch (op) {
00359 
00360     case ARPOP_REQUEST:
