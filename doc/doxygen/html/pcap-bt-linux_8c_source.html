00101     }
00102 
00103     dev_list->dev_num = HCI_MAX_DEV;
00104 
00105     if (ioctl(sock, HCIGETDEVLIST, (void *) dev_list) < 0)

00108             errno, strerror(errno));
00109         ret = -1;
00110         goto free;
00111     }
00112 
00113     dev_req = dev_list->dev_req;
00114     for (i = 0; i < dev_list->dev_num; i++, dev_req++) {
00115         char dev_name[20], dev_descr[30];
00116         
00117         snprintf(dev_name, 20, BT_IFACE"%d", dev_req->dev_id);
00118         snprintf(dev_descr, 30, "Bluetooth adapter number %d", i);
00119             
00120         if (pcap_add_if(&found_dev, dev_name, 0, 
00121                dev_descr, err_str) < 0)
00122         {
00123             ret = -1;
00124             break;
00125         }
00126 
00127     }
00128 
00129 free:
00130     free(dev_list);
00131 
00132 done:
00133     close(sock);
00134     return ret;

00214     memset(&flt, 0, sizeof(flt));

00223

00225     addr.hci_family = AF_BLUETOOTH;

00242         /*
00243          * Set the socket buffer size to the specified value.

00246             &handle->opt.buffer_size,
00247             sizeof(handle->opt.buffer_size)) == -1) {

00263 bt_read_linux(pcap_t *handle, int max_packets, pcap_handler callback, u_char *user)
00264 {
00265     struct cmsghdr *cmsg;
00266     struct msghdr msg;
00267     struct iovec  iv;
00268     struct pcap_pkthdr pkth;
00269     pcap_bluetooth_h4_header* bthdr;
00270 
00271     bthdr = (pcap_bluetooth_h4_header*) &handle->buffer[handle->offset];
00272     iv.iov_base = &handle->buffer[handle->offset+sizeof(pcap_bluetooth_h4_header)];
00273     iv.iov_len  = handle->snapshot;
00274     
00275     memset(&msg, 0, sizeof(msg));
00276     msg.msg_iov = &iv;
00277     msg.msg_iovlen = 1;
00278     msg.msg_control = handle->buffer;
00279     msg.msg_controllen = handle->offset;
00280 
00281     /* ignore interrupt system call error */
00282     do {
00283         pkth.caplen = recvmsg(handle->fd, &msg, 0);
00284         if (handle->break_loop)
00285         {
00286             handle->break_loop = 0;
00287             return -2;
00288         }
00289     } while ((pkth.caplen == -1) && (errno == EINTR));
00290 
00291         
00292     if (pkth.caplen < 0) {
00293         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE, "Can't receive packet %d:%s",
00294             errno, strerror(errno));
00295         return -1;
00296     }
00297 
00298     /* get direction and timestamp*/ 
00299     cmsg = CMSG_FIRSTHDR(&msg);
00300     int in=0;
00301     while (cmsg) {
00302         switch (cmsg->cmsg_type) {
00303             case HCI_CMSG_DIR:
00304                 in = *((int *) CMSG_DATA(cmsg));
00305                 break;
00306                         case HCI_CMSG_TSTAMP:
00307                 pkth.ts = *((struct timeval *) CMSG_DATA(cmsg));
00308                 break;
00309         }
00310         cmsg = CMSG_NXTHDR(&msg, cmsg);
00311     }
00312     if ((in && (handle->direction == PCAP_D_OUT)) || 
00313                 ((!in) && (handle->direction == PCAP_D_IN)))
00314         return 0;
00315 
00316     bthdr->direction = htonl(in != 0);
00317     pkth.caplen+=sizeof(pcap_bluetooth_h4_header);
00318     pkth.len = pkth.caplen;
00319     callback(user, &pkth, &handle->buffer[handle->offset]);
00320     return 1;
00321 }
