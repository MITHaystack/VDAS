00041 static u_int8_t *
00042 cpack_next_boundary(u_int8_t *buf, u_int8_t *p, size_t alignment)
00043 {
00044     size_t misalignment = (size_t)(p - buf) % alignment;
00045 
00046     if (misalignment == 0)
00047         return p;
00048 
00049     return p + (alignment - misalignment);
00050 }

00052 /* Advance to the next wordsize boundary. Return NULL if fewer than
00053  * wordsize bytes remain in the buffer after the boundary.  Otherwise,
00054  * return a pointer to the boundary.
00055  */
00056 static u_int8_t *
00057 cpack_align_and_reserve(struct cpack_state *cs, size_t wordsize)
00058 {
00059     u_int8_t *next;
00060 
00061     /* Ensure alignment. */
00062     next = cpack_next_boundary(cs->c_buf, cs->c_next, wordsize);
00063 
00064     /* Too little space for wordsize bytes? */
00065     if (next - cs->c_buf + wordsize > cs->c_len)
00066         return NULL;
00067 
00068     return next;
00069 }

00083 /* Unpack a 64-bit unsigned integer. */
00084 int
00085 cpack_uint64(struct cpack_state *cs, u_int64_t *u)
00086 {
00087     u_int8_t *next;
00088 
00089     if ((next = cpack_align_and_reserve(cs, sizeof(*u))) == NULL)
00090         return -1;
00091 
00092     *u = EXTRACT_LE_64BITS(next);
00093 
00094     /* Move pointer past the u_int64_t. */
00095     cs->c_next = next + sizeof(*u);
00096     return 0;
00097 }
