00408  * This is the OTHER way we may receive packets
00409  */
00410 CALLBACK (NdisReceiveChain (WORD macId, WORD frameSize, WORD reqHandle,
00411                             struct _RxBufDescr *rxBufDescr,
00412                             BYTE *indicate, WORD protDS))
00413 {
00414   struct _PktBuf *pktBuf;
00415   int     i;
00416 
00417   /*
00418    * For now we copy the entire packet over to a PktBuf structure. This may be
00419    * a performance hit but this routine probably isn't called very much, and
00420    * it is a lot of work to do it otherwise. Also if it is a filter protocol
00421    * packet we could end up sucking up MAC buffes.
00422    */
00423 
00424   if ((pktBuf = AllocPktBuf()) == NULL)
00425   {
00426     droppedPackets++;
00427     return (ERR_FRAME_REJECTED);
00428   }
00429   pktBuf->packetLength = 0;
00430 
00431   /* Copy the packet to the buffer
00432    */
00433   for (i = 0; i < rxBufDescr->rxDataCount; ++i)
00434   {
00435     struct _RxBufDescrRec *rxDescr = &rxBufDescr->rxBufDescrRec[i];
00436 
00437     memcpy (pktBuf->buffer + pktBuf->packetLength,
00438             rxDescr->rxDataPtr, rxDescr->rxDataLen);
00439     pktBuf->packetLength += rxDescr->rxDataLen;
00440   }
00441 
00442   EnquePktBuf (pktBuf);
00443 
00444   ARGSUSED (frameSize);
00445   ARGSUSED (reqHandle);
00446   ARGSUSED (indicate);
00447   ARGSUSED (protDS);   
00448 
00449   /* This frees up the buffer for the MAC to use
00450    */
00451   return (ERR_SUCCESS);
00452 }

00483 /*
00484  * Tell the NDIS driver to start the delivery of the packet
00485  */
00486 int NdisSendPacket (struct _PktBuf *pktBuf, int macId)
00487 { 
00488   struct _TxBufDescr txBufDescr;
00489   int     result;
00490 
00491   xmitPending++;
00492   txBufPending = pktBuf;    /* we only have 1 pending Tx at a time */
00493 
00494   txBufDescr.txImmedLen  = 0;
00495   txBufDescr.txImmedPtr  = NULL;
00496   txBufDescr.txDataCount = 1;
00497   txBufDescr.txBufDescrRec[0].txPtrType = NDIS_PTR_PHYSICAL;
00498   txBufDescr.txBufDescrRec[0].dummy     = 0;
00499   txBufDescr.txBufDescrRec[0].txDataLen = pktBuf->packetLength;
00500   txBufDescr.txBufDescrRec[0].txDataPtr = pktBuf->buffer;
00501 
00502   result = MAC_DISPATCH(handle)->transmitChain (common.moduleId,
00503                                                 pktBuf->handle,
00504                                                 &txBufDescr,
00505                                                 handle->common->moduleDS);
00506   switch (result)
00507   {
00508     case ERR_OUT_OF_RESOURCE:
00509          /* Note that this should not happen but if it does there is not
00510           * much we can do about it
00511           */
00512          printf ("ERROR: transmit queue overflowed\n");
00513          return (0);
00514 
00515     case ERR_SUCCESS:
00516          /* Everything was hunky dory and synchronous. Free up the 
00517           * packet buffer
00518           */
00519          xmitPending--;
00520          FreePktBuf (pktBuf);
00521          return (1);
00522 
00523     case ERR_REQUEST_QUEUED:
00524          /* Everything was hunky dory and asynchronous. Do nothing
00525           */
00526          return (1);
00527 
00528     default:
00529          printf ("Tx fail, code = %04X\n", result);
00530          return (0);
00531   }
00532 }
