00196 static u32 __iomem *
00197 myri10ge_mmio_ext_config(struct pci_dev *dev, int where)
00198 {
00199     u32 __iomem *ptr32 = NULL;
00200     struct pci_dev *mcp55, *ck804;
00201     unsigned long base = 0UL;
00202     static unsigned long config_phys = 0;
00203     u32 pci_id;
00204     u16 word;
00205     unsigned long dev_base;
00206 
00207     if (config_phys == 0) {
00208         if (dev->vendor != PCI_VENDOR_ID_NVIDIA)
00209             goto unknown;
00210 
00211         if (dev->device == PCI_DEVICE_ID_NVIDIA_NFORCE_CK804_PCIE) {
00212             ck804 = myri10ge_pci_get_bus_and_slot(0, 0);
00213             if (ck804) {
00214                 if (ck804->vendor == PCI_VENDOR_ID_NVIDIA &&
00215                     ck804->device == PCI_DEVICE_ID_NVIDIA_NFORCE_CK804_ROOT) {
00216                     pci_read_config_word(ck804, 0x90, &word);
00217                     base = ((unsigned long)word & 0xfff) << 28;
00218                 }
00219                 myri10ge_pci_get_dev_put(ck804);
00220             }
00221         } else if (dev->device >= PCI_DEVICE_ID_NVIDIA_NFORCE_MCP55_PCIE_374 &&
00222                dev->device <= PCI_DEVICE_ID_NVIDIA_NFORCE_MCP55_PCIE_378) {
00223             mcp55 = myri10ge_pci_get_bus_and_slot(0, 0);
00224             if (mcp55) {
00225                 if (mcp55->vendor == PCI_VENDOR_ID_NVIDIA &&
00226                     mcp55->device == PCI_DEVICE_ID_NVIDIA_NFORCE_MCP55_PCIE_369) {
00227                     pci_read_config_word(mcp55, 0x90, &word);
00228                     base = ((unsigned long)word & 0x7ffeU) << 25;
00229                 }
00230                 myri10ge_pci_get_dev_put(mcp55);
00231             }
00232         } else {
00233             goto unknown;
00234         }
00235         if (!base)
00236             return NULL;
00237         dev_base = (base 
00238                 + (unsigned long)dev->bus->number * 0x00100000UL
00239                 + (unsigned long)dev->devfn * 0x00001000UL);
00240         ptr32 = (uint32_t *) ioremap(dev_base, 4);
00241         if (!ptr32)
00242             return NULL;
00243 
00244         pci_id = *ptr32;
00245         iounmap(ptr32);
00246         if (pci_id != dev->vendor + (dev->device << 16)) {
00247             printk("Ext-conf-space at unknown address, contact help@myri.com\n");
00248             return NULL;
00249         }
00250         config_phys = base;
00251     }
00252 
00253 
00254     dev_base = (config_phys
00255             + (unsigned long)dev->bus->number * 0x00100000UL
00256             + (unsigned long)dev->devfn * 0x00001000UL);
00257     ptr32 = (u32 __iomem *) ioremap(dev_base + where, 4);
00258     return ptr32;
00259 
00260 unknown:
00261     printk("Ext-conf-space with unknown chipset %x:%x\n",
00262            dev->vendor, dev->device);
00263     printk("contact help@myri.com\n");
00264     return NULL;
