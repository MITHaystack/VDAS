00088 vrrp_print(register const u_char *bp, register u_int len, int ttl)
00089 {
00090     int version, type, auth_type;
00091     const char *type_s;
00092 
00093     TCHECK(bp[0]);
00094     version = (bp[0] & 0xf0) >> 4;
00095     type = bp[0] & 0x0f;
00096     type_s = tok2str(type2str, "unknown type (%u)", type);
00097     printf("VRRPv%u, %s", version, type_s);
00098     if (ttl != 255)
00099         printf(", (ttl %u)", ttl);
00100     if (version != 2 || type != VRRP_TYPE_ADVERTISEMENT)
00101         return;
00102     TCHECK(bp[2]);
00103     printf(", vrid %u, prio %u", bp[1], bp[2]);
00104     TCHECK(bp[5]);
00105     auth_type = bp[4];
00106     printf(", authtype %s", tok2str(auth2str, NULL, auth_type));
00107     printf(", intvl %us, length %u", bp[5],len);
00108     if (vflag) {
00109         int naddrs = bp[3];
00110         int i;
00111         char c;
00112 
00113         if (TTEST2(bp[0], len) && in_cksum((const u_short*)bp, len, 0))
00114             printf(", (bad vrrp cksum %x)",
00115                 EXTRACT_16BITS(&bp[6]));
00116         printf(", addrs");
00117         if (naddrs > 1)
00118             printf("(%d)", naddrs);
00119         printf(":");
00120         c = ' ';
00121         bp += 8;
00122         for (i = 0; i < naddrs; i++) {
00123             TCHECK(bp[3]);
00124             printf("%c%s", c, ipaddr_string(bp));
00125             c = ',';
00126             bp += 4;
00127         }
00128         if (auth_type == VRRP_AUTH_SIMPLE) { /* simple text password */
00129             TCHECK(bp[7]);
00130             printf(" auth \"");
00131             if (fn_printn(bp, 8, snapend)) {
00132                 printf("\"");
00133                 goto trunc;
00134             }
00135             printf("\"");
00136         }
00137     }
00138     return;

00141 }
