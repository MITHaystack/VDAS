00060  * The time difference in millisecond
00061  */
00062 double delta_time (struct timeval * now,
00063            struct timeval * before) {
00064   time_t delta_seconds;
00065   time_t delta_microseconds;
00066 
00067   /*
00068    * compute delta in second, 1/10's and 1/1000's second units
00069    */
00070   delta_seconds      = now -> tv_sec  - before -> tv_sec;
00071   delta_microseconds = now -> tv_usec - before -> tv_usec;
00072 
00073   if(delta_microseconds < 0) {
00074     /* manually carry a one from the seconds field */
00075     delta_microseconds += 1000000;  /* 1e6 */
00076     -- delta_seconds;
00077   }
00078   return((double)(delta_seconds * 1000) + (double)delta_microseconds/1000);
00079 }

00200     if (byte > 0) {
00201       *--cp = byte % 10 + '0';
00202       byte /= 10;

00206     *--cp = '.';

00208   } while (--n > 0);
00209 
00210   /* Convert the string to lowercase */
00211   retStr = (char*)(cp+1);
00212 
00213   return(retStr);

00294 int32_t gmt2local(time_t t) {
00295   int dt, dir;
00296   struct tm *gmt, *loc;
00297   struct tm sgmt;
00298 
00299   if (t == 0)
00300     t = time(NULL);
00301   gmt = &sgmt;
00302   *gmt = *gmtime(&t);
00303   loc = localtime(&t);
00304   dt = (loc->tm_hour - gmt->tm_hour) * 60 * 60 +
00305     (loc->tm_min - gmt->tm_min) * 60;
00306 
00307   /*
00308    * If the year or julian day is different, we span 00:00 GMT
00309    * and must add or subtract a day. Check the year first to
00310    * avoid problems when the julian day wraps.
00311    */
00312   dir = loc->tm_year - gmt->tm_year;
00313   if (dir == 0)
00314     dir = loc->tm_yday - gmt->tm_yday;
00315   dt += dir * 24 * 60 * 60;
00316 
00317   return (dt);
00318 }
