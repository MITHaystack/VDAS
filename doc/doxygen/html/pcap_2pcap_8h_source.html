00086 /*
00087  * The first record in the file contains saved values for some
00088  * of the flags used in the printout phases of tcpdump.
00089  * Many fields here are 32 bit ints so compilers won't insert unwanted
00090  * padding; these files need to be interchangeable across architectures.
00091  *
00092  * Do not change the layout of this structure, in any way (this includes
00093  * changes that only affect the length of fields in this structure).
00094  *
00095  * Also, do not change the interpretation of any of the members of this
00096  * structure, in any way (this includes using values other than
00097  * LINKTYPE_ values, as defined in "savefile.c", in the "linktype"
00098  * field).
00099  *
00100  * Instead:
00101  *
00102  *  introduce a new structure for the new format, if the layout
00103  *  of the structure changed;
00104  *
00105  *  send mail to "tcpdump-workers@lists.tcpdump.org", requesting
00106  *  a new magic number for your new capture file format, and, when
00107  *  you get the new magic number, put it in "savefile.c";
00108  *
00109  *  use that magic number for save files with the changed file
00110  *  header;
00111  *
00112  *  make the code in "savefile.c" capable of reading files with
00113  *  the old file header as well as files with the new file header
00114  *  (using the magic number to determine the header format).
00115  *
00116  * Then supply the changes as a patch at
00117  *
00118  *  http://sourceforge.net/projects/libpcap/
00119  *
00120  * so that future versions of libpcap and programs that use it (such as
00121  * tcpdump) will be able to read your new capture file format.
00122  */
00123 struct pcap_file_header {
00124     bpf_u_int32 magic;
00125     u_short version_major;
00126     u_short version_minor;
00127     bpf_int32 thiszone; /* gmt to local correction */
00128     bpf_u_int32 sigfigs;    /* accuracy of timestamps */
00129     bpf_u_int32 snaplen;    /* max length saved portion of each pkt */
00130     bpf_u_int32 linktype;   /* data link type (LINKTYPE_*) */
00131 };

00149 /*
00150  * Generic per-packet information, as supplied by libpcap.
00151  *
00152  * The time stamp can and should be a "struct timeval", regardless of
00153  * whether your system supports 32-bit tv_sec in "struct timeval",
00154  * 64-bit tv_sec in "struct timeval", or both if it supports both 32-bit
00155  * and 64-bit applications.  The on-disk format of savefiles uses 32-bit
00156  * tv_sec (and tv_usec); this structure is irrelevant to that.  32-bit
00157  * and 64-bit versions of libpcap, even if they're on the same platform,
00158  * should supply the appropriate version of "struct timeval", even if
00159  * that's not what the underlying packet capture mechanism supplies.
00160  */
00161 struct pcap_pkthdr {
00162     struct timeval ts;  /* time stamp */
00163     bpf_u_int32 caplen; /* length of portion present */
00164     bpf_u_int32 len;    /* length this packet (off wire) */
00165 };
