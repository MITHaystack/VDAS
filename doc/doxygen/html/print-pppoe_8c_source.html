00101 u_int
00102 pppoe_print(register const u_char *bp, u_int length)
00103 {
00104     u_int16_t pppoe_ver, pppoe_type, pppoe_code, pppoe_sessionid;
00105     u_int pppoe_length;
00106     const u_char *pppoe_packet, *pppoe_payload;
00107 
00108     if (length < PPPOE_HDRLEN) {
00109         (void)printf("truncated-pppoe %u", length);
00110         return (length);
00111     }
00112     length -= PPPOE_HDRLEN;
00113     pppoe_packet = bp;
00114     TCHECK2(*pppoe_packet, PPPOE_HDRLEN);
00115     pppoe_ver  = (pppoe_packet[0] & 0xF0) >> 4;
00116     pppoe_type  = (pppoe_packet[0] & 0x0F);
00117     pppoe_code = pppoe_packet[1];
00118     pppoe_sessionid = EXTRACT_16BITS(pppoe_packet + 2);
00119     pppoe_length    = EXTRACT_16BITS(pppoe_packet + 4);
00120     pppoe_payload = pppoe_packet + PPPOE_HDRLEN;
00121 
00122     if (pppoe_ver != 1) {
00123         printf(" [ver %d]",pppoe_ver);
00124     }
00125     if (pppoe_type != 1) {
00126         printf(" [type %d]",pppoe_type);
00127     }
00128 
00129     printf("PPPoE %s", tok2str(pppoecode2str, "PAD-%x", pppoe_code));
00130     if (pppoe_code == PPPOE_PADI && pppoe_length > 1484 - PPPOE_HDRLEN) {
00131         printf(" [len %u!]",pppoe_length);
00132     }
00133     if (pppoe_length > length) {
00134         printf(" [len %u > %u!]", pppoe_length, length);
00135         pppoe_length = length;
00136     }
00137     if (pppoe_sessionid) {
00138         printf(" [ses 0x%x]", pppoe_sessionid);
00139     }
00140 
00141     if (pppoe_code) {
00142         /* PPP session packets don't contain tags */
00143         u_short tag_type = 0xffff, tag_len;
00144         const u_char *p = pppoe_payload;
00145 
00146         /*
00147          * loop invariant:
00148          * p points to current tag,
00149          * tag_type is previous tag or 0xffff for first iteration
00150          */
00151         while (tag_type && p < pppoe_payload + pppoe_length) {
00152             TCHECK2(*p, 4);
00153             tag_type = EXTRACT_16BITS(p);
00154             tag_len = EXTRACT_16BITS(p + 2);
00155             p += 4;
00156             /* p points to tag_value */
00157 
00158             if (tag_len) {
00159                 unsigned isascii = 0, isgarbage = 0;
00160                 const u_char *v = p;
00161                 char tag_str[MAXTAGPRINT];
00162                 unsigned tag_str_len = 0;
00163 
00164                 /* TODO print UTF-8 decoded text */
00165                 TCHECK2(*p, tag_len);
00166                 for (v = p; v < p + tag_len && tag_str_len < MAXTAGPRINT-1; v++)
00167                     if (*v >= 32 && *v < 127) {
00168                         tag_str[tag_str_len++] = *v;
00169                         isascii++;
00170                     } else {
00171                         tag_str[tag_str_len++] = '.';
00172                         isgarbage++;
00173                     }
00174                 tag_str[tag_str_len] = 0;
00175 
00176                 if (isascii > isgarbage) {
00177                     printf(" [%s \"%*.*s\"]",
00178                            tok2str(pppoetag2str, "TAG-0x%x", tag_type),
00179                            (int)tag_str_len,
00180                            (int)tag_str_len,
00181                            tag_str);
00182                 } else {
00183                     /* Print hex, not fast to abuse printf but this doesn't get used much */
00184                     printf(" [%s 0x", tok2str(pppoetag2str, "TAG-0x%x", tag_type));
00185                     for (v=p; v<p+tag_len; v++) {
00186                         printf("%02X", *v);
00187                     }
00188                     printf("]");
00189                 }
00190                 
00191 
00192             } else
00193                 printf(" [%s]", tok2str(pppoetag2str,
00194                     "TAG-0x%x", tag_type));
00195 
00196             p += tag_len;
00197             /* p points to next tag */
00198         }
00199         return (0);
00200     } else {
00201         /* PPPoE data */
00202         printf(" ");
00203         return (PPPOE_HDRLEN + ppp_print(pppoe_payload, pppoe_length));
00204     }
00205 
00206 trunc:
00207     printf("[|pppoe]");
00208     return (PPPOE_HDRLEN);
00209 }
