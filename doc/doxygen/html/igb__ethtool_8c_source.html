00230 static int igb_set_settings(struct net_device *netdev, struct ethtool_cmd *ecmd)
00231 {
00232     struct igb_adapter *adapter = netdev_priv(netdev);
00233     struct e1000_hw *hw = &adapter->hw;
00234 
00235     /* When SoL/IDER sessions are active, autoneg/speed/duplex
00236      * cannot be changed */
00237     if (e1000_check_reset_block(hw)) {
00238         dev_err(pci_dev_to_dev(adapter->pdev), "Cannot change link "
00239             "characteristics when SoL/IDER is active.\n");
00240         return -EINVAL;
00241     }
00242 
00243     while (test_and_set_bit(__IGB_RESETTING, &adapter->state))
00244         msleep(1);
00245 
00246     if (ecmd->autoneg == AUTONEG_ENABLE) {
00247         hw->mac.autoneg = 1;
00248         hw->phy.autoneg_advertised = ecmd->advertising |
00249                          ADVERTISED_TP |
00250                          ADVERTISED_Autoneg;
00251         ecmd->advertising = hw->phy.autoneg_advertised;
00252         if (adapter->fc_autoneg)
00253             hw->fc.requested_mode = e1000_fc_default;
00254     } else {
00255         if (igb_set_spd_dplx(adapter, ecmd->speed + ecmd->duplex)) {
00256             clear_bit(__IGB_RESETTING, &adapter->state);
00257             return -EINVAL;
00258         }
00259     }
00260 
00261 #ifdef ETH_TP_MDI_X
00262     /* MDI-X =>2; MDI=>1; Invalid =>0 */
00263     if (hw->phy.media_type == e1000_media_type_copper) {
00264         switch (ecmd->eth_tp_mdix) {
00265         case ETH_TP_MDI_X:
00266             hw->phy.mdix = 2;
00267             break;
00268         case ETH_TP_MDI:
00269             hw->phy.mdix = 1;
00270             break;
00271         case ETH_TP_MDI_INVALID:
00272         default:
00273             hw->phy.mdix = 0;
00274             break;
00275         }
00276     }
00277 
00278 #endif /* ETH_TP_MDI_X */
00279     /* reset the link */
00280     if (netif_running(adapter->netdev)) {
00281         igb_down(adapter);
00282         igb_up(adapter);
00283     } else
00284         igb_reset(adapter);
00285 
00286     clear_bit(__IGB_RESETTING, &adapter->state);
00287     return 0;
00288 }

00707 static int igb_get_eeprom(struct net_device *netdev,
00708               struct ethtool_eeprom *eeprom, u8 *bytes)
00709 {
00710     struct igb_adapter *adapter = netdev_priv(netdev);
00711     struct e1000_hw *hw = &adapter->hw;
00712     u16 *eeprom_buff;
00713     int first_word, last_word;
00714     int ret_val = 0;
00715     u16 i;
00716 
00717     if (eeprom->len == 0)
00718         return -EINVAL;
00719 
00720     eeprom->magic = hw->vendor_id | (hw->device_id << 16);
00721 
00722     first_word = eeprom->offset >> 1;
00723     last_word = (eeprom->offset + eeprom->len - 1) >> 1;
00724 
00725     eeprom_buff = kmalloc(sizeof(u16) *
00726             (last_word - first_word + 1), GFP_KERNEL);
00727     if (!eeprom_buff)
00728         return -ENOMEM;
00729 
00730     if (hw->nvm.type == e1000_nvm_eeprom_spi)
00731         ret_val = e1000_read_nvm(hw, first_word,
00732                      last_word - first_word + 1,
00733                      eeprom_buff);
00734     else {
00735         for (i = 0; i < last_word - first_word + 1; i++) {
00736             ret_val = e1000_read_nvm(hw, first_word + i, 1,
00737                          &eeprom_buff[i]);
00738             if (ret_val)
00739                 break;
00740         }
00741     }
00742 
00743     /* Device's eeprom is always little-endian, word addressable */
00744     for (i = 0; i < last_word - first_word + 1; i++)
00745         eeprom_buff[i] = le16_to_cpu(eeprom_buff[i]);
00746 
00747     memcpy(bytes, (u8 *)eeprom_buff + (eeprom->offset & 1),
00748             eeprom->len);
00749     kfree(eeprom_buff);
00750 
00751     return ret_val;
00752 }

00754 static int igb_set_eeprom(struct net_device *netdev,
00755               struct ethtool_eeprom *eeprom, u8 *bytes)
00756 {
00757     struct igb_adapter *adapter = netdev_priv(netdev);
00758     struct e1000_hw *hw = &adapter->hw;
00759     u16 *eeprom_buff;
00760     void *ptr;
00761     int max_len, first_word, last_word, ret_val = 0;
00762     u16 i;
00763 
00764     if (eeprom->len == 0)
00765         return -EOPNOTSUPP;
00766 
00767     if (eeprom->magic != (hw->vendor_id | (hw->device_id << 16)))
00768         return -EFAULT;
00769 
00770     max_len = hw->nvm.word_size * 2;
00771 
00772     first_word = eeprom->offset >> 1;
00773     last_word = (eeprom->offset + eeprom->len - 1) >> 1;
00774     eeprom_buff = kmalloc(max_len, GFP_KERNEL);
00775     if (!eeprom_buff)
00776         return -ENOMEM;
00777 
00778     ptr = (void *)eeprom_buff;
00779 
00780     if (eeprom->offset & 1) {
00781         /* need read/modify/write of first changed EEPROM word */
00782         /* only the second byte of the word is being modified */
00783         ret_val = e1000_read_nvm(hw, first_word, 1,
00784                         &eeprom_buff[0]);
00785         ptr++;
00786     }
00787     if (((eeprom->offset + eeprom->len) & 1) && (ret_val == 0)) {
00788         /* need read/modify/write of last changed EEPROM word */
00789         /* only the first byte of the word is being modified */
00790         ret_val = e1000_read_nvm(hw, last_word, 1,
00791               &eeprom_buff[last_word - first_word]);
00792     }
00793 
00794     /* Device's eeprom is always little-endian, word addressable */
00795     for (i = 0; i < last_word - first_word + 1; i++)
00796         le16_to_cpus(&eeprom_buff[i]);
00797 
00798     memcpy(ptr, bytes, eeprom->len);
00799 
00800     for (i = 0; i < last_word - first_word + 1; i++)
00801         cpu_to_le16s(&eeprom_buff[i]);
00802 
00803     ret_val = e1000_write_nvm(hw, first_word,
00804                   last_word - first_word + 1, eeprom_buff);
00805 
00806     /* Update the checksum over the first part of the EEPROM if needed
00807      * and flush shadow RAM for 82573 controllers */
00808     if ((ret_val == 0) && ((first_word <= NVM_CHECKSUM_REG)))
00809         e1000_update_nvm_checksum(hw);
00810 
00811     kfree(eeprom_buff);
00812     return ret_val;
00813 }

01022 static int igb_reg_test(struct igb_adapter *adapter, u64 *data)
01023 {
01024     struct e1000_hw *hw = &adapter->hw;
01025     struct igb_reg_test *test;
01026     u32 value, before, after;
01027     u32 i, toggle;
01028 
01029     switch (adapter->hw.mac.type) {
01030     case e1000_i350:
01031         test = reg_test_i350;
01032         toggle = 0x7FEFF3FF;
01033         break;
01034     case e1000_82580:
01035         test = reg_test_82580;
01036         toggle = 0x7FEFF3FF;
01037         break;
01038     case e1000_82576:
01039         test = reg_test_82576;
01040         toggle = 0x7FFFF3FF;
01041         break;
01042     default:
01043         test = reg_test_82575;
01044         toggle = 0x7FFFF3FF;
01045         break;
01046     }
01047 
01048     /* Because the status register is such a special case,
01049      * we handle it separately from the rest of the register
01050      * tests.  Some bits are read-only, some toggle, and some
01051      * are writable on newer MACs.
01052      */
01053     before = E1000_READ_REG(hw, E1000_STATUS);
01054     value = (E1000_READ_REG(hw, E1000_STATUS) & toggle);
01055     E1000_WRITE_REG(hw, E1000_STATUS, toggle);

01058         dev_err(pci_dev_to_dev(adapter->pdev), "failed STATUS register test "

01064     E1000_WRITE_REG(hw, E1000_STATUS, before);
01065

01067      * the test table until we either fail or reach the null entry.
01068      */

01070         for (i = 0; i < test->array_len; i++) {
01071             switch (test->test_type) {
01072             case PATTERN_TEST:
01073                 REG_PATTERN_TEST(test->reg +
01074                         (i * test->reg_offset),
01075                         test->mask,
01076                         test->write);
01077                 break;
01078             case SET_READ_TEST:
01079                 REG_SET_AND_CHECK(test->reg +
01080                         (i * test->reg_offset),
01081                         test->mask,
01082                         test->write);
01083                 break;
01084             case WRITE_NO_TEST:
01085                 writel(test->write,
01086                        (adapter->hw.hw_addr + test->reg)
01087                     + (i * test->reg_offset));
01088                 break;
01089             case TABLE32_TEST:
01090                 REG_PATTERN_TEST(test->reg + (i * 4),
01091                         test->mask,
01092                         test->write);
01093                 break;
01094             case TABLE64_TEST_LO:
01095                 REG_PATTERN_TEST(test->reg + (i * 8),
01096                         test->mask,
01097                         test->write);
01098                 break;
01099             case TABLE64_TEST_HI:
01100                 REG_PATTERN_TEST((test->reg + 4) + (i * 8),
01101                         test->mask,
01102                         test->write);
01103                 break;
01104             }
01105         }
01106         test++;
01107     }
01108 
01109     *data = 0;
01110     return 0;
01111 }
