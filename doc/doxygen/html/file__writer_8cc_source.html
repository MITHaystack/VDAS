00089 void FileWriter::run() {
00090   LOG4CXX_INFO(logger, "FileWriter Running...");
00091 
00092   Timer run_timer;
00093   Timer command_timer;
00094   
00095   try {
00096     // Main processing loop.
00097     while (_running) {
00098 
00099       // State machine.
00100       switch (_state) {
00101       case WRITE_TO_DISK:
00102     if (command_timer.elapsed() > _command_interval) {
00103       command_timer.restart();
00104       continue;
00105     }
00106     handle_write_to_disk();
00107     break;
00108 
00109       case IDLE:
00110     if (command_timer.elapsed() > _command_interval) {
00111       command_timer.restart();
00112       continue;
00113     }
00114     handle_idle();
00115     break;
00116 
00117       case STOP:
00118     handle_stop();
00119     break;
00120 
00121       default:
00122     LOG4CXX_ERROR(logger, "Unknown state.");
00123     break;
00124       }
00125     }
00126     LOG4CXX_DEBUG(logger, "elapsed run time: " << run_timer.elapsed());
00127   } catch(std::exception &ex) {
00128     LOG4CXX_ERROR(logger, "error: " << ex.what());
00129   }
00130 }

00244 void FileWriter::write_block() {
00245   boost::uint8_t* buf;
00246   boost::uint64_t buf_len;
00247   if (_write_bufs.empty()) {
00248     return;
00249   } else {
00250     boost::mutex::scoped_lock lock(_write_bufs_mutex);
00251     buf = _write_bufs.front();
00252     buf_len = _write_bufs.size();
00253     _write_bufs.pop_front();
00254   }
00255 
00256   // Write buffer to disk.
00257   int bytes_left = _WRITE_BLOCK_SIZE;
00258   int bytes_written = 0;
00259   while (bytes_left) {
00260     int nb = ::write(_pfd.fd, &buf[bytes_written], bytes_left);
00261     if (nb > 0) {
00262       bytes_left -= nb;
00263       bytes_written += nb;
00264     } else {
00265       LOG4CXX_ERROR(logger, "Write error: " << strerror(errno));
00266     }
00267   }
00268   if (_sw)
00269     _sw->update(1, bytes_written, 0, buf_len);
00270   free_buffer(buf);
00271 }
