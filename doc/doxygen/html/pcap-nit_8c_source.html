00074 static int
00075 pcap_stats_nit(pcap_t *p, struct pcap_stat *ps)
00076 {
00077 
00078     /*
00079      * "ps_recv" counts packets handed to the filter, not packets
00080      * that passed the filter.  As filtering is done in userland,
00081      * this does not include packets dropped because we ran out
00082      * of buffer space.
00083      *
00084      * "ps_drop" presumably counts packets dropped by the socket
00085      * because of flow control requirements or resource exhaustion;
00086      * it doesn't count packets dropped by the interface driver.
00087      * As filtering is done in userland, it counts packets regardless
00088      * of whether they would've passed the filter.
00089      *
00090      * These statistics don't include packets not yet read from the
00091      * kernel by libpcap or packets not yet read from libpcap by the
00092      * application.
00093      */
00094     *ps = p->md.stat;
00095     return (0);

00098 static int
00099 pcap_read_nit(pcap_t *p, int cnt, pcap_handler callback, u_char *user)
00100 {
00101     register int cc, n;
00102     register u_char *bp, *cp, *ep;
00103     register struct nit_hdr *nh;
00104     register int caplen;
00105 
00106     cc = p->cc;
00107     if (cc == 0) {
00108         cc = read(p->fd, (char *)p->buffer, p->bufsize);
00109         if (cc < 0) {
00110             if (errno == EWOULDBLOCK)
00111                 return (0);
00112             snprintf(p->errbuf, sizeof(p->errbuf), "pcap_read: %s",
00113                 pcap_strerror(errno));
00114             return (-1);
00115         }
00116         bp = p->buffer;
00117     } else
00118         bp = p->bp;
00119 
00120     /*
00121      * Loop through each packet.  The increment expression
00122      * rounds up to the next int boundary past the end of
00123      * the previous packet.
00124      */
00125     n = 0;
00126     ep = bp + cc;
00127     while (bp < ep) {
00128         /*
00129          * Has "pcap_breakloop()" been called?
00130          * If so, return immediately - if we haven't read any
00131          * packets, clear the flag and return -2 to indicate
00132          * that we were told to break out of the loop, otherwise
00133          * leave the flag set, so that the *next* call will break
00134          * out of the loop without having read any packets, and
00135          * return the number of packets we've processed so far.
00136          */
00137         if (p->break_loop) {
00138             if (n == 0) {
00139                 p->break_loop = 0;
00140                 return (-2);
00141             } else {
00142                 p->cc = ep - bp;
00143                 p->bp = bp;
00144                 return (n);
00145             }
00146         }
00147 
00148         nh = (struct nit_hdr *)bp;
00149         cp = bp + sizeof(*nh);
00150 
00151         switch (nh->nh_state) {
00152 
00153         case NIT_CATCH:
00154             break;
00155 
00156         case NIT_NOMBUF:
00157         case NIT_NOCLUSTER:
00158         case NIT_NOSPACE:
00159             p->md.stat.ps_drop = nh->nh_dropped;
00160             continue;
00161 
00162         case NIT_SEQNO:
00163             continue;
00164 
00165         default:
00166             snprintf(p->errbuf, sizeof(p->errbuf),
00167                 "bad nit state %d", nh->nh_state);
00168             return (-1);
00169         }
00170         ++p->md.stat.ps_recv;
00171         bp += ((sizeof(struct nit_hdr) + nh->nh_datalen +
00172             sizeof(int) - 1) & ~(sizeof(int) - 1));
00173 
00174         caplen = nh->nh_wirelen;
00175         if (caplen > p->snapshot)
00176             caplen = p->snapshot;
00177         if (bpf_filter(p->fcode.bf_insns, cp, nh->nh_wirelen, caplen)) {
00178             struct pcap_pkthdr h;
00179             h.ts = nh->nh_timestamp;
00180             h.len = nh->nh_wirelen;
00181             h.caplen = caplen;
00182             (*callback)(user, &h, cp);
00183             if (++n >= cnt && cnt > 0) {
00184                 p->cc = ep - bp;
00185                 p->bp = bp;
00186                 return (n);
00187             }
00188         }
00189     }
00190     p->cc = 0;
00191     return (n);
00192 }

00273             "bind: %s: %s", snit.snit_ifname, pcap_strerror(errno));
00274         goto bad;

00277 
00278     /*
00279      * NIT supports only ethernets.
00280      */
00281     p->linktype = DLT_EN10MB;
00282

00306     /*

00309     if (p->dlt_list != NULL) {
00310         p->dlt_list[0] = DLT_EN10MB;
00311         p->dlt_list[1] = DLT_DOCSIS;
