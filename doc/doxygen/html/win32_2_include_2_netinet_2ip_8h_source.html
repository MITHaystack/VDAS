00051 /*
00052  * Structure of an internet header, naked of options.
00053  *
00054  * We declare ip_len and ip_off to be short, rather than u_short
00055  * pragmatically since otherwise unsigned comparisons can result
00056  * against negative integers quite easily, and fail in subtle ways.
00057  */
00058 struct ip {
00059 #if BYTE_ORDER == LITTLE_ENDIAN 
00060     u_char  ip_hl:4,        /* header length */
00061         ip_v:4;         /* version */
00062 #endif
00063 #if BYTE_ORDER == BIG_ENDIAN 
00064     u_char  ip_v:4,         /* version */
00065         ip_hl:4;        /* header length */
00066 #endif
00067     u_char  ip_tos;         /* type of service */
00068     short   ip_len;         /* total length */
00069     u_short ip_id;          /* identification */
00070     short   ip_off;         /* fragment offset field */
00071 #define IP_DF 0x4000            /* dont fragment flag */
00072 #define IP_MF 0x2000            /* more fragments flag */
00073 #define IP_OFFMASK 0x1fff       /* mask for fragmenting bits */
00074     u_char  ip_ttl;         /* time to live */
00075     u_char  ip_p;           /* protocol */
00076     u_short ip_sum;         /* checksum */
00077     struct  in_addr ip_src,ip_dst;  /* source and dest address */
00078 };
