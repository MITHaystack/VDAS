00443 /*
00444  * Handle the rx-level packet.  See if we know what port it's going to so
00445  * we can peek at the afs call inside
00446  */
00447 
00448 void
00449 rx_print(register const u_char *bp, int length, int sport, int dport,
00450      u_char *bp2)
00451 {
00452     register struct rx_header *rxh;
00453     int i;
00454     int32_t opcode;
00455 
00456     if (snapend - bp < (int)sizeof (struct rx_header)) {
00457         printf(" [|rx] (%d)", length);
00458         return;
00459     }
00460 
00461     rxh = (struct rx_header *) bp;
00462 
00463     printf(" rx %s", tok2str(rx_types, "type %d", rxh->type));
00464 
00465     if (vflag) {
00466         int firstflag = 0;
00467 
00468         if (vflag > 1)
00469             printf(" cid %08x call# %d",
00470                    (int) EXTRACT_32BITS(&rxh->cid),
00471                    (int) EXTRACT_32BITS(&rxh->callNumber));
00472 
00473         printf(" seq %d ser %d",
00474                (int) EXTRACT_32BITS(&rxh->seq),
00475                (int) EXTRACT_32BITS(&rxh->serial));
00476 
00477         if (vflag > 2)
00478             printf(" secindex %d serviceid %hu",
00479                 (int) rxh->securityIndex,
00480                 EXTRACT_16BITS(&rxh->serviceId));
00481 
00482         if (vflag > 1)
00483             for (i = 0; i < NUM_RX_FLAGS; i++) {
00484                 if (rxh->flags & rx_flags[i].flag &&
00485                     (!rx_flags[i].packetType ||
00486                      rxh->type == rx_flags[i].packetType)) {
00487                     if (!firstflag) {
00488                         firstflag = 1;
00489                         printf(" ");
00490                     } else {
00491                         printf(",");
00492                     }
00493                     printf("<%s>", rx_flags[i].s);
00494                 }
00495             }
00496     }
00497 
00498     /*
00499      * Try to handle AFS calls that we know about.  Check the destination
00500      * port and make sure it's a data packet.  Also, make sure the
00501      * seq number is 1 (because otherwise it's a continuation packet,
00502      * and we can't interpret that).  Also, seems that reply packets
00503      * do not have the client-init flag set, so we check for that
00504      * as well.
00505      */
00506 
00507     if (rxh->type == RX_PACKET_TYPE_DATA &&
00508         EXTRACT_32BITS(&rxh->seq) == 1 &&
00509         rxh->flags & RX_CLIENT_INITIATED) {
00510 
00511         /*
00512          * Insert this call into the call cache table, so we
00513          * have a chance to print out replies
00514          */
00515 
00516         rx_cache_insert(bp, (const struct ip *) bp2, dport);
00517 
00518         switch (dport) {
00519             case FS_RX_PORT:    /* AFS file service */
00520                 fs_print(bp, length);
00521                 break;
00522             case CB_RX_PORT:    /* AFS callback service */
00523                 cb_print(bp, length);
00524                 break;
00525             case PROT_RX_PORT:  /* AFS protection service */
00526                 prot_print(bp, length);
00527                 break;
00528             case VLDB_RX_PORT:  /* AFS VLDB service */
00529                 vldb_print(bp, length);
00530                 break;
00531             case KAUTH_RX_PORT: /* AFS Kerberos auth service */
00532                 kauth_print(bp, length);
00533                 break;
00534             case VOL_RX_PORT:   /* AFS Volume service */
00535                 vol_print(bp, length);
00536                 break;
00537             case BOS_RX_PORT:   /* AFS BOS service */
00538                 bos_print(bp, length);
00539                 break;
00540             default:
00541                 ;
00542         }
00543 
00544     /*
00545      * If it's a reply (client-init is _not_ set, but seq is one)
00546      * then look it up in the cache.  If we find it, call the reply
00547      * printing functions  Note that we handle abort packets here,
00548      * because printing out the return code can be useful at times.
00549      */
00550 
00551     } else if (((rxh->type == RX_PACKET_TYPE_DATA &&
00552                     EXTRACT_32BITS(&rxh->seq) == 1) ||
00553             rxh->type == RX_PACKET_TYPE_ABORT) &&
00554            (rxh->flags & RX_CLIENT_INITIATED) == 0 &&
00555            rx_cache_find(rxh, (const struct ip *) bp2,
00556                  sport, &opcode)) {
00557 
00558         switch (sport) {
00559             case FS_RX_PORT:    /* AFS file service */
00560                 fs_reply_print(bp, length, opcode);
00561                 break;
00562             case CB_RX_PORT:    /* AFS callback service */
00563                 cb_reply_print(bp, length, opcode);
00564                 break;
00565             case PROT_RX_PORT:  /* AFS PT service */
