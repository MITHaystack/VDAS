00163 static int
00164 pcap_read_win32_npf(pcap_t *p, int cnt, pcap_handler callback, u_char *user)
00165 {
00166     int cc;
00167     int n = 0;
00168     register u_char *bp, *ep;
00169 
00170     cc = p->cc;
00171     if (p->cc == 0) {
00172         /*
00173          * Has "pcap_breakloop()" been called?
00174          */
00175         if (p->break_loop) {
00176             /*
00177              * Yes - clear the flag that indicates that it
00178              * has, and return -2 to indicate that we were
00179              * told to break out of the loop.
00180              */
00181             p->break_loop = 0;
00182             return (-2);
00183         }
00184 
00185         /* capture the packets */
00186         if(PacketReceivePacket(p->adapter,p->Packet,TRUE)==FALSE){
00187             snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "read error: PacketReceivePacket failed");
00188             return (-1);
00189         }
00190             
00191         cc = p->Packet->ulBytesReceived;
00192 
00193         bp = p->Packet->Buffer;
00194     } 
00195     else
00196         bp = p->bp;
00197 
00198     /*
00199      * Loop through each packet.
00200      */
00201 #define bhp ((struct bpf_hdr *)bp)
00202     ep = bp + cc;
00203     while (1) {
00204         register int caplen, hdrlen;
00205 
00206         /*
00207          * Has "pcap_breakloop()" been called?
00208          * If so, return immediately - if we haven't read any
00209          * packets, clear the flag and return -2 to indicate
00210          * that we were told to break out of the loop, otherwise
00211          * leave the flag set, so that the *next* call will break
00212          * out of the loop without having read any packets, and
00213          * return the number of packets we've processed so far.
00214          */
00215         if (p->break_loop) {
00216             if (n == 0) {
00217                 p->break_loop = 0;
00218                 return (-2);
00219             } else {
00220                 p->bp = bp;
00221                 p->cc = ep - bp;
00222                 return (n);
00223             }
00224         }
00225         if (bp >= ep)
00226             break;
00227 
00228         caplen = bhp->bh_caplen;
00229         hdrlen = bhp->bh_hdrlen;
00230 
00231         /*
00232          * XXX A bpf_hdr matches a pcap_pkthdr.
00233          */
00234         (*callback)(user, (struct pcap_pkthdr*)bp, bp + hdrlen);
00235         bp += BPF_WORDALIGN(caplen + hdrlen);
00236         if (++n >= cnt && cnt > 0) {
00237             p->bp = bp;
00238             p->cc = ep - bp;
00239             return (n);
00240         }
00241     }
00242 #undef bhp
00243     p->cc = 0;
00244     return (n);
00245 }

00249 pcap_read_win32_dag(pcap_t *p, int cnt, pcap_handler callback, u_char *user)
00250 {
00251     u_char *dp = NULL;
00252     int packet_len = 0, caplen = 0;
00253     struct pcap_pkthdr  pcap_header;
00254     u_char *endofbuf;
00255     int n = 0;
00256     dag_record_t *header;
00257     unsigned erf_record_len;
00258     ULONGLONG ts;
00259     int cc;
00260     unsigned swt;
00261     unsigned dfp = p->adapter->DagFastProcess;
00262 
00263     cc = p->cc;
00264     if (cc == 0) /* Get new packets only if we have processed all the ones of the previous read */
00265     {
00266         /* Get new packets from the network */
00267         if(PacketReceivePacket(p->adapter, p->Packet, TRUE)==FALSE){
00268             snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "read error: PacketReceivePacket failed");
00269             return (-1);
00270         }
00271 
00272         cc = p->Packet->ulBytesReceived;
00273         if(cc == 0)
00274             /* The timeout has expired but we no packets arrived */
00275             return 0;
00276         header = (dag_record_t*)p->adapter->DagBuffer;
00277     } 
00278     else
00279         header = (dag_record_t*)p->bp;
00280     
00281     endofbuf = (char*)header + cc;
00282     
00283     /* 
00284      * Cycle through the packets 
00285      */
00286     do
00287     {
00288         erf_record_len = SWAPS(header->rlen);
00289         if((char*)header + erf_record_len > endofbuf)
00290             break;
00291 
00292         /* Increase the number of captured packets */
00293         p->md.stat.ps_recv++;
00294         
00295         /* Find the beginning of the packet */
00296         dp = ((u_char *)header) + dag_record_size;
00297 
00298         /* Determine actual packet len */
00299         switch(header->type) 
00300         {
00301         case TYPE_ATM:
00302             packet_len = ATM_SNAPLEN;
00303             caplen = ATM_SNAPLEN;
00304             dp += 4;
00305 
00306             break;
00307 
00308         case TYPE_ETH:
00309             swt = SWAPS(header->wlen);
00310             packet_len = swt - (p->md.dag_fcs_bits);
00311             caplen = erf_record_len - dag_record_size - 2;
00312             if (caplen > packet_len)
00313             {
00314                 caplen = packet_len;
00315             }
00316             dp += 2;
00317             
00318             break;
00319         
00320         case TYPE_HDLC_POS:
00321             swt = SWAPS(header->wlen);
00322             packet_len = swt - (p->md.dag_fcs_bits);
00323             caplen = erf_record_len - dag_record_size;
00324             if (caplen > packet_len)
00325             {
00326                 caplen = packet_len;
00327             }
00328             
00329             break;
00330         }
00331         
00332         if(caplen > p->snapshot)
00333             caplen = p->snapshot;
00334 
00335         /*
00336          * Has "pcap_breakloop()" been called?
00337          * If so, return immediately - if we haven't read any
00338          * packets, clear the flag and return -2 to indicate
00339          * that we were told to break out of the loop, otherwise
00340          * leave the flag set, so that the *next* call will break
00341          * out of the loop without having read any packets, and
00342          * return the number of packets we've processed so far.
00343          */
00344         if (p->break_loop) 
00345         {
00346             if (n == 0) 
00347             {
00348                 p->break_loop = 0;
00349                 return (-2);
00350             } 
00351             else 
00352             {
00353                 p->bp = (char*)header;
00354                 p->cc = endofbuf - (char*)header;
00355                 return (n);
00356             }
00357         }
00358 
00359         if(!dfp)
00360         {
00361             /* convert between timestamp formats */
00362             ts = header->ts;
00363             pcap_header.ts.tv_sec = (int)(ts >> 32);
00364             ts = (ts & 0xffffffffi64) * 1000000;
00365             ts += 0x80000000; /* rounding */
00366             pcap_header.ts.tv_usec = (int)(ts >> 32);
00367             if (pcap_header.ts.tv_usec >= 1000000) {
00368                 pcap_header.ts.tv_usec -= 1000000;
00369                 pcap_header.ts.tv_sec++;
00370             }
00371         }
00372         
00373         /* No underlaying filtering system. We need to filter on our own */
00374         if (p->fcode.bf_insns) 
00375         {
00376             if (bpf_filter(p->fcode.bf_insns, dp, packet_len, caplen) == 0) 
00377             {
00378                 /* Move to next packet */
00379                 header = (dag_record_t*)((char*)header + erf_record_len);
00380                 continue;
00381             }
00382         }
00383         
00384         /* Fill the header for the user suppplied callback function */
00385         pcap_header.caplen = caplen;
00386         pcap_header.len = packet_len;
00387         
00388         /* Call the callback function */
00389         (*callback)(user, &pcap_header, dp);
00390         
00391         /* Move to next packet */
00392         header = (dag_record_t*)((char*)header + erf_record_len);
00393 
00394         /* Stop if the number of packets requested by user has been reached*/
00395         if (++n >= cnt && cnt > 0) 
00396         {
00397             p->bp = (char*)header;
00398             p->cc = endofbuf - (char*)header;
00399             return (n);
00400         }
00401     }
00402     while((u_char*)header < endofbuf);
00403     
00404   return 1;
00405 }

00430     /*
00431      * We assume it all got sent if "PacketSendPacket()" succeeded.
00432      * "pcap_inject()" is expected to return the number of bytes

00435     return size;
