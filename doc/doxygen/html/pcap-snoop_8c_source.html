00062 pcap_read_snoop(pcap_t *p, int cnt, pcap_handler callback, u_char *user)
00063 {
00064     int cc;
00065     register struct snoopheader *sh;
00066     register u_int datalen;
00067     register u_int caplen;
00068     register u_char *cp;
00069 
00070 again:
00071     /*
00072      * Has "pcap_breakloop()" been called?
00073      */
00074     if (p->break_loop) {
00075         /*
00076          * Yes - clear the flag that indicates that it
00077          * has, and return -2 to indicate that we were
00078          * told to break out of the loop.
00079          */
00080         p->break_loop = 0;
00081         return (-2);
00082     }
00083     cc = read(p->fd, (char *)p->buffer, p->bufsize);
00084     if (cc < 0) {
00085         /* Don't choke when we get ptraced */
00086         switch (errno) {
00087 
00088         case EINTR:
00089             goto again;
00090 
00091         case EWOULDBLOCK:
00092             return (0);         /* XXX */
00093         }
00094         snprintf(p->errbuf, sizeof(p->errbuf),
00095             "read: %s", pcap_strerror(errno));
00096         return (-1);
00097     }
00098     sh = (struct snoopheader *)p->buffer;
00099     datalen = sh->snoop_packetlen;
00100 
00101     /*
00102      * XXX - Sigh, snoop_packetlen is a 16 bit quantity.  If we
00103      * got a short length, but read a full sized snoop pakcet,
00104      * assume we overflowed and add back the 64K...
00105      */
00106     if (cc == (p->snapshot + sizeof(struct snoopheader)) &&
00107         (datalen < p->snapshot))
00108         datalen += (64 * 1024);
00109 
00110     caplen = (datalen < p->snapshot) ? datalen : p->snapshot;
00111     cp = (u_char *)(sh + 1) + p->offset;        /* XXX */
00112 
00113     /* 
00114      * XXX unfortunately snoop loopback isn't exactly like
00115      * BSD's.  The address family is encoded in the first 2
00116      * bytes rather than the first 4 bytes!  Luckily the last
00117      * two snoop loopback bytes are zeroed.
00118      */
00119     if (p->linktype == DLT_NULL && *((short *)(cp + 2)) == 0) {
00120         u_int *uip = (u_int *)cp;
00121         *uip >>= 16;
00122     }
00123 
00124     if (p->fcode.bf_insns == NULL ||
00125         bpf_filter(p->fcode.bf_insns, cp, datalen, caplen)) {
00126         struct pcap_pkthdr h;
00127         ++p->md.stat.ps_recv;
00128         h.ts.tv_sec = sh->snoop_timestamp.tv_sec;
00129         h.ts.tv_usec = sh->snoop_timestamp.tv_usec;
00130         h.len = datalen;
00131         h.caplen = caplen;
00132         (*callback)(user, &h, cp);
00133         return (1);
00134     }
00135     return (0);
00136 }

00158 {
00159     register struct rawstats *rs;
00160     struct rawstats rawstats;
00161 
00162     rs = &rawstats;
00163     memset(rs, 0, sizeof(*rs));
00164     if (ioctl(p->fd, SIOCRAWSTATS, (char *)rs) < 0) {
00165         snprintf(p->errbuf, sizeof(p->errbuf),
00166             "SIOCRAWSTATS: %s", pcap_strerror(errno));
00167         return (-1);
00168     }
00169 
00170     /*
00171      * "ifdrops" are those dropped by the network interface
00172      * due to resource shortages or hardware errors.
00173      *
00174      * "sbdrops" are those dropped due to socket buffer limits.
00175      *
00176      * As filter is done in userland, "sbdrops" counts packets
00177      * regardless of whether they would've passed the filter.
00178      *
00179      * XXX - does this count *all* Snoop or Drain sockets,
00180      * rather than just this socket?  If not, why does it have
00181      * both Snoop and Drain statistics?
00182      */
00183     p->md.stat.ps_drop =
00184         rs->rs_snoop.ss_ifdrops + rs->rs_snoop.ss_sbdrops +
00185         rs->rs_drain.ds_ifdrops + rs->rs_drain.ds_sbdrops;
00186 
00187     /*
00188      * "ps_recv" counts only packets that passed the filter.
00189      * As filtering is done in userland, this does not include
00190      * packets dropped because we ran out of buffer space.
00191      */
00192     *ps = p->md.stat;
00193     return (0);

00252         ll_hdrlen = sizeof(struct ether_header);

00256          * that an application can let you choose it, in case you're
00257          * capturing DOCSIS traffic that a Cisco Cable Modem
00258          * Termination System is putting out onto an Ethernet (it
00259          * doesn't put an Ethernet header onto the wire, it puts raw

00268          * the same applies for "qaa" classical IP devices.  If
00269          * "fa" devices are for FORE SPANS, that'd apply to them
00270          * as well; what are "cip" devices - some other ATM
00271          * Classical IP devices?
00272          */
00273         p->dlt_list = (u_int *) malloc(sizeof(u_int) * 2);
00274         /*

00298         p->linktype = DLT_NULL;

00302             "snoop: unknown physical layer type");

00305 
00306     if (p->opt.rfmon) {
00307         /*

00315     /*
00316      * XXX - IRIX appears to give you an error if you try to set the
00317      * capture length to be greater than the MTU, so let's try to get
00318      * the MTU first and, if that succeeds, trim the snap length
