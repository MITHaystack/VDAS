00033 struct _daq_module
00034 {
00035     /* The version of the API this module implements.
00036        This *must* be the first element in the structure. */
00037     const uint32_t api_version;
00038     /* The version of the DAQ module itself - can be completely arbitrary. */
00039     const uint32_t module_version;
00040     /* The name of the module (sfpacket, xvnim, pcap, etc.) */
00041     const char *name;
00042     /* Various flags describing the module and its capabilities (Inline-capabale, etc.) */
00043     const uint32_t type;
00044     /* Initialize the device for packet acquisition with the supplied configuration.
00045        This should not start queuing packets for the application. */
00046     int (*initialize) (const DAQ_Config_t *config, void **ctxt_ptr, char *errbuf, size_t len);
00047     /* Set the module's BPF based on the given string */
00048     int (*set_filter) (void *handle, const char *filter);
00049     /* Complete device opening and begin queuing packets if they have not been already. */
00050     int (*start) (void *handle);
00051     /* Acquire up to <cnt> packets and call <callback> for each with <user> as the final argument.
00052        The return value of the callback will determine the action taken by the DAQ for each packet.
00053        If <cnt> is 0, packets will continue to be acquired until some other factor breaks the
00054        acquisition loop. */
00055     int (*acquire) (void *handle, int cnt, DAQ_Analysis_Func_t callback, void *user);
00056     /* Inject a new packet going either the same or opposite direction as the specified packet. */
00057     int (*inject) (void *handle, const DAQ_PktHdr_t *hdr, const uint8_t *packet_data, uint32_t len, int reverse);
00058     /* Force breaking out of the acquisition loop after the current iteration. */
00059     int (*breakloop) (void *handle);
00060     /* Stop queuing packets, if possible */
00061     int (*stop) (void *handle);
00062     /* Close the device and clean up */
00063     void (*shutdown) (void *handle);
00064     /* Get the status of the module (one of DAQ_STATE_*). */
00065     DAQ_State (*check_status) (void *handle);
00066     /* Populates the <stats> structure with the current DAQ stats.  These stats are cumulative. */
00067     int (*get_stats) (void *handle, DAQ_Stats_t *stats);
00068     /* Resets the DAQ module's internal stats. */
00069     void (*reset_stats) (void *handle);
00070     /* Return the configured snaplen */
00071     int (*get_snaplen) (void *handle);
00072     /* Return a bitfield of the device's capabilities */
00073     uint32_t (*get_capabilities) (void *handle);
00074     /* Return the instance's Data Link Type */
00075     int (*get_datalink_type) (void *handle);
00076     /* Return a pointer to the module's internal error buffer */
00077     const char * (*get_errbuf) (void *handle);
00078     /* Write a string to the module instance's internal error buffer */
00079     void (*set_errbuf) (void *handle, const char *string);
00080     /* Return the index of the given named device if possible. */
00081     int (*get_device_index) (void *handle, const char *device);
00082 };
