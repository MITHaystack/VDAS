00065 /* Describe the long-named options requested by the application.
00066    The LONG_OPTIONS argument to getopt_long or getopt_long_only is a vector
00067    of `struct option' terminated by an element containing a name which is
00068    zero.
00069 
00070    The field `has_arg' is:
00071    no_argument      (or 0) if the option does not take an argument,
00072    required_argument    (or 1) if the option requires an argument,
00073    optional_argument    (or 2) if the option takes an optional argument.
00074 
00075    If the field `flag' is not NULL, it points to a variable that is set
00076    to the value given in the field `val' when the option is found, but
00077    left unchanged if the option is not found.
00078 
00079    To have a long-named option do something other than set an `int' to
00080    a compiled-in constant, such as set a value from `optarg', set the
00081    option's `flag' field to zero and its `val' field to a nonzero
00082    value (the equivalent single-letter option character, if there is
00083    one).  For long options that have a zero `flag' field, `getopt'
00084    returns the contents of the `val' field.  */
00085 
00086 struct option
00087 {
00088 #if __STDC__
00089   const char *name;
00090 #else
00091   char *name;
00092 #endif
00093   /* has_arg can't be an enum because some compilers complain about
00094      type mismatches in all the code that assumes it is an int.  */
00095   int has_arg;
00096   int *flag;
00097   int val;
00098 };
