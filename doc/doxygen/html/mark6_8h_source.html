00052   friend std::ostream& operator<<(std::ostream& out, const IPEndpoint& ep) {
00053     out
00054       << "IPEndpoint {\n"
00055       << "_ip_address:" << ep._ip_address << std::endl
00056       << "_port:" << ep._port << std::endl
00057       << "}";
00058     return out;
00059   }
00060 
00061   bool operator() (IPEndpoint ep1, IPEndpoint ep2) {
00062     if (ep1._ip_address == ep2._ip_address)
00063       return ep1._port < ep2._port;
00064     return ep1._ip_address < ep2._ip_address;
