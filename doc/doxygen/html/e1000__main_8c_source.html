00477 {
00478     struct net_device *netdev = adapter->netdev;

00480 
00481     /* disable receives in the hardware */
00482     rctl = E1000_READ_REG(&adapter->hw, E1000_RCTL);
00483     E1000_WRITE_REG(&adapter->hw, E1000_RCTL, rctl & ~E1000_RCTL_EN);

00491     E1000_WRITE_REG(&adapter->hw, E1000_TCTL, tctl);

00493     E1000_WRITE_FLUSH(&adapter->hw);
00494     msleep(10);
00495 
00496 #ifdef CONFIG_E1000_NAPI
00497     napi_disable(&adapter->rx_ring->napi);
00498 
00499 #endif
00500     e1000_irq_disable(adapter);
00501 
00502     /* signal that we're down so the interrupt handler does not
00503      * reschedule our watchdog timer */

00505 
00506     del_timer_sync(&adapter->tx_fifo_stall_timer);

00571     case e1000_82547_rev_2:
00572         legacy_pba_adjust = true;
00573         pba = E1000_PBA_30K;
00574         break;
00575     case e1000_undefined:
00576     case e1000_num_macs:
00577         break;
00578     }
00579 
00580     if (legacy_pba_adjust == true) {
00581         if (adapter->max_frame_size > E1000_RXBUFFER_8192)
00582             pba -= 8; /* allocate more FIFO for Tx */

00584         if (mac->type == e1000_82547) {
00585             adapter->tx_fifo_head = 0;
00586             adapter->tx_head_addr = pba << E1000_TX_HEAD_ADDR_SHIFT;
00587             adapter->tx_fifo_size =
00588                 (E1000_PBA_40K - pba) << E1000_PBA_BYTES_SHIFT;
00589             atomic_set(&adapter->tx_fifo_stall, 0);
00590         }
00591     } else if (adapter->max_frame_size > ETH_FRAME_LEN + ETH_FCS_LEN) {
00592         /* adjust PBA for jumbo frames */
00593         E1000_WRITE_REG(&adapter->hw, E1000_PBA, pba);
00594 
00595         /* To maintain wire speed transmits, the Tx FIFO should be
00596          * large enough to accommodate two full transmit packets,
00597          * rounded up to the next 1KB and expressed in KB.  Likewise,
00598          * the Rx FIFO should be large enough to accommodate at least
00599          * one full receive packet and is similarly rounded up and
00600          * expressed in KB. */

00602         /* upper 16 bits has Tx packet buffer allocation size in KB */
00603         tx_space = pba >> 16;
00604         /* lower 16 bits has Rx packet buffer allocation size in KB */
00605         pba &= 0xffff;
00606         /* the tx fifo also stores 16 bytes of information about the tx
00607          * but don't include ethernet FCS because hardware appends it */

00609                         sizeof(struct e1000_tx_desc) -

00615         min_rx_space = ALIGN(min_rx_space, 1024);
00616         min_rx_space >>= 10;

00619          * and the min Tx FIFO size is less than the current Rx FIFO
00620          * allocation, take space away from current Rx allocation */

00622             ((min_tx_space - tx_space) < pba)) {
00623             pba = pba - (min_tx_space - tx_space);
00624 
00625             /* PCI/PCIx hardware has PBA alignment constraints */

00627             case e1000_82545 ... e1000_82546_rev_3:

00629                 break;
00630             default:
00631                 break;
00632             }

00635              * adjustment or use Early Receive if available */
00636             if (pba < min_rx_space) {
00637                 pba = min_rx_space;
00638             }
00639         }
00640     }
00641 
00642     E1000_WRITE_REG(&adapter->hw, E1000_PBA, pba);
00643 
00644     /* flow control settings */
00645     /* The high water mark must be low enough to fit one full frame
00646      * (or the size used for early receive) above it in the Rx FIFO.
00647      * Set it to the lower of:
00648      * - 90% of the Rx FIFO size, and
00649      * - the full Rx FIFO size minus the early receive size (for parts
00650      *   with ERT support assuming ERT set to E1000_ERT_2048), or
00651      * - the full Rx FIFO size minus one full frame */

00653           ((pba << 10) - adapter->max_frame_size));

00656     fc->low_water = fc->high_water - 8;
00657 
00658     fc->pause_time = E1000_FC_PAUSE_TIME;
00659     fc->send_xon = 1;
00660     fc->current_mode = fc->requested_mode;

00662     /* Allow time for pending master requests to run */
00663     e1000_reset_hw(&adapter->hw);
00664 
00665     if (mac->type >= e1000_82544)
00666         E1000_WRITE_REG(&adapter->hw, E1000_WUC, 0);
00667 
00668     if (e1000_init_hw(&adapter->hw))
00669         DPRINTK(PROBE, ERR, "Hardware Error\n");
00670 #ifdef NETIF_F_HW_VLAN_TX
00671     e1000_update_mng_vlan(adapter);
00672 #endif
00673     /* if (adapter->hwflags & HWFLAGS_PHY_PWR_BIT) { */
00674     if (mac->type >= e1000_82544 &&
00675         mac->type <= e1000_82547_rev_2 &&
00676         mac->autoneg == 1 &&
00677         adapter->hw.phy.autoneg_advertised == ADVERTISE_1000_FULL) {
00678         u32 ctrl = E1000_READ_REG(&adapter->hw, E1000_CTRL);
00679         /* clear phy power management bit if we are in gig only mode,
00680          * which if enabled will attempt negotiation to 100Mb, which
00681          * can cause a loss of link at power off or driver unload */
00682         ctrl &= ~E1000_CTRL_SWDPIN3;
00683         E1000_WRITE_REG(&adapter->hw, E1000_CTRL, ctrl);
00684     }
00685 
00686     /* Enable h/w to recognize an 802.1Q VLAN Ethernet packet */
00687     E1000_WRITE_REG(&adapter->hw, E1000_VET, ETHERNET_IEEE_VLAN_TYPE);

00689     e1000_reset_adaptive(&adapter->hw);
00690     e1000_get_phy_info(&adapter->hw);
00691 
00692     e1000_release_manageability(adapter);
00693 }

01349                                        void *start, unsigned long len)
01350 {

01352     unsigned long end = begin + len;

01371 static int e1000_setup_tx_resources(struct e1000_adapter *adapter,
01372                                     struct e1000_tx_ring *tx_ring)
01373 {
01374     struct pci_dev *pdev = adapter->pdev;
01375     int size;
01376 
01377     size = sizeof(struct e1000_buffer) * tx_ring->count;
01378     tx_ring->buffer_info = vmalloc(size);
01379     if (!tx_ring->buffer_info) {
01380         DPRINTK(PROBE, ERR,
01381         "Unable to allocate memory for the transmit descriptor ring\n");
01382         return -ENOMEM;
01383     }
01384     memset(tx_ring->buffer_info, 0, size);
01385 
01386     /* round up to nearest 4K */
01387 
01388     tx_ring->size = tx_ring->count * sizeof(struct e1000_tx_desc);
01389     tx_ring->size = ALIGN(tx_ring->size, 4096);
01390 
01391     tx_ring->desc = dma_alloc_coherent(pci_dev_to_dev(pdev), tx_ring->size,
01392                                        &tx_ring->dma, GFP_KERNEL);
01393     if (!tx_ring->desc) {
01394 setup_tx_desc_die:
01395         vfree(tx_ring->buffer_info);

01401     /* Fix for errata 23, can't cross 64kB boundary */
01402     if (!e1000_check_64k_bound(adapter, tx_ring->desc, tx_ring->size)) {

01408         tx_ring->desc = dma_alloc_coherent(pci_dev_to_dev(pdev),
01409                            tx_ring->size,
01410                                            &tx_ring->dma, GFP_KERNEL);
01411         /* Failed allocation, critical failure */
01412         if (!tx_ring->desc) {
01413             dma_free_coherent(pci_dev_to_dev(pdev), tx_ring->size,
01414                       olddesc, olddma);
01415             goto setup_tx_desc_die;
01416         }
01417 
01418         if (!e1000_check_64k_bound(adapter, tx_ring->desc,
01419                                    tx_ring->size)) {
01420             /* give up */
01421             dma_free_coherent(pci_dev_to_dev(pdev), tx_ring->size,
01422                       tx_ring->desc, tx_ring->dma);
01423             dma_free_coherent(pci_dev_to_dev(pdev), tx_ring->size,
01424                       olddesc, olddma);
01425             DPRINTK(PROBE, ERR,
01426                 "Unable to allocate aligned memory "
01427                 "for the transmit descriptor ring\n");
01428             vfree(tx_ring->buffer_info);
01429             return -ENOMEM;
01430         } else {
01431             /* Free old allocation, new allocation was successful */
01432             dma_free_coherent(pci_dev_to_dev(pdev), tx_ring->size,
01433                       olddesc, olddma);
01434         }
01435     }
01436     memset(tx_ring->desc, 0, tx_ring->size);
01437 
01438     tx_ring->next_to_use = 0;
01439     tx_ring->next_to_clean = 0;
01440 
01441     return 0;
01442 }

01468 static void e1000_configure_tx(struct e1000_adapter *adapter)
01469 {
01470     u64 tdba;
01471     struct e1000_hw *hw = &adapter->hw;
01472     u32 tdlen, tctl, tipg;
01473     u32 ipgr1, ipgr2;
01474 
01475     /* Setup the HW Tx Head and Tail descriptor pointers */
01476     tdba = adapter->tx_ring->dma;
01477     tdlen = adapter->tx_ring->count * sizeof(struct e1000_tx_desc);
01478     E1000_WRITE_REG(hw, E1000_TDBAL(0), (tdba & 0x00000000ffffffffULL));
01479     E1000_WRITE_REG(hw, E1000_TDBAH(0), (tdba >> 32));
01480     E1000_WRITE_REG(hw, E1000_TDLEN(0), tdlen);
01481     E1000_WRITE_REG(hw, E1000_TDH(0), 0);
01482     E1000_WRITE_REG(hw, E1000_TDT(0), 0);
01483     adapter->tx_ring->tdh = E1000_REGISTER(hw, E1000_TDH(0));
01484     adapter->tx_ring->tdt = E1000_REGISTER(hw, E1000_TDT(0));
01485 
01486 
01487     /* Set the default values for the Tx Inter Packet Gap timer */
01488     if (adapter->hw.mac.type <= e1000_82547_rev_2 &&
01489         (hw->phy.media_type == e1000_media_type_fiber ||
01490          hw->phy.media_type == e1000_media_type_internal_serdes))
01491         tipg = DEFAULT_82543_TIPG_IPGT_FIBER;
01492     else
01493         tipg = DEFAULT_82543_TIPG_IPGT_COPPER;
01494 
01495     switch (hw->mac.type) {
01496     case e1000_82542:
01497         tipg = DEFAULT_82542_TIPG_IPGT;
01498         ipgr1 = DEFAULT_82542_TIPG_IPGR1;
01499         ipgr2 = DEFAULT_82542_TIPG_IPGR2;
01500         break;
01501     default:
01502         ipgr1 = DEFAULT_82543_TIPG_IPGR1;
01503         ipgr2 = DEFAULT_82543_TIPG_IPGR2;
01504         break;
01505     }
01506     tipg |= ipgr1 << E1000_TIPG_IPGR1_SHIFT;
01507     tipg |= ipgr2 << E1000_TIPG_IPGR2_SHIFT;
01508     E1000_WRITE_REG(hw, E1000_TIPG, tipg);
01509 
01510     /* Set the Tx Interrupt Delay register */
01511 
01512     E1000_WRITE_REG(hw, E1000_TIDV, adapter->tx_int_delay);
01513     if (adapter->flags & E1000_FLAG_HAS_INTR_MODERATION)
01514         E1000_WRITE_REG(hw, E1000_TADV, adapter->tx_abs_int_delay);
01515 
01516     /* Program the Transmit Control Register */
01517 
01518     tctl = E1000_READ_REG(hw, E1000_TCTL);
01519     tctl &= ~E1000_TCTL_CT;
01520     tctl |= E1000_TCTL_PSP | E1000_TCTL_RTLC |
01521         (E1000_COLLISION_THRESHOLD << E1000_CT_SHIFT);
01522 
01523     e1000_config_collision_dist(hw);
01524 
01525     /* Setup Transmit Descriptor Settings for eop descriptor */
01526     adapter->txd_cmd = E1000_TXD_CMD_EOP | E1000_TXD_CMD_IFCS;
01527 
01528     /* only set IDE if we are delaying interrupts using the timers */
01529     if (adapter->tx_int_delay)
01530         adapter->txd_cmd |= E1000_TXD_CMD_IDE;
01531 
01532     if (hw->mac.type < e1000_82543)
01533         adapter->txd_cmd |= E1000_TXD_CMD_RPS;
01534     else
01535         adapter->txd_cmd |= E1000_TXD_CMD_RS;
01536 
01537     /* Cache if we're 82544 running in PCI-X because we'll
01538      * need this to apply a workaround later in the send path. */
01539     if (hw->mac.type == e1000_82544 &&
01540         hw->bus.type == e1000_bus_type_pcix)
01541         adapter->pcix_82544 = 1;
01542 
01543     E1000_WRITE_REG(hw, E1000_TCTL, tctl);
01544 
01545 }

01718     struct e1000_hw *hw = &adapter->hw;
01719     u32 rdlen, rctl, rxcsum;
01720 #ifdef CONFIG_E1000_NAPI
01721     if ((adapter->netdev->mtu > MAXIMUM_ETHERNET_VLAN_SIZE) &&
01722        (hw->mac.type != e1000_82544 )) {

01724                 sizeof(struct e1000_rx_desc);
01725         adapter->clean_rx = e1000_clean_jumbo_rx_irq;
01726         adapter->alloc_rx_buf = e1000_alloc_jumbo_rx_buffers;
01727     } else
01728 #endif /* CONFIG_E1000_NAPI */

01732         adapter->clean_rx = e1000_clean_rx_irq;
01733         adapter->alloc_rx_buf = e1000_alloc_rx_buffers;
01734     }
01735

01737     rctl = E1000_READ_REG(hw, E1000_RCTL);
01738     E1000_WRITE_REG(hw, E1000_RCTL, rctl & ~E1000_RCTL_EN);
01739     /* do not flush or delay here, causes some strange problem
01740      * on SERDES connected SFP modules */
01741

01744

01746         E1000_WRITE_REG(hw, E1000_RADV, adapter->rx_abs_int_delay);

01749                 1000000000 / (adapter->itr * 256));
01750     }
01751 
01752     /* Setup the HW Rx Head and Tail Descriptor Pointers and
01753      * the Base and Length of the Rx Descriptor Ring */
01754     rdba = adapter->rx_ring->dma;
01755     E1000_WRITE_REG(hw, E1000_RDBAL(0), (rdba & 0x00000000ffffffffULL));
01756     E1000_WRITE_REG(hw, E1000_RDBAH(0), (rdba >> 32));
01757     E1000_WRITE_REG(hw, E1000_RDLEN(0), rdlen);
01758     E1000_WRITE_REG(hw, E1000_RDH(0), 0);
01759     E1000_WRITE_REG(hw, E1000_RDT(0), 0);
01760     adapter->rx_ring->rdh = E1000_REGISTER(hw, E1000_RDH(0));
01761     adapter->rx_ring->rdt = E1000_REGISTER(hw, E1000_RDT(0));
01762 
01763     if (hw->mac.type >= e1000_82543) {
01764         /* Enable 82543 Receive Checksum Offload for TCP and UDP */
01765         rxcsum = E1000_READ_REG(hw, E1000_RXCSUM);
01766         if (adapter->rx_csum == true) {

02211 bool e1000_has_link(struct e1000_adapter *adapter)
02212 {
02213     struct e1000_hw *hw = &adapter->hw;
02214     bool link_active = false;
02215     s32 ret_val = 0;
02216 
02217     /* get_link_status is set on LSC (link status) interrupt or
02218      * rx sequence error interrupt.  get_link_status will stay
02219      * false until the e1000_check_for_link establishes link
02220      * for copper adapters ONLY
02221      */
02222     switch (hw->phy.media_type) {
02223     case e1000_media_type_copper:
02224         if (hw->mac.get_link_status) {
02225             ret_val = e1000_check_for_link(hw);
02226             link_active = !hw->mac.get_link_status;
02227         } else {
02228             link_active = true;
02229         }
02230         break;
02231     case e1000_media_type_fiber:
02232         ret_val = e1000_check_for_link(hw);
02233         link_active = !!(E1000_READ_REG(hw, E1000_STATUS) &
02234                          E1000_STATUS_LU);
02235         break;
02236     case e1000_media_type_internal_serdes:
02237         ret_val = e1000_check_for_link(hw);
02238         link_active = adapter->hw.mac.serdes_has_link;
02239         break;
02240     default:
02241     case e1000_media_type_unknown:
02242         break;
02243     }
02244 
02245     if ((ret_val == E1000_ERR_PHY) && (hw->phy.type == e1000_phy_igp_3) &&
02246         (E1000_READ_REG(&adapter->hw, E1000_CTRL) & E1000_PHY_CTRL_GBE_DISABLE)) {
02247         DPRINTK(LINK, INFO,
02248             "Gigabit has been disabled, downgrading speed\n");
02249     }
02250 
02251     return link_active;
02252 }

02266 static void e1000_watchdog_task(struct work_struct *work)
02267 {
02268     struct e1000_adapter *adapter = container_of(work,
02269                                     struct e1000_adapter, watchdog_task);
02270     struct net_device *netdev = adapter->netdev;
02271     struct e1000_mac_info *mac = &adapter->hw.mac;
02272     struct e1000_tx_ring *tx_ring;
02273     u32 link, tctl;
02274     int tx_pending = 0;
02275 
02276     link = e1000_has_link(adapter);
02277     if ((netif_carrier_ok(netdev)) && link)
02278         goto link_up;
02279 
02280     if (link) {
02281         if (!netif_carrier_ok(netdev)) {
02282             u32 ctrl;
02283 #ifdef SIOCGMIIPHY
02284             /* update snapshot of PHY registers on LSC */
02285             e1000_phy_read_status(adapter);
02286 #endif
02287             e1000_get_speed_and_duplex(&adapter->hw,
02288                                        &adapter->link_speed,
02289                                        &adapter->link_duplex);
02290 
02291             ctrl = E1000_READ_REG(&adapter->hw, E1000_CTRL);
02292             DPRINTK(LINK, INFO, "NIC Link is Up %d Mbps %s, "
02293                     "Flow Control: %s\n",
02294                     adapter->link_speed,
02295                     adapter->link_duplex == FULL_DUPLEX ?
02296                     "Full Duplex" : "Half Duplex",
02297                     ((ctrl & E1000_CTRL_TFCE) && (ctrl &
02298                     E1000_CTRL_RFCE)) ? "RX/TX" : ((ctrl &
02299                     E1000_CTRL_RFCE) ? "RX" : ((ctrl &
02300                     E1000_CTRL_TFCE) ? "TX" : "None" )));
02301 
02302             /* adjust the timeout factor according to speed/duplex */
02303             adapter->tx_timeout_factor = 1;
02304             switch (adapter->link_speed) {
02305             case SPEED_10:
02306                 adapter->tx_timeout_factor = 16;
02307                 break;
02308             case SPEED_100:
02309                 /* maybe add some timeout factor ? */
02310                 break;
02311             }
02312 
02313 #ifdef NETIF_F_TSO
02314             /* disable TSO for pcie and 10/100 speeds, to avoid
02315              * some hardware issues */
02316             if (!(adapter->flags & E1000_FLAG_TSO_FORCE) &&
02317                 adapter->hw.bus.type == e1000_bus_type_pci_express){
02318                 switch (adapter->link_speed) {
02319                 case SPEED_10:
02320                 case SPEED_100:
02321                     DPRINTK(PROBE,INFO,
02322                         "10/100 speed: disabling TSO\n");
02323                     netdev->features &= ~NETIF_F_TSO;
02324 #ifdef NETIF_F_TSO6
02325                     netdev->features &= ~NETIF_F_TSO6;
02326 #endif
02327                     break;
02328                 case SPEED_1000:
02329                     netdev->features |= NETIF_F_TSO;
02330 #ifdef NETIF_F_TSO6
02331                     netdev->features |= NETIF_F_TSO6;
02332 #endif
02333                     break;
02334                 default:
02335                     /* oops */
02336                     break;
02337                 }
02338             }
02339 #endif
02340 
02341             /* enable transmits in the hardware, need to do this
02342              * after setting TARC0 */
02343             tctl = E1000_READ_REG(&adapter->hw, E1000_TCTL);
02344             tctl |= E1000_TCTL_EN;
02345             E1000_WRITE_REG(&adapter->hw, E1000_TCTL, tctl);
02346 
02347             netif_carrier_on(netdev);
02348             netif_wake_queue(netdev);
02349 
02350             if (!test_bit(__E1000_DOWN, &adapter->state))
02351                 mod_timer(&adapter->phy_info_timer,
02352                           round_jiffies(jiffies + 2 * HZ));
02353             adapter->smartspeed = 0;
02354         }
02355     } else {
02356         if (netif_carrier_ok(netdev)) {
02357             adapter->link_speed = 0;
02358             adapter->link_duplex = 0;
02359             DPRINTK(LINK, INFO, "NIC Link is Down\n");
02360             netif_carrier_off(netdev);
02361             netif_stop_queue(netdev);
02362 
02363             if (!test_bit(__E1000_DOWN, &adapter->state))
02364                 mod_timer(&adapter->phy_info_timer,
02365                           round_jiffies(jiffies + 2 * HZ));
02366         }
02367 
02368         e1000_smartspeed(adapter);
02369     }
02370 
02371 link_up:
02372     e1000_update_stats(adapter);
02373 
02374     mac->tx_packet_delta = adapter->stats.tpt - adapter->tpt_old;
02375     adapter->tpt_old = adapter->stats.tpt;
02376     mac->collision_delta = adapter->stats.colc - adapter->colc_old;
02377     adapter->colc_old = adapter->stats.colc;
02378 
02379     adapter->gorc = adapter->stats.gorc - adapter->gorc_old;
02380     adapter->gorc_old = adapter->stats.gorc;
02381     adapter->gotc = adapter->stats.gotc - adapter->gotc_old;
02382     adapter->gotc_old = adapter->stats.gotc;
02383 
02384     e1000_update_adaptive(&adapter->hw);
02385 
02386     if (!netif_carrier_ok(netdev)) {
02387         tx_ring = adapter->tx_ring;
02388         tx_pending |= (E1000_DESC_UNUSED(tx_ring) +
02389                        tx_ring->step  < tx_ring->count);
02390         if (tx_pending) {
02391             /* We've lost link, so the controller stops DMA,
02392              * but we've got queued Tx work that's never going
02393              * to get done, so reset controller to flush Tx.
02394              * (Do the reset outside of interrupt context). */
02395             adapter->tx_timeout_count++;
02396             schedule_work(&adapter->reset_task);
02397         }
02398     }
02399 
02400     /* Simple mode for Interrupt Throttle Rate (ITR) */
02401     if (adapter->hw.mac.type >= e1000_82540 && adapter->itr_setting == 4) {
02402         /*
02403          * Symmetric Tx/Rx gets a reduced ITR=2000;
02404          * Total asymmetrical Tx or Rx gets ITR=8000;
02405          * everyone else is between 2000-8000.
02406          */
02407         u32 goc = (adapter->gotc + adapter->gorc) / 10000;
02408         u32 dif = (adapter->gotc > adapter->gorc ?
02409                 adapter->gotc - adapter->gorc :
02410                 adapter->gorc - adapter->gotc) / 10000;
02411         u32 itr = goc > 0 ? (dif * 6000 / goc + 2000) : 8000;
02412 
02413         E1000_WRITE_REG(&adapter->hw, E1000_ITR, 1000000000 / (itr * 256));
02414     }
02415 
02416     /* Cause software interrupt to ensure rx ring is cleaned */
02417     E1000_WRITE_REG(&adapter->hw, E1000_ICS, E1000_ICS_RXDMT0);
02418 
02419     /* Force detection of hung controller every watchdog period */
02420     adapter->detect_tx_hung = true;
02421 
02422     /* Reset the timer */
02423     if (!test_bit(__E1000_DOWN, &adapter->state))
02424         mod_timer(&adapter->watchdog_timer,
02425                   round_jiffies(jiffies + 2 * HZ));
02426 }

02452 static unsigned int e1000_update_itr(struct e1000_adapter *adapter,
02453                                      u16 itr_setting, int packets,
02454                                      int bytes)
02455 {
02456     unsigned int retval = itr_setting;
02457 
02458     if (unlikely(!(adapter->flags & E1000_FLAG_HAS_INTR_MODERATION)))
02459         goto update_itr_done;
02460 
02461     if (packets == 0)
02462         goto update_itr_done;
02463 
02464     switch (itr_setting) {
02465     case lowest_latency:
02466         /* handle TSO and jumbo frames */
02467         if (bytes/packets > 8000)
02468             retval = bulk_latency;
02469         else if ((packets < 5) && (bytes > 512)) {
02470             retval = low_latency;
02471         }
02472         break;
02473     case low_latency:  /* 50 usec aka 20000 ints/s */
02474         if (bytes > 10000) {
02475             /* this if handles the TSO accounting */
02476             if (bytes/packets > 8000) {
02477                 retval = bulk_latency;
02478             } else if ((packets < 10) || ((bytes/packets) > 1200)) {
02479                 retval = bulk_latency;
02480             } else if ((packets > 35)) {
02481                 retval = lowest_latency;
02482             }
02483         } else if (bytes/packets > 2000) {
02484             retval = bulk_latency;
02485         } else if (packets <= 2 && bytes < 512) {
02486             retval = lowest_latency;
02487         }
02488         break;
02489     case bulk_latency: /* 250 usec aka 4000 ints/s */
02490         if (bytes > 25000) {
02491             if (packets > 35) {
02492                 retval = low_latency;
02493             }
02494         } else if (bytes < 6000) {
02495             retval = low_latency;
02496         }
02497         break;
02498     }
02499 
02500 update_itr_done:
02501     return retval;
02502 }

02504 static void e1000_set_itr(struct e1000_adapter *adapter)
02505 {
02506     struct e1000_hw *hw = &adapter->hw;
02507     u16 current_itr;
02508     u32 new_itr = adapter->itr;
02509 
02510     if (unlikely(!(adapter->flags & E1000_FLAG_HAS_INTR_MODERATION)))
02511         return;
02512 
02513     /* for non-gigabit speeds, just fix the interrupt rate at 4000 */
02514     if (unlikely(adapter->link_speed != SPEED_1000)) {
02515         current_itr = 0;
02516         new_itr = 4000;
02517         goto set_itr_now;
02518     }
02519 
02520     adapter->tx_itr = e1000_update_itr(adapter,
02521                                 adapter->tx_itr,
02522                                 adapter->total_tx_packets,
02523                                 adapter->total_tx_bytes);
02524     /* conservative mode (itr 3) eliminates the lowest_latency setting */
02525     if (adapter->itr_setting == 3 && adapter->tx_itr == lowest_latency)
02526         adapter->tx_itr = low_latency;
02527 
02528     adapter->rx_itr = e1000_update_itr(adapter,
02529                                 adapter->rx_itr,
02530                                 adapter->total_rx_packets,
02531                                 adapter->total_rx_bytes);
02532     /* conservative mode (itr 3) eliminates the lowest_latency setting */
02533     if (adapter->itr_setting == 3 && adapter->rx_itr == lowest_latency)
02534         adapter->rx_itr = low_latency;
02535 
02536     current_itr = max(adapter->rx_itr, adapter->tx_itr);
02537 
02538     switch (current_itr) {
02539     /* counts and packets in update_itr are dependent on these numbers */
02540     case lowest_latency:
02541         new_itr = 70000;
02542         break;
02543     case low_latency:
02544         new_itr = 20000; /* aka hwitr = ~200 */
02545         break;
02546     case bulk_latency:
02547         new_itr = 4000;
02548         break;
02549     default:
02550         break;
02551     }
02552 
02553 set_itr_now:
02554     if (new_itr != adapter->itr) {
02555         /* this attempts to bias the interrupt rate towards Bulk
02556          * by adding intermediate steps when interrupt rate is
02557          * increasing */
02558         new_itr = new_itr > adapter->itr ?
02559                      min(adapter->itr + (new_itr >> 2), new_itr) :
02560                      new_itr;
02561         adapter->itr = new_itr;
02562         E1000_WRITE_REG(hw, E1000_ITR, 1000000000 / (new_itr * 256));
02563     }
02564 
02565     return;
02566 }

02706 static int e1000_tx_map(struct e1000_adapter *adapter,
02707                         struct e1000_tx_ring *tx_ring,
02708                         struct sk_buff *skb, unsigned int first,
02709                         unsigned int max_per_txd, unsigned int nr_frags,
02710                         unsigned int mss)
02711 {
02712     struct e1000_buffer *buffer_info;
02713     unsigned int len = skb->len;
02714     unsigned int offset = 0, size, count = 0, i;
02715 #ifdef MAX_SKB_FRAGS
02716     unsigned int f;
02717     len -= skb->data_len;
02718 #endif
02719 
02720     i = tx_ring->next_to_use;
02721 
02722     while (len) {
02723         buffer_info = &tx_ring->buffer_info[i];
02724         size = min(len, max_per_txd);
02725 #ifdef NETIF_F_TSO
02726         /* Workaround for Controller erratum --
02727          * descriptor for non-tso packet in a linear SKB that follows a
02728          * tso gets written back prematurely before the data is fully
02729          * DMA'd to the controller */
02730         if (tx_ring->last_tx_tso && !skb_is_gso(skb)) {
02731             tx_ring->last_tx_tso = 0;
02732             if (!skb->data_len)
02733                 size -= 4;
02734         }
02735 
02736         /* Workaround for premature desc write-backs
02737          * in TSO mode.  Append 4-byte sentinel desc */
02738         if (unlikely(mss && !nr_frags && size == len && size > 8))
02739             size -= 4;
02740 #endif
02741         /* work-around for errata 10 and it applies
02742          * to all controllers in PCI-X mode
02743          * The fix is to make sure that the first descriptor of a
02744          * packet is smaller than 2048 - 16 - 16 (or 2016) bytes
02745          */
02746         if (unlikely((adapter->hw.bus.type == e1000_bus_type_pcix) &&
02747                         (size > 2015) && count == 0))
02748                 size = 2015;
02749 
02750         /* Workaround for potential 82544 hang in PCI-X.  Avoid
02751          * terminating buffers within evenly-aligned dwords. */
02752         if (unlikely(adapter->pcix_82544 &&
02753            !((unsigned long)(skb->data + offset + size - 1) & 4) &&
02754            size > 4))
02755             size -= 4;
02756 
02757         buffer_info->length = size;
02758         /* set time_stamp *before* dma to help avoid a possible race */
02759         buffer_info->time_stamp = jiffies;
02760         buffer_info->dma =
02761             dma_map_single(pci_dev_to_dev(adapter->pdev),
02762                        skb->data + offset,
02763                        size,
02764                        DMA_TO_DEVICE);
02765         buffer_info->mapped_as_page = false;
02766         buffer_info->next_to_watch = i;
02767 
02768         len -= size;
02769         offset += size;
02770         count++;
02771         E1000_TX_DESC_INC(tx_ring,i);
02772     }
02773 
02774 #ifdef MAX_SKB_FRAGS
02775     for (f = 0; f < nr_frags; f++) {
02776         struct skb_frag_struct *frag;
02777 
02778         frag = &skb_shinfo(skb)->frags[f];
02779         len = frag->size;
02780         offset = frag->page_offset;
02781 
02782         while (len) {
02783             buffer_info = &tx_ring->buffer_info[i];
02784             size = min(len, max_per_txd);
02785 #ifdef NETIF_F_TSO
02786             /* Workaround for premature desc write-backs
02787              * in TSO mode.  Append 4-byte sentinel desc */
02788             if (unlikely(mss && f == (nr_frags-1) && size == len && size > 8))
02789                 size -= 4;
02790 #endif
02791             /* Workaround for potential 82544 hang in PCI-X.
02792              * Avoid terminating buffers within evenly-aligned
02793              * dwords. */
02794             if (unlikely(adapter->pcix_82544 &&
02795                 !((unsigned long)(page_to_phys(frag->page) + offset
02796                                   + size - 1) & 4) &&
02797                 size > 4))
02798                 size -= 4;
02799 
02800             buffer_info->length = size;
02801             buffer_info->time_stamp = jiffies;
02802             buffer_info->dma =
02803                 dma_map_page(pci_dev_to_dev(adapter->pdev),
02804                          frag->page,
02805                          offset,
02806                          size,
02807                          DMA_TO_DEVICE);
02808             buffer_info->mapped_as_page = true;
02809             buffer_info->next_to_watch = i;
02810 
02811             len -= size;
02812             offset += size;
02813             count++;
02814             E1000_TX_DESC_INC(tx_ring,i);
02815         }
02816     }
02817 #endif
02818     E1000_TX_DESC_DEC(tx_ring,i);
02819     tx_ring->buffer_info[i].skb = skb;
02820     tx_ring->buffer_info[first].next_to_watch = i;
02821 
02822     return count;
02823 }

02825 static void e1000_tx_queue(struct e1000_adapter *adapter,
02826                            struct e1000_tx_ring *tx_ring,
02827                            int tx_flags, int count)
02828 {
02829     struct e1000_tx_desc *tx_desc = NULL;
02830     struct e1000_buffer *buffer_info;
02831     u32 txd_upper = 0, txd_lower = E1000_TXD_CMD_IFCS;
02832     unsigned int i;
02833 
02834     if (likely(tx_flags & E1000_TX_FLAGS_TSO)) {
02835         txd_lower |= E1000_TXD_CMD_DEXT | E1000_TXD_DTYP_D |
02836                      E1000_TXD_CMD_TSE;
02837         txd_upper |= E1000_TXD_POPTS_TXSM << 8;
02838 
02839         if (likely(tx_flags & E1000_TX_FLAGS_IPV4))
02840             txd_upper |= E1000_TXD_POPTS_IXSM << 8;
02841     }
02842 
02843     if (likely(tx_flags & E1000_TX_FLAGS_CSUM)) {
02844         txd_lower |= E1000_TXD_CMD_DEXT | E1000_TXD_DTYP_D;
02845         txd_upper |= E1000_TXD_POPTS_TXSM << 8;
02846     }
02847 
02848     if (unlikely(tx_flags & E1000_TX_FLAGS_VLAN)) {
02849         txd_lower |= E1000_TXD_CMD_VLE;
02850         txd_upper |= (tx_flags & E1000_TX_FLAGS_VLAN_MASK);
02851     }
02852 
02853     i = tx_ring->next_to_use;
02854 
02855     while (count--) {
02856         buffer_info = &tx_ring->buffer_info[i];
02857         tx_desc = E1000_TX_DESC(*tx_ring, i);
02858         tx_desc->buffer_addr = cpu_to_le64(buffer_info->dma);
02859         tx_desc->lower.data =
02860             cpu_to_le32(txd_lower | buffer_info->length);
02861         tx_desc->upper.data = cpu_to_le32(txd_upper);
02862         E1000_TX_DESC_INC(tx_ring,i);
02863     }
02864 
02865     tx_desc->lower.data |= cpu_to_le32(adapter->txd_cmd);
02866 
02867     /* Force memory writes to complete before letting h/w
02868      * know there are new descriptors to fetch.  (Only
02869      * applicable for weak-ordered memory model archs,
02870      * such as IA-64). */
02871     wmb();
02872 
02873     tx_ring->next_to_use = i;
02874     writel(i, adapter->hw.hw_addr + tx_ring->tdt);
02875     /* we need this if more than one processor can write to our tail
02876      * at a time, it synchronizes IO on IA64/Altix systems */
02877     mmiowb();
02878 }

02947 #define TXD_USE_COUNT(S, X) (((S) >> (X)) + 1 )
02948 static int e1000_xmit_frame(struct sk_buff *skb, struct net_device *netdev)
02949 {
02950     struct e1000_adapter *adapter = netdev_priv(netdev);
02951     struct e1000_tx_ring *tx_ring = adapter->tx_ring;
02952     unsigned int max_txd_pwr = adapter->tx_desc_pwr;
02953     unsigned int first, max_per_txd = (1 << max_txd_pwr);
02954     unsigned int tx_flags = 0;
02955     unsigned int len = skb->len;
02956     unsigned int nr_frags = 0;
02957     unsigned int mss = 0;
02958     int count = 0;
02959     int tso;
02960 #ifdef MAX_SKB_FRAGS
02961     unsigned int f;
02962     len -= skb->data_len;
02963 #endif
02964 
02965     if (test_bit(__E1000_DOWN, &adapter->state)) {
02966         dev_kfree_skb_any(skb);
02967         return NETDEV_TX_OK;
02968     }
02969 
02970     if (unlikely(skb->len <= 0)) {
02971         dev_kfree_skb_any(skb);
02972         return NETDEV_TX_OK;
02973     }
02974     /* On PCIX HW, there have been rare TXHangs caused by custom apps
02975      * sending frames with skb->len == 16 (macdest+macsrc+protocol+2bytes).
02976      * Such frames are not transmitted by registered protocols, so are
02977      * only a problem for experimental code.
02978      * Pad all 16 byte packets with an additional byte to work-around this
02979      * problem case.
02980      */
02981     if (unlikely(skb->len == 16)) {
02982            skb_pad(skb,1);
02983            skb_put(skb,1);
02984     }
02985 
02986 
02987 #ifdef NETIF_F_TSO
02988     mss = skb_shinfo(skb)->gso_size;
02989     /* The controller does a simple calculation to
02990      * make sure there is enough room in the FIFO before
02991      * initiating the DMA for each buffer.  The calc is:
02992      * 4 = ceil(buffer len/mss).  To make sure we don't
02993      * overrun the FIFO, adjust the max buffer len if mss
02994      * drops. */
02995     if (mss) {
02996         u8 hdr_len;
02997         max_per_txd = min(mss << 2, max_per_txd);
02998         max_txd_pwr = fls(max_per_txd) - 1;
02999 
03000         /* TSO Workaround */
03001         hdr_len = skb_transport_offset(skb) + tcp_hdrlen(skb);
03002         if (skb->data_len && (hdr_len == (skb->len - skb->data_len))) {
03003             switch (adapter->hw.mac.type) {
03004                 unsigned int pull_size;
03005             case e1000_82544:
03006                 /* Make sure we have room to chop off 4 bytes,
03007                  * and that the end alignment will work out to
03008                  * this hardware's requirements
03009                  * NOTE: this is a TSO only workaround
03010                  * if end byte alignment not correct move us
03011                  * into the next dword */
03012                 if ((unsigned long)(skb_tail_pointer(skb) - 1) & 4)
03013                     break;
03014                 pull_size = min((unsigned int)4, skb->data_len);
03015                 if (!__pskb_pull_tail(skb, pull_size)) {
03016                     DPRINTK(DRV, ERR,
03017                         "__pskb_pull_tail failed.\n");
03018                     dev_kfree_skb_any(skb);
03019                     return NETDEV_TX_OK;
03020                 }
03021                 len = skb->len - skb->data_len;
03022                 break;
03023             default:
03024                 /* do nothing */
03025                 break;
03026             }
03027         }
03028     }
03029 
03030     /* reserve a descriptor for the offload context */
03031     if ((mss) || (skb->ip_summed == CHECKSUM_PARTIAL))
03032         count++;
03033     count++;
03034 #else
03035     if (skb->ip_summed == CHECKSUM_PARTIAL)
03036         count++;
03037 #endif
03038 
03039 #ifdef NETIF_F_TSO
03040     /* Controller Erratum workaround */
03041     if (!skb->data_len && tx_ring->last_tx_tso && !skb_is_gso(skb))
03042         count++;
03043 #endif
03044 
03045     count += TXD_USE_COUNT(len, max_txd_pwr);
03046 
03047     if (adapter->pcix_82544)
03048         count++;
03049 
03050     /* work-around for errata 10 and it applies to all controllers
03051      * in PCI-X mode, so add one more descriptor to the count
03052      */
03053     if (unlikely((adapter->hw.bus.type == e1000_bus_type_pcix) &&
03054             (len > 2015)))
03055         count++;
03056 
03057 #ifdef MAX_SKB_FRAGS
03058     nr_frags = skb_shinfo(skb)->nr_frags;
03059     for (f = 0; f < nr_frags; f++)
03060         count += TXD_USE_COUNT(skb_shinfo(skb)->frags[f].size,
03061                        max_txd_pwr);
03062     if (adapter->pcix_82544)
03063         count += nr_frags;
03064 
03065 #endif
03066 
03067     /* need: count + 2 desc gap to keep tail from touching
03068      * head, otherwise try next time */
03069     if (unlikely(e1000_maybe_stop_tx(netdev, tx_ring, count + 2)))
03070         return NETDEV_TX_BUSY;
03071 
03072     if (unlikely(adapter->hw.mac.type == e1000_82547)) {
03073         if (unlikely(e1000_82547_fifo_workaround(adapter, skb))) {
03074             netif_stop_queue(netdev);
03075             if (!test_bit(__E1000_DOWN, &adapter->state))
03076                 mod_timer(&adapter->tx_fifo_stall_timer,
03077                           jiffies + 1);
03078             return NETDEV_TX_BUSY;
03079         }
03080     }
03081 
03082 #ifdef NETIF_F_HW_VLAN_TX
03083     if (unlikely(adapter->vlgrp && vlan_tx_tag_present(skb))) {
03084         tx_flags |= E1000_TX_FLAGS_VLAN;
03085         tx_flags |= (vlan_tx_tag_get(skb) << E1000_TX_FLAGS_VLAN_SHIFT);
03086     }
03087 #endif
03088 
03089     first = tx_ring->next_to_use;
03090 
03091     tso = e1000_tso(adapter, tx_ring, skb);
03092     if (tso < 0) {
03093         dev_kfree_skb_any(skb);
03094         return NETDEV_TX_OK;
03095     }
03096 
03097     if (likely(tso)) {
03098         if (likely(adapter->hw.mac.type != e1000_82544))
03099             tx_ring->last_tx_tso = 1;
03100         tx_flags |= E1000_TX_FLAGS_TSO;
03101     } else if (likely(e1000_tx_csum(adapter, tx_ring, skb)))
03102         tx_flags |= E1000_TX_FLAGS_CSUM;
03103 
03104     if (likely(skb->protocol == htons(ETH_P_IP)))
03105         tx_flags |= E1000_TX_FLAGS_IPV4;
03106 
03107     e1000_tx_queue(adapter, tx_ring, tx_flags,
03108                    e1000_tx_map(adapter, tx_ring, skb, first,
03109                                 max_per_txd, nr_frags, mss));
03110 
03111     netdev->trans_start = jiffies;
03112 
03113     /* Make sure there is space in the ring for the next send. */
03114     e1000_maybe_stop_tx(netdev, tx_ring, MAX_SKB_FRAGS + 2);
03115 
03116     return NETDEV_TX_OK;
03117 }

03163 static int e1000_change_mtu(struct net_device *netdev, int new_mtu)
03164 {
03165     struct e1000_adapter *adapter = netdev_priv(netdev);
03166     int max_frame = new_mtu + ETH_HLEN + ETH_FCS_LEN;
03167 
03168     if ((new_mtu < 68) || (max_frame > MAX_JUMBO_FRAME_SIZE)) {
03169         DPRINTK(PROBE, ERR, "Invalid MTU setting\n");
03170         return -EINVAL;
03171     }
03172 
03173     /* Adapter-specific max frame size limits. */
03174     switch (adapter->hw.mac.type) {
03175     case e1000_undefined:
03176     case e1000_82542:
03177         if (max_frame > ETH_FRAME_LEN + ETH_FCS_LEN) {
03178             DPRINTK(PROBE, ERR, "Jumbo Frames not supported.\n");
03179             return -EINVAL;
03180         }
03181         break;
03182     default:
03183         /* Capable of supporting up to MAX_JUMBO_FRAME_SIZE limit. */
03184         break;
03185     }
03186 
03187     while (test_and_set_bit(__E1000_RESETTING, &adapter->state))
03188         msleep(1);
03189     /* e1000_down has a dependency on max_frame_size */
03190     adapter->max_frame_size = max_frame;
03191     if (netif_running(netdev))
03192         e1000_down(adapter);
03193 
03194     /* NOTE: netdev_alloc_skb reserves 16 bytes, and typically NET_IP_ALIGN
03195      * means we reserve 2 more, this pushes us to allocate from the next
03196      * larger slab size.
03197      * i.e. RXBUFFER_2048 --> size-4096 slab
03198      *  however with the new *_jumbo_rx* routines, jumbo receives will use
03199      *  fragmented skbs */
03200 
03201     /* Due to a HW errata, the 82544 should not use the jumbo routines, and
03202      * so will have to handle jumbo frames within one descriptor.
03203      */
03204     if (adapter->hw.mac.type == e1000_82544) {
03205         if (max_frame <= E1000_RXBUFFER_2048)
03206             adapter->rx_buffer_len = E1000_RXBUFFER_2048;
03207         else if (max_frame <= E1000_RXBUFFER_4096)
03208             adapter->rx_buffer_len = E1000_RXBUFFER_4096;
03209         else if (max_frame <= E1000_RXBUFFER_8192)
03210             adapter->rx_buffer_len = E1000_RXBUFFER_8192;
03211         else if (max_frame <= E1000_RXBUFFER_16384)
03212             adapter->rx_buffer_len = E1000_RXBUFFER_16384;
03213     } else {
03214         if (max_frame <= E1000_RXBUFFER_2048)
03215             adapter->rx_buffer_len = E1000_RXBUFFER_2048;
03216     #ifdef CONFIG_E1000_NAPI
03217         else
03218             adapter->rx_buffer_len = E1000_RXBUFFER_4096;
03219     #else
03220         else if (max_frame <= E1000_RXBUFFER_4096)
03221             adapter->rx_buffer_len = E1000_RXBUFFER_4096;
03222         else if (max_frame <= E1000_RXBUFFER_8192)
03223             adapter->rx_buffer_len = E1000_RXBUFFER_8192;
03224         else if (max_frame <= E1000_RXBUFFER_16384)
03225             adapter->rx_buffer_len = E1000_RXBUFFER_16384;
03226     #endif
03227     }
03228     /* adjust allocation if LPE protects us, and we aren't using SBP */
03229     if (!e1000_tbi_sbp_enabled_82543(&adapter->hw) &&
03230         ((max_frame == ETH_FRAME_LEN + ETH_FCS_LEN) ||
03231          (max_frame == MAXIMUM_ETHERNET_VLAN_SIZE)))
03232         adapter->rx_buffer_len = MAXIMUM_ETHERNET_VLAN_SIZE;
03233 
03234     DPRINTK(PROBE, INFO, "changing MTU from %d to %d\n",
03235             netdev->mtu, new_mtu);
03236     netdev->mtu = new_mtu;
03237 
03238     if (netif_running(netdev))
03239         e1000_up(adapter);
03240     else
03241         e1000_reset(adapter);
03242 
03243     clear_bit(__E1000_RESETTING, &adapter->state);
03244 
03245     return 0;
03246 }

03437 static irqreturn_t e1000_intr(int irq, void *data)
03438 {
03439     struct net_device *netdev = data;
03440     struct e1000_adapter *adapter = netdev_priv(netdev);
03441     struct e1000_hw *hw = &adapter->hw;
03442     u32 icr = E1000_READ_REG(hw, E1000_ICR);
03443 
03444     if (unlikely((!icr) || test_bit(__E1000_DOWN, &adapter->state)))
03445         return IRQ_NONE;  /* Not our interrupt */
03446 
03447     if (unlikely(icr & (E1000_ICR_RXSEQ | E1000_ICR_LSC))) {
03448         hw->mac.get_link_status = 1;
03449 
03450         /* guard against interrupt when we're going down */
03451         if (!test_bit(__E1000_DOWN, &adapter->state))
03452             mod_timer(&adapter->watchdog_timer, jiffies + 1);
03453     }
03454 
03455 #ifdef CONFIG_E1000_NAPI
03456     if (likely(napi_schedule_prep(&adapter->rx_ring->napi))) {
03457         /* disable interrupts, without the synchronize_irq bit */
03458         E1000_WRITE_REG(hw, E1000_IMC, ~0);
03459         E1000_WRITE_FLUSH(&adapter->hw);
03460 
03461         adapter->total_tx_bytes = 0;
03462         adapter->total_tx_packets = 0;
03463         adapter->total_rx_bytes = 0;
03464         adapter->total_rx_packets = 0;
03465         __napi_schedule(&adapter->rx_ring->napi);
03466     }
03467 #else
03468     /* Writing IMC and IMS is needed for 82547.
03469      * Due to Hub Link bus being occupied, an interrupt
03470      * de-assertion message is not able to be sent.
03471      * When an interrupt assertion message is generated later,
03472      * two messages are re-ordered and sent out.
03473      * That causes APIC to think 82547 is in de-assertion
03474      * state, while 82547 is in assertion state, resulting
03475      * in dead lock. Writing IMC forces 82547 into
03476      * de-assertion state.
03477      */
03478     if (hw->mac.type == e1000_82547 || hw->mac.type == e1000_82547_rev_2)
03479         E1000_WRITE_REG(hw, E1000_IMC, ~0);
03480 
03481     adapter->total_tx_bytes = 0;
03482     adapter->total_rx_bytes = 0;
03483     adapter->total_tx_packets = 0;
03484     adapter->total_rx_packets = 0;
03485 
03486     adapter->clean_rx(adapter, adapter->rx_ring);
03487     e1000_clean_tx_irq(adapter, adapter->tx_ring);
03488 
03489     if (likely(adapter->itr_setting & 3))
03490         e1000_set_itr(adapter);
03491 
03492     if (hw->mac.type == e1000_82547 || hw->mac.type == e1000_82547_rev_2)
03493         if (!test_bit(__E1000_DOWN, &adapter->state))
03494             e1000_irq_enable(adapter);
03495 
03496 #endif
03497     return IRQ_HANDLED;
03498 }

03546 static bool e1000_clean_tx_irq(struct e1000_adapter *adapter,
03547                                struct e1000_tx_ring *tx_ring)
03548 {
03549     struct net_device *netdev = adapter->netdev;
03550     struct e1000_tx_desc *tx_desc, *eop_desc;
03551     struct e1000_buffer *buffer_info;
03552     unsigned int i, eop;
03553 #ifdef CONFIG_E1000_NAPI
03554     unsigned int count = 0;
03555 #endif
03556     bool cleaned = false;
03557     bool retval = true;
03558     unsigned int total_tx_bytes=0, total_tx_packets=0;
03559 
03560 
03561     i = tx_ring->next_to_clean;
03562     eop = tx_ring->buffer_info[i].next_to_watch;
03563     eop_desc = E1000_TX_DESC(*tx_ring, eop);
03564 
03565     while (eop_desc->upper.data & cpu_to_le32(E1000_TXD_STAT_DD)) {
03566         for (cleaned = false; !cleaned; ) {
03567             tx_desc = E1000_TX_DESC(*tx_ring, i);
03568             buffer_info = &tx_ring->buffer_info[i];
03569             cleaned = (i == eop);
03570 
03571             if (cleaned) {
03572                 struct sk_buff *skb = buffer_info->skb;
03573 #ifdef NETIF_F_TSO
03574                 unsigned int segs, bytecount;
03575                 segs = skb_shinfo(skb)->gso_segs ?: 1;
03576                 /* multiply data chunks by size of headers */
03577                 bytecount = ((segs - 1) * skb_headlen(skb)) +
03578                             skb->len;
03579                 total_tx_packets += segs;
03580                 total_tx_bytes += bytecount;
03581 #else
03582                 total_tx_packets++;
03583                 total_tx_bytes += skb->len;
03584 #endif
03585             }
03586             e1000_unmap_and_free_tx_resource(adapter, buffer_info);
03587             tx_desc->upper.data = 0;
03588             E1000_TX_DESC_INC(tx_ring,i);
03589         }
03590 
03591         eop = tx_ring->buffer_info[i].next_to_watch;
03592         eop_desc = E1000_TX_DESC(*tx_ring, eop);
03593 #ifdef CONFIG_E1000_NAPI
03594 #define E1000_TX_WEIGHT 64
03595         /* weight of a sort for tx, to avoid endless transmit cleanup */
03596         if (count++ == E1000_TX_WEIGHT) {
03597             retval = false;
03598             break;
03599         }
03600 #endif
03601     }
03602 
03603     tx_ring->next_to_clean = i;
03604 
03605 #define TX_WAKE_THRESHOLD 32
03606     if (unlikely(cleaned && netif_carrier_ok(netdev) &&
03607              E1000_DESC_UNUSED(tx_ring) >= TX_WAKE_THRESHOLD)) {
03608         /* Make sure that anybody stopping the queue after this
03609          * sees the new next_to_clean.
03610          */

03669 static void e1000_rx_checksum(struct e1000_adapter *adapter, u32 status_err,
03670                               u32 csum, struct sk_buff *skb)
03671 {
03672     u16 status = (u16)status_err;
03673     u8 errors = (u8)(status_err >> 24);
03674     skb->ip_summed = CHECKSUM_NONE;
03675 
03676     /* 82543 or newer only */
03677     if (unlikely(adapter->hw.mac.type < e1000_82543)) return;
03678     /* Ignore Checksum bit is set */
03679     if (unlikely(status & E1000_RXD_STAT_IXSM)) return;
03680     /* TCP/UDP checksum error bit is set */
03681     if (unlikely(errors & E1000_RXD_ERR_TCPE)) {
03682         /* let the stack verify checksum errors */
03683         adapter->hw_csum_err++;
03684         return;
03685     }
03686     /* TCP/UDP Checksum has not been calculated */
03687     if (adapter->hw.mac.type <= e1000_82547_rev_2) {
03688         if (!(status & E1000_RXD_STAT_TCPCS))
03689             return;
03690     } else {
03691         if (!(status & (E1000_RXD_STAT_TCPCS | E1000_RXD_STAT_UDPCS)))
03692             return;
03693     }
03694     /* It must be a TCP or UDP packet with a valid checksum */
03695     if (likely(status & E1000_RXD_STAT_TCPCS)) {
03696         /* TCP checksum is good */
03697         skb->ip_summed = CHECKSUM_UNNECESSARY;
03698     }
03699     adapter->hw_csum_good++;
03700 }

03810

03840         if (unlikely((status & E1000_RXD_STAT_EOP) &&
03841             (rx_desc->errors & E1000_RXD_ERR_FRAME_ERR_MASK))) {

03857                 /* recycle both page and skb */

03865             }
03866         }
03867 
03868 #define rxtop rx_ring->rx_skb_top

03870             /* this descriptor is only the beginning (or middle) */
03871             if (!rxtop) {
03872                 /* this is the beginning of a chain */
03873                 rxtop = skb;
03874                 skb_fill_page_desc(rxtop, 0, buffer_info->page,
03875                                    0, length);
03876             } else {
03877                 /* this is the middle of a chain */
03878                 skb_fill_page_desc(rxtop,
03879                     skb_shinfo(rxtop)->nr_frags,
03880                     buffer_info->page, 0, length);
03881                 /* re-use the skb, only consumed the page */
03882                 buffer_info->skb = skb;
03883             }
03884             e1000_consume_page(buffer_info, rxtop, length);
03885             goto next_desc;
03886         } else {
03887             if (rxtop) {
03888                 /* end of the chain */
03889                 skb_fill_page_desc(rxtop,
03890                     skb_shinfo(rxtop)->nr_frags,
03891                     buffer_info->page, 0, length);
03892                 /* re-use the current skb, we only consumed the
03893                  * page */
03894                 buffer_info->skb = skb;
03895                 skb = rxtop;
03896                 rxtop = NULL;
03897                 e1000_consume_page(buffer_info, skb, length);

03900                  * copybreak to save the put_page/alloc_page */
03901                 if (length <= copybreak &&
03902                     skb_tailroom(skb) >= length) {
03903                     u8 *vaddr;
03904                     vaddr = kmap_atomic(buffer_info->page,
03905                                        KM_SKB_DATA_SOFTIRQ);
03906                     memcpy(skb_tail_pointer(skb), vaddr, length);
03907                     kunmap_atomic(vaddr,
03908                                   KM_SKB_DATA_SOFTIRQ);
03909                     /* re-use the page, so don't erase
03910                      * buffer_info->page */
03911                     skb_put(skb, length);
03912                 } else {
03913                     skb_fill_page_desc(skb, 0,
03914                                        buffer_info->page, 0,
03915                                            length);
03916                     e1000_consume_page(buffer_info, skb,
03917                                        length);
03918                 }
03919             }
03920         }
03921 
03922         /* Receive Checksum Offload XXX recompute due to CRC strip? */
03923         e1000_rx_checksum(adapter,
03924                           (u32)(status) |
03925                           ((u32)(rx_desc->errors) << 24),
03926                           le16_to_cpu(rx_desc->csum), skb);
03927 
03928         pskb_trim(skb, skb->len - 4);
03929 
03930         /* probably a little skewed due to removing CRC */
03931         total_rx_bytes += skb->len;
03932         total_rx_packets++;
03933 
03934         /* eth type trans needs skb->data to point to something */
03935         if (!pskb_may_pull(skb, ETH_HLEN)) {
03936             DPRINTK(DRV, ERR, "__pskb_pull_tail failed.\n");
03937             dev_kfree_skb(skb);
03938             goto next_desc;
03939         }
03940 
03941         e1000_receive_skb(adapter, status, rx_desc->special, skb);
03942 
03943         netdev->last_rx = jiffies;
03944 
03945 next_desc:
03946         rx_desc->status = 0;
03947 
03948         /* return some buffers to hardware, one at a time is too slow */
03949         if (unlikely(cleaned_count >= E1000_RX_BUFFER_WRITE)) {
03950             adapter->alloc_rx_buf(adapter, rx_ring, cleaned_count);
03951             cleaned_count = 0;
03952         }
03953 
03954         /* use prefetched values */
03955         rx_desc = next_rxd;
03956         buffer_info = next_buffer;
03957     }
03958     rx_ring->next_to_clean = i;
03959 
03960     cleaned_count = E1000_DESC_UNUSED(rx_ring);
03961     if (cleaned_count)
03962         adapter->alloc_rx_buf(adapter, rx_ring, cleaned_count);
03963

03967     adapter->net_stats.rx_packets += total_rx_packets;

04150 static void e1000_alloc_jumbo_rx_buffers(struct e1000_adapter *adapter,
04151                                          struct e1000_rx_ring *rx_ring,
04152                                          int cleaned_count)
04153 {
04154     struct net_device *netdev = adapter->netdev;
04155     struct pci_dev *pdev = adapter->pdev;
04156     struct e1000_rx_desc *rx_desc;
04157     struct e1000_rx_buffer *buffer_info;
04158     struct sk_buff *skb;
04159     unsigned int i;
04160     unsigned int bufsz = 256 -
04161                          NET_SKB_PAD /*for skb_reserve */ -
04162                          NET_IP_ALIGN;
04163 
04164     i = rx_ring->next_to_use;
04165     buffer_info = &rx_ring->buffer_info[i];
04166 
04167     while (cleaned_count--) {
04168         skb = buffer_info->skb;
04169         if (skb) {
04170             skb_trim(skb, 0);
04171             goto check_page;
04172         }
04173 
04174         skb = netdev_alloc_skb_ip_align(netdev, bufsz);
04175         if (unlikely(!skb)) {
04176             /* Better luck next round */
04177             adapter->alloc_rx_buff_failed++;
04178             break;
04179         }
04180 
04181         /* Fix for errata 23, can't cross 64kB boundary */
04182         if (!e1000_check_64k_bound(adapter, skb->data, bufsz)) {
04183             struct sk_buff *oldskb = skb;
04184             DPRINTK(PROBE, ERR, "skb align check failed: %u bytes "
04185                          "at %p\n", bufsz, skb->data);
04186             /* Try again, without freeing the previous */
04187             skb = netdev_alloc_skb_ip_align(netdev, bufsz);
04188             /* Failed allocation, critical failure */
04189             if (!skb) {
04190                 dev_kfree_skb(oldskb);
04191                 adapter->alloc_rx_buff_failed++;
04192                 break;
04193             }
04194 
04195             if (!e1000_check_64k_bound(adapter, skb->data, bufsz)) {
04196                 /* give up */
04197                 dev_kfree_skb(skb);
04198                 dev_kfree_skb(oldskb);
04199                 adapter->alloc_rx_buff_failed++;
04200                 break; /* while !buffer_info->skb */
04201             }
04202 
04203             /* Use new allocation */
04204             dev_kfree_skb(oldskb);
04205         }
04206 
04207         buffer_info->skb = skb;
04208 check_page:
04209         /* allocate a new page if necessary */
04210         if (!buffer_info->page) {
04211             buffer_info->page = alloc_page(GFP_ATOMIC);
04212             if (unlikely(!buffer_info->page)) {
04213                 adapter->alloc_rx_buff_failed++;
04214                 break;
04215             }
04216         }
04217 
04218         if (!buffer_info->dma)
04219             buffer_info->dma = dma_map_page(pci_dev_to_dev(pdev),
04220                                             buffer_info->page, 0,
04221                                             PAGE_SIZE,
04222                                             DMA_FROM_DEVICE);
04223 
04224         rx_desc = E1000_RX_DESC(*rx_ring, i);
04225         rx_desc->buffer_addr = cpu_to_le64(buffer_info->dma);
04226 
04227         if (unlikely(++i == rx_ring->count))
04228             i = 0;
04229         buffer_info = &rx_ring->buffer_info[i];
04230     }
04231 
04232     if (likely(rx_ring->next_to_use != i)) {
04233         rx_ring->next_to_use = i;
04234         if (unlikely(i-- == 0))
04235             i = (rx_ring->count - 1);
04236 
04237         /* Force memory writes to complete before letting h/w
04238          * know there are new descriptors to fetch.  (Only
04239          * applicable for weak-ordered memory model archs,
04240          * such as IA-64). */

04251 static void e1000_alloc_rx_buffers(struct e1000_adapter *adapter,
04252                                    struct e1000_rx_ring *rx_ring,
04253                                    int cleaned_count)
04254 {
04255     struct net_device *netdev = adapter->netdev;
04256     struct pci_dev *pdev = adapter->pdev;
04257     struct e1000_rx_desc *rx_desc;
04258     struct e1000_rx_buffer *buffer_info;
04259     struct sk_buff *skb;
04260     unsigned int i;
04261     unsigned int bufsz = adapter->rx_buffer_len;
04262 
04263     i = rx_ring->next_to_use;
04264     buffer_info = &rx_ring->buffer_info[i];
04265 
04266     while (cleaned_count--) {
04267         skb = buffer_info->skb;
04268         if (skb) {
04269             skb_trim(skb, 0);
04270             goto map_skb;
04271         }
04272 
04273         skb = netdev_alloc_skb_ip_align(netdev, bufsz);
04274         if (unlikely(!skb)) {
04275             /* Better luck next round */
04276             adapter->alloc_rx_buff_failed++;
04277             break;
04278         }
04279 
04280         /* Fix for errata 23, can't cross 64kB boundary */
04281         if (!e1000_check_64k_bound(adapter, skb->data, bufsz)) {
04282             struct sk_buff *oldskb = skb;
04283             DPRINTK(RX_ERR, ERR, "skb align check failed: %u bytes "
04284                          "at %p\n", bufsz, skb->data);
04285             /* Try again, without freeing the previous */
04286             skb = netdev_alloc_skb_ip_align(netdev, bufsz);
04287             /* Failed allocation, critical failure */
04288             if (!skb) {
04289                 dev_kfree_skb(oldskb);
04290                 adapter->alloc_rx_buff_failed++;
04291                 break;
04292             }
04293 
04294             if (!e1000_check_64k_bound(adapter, skb->data, bufsz)) {
04295                 /* give up */
04296                 dev_kfree_skb(skb);
04297                 dev_kfree_skb(oldskb);
04298                 adapter->alloc_rx_buff_failed++;
04299                 break; /* while !buffer_info->skb */
04300             }
04301 
04302             /* Use new allocation */
04303             dev_kfree_skb(oldskb);
04304         }
04305 
04306         buffer_info->skb = skb;
04307 map_skb:
04308         buffer_info->dma = dma_map_single(pci_dev_to_dev(pdev),
04309                           skb->data,
04310                           adapter->rx_buffer_len,
04311                           DMA_FROM_DEVICE);
04312 
04313         /* Fix for errata 23, can't cross 64kB boundary */
04314         if (!e1000_check_64k_bound(adapter,
04315                     (void *)(unsigned long)buffer_info->dma,
04316                     adapter->rx_buffer_len)) {
04317             DPRINTK(RX_ERR, ERR,
04318                 "dma align check failed: %u bytes at %p\n",
04319                 adapter->rx_buffer_len,
04320                 (void *)(unsigned long)buffer_info->dma);
04321             dev_kfree_skb(skb);
04322             buffer_info->skb = NULL;
04323 
04324             dma_unmap_single(pci_dev_to_dev(pdev), buffer_info->dma,
04325                      adapter->rx_buffer_len,
04326                      DMA_FROM_DEVICE);
04327             buffer_info->dma = 0;
04328 
04329             adapter->alloc_rx_buff_failed++;
04330             break; /* while !buffer_info->skb */
04331         }
04332         rx_desc = E1000_RX_DESC(*rx_ring, i);
04333         rx_desc->buffer_addr = cpu_to_le64(buffer_info->dma);
04334 
04335         if (unlikely(++i == rx_ring->count))
04336             i = 0;
04337         buffer_info = &rx_ring->buffer_info[i];
04338     }
04339 
04340     if (likely(rx_ring->next_to_use != i)) {
04341         rx_ring->next_to_use = i;
04342         if (unlikely(i-- == 0))
04343             i = (rx_ring->count - 1);
04344 
04345         /* Force memory writes to complete before letting h/w
04346          * know there are new descriptors to fetch.  (Only
04347          * applicable for weak-ordered memory model archs,
04348          * such as IA-64). */
04349         wmb();
04350         writel(i, adapter->hw.hw_addr + rx_ring->rdt);
04351     }
04352 }

04358 static void e1000_smartspeed(struct e1000_adapter *adapter)
04359 {
04360     struct e1000_mac_info *mac = &adapter->hw.mac;
04361     struct e1000_phy_info *phy = &adapter->hw.phy;
04362     u16 phy_status;
04363     u16 phy_ctrl;
04364 
04365     if ((phy->type != e1000_phy_igp) || !mac->autoneg ||
04366         !(phy->autoneg_advertised & ADVERTISE_1000_FULL))
04367         return;
04368 
04369     if (adapter->smartspeed == 0) {
04370         /* If Master/Slave config fault is asserted twice,
04371          * we assume back-to-back */
04372         e1000_read_phy_reg(&adapter->hw, PHY_1000T_STATUS, &phy_status);
04373         if (!(phy_status & SR_1000T_MS_CONFIG_FAULT)) return;
04374         e1000_read_phy_reg(&adapter->hw, PHY_1000T_STATUS, &phy_status);
04375         if (!(phy_status & SR_1000T_MS_CONFIG_FAULT)) return;
04376         e1000_read_phy_reg(&adapter->hw, PHY_1000T_CTRL, &phy_ctrl);
04377         if (phy_ctrl & CR_1000T_MS_ENABLE) {
04378             phy_ctrl &= ~CR_1000T_MS_ENABLE;
04379             e1000_write_phy_reg(&adapter->hw, PHY_1000T_CTRL,
04380                         phy_ctrl);
04381             adapter->smartspeed++;
04382             if (!e1000_phy_setup_autoneg(&adapter->hw) &&
04383                !e1000_read_phy_reg(&adapter->hw, PHY_CONTROL,
04384                            &phy_ctrl)) {
04385                 phy_ctrl |= (MII_CR_AUTO_NEG_EN |
04386                          MII_CR_RESTART_AUTO_NEG);
04387                 e1000_write_phy_reg(&adapter->hw, PHY_CONTROL,
04388                             phy_ctrl);
04389             }
04390         }
04391         return;
04392     } else if (adapter->smartspeed == E1000_SMARTSPEED_DOWNSHIFT) {
04393         /* If still no link, perhaps using 2/3 pair cable */
04394         e1000_read_phy_reg(&adapter->hw, PHY_1000T_CTRL, &phy_ctrl);
04395         phy_ctrl |= CR_1000T_MS_ENABLE;
04396         e1000_write_phy_reg(&adapter->hw, PHY_1000T_CTRL, phy_ctrl);
04397         if (!e1000_phy_setup_autoneg(&adapter->hw) &&
04398            !e1000_read_phy_reg(&adapter->hw, PHY_CONTROL, &phy_ctrl)) {
04399             phy_ctrl |= (MII_CR_AUTO_NEG_EN |
04400                      MII_CR_RESTART_AUTO_NEG);
04401             e1000_write_phy_reg(&adapter->hw, PHY_CONTROL, phy_ctrl);
04402         }
04403     }
04404     /* Restart process after E1000_SMARTSPEED_MAX iterations */
04405     if (adapter->smartspeed++ == E1000_SMARTSPEED_MAX)
04406         adapter->smartspeed = 0;
04407 }

04708 static int __e1000_shutdown(struct pci_dev *pdev, bool *enable_wake)
04709 {
04710     struct net_device *netdev = pci_get_drvdata(pdev);
04711     struct e1000_adapter *adapter = netdev_priv(netdev);
04712     u32 ctrl, ctrl_ext, rctl, status;
04713     u32 wufc = adapter->wol;
04714 #ifdef CONFIG_PM
04715     int retval = 0;
04716 #endif
04717 
04718     netif_device_detach(netdev);
04719 
04720     if (netif_running(netdev)) {
04721         WARN_ON(test_bit(__E1000_RESETTING, &adapter->state));
04722         e1000_down(adapter);
04723         e1000_free_irq(adapter);
04724     }
04725 
04726 #ifdef CONFIG_PM
04727     retval = pci_save_state(pdev);
04728     if (retval)
04729         return retval;
04730 #endif
04731 
04732     status = E1000_READ_REG(&adapter->hw, E1000_STATUS);
04733     if (status & E1000_STATUS_LU)
04734         wufc &= ~E1000_WUFC_LNKC;
04735 
04736     if (wufc) {
04737         e1000_setup_rctl(adapter);
04738         e1000_set_multi(netdev);
04739 
04740         /* turn on all-multi mode if wake on multicast is enabled */
04741         if (wufc & E1000_WUFC_MC) {
04742             rctl = E1000_READ_REG(&adapter->hw, E1000_RCTL);
04743             rctl |= E1000_RCTL_MPE;
04744             E1000_WRITE_REG(&adapter->hw, E1000_RCTL, rctl);
04745         }
04746 
04747         if (adapter->hw.mac.type >= e1000_82540) {
04748             ctrl = E1000_READ_REG(&adapter->hw, E1000_CTRL);
04749             /* advertise wake from D3Cold */
04750             #define E1000_CTRL_ADVD3WUC 0x00100000
04751             /* phy power management enable */
04752             #define E1000_CTRL_EN_PHY_PWR_MGMT 0x00200000
04753             ctrl |= E1000_CTRL_ADVD3WUC |
04754                 E1000_CTRL_EN_PHY_PWR_MGMT;
04755             E1000_WRITE_REG(&adapter->hw, E1000_CTRL, ctrl);
04756         }
04757 
04758         if (adapter->hw.phy.media_type == e1000_media_type_fiber ||
04759            adapter->hw.phy.media_type == e1000_media_type_internal_serdes) {
04760             /* keep the laser running in D3 */
04761             ctrl_ext = E1000_READ_REG(&adapter->hw, E1000_CTRL_EXT);
04762             ctrl_ext |= E1000_CTRL_EXT_SDP3_DATA;
04763             E1000_WRITE_REG(&adapter->hw, E1000_CTRL_EXT, ctrl_ext);
04764         }
04765 
04766         /* Allow time for pending master requests to run */
04767         e1000_disable_pcie_master(&adapter->hw);
04768 
04769         E1000_WRITE_REG(&adapter->hw, E1000_WUC, E1000_WUC_PME_EN);
04770         E1000_WRITE_REG(&adapter->hw, E1000_WUFC, wufc);
04771     } else {
04772         E1000_WRITE_REG(&adapter->hw, E1000_WUC, 0);
04773         E1000_WRITE_REG(&adapter->hw, E1000_WUFC, 0);
04774     }
04775 
04776     *enable_wake =!!wufc;
04777 
04778     e1000_release_manageability(adapter);
04779 
04780     /* make sure adapter isn't asleep if manageability is enabled */
04781     if (adapter->en_mng_pt)
04782         *enable_wake = true;
04783 
04784     pci_disable_device(pdev);
04785 
04786     return 0;
04787 }
