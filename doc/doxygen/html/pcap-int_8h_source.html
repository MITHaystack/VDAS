00347 /*
00348  * This is a timeval as stored in a savefile.
00349  * It has to use the same types everywhere, independent of the actual
00350  * `struct timeval'; `struct timeval' has 32-bit tv_sec values on some
00351  * platforms and 64-bit tv_sec values on other platforms, and writing
00352  * out native `struct timeval' values would mean files could only be
00353  * read on systems with the same tv_sec size as the system on which
00354  * the file was written.
00355  */
00356 
00357 struct pcap_timeval {
00358     bpf_int32 tv_sec;       /* seconds */
00359     bpf_int32 tv_usec;      /* microseconds */
00360 };

00362 /*
00363  * This is a `pcap_pkthdr' as actually stored in a savefile.
00364  *
00365  * Do not change the format of this structure, in any way (this includes
00366  * changes that only affect the length of fields in this structure),
00367  * and do not make the time stamp anything other than seconds and
00368  * microseconds (e.g., seconds and nanoseconds).  Instead:
00369  *
00370  *  introduce a new structure for the new format;
00371  *
00372  *  send mail to "tcpdump-workers@lists.tcpdump.org", requesting
00373  *  a new magic number for your new capture file format, and, when
00374  *  you get the new magic number, put it in "savefile.c";
00375  *
00376  *  use that magic number for save files with the changed record
00377  *  header;
00378  *
00379  *  make the code in "savefile.c" capable of reading files with
00380  *  the old record header as well as files with the new record header
00381  *  (using the magic number to determine the header format).
00382  *
00383  * Then supply the changes as a patch at
00384  *
00385  *  http://sourceforge.net/projects/libpcap/
00386  *
00387  * so that future versions of libpcap and programs that use it (such as
00388  * tcpdump) will be able to read your new capture file format.
00389  */
00390 
00391 struct pcap_sf_pkthdr {
00392     struct pcap_timeval ts; /* time stamp */
00393     bpf_u_int32 caplen;     /* length of portion present */
00394     bpf_u_int32 len;        /* length this packet (off wire) */
00395 };
