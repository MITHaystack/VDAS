00101 u_int
00102 token_print(const u_char *p, u_int length, u_int caplen)
00103 {
00104     const struct token_header *trp;
00105     u_short extracted_ethertype;
00106     struct ether_header ehdr;
00107     u_int route_len = 0, hdr_len = TOKEN_HDRLEN;
00108     int seg;
00109 
00110     trp = (const struct token_header *)p;
00111 
00112     if (caplen < TOKEN_HDRLEN) {
00113         printf("[|token-ring]");
00114         return hdr_len;
00115     }
00116 
00117     /*
00118      * Get the TR addresses into a canonical form
00119      */
00120     extract_token_addrs(trp, (char*)ESRC(&ehdr), (char*)EDST(&ehdr));
00121 
00122     /* Adjust for source routing information in the MAC header */
00123     if (IS_SOURCE_ROUTED(trp)) {
00124         /* Clear source-routed bit */
00125         *ESRC(&ehdr) &= 0x7f;
00126 
00127         if (eflag)
00128             token_hdr_print(trp, length, ESRC(&ehdr), EDST(&ehdr));
00129 
00130         if (caplen < TOKEN_HDRLEN + 2) {
00131             printf("[|token-ring]");
00132             return hdr_len;
00133         }
00134         route_len = RIF_LENGTH(trp);
00135         hdr_len += route_len;
00136         if (caplen < hdr_len) {
00137             printf("[|token-ring]");
00138             return hdr_len;
00139         }
00140         if (vflag) {
00141             printf("%s ", broadcast_indicator[BROADCAST(trp)]);
00142             printf("%s", direction[DIRECTION(trp)]);
00143 
00144             for (seg = 0; seg < SEGMENT_COUNT(trp); seg++)
00145                 printf(" [%d:%d]", RING_NUMBER(trp, seg),
00146                     BRIDGE_NUMBER(trp, seg));
00147         } else {
00148             printf("rt = %x", EXTRACT_16BITS(&trp->token_rcf));
00149 
00150             for (seg = 0; seg < SEGMENT_COUNT(trp); seg++)
00151                 printf(":%x", EXTRACT_16BITS(&trp->token_rseg[seg]));
00152         }
00153         printf(" (%s) ", largest_frame[LARGEST_FRAME(trp)]);
00154     } else {
00155         if (eflag)
00156             token_hdr_print(trp, length, ESRC(&ehdr), EDST(&ehdr));
00157     }
00158 
00159     /* Skip over token ring MAC header and routing information */
00160     length -= hdr_len;
00161     p += hdr_len;
00162     caplen -= hdr_len;
00163 
00164     /* Frame Control field determines interpretation of packet */
00165     if (FRAME_TYPE(trp) == TOKEN_FC_LLC) {
00166         /* Try to print the LLC-layer header & higher layers */
00167         if (llc_print(p, length, caplen, ESRC(&ehdr), EDST(&ehdr),
00168             &extracted_ethertype) == 0) {
00169             /* ether_type not known, print raw packet */
00170             if (!eflag)
00171                 token_hdr_print(trp,
00172                     length + TOKEN_HDRLEN + route_len,
00173                     ESRC(&ehdr), EDST(&ehdr));
00174             if (extracted_ethertype) {
00175                 printf("(LLC %s) ",
00176             etherproto_string(htons(extracted_ethertype)));
00177             }
00178             if (!suppress_default_print)
00179                 default_print(p, caplen);
00180         }
00181     } else {
00182         /* Some kinds of TR packet we cannot handle intelligently */
00183         /* XXX - dissect MAC packets if frame type is 0 */
00184         if (!eflag)
00185             token_hdr_print(trp, length + TOKEN_HDRLEN + route_len,
00186                 ESRC(&ehdr), EDST(&ehdr));
00187         if (!suppress_default_print)
00188             default_print(p, caplen);
00189     }
00190     return (hdr_len);
00191 }
