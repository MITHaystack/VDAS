00060 /* The time difference in milliseconds */
00061 time_t delta_time_in_milliseconds (struct timeval * t2, struct timeval * t1)
00062 {
00063   /* Compute delta in second, 1/10's and 1/1000's second units */
00064   time_t delta_seconds      = t2 -> tv_sec - t1 -> tv_sec;
00065   time_t delta_milliseconds = (t2 -> tv_usec - t1 -> tv_usec) / 1000;
00066 
00067   if (delta_milliseconds < 0)
00068     { /* manually carry a one from the seconds field */
00069       delta_milliseconds += 1000;                              /* 1e3 */
00070       -- delta_seconds;
00071     }
00072   return (delta_seconds * 1000) + delta_milliseconds;
00073 }
