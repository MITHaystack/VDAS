00073 /*
00074  * Known problems with 2.0[.x] kernels:
00075  *
00076  *   - The loopback device gives every packet twice; on 2.2[.x] kernels,
00077  *     if we use PF_PACKET, we can filter out the transmitted version
00078  *     of the packet by using data in the "sockaddr_ll" returned by
00079  *     "recvfrom()", but, on 2.0[.x] kernels, we have to use
00080  *     PF_INET/SOCK_PACKET, which means "recvfrom()" supplies a
00081  *     "sockaddr_pkt" which doesn't give us enough information to let
00082  *     us do that.
00083  *
00084  *   - We have to set the interface's IFF_PROMISC flag ourselves, if
00085  *     we're to run in promiscuous mode, which means we have to turn
00086  *     it off ourselves when we're done; the kernel doesn't keep track
00087  *     of how many sockets are listening promiscuously, which means
00088  *     it won't get turned off automatically when no sockets are
00089  *     listening promiscuously.  We catch "pcap_close()" and, for
00090  *     interfaces we put into promiscuous mode, take them out of
00091  *     promiscuous mode - which isn't necessarily the right thing to
00092  *     do, if another socket also requested promiscuous mode between
00093  *     the time when we opened the socket and the time when we close
00094  *     the socket.
00095  *
00096  *   - MSG_TRUNC isn't supported, so you can't specify that "recvfrom()"
00097  *     return the amount of data that you could have read, rather than
00098  *     the amount that was returned, so we can't just allocate a buffer
00099  *     whose size is the snapshot length and pass the snapshot length
00100  *     as the byte count, and also pass MSG_TRUNC, so that the return
00101  *     value tells us how long the packet was on the wire.
00102  *
00103  *     This means that, if we want to get the actual size of the packet,
00104  *     so we can return it in the "len" field of the packet header,
00105  *     we have to read the entire packet, not just the part that fits
00106  *     within the snapshot length, and thus waste CPU time copying data
00107  *     from the kernel that our caller won't see.
00108  *
00109  *     We have to get the actual size, and supply it in "len", because
00110  *     otherwise, the IP dissector in tcpdump, for example, will complain
00111  *     about "truncated-ip", as the packet will appear to have been
00112  *     shorter, on the wire, than the IP header said it should have been.
00113  */

00418 /*
00419      *
00420      * If interface {if} is a mac80211 driver, the file
00421      * /sys/class/net/{if}/phy80211 is a symlink to
00422      * /sys/class/ieee80211/{phydev}, for some {phydev}.
00423      *
00424      * On Fedora 9, with a 2.6.26.3-29 kernel, my Zydas stick, at
00425      * least, has a "wmaster0" device and a "wlan0" device; the
00426      * latter is the one with the IP address.  Both show up in
00427      * "tcpdump -D" output.  Capturing on the wmaster0 device
00428      * captures with 802.11 headers.
00429      *
00430      * airmon-ng searches through /sys/class/net for devices named
00431      * monN, starting with mon0; as soon as one *doesn't* exist,
00432      * it chooses that as the monitor device name.  If the "iw"
00433      * command exists, it does "iw dev {if} interface add {monif}
00434      * type monitor", where {monif} is the monitor device.  It
00435      * then (sigh) sleeps .1 second, and then configures the
00436      * device up.  Otherwise, if /sys/class/ieee80211/{phydev}/add_iface
00437      * is a file, it writes {mondev}, without a newline, to that file,
00438      * and again (sigh) sleeps .1 second, and then iwconfig's that
00439      * device into monitor mode and configures it up.  Otherwise,
00440      * you can't do monitor mode.
00441      *
00442      * All these devices are "glued" together by having the
00443      * /sys/class/net/{device}/phy80211 links pointing to the same
00444      * place, so, given a wmaster, wlan, or mon device, you can
00445      * find the other devices by looking for devices with
00446      * the same phy80211 link.
00447      *
00448      * To turn monitor mode off, delete the monitor interface,
00449      * either with "iw dev {monif} interface del" or by sending
00450      * {monif}, with no NL, down /sys/class/ieee80211/{phydev}/remove_iface
00451      *
00452      * Note: if you try to create a monitor device named "monN", and
00453      * there's already a "monN" device, it fails, as least with
00454      * the netlink interface (which is what iw uses), with a return
00455      * value of -ENFILE.  (Return values are negative errnos.)  We
00456      * could probably use that to find an unused device.
00457      *
00458      * Yes, you can have multiple monitor devices for a given
00459      * physical device.
00460 */

00462 /*
00463  * Is this a mac80211 device?  If so, fill in the physical device path and
00464  * return 1; if not, return 0.  On an error, fill in handle->errbuf and
00465  * return PCAP_ERROR.
00466  */
00467 static int
00468 get_mac80211_phydev(pcap_t *handle, const char *device, char *phydev_path,
00469     size_t phydev_max_pathlen)
00470 {
00471     char *pathstr;
00472     ssize_t bytes_read;
00473 
00474     /*
00475      * Generate the path string for the symlink to the physical device.
00476      */
00477     if (asprintf(&pathstr, "/sys/class/net/%s/phy80211", device) == -1) {
00478         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
00479             "%s: Can't generate path name string for /sys/class/net device",
00480             device);
00481         return PCAP_ERROR;
00482     }
00483     bytes_read = readlink(pathstr, phydev_path, phydev_max_pathlen);
00484     if (bytes_read == -1) {
00485         if (errno == ENOENT || errno == EINVAL) {
00486             /*
00487              * Doesn't exist, or not a symlink; assume that
00488              * means it's not a mac80211 device.
00489              */
00490             free(pathstr);
00491             return 0;
00492         }
00493         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
00494             "%s: Can't readlink %s: %s", device, pathstr,
00495             strerror(errno));
00496         free(pathstr);
00497         return PCAP_ERROR;
00498     }
00499     free(pathstr);
00500     phydev_path[bytes_read] = '\0';
00501     return 1;
00502 }

00917  * Grabs the number of dropped packets by the interface from /proc/net/dev.
00918  *
00919  * XXX - what about /sys/class/net/{interface name}/rx_*?  There are
00920  * individual devices giving, in ASCII, various rx_ and tx_ statistics.
00921  *
00922  * Or can we get them in binary form from netlink?
00923  */
00924 static long int
00925 linux_if_drops(const char * if_name)
00926 {
00927     char buffer[512];
00928     char * bufptr;
00929     FILE * file;
00930     int field_to_convert = 3, if_name_sz = strlen(if_name);
00931     long int dropped_pkts = 0;
00932 
00933     file = fopen("/proc/net/dev", "r");
00934     if (!file)
00935         return 0;
00936 
00937     while (!dropped_pkts && fgets( buffer, sizeof(buffer), file ))
00938     {
00939         /*  search for 'bytes' -- if its in there, then
00940             that means we need to grab the fourth field. otherwise
00941             grab the third field. */
00942         if (field_to_convert != 4 && strstr(buffer, "bytes"))
00943         {
00944             field_to_convert = 4;
00945             continue;
00946         }
00947 
00948         /* find iface and make sure it actually matches -- space before the name and : after it */
00949         if ((bufptr = strstr(buffer, if_name)) &&
00950             (bufptr == buffer || *(bufptr-1) == ' ') &&
00951             *(bufptr + if_name_sz) == ':')
00952         {
00953             bufptr = bufptr + if_name_sz + 1;
00954 
00955             /* grab the nth field from it */
00956             while( --field_to_convert && *bufptr != '\0')
00957             {
00958                 while (*bufptr != '\0' && *(bufptr++) == ' ');
00959                 while (*bufptr != '\0' && *(bufptr++) != ' ');
00960             }
00961 
00962             /* get rid of any final spaces */
00963             while (*bufptr != '\0' && *bufptr == ' ') bufptr++;
00964 
00965             if (*bufptr != '\0')
00966                 dropped_pkts = strtol(bufptr, NULL, 10);
00967 
00968             break;
00969         }
00970     }
00971

01828          * argument is supported on PF_PACKET sockets:
01829          *
01830          *  "ps_recv" counts only packets that *passed* the
01831          *  filter, not packets that didn't pass the filter.
01832          *  This includes packets later dropped because we
01833          *  ran out of buffer space.
01834          *
01835          *  "ps_drop" counts packets dropped because we ran
01836          *  out of buffer space.  It doesn't count packets
01837          *  dropped by the interface driver.  It counts only
01838          *  packets that passed the filter.
01839          *
01840          *  See above for ps_ifdrop.
01841          *
01842          *  Both statistics include packets not yet read from
01843          *  the kernel by libpcap, and thus not yet seen by
01844          *  the application.
01845          *
01846          * In "linux/net/packet/af_packet.c", at least in the
01847          * 2.4.9 kernel, "tp_packets" is incremented for every
01848          * packet that passes the packet filter *and* is
01849          * successfully queued on the socket; "tp_drops" is
01850          * incremented for every packet dropped because there's
01851          * not enough free space in the socket buffer.
01852          *
01853          * When the statistics are returned for a PACKET_STATISTICS
01854          * "getsockopt()" call, "tp_drops" is added to "tp_packets",
01855          * so that "tp_packets" counts all packets handed to
01856          * the PF_PACKET socket, including packets dropped because
01857          * there wasn't room on the socket buffer - but not
01858          * including packets that didn't pass the filter.
01859          *
01860          * In the BSD BPF, the count of received packets is
01861          * incremented for every packet handed to BPF, regardless
01862          * of whether it passed the filter.
01863          *
01864          * We can't make "pcap_stats()" work the same on both
01865          * platforms, but the best approximation is to return
01866          * "tp_packets" as the count of packets and "tp_drops"
01867          * as the count of drops.
01868          *
01869          * Keep a running total because each call to
01870          *    getsockopt(handle->fd, SOL_PACKET, PACKET_STATISTICS, ....
01871          * resets the counters to zero.
01872          */
01873         handle->md.stat.ps_recv += kstats.tp_packets;
01874         handle->md.stat.ps_drop += kstats.tp_drops;
01875         *stats = handle->md.stat;
01876         return 0;
01877     }
01878     else
01879     {
01880         /*
01881          * If the error was EOPNOTSUPP, fall through, so that
01882          * if you build the library on a system with
01883          * "struct tpacket_stats" and run it on a system
01884          * that doesn't, it works as it does if the library
01885          * is built on a system without "struct tpacket_stats".
01886          */
01887         if (errno != EOPNOTSUPP) {
01888             snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
01889                 "pcap_stats: %s", pcap_strerror(errno));
01890             return -1;
01891         }
01892     }
01893 #endif
01894     /*
01895      * On systems where the PACKET_STATISTICS "getsockopt()" argument
01896      * is not supported on PF_PACKET sockets:
01897      *
01898      *  "ps_recv" counts only packets that *passed* the filter,
01899      *  not packets that didn't pass the filter.  It does not
01900      *  count packets dropped because we ran out of buffer
01901      *  space.
01902      *
01903      *  "ps_drop" is not supported.
01904      *
01905      *  "ps_ifdrop" is supported. It will return the number
01906      *  of drops the interface reports in /proc/net/dev,
01907      *  if that is available.
01908      *
01909      *  "ps_recv" doesn't include packets not yet read from
01910      *  the kernel by libpcap.
01911      *
01912      * We maintain the count of packets processed by libpcap in
01913      * "md.packets_read", for reasons described in the comment
01914      * at the end of pcap_read_packet().  We have no idea how many
01915      * packets were dropped by the kernel buffers -- but we know
01916      * how many the interface dropped, so we can return that.
01917      */
01918 
01919     stats->ps_recv = handle->md.packets_read;
01920     stats->ps_drop = 0;
01921     stats->ps_ifdrop = handle->md.stat.ps_ifdrop;
01922     return 0;
01923 }

01926  * Get from "/sys/class/net" all interfaces listed there; if they're
01927  * already in the list of interfaces we have, that won't add another
01928  * instance, but if they're not, that'll add them.
01929  *
01930  * We don't bother getting any addresses for them; it appears you can't
01931  * use SIOCGIFADDR on Linux to get IPv6 addresses for interfaces, and,
01932  * although some other types of addresses can be fetched with SIOCGIFADDR,
01933  * we don't bother with them for now.
01934  *
01935  * We also don't fail if we couldn't open "/sys/class/net"; we just leave
01936  * the list of interfaces as is, and return 0, so that we can try
01937  * scanning /proc/net/dev.
01938  */
01939 static int
01940 scan_sys_class_net(pcap_if_t **devlistp, char *errbuf)
01941 {
01942     DIR *sys_class_net_d;
01943     int fd;
01944     struct dirent *ent;
01945     char *p;
01946     char name[512]; /* XXX - pick a size */
01947     char *q, *saveq;
01948     struct ifreq ifrflags;
01949     int ret = 1;
01950 
01951     sys_class_net_d = opendir("/sys/class/net");
01952     if (sys_class_net_d == NULL && errno == ENOENT)
01953         return (0);
01954 
01955     /*
01956      * Create a socket from which to fetch interface information.
01957      */
01958     fd = socket(AF_INET, SOCK_DGRAM, 0);
01959     if (fd < 0) {
01960         (void)snprintf(errbuf, PCAP_ERRBUF_SIZE,
01961             "socket: %s", pcap_strerror(errno));
01962         return (-1);
01963     }
01964 
01965     for (;;) {
01966         errno = 0;
01967         ent = readdir(sys_class_net_d);
01968         if (ent == NULL) {
01969             /*
01970              * Error or EOF; if errno != 0, it's an error.
01971              */
01972             break;
01973         }
01974 
01975         /*
01976          * Ignore directories (".", "..", and any subdirectories).
01977          */
01978         if (ent->d_type == DT_DIR)
01979             continue;
01980 
01981         /*
01982          * Get the interface name.
01983          */
01984         p = &ent->d_name[0];
01985         q = &name[0];
01986         while (*p != '\0' && isascii(*p) && !isspace(*p)) {
01987             if (*p == ':') {
01988                 /*
01989                  * This could be the separator between a
01990                  * name and an alias number, or it could be
01991                  * the separator between a name with no
01992                  * alias number and the next field.
01993                  *
01994                  * If there's a colon after digits, it
01995                  * separates the name and the alias number,
01996                  * otherwise it separates the name and the
01997                  * next field.
01998                  */
01999                 saveq = q;
02000                 while (isascii(*p) && isdigit(*p))
02001                     *q++ = *p++;
02002                 if (*p != ':') {
02003                     /*
02004                      * That was the next field,
02005                      * not the alias number.
02006                      */
02007                     q = saveq;
02008                 }
02009                 break;
02010             } else
02011                 *q++ = *p++;
02012         }
02013         *q = '\0';
02014 
02015         /*
02016          * Get the flags for this interface, and skip it if
02017          * it's not up.
02018          */
02019         strncpy(ifrflags.ifr_name, name, sizeof(ifrflags.ifr_name));
02020         if (ioctl(fd, SIOCGIFFLAGS, (char *)&ifrflags) < 0) {
02021             if (errno == ENXIO)
02022                 continue;
02023             (void)snprintf(errbuf, PCAP_ERRBUF_SIZE,
02024                 "SIOCGIFFLAGS: %.*s: %s",
02025                 (int)sizeof(ifrflags.ifr_name),
02026                 ifrflags.ifr_name,
02027                 pcap_strerror(errno));
02028             ret = -1;
02029             break;
02030         }
02031         if (!(ifrflags.ifr_flags & IFF_UP))
02032             continue;
02033 
02034         /*
02035          * Add an entry for this interface, with no addresses.
02036          */

02039             /*
02040              * Failure.
02041              */
02042             ret = -1;
02043             break;
02044         }
02045     }
02046     if (ret != -1) {
02047         /*
02048          * Well, we didn't fail for any other reason; did we
02049          * fail due to an error reading the directory?
02050          */

02052             (void)snprintf(errbuf, PCAP_ERRBUF_SIZE,
02053                 "Error reading /sys/class/net: %s",
02054                 pcap_strerror(errno));

02056         }
02057     }
02058 
02059     (void)close(fd);

02061     return (ret);

02273 /*
02274  *  Attach the given BPF code to the packet capture device.
02275  */
02276 static int
02277 pcap_setfilter_linux_common(pcap_t *handle, struct bpf_program *filter,
02278     int is_mmapped)
02279 {
02280 #ifdef SO_ATTACH_FILTER
02281     struct sock_fprog   fcode;
02282     int         can_filter_in_kernel;
02283     int         err = 0;
02284 #endif
02285 
02286     if (!handle)
02287         return -1;
02288     if (!filter) {
02289             strncpy(handle->errbuf, "setfilter: No filter specified",
02290             PCAP_ERRBUF_SIZE);
02291         return -1;
02292     }
02293 
02294     /* Make our private copy of the filter */
02295 
02296     if (install_bpf_program(handle, filter) < 0)
02297         /* install_bpf_program() filled in errbuf */
02298         return -1;
02299 
02300     /*
02301      * Run user level packet filter by default. Will be overriden if
02302      * installing a kernel filter succeeds.
02303      */
02304     handle->md.use_bpf = 0;
02305 
02306     /* Install kernel level filter if possible */
02307 
02308 #ifdef SO_ATTACH_FILTER
02309 #ifdef USHRT_MAX
02310     if (handle->fcode.bf_len > USHRT_MAX) {
02311         /*
02312          * fcode.len is an unsigned short for current kernel.
02313          * I have yet to see BPF-Code with that much
02314          * instructions but still it is possible. So for the
02315          * sake of correctness I added this check.
02316          */
02317         fprintf(stderr, "Warning: Filter too complex for kernel\n");
02318         fcode.len = 0;
02319         fcode.filter = NULL;
02320         can_filter_in_kernel = 0;
02321     } else
02322 #endif /* USHRT_MAX */
02323     {
02324         /*
02325          * Oh joy, the Linux kernel uses struct sock_fprog instead
02326          * of struct bpf_program and of course the length field is
02327          * of different size. Pointed out by Sebastian
02328          *
02329          * Oh, and we also need to fix it up so that all "ret"
02330          * instructions with non-zero operands have 65535 as the
02331          * operand if we're not capturing in memory-mapped modee,
02332          * and so that, if we're in cooked mode, all memory-reference
02333          * instructions use special magic offsets in references to
02334          * the link-layer header and assume that the link-layer
02335          * payload begins at 0; "fix_program()" will do that.
02336          */
02337         switch (fix_program(handle, &fcode, is_mmapped)) {
02338 
02339         case -1:
02340         default:
02341             /*
02342              * Fatal error; just quit.
02343              * (The "default" case shouldn't happen; we
02344              * return -1 for that reason.)
02345              */
02346             return -1;
02347 
02348         case 0:
02349             /*
02350              * The program performed checks that we can't make
02351              * work in the kernel.
02352              */
02353             can_filter_in_kernel = 0;
02354             break;
02355 
02356         case 1:
02357             /*
02358              * We have a filter that'll work in the kernel.
02359              */
02360             can_filter_in_kernel = 1;
02361             break;
02362         }
02363     }
02364 
02365     if (can_filter_in_kernel) {
02366         if ((err = set_kernel_filter(handle, &fcode)) == 0)
02367         {
02368             /* Installation succeded - using kernel filter. */
02369             handle->md.use_bpf = 1;
02370         }
02371         else if (err == -1) /* Non-fatal error */
02372         {
02373             /*
02374              * Print a warning if we weren't able to install
02375              * the filter for a reason other than "this kernel
02376              * isn't configured to support socket filters.
02377              */
02378             if (errno != ENOPROTOOPT && errno != EOPNOTSUPP) {
02379                 fprintf(stderr,
02380                     "Warning: Kernel filter failed: %s\n",
02381                     pcap_strerror(errno));
02382             }
02383         }
02384     }
02385 
02386     /*
02387      * If we're not using the kernel filter, get rid of any kernel
02388      * filter that might've been there before, e.g. because the
02389      * previous filter could work in the kernel, or because some other
02390      * code attached a filter to the socket by some means other than
02391      * calling "pcap_setfilter()".  Otherwise, the kernel filter may
02392      * filter out packets that would pass the new userland filter.
02393      */
02394     if (!handle->md.use_bpf)
02395         reset_kernel_filter(handle);
02396 
02397     /*
02398      * Free up the copy of the filter that was made by "fix_program()".
02399      */
02400     if (fcode.filter != NULL)
02401         free(fcode.filter);
02402 
02403     if (err == -2)
02404         /* Fatal error */
02405         return -1;
02406 #endif /* SO_ATTACH_FILTER */
02407 
02408     return 0;
02409 }

02795 /*
02796  * Try to open a packet socket using the new kernel PF_PACKET interface.
02797  * Returns 1 on success, 0 on an error that means the new interface isn't
02798  * present (so the old SOCK_PACKET interface should be tried), and a
02799  * PCAP_ERROR_ value on an error that means that the old mechanism won't
02800  * work either (so it shouldn't be tried).
02801  */
02802 static int
02803 activate_new(pcap_t *handle)
02804 {
02805 #ifdef HAVE_PF_PACKET_SOCKETS
02806     const char      *device = handle->opt.source;
02807     int         is_any_device = (strcmp(device, "any") == 0);
02808     int         sock_fd = -1, arptype;
02809 #ifdef HAVE_PACKET_AUXDATA
02810     int         val;
02811 #endif
02812     int         err = 0;
02813     struct packet_mreq  mr;
02814 
02815     /*
02816      * Open a socket with protocol family packet. If the
02817      * "any" device was specified, we open a SOCK_DGRAM
02818      * socket for the cooked interface, otherwise we first
02819      * try a SOCK_RAW socket for the raw interface.
02820      */
02821     sock_fd = is_any_device ?
02822         socket(PF_PACKET, SOCK_DGRAM, htons(ETH_P_ALL)) :
02823         socket(PF_PACKET, SOCK_RAW, htons(ETH_P_ALL));
02824 
02825     if (sock_fd == -1) {
02826         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE, "socket: %s",
02827              pcap_strerror(errno) );
02828         return 0;   /* try old mechanism */
02829     }
02830 
02831     /* It seems the kernel supports the new interface. */
02832     handle->md.sock_packet = 0;
02833 
02834     /*
02835      * Get the interface index of the loopback device.
02836      * If the attempt fails, don't fail, just set the
02837      * "md.lo_ifindex" to -1.
02838      *
02839      * XXX - can there be more than one device that loops
02840      * packets back, i.e. devices other than "lo"?  If so,
02841      * we'd need to find them all, and have an array of
02842      * indices for them, and check all of them in
02843      * "pcap_read_packet()".
02844      */
02845     handle->md.lo_ifindex = iface_get_id(sock_fd, "lo", handle->errbuf);
02846 
02847     /*
02848      * Default value for offset to align link-layer payload
02849      * on a 4-byte boundary.
02850      */
02851     handle->offset   = 0;
02852 
02853     /*
02854      * What kind of frames do we have to deal with? Fall back
02855      * to cooked mode if we have an unknown interface type
02856      * or a type we know doesn't work well in raw mode.
02857      */
02858     if (!is_any_device) {
02859         /* Assume for now we don't need cooked mode. */
02860         handle->md.cooked = 0;
02861 
02862         if (handle->opt.rfmon) {
02863             /*
02864              * We were asked to turn on monitor mode.
02865              * Do so before we get the link-layer type,
02866              * because entering monitor mode could change
02867              * the link-layer type.
02868              */
02869             err = enter_rfmon_mode(handle, sock_fd, device);
02870             if (err < 0) {
02871                 /* Hard failure */
02872                 close(sock_fd);
02873                 return err;
02874             }
02875             if (err == 0) {
02876                 /*
02877                  * Nothing worked for turning monitor mode
02878                  * on.
02879                  */
02880                 close(sock_fd);
02881                 return PCAP_ERROR_RFMON_NOTSUP;
02882             }
02883 
02884             /*
02885              * Either monitor mode has been turned on for
02886              * the device, or we've been given a different
02887              * device to open for monitor mode.  If we've
02888              * been given a different device, use it.
02889              */
02890             if (handle->md.mondevice != NULL)
02891                 device = handle->md.mondevice;
02892         }
02893         arptype = iface_get_arptype(sock_fd, device, handle->errbuf);
02894         if (arptype < 0) {
02895             close(sock_fd);
02896             return arptype;
02897         }
02898         map_arphrd_to_dlt(handle, arptype, 1);
02899         if (handle->linktype == -1 ||
02900             handle->linktype == DLT_LINUX_SLL ||
02901             handle->linktype == DLT_LINUX_IRDA ||
02902             handle->linktype == DLT_LINUX_LAPD ||
02903             (handle->linktype == DLT_EN10MB &&
02904              (strncmp("isdn", device, 4) == 0 ||
02905               strncmp("isdY", device, 4) == 0))) {
02906             /*
02907              * Unknown interface type (-1), or a
02908              * device we explicitly chose to run
02909              * in cooked mode (e.g., PPP devices),
02910              * or an ISDN device (whose link-layer
02911              * type we can only determine by using
02912              * APIs that may be different on different
02913              * kernels) - reopen in cooked mode.
02914              */
02915             if (close(sock_fd) == -1) {
02916                 snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
02917                      "close: %s", pcap_strerror(errno));
02918                 return PCAP_ERROR;
02919             }
02920             sock_fd = socket(PF_PACKET, SOCK_DGRAM,
02921                 htons(ETH_P_ALL));
02922             if (sock_fd == -1) {
02923                 snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
02924                     "socket: %s", pcap_strerror(errno));
02925                 return PCAP_ERROR;
02926             }
02927             handle->md.cooked = 1;
02928 
02929             /*
02930              * Get rid of any link-layer type list
02931              * we allocated - this only supports cooked
02932              * capture.
02933              */
02934             if (handle->dlt_list != NULL) {
02935                 free(handle->dlt_list);
02936                 handle->dlt_list = NULL;
02937                 handle->dlt_count = 0;
02938             }
02939 
02940             if (handle->linktype == -1) {
02941                 /*
02942                  * Warn that we're falling back on
02943                  * cooked mode; we may want to
02944                  * update "map_arphrd_to_dlt()"
02945                  * to handle the new type.
02946                  */
02947                 snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
02948                     "arptype %d not "
02949                     "supported by libpcap - "
02950                     "falling back to cooked "
02951                     "socket",
02952                     arptype);
02953             }
02954 
02955             /*
02956              * IrDA capture is not a real "cooked" capture,
02957              * it's IrLAP frames, not IP packets.  The
02958              * same applies to LAPD capture.
02959              */
02960             if (handle->linktype != DLT_LINUX_IRDA &&
02961                 handle->linktype != DLT_LINUX_LAPD)
02962                 handle->linktype = DLT_LINUX_SLL;
02963         }
02964 
02965         handle->md.ifindex = iface_get_id(sock_fd, device,
02966             handle->errbuf);
02967         if (handle->md.ifindex == -1) {
02968             close(sock_fd);
02969             return PCAP_ERROR;
02970         }
02971 
02972         if ((err = iface_bind(sock_fd, handle->md.ifindex,
02973             handle->errbuf)) != 1) {
02974                 close(sock_fd);
02975             if (err < 0)
02976                 return err;
02977             else
02978                 return 0;   /* try old mechanism */
02979         }
02980     } else {
02981         /*
02982          * The "any" device.
02983          */
02984         if (handle->opt.rfmon) {
02985             /*
02986              * It doesn't support monitor mode.
02987              */
02988             return PCAP_ERROR_RFMON_NOTSUP;
02989         }
02990 
02991         /*
02992          * It uses cooked mode.
02993          */
02994         handle->md.cooked = 1;
02995         handle->linktype = DLT_LINUX_SLL;
02996 
02997         /*
02998          * We're not bound to a device.
02999          * For now, we're using this as an indication
03000          * that we can't transmit; stop doing that only
03001          * if we figure out how to transmit in cooked
03002          * mode.
03003          */
03004         handle->md.ifindex = -1;
03005     }
03006 
03007     /*
03008      * Select promiscuous mode on if "promisc" is set.
03009      *
03010      * Do not turn allmulti mode on if we don't select
03011      * promiscuous mode - on some devices (e.g., Orinoco
03012      * wireless interfaces), allmulti mode isn't supported
03013      * and the driver implements it by turning promiscuous
03014      * mode on, and that screws up the operation of the
03015      * card as a normal networking interface, and on no
03016      * other platform I know of does starting a non-
03017      * promiscuous capture affect which multicast packets
03018      * are received by the interface.
03019      */
03020 
03021     /*
03022      * Hmm, how can we set promiscuous mode on all interfaces?
03023      * I am not sure if that is possible at all.  For now, we
03024      * silently ignore attempts to turn promiscuous mode on
03025      * for the "any" device (so you don't have to explicitly
03026      * disable it in programs such as tcpdump).
03027      */
03028 
03029     if (!is_any_device && handle->opt.promisc) {
03030         memset(&mr, 0, sizeof(mr));
03031         mr.mr_ifindex = handle->md.ifindex;
03032         mr.mr_type    = PACKET_MR_PROMISC;
03033         if (setsockopt(sock_fd, SOL_PACKET, PACKET_ADD_MEMBERSHIP,
03034             &mr, sizeof(mr)) == -1) {
03035             snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
03036                 "setsockopt: %s", pcap_strerror(errno));
03037             close(sock_fd);
03038             return PCAP_ERROR;
03039         }
03040     }
03041 
03042     /* Enable auxillary data if supported and reserve room for
03043      * reconstructing VLAN headers. */
03044 #ifdef HAVE_PACKET_AUXDATA
03045     val = 1;
03046     if (setsockopt(sock_fd, SOL_PACKET, PACKET_AUXDATA, &val,
03047                sizeof(val)) == -1 && errno != ENOPROTOOPT) {
03048         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
03049              "setsockopt: %s", pcap_strerror(errno));
03050         close(sock_fd);
03051         return PCAP_ERROR;
03052     }
03053     handle->offset += VLAN_TAG_LEN;
03054 #endif /* HAVE_PACKET_AUXDATA */
03055 
03056     /*
03057      * This is a 2.2[.x] or later kernel (we know that
03058      * because we're not using a SOCK_PACKET socket -
03059      * PF_PACKET is supported only in 2.2 and later
03060      * kernels).
03061      *
03062      * We can safely pass "recvfrom()" a byte count
03063      * based on the snapshot length.
03064      *
03065      * If we're in cooked mode, make the snapshot length
03066      * large enough to hold a "cooked mode" header plus
03067      * 1 byte of packet data (so we don't pass a byte
03068      * count of 0 to "recvfrom()").
03069      */
03070     if (handle->md.cooked) {
03071         if (handle->snapshot < SLL_HDR_LEN + 1)
03072             handle->snapshot = SLL_HDR_LEN + 1;
03073     }
03074     handle->bufsize = handle->snapshot;
03075 
03076     /* Save the socket FD in the pcap structure */
03077     handle->fd = sock_fd;
03078 
03079     return 1;
03080 #else
03081     strncpy(ebuf,
03082         "New packet capturing interface not supported by build "
03083         "environment", PCAP_ERRBUF_SIZE);
03084     return 0;
03085 #endif
03086 }

03190 static int
03191 create_ring(pcap_t *handle)
03192 {
03193     unsigned i, j, frames_per_block;
03194     struct tpacket_req req;
03195 
03196     /* Note that with large snapshot (say 64K) only a few frames
03197      * will be available in the ring even with pretty large ring size
03198      * (and a lot of memory will be unused).
03199      * The snap len should be carefully chosen to achive best
03200      * performance */
03201     req.tp_frame_size = TPACKET_ALIGN(handle->snapshot +
03202                       TPACKET_ALIGN(handle->md.tp_hdrlen) +
03203                       sizeof(struct sockaddr_ll));
03204     req.tp_frame_nr = handle->opt.buffer_size/req.tp_frame_size;
03205 
03206     /* compute the minumum block size that will handle this frame.
03207      * The block has to be page size aligned.
03208      * The max block size allowed by the kernel is arch-dependent and
03209      * it's not explicitly checked here. */
03210     req.tp_block_size = getpagesize();
03211     while (req.tp_block_size < req.tp_frame_size)
03212         req.tp_block_size <<= 1;
03213 
03214     frames_per_block = req.tp_block_size/req.tp_frame_size;
03215 
03216     /* ask the kernel to create the ring */
03217 retry:
03218     req.tp_block_nr = req.tp_frame_nr / frames_per_block;
03219 
03220     /* req.tp_frame_nr is requested to match frames_per_block*req.tp_block_nr */
03221     req.tp_frame_nr = req.tp_block_nr * frames_per_block;
03222 
03223     if (setsockopt(handle->fd, SOL_PACKET, PACKET_RX_RING,
03224                     (void *) &req, sizeof(req))) {
03225         if ((errno == ENOMEM) && (req.tp_block_nr > 1)) {
03226             /*
03227              * Memory failure; try to reduce the requested ring
03228              * size.
03229              *
03230              * We used to reduce this by half -- do 5% instead.
03231              * That may result in more iterations and a longer
03232              * startup, but the user will be much happier with
03233              * the resulting buffer size.
03234              */
03235             if (req.tp_frame_nr < 20)
03236                 req.tp_frame_nr -= 1;
03237             else
03238                 req.tp_frame_nr -= req.tp_frame_nr/20;
03239             goto retry;
03240         }
03241         if (errno == ENOPROTOOPT) {
03242             /*
03243              * We don't have ring buffer support in this kernel.
03244              */
03245             return 0;
03246         }
03247         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
03248             "can't create rx ring on packet socket: %s",
03249             pcap_strerror(errno));
03250         return -1;
03251     }
03252 
03253     /* memory map the rx ring */
03254     handle->md.mmapbuflen = req.tp_block_nr * req.tp_block_size;
03255     handle->md.mmapbuf = mmap(0, handle->md.mmapbuflen,
03256         PROT_READ|PROT_WRITE, MAP_SHARED, handle->fd, 0);
03257     if (handle->md.mmapbuf == MAP_FAILED) {
03258         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
03259             "can't mmap rx ring: %s", pcap_strerror(errno));
03260 
03261         /* clear the allocated ring on error*/
03262         destroy_ring(handle);
03263         return -1;
03264     }
03265 
03266     /* allocate a ring for each frame header pointer*/
03267     handle->cc = req.tp_frame_nr;
03268     handle->buffer = malloc(handle->cc * sizeof(union thdr *));
03269     if (!handle->buffer) {
03270         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
03271             "can't allocate ring of frame headers: %s",
03272             pcap_strerror(errno));
03273 
03274         destroy_ring(handle);
03275         return -1;
03276     }
03277 
03278     /* fill the header ring with proper frame ptr*/
03279     handle->offset = 0;
03280     for (i=0; i<req.tp_block_nr; ++i) {
03281         void *base = &handle->md.mmapbuf[i*req.tp_block_size];
03282         for (j=0; j<frames_per_block; ++j, ++handle->offset) {
03283             RING_GET_FRAME(handle) = base;
03284             base += req.tp_frame_size;
03285         }
03286     }
03287 
03288     handle->bufsize = req.tp_frame_size;
03289     handle->offset = 0;
03290     return 1;
03291 }

03293 /* free all ring related resources*/
03294 static void
03295 destroy_ring(pcap_t *handle)
03296 {
03297     /* tell the kernel to destroy the ring*/
03298     struct tpacket_req req;
03299     memset(&req, 0, sizeof(req));
03300     setsockopt(handle->fd, SOL_PACKET, PACKET_RX_RING,
03301                 (void *) &req, sizeof(req));
03302 
03303     /* if ring is mapped, unmap it*/
03304     if (handle->md.mmapbuf) {
03305         /* do not test for mmap failure, as we can't recover from any error */
03306         munmap(handle->md.mmapbuf, handle->md.mmapbuflen);
03307         handle->md.mmapbuf = NULL;
03308     }
03309 }

03311 /*
03312  * Special one-shot callback, used for pcap_next() and pcap_next_ex(),
03313  * for Linux mmapped capture.
03314  *
03315  * The problem is that pcap_next() and pcap_next_ex() expect the packet
03316  * data handed to the callback to be valid after the callback returns,
03317  * but pcap_read_linux_mmap() has to release that packet as soon as
03318  * the callback returns (otherwise, the kernel thinks there's still
03319  * at least one unprocessed packet available in the ring, so a select()
03320  * will immediately return indicating that there's data to process), so,
03321  * in the callback, we have to make a copy of the packet.
03322  *
03323  * Yes, this means that, if the capture is using the ring buffer, using
03324  * pcap_next() or pcap_next_ex() requires more copies than using
03325  * pcap_loop() or pcap_dispatch().  If that bothers you, don't use
03326  * pcap_next() or pcap_next_ex().
03327  */
03328 static void
03329 pcap_oneshot_mmap(u_char *user, const struct pcap_pkthdr *h,
03330     const u_char *bytes)
03331 {
03332     struct oneshot_userdata *sp = (struct oneshot_userdata *)user;
03333 
03334     *sp->hdr = *h;
03335     memcpy(sp->pd->md.oneshot_buffer, bytes, h->caplen);
03336     *sp->pkt = sp->pd->md.oneshot_buffer;
03337 }

03342     destroy_ring(handle);
03343     if (handle->md.oneshot_buffer != NULL) {
03344         free(handle->md.oneshot_buffer);
03345         handle->md.oneshot_buffer = NULL;
03346     }
03347     pcap_cleanup_linux(handle);
03348 }

03410 #ifndef POLLRDHUP
03411 #define POLLRDHUP 0
03412 #endif
03413 
03414 static int
03415 pcap_read_linux_mmap(pcap_t *handle, int max_packets, pcap_handler callback,
03416         u_char *user)
03417 {
03418     int timeout;
03419     int pkts = 0;
03420     char c;
03421 
03422     /* wait for frames availability.*/
03423     if (!pcap_get_ring_frame(handle, TP_STATUS_USER)) {
03424         struct pollfd pollinfo;
03425         int ret;
03426 
03427         pollinfo.fd = handle->fd;
03428         pollinfo.events = POLLIN;
03429 
03430         if (handle->md.timeout == 0)
03431             timeout = -1;   /* block forever */
03432         else if (handle->md.timeout > 0)
03433             timeout = handle->md.timeout;   /* block for that amount of time */
03434         else
03435             timeout = 0;    /* non-blocking mode - poll to pick up errors */
03436         do {
03437             ret = poll(&pollinfo, 1, timeout);
03438             if (ret < 0 && errno != EINTR) {
03439                 snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
03440                     "can't poll on packet socket: %s",
03441                     pcap_strerror(errno));
03442                 return PCAP_ERROR;
03443             } else if (ret > 0 &&
03444                 (pollinfo.revents & (POLLHUP|POLLRDHUP|POLLERR|POLLNVAL))) {
03445                 /*
03446                  * There's some indication other than
03447                  * "you can read on this descriptor" on
03448                  * the descriptor.
03449                  */
03450                 if (pollinfo.revents & (POLLHUP | POLLRDHUP)) {
03451                     snprintf(handle->errbuf,
03452                         PCAP_ERRBUF_SIZE,
03453                         "Hangup on packet socket");
03454                     return PCAP_ERROR;
03455                 }
03456                 if (pollinfo.revents & POLLERR) {
03457                     /*
03458                      * A recv() will give us the
03459                      * actual error code.
03460                      *
03461                      * XXX - make the socket non-blocking?
03462                      */
03463                     if (recv(handle->fd, &c, sizeof c,
03464                         MSG_PEEK) != -1)
03465                         continue;   /* what, no error? */
03466                     if (errno == ENETDOWN) {
03467                         /*
03468                          * The device on which we're
03469                          * capturing went away.
03470                          *
03471                          * XXX - we should really return
03472                          * PCAP_ERROR_IFACE_NOT_UP,
03473                          * but pcap_dispatch() etc.
03474                          * aren't defined to return
03475                          * that.
03476                          */
03477                         snprintf(handle->errbuf,
03478                             PCAP_ERRBUF_SIZE,
03479                             "The interface went down");
03480                     } else {
03481                         snprintf(handle->errbuf,
03482                             PCAP_ERRBUF_SIZE,
03483                             "Error condition on packet socket: %s",
03484                             strerror(errno));
03485                     }
03486                     return PCAP_ERROR;
03487                 }
03488                 if (pollinfo.revents & POLLNVAL) {
03489                     snprintf(handle->errbuf,
03490                         PCAP_ERRBUF_SIZE,
03491                         "Invalid polling request on packet socket");
03492                     return PCAP_ERROR;
03493                 }
03494             }
03495             /* check for break loop condition on interrupted syscall*/
03496             if (handle->break_loop) {
03497                 handle->break_loop = 0;
03498                 return PCAP_ERROR_BREAK;
03499             }
03500         } while (ret < 0);
03501     }
03502 
03503     /* non-positive values of max_packets are used to require all
03504      * packets currently available in the ring */
03505     while ((pkts < max_packets) || (max_packets <= 0)) {
03506         int run_bpf;
03507         struct sockaddr_ll *sll;
03508         struct pcap_pkthdr pcaphdr;
03509         unsigned char *bp;
03510         union thdr h;
03511         unsigned int tp_len;
03512         unsigned int tp_mac;
03513         unsigned int tp_snaplen;
03514         unsigned int tp_sec;
03515         unsigned int tp_usec;
03516 
03517         h.raw = pcap_get_ring_frame(handle, TP_STATUS_USER);
03518         if (!h.raw)
03519             break;
03520 
03521         switch (handle->md.tp_version) {
03522         case TPACKET_V1:
03523             tp_len     = h.h1->tp_len;
03524             tp_mac     = h.h1->tp_mac;
03525             tp_snaplen = h.h1->tp_snaplen;
03526             tp_sec     = h.h1->tp_sec;
03527             tp_usec    = h.h1->tp_usec;
03528             break;
03529 #ifdef HAVE_TPACKET2
03530         case TPACKET_V2:
03531             tp_len     = h.h2->tp_len;
03532             tp_mac     = h.h2->tp_mac;
03533             tp_snaplen = h.h2->tp_snaplen;
03534             tp_sec     = h.h2->tp_sec;
03535             tp_usec    = h.h2->tp_nsec / 1000;
03536             break;
03537 #endif
03538         default:
03539             snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
03540                 "unsupported tpacket version %d",
03541                 handle->md.tp_version);
03542             return -1;
03543         }
03544         /* perform sanity check on internal offset. */
03545         if (tp_mac + tp_snaplen > handle->bufsize) {
03546             snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
03547                 "corrupted frame on kernel ring mac "
03548                 "offset %d + caplen %d > frame len %d",
03549                 tp_mac, tp_snaplen, handle->bufsize);
03550             return -1;
03551         }
03552 
03553         /* run filter on received packet
03554          * If the kernel filtering is enabled we need to run the
03555          * filter until all the frames present into the ring
03556          * at filter creation time are processed.
03557          * In such case md.use_bpf is used as a counter for the
03558          * packet we need to filter.
03559          * Note: alternatively it could be possible to stop applying
03560          * the filter when the ring became empty, but it can possibly
03561          * happen a lot later... */
03562         bp = (unsigned char*)h.raw + tp_mac;
03563         run_bpf = (!handle->md.use_bpf) ||
03564             ((handle->md.use_bpf>1) && handle->md.use_bpf--);
03565         if (run_bpf && handle->fcode.bf_insns &&
03566                 (bpf_filter(handle->fcode.bf_insns, bp,
03567                     tp_len, tp_snaplen) == 0))
03568             goto skip;
03569 
03570         /*
03571          * Do checks based on packet direction.
03572          */
03573         sll = (void *)h.raw + TPACKET_ALIGN(handle->md.tp_hdrlen);
03574         if (sll->sll_pkttype == PACKET_OUTGOING) {
03575             /*
03576              * Outgoing packet.
03577              * If this is from the loopback device, reject it;
03578              * we'll see the packet as an incoming packet as well,
03579              * and we don't want to see it twice.
03580              */
03581             if (sll->sll_ifindex == handle->md.lo_ifindex)
03582                 goto skip;
03583 
03584             /*
03585              * If the user only wants incoming packets, reject it.
03586              */
03587             if (handle->direction == PCAP_D_IN)
03588                 goto skip;
03589         } else {
03590             /*
03591              * Incoming packet.
03592              * If the user only wants outgoing packets, reject it.
03593              */
03594             if (handle->direction == PCAP_D_OUT)
03595                 goto skip;
03596         }
03597 
03598         /* get required packet info from ring header */
03599         pcaphdr.ts.tv_sec = tp_sec;
03600         pcaphdr.ts.tv_usec = tp_usec;
03601         pcaphdr.caplen = tp_snaplen;
03602         pcaphdr.len = tp_len;
03603 
03604         /* if required build in place the sll header*/
03605         if (handle->md.cooked) {
03606             struct sll_header *hdrp;
03607 
03608             /*
03609              * The kernel should have left us with enough
03610              * space for an sll header; back up the packet
03611              * data pointer into that space, as that'll be
03612              * the beginning of the packet we pass to the
03613              * callback.
03614              */
03615             bp -= SLL_HDR_LEN;
03616 
03617             /*
03618              * Let's make sure that's past the end of
03619              * the tpacket header, i.e. >=
03620              * ((u_char *)thdr + TPACKET_HDRLEN), so we
03621              * don't step on the header when we construct
03622              * the sll header.
03623              */
03624             if (bp < (u_char *)h.raw +
03625                        TPACKET_ALIGN(handle->md.tp_hdrlen) +
03626                        sizeof(struct sockaddr_ll)) {
03627                 snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
03628                     "cooked-mode frame doesn't have room for sll header");
03629                 return -1;
03630             }
03631 
03632             /*
03633              * OK, that worked; construct the sll header.
03634              */
03635             hdrp = (struct sll_header *)bp;
03636             hdrp->sll_pkttype = map_packet_type_to_sll_type(
03637                             sll->sll_pkttype);
03638             hdrp->sll_hatype = htons(sll->sll_hatype);
03639             hdrp->sll_halen = htons(sll->sll_halen);
03640             memcpy(hdrp->sll_addr, sll->sll_addr, SLL_ADDRLEN);
03641             hdrp->sll_protocol = sll->sll_protocol;
03642 
03643             /* update packet len */
03644             pcaphdr.caplen += SLL_HDR_LEN;
03645             pcaphdr.len += SLL_HDR_LEN;
03646         }
03647 
03648 #ifdef HAVE_TPACKET2
03649         if (handle->md.tp_version == TPACKET_V2 && h.h2->tp_vlan_tci &&
03650             tp_snaplen >= 2 * ETH_ALEN) {
03651             struct vlan_tag *tag;
03652 
03653             bp -= VLAN_TAG_LEN;
03654             memmove(bp, bp + VLAN_TAG_LEN, 2 * ETH_ALEN);
03655 
03656             tag = (struct vlan_tag *)(bp + 2 * ETH_ALEN);
03657             tag->vlan_tpid = htons(ETH_P_8021Q);
03658             tag->vlan_tci = htons(h.h2->tp_vlan_tci);
03659 
03660             pcaphdr.caplen += VLAN_TAG_LEN;
03661             pcaphdr.len += VLAN_TAG_LEN;
03662         }
03663 #endif
03664 
03665         /*
03666          * The only way to tell the kernel to cut off the
03667          * packet at a snapshot length is with a filter program;
03668          * if there's no filter program, the kernel won't cut
03669          * the packet off.
03670          *
03671          * Trim the snapshot length to be no longer than the
03672          * specified snapshot length.
03673          */
03674         if (pcaphdr.caplen > handle->snapshot)
03675             pcaphdr.caplen = handle->snapshot;
03676 
03677         /* pass the packet to the user */
03678         pkts++;
03679 
03680 #if defined(HAVE_PF_RING)
03681         {
03682           struct ns_pcaphdr myhdr;
03683 
03684           myhdr.ts.tv_sec = pcaphdr.ts.tv_sec;
03685           myhdr.ts.tv_usec = pcaphdr.ts.tv_usec;
03686           
03687           myhdr.caplen = pcaphdr.caplen, myhdr.len = pcaphdr.len;
03688 
03689           if(handle->ring != NULL) {
03690             myhdr.ns = 0;
03691           } else {
03692 #if defined(HAVE_TPACKET2)
03693             myhdr.ns = ((u_int64_t)h.h2->tp_sec * 1000000000) + h.h2->tp_nsec;
03694 #else
03695             myhdr.ns = 0;
03696 #endif
03697           }
03698 
03699           callback(user, (struct pcap_pkthdr*)&myhdr, bp);
03700         }
03701 #else
03702         callback(user, &pcaphdr, bp);
03703 #endif
03704 
03705         handle->md.packets_read++;
03706 
03707 skip:
03708         /* next packet */
03709         switch (handle->md.tp_version) {
03710         case TPACKET_V1:
03711             h.h1->tp_status = TP_STATUS_KERNEL;
03712             break;
03713 #ifdef HAVE_TPACKET2
03714         case TPACKET_V2:
03715             h.h2->tp_status = TP_STATUS_KERNEL;
03716             break;
03717 #endif
03718         }
03719         if (++handle->offset >= handle->cc)
03720             handle->offset = 0;
03721 
03722         /* check for break loop condition*/
03723         if (handle->break_loop) {
03724             handle->break_loop = 0;
03725             return PCAP_ERROR_BREAK;
03726         }
03727     }
03728     return pkts;
03729 }

03733 {
03734     int n, offset;
03735     int ret;
03736 
03737     /*
03738      * Don't rewrite "ret" instructions; we don't need to, as

03748      * all packets present into the ring. Get an upper bound of their number

03750     if (!handle->md.use_bpf)

03759             handle->offset = handle->cc - 1;
03760         if (!pcap_get_ring_frame(handle, TP_STATUS_KERNEL))
03761             break;

03765     handle->offset = offset;
03766 
03767     /* store the number of packets currently present in the ring */
03768     handle->md.use_bpf = 1 + (handle->cc - n);

04507 /*
04508  * Try to open a packet socket using the old kernel interface.
04509  * Returns 1 on success and a PCAP_ERROR_ value on an error.
04510  */
04511 static int
04512 activate_old(pcap_t *handle)
04513 {
04514     int     arptype;
04515     struct ifreq    ifr;
04516     const char  *device = handle->opt.source;
04517     struct utsname  utsname;
04518     int     mtu;
04519 
04520     /* Open the socket */
04521 
04522     handle->fd = socket(PF_INET, SOCK_PACKET, htons(ETH_P_ALL));
04523     if (handle->fd == -1) {
04524         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
04525              "socket: %s", pcap_strerror(errno));
04526         return PCAP_ERROR_PERM_DENIED;
04527     }
04528 
04529     /* It worked - we are using the old interface */
04530     handle->md.sock_packet = 1;
04531 
04532     /* ...which means we get the link-layer header. */
04533     handle->md.cooked = 0;
04534 
04535     /* Bind to the given device */
04536 
04537     if (strcmp(device, "any") == 0) {
04538         strncpy(handle->errbuf, "pcap_activate: The \"any\" device isn't supported on 2.0[.x]-kernel systems",
04539             PCAP_ERRBUF_SIZE);
04540         return PCAP_ERROR;
04541     }
04542     if (iface_bind_old(handle->fd, device, handle->errbuf) == -1)
04543         return PCAP_ERROR;
04544 
04545     /*
04546      * Try to get the link-layer type.
04547      */
04548     arptype = iface_get_arptype(handle->fd, device, handle->errbuf);
04549     if (arptype < 0)
04550         return PCAP_ERROR;
04551 
04552     /*
04553      * Try to find the DLT_ type corresponding to that
04554      * link-layer type.
04555      */
04556     map_arphrd_to_dlt(handle, arptype, 0);
04557     if (handle->linktype == -1) {
04558         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
04559              "unknown arptype %d", arptype);
04560         return PCAP_ERROR;
04561     }
04562 
04563     /* Go to promisc mode if requested */
04564 
04565     if (handle->opt.promisc) {
04566         memset(&ifr, 0, sizeof(ifr));
04567         strncpy(ifr.ifr_name, device, sizeof(ifr.ifr_name));
04568         if (ioctl(handle->fd, SIOCGIFFLAGS, &ifr) == -1) {
04569             snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
04570                  "SIOCGIFFLAGS: %s", pcap_strerror(errno));
04571             return PCAP_ERROR;
04572         }
04573         if ((ifr.ifr_flags & IFF_PROMISC) == 0) {
04574             /*
04575              * Promiscuous mode isn't currently on,
04576              * so turn it on, and remember that
04577              * we should turn it off when the
04578              * pcap_t is closed.
04579              */
04580 
04581             /*
04582              * If we haven't already done so, arrange
04583              * to have "pcap_close_all()" called when
04584              * we exit.
04585              */
04586             if (!pcap_do_addexit(handle)) {
04587                 /*
04588                  * "atexit()" failed; don't put
04589                  * the interface in promiscuous
04590                  * mode, just give up.
04591                  */
04592                 return PCAP_ERROR;
04593             }
04594 
04595             ifr.ifr_flags |= IFF_PROMISC;
04596             if (ioctl(handle->fd, SIOCSIFFLAGS, &ifr) == -1) {
04597                     snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
04598                      "SIOCSIFFLAGS: %s",
04599                      pcap_strerror(errno));
04600                 return PCAP_ERROR;
04601             }
04602             handle->md.must_do_on_close |= MUST_CLEAR_PROMISC;
04603 
04604             /*
04605              * Add this to the list of pcaps
04606              * to close when we exit.
04607              */
04608             pcap_add_to_pcaps_to_close(handle);
04609         }
04610     }
04611 
04612     /*
04613      * Compute the buffer size.
04614      *
04615      * We're using SOCK_PACKET, so this might be a 2.0[.x]
04616      * kernel, and might require special handling - check.
04617      */
04618     if (uname(&utsname) < 0 ||
04619         strncmp(utsname.release, "2.0", 3) == 0) {
04620         /*
04621          * Either we couldn't find out what kernel release
04622          * this is, or it's a 2.0[.x] kernel.
04623          *
04624          * In the 2.0[.x] kernel, a "recvfrom()" on
04625          * a SOCK_PACKET socket, with MSG_TRUNC set, will
04626          * return the number of bytes read, so if we pass
04627          * a length based on the snapshot length, it'll
04628          * return the number of bytes from the packet
04629          * copied to userland, not the actual length
04630          * of the packet.
04631          *
04632          * This means that, for example, the IP dissector
04633          * in tcpdump will get handed a packet length less
04634          * than the length in the IP header, and will
04635          * complain about "truncated-ip".
04636          *
04637          * So we don't bother trying to copy from the
04638          * kernel only the bytes in which we're interested,
04639          * but instead copy them all, just as the older
04640          * versions of libpcap for Linux did.
04641          *
04642          * The buffer therefore needs to be big enough to
04643          * hold the largest packet we can get from this
04644          * device.  Unfortunately, we can't get the MRU
04645          * of the network; we can only get the MTU.  The
04646          * MTU may be too small, in which case a packet larger
04647          * than the buffer size will be truncated *and* we
04648          * won't get the actual packet size.
04649          *
04650          * However, if the snapshot length is larger than
04651          * the buffer size based on the MTU, we use the
04652          * snapshot length as the buffer size, instead;
04653          * this means that with a sufficiently large snapshot
04654          * length we won't artificially truncate packets
04655          * to the MTU-based size.
04656          *
04657          * This mess just one of many problems with packet
04658          * capture on 2.0[.x] kernels; you really want a
04659          * 2.2[.x] or later kernel if you want packet capture
04660          * to work well.
04661          */
04662         mtu = iface_get_mtu(handle->fd, device, handle->errbuf);
04663         if (mtu == -1)
04664             return PCAP_ERROR;
04665         handle->bufsize = MAX_LINKHEADER_SIZE + mtu;
04666         if (handle->bufsize < handle->snapshot)
04667             handle->bufsize = handle->snapshot;
04668     } else {
04669         /*
04670          * This is a 2.2[.x] or later kernel.
04671          *
04672          * We can safely pass "recvfrom()" a byte count
04673          * based on the snapshot length.
04674          */
04675         handle->bufsize = handle->snapshot;
04676     }
04677 
04678     /*
04679      * Default value for offset to align link-layer payload
04680      * on a 4-byte boundary.
04681      */
04682     handle->offset   = 0;
04683 
04684     return 1;
04685 }

04794 static int
04795 fix_program(pcap_t *handle, struct sock_fprog *fcode, int is_mmapped)
04796 {
04797     size_t prog_size;
04798     register int i;
04799     register struct bpf_insn *p;
04800     struct bpf_insn *f;

04802 
04803     /*
04804      * Make a copy of the filter, and modify that copy if
04805      * necessary.
04806      */
04807     prog_size = sizeof(*handle->fcode.bf_insns) * handle->fcode.bf_len;
04808     len = handle->fcode.bf_len;
04809     f = (struct bpf_insn *)malloc(prog_size);
04810     if (f == NULL) {
04811         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,

04813         return -1;

04816     fcode->len = len;

04818 
04819     for (i = 0; i < len; ++i) {

04822          * What type of instruction is this?
04823          */
04824         switch (BPF_CLASS(p->code)) {
04825

04828              * It's a return instruction; are we capturing
04829              * in memory-mapped mode?
04830              */
04831             if (!is_mmapped) {
04832                 /*
04833                  * No; is the snapshot length a constant,
04834                  * rather than the contents of the
04835                  * accumulator?
04836                  */
04837                 if (BPF_MODE(p->code) == BPF_K) {
04838                     /*
04839                      * Yes - if the value to be returned,
04840                      * i.e. the snapshot length, is
04841                      * anything other than 0, make it
04842                      * 65535, so that the packet is
04843                      * truncated by "recvfrom()",
04844                      * not by the filter.
04845                      *
04846                      * XXX - there's nothing we can
04847                      * easily do if it's getting the
04848                      * value from the accumulator; we'd
04849                      * have to insert code to force
04850                      * non-zero values to be 65535.
04851                      */
04852                     if (p->k != 0)
04853                         p->k = 65535;
04854                 }
04855             }
04856             break;
04857 
04858         case BPF_LD:
04859         case BPF_LDX:
04860             /*
04861              * It's a load instruction; is it loading
04862              * from the packet?
04863              */
04864             switch (BPF_MODE(p->code)) {
04865 
04866             case BPF_ABS:
04867             case BPF_IND:
04868             case BPF_MSH:

04871                  */

04875                      * instruction.

04880                          * Return 0, so our caller
04881                          * knows to punt to userland.
04882                          */
04883                         return 0;
04884                     }
04885                 }
04886                 break;
04887             }
04888             break;
04889         }
04890     }
04891     return 1;   /* we succeeded */

04894 static int
04895 fix_offset(struct bpf_insn *p)
04896 {
04897     /*
04898      * What's the offset?
04899      */
04900     if (p->k >= SLL_HDR_LEN) {
04901         /*
04902          * It's within the link-layer payload; that starts at an
04903          * offset of 0, as far as the kernel packet filter is
04904          * concerned, so subtract the length of the link-layer
04905          * header.
04906          */
04907         p->k -= SLL_HDR_LEN;
04908     } else if (p->k == 14) {
04909         /*
04910          * It's the protocol field; map it to the special magic
04911          * kernel offset for that field.
04912          */
04913         p->k = SKF_AD_OFF + SKF_AD_PROTOCOL;
04914     } else {
04915         /*
04916          * It's within the header, but it's not one of those
04917          * fields; we can't do that in the kernel, so punt
04918          * to userland.
04919          */
04920         return -1;
04921     }
04922     return 0;
04923 }

04925 static int
04926 set_kernel_filter(pcap_t *handle, struct sock_fprog *fcode)
04927 {
04928     int total_filter_on = 0;
04929     int save_mode;
04930     int ret;
04931     int save_errno;
04932 
04933     /*
04934      * The socket filter code doesn't discard all packets queued
04935      * up on the socket when the filter is changed; this means
04936      * that packets that don't match the new filter may show up
04937      * after the new filter is put onto the socket, if those
04938      * packets haven't yet been read.
04939      *
04940      * This means, for example, that if you do a tcpdump capture
04941      * with a filter, the first few packets in the capture might
04942      * be packets that wouldn't have passed the filter.
04943      *
04944      * We therefore discard all packets queued up on the socket
04945      * when setting a kernel filter.  (This isn't an issue for
04946      * userland filters, as the userland filtering is done after
04947      * packets are queued up.)
04948      *
04949      * To flush those packets, we put the socket in read-only mode,
04950      * and read packets from the socket until there are no more to
04951      * read.
04952      *
04953      * In order to keep that from being an infinite loop - i.e.,
04954      * to keep more packets from arriving while we're draining
04955      * the queue - we put the "total filter", which is a filter
04956      * that rejects all packets, onto the socket before draining
04957      * the queue.
04958      *
04959      * This code deliberately ignores any errors, so that you may
04960      * get bogus packets if an error occurs, rather than having
04961      * the filtering done in userland even if it could have been
04962      * done in the kernel.
04963      */

05003 #endif
05004     }
05005     /*
05006      * Now attach the new filter.
05007      */
05008     ret = setsockopt(handle->fd,
05009 #ifdef HAVE_PF_RING
05010              0,
05011 #else
05012              SOL_SOCKET,
05013 #endif
05014              SO_ATTACH_FILTER,
05015              fcode, sizeof(*fcode));
05016     if (ret == -1 && total_filter_on) {
05017         /*
05018          * Well, we couldn't set that filter on the socket,
05019          * but we could set the total filter on the socket.
05020          *
05021          * This could, for example, mean that the filter was
05022          * too big to put into the kernel, so we'll have to
05023          * filter in userland; in any case, we'll be doing
05024          * filtering in userland, so we need to remove the
05025          * total filter so we see packets.
05026          */
05027         save_errno = errno;
05028 
05029         /*
05030          * XXX - if this fails, we're really screwed;
05031          * we have the total filter on the socket,
05032          * and it won't come off.  What do we do then?
05033          */
05034         reset_kernel_filter(handle);
05035 
05036         errno = save_errno;
05037     }
05038     return ret;
