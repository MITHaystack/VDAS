00059  * The time difference in millisecond
00060  */
00061 double delta_time (struct timeval * now,
00062            struct timeval * before) {
00063   time_t delta_seconds;
00064   time_t delta_microseconds;
00065 
00066   /*
00067    * compute delta in second, 1/10's and 1/1000's second units
00068    */
00069   delta_seconds      = now -> tv_sec  - before -> tv_sec;
00070   delta_microseconds = now -> tv_usec - before -> tv_usec;
00071 
00072   if(delta_microseconds < 0) {
00073     /* manually carry a one from the seconds field */
00074     delta_microseconds += 1000000;  /* 1e6 */
00075     -- delta_seconds;
00076   }
00077   return((double)(delta_seconds * 1000) + (double)delta_microseconds/1000);
00078 }

00179 char* _intoa(unsigned int addr, char* buf, u_short bufLen) {
00180   char *cp, *retStr;
00181   u_int byte;
00182   int n;
00183 
00184   cp = &buf[bufLen];
00185   *--cp = '\0';
00186 
00187   n = 4;
00188   do {
00189     byte = addr & 0xff;
00190     *--cp = byte % 10 + '0';
00191     byte /= 10;
00192     if (byte > 0) {
00193       *--cp = byte % 10 + '0';
00194       byte /= 10;
00195       if (byte > 0)
00196     *--cp = byte + '0';
00197     }
00198     *--cp = '.';
00199     addr >>= 8;
00200   } while (--n > 0);
00201 
00202   /* Convert the string to lowercase */
00203   retStr = (char*)(cp+1);
00204 
00205   return(retStr);

00282 int32_t gmt2local(time_t t) {
00283   int dt, dir;
00284   struct tm *gmt, *loc;
00285   struct tm sgmt;
00286 
00287   if (t == 0)
00288     t = time(NULL);
00289   gmt = &sgmt;
00290   *gmt = *gmtime(&t);
00291   loc = localtime(&t);
00292   dt = (loc->tm_hour - gmt->tm_hour) * 60 * 60 +
00293     (loc->tm_min - gmt->tm_min) * 60;
00294 
00295   /*
00296    * If the year or julian day is different, we span 00:00 GMT
00297    * and must add or subtract a day. Check the year first to
00298    * avoid problems when the julian day wraps.
00299    */
00300   dir = loc->tm_year - gmt->tm_year;
00301   if (dir == 0)
00302     dir = loc->tm_yday - gmt->tm_yday;
00303   dt += dir * 24 * 60 * 60;
00304 
00305   return (dt);
00306 }
