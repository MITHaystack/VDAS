00208 read_bytes(FILE *fp, void *buf, size_t bytes_to_read, int fail_on_eof,
00209     char *errbuf)
00210 {
00211     size_t amt_read;
00212 
00213     amt_read = fread(buf, 1, bytes_to_read, fp);
00214     if (amt_read != bytes_to_read) {
00215         if (ferror(fp)) {
00216             snprintf(errbuf, PCAP_ERRBUF_SIZE,
00217                 "error reading dump file: %s",
00218                 pcap_strerror(errno));
00219         } else {
00220             if (amt_read == 0 && !fail_on_eof)
00221                 return (0); /* EOF */

00223                 "truncated dump file; tried to read %lu bytes, only got %lu",
00224                 (unsigned long)bytes_to_read,

00226         }
00227         return (-1);

00232 static int
00233 read_block(FILE *fp, pcap_t *p, struct block_cursor *cursor, char *errbuf)
00234 {
00235     int status;
00236     struct block_header bhdr;
00237 
00238     status = read_bytes(fp, &bhdr, sizeof(bhdr), 0, errbuf);
00239     if (status <= 0)
00240         return (status);    /* error or EOF */
00241 
00242     if (p->sf.swapped) {
00243         bhdr.block_type = SWAPLONG(bhdr.block_type);
00244         bhdr.total_length = SWAPLONG(bhdr.total_length);
00245     }
00246 
00247     /*
00248      * Is this block "too big"?
00249      *
00250      * We choose 16MB as "too big", for now, so that we handle
00251      * "reasonably" large buffers but don't chew up all the
00252      * memory if we read a malformed file.
00253      */
00254     if (bhdr.total_length > 16*1024*1024) {
00255         snprintf(errbuf, PCAP_ERRBUF_SIZE,
00256             "pcap-ng block size %u > maximum %u",
00257             bhdr.total_length, 16*1024*1024);
00258             return (-1);
00259     }
00260 
00261     /*
00262      * Is this block "too small" - i.e., is it shorter than a block
00263      * header plus a block trailer?
00264      */
00265     if (bhdr.total_length < sizeof(struct block_header) +
00266         sizeof(struct block_trailer)) {
00267         snprintf(errbuf, PCAP_ERRBUF_SIZE,
00268             "block in pcap-ng dump file has a length of %u < %lu",
00269             bhdr.total_length,
00270             (unsigned long)(sizeof(struct block_header) + sizeof(struct block_trailer)));
00271         return (-1);
00272     }
00273 
00274     /*
00275      * Is the buffer big enough?
00276      */
00277     if (p->bufsize < bhdr.total_length) {
00278         /*
00279          * No - make it big enough.
00280          */
00281         p->buffer = realloc(p->buffer, bhdr.total_length);
00282         if (p->buffer == NULL) {
00283             snprintf(errbuf, PCAP_ERRBUF_SIZE, "out of memory");
00284             return (-1);
00285         }
00286     }
00287 
00288     /*
00289      * Copy the stuff we've read to the buffer, and read the rest
00290      * of the block.
00291      */
00292     memcpy(p->buffer, &bhdr, sizeof(bhdr));
00293     if (read_bytes(fp, p->buffer + sizeof(bhdr),
00294         bhdr.total_length - sizeof(bhdr), 1, errbuf) == -1)
00295         return (-1);
00296 
00297     /*
00298      * Initialize the cursor.
00299      */
00300     cursor->data = p->buffer + sizeof(bhdr);
00301     cursor->data_remaining = bhdr.total_length - sizeof(bhdr) -
00302         sizeof(struct block_trailer);
00303     cursor->block_type = bhdr.block_type;
00304     return (1);
00305 }

00307 static void *
00308 get_from_block_data(struct block_cursor *cursor, size_t chunk_size,
00309     char *errbuf)
00310 {
00311     void *data;
00312 
00313     /*
00314      * Make sure we have the specified amount of data remaining in
00315      * the block data.
00316      */
00317     if (cursor->data_remaining < chunk_size) {
00318         snprintf(errbuf, PCAP_ERRBUF_SIZE,
00319             "block of type %u in pcap-ng dump file is too short",
00320             cursor->block_type);
00321         return (NULL);
00322     }
00323 
00324     /*
00325      * Return the current pointer, and skip past the chunk.
00326      */
00327     data = cursor->data;
00328     cursor->data += chunk_size;
00329     cursor->data_remaining -= chunk_size;
00330     return (data);
00331 }

00379 static int
00380 process_idb_options(pcap_t *p, struct block_cursor *cursor, u_int *tsresol,
00381     u_int64_t *tsoffset, char *errbuf)
00382 {
00383     struct option_header *opthdr;
00384     void *optvalue;
00385     int saw_tsresol, saw_tsoffset;
00386     u_char tsresol_opt;
00387     u_int i;
00388 
00389     saw_tsresol = 0;
00390     saw_tsoffset = 0;
00391     while (cursor->data_remaining != 0) {
00392         /*
00393          * Get the option header.
00394          */
00395         opthdr = get_opthdr_from_block_data(p, cursor, errbuf);
00396         if (opthdr == NULL) {
00397             /*
00398              * Option header is cut short.
00399              */
00400             return (-1);
00401         }
00402 
00403         /*
00404          * Get option value.
00405          */
00406         optvalue = get_optvalue_from_block_data(cursor, opthdr,
00407             errbuf);
00408         if (optvalue == NULL) {
00409             /*
00410              * Option value is cut short.
00411              */
00412             return (-1);
00413         }
00414 
00415         switch (opthdr->option_code) {
00416 
00417         case OPT_ENDOFOPT:
00418             if (opthdr->option_length != 0) {
00419                 snprintf(errbuf, PCAP_ERRBUF_SIZE,
00420                     "Interface Description Block has opt_endofopt option with length %u != 0",
00421                     opthdr->option_length);
00422                 return (-1);
00423             }
00424             goto done;
00425 
00426         case IF_TSRESOL:
00427             if (opthdr->option_length != 1) {
00428                 snprintf(errbuf, PCAP_ERRBUF_SIZE,
00429                     "Interface Description Block has if_tsresol option with length %u != 1",
00430                     opthdr->option_length);
00431                 return (-1);
00432             }
00433             if (saw_tsresol) {
00434                 snprintf(errbuf, PCAP_ERRBUF_SIZE,
00435                     "Interface Description Block has more than one if_tsresol option");
00436                 return (-1);
00437             }
00438             saw_tsresol = 1;
00439             tsresol_opt = *(u_int *)optvalue;
00440             if (tsresol_opt & 0x80) {
00441                 /*
00442                  * Resolution is negative power of 2.
00443                  */
00444                 *tsresol = 1 << (tsresol_opt & 0x7F);
00445             } else {
00446                 /*
00447                  * Resolution is negative power of 10.
00448                  */
00449                 *tsresol = 1;
00450                 for (i = 0; i < tsresol_opt; i++)
00451                     *tsresol *= 10;
00452             }
00453             if (*tsresol == 0) {
00454                 /*
00455                  * Resolution is too high.
00456                  */
00457                 if (tsresol_opt & 0x80) {
00458                     snprintf(errbuf, PCAP_ERRBUF_SIZE,
00459                         "Interface Description Block if_tsresol option resolution 2^-%u is too high",
00460                         tsresol_opt & 0x7F);
00461                 } else {
00462                     snprintf(errbuf, PCAP_ERRBUF_SIZE,
00463                         "Interface Description Block if_tsresol option resolution 10^-%u is too high",
00464                         tsresol_opt);
00465                 }
00466                 return (-1);
00467             }
00468             break;
00469 
00470         case IF_TSOFFSET:
00471             if (opthdr->option_length != 8) {
00472                 snprintf(errbuf, PCAP_ERRBUF_SIZE,
00473                     "Interface Description Block has if_tsoffset option with length %u != 8",
00474                     opthdr->option_length);
00475                 return (-1);
00476             }
00477             if (saw_tsoffset) {
00478                 snprintf(errbuf, PCAP_ERRBUF_SIZE,
00479                     "Interface Description Block has more than one if_tsoffset option");
00480                 return (-1);
00481             }
00482             saw_tsoffset = 1;
00483             memcpy(tsoffset, optvalue, sizeof(*tsoffset));
00484             if (p->sf.swapped)
00485                 *tsoffset = SWAPLL(*tsoffset);
00486             break;
00487 
00488         default:
00489             break;
00490         }
00491     }
00492 
00493 done:
00494     return (0);
00495 }

00497 /*
00498  * Check whether this is a pcap-ng savefile and, if it is, extract the
00499  * relevant information from the header.
00500  */
00501 int
00502 pcap_ng_check_header(pcap_t *p, bpf_u_int32 magic, FILE *fp, char *errbuf)
00503 {
00504     size_t amt_read;
00505     bpf_u_int32 total_length;
00506     bpf_u_int32 byte_order_magic;
00507     struct block_header *bhdrp;
00508     struct section_header_block *shbp;
00509     int status;
00510     struct block_cursor cursor;
00511     struct interface_description_block *idbp;
00512 
00513     /*
00514      * Check whether the first 4 bytes of the file are the block
00515      * type for a pcap-ng savefile. 
00516      */
00517     if (magic != BT_SHB) {
00518         /*
00519          * XXX - check whether this looks like what the block
00520          * type would be after being munged by mapping between
00521          * UN*X and DOS/Windows text file format and, if it
00522          * does, look for the byte-order magic number in
00523          * the appropriate place and, if we find it, report
00524          * this as possibly being a pcap-ng file transferred
00525          * between UN*X and Windows in text file format?
00526          */
00527         return (0); /* nope */
00528     }
00529 
00530     /*
00531      * OK, they are.  However, that's just \n\r\r\n, so it could,
00532      * conceivably, be an ordinary text file.
00533      *
00534      * It could not, however, conceivably be any other type of
00535      * capture file, so we can read the rest of the putative
00536      * Section Header Block; put the block type in the common
00537      * header, read the rest of the common header and the
00538      * fixed-length portion of the SHB, and look for the byte-order
00539      * magic value.
00540      */
00541     amt_read = fread(&total_length, 1, sizeof(total_length), fp);
00542     if (amt_read < sizeof(total_length)) {
00543         if (ferror(fp)) {
00544             snprintf(errbuf, PCAP_ERRBUF_SIZE,
00545                 "error reading dump file: %s",
00546                 pcap_strerror(errno));
00547             return (-1);    /* fail */
00548         }
00549 
00550         /*
00551          * Possibly a weird short text file, so just say
00552          * "not pcap-ng".
00553          */
00554         return (0);
00555     }
00556     amt_read = fread(&byte_order_magic, 1, sizeof(byte_order_magic), fp);
00557     if (amt_read < sizeof(byte_order_magic)) {
00558         if (ferror(fp)) {
00559             snprintf(errbuf, PCAP_ERRBUF_SIZE,
00560                 "error reading dump file: %s",
00561                 pcap_strerror(errno));
00562             return (-1);    /* fail */
00563         }
00564 
00565         /*
00566          * Possibly a weird short text file, so just say
00567          * "not pcap-ng".
00568          */
00569         return (0);
00570     }
00571     if (byte_order_magic != BYTE_ORDER_MAGIC) {
00572         byte_order_magic = SWAPLONG(byte_order_magic);
00573         if (byte_order_magic != BYTE_ORDER_MAGIC) {
00574             /*
00575              * Not a pcap-ng file.
00576              */
00577             return (0);
00578         }
00579         p->sf.swapped = 1;
00580         total_length = SWAPLONG(total_length);
00581     }
00582 
00583     /*
00584      * Check the sanity of the total length.
00585      */
00586     if (total_length < sizeof(*bhdrp) + sizeof(*shbp) + sizeof(struct block_trailer)) {
00587         snprintf(errbuf, PCAP_ERRBUF_SIZE,
00588             "Section Header Block in pcap-ng dump file has a length of %u < %lu",
00589             total_length,
00590             (unsigned long)(sizeof(*bhdrp) + sizeof(*shbp) + sizeof(struct block_trailer)));
00591         return (-1);
00592     }
00593 
00594     /*
00595      * Allocate a buffer into which to read blocks.  We default to
00596      * the maximum of:
00597      *
00598      *  the total length of the SHB for which we read the header;
00599      *
00600      *  2K, which should be more than large enough for an Enhanced
00601      *  Packet Block containing a full-size Ethernet frame, and
00602      *  leaving room for some options.
00603      *
00604      * If we find a bigger block, we reallocate the buffer.
00605      */
00606     p->bufsize = 2048;
00607     if (p->bufsize < total_length)
00608         p->bufsize = total_length;
00609     p->buffer = malloc(p->bufsize);
00610     if (p->buffer == NULL) {
00611         snprintf(errbuf, PCAP_ERRBUF_SIZE, "out of memory");
00612         return (-1);
00613     }
00614 
00615     /*
00616      * Copy the stuff we've read to the buffer, and read the rest
00617      * of the SHB.
00618      */
00619     bhdrp = (struct block_header *)p->buffer;
00620     shbp = (struct section_header_block *)(p->buffer + sizeof(struct block_header));
00621     bhdrp->block_type = magic;
00622     bhdrp->total_length = total_length;
00623     shbp->byte_order_magic = byte_order_magic;
00624     if (read_bytes(fp,
00625         p->buffer + (sizeof(magic) + sizeof(total_length) + sizeof(byte_order_magic)),
00626         total_length - (sizeof(magic) + sizeof(total_length) + sizeof(byte_order_magic)),
00627         1, errbuf) == -1)
00628         goto fail;
00629 
00630     if (p->sf.swapped) {
00631         /*
00632          * Byte-swap the fields we've read.
00633          */
00634         shbp->major_version = SWAPSHORT(shbp->major_version);
00635         shbp->minor_version = SWAPSHORT(shbp->minor_version);
00636 
00637         /*
00638          * XXX - we don't care about the section length.
00639          */
00640     }
00641     if (shbp->major_version != PCAP_NG_VERSION_MAJOR) {
00642         snprintf(errbuf, PCAP_ERRBUF_SIZE,
00643             "unknown pcap-ng savefile major version number %u",
00644             shbp->major_version);
00645         goto fail;
00646     }
00647     p->sf.version_major = shbp->major_version;
00648     p->sf.version_minor = shbp->minor_version;
00649 
00650     /*
00651      * Set the default time stamp resolution and offset.
00652      */
00653     p->sf.tsresol = 1000000;    /* microsecond resolution */
00654     p->sf.tsscale = 1;      /* multiply by 1 to scale to microseconds */
00655     p->sf.tsoffset = 0;     /* absolute timestamps */
00656 
00657     /*
00658      * Now start looking for an Interface Description Block.
00659      */
00660     for (;;) {
00661         /*
00662          * Read the next block.
00663          */
00664         status = read_block(fp, p, &cursor, errbuf);
00665         if (status == 0) {
