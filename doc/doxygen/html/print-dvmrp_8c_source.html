00075 void
00076 dvmrp_print(register const u_char *bp, register u_int len)
00077 {
00078     register const u_char *ep;
00079     register u_char type;
00080 
00081     ep = (const u_char *)snapend;
00082     if (bp >= ep)
00083         return;
00084 
00085     TCHECK(bp[1]);
00086     type = bp[1];
00087 
00088     /* Skip IGMP header */
00089     bp += 8;
00090     len -= 8;
00091 
00092     switch (type) {
00093 
00094     case DVMRP_PROBE:
00095         printf(" Probe");
00096         if (vflag) {
00097             if (print_probe(bp, ep, len) < 0)
00098                 goto trunc;
00099         }
00100         break;
00101 
00102     case DVMRP_REPORT:
00103         printf(" Report");
00104         if (vflag > 1) {
00105             if (print_report(bp, ep, len) < 0)
00106                 goto trunc;
00107         }
00108         break;
00109 
00110     case DVMRP_ASK_NEIGHBORS:
00111         printf(" Ask-neighbors(old)");
00112         break;
00113 
00114     case DVMRP_NEIGHBORS:
00115         printf(" Neighbors(old)");
00116         if (print_neighbors(bp, ep, len) < 0)
00117             goto trunc;
00118         break;
00119 
00120     case DVMRP_ASK_NEIGHBORS2:
00121         printf(" Ask-neighbors2");
00122         break;
00123 
00124     case DVMRP_NEIGHBORS2:
00125         printf(" Neighbors2");
00126         /*
00127          * extract version and capabilities from IGMP group
00128          * address field
00129          */
00130         bp -= 4;
00131         TCHECK2(bp[0], 4);
00132         target_level = (bp[0] << 24) | (bp[1] << 16) |
00133             (bp[2] << 8) | bp[3];
00134         bp += 4;
00135         if (print_neighbors2(bp, ep, len) < 0)
00136             goto trunc;
00137         break;
00138 
00139     case DVMRP_PRUNE:
00140         printf(" Prune");
00141         if (print_prune(bp) < 0)
00142             goto trunc;
00143         break;
00144 
00145     case DVMRP_GRAFT:
00146         printf(" Graft");
00147         if (print_graft(bp) < 0)
00148             goto trunc;
00149         break;
00150 
00151     case DVMRP_GRAFT_ACK:
00152         printf(" Graft-ACK");
00153         if (print_graft_ack(bp) < 0)
00154             goto trunc;
00155         break;
00156 
00157     default:
00158         printf(" [type %d]", type);
00159         break;
00160     }
00161     return;
00162 
00163 trunc:
00164     printf("[|dvmrp]");
00165     return;
00166 }

00169 print_report(register const u_char *bp, register const u_char *ep,
00170     register u_int len)
00171 {
00172     register u_int32_t mask, origin;
00173     register int metric, done;
00174     register u_int i, width;
00175 
00176     while (len > 0) {
00177         if (len < 3) {
00178             printf(" [|]");
00179             return (0);
00180         }
00181         TCHECK2(bp[0], 3);
00182         mask = (u_int32_t)0xff << 24 | bp[0] << 16 | bp[1] << 8 | bp[2];
00183         width = 1;
00184         if (bp[0])
00185             width = 2;
00186         if (bp[1])
00187             width = 3;
00188         if (bp[2])
00189             width = 4;
00190 
00191         printf("\n\tMask %s", intoa(htonl(mask)));
00192         bp += 3;
00193         len -= 3;
00194         do {
00195             if (bp + width + 1 > ep) {
00196                 printf(" [|]");
00197                 return (0);
00198             }
00199             if (len < width + 1) {
00200                 printf("\n\t  [Truncated Report]");
00201                 return (0);
00202             }
00203             origin = 0;
00204             for (i = 0; i < width; ++i) {
00205                 TCHECK(*bp);
00206                 origin = origin << 8 | *bp++;
00207             }
00208             for ( ; i < 4; ++i)
00209                 origin <<= 8;
00210 
00211             TCHECK(*bp);
00212             metric = *bp++;
00213             done = metric & 0x80;
00214             metric &= 0x7f;
00215             printf("\n\t  %s metric %d", intoa(htonl(origin)),
00216                 metric);
00217             len -= width + 1;
00218         } while (!done);
00219     }
00220     return (0);
00221 trunc:
00222     return (-1);
00223 }

00226 print_probe(register const u_char *bp, register const u_char *ep,
00227     register u_int len)
00228 {
00229     register u_int32_t genid;
00230 
00231     TCHECK2(bp[0], 4);
00232     if ((len < 4) || ((bp + 4) > ep)) {
00233         /* { (ctags) */
00234         printf(" [|}");
00235         return (0);
00236     }
00237     genid = (bp[0] << 24) | (bp[1] << 16) | (bp[2] << 8) | bp[3];
00238     bp += 4;
00239     len -= 4;
00240     if (vflag > 1)
00241         printf("\n\t");
00242     else
00243         printf(" ");
00244     printf("genid %u", genid);
00245     if (vflag < 2)
00246         return (0);
00247 
00248     while ((len > 0) && (bp < ep)) {
00249         TCHECK2(bp[0], 4);
00250         printf("\n\tneighbor %s", ipaddr_string(bp));
00251         bp += 4; len -= 4;
00252     }
00253     return (0);
