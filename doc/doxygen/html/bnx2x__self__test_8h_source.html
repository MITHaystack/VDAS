00052 static void bnx2x_idle_chk6(struct bnx2x *bp, u32 chip_mask, u32 offset,
00053                 int loop, int inc, int severity)
00054 {
00055     int is_e1  = CHIP_IS_E1(bp);
00056     int is_e1h = CHIP_IS_E1H(bp);
00057     int i;
00058     u32 val1, val2;
00059     u32 rd_ptr, wr_ptr, rd_bank, wr_bank;
00060 
00061     if (!CHIP_MASK_CHK(chip_mask))
00062         return;
00063 
00064     for (i = 0; i < loop; i++) {
00065         val1 = REG_RD(bp, offset + i*inc);
00066         val2 = REG_RD(bp, offset + i*inc + 4);
00067         rd_ptr = ((val1 & 0x3FFFFFC0) >> 6);
00068         wr_ptr = ((((val1 & 0xC0000000) >> 30) & 0x3) |
00069               ((val2 & 0x3FFFFF) << 2));
00070         CONDITION_CHK((rd_ptr != wr_ptr), severity,
00071                 "QM: PTRTBL entry %d - rd_ptr is not"
00072                 " equal to wr_ptr. Values are 0x%x 0x%x\n",
00073                 i, rd_ptr, wr_ptr);
00074         rd_bank = ((val1 & 0x30) >> 4);
00075         wr_bank = (val1 & 0x03);
00076         CONDITION_CHK((rd_bank != wr_bank), severity,
00077                 "QM: PTRTBL entry %d - rd_bank is not"
00078                 " equal to wr_bank. Values are 0x%x 0x%x\n",
00079                 i, rd_bank, wr_bank);
00080     }
00081 }
