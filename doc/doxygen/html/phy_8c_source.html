00640     u16 phy_data;
00641

00644     if (ret_val)

00648     if (phy->type != e1000_phy_bm)
00649         phy_data |= M88E1000_PSCR_ASSERT_CRS_ON_TX;
00650 
00651     /*
00652      * Options:
00653      *   MDI/MDI-X = 0 (default)
00654      *   0 - Auto for all speeds
00655      *   1 - MDI mode
00656      *   2 - MDI-X mode
00657      *   3 - Auto for 1000Base-T only (MDI-X for 10/100Base-T modes)

00660

00664         break;

00667         break;
00668     case 3:

00671     case 0:
00672     default:
00673         phy_data |= M88E1000_PSCR_AUTO_X_MODE;
00674         break;

00676 
00677     /*
00678      * Options:
00679      *   disable_polarity_correction = 0 (default)
00680      *       Automatic Correction for Reversed Cable Polarity
00681      *   0 - Disabled

00684     phy_data &= ~M88E1000_PSCR_POLARITY_REVERSAL;
00685     if (phy->disable_polarity_correction == 1)

00965 static s32 e1000_phy_setup_autoneg(struct e1000_hw *hw)
00966 {
00967     struct e1000_phy_info *phy = &hw->phy;
00968     s32 ret_val;
00969     u16 mii_autoneg_adv_reg;
00970     u16 mii_1000t_ctrl_reg = 0;
00971 
00972     phy->autoneg_advertised &= phy->autoneg_mask;
00973 
00974     /* Read the MII Auto-Neg Advertisement Register (Address 4). */
00975     ret_val = e1e_rphy(hw, PHY_AUTONEG_ADV, &mii_autoneg_adv_reg);
00976     if (ret_val)
00977         goto out;
00978 
00979     if (phy->autoneg_mask & ADVERTISE_1000_FULL) {
00980         /* Read the MII 1000Base-T Control Register (Address 9). */
00981         ret_val = e1e_rphy(hw, PHY_1000T_CTRL,
00982                                     &mii_1000t_ctrl_reg);
00983         if (ret_val)
00984             goto out;
00985     }
00986 
00987     /*
00988      * Need to parse both autoneg_advertised and fc and set up
00989      * the appropriate PHY registers.  First we will parse for
00990      * autoneg_advertised software override.  Since we can advertise
00991      * a plethora of combinations, we need to check each bit
00992      * individually.
00993      */
00994 
00995     /*
00996      * First we clear all the 10/100 mb speed bits in the Auto-Neg
00997      * Advertisement Register (Address 4) and the 1000 mb speed bits in
00998      * the  1000Base-T Control Register (Address 9).
00999      */
01000     mii_autoneg_adv_reg &= ~(NWAY_AR_100TX_FD_CAPS |
01001                              NWAY_AR_100TX_HD_CAPS |
01002                              NWAY_AR_10T_FD_CAPS   |
01003                              NWAY_AR_10T_HD_CAPS);
01004     mii_1000t_ctrl_reg &= ~(CR_1000T_HD_CAPS | CR_1000T_FD_CAPS);
01005 
01006     e_dbg("autoneg_advertised %x\n", phy->autoneg_advertised);
01007 
01008     /* Do we want to advertise 10 Mb Half Duplex? */
01009     if (phy->autoneg_advertised & ADVERTISE_10_HALF) {
01010         e_dbg("Advertise 10mb Half duplex\n");
01011         mii_autoneg_adv_reg |= NWAY_AR_10T_HD_CAPS;
01012     }
01013 
01014     /* Do we want to advertise 10 Mb Full Duplex? */
01015     if (phy->autoneg_advertised & ADVERTISE_10_FULL) {
01016         e_dbg("Advertise 10mb Full duplex\n");
01017         mii_autoneg_adv_reg |= NWAY_AR_10T_FD_CAPS;
01018     }
01019 
01020     /* Do we want to advertise 100 Mb Half Duplex? */
01021     if (phy->autoneg_advertised & ADVERTISE_100_HALF) {
01022         e_dbg("Advertise 100mb Half duplex\n");
01023         mii_autoneg_adv_reg |= NWAY_AR_100TX_HD_CAPS;
01024     }
01025 
01026     /* Do we want to advertise 100 Mb Full Duplex? */
01027     if (phy->autoneg_advertised & ADVERTISE_100_FULL) {
01028         e_dbg("Advertise 100mb Full duplex\n");
01029         mii_autoneg_adv_reg |= NWAY_AR_100TX_FD_CAPS;
01030     }
01031 
01032     /* We do not allow the Phy to advertise 1000 Mb Half Duplex */
01033     if (phy->autoneg_advertised & ADVERTISE_1000_HALF)
01034         e_dbg("Advertise 1000mb Half duplex request denied!\n");
01035 
01036     /* Do we want to advertise 1000 Mb Full Duplex? */
01037     if (phy->autoneg_advertised & ADVERTISE_1000_FULL) {
01038         e_dbg("Advertise 1000mb Full duplex\n");
01039         mii_1000t_ctrl_reg |= CR_1000T_FD_CAPS;
01040     }
01041 
01042     /*
01043      * Check for a software override of the flow control settings, and
01044      * setup the PHY advertisement registers accordingly.  If
01045      * auto-negotiation is enabled, then software will have to set the
01046      * "PAUSE" bits to the correct value in the Auto-Negotiation
01047      * Advertisement Register (PHY_AUTONEG_ADV) and re-start auto-
01048      * negotiation.
01049      *
01050      * The possible values of the "fc" parameter are:
01051      *      0:  Flow control is completely disabled
01052      *      1:  Rx flow control is enabled (we can receive pause frames
01053      *          but not send pause frames).
01054      *      2:  Tx flow control is enabled (we can send pause frames
01055      *          but we do not support receiving pause frames).
01056      *      3:  Both Rx and Tx flow control (symmetric) are enabled.
01057      *  other:  No software override.  The flow control configuration
01058      *          in the EEPROM is used.
01059      */
01060     switch (hw->fc.current_mode) {
01061     case e1000_fc_none:
01062         /*
01063          * Flow control (Rx & Tx) is completely disabled by a
01064          * software over-ride.
01065          */
01066         mii_autoneg_adv_reg &= ~(NWAY_AR_ASM_DIR | NWAY_AR_PAUSE);
01067         break;
01068     case e1000_fc_rx_pause:
01069         /*
01070          * Rx Flow control is enabled, and Tx Flow control is
01071          * disabled, by a software over-ride.
01072          *
01073          * Since there really isn't a way to advertise that we are
01074          * capable of Rx Pause ONLY, we will advertise that we
01075          * support both symmetric and asymmetric Rx PAUSE.  Later
01076          * (in e1000e_config_fc_after_link_up) we will disable the
01077          * hw's ability to send PAUSE frames.
01078          */
01079         mii_autoneg_adv_reg |= (NWAY_AR_ASM_DIR | NWAY_AR_PAUSE);
01080         break;
01081     case e1000_fc_tx_pause:
01082         /*
01083          * Tx Flow control is enabled, and Rx Flow control is
01084          * disabled, by a software over-ride.
01085          */
01086         mii_autoneg_adv_reg |= NWAY_AR_ASM_DIR;
01087         mii_autoneg_adv_reg &= ~NWAY_AR_PAUSE;
01088         break;
01089     case e1000_fc_full:
01090         /*
01091          * Flow control (both Rx and Tx) is enabled by a software
01092          * over-ride.
01093          */
01094         mii_autoneg_adv_reg |= (NWAY_AR_ASM_DIR | NWAY_AR_PAUSE);
01095         break;
01096     default:
01097         e_dbg("Flow control param set incorrectly\n");
01098         ret_val = -E1000_ERR_CONFIG;
01099         goto out;
01100     }
01101 
01102     ret_val = e1e_wphy(hw, PHY_AUTONEG_ADV, mii_autoneg_adv_reg);
01103     if (ret_val)
01104         goto out;
01105 
01106     e_dbg("Auto-Neg Advertising %x\n", mii_autoneg_adv_reg);
01107 
01108     if (phy->autoneg_mask & ADVERTISE_1000_FULL) {
01109         ret_val = e1e_wphy(hw,
01110                                       PHY_1000T_CTRL,
01111                                       mii_1000t_ctrl_reg);
01112         if (ret_val)
01113             goto out;
01114     }
01115 
01116 out:
01117     return ret_val;
01118 }

01440 void e1000e_phy_force_speed_duplex_setup(struct e1000_hw *hw, u16 *phy_ctrl)
01441 {
01442     struct e1000_mac_info *mac = &hw->mac;
01443     u32 ctrl;
01444 
01445     /* Turn off flow control when forcing speed/duplex */
01446     hw->fc.current_mode = e1000_fc_none;
01447 
01448     /* Force speed/duplex on the mac */
01449     ctrl = er32(CTRL);
01450     ctrl |= (E1000_CTRL_FRCSPD | E1000_CTRL_FRCDPX);
01451     ctrl &= ~E1000_CTRL_SPD_SEL;
01452 
01453     /* Disable Auto Speed Detection */
01454     ctrl &= ~E1000_CTRL_ASDE;
01455 
01456     /* Disable autoneg on the phy */
01457     *phy_ctrl &= ~MII_CR_AUTO_NEG_EN;
01458 
01459     /* Forcing Full or Half Duplex? */
01460     if (mac->forced_speed_duplex & E1000_ALL_HALF_DUPLEX) {
01461         ctrl &= ~E1000_CTRL_FD;
01462         *phy_ctrl &= ~MII_CR_FULL_DUPLEX;
01463         e_dbg("Half Duplex\n");
01464     } else {
01465         ctrl |= E1000_CTRL_FD;
01466         *phy_ctrl |= MII_CR_FULL_DUPLEX;
01467         e_dbg("Full Duplex\n");
01468     }
01469 
01470     /* Forcing 10mb or 100mb? */
01471     if (mac->forced_speed_duplex & E1000_ALL_100_SPEED) {
01472         ctrl |= E1000_CTRL_SPD_100;
01473         *phy_ctrl |= MII_CR_SPEED_100;
01474         *phy_ctrl &= ~(MII_CR_SPEED_1000 | MII_CR_SPEED_10);
01475         e_dbg("Forcing 100mb\n");
01476     } else {

01770         /*

01855     u16 phy_data, i, agc_value = 0;
01856     u16 cur_agc_index, max_agc_index = 0;
01857     u16 min_agc_index = IGP02E1000_CABLE_LENGTH_TABLE_SIZE - 1;

01859            IGP02E1000_PHY_AGC_A,

01861            IGP02E1000_PHY_AGC_C,
01862            IGP02E1000_PHY_AGC_D
01863     };
01864

01866     for (i = 0; i < IGP02E1000_PHY_CHANNEL_NUM; i++) {

01868         if (ret_val)

01872          * Getting bits 15:9, which represent the combination of
01873          * coarse and fine gain values.  The result is a number
01874          * that can be put into the lookup table to obtain the
01875          * approximate cable length.
01876          */

01878                         IGP02E1000_AGC_LENGTH_MASK;

01882             (cur_agc_index == 0)) {

02613 static s32 e1000_access_phy_wakeup_reg_bm(struct e1000_hw *hw, u32 offset,
02614                                           u16 *data, bool read)
02615 {
02616     s32 ret_val;
02617     u16 reg = BM_PHY_REG_NUM(offset);
02618     u16 phy_reg = 0;
02619 
02620     /* Gig must be disabled for MDIO accesses to page 800 */
02621     if ((hw->mac.type == e1000_pchlan) &&
02622        (!(er32(PHY_CTRL) & E1000_PHY_CTRL_GBE_DISABLE)))
02623         e_dbg("Attempting to access page 800 while gig enabled.\n");
02624 
02625     /* All operations in this function are phy address 1 */
02626     hw->phy.addr = 1;
02627 
02628     /* Set page 769 */
02629     e1000e_write_phy_reg_mdic(hw, IGP01E1000_PHY_PAGE_SELECT,
02630                              (BM_WUC_ENABLE_PAGE << IGP_PAGE_SHIFT));
02631 
02632     ret_val = e1000e_read_phy_reg_mdic(hw, BM_WUC_ENABLE_REG, &phy_reg);
02633     if (ret_val) {
02634         e_dbg("Could not read PHY page 769\n");
02635         goto out;
02636     }
02637 
02638     /* First clear bit 4 to avoid a power state change */
02639     phy_reg &= ~(BM_WUC_HOST_WU_BIT);
02640     ret_val = e1000e_write_phy_reg_mdic(hw, BM_WUC_ENABLE_REG, phy_reg);
02641     if (ret_val) {
02642         e_dbg("Could not clear PHY page 769 bit 4\n");
02643         goto out;
02644     }
02645 
02646     /* Write bit 2 = 1, and clear bit 4 to 769_17 */
02647     ret_val = e1000e_write_phy_reg_mdic(hw, BM_WUC_ENABLE_REG,
02648                                        phy_reg | BM_WUC_ENABLE_BIT);
02649     if (ret_val) {
02650         e_dbg("Could not write PHY page 769 bit 2\n");
02651         goto out;
02652     }
02653 
02654     /* Select page 800 */
02655     ret_val = e1000e_write_phy_reg_mdic(hw, IGP01E1000_PHY_PAGE_SELECT,
02656                                        (BM_WUC_PAGE << IGP_PAGE_SHIFT));
02657 
02658     /* Write the page 800 offset value using opcode 0x11 */
02659     ret_val = e1000e_write_phy_reg_mdic(hw, BM_WUC_ADDRESS_OPCODE, reg);
02660     if (ret_val) {
02661         e_dbg("Could not write address opcode to page 800\n");
02662         goto out;
02663     }
02664 
02665     if (read) {
02666             /* Read the page 800 value using opcode 0x12 */
02667         ret_val = e1000e_read_phy_reg_mdic(hw, BM_WUC_DATA_OPCODE,
02668                                           data);
02669     } else {
02670             /* Write the page 800 value using opcode 0x12 */
02671         ret_val = e1000e_write_phy_reg_mdic(hw, BM_WUC_DATA_OPCODE,
02672                                            *data);
02673     }
02674 
02675     if (ret_val) {
02676         e_dbg("Could not access data value from page 800\n");
02677         goto out;
02678     }
02679 
02680     /*
02681      * Restore 769_17.2 to its original value
02682      * Set page 769
02683      */
02684     e1000e_write_phy_reg_mdic(hw, IGP01E1000_PHY_PAGE_SELECT,
02685                              (BM_WUC_ENABLE_PAGE << IGP_PAGE_SHIFT));
02686 
02687     /* Clear 769_17.2 */
02688     ret_val = e1000e_write_phy_reg_mdic(hw, BM_WUC_ENABLE_REG, phy_reg);
02689     if (ret_val) {
02690         e_dbg("Could not clear PHY page 769 bit 2\n");
02691         goto out;
02692     }
02693 
02694 out:
02695     return ret_val;
02696 }
