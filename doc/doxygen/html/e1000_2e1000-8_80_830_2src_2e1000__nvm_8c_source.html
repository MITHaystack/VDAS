00136 static void e1000_shift_out_eec_bits(struct e1000_hw *hw, u16 data, u16 count)
00137 {
00138     struct e1000_nvm_info *nvm = &hw->nvm;
00139     u32 eecd = E1000_READ_REG(hw, E1000_EECD);
00140     u32 mask;
00141 
00142     DEBUGFUNC("e1000_shift_out_eec_bits");
00143 
00144     mask = 0x01 << (count - 1);
00145     if (nvm->type == e1000_nvm_eeprom_microwire)
00146         eecd &= ~E1000_EECD_DO;
00147     else
00148     if (nvm->type == e1000_nvm_eeprom_spi)
00149         eecd |= E1000_EECD_DO;
00150 
00151     do {
00152         eecd &= ~E1000_EECD_DI;
00153 
00154         if (data & mask)
00155             eecd |= E1000_EECD_DI;
00156 
00157         E1000_WRITE_REG(hw, E1000_EECD, eecd);
00158         E1000_WRITE_FLUSH(hw);
00159 
00160         usec_delay(nvm->delay_usec);
00161 
00162         e1000_raise_eec_clk(hw, &eecd);
00163         e1000_lower_eec_clk(hw, &eecd);
00164 
00165         mask >>= 1;
00166     } while (mask);
00167 
00168     eecd &= ~E1000_EECD_DI;
00169     E1000_WRITE_REG(hw, E1000_EECD, eecd);
00170 }

00289 {
00290     struct e1000_nvm_info *nvm = &hw->nvm;
00291     u32 eecd = E1000_READ_REG(hw, E1000_EECD);

00296         eecd &= ~(E1000_EECD_CS | E1000_EECD_SK);
00297         E1000_WRITE_REG(hw, E1000_EECD, eecd);
00298         E1000_WRITE_FLUSH(hw);

00300 
00301         e1000_raise_eec_clk(hw, &eecd);

00303         /* Select EEPROM */
00304         eecd |= E1000_EECD_CS;
00305         E1000_WRITE_REG(hw, E1000_EECD, eecd);
00306         E1000_WRITE_FLUSH(hw);

00375 static s32 e1000_ready_nvm_eeprom(struct e1000_hw *hw)
00376 {
00377     struct e1000_nvm_info *nvm = &hw->nvm;
00378     u32 eecd = E1000_READ_REG(hw, E1000_EECD);
00379     s32 ret_val = E1000_SUCCESS;
00380     u8 spi_stat_reg;
00381 
00382     DEBUGFUNC("e1000_ready_nvm_eeprom");
00383 
00384     if (nvm->type == e1000_nvm_eeprom_microwire) {
00385         /* Clear SK and DI */
00386         eecd &= ~(E1000_EECD_DI | E1000_EECD_SK);
00387         E1000_WRITE_REG(hw, E1000_EECD, eecd);
00388         /* Set CS */
00389         eecd |= E1000_EECD_CS;
00390         E1000_WRITE_REG(hw, E1000_EECD, eecd);
00391     } else
00392     if (nvm->type == e1000_nvm_eeprom_spi) {
00393         u16 timeout = NVM_MAX_RETRY_SPI;
00394 
00395         /* Clear SK and CS */
00396         eecd &= ~(E1000_EECD_CS | E1000_EECD_SK);
00397         E1000_WRITE_REG(hw, E1000_EECD, eecd);
00398         usec_delay(1);
00399 
00400         /*
00401          * Read "Status Register" repeatedly until the LSB is cleared.
00402          * The EEPROM will signal that the command has been completed
00403          * by clearing bit 0 of the internal status register.  If it's
00404          * not cleared within 'timeout', then error out.
00405          */
00406         while (timeout) {
00407             e1000_shift_out_eec_bits(hw, NVM_RDSR_OPCODE_SPI,
00408                                      hw->nvm.opcode_bits);
00409             spi_stat_reg = (u8)e1000_shift_in_eec_bits(hw, 8);
00410             if (!(spi_stat_reg & NVM_STATUS_RDY_SPI))
00411                 break;
00412 
00413             usec_delay(5);
00414             e1000_standby_nvm(hw);
00415             timeout--;
00416         }
00417 
00418         if (!timeout) {
00419             DEBUGOUT("SPI NVM Status error\n");
00420             ret_val = -E1000_ERR_NVM;
00421             goto out;
00422         }
00423     }
00424 
00425 out:
00426     return ret_val;
00427 }

00438 s32 e1000_read_nvm_spi(struct e1000_hw *hw, u16 offset, u16 words, u16 *data)
00439 {
00440     struct e1000_nvm_info *nvm = &hw->nvm;
00441     u32 i = 0;
00442     s32 ret_val;
00443     u16 word_in;
00444     u8 read_opcode = NVM_READ_OPCODE_SPI;
00445 
00446     DEBUGFUNC("e1000_read_nvm_spi");
00447 
00448     /*
00449      * A check for invalid values:  offset too large, too many words,
00450      * and not enough words.
00451      */
00452     if ((offset >= nvm->word_size) || (words > (nvm->word_size - offset)) ||
00453         (words == 0)) {
00454         DEBUGOUT("nvm parameter(s) out of bounds\n");
00455         ret_val = -E1000_ERR_NVM;
00456         goto out;
00457     }
00458 
00459     ret_val = nvm->ops.acquire(hw);
00460     if (ret_val)
00461         goto out;
00462 
00463     ret_val = e1000_ready_nvm_eeprom(hw);
00464     if (ret_val)
00465         goto release;
00466 
00467     e1000_standby_nvm(hw);
00468 
00469     if ((nvm->address_bits == 8) && (offset >= 128))
00470         read_opcode |= NVM_A8_OPCODE_SPI;
00471 
00472     /* Send the READ command (opcode + addr) */
00473     e1000_shift_out_eec_bits(hw, read_opcode, nvm->opcode_bits);
00474     e1000_shift_out_eec_bits(hw, (u16)(offset*2), nvm->address_bits);
00475 
00476     /*
00477      * Read the data.  SPI NVMs increment the address with each byte
00478      * read and will roll over if reading beyond the end.  This allows
00479      * us to read the whole NVM from any offset
00480      */
00481     for (i = 0; i < words; i++) {
00482         word_in = e1000_shift_in_eec_bits(hw, 16);
00483         data[i] = (word_in >> 8) | (word_in << 8);
00484     }
00485 
00486 release:
00487     nvm->ops.release(hw);
00488 
00489 out:
00490     return ret_val;
00491 }

00667                 e1000_standby_nvm(hw);

00670         }
00671     }
00672 
00673     msec_delay(10);
00674 release:
00675     nvm->ops.release(hw);
00676 
00677 out:
00678     return ret_val;
