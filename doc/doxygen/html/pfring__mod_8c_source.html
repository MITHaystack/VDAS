00274 int pfring_mod_bind(pfring *ring, char *device_name) {
00275   struct sockaddr sa;
00276   char *at;
00277   u_int32_t channel_id = RING_ANY_CHANNEL;
00278   int rc = 0;
00279 
00280   if((device_name == NULL) || (strcmp(device_name, "none") == 0))
00281     return(-1);
00282 
00283   at = strchr(device_name, '@');
00284   if(at != NULL) {
00285     char *tok, *pos = NULL;
00286 
00287     at[0] = '\0';
00288 
00289     /* Syntax
00290        ethX@1,5       channel 1 and 5
00291        ethX@1-5       channel 1,2...5
00292        ethX@1-3,5-7   channel 1,2,3,5,6,7
00293     */
00294 
00295     tok = strtok_r(&at[1], ",", &pos);
00296     channel_id = 0;
00297 
00298     while(tok != NULL) {
00299       char *dash = strchr(tok, '-');
00300       int32_t min_val, max_val, i;
00301 
00302       if(dash) {
00303     dash[0] = '\0';
00304     min_val = atoi(tok);
00305     max_val = atoi(&dash[1]);
00306 
00307       } else
00308     min_val = max_val = atoi(tok);
00309 
00310       for(i = min_val; i <= max_val; i++)
00311     channel_id |= 1 << i;
00312 
00313       tok = strtok_r(NULL, ",", &pos);
00314     }
00315   }
00316 
00317   /* Setup TX */
00318   ring->sock_tx.sll_family = PF_PACKET;
00319   ring->sock_tx.sll_protocol = htons(ETH_P_ALL);
00320 
00321   sa.sa_family = PF_RING;
00322   snprintf(sa.sa_data, sizeof(sa.sa_data), "%s", device_name);
00323 
00324   rc = bind(ring->fd, (struct sockaddr *)&sa, sizeof(sa));
00325 
00326   if(rc == 0) {
00327     rc = pfring_set_channel_id(ring, channel_id);
00328 
00329     if(rc != 0)
00330       printf("pfring_set_channel_id() failed: %d\n", rc);
00331   }
00332 
00333   return(rc);
00334 }

00400 int pfring_mod_recv(pfring *ring, u_char** buffer, u_int buffer_len,
00401             struct pfring_pkthdr *hdr,
00402             u_int8_t wait_for_incoming_packet) {
00403 
00404   if(ring->is_shutting_down) return(-1);
00405 
00406   int rc = 0;
00407 
00408   if(ring->buffer == NULL) return(-1);
00409 
00410   ring->break_recv_loop = 0;
00411 
00412   do_pfring_recv:
00413     if(ring->break_recv_loop)
00414       return(0);
00415 
00416     if(ring->reentrant)
00417       pthread_spin_lock(&ring->spinlock);
00418 
00419     //rmb();
00420 
00421     if(pfring_there_is_pkt_available(ring)) {
00422       char *bucket = &ring->slots[ring->slots_info->remove_off];
00423       u_int32_t next_off, real_slot_len, insert_off, bktLen;
00424 
00425       memcpy(hdr, bucket, ring->slot_header_len);
00426 
00427       if(ring->slot_header_len != sizeof(struct pfring_pkthdr))
00428     bktLen = hdr->caplen;
00429       else
00430     bktLen = hdr->caplen+hdr->extended_hdr.parsed_header_len;
00431 
00432       real_slot_len = ring->slot_header_len + bktLen;
00433       insert_off = ring->slots_info->insert_off;
00434       if(bktLen > buffer_len) bktLen = buffer_len;
00435 
00436       if(buffer_len == 0)
00437     *buffer = (u_char*)&bucket[ring->slot_header_len];
00438       else
00439     memcpy(*buffer, &bucket[ring->slot_header_len], bktLen);            
00440 
00441       next_off = ring->slots_info->remove_off + real_slot_len;
00442       if((next_off + ring->slots_info->slot_len) > (ring->slots_info->tot_mem - sizeof(FlowSlotInfo))) {
00443         next_off = 0;
00444       }
00445 
00446 #ifdef USE_MB
00447       /* This prevents the compiler from reordering instructions.
00448        * http://en.wikipedia.org/wiki/Memory_ordering#Compiler_memory_barrier */
00449       gcc_mb();
00450 #endif
00451 
00452       ring->slots_info->tot_read++;
00453       ring->slots_info->remove_off = next_off;
00454 
00455       /* Ugly safety check */
00456       if((ring->slots_info->tot_insert == ring->slots_info->tot_read)
00457      && (ring->slots_info->remove_off > ring->slots_info->insert_off)) {
00458     ring->slots_info->remove_off = ring->slots_info->insert_off;
00459       }
00460 
00461       if(ring->reentrant) pthread_spin_unlock(&ring->spinlock);
00462       return(1);
00463     }
00464 
00465     /* Nothing to do: we need to wait */
00466     if(ring->reentrant) pthread_spin_unlock(&ring->spinlock);
00467 
00468     if(wait_for_incoming_packet) {
00469       rc = pfring_poll(ring, ring->poll_duration);
00470 
00471       if((rc == -1) && (errno != EINTR))
00472     return(-1);
00473       else
00474     goto do_pfring_recv;
00475     }
00476 
00477   return(-1);
00478 }
