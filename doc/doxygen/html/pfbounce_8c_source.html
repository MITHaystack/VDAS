00054 /*
00055  * The time difference in millisecond
00056  */
00057 double delta_time (struct timeval * now,
00058            struct timeval * before) {
00059   time_t delta_seconds;
00060   time_t delta_microseconds;
00061 
00062   /*
00063    * compute delta in second, 1/10's and 1/1000's second units
00064    */
00065   delta_seconds      = now -> tv_sec  - before -> tv_sec;
00066   delta_microseconds = now -> tv_usec - before -> tv_usec;
00067 
00068   if(delta_microseconds < 0) {
00069     /* manually carry a one from the seconds field */
00070     delta_microseconds += 1000000;  /* 1e6 */
00071     -- delta_seconds;
00072   }
00073   return((double)(delta_seconds * 1000) + (double)delta_microseconds/1000);
00074 }

00107   filtering_rule rule;

00109   /* Bounce all packets received on in_dev -> in_dev */
00110   memset(&rule, 0, sizeof(rule));
00111   rule.rule_id = 1;
00112   rule.rule_action = bounce_packet_and_stop_rule_evaluation;
00113   rule.core_fields.proto = 0; /* any */
00114   snprintf(rule.reflector_device_name, REFLECTOR_NAME_LEN, "%s", in_dev);

00116   if(pfring_add_filtering_rule(pd, &rule) < 0) {
00117     printf("pfring_add_filtering_rule() failed\n");
