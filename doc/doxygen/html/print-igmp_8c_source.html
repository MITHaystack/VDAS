00049 struct tr_query {
00050     u_int32_t  tr_src;          /* traceroute source */

00052     u_int32_t  tr_raddr;        /* traceroute response address */

00059 /*
00060  * Traceroute response format.  A traceroute response has a tr_query at the
00061  * beginning, followed by one tr_resp for each hop taken.
00062  */
00063 struct tr_resp {
00064     u_int32_t tr_qarr;          /* query arrival time */
00065     u_int32_t tr_inaddr;        /* incoming interface address */
00066     u_int32_t tr_outaddr;       /* outgoing interface address */
00067     u_int32_t tr_rmtaddr;       /* parent address in source tree */
00068     u_int32_t tr_vifin;         /* input packet count on interface */
00069     u_int32_t tr_vifout;        /* output packet count on interface */
00070     u_int32_t tr_pktcnt;        /* total incoming packets for src-grp */
00071     u_int8_t  tr_rproto;      /* routing proto deployed on router */
00072     u_int8_t  tr_fttl;        /* ttl required to forward on outvif */
00073     u_int8_t  tr_smask;       /* subnet mask for src addr */
00074     u_int8_t  tr_rflags;      /* forwarding error codes */
00075 };

00153 static void
00154 print_igmpv3_report(register const u_char *bp, register u_int len)
00155 {
00156     u_int group, nsrcs, ngroups;
00157     register u_int i, j;
00158 
00159     /* Minimum len is 16, and should be a multiple of 4 */
00160     if (len < 16 || len & 0x03) {
00161     (void)printf(" [invalid len %d]", len);
00162     return;
00163     }
00164     TCHECK2(bp[6], 2);
00165     ngroups = EXTRACT_16BITS(&bp[6]);
00166     (void)printf(", %d group record(s)", ngroups);
00167     if (vflag > 0) {
00168     /* Print the group records */
00169     group = 8;
00170         for (i=0; i<ngroups; i++) {
00171         if (len < group+8) {
00172         (void)printf(" [invalid number of groups]");
00173         return;
00174         }
00175         TCHECK2(bp[group+4], 4);
00176             (void)printf(" [gaddr %s", ipaddr_string(&bp[group+4]));
00177         (void)printf(" %s", tok2str(igmpv3report2str, " [v3-report-#%d]",
00178                                 bp[group]));
00179             nsrcs = EXTRACT_16BITS(&bp[group+2]);
00180         /* Check the number of sources and print them */
00181         if (len < group+8+(nsrcs<<2)) {
00182         (void)printf(" [invalid number of sources %d]", nsrcs);
00183         return;
00184         }
00185             if (vflag == 1)
00186                 (void)printf(", %d source(s)", nsrcs);
00187             else {
00188         /* Print the sources */
00189                 (void)printf(" {");
00190                 for (j=0; j<nsrcs; j++) {
00191             TCHECK2(bp[group+8+(j<<2)], 4);
00192             (void)printf(" %s", ipaddr_string(&bp[group+8+(j<<2)]));
00193         }
00194                 (void)printf(" }");
00195             }
00196         /* Next group record */
00197             group += 8 + (nsrcs << 2);
00198         (void)printf("]");
00199         }
00200     }
00201     return;
00202 trunc:
00203     (void)printf("[|igmp]");
00204     return;
00205 }

00207 static void
00208 print_igmpv3_query(register const u_char *bp, register u_int len)
00209 {
00210     u_int mrc;
00211     int mrt;
00212     u_int nsrcs;
00213     register u_int i;
00214 
00215     (void)printf(" v3");
00216     /* Minimum len is 12, and should be a multiple of 4 */
00217     if (len < 12 || len & 0x03) {
00218     (void)printf(" [invalid len %d]", len);
00219     return;
00220     }
00221     TCHECK(bp[1]);
00222     mrc = bp[1];
00223     if (mrc < 128) {
00224     mrt = mrc;
00225     } else {
00226         mrt = ((mrc & 0x0f) | 0x10) << (((mrc & 0x70) >> 4) + 3);
00227     }
00228     if (mrc != 100) {
00229     (void)printf(" [max resp time ");
00230     relts_print(mrt);
00231     (void)printf("]");
00232     }
00233     TCHECK2(bp[4], 4);
00234     if (EXTRACT_32BITS(&bp[4]) == 0)
00235     return;
00236     (void)printf(" [gaddr %s", ipaddr_string(&bp[4]));
00237     TCHECK2(bp[10], 2);
00238     nsrcs = EXTRACT_16BITS(&bp[10]);
00239     if (nsrcs > 0) {
00240     if (len < 12 + (nsrcs << 2))
00241         (void)printf(" [invalid number of sources]");
00242     else if (vflag > 1) {
00243         (void)printf(" {");
00244         for (i=0; i<nsrcs; i++) {
00245         TCHECK2(bp[12+(i<<2)], 4);
00246         (void)printf(" %s", ipaddr_string(&bp[12+(i<<2)]));
00247         }
00248         (void)printf(" }");
00249     } else
00250         (void)printf(", %d source(s)", nsrcs);
00251     }
00252     (void)printf("]");
00253     return;
00254 trunc:
00255     (void)printf("[|igmp]");
00256     return;

00259 void
00260 igmp_print(register const u_char *bp, register u_int len)
00261 {
00262     if (qflag) {
00263         (void)printf("igmp");
00264         return;
00265     }
00266 
00267     TCHECK(bp[0]);
00268     switch (bp[0]) {
00269     case 0x11:
00270         (void)printf("igmp query");
00271     if (len >= 12)
00272         print_igmpv3_query(bp, len);
00273     else {
00274             TCHECK(bp[1]);
00275         if (bp[1]) {
00276         (void)printf(" v2");
00277         if (bp[1] != 100)
00278             (void)printf(" [max resp time %d]", bp[1]);
00279         } else
00280         (void)printf(" v1");
00281             TCHECK2(bp[4], 4);
00282         if (EXTRACT_32BITS(&bp[4]))
00283                 (void)printf(" [gaddr %s]", ipaddr_string(&bp[4]));
00284             if (len != 8)
00285                 (void)printf(" [len %d]", len);
00286     }
00287         break;
00288     case 0x12:
00289         TCHECK2(bp[4], 4);
00290         (void)printf("igmp v1 report %s", ipaddr_string(&bp[4]));
00291         if (len != 8)
00292             (void)printf(" [len %d]", len);
00293         break;
00294     case 0x16:
00295         TCHECK2(bp[4], 4);
00296         (void)printf("igmp v2 report %s", ipaddr_string(&bp[4]));
00297         break;
00298     case 0x22:
00299         (void)printf("igmp v3 report");
00300     print_igmpv3_report(bp, len);
00301         break;
00302     case 0x17:
00303         TCHECK2(bp[4], 4);
00304         (void)printf("igmp leave %s", ipaddr_string(&bp[4]));
00305         break;
00306     case 0x13:
00307         (void)printf("igmp dvmrp");
00308         if (len < 8)
00309             (void)printf(" [len %d]", len);
00310         else
00311             dvmrp_print(bp, len);
00312         break;
00313     case 0x14:
00314         (void)printf("igmp pimv1");
00315         pimv1_print(bp, len);
00316         break;
00317     case 0x1e:
00318         print_mresp(bp, len);
00319         break;
00320     case 0x1f:
00321         print_mtrace(bp, len);
00322         break;
00323     default:
00324         (void)printf("igmp-%d", bp[0]);
00325         break;
00326     }
00327 
00328     if (vflag && TTEST2(bp[0], len)) {
00329         /* Check the IGMP checksum */
00330         if (in_cksum((const u_short*)bp, len, 0))
00331             printf(" bad igmp cksum %x!", EXTRACT_16BITS(&bp[2]));
00332     }
00333     return;
00334 trunc:
00335     fputs("[|igmp]", stdout);
00336 }
