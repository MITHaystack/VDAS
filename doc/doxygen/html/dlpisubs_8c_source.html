00076 /*
00077  * Get the packet statistics.
00078  */
00079 int
00080 pcap_stats_dlpi(pcap_t *p, struct pcap_stat *ps)
00081 {
00082 
00083     /*
00084      * "ps_recv" counts packets handed to the filter, not packets
00085      * that passed the filter.  As filtering is done in userland,
00086      * this would not include packets dropped because we ran out
00087      * of buffer space; in order to make this more like other
00088      * platforms (Linux 2.4 and later, BSDs with BPF), where the
00089      * "packets received" count includes packets received but dropped
00090      * due to running out of buffer space, and to keep from confusing
00091      * applications that, for example, compute packet drop percentages,
00092      * we also make it count packets dropped by "bufmod" (otherwise we
00093      * might run the risk of the packet drop count being bigger than
00094      * the received-packet count).
00095      *
00096      * "ps_drop" counts packets dropped by "bufmod" because of
00097      * flow control requirements or resource exhaustion; it doesn't
00098      * count packets dropped by the interface driver, or packets
00099      * dropped upstream.  As filtering is done in userland, it counts
00100      * packets regardless of whether they would've passed the filter.
00101      *
00102      * These statistics don't include packets not yet read from
00103      * the kernel by libpcap, but they may include packets not
00104      * yet read from libpcap by the application.
00105      */
00106     *ps = p->md.stat;
00107 
00108     /*
00109      * Add in the drop count, as per the above comment.
00110      */
00111     ps->ps_recv += ps->ps_drop;
00112     return (0);
00113 }

00116  * Loop through the packets and call the callback for each packet.
00117  * Return the number of packets read.
00118  */
00119 int
00120 pcap_process_pkts(pcap_t *p, pcap_handler callback, u_char *user,
00121     int count, u_char *bufp, int len)
00122 {
00123     int n, caplen, origlen;
00124     u_char *ep, *pk;
00125     struct pcap_pkthdr pkthdr;
00126 #ifdef HAVE_SYS_BUFMOD_H
00127     struct sb_hdr *sbp;
00128 #ifdef LBL_ALIGN
00129     struct sb_hdr sbhdr;
00130 #endif
00131 #endif
00132 
00133     /* Loop through packets */
00134     ep = bufp + len;
00135     n = 0;
00136 
00137 #ifdef HAVE_SYS_BUFMOD_H
00138     while (bufp < ep) {
00139         /*
00140          * Has "pcap_breakloop()" been called?
00141          * If so, return immediately - if we haven't read any
00142          * packets, clear the flag and return -2 to indicate
00143          * that we were told to break out of the loop, otherwise
00144          * leave the flag set, so that the *next* call will break
00145          * out of the loop without having read any packets, and
00146          * return the number of packets we've processed so far.
00147          */
00148         if (p->break_loop) {
00149             if (n == 0) {
00150                 p->break_loop = 0;
00151                 return (-2);
00152             } else {
00153                 p->bp = bufp;
00154                 p->cc = ep - bufp;
00155                 return (n);
00156             }
00157         }
00158 #ifdef LBL_ALIGN
00159         if ((long)bufp & 3) {
00160             sbp = &sbhdr;
00161             memcpy(sbp, bufp, sizeof(*sbp));
00162         } else
00163 #endif
00164             sbp = (struct sb_hdr *)bufp;
00165         p->md.stat.ps_drop = sbp->sbh_drops;
00166         pk = bufp + sizeof(*sbp);
00167         bufp += sbp->sbh_totlen;
00168         origlen = sbp->sbh_origlen;
00169         caplen = sbp->sbh_msglen;
00170 #else
00171         origlen = len;
00172         caplen = min(p->snapshot, len);
00173         pk = bufp;
00174         bufp += caplen;
00175 #endif
00176         ++p->md.stat.ps_recv;
00177         if (bpf_filter(p->fcode.bf_insns, pk, origlen, caplen)) {
00178 #ifdef HAVE_SYS_BUFMOD_H
00179             pkthdr.ts.tv_sec = sbp->sbh_timestamp.tv_sec;
00180             pkthdr.ts.tv_usec = sbp->sbh_timestamp.tv_usec;
00181 #else
00182             (void) gettimeofday(&pkthdr.ts, NULL);
00183 #endif
00184             pkthdr.len = origlen;
00185             pkthdr.caplen = caplen;
00186             /* Insure caplen does not exceed snapshot */
00187             if (pkthdr.caplen > p->snapshot)
00188                 pkthdr.caplen = p->snapshot;
00189             (*callback)(user, &pkthdr, pk);
00190             if (++n >= count && count >= 0) {
00191                 p->cc = ep - bufp;
00192                 p->bp = bufp;
00193                 return (n);
00194             }
00195         }
00196 #ifdef HAVE_SYS_BUFMOD_H
00197     }
00198 #endif
00199     p->cc = 0;
00200     return (n);
00201 }

00203 /*
00204  * Process the mac type. Returns -1 if no matching mac type found, otherwise 0.
00205  */
00206 int
00207 pcap_process_mactype(pcap_t *p, u_int mactype)
00208 {
00209     int retv = 0;
00210 
00211     switch (mactype) {
00212 
00213     case DL_CSMACD:
00214     case DL_ETHER:
00215         p->linktype = DLT_EN10MB;
00216         p->offset = 2;
00217         /*
00218          * This is (presumably) a real Ethernet capture; give it a
00219          * link-layer-type list with DLT_EN10MB and DLT_DOCSIS, so
00220          * that an application can let you choose it, in case you're
00221          * capturing DOCSIS traffic that a Cisco Cable Modem
00222          * Termination System is putting out onto an Ethernet (it
00223          * doesn't put an Ethernet header onto the wire, it puts raw
00224          * DOCSIS frames out on the wire inside the low-level
00225          * Ethernet framing).
00226          */
00227         p->dlt_list = (u_int *)malloc(sizeof(u_int) * 2);
00228         /*
00229          * If that fails, just leave the list empty.
00230          */
00231         if (p->dlt_list != NULL) {
00232             p->dlt_list[0] = DLT_EN10MB;
00233             p->dlt_list[1] = DLT_DOCSIS;
00234             p->dlt_count = 2;
00235         }
00236         break;
00237 
00238     case DL_FDDI:
00239         p->linktype = DLT_FDDI;
00240         p->offset = 3;
00241         break;
00242 
00243     case DL_TPR:
00244         /* XXX - what about DL_TPB?  Is that Token Bus?  */
00245         p->linktype = DLT_IEEE802;
00246         p->offset = 2;
00247         break;
00248 
00249 #ifdef HAVE_SOLARIS
00250     case DL_IPATM:
00251         p->linktype = DLT_SUNATM;
00252         p->offset = 0;  /* works for LANE and LLC encapsulation */
00253         break;
00254 #endif
00255 
00256     default:
00257         snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "unknown mactype %u",
00258             mactype);
00259         retv = -1;
00260     }
00261 
00262     return (retv);
00263 }

00269 int
00270 pcap_conf_bufmod(pcap_t *p, int snaplen, int timeout)
00271 {
00272     int retv = 0;
00273 
00274     bpf_u_int32 ss, chunksize;
00275 
00276     /* Non-standard call to get the data nicely buffered. */
00277     if (ioctl(p->fd, I_PUSH, "bufmod") != 0) {
00278         pcap_stream_err("I_PUSH bufmod", errno, p->errbuf);
00279         retv = -1;
00280     }
00281 
00282     ss = snaplen;
00283     if (ss > 0 &&
00284         strioctl(p->fd, SBIOCSSNAP, sizeof(ss), (char *)&ss) != 0) {
00285         pcap_stream_err("SBIOCSSNAP", errno, p->errbuf);
00286         retv = -1;
00287     }
00288 
00289     /* Set up the bufmod timeout. */
00290     if (timeout != 0) {
00291         struct timeval to;
00292 
00293         to.tv_sec = timeout / 1000;
00294         to.tv_usec = (timeout * 1000) % 1000000;
00295         if (strioctl(p->fd, SBIOCSTIME, sizeof(to), (char *)&to) != 0) {
00296             pcap_stream_err("SBIOCSTIME", errno, p->errbuf);
00297             retv = -1;
00298         }
00299     }
00300 
00301     /* Set the chunk length. */
