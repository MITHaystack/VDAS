00092 {
00093     int length;
00094     int len;
00095     const u_char *cp;
00096     u_int32_t retval;
00097 
00098     cp = (const u_char *)(ip + 1);
00099     length = (IP_HL(ip) << 2) - sizeof(struct ip);
00100 
00101     for (; length > 0; cp += len, length -= len) {
00102         int tt;
00103 
00104         TCHECK(*cp);
00105         tt = *cp;
00106         if (tt == IPOPT_EOL)
00107             break;
00108         else if (tt == IPOPT_NOP)
00109             len = 1;
00110         else {
00111             TCHECK(cp[1]);
00112             len = cp[1];
00113             if (len < 2)
00114                 break;

00116         TCHECK2(*cp, len);
00117         switch (tt) {
00118 
00119         case IPOPT_SSRR:
00120         case IPOPT_LSRR:
00121             if (len < 7)
00122                 break;
00123             memcpy(&retval, cp + len - 4, 4);
00124             return retval;

00271 /*
00272  * compute an IP header checksum.
00273  * don't modifiy the packet.
00274  */
00275 u_short
00276 in_cksum(const u_short *addr, register u_int len, int csum)
00277 {
00278     int nleft = len;
00279     const u_short *w = addr;
00280     u_short answer;
00281     int sum = csum;
00282 
00283     /*
00284      *  Our algorithm is simple, using a 32 bit accumulator (sum),
00285      *  we add sequential 16 bit words to it, and at the end, fold
00286      *  back all the carry bits from the top 16 bits into the lower
00287      *  16 bits.
00288      */
00289     while (nleft > 1)  {
00290         sum += *w++;
00291         nleft -= 2;
00292     }
00293     if (nleft == 1)
00294         sum += htons(*(u_char *)w<<8);
00295 
00296     /*
00297      * add back carry outs from top 16 bits to low 16 bits
00298      */
00299     sum = (sum >> 16) + (sum & 0xffff); /* add hi 16 to low 16 */
00300     sum += (sum >> 16);         /* add carry */
00301     answer = ~sum;              /* truncate to 16 bits */
00302     return (answer);
00303 }

00305 /*
00306  * Given the host-byte-order value of the checksum field in a packet
00307  * header, and the network-byte-order computed checksum of the data
00308  * that the checksum covers (including the checksum itself), compute
00309  * what the checksum field *should* have been.
00310  */
00311 u_int16_t
00312 in_cksum_shouldbe(u_int16_t sum, u_int16_t computed_sum)
00313 {
00314     u_int32_t shouldbe;
00315 
00316     /*
00317      * The value that should have gone into the checksum field
00318      * is the negative of the value gotten by summing up everything
00319      * *but* the checksum field.
00320      *
00321      * We can compute that by subtracting the value of the checksum
00322      * field from the sum of all the data in the packet, and then
00323      * computing the negative of that value.
00324      *
00325      * "sum" is the value of the checksum field, and "computed_sum"
00326      * is the negative of the sum of all the data in the packets,
00327      * so that's -(-computed_sum - sum), or (sum + computed_sum).
00328      *
00329      * All the arithmetic in question is one's complement, so the
00330      * addition must include an end-around carry; we do this by
00331      * doing the arithmetic in 32 bits (with no sign-extension),
00332      * and then adding the upper 16 bits of the sum, which contain
00333      * the carry, to the lower 16 bits of the sum, and then do it
00334      * again in case *that* sum produced a carry.
00335      *
00336      * As RFC 1071 notes, the checksum can be computed without
00337      * byte-swapping the 16-bit words; summing 16-bit words
00338      * on a big-endian machine gives a big-endian checksum, which
00339      * can be directly stuffed into the big-endian checksum fields
00340      * in protocol headers, and summing words on a little-endian
00341      * machine gives a little-endian checksum, which must be
00342      * byte-swapped before being stuffed into a big-endian checksum
00343      * field.
00344      *
00345      * "computed_sum" is a network-byte-order value, so we must put
00346      * it in host byte order before subtracting it from the
00347      * host-byte-order value from the header; the adjusted checksum
00348      * will be in host byte order, which is what we'll return.
00349      */
00350     shouldbe = sum;
00351     shouldbe += ntohs(computed_sum);
00352     shouldbe = (shouldbe & 0xFFFF) + (shouldbe >> 16);
00353     shouldbe = (shouldbe & 0xFFFF) + (shouldbe >> 16);
00354     return shouldbe;
00355 }
