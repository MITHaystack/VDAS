00050 enc_if_print(const struct pcap_pkthdr *h, register const u_char *p)
00051 {
00052     register u_int length = h->len;
00053     register u_int caplen = h->caplen;
00054     int flags;
00055     const struct enchdr *hdr;
00056 
00057     if (caplen < ENC_HDRLEN) {
00058         printf("[|enc]");
00059         goto out;
00060     }
00061 
00062     hdr = (struct enchdr *)p;
00063     flags = hdr->flags;
00064     if (flags == 0)
00065         printf("(unprotected): ");
00066     else
00067         printf("(");
00068     ENC_PRINT_TYPE(flags, M_AUTH, "authentic");
00069     ENC_PRINT_TYPE(flags, M_CONF, "confidential");
00070     /* ENC_PRINT_TYPE(flags, M_TUNNEL, "tunnel"); */
00071     printf("SPI 0x%08x: ", EXTRACT_32BITS(&hdr->spi));
00072 
00073     length -= ENC_HDRLEN;
00074     caplen -= ENC_HDRLEN;
00075     p += ENC_HDRLEN;
00076     
00077     switch (hdr->af) {
00078     case AF_INET:
00079         ip_print(gndo, p, length);
00080         break;
00081 #ifdef INET6
00082     case AF_INET6:
00083         ip6_print(p, length);
00084         break;
00085 #endif /*INET6*/
00086     }
00087 
00088 out:
00089     return (ENC_HDRLEN);
