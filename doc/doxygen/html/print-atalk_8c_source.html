00091  * Print AppleTalk LLAP packets.
00092  */
00093 u_int
00094 llap_print(register const u_char *bp, u_int length)
00095 {
00096     register const struct LAP *lp;
00097     register const struct atDDP *dp;
00098     register const struct atShortDDP *sdp;
00099     u_short snet;
00100     u_int hdrlen;
00101 
00102     /*
00103      * Our packet is on a 4-byte boundary, as we're either called
00104      * directly from a top-level link-layer printer (ltalk_if_print)
00105      * or from the UDP printer.  The LLAP+DDP header is a multiple
00106      * of 4 bytes in length, so the DDP payload is also on a 4-byte
00107      * boundary, and we don't need to align it before calling
00108      * "ddp_print()".
00109      */
00110     lp = (const struct LAP *)bp;
00111     bp += sizeof(*lp);
00112     length -= sizeof(*lp);
00113     hdrlen = sizeof(*lp);
00114     switch (lp->type) {
00115 
00116     case lapShortDDP:
00117         if (length < ddpSSize) {
00118             (void)printf(" [|sddp %d]", length);
00119             return (length);
00120         }
00121         sdp = (const struct atShortDDP *)bp;
00122         printf("%s.%s",
00123             ataddr_string(0, lp->src), ddpskt_string(sdp->srcSkt));
00124         printf(" > %s.%s:",
00125             ataddr_string(0, lp->dst), ddpskt_string(sdp->dstSkt));
00126         bp += ddpSSize;
00127         length -= ddpSSize;
00128         hdrlen += ddpSSize;
00129         ddp_print(bp, length, sdp->type, 0, lp->src, sdp->srcSkt);
00130         break;
00131 
00132     case lapDDP:
00133         if (length < ddpSize) {
00134             (void)printf(" [|ddp %d]", length);
00135             return (length);
00136         }
00137         dp = (const struct atDDP *)bp;
00138         snet = EXTRACT_16BITS(&dp->srcNet);
00139         printf("%s.%s", ataddr_string(snet, dp->srcNode),
00140             ddpskt_string(dp->srcSkt));
00141         printf(" > %s.%s:",
00142             ataddr_string(EXTRACT_16BITS(&dp->dstNet), dp->dstNode),
00143             ddpskt_string(dp->dstSkt));
00144         bp += ddpSize;
00145         length -= ddpSize;
00146         hdrlen += ddpSize;
00147         ddp_print(bp, length, dp->type, snet, dp->srcNode, dp->srcSkt);
00148         break;
00149 
00150 #ifdef notdef
00151     case lapKLAP:
00152         klap_print(bp, length);
00153         break;
00154 #endif
00155 
00156     default:
00157         printf("%d > %d at-lap#%d %d",
00158             lp->src, lp->dst, lp->type, length);
00159         break;
00160     }
00161     return (hdrlen);
00162 }

00260     char c;
00261     u_int32_t data;

00299         case atpXO:
00300             (void)printf(" [XO]");
00301             break;
00302         case atpSTS:
00303             (void)printf(" [STS]");
00304             break;
00305         case atpXO|atpSTS:
00306             (void)printf(" [XO,STS]");
00307             break;
00308         }
00309         break;
00310 
00311     case atpRelCode:
00312         (void)printf(" atp-rel  %d", EXTRACT_16BITS(&ap->transID));
00313 
00314         atp_bitmap_print(ap->bitmap);
00315 
00316         /* length should be zero */
00317         if (length)
00318             (void)printf(" [len=%d]", length);
00319 
00320         /* there shouldn't be any control flags */
00321         if (ap->control & (atpXO|atpEOM|atpSTS)) {
00322             c = '[';
00323             if (ap->control & atpXO) {
00324                 (void)printf("%cXO", c);
00325                 c = ',';
00326             }
00327             if (ap->control & atpEOM) {
00328                 (void)printf("%cEOM", c);
00329                 c = ',';
00330             }
00331             if (ap->control & atpSTS) {
00332                 (void)printf("%cSTS", c);
00333                 c = ',';
00334             }
00335             (void)printf("]");
00336         }
00337         break;
00338 
00339     default:
00340         (void)printf(" atp-0x%x  %d (%d)", ap->control,
00341                  EXTRACT_16BITS(&ap->transID), length);

00343     }

00349 static void
00350 atp_bitmap_print(register u_char bm)
00351 {
00352     register char c;
00353     register int i;
00354 
00355     /*
00356      * The '& 0xff' below is needed for compilers that want to sign
00357      * extend a u_char, which is the case with the Ultrix compiler.
00358      * (gcc is smart enough to eliminate it, at least on the Sparc).
00359      */
00360     if ((bm + 1) & (bm & 0xff)) {
00361         c = '<';
00362         for (i = 0; bm; ++i) {
00363             if (bm & 1) {
00364                 (void)printf("%c%d", c, i);
00365                 c = ',';
00366             }
00367             bm >>= 1;
00368         }
00369         (void)printf(">");
00370     } else {
00371         for (i = 0; bm; ++i)
00372             bm >>= 1;
00373         if (i > 1)
00374             (void)printf("<0-%d>", i - 1);
00375         else
00376             (void)printf("<0>");
00377     }
00378 }

00380 static void
00381 nbp_print(register const struct atNBP *np, u_int length, register u_short snet,
00382       register u_char snode, register u_char skt)
00383 {
00384     register const struct atNBPtuple *tp =
00385         (const struct atNBPtuple *)((u_char *)np + nbpHeaderSize);
00386     int i;
00387     const u_char *ep;
00388 
00389     if (length < nbpHeaderSize) {
00390         (void)printf(" truncated-nbp %d", length);
00391         return;
00392     }
00393 
00394     length -= nbpHeaderSize;
00395     if (length < 8) {
00396         /* must be room for at least one tuple */
00397         (void)printf(" truncated-nbp %d", length + nbpHeaderSize);
00398         return;
00399     }
00400     /* ep points to end of available data */
00401     ep = snapend;
00402     if ((const u_char *)tp > ep) {
00403         fputs(tstr, stdout);
00404         return;
00405     }
00406     switch (i = np->control & 0xf0) {
00407 
00408     case nbpBrRq:
00409     case nbpLkUp:
00410         (void)printf(i == nbpLkUp? " nbp-lkup %d:":" nbp-brRq %d:",
00411                  np->id);
00412         if ((const u_char *)(tp + 1) > ep) {
00413             fputs(tstr, stdout);
00414             return;
00415         }
00416         (void)nbp_name_print(tp, ep);
00417         /*
00418          * look for anomalies: the spec says there can only
00419          * be one tuple, the address must match the source
00420          * address and the enumerator should be zero.
00421          */
00422         if ((np->control & 0xf) != 1)
00423             (void)printf(" [ntup=%d]", np->control & 0xf);
00424         if (tp->enumerator)
00425             (void)printf(" [enum=%d]", tp->enumerator);
00426         if (EXTRACT_16BITS(&tp->net) != snet ||
00427             tp->node != snode || tp->skt != skt)
00428             (void)printf(" [addr=%s.%d]",
00429                 ataddr_string(EXTRACT_16BITS(&tp->net),
00430                 tp->node), tp->skt);
00431         break;
00432 
00433     case nbpLkUpReply:
00434         (void)printf(" nbp-reply %d:", np->id);
00435 
00436         /* print each of the tuples in the reply */
00437         for (i = np->control & 0xf; --i >= 0 && tp; )
00438             tp = nbp_tuple_print(tp, ep, snet, snode, skt);
00439         break;
00440 
00441     default:
00442         (void)printf(" nbp-0x%x  %d (%d)", np->control, np->id,
00443                 length);
00444         break;
00445     }
00446 }

00543     char nambuf[MAXHOSTNAMELEN + 20];
00544     static int first = 1;
00545     FILE *fp;
00546 
00547     /*
00548      * if this is the first call, see if there's an AppleTalk
00549      * number to name map file.
00550      */
00551     if (first && (first = 0, !nflag)
00552         && (fp = fopen("/etc/atalk.names", "r"))) {
00553         char line[256];
00554         int i1, i2, i3;
00555 
00556         while (fgets(line, sizeof(line), fp)) {
00557             if (line[0] == '\n' || line[0] == 0 || line[0] == '#')
00558                 continue;

00560                      nambuf) == 4)
00561                 /* got a hostname. */
00562                 i3 |= ((i1 << 8) | i2) << 8;
00563             else if (sscanf(line, "%d.%d %256s", &i1, &i2,
00564                     nambuf) == 3)
00565                 /* got a net name */
00566                 i3 = (((i1 << 8) | i2) << 8) | 255;
00567             else
00568                 continue;
00569 
00570             for (tp = &hnametable[i3 & (HASHNAMESIZE-1)];
00571                  tp->nxt; tp = tp->nxt)
00572                 ;
00573             tp->addr = i3;
00574             tp->nxt = newhnamemem();
00575             tp->name = strdup(nambuf);
00576         }
00577         fclose(fp);
00578     }
00579 
00580     for (tp = &hnametable[i & (HASHNAMESIZE-1)]; tp->nxt; tp = tp->nxt)
00581         if (tp->addr == i)
00582             return (tp->name);
00583 
00584     /* didn't have the node name -- see if we've got the net name */
00585     i |= 255;
00586     for (tp2 = &hnametable[i & (HASHNAMESIZE-1)]; tp2->nxt; tp2 = tp2->nxt)
00587         if (tp2->addr == i) {
00588             tp->addr = (atnet << 8) | athost;
00589             tp->nxt = newhnamemem();
00590             (void)snprintf(nambuf, sizeof(nambuf), "%s.%d",
00591                 tp2->name, athost);
00592             tp->name = strdup(nambuf);
00593             return (tp->name);
00594         }
00595 
00596     tp->addr = (atnet << 8) | athost;
00597     tp->nxt = newhnamemem();
00598     if (athost != 255)
00599         (void)snprintf(nambuf, sizeof(nambuf), "%d.%d.%d",
00600             atnet >> 8, atnet & 0xff, athost);
00601     else
00602         (void)snprintf(nambuf, sizeof(nambuf), "%d.%d", atnet >> 8,
