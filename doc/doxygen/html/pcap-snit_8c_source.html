00088 pcap_stats_snit(pcap_t *p, struct pcap_stat *ps)
00089 {
00090 
00091     /*
00092      * "ps_recv" counts packets handed to the filter, not packets
00093      * that passed the filter.  As filtering is done in userland,
00094      * this does not include packets dropped because we ran out
00095      * of buffer space.
00096      *
00097      * "ps_drop" counts packets dropped inside the "/dev/nit"
00098      * device because of flow control requirements or resource
00099      * exhaustion; it doesn't count packets dropped by the
00100      * interface driver, or packets dropped upstream.  As filtering
00101      * is done in userland, it counts packets regardless of whether
00102      * they would've passed the filter.
00103      *
00104      * These statistics don't include packets not yet read from the
00105      * kernel by libpcap or packets not yet read from libpcap by the
00106      * application.
00107      */
00108     *ps = p->md.stat;
00109     return (0);

00113 pcap_read_snit(pcap_t *p, int cnt, pcap_handler callback, u_char *user)
00114 {
00115     register int cc, n;
00116     register u_char *bp, *cp, *ep;
00117     register struct nit_bufhdr *hdrp;
00118     register struct nit_iftime *ntp;
00119     register struct nit_iflen *nlp;
00120     register struct nit_ifdrops *ndp;
00121     register int caplen;
00122 
00123     cc = p->cc;
00124     if (cc == 0) {
00125         cc = read(p->fd, (char *)p->buffer, p->bufsize);
00126         if (cc < 0) {
00127             if (errno == EWOULDBLOCK)
00128                 return (0);
00129             snprintf(p->errbuf, sizeof(p->errbuf), "pcap_read: %s",
00130                 pcap_strerror(errno));
00131             return (-1);
00132         }
00133         bp = p->buffer;
00134     } else
00135         bp = p->bp;
00136 
00137     /*
00138      * loop through each snapshot in the chunk
00139      */
00140     n = 0;
00141     ep = bp + cc;
00142     while (bp < ep) {
00143         /*
00144          * Has "pcap_breakloop()" been called?
00145          * If so, return immediately - if we haven't read any
00146          * packets, clear the flag and return -2 to indicate
00147          * that we were told to break out of the loop, otherwise
00148          * leave the flag set, so that the *next* call will break
00149          * out of the loop without having read any packets, and
00150          * return the number of packets we've processed so far.
00151          */
00152         if (p->break_loop) {
00153             if (n == 0) {
00154                 p->break_loop = 0;
00155                 return (-2);
00156             } else {
00157                 p->bp = bp;
00158                 p->cc = ep - bp;
00159                 return (n);
00160             }
00161         }
00162 
00163         ++p->md.stat.ps_recv;
00164         cp = bp;
00165 
00166         /* get past NIT buffer  */
00167         hdrp = (struct nit_bufhdr *)cp;
00168         cp += sizeof(*hdrp);
00169 
00170         /* get past NIT timer   */
00171         ntp = (struct nit_iftime *)cp;
00172         cp += sizeof(*ntp);
00173 
00174         ndp = (struct nit_ifdrops *)cp;
00175         p->md.stat.ps_drop = ndp->nh_drops;
00176         cp += sizeof *ndp;
00177 
00178         /* get past packet len  */
00179         nlp = (struct nit_iflen *)cp;
00180         cp += sizeof(*nlp);
00181 
00182         /* next snapshot        */
00183         bp += hdrp->nhb_totlen;
00184 
00185         caplen = nlp->nh_pktlen;
00186         if (caplen > p->snapshot)
00187             caplen = p->snapshot;
00188 
00189         if (bpf_filter(p->fcode.bf_insns, cp, nlp->nh_pktlen, caplen)) {
00190             struct pcap_pkthdr h;
00191             h.ts = ntp->nh_timestamp;
00192             h.len = nlp->nh_pktlen;
00193             h.caplen = caplen;
00194             (*callback)(user, &h, cp);
00195             if (++n >= cnt && cnt > 0) {
00196                 p->cc = ep - bp;
00197                 p->bp = bp;
00198                 return (n);
00199             }
00200         }
00201     }
00202     p->cc = 0;
00203     return (n);
00204 }

00272     if (p->opt.rfmon) {
00273         /*
00274          * No monitor mode on SunOS 4.x (no Wi-Fi devices on
00275          * hardware supported by SunOS 4.x).
00276          */
00277         return (PCAP_ERROR_RFMON_NOTSUP);
00278     }
00279 
00280     if (p->snapshot < 96)
00281         /*
00282          * NIT requires a snapshot length of at least 96.
00283          */
00284         p->snapshot = 96;
00285 
00286     /*
00287      * Initially try a read/write open (to allow the inject
00288      * method to work).  If that fails due to permission
00289      * issues, fall back to read-only.  This allows a
00290      * non-root user to be granted specific access to pcap
00291      * capabilities via file permissions.
00292      *
00293      * XXX - we should have an API that has a flag that
00294      * controls whether to open read-only or read-write,
00295      * so that denial of permission to send (or inability
00296      * to send, if sending packets isn't supported on
00297      * the device in question) can be indicated at open
00298      * time.
00299      */
00300     p->fd = fd = open(dev, O_RDWR);
00301     if (fd < 0 && errno == EACCES)
00302         p->fd = fd = open(dev, O_RDONLY);
00303     if (fd < 0) {
00304         snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "%s: %s", dev,
00305             pcap_strerror(errno));
00306         goto bad;
00307     }
00308 
00309     /* arrange to get discrete messages from the STREAM and use NIT_BUF */
00310     if (ioctl(fd, I_SRDOPT, (char *)RMSGD) < 0) {
00311         snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "I_SRDOPT: %s",
00312             pcap_strerror(errno));
00313         goto bad;
00314     }
00315     if (ioctl(fd, I_PUSH, "nbuf") < 0) {
00316         snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "push nbuf: %s",
00317             pcap_strerror(errno));
00318         goto bad;
00319     }
00320     /* set the chunksize */
00321     si.ic_cmd = NIOCSCHUNK;
00322     si.ic_timout = INFTIM;
00323     si.ic_len = sizeof(chunksize);
00324     si.ic_dp = (char *)&chunksize;
00325     if (ioctl(fd, I_STR, (char *)&si) < 0) {
00326         snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "NIOCSCHUNK: %s",
00327             pcap_strerror(errno));
00328         goto bad;
00329     }
00330 
00331     /* request the interface */
00332     strncpy(ifr.ifr_name, p->opt.source, sizeof(ifr.ifr_name));
00333     ifr.ifr_name[sizeof(ifr.ifr_name) - 1] = '\0';

00335     si.ic_len = sizeof(ifr);
00336     si.ic_dp = (char *)&ifr;
00337     if (ioctl(fd, I_STR, (char *)&si) < 0) {
00338         snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "NIOCBIND: %s: %s",
00339             ifr.ifr_name, pcap_strerror(errno));
00340         goto bad;
00341     }
00342 
00343     /* set the snapshot length */
00344     si.ic_cmd = NIOCSSNAP;
00345     si.ic_len = sizeof(p->snapshot);
00346     si.ic_dp = (char *)&p->snapshot;
00347     if (ioctl(fd, I_STR, (char *)&si) < 0) {
00348         snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "NIOCSSNAP: %s",
00349             pcap_strerror(errno));
00350         goto bad;
00351     }
00352     if (nit_setflags(p->fd, p->opt.promisc, p->md.timeout, p->errbuf) < 0)
00353         goto bad;
00354 
00355     (void)ioctl(fd, I_FLUSH, (char *)FLUSHR);
00356     /*
00357      * NIT supports only ethernets.
00358      */
00359     p->linktype = DLT_EN10MB;
00360 
00361     p->bufsize = BUFSPACE;
00362     p->buffer = (u_char *)malloc(p->bufsize);
00363     if (p->buffer == NULL) {
00364         strlcpy(p->errbuf, pcap_strerror(errno), PCAP_ERRBUF_SIZE);
00365         goto bad;
00366     }
00367 
00368     /*
00369      * "p->fd" is an FD for a STREAMS device, so "select()" and
00370      * "poll()" should work on it.
00371      */
00372     p->selectable_fd = p->fd;
00373 
00374     /*
00375      * This is (presumably) a real Ethernet capture; give it a
00376      * link-layer-type list with DLT_EN10MB and DLT_DOCSIS, so
00377      * that an application can let you choose it, in case you're
00378      * capturing DOCSIS traffic that a Cisco Cable Modem
00379      * Termination System is putting out onto an Ethernet (it

00384     p->dlt_list = (u_int *) malloc(sizeof(u_int) * 2);

00397     p->setdirection_op = NULL;  /* Not implemented. */
00398     p->set_datalink_op = NULL;  /* can't change data link type */
00399     p->getnonblock_op = pcap_getnonblock_fd;
00400     p->setnonblock_op = pcap_setnonblock_fd;

00403     return (0);
