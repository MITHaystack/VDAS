00061  * Print bootp requests
00062  */
00063 void
00064 bootp_print(register const u_char *cp, u_int length)
00065 {
00066     register const struct bootp *bp;
00067     static const u_char vm_cmu[4] = VM_CMU;
00068     static const u_char vm_rfc1048[4] = VM_RFC1048;
00069 
00070     bp = (const struct bootp *)cp;
00071     TCHECK(bp->bp_op);
00072 
00073         printf("BOOTP/DHCP, %s",
00074            tok2str(bootp_op_values, "unknown (0x%02x)", bp->bp_op));
00075 
00076     if (bp->bp_htype == 1 && bp->bp_hlen == 6 && bp->bp_op == BOOTPREQUEST) {
00077         TCHECK2(bp->bp_chaddr[0], 6);
00078         printf(" from %s", etheraddr_string(bp->bp_chaddr));
00079     }
00080 
00081         printf(", length %u", length);
00082 
00083         if (!vflag)
00084             return;
00085 
00086     TCHECK(bp->bp_secs);
00087 
00088     /* The usual hardware address type is 1 (10Mb Ethernet) */
00089     if (bp->bp_htype != 1)
00090         printf(", htype %d", bp->bp_htype);
00091 
00092     /* The usual length for 10Mb Ethernet address is 6 bytes */
00093     if (bp->bp_htype != 1 || bp->bp_hlen != 6)
00094         printf(", hlen %d", bp->bp_hlen);
00095 
00096     /* Only print interesting fields */
00097     if (bp->bp_hops)
00098         printf(", hops %d", bp->bp_hops);
00099     if (bp->bp_xid)
00100         printf(", xid 0x%x", EXTRACT_32BITS(&bp->bp_xid));
00101     if (bp->bp_secs)
00102         printf(", secs %d", EXTRACT_16BITS(&bp->bp_secs));
00103 
00104     printf(", Flags [%s]",
00105         bittok2str(bootp_flag_values, "none", EXTRACT_16BITS(&bp->bp_flags)));
00106     if (vflag > 1)
00107         printf(" (0x%04x)", EXTRACT_16BITS(&bp->bp_flags));
00108 
00109     /* Client's ip address */
00110     TCHECK(bp->bp_ciaddr);
00111     if (bp->bp_ciaddr.s_addr)
00112         printf("\n\t  Client-IP %s", ipaddr_string(&bp->bp_ciaddr));
00113 
00114     /* 'your' ip address (bootp client) */
00115     TCHECK(bp->bp_yiaddr);
00116     if (bp->bp_yiaddr.s_addr)
00117         printf("\n\t  Your-IP %s", ipaddr_string(&bp->bp_yiaddr));
00118 
00119     /* Server's ip address */
00120     TCHECK(bp->bp_siaddr);
00121     if (bp->bp_siaddr.s_addr)
00122         printf("\n\t  Server-IP %s", ipaddr_string(&bp->bp_siaddr));
00123 
00124     /* Gateway's ip address */
00125     TCHECK(bp->bp_giaddr);
00126     if (bp->bp_giaddr.s_addr)
00127         printf("\n\t  Gateway-IP %s", ipaddr_string(&bp->bp_giaddr));
00128 
00129     /* Client's Ethernet address */
00130     if (bp->bp_htype == 1 && bp->bp_hlen == 6) {
00131         TCHECK2(bp->bp_chaddr[0], 6);
00132         printf("\n\t  Client-Ethernet-Address %s", etheraddr_string(bp->bp_chaddr));
00133     }
00134 
00135     TCHECK2(bp->bp_sname[0], 1);        /* check first char only */
00136     if (*bp->bp_sname) {
00137         printf("\n\t  sname \"");
00138         if (fn_print(bp->bp_sname, snapend)) {
00139             putchar('"');
00140             fputs(tstr + 1, stdout);
00141             return;
00142         }
00143         putchar('"');
00144     }
00145     TCHECK2(bp->bp_file[0], 1);     /* check first char only */
00146     if (*bp->bp_file) {
00147         printf("\n\t  file \"");
00148         if (fn_print(bp->bp_file, snapend)) {
00149             putchar('"');
00150             fputs(tstr + 1, stdout);
00151             return;
00152         }
00153         putchar('"');
00154     }
00155 
00156     /* Decode the vendor buffer */
00157     TCHECK(bp->bp_vend[0]);
00158     if (memcmp((const char *)bp->bp_vend, vm_rfc1048,
00159          sizeof(u_int32_t)) == 0)
00160         rfc1048_print(bp->bp_vend);
00161     else if (memcmp((const char *)bp->bp_vend, vm_cmu,
00162               sizeof(u_int32_t)) == 0)
00163         cmu_print(bp->bp_vend);
00164     else {
00165         u_int32_t ul;
00166 
00167         ul = EXTRACT_32BITS(&bp->bp_vend);
00168         if (ul != 0)
00169             printf("\n\t  Vendor-#0x%x", ul);
00170     }
00171 
00172     return;
00173 trunc:
00174     fputs(tstr, stdout);
00175 }

00366 static void
00367 rfc1048_print(register const u_char *bp)
00368 {
00369     register u_int16_t tag;
00370     register u_int len;
00371     register const char *cp;
00372     register char c;
00373     int first, idx;
00374     u_int32_t ul;
00375     u_int16_t us;
00376     u_int8_t uc, subopt, suboptlen;
00377 
00378     printf("\n\t  Vendor-rfc1048 Extensions");
00379 
00380     /* Step over magic cookie */
00381         printf("\n\t    Magic Cookie 0x%08x", EXTRACT_32BITS(bp));
00382     bp += sizeof(int32_t);
00383 
00384     /* Loop while we there is a tag left in the buffer */
00385     while (TTEST2(*bp, 1)) {
00386         tag = *bp++;
00387         if (tag == TAG_PAD && vflag < 3)
00388             continue;
00389         if (tag == TAG_END && vflag < 3)
00390             return;
00391         if (tag == TAG_EXTENDED_OPTION) {
00392             TCHECK2(*(bp + 1), 2);
00393             tag = EXTRACT_16BITS(bp + 1);
00394             /* XXX we don't know yet if the IANA will
00395              * preclude overlap of 1-byte and 2-byte spaces.
00396              * If not, we need to offset tag after this step.
00397              */
00398             cp = tok2str(xtag2str, "?xT%u", tag);
00399         } else
00400             cp = tok2str(tag2str, "?T%u", tag);
00401         c = *cp++;
00402 
00403         if (tag == TAG_PAD || tag == TAG_END)
00404             len = 0;
00405         else {
00406             /* Get the length; check for truncation */
00407             TCHECK2(*bp, 1);
00408             len = *bp++;
00409         }
00410 
00411         printf("\n\t    %s Option %u, length %u%s", cp, tag, len,
00412             len > 0 ? ": " : "");
00413 
00414         if (tag == TAG_PAD && vflag > 2) {
00415             u_int ntag = 1;
00416             while (TTEST2(*bp, 1) && *bp == TAG_PAD) {
00417                 bp++;
00418                 ntag++;
00419             }
00420             if (ntag > 1)
00421                 printf(", occurs %u", ntag);
00422         }
00423 
00424         if (!TTEST2(*bp, len)) {
00425             printf("[|rfc1048 %u]", len);
00426             return;
00427         }
00428 
00429         if (tag == TAG_DHCP_MESSAGE && len == 1) {
00430             uc = *bp++;
00431                         printf("%s", tok2str(dhcp_msg_values, "Unknown (%u)", uc));
00432                         continue;
00433         }
00434 
00435         if (tag == TAG_PARM_REQUEST) {
00436             idx = 0;
00437             while (len-- > 0) {
00438                 uc = *bp++;
00439                 cp = tok2str(tag2str, "?Option %u", uc);
00440                 if (idx % 4 == 0)
00441                     printf("\n\t      ");
00442                 else
00443                     printf(", ");
00444                 printf("%s", cp + 1);
00445                 idx++;
00446             }
00447             continue;
00448         }
00449 
00450         if (tag == TAG_EXTENDED_REQUEST) {
00451             first = 1;
00452             while (len > 1) {
00453                 len -= 2;
00454                 us = EXTRACT_16BITS(bp);
00455                 bp += 2;
00456                 cp = tok2str(xtag2str, "?xT%u", us);
00457                 if (!first)
00458                     putchar('+');
00459                 printf("%s", cp + 1);
00460                 first = 0;
00461             }
00462             continue;
00463         }
00464 
00465         /* Print data */
00466         if (c == '?') {
00467             /* Base default formats for unknown tags on data size */
00468             if (len & 1)
00469                 c = 'b';
00470             else if (len & 2)
00471                 c = 's';
00472             else
00473                 c = 'l';
00474         }
00475         first = 1;
00476         switch (c) {
00477 
00478         case 'a':
00479             /* ascii strings */
00480             putchar('"');
00481             if (fn_printn(bp, len, snapend)) {
00482                 putchar('"');
00483                 goto trunc;
00484             }
00485             putchar('"');
00486             bp += len;
00487             len = 0;
00488             break;
00489 
00490         case 'i':
00491         case 'l':
00492         case 'L':
00493             /* ip addresses/32-bit words */
00494             while (len >= sizeof(ul)) {
00495                 if (!first)
00496                     putchar(',');
00497                 ul = EXTRACT_32BITS(bp);
00498                 if (c == 'i') {
00499                     ul = htonl(ul);
00500                     printf("%s", ipaddr_string(&ul));
00501                 } else if (c == 'L')
00502                     printf("%d", ul);
00503                 else
00504                     printf("%u", ul);
00505                 bp += sizeof(ul);
00506                 len -= sizeof(ul);
00507                 first = 0;
00508             }
00509             break;
00510 
00511         case 'p':
00512             /* IP address pairs */
00513             while (len >= 2*sizeof(ul)) {
00514                 if (!first)
00515                     putchar(',');
00516                 memcpy((char *)&ul, (const char *)bp, sizeof(ul));
00517                 printf("(%s:", ipaddr_string(&ul));
00518                 bp += sizeof(ul);
00519                 memcpy((char *)&ul, (const char *)bp, sizeof(ul));
00520                 printf("%s)", ipaddr_string(&ul));
00521                 bp += sizeof(ul);
00522                 len -= 2*sizeof(ul);
00523                 first = 0;
00524             }
00525             break;
00526 
00527         case 's':
00528             /* shorts */
00529             while (len >= sizeof(us)) {
00530                 if (!first)
00531                     putchar(',');
00532                 us = EXTRACT_16BITS(bp);
00533                 printf("%u", us);
00534                 bp += sizeof(us);
00535                 len -= sizeof(us);
00536                 first = 0;
00537             }
00538             break;
00539 
00540         case 'B':
00541             /* boolean */
00542             while (len > 0) {
00543                 if (!first)
00544                     putchar(',');
00545                 switch (*bp) {
00546                 case 0:
00547                     putchar('N');
00548                     break;
00549                 case 1:
00550                     putchar('Y');
00551                     break;
00552                 default:
00553                     printf("%u?", *bp);
00554                     break;
00555                 }
00556                 ++bp;
00557                 --len;
00558                 first = 0;
00559             }
00560             break;
00561 
00562         case 'b':
00563         case 'x':
