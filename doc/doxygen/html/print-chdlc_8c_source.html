00066 u_int
00067 chdlc_print(register const u_char *p, u_int length) {
00068     u_int proto;
00069 
00070     proto = EXTRACT_16BITS(&p[2]);
00071     if (eflag) {
00072                 printf("%s, ethertype %s (0x%04x), length %u: ",
00073                        tok2str(chdlc_cast_values, "0x%02x", p[0]),
00074                        tok2str(ethertype_values, "Unknown", proto),
00075                        proto,
00076                        length);
00077     }
00078 
00079     length -= CHDLC_HDRLEN;
00080     p += CHDLC_HDRLEN;
00081 
00082     switch (proto) {
00083     case ETHERTYPE_IP:
00084         ip_print(gndo, p, length);
00085         break;
00086 #ifdef INET6
00087     case ETHERTYPE_IPV6:
00088         ip6_print(p, length);
00089         break;
00090 #endif
00091     case CHDLC_TYPE_SLARP:
00092         chdlc_slarp_print(p, length);
00093         break;
00094 #if 0
00095     case CHDLC_TYPE_CDP:
00096         chdlc_cdp_print(p, length);
00097         break;
00098 #endif
00099         case ETHERTYPE_MPLS:
00100         case ETHERTYPE_MPLS_MULTI:
00101                 mpls_print(p, length);
00102         break;
00103         case ETHERTYPE_ISO:
00104                 /* is the fudge byte set ? lets verify by spotting ISO headers */
00105                 if (*(p+1) == 0x81 ||
00106                     *(p+1) == 0x82 ||
00107                     *(p+1) == 0x83)
00108                     isoclns_print(p+1, length-1, length-1);
00109                 else
00110                     isoclns_print(p, length, length);
00111                 break;
00112     default:
00113                 if (!eflag)
00114                         printf("unknown CHDLC protocol (0x%04x)", proto);
00115                 break;
00116     }
00117 
00118     return (CHDLC_HDRLEN);
00119 }

00124 struct cisco_slarp {
00125     u_int8_t code[4];
00126 #define SLARP_REQUEST   0
00127 #define SLARP_REPLY 1
00128 #define SLARP_KEEPALIVE 2
00129     union {
00130         struct {
00131             u_int8_t addr[4];
00132             u_int8_t mask[4];
00133         } addr;
00134         struct {
00135             u_int8_t myseq[4];
00136             u_int8_t yourseq[4];
00137             u_int8_t rel[2];
00138         } keep;
00139     } un;
00140 };

00142 #define SLARP_MIN_LEN   14
00143 #define SLARP_MAX_LEN   18

00145 static void
00146 chdlc_slarp_print(const u_char *cp, u_int length)
00147 {
00148     const struct cisco_slarp *slarp;
00149         u_int sec,min,hrs,days;
00150 
00151         printf("SLARP (length: %u), ",length);
00152     if (length < SLARP_MIN_LEN)
00153         goto trunc;
00154 
00155     slarp = (const struct cisco_slarp *)cp;
00156     TCHECK2(*slarp, SLARP_MIN_LEN);
00157     switch (EXTRACT_32BITS(&slarp->code)) {
00158     case SLARP_REQUEST:
00159         printf("request");
00160         /*
00161          * At least according to William "Chops" Westfield's
00162          * message in
00163          *
00164          *  http://www.nethelp.no/net/cisco-hdlc.txt
00165          *
00166          * the address and mask aren't used in requests -
00167          * they're just zero.
00168          */
00169         break;
00170     case SLARP_REPLY:
00171         printf("reply %s/%s",
00172             ipaddr_string(&slarp->un.addr.addr),
00173             ipaddr_string(&slarp->un.addr.mask));
00174         break;
00175     case SLARP_KEEPALIVE:
00176         printf("keepalive: mineseen=0x%08x, yourseen=0x%08x, reliability=0x%04x",
00177                        EXTRACT_32BITS(&slarp->un.keep.myseq),
00178                        EXTRACT_32BITS(&slarp->un.keep.yourseq),
00179                        EXTRACT_16BITS(&slarp->un.keep.rel));
00180 
00181                 if (length >= SLARP_MAX_LEN) { /* uptime-stamp is optional */
00182                         cp += SLARP_MIN_LEN;
00183                         if (!TTEST2(*cp, 4))
00184                                 goto trunc;
00185                         sec = EXTRACT_32BITS(cp) / 1000;
00186                         min = sec / 60; sec -= min * 60;
00187                         hrs = min / 60; min -= hrs * 60;
00188                         days = hrs / 24; hrs -= days * 24;
00189                         printf(", link uptime=%ud%uh%um%us",days,hrs,min,sec);
00190                 }
00191         break;
00192     default:
00193         printf("0x%02x unknown", EXTRACT_32BITS(&slarp->code));
00194                 if (vflag <= 1)
00195                     print_unknown_data(cp+4,"\n\t",length-4);
00196         break;
00197     }
00198 
00199     if (SLARP_MAX_LEN < length && vflag)
00200         printf(", (trailing junk: %d bytes)", length - SLARP_MAX_LEN);
00201         if (vflag > 1)
00202             print_unknown_data(cp+4,"\n\t",length-4);
00203     return;
00204 
00205 trunc:
00206     printf("[|slarp]");
00207 }
