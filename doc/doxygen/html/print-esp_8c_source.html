00088  * this will adjust ndo_packetp and ndo_snapend to new buffer!
00089  */
00090 int esp_print_decrypt_buffer_by_ikev2(netdissect_options *ndo,
00091                       int initiator,
00092                       u_char spii[8], u_char spir[8],
00093                       u_char *buf, u_char *end)
00094 {
00095     struct sa_list *sa;
00096     u_char *iv;
00097     int len;
00098     EVP_CIPHER_CTX ctx;
00099 
00100     /* initiator arg is any non-zero value */
00101     if(initiator) initiator=1;
00102                        
00103     /* see if we can find the SA, and if so, decode it */
00104     for (sa = ndo->ndo_sa_list_head; sa != NULL; sa = sa->next) {
00105         if (sa->spi == 0
00106             && initiator == sa->initiator
00107             && memcmp(spii, sa->spii, 8) == 0
00108             && memcmp(spir, sa->spir, 8) == 0)
00109             break;
00110     }
00111 
00112     if(sa == NULL) return 0;
00113     if(sa->evp == NULL) return 0;
00114 
00115     /*
00116      * remove authenticator, and see if we still have something to
00117      * work with
00118      */
00119     end = end - sa->authlen;
00120     iv  = buf;
00121     buf = buf + sa->ivlen;
00122     len = end-buf;
00123 
00124     if(end <= buf) return 0;
00125 
00126     memset(&ctx, 0, sizeof(ctx));
00127     if (EVP_CipherInit(&ctx, sa->evp, sa->secret, NULL, 0) < 0)
00128         (*ndo->ndo_warning)(ndo, "espkey init failed");
00129     EVP_CipherInit(&ctx, NULL, NULL, iv, 0);
00130     EVP_Cipher(&ctx, buf, buf, len);
00131     EVP_CIPHER_CTX_cleanup(&ctx);
00132 
00133     ndo->ndo_packetp = buf;
00134     ndo->ndo_snapend = end;
00135 
00136     return 1;
00137     
00138 }

00215 static int
00216 espprint_decode_encalgo(netdissect_options *ndo,
00217             char *decode, struct sa_list *sa)
00218 {
00219     int len;
00220     size_t i;
00221     const EVP_CIPHER *evp;
00222     int authlen = 0;
00223     char *colon, *p;
00224     
00225     colon = strchr(decode, ':');
00226     if (colon == NULL) {
00227         (*ndo->ndo_warning)(ndo, "failed to decode espsecret: %s\n", decode);
00228         return 0;
00229     }
00230     *colon = '\0';
00231     
00232     len = colon - decode;
00233     if (strlen(decode) > strlen("-hmac96") &&
00234         !strcmp(decode + strlen(decode) - strlen("-hmac96"),
00235             "-hmac96")) {
00236         p = strstr(decode, "-hmac96");
00237         *p = '\0';
00238         authlen = 12;
00239     }
00240     if (strlen(decode) > strlen("-cbc") &&
00241         !strcmp(decode + strlen(decode) - strlen("-cbc"), "-cbc")) {
00242         p = strstr(decode, "-cbc");
00243         *p = '\0';
00244     }
00245     evp = EVP_get_cipherbyname(decode);
00246 
00247     if (!evp) {
00248         (*ndo->ndo_warning)(ndo, "failed to find cipher algo %s\n", decode);
00249         sa->evp = NULL;
00250         sa->authlen = 0;
00251         sa->ivlen = 0;
00252         return 0;
00253     }
00254     
00255     sa->evp = evp;
00256     sa->authlen = authlen;
00257     sa->ivlen = EVP_CIPHER_iv_length(evp);
00258     
00259     colon++;
00260     if (colon[0] == '0' && colon[1] == 'x') {
00261         /* decode some hex! */
00262 
00263         colon += 2;
00264         sa->secretlen = espprint_decode_hex(ndo, sa->secret, sizeof(sa->secret), colon);
00265         if(sa->secretlen == 0) return 0;
00266     } else {
00267         i = strlen(colon);
00268         
00269         if (i < sizeof(sa->secret)) {
00270             memcpy(sa->secret, colon, i);
00271             sa->secretlen = i;
00272         } else {
00273             memcpy(sa->secret, colon, sizeof(sa->secret));
00274             sa->secretlen = sizeof(sa->secret);
00275         }
00276     }
00277 
00278     return 1;
00279 }

00305 static void esp_print_decode_ikeline(netdissect_options *ndo, char *line,
00306                      const char *file, int lineno)
00307 {
00308     /* it's an IKEv2 secret, store it instead */
00309     struct sa_list sa1;
00310 
00311     char *init;
00312     char *icookie, *rcookie;
00313     int   ilen, rlen;
00314     char *authkey;
00315     char *enckey;
00316     
00317     init = strsep(&line, " \t");
00318     icookie = strsep(&line, " \t");
00319     rcookie = strsep(&line, " \t");
00320     authkey = strsep(&line, " \t");
00321     enckey  = strsep(&line, " \t");
00322     
00323     /* if any fields are missing */
00324     if(!init || !icookie || !rcookie || !authkey || !enckey) {
00325         (*ndo->ndo_warning)(ndo, "print_esp: failed to find all fields for ikev2 at %s:%u",
00326                     file, lineno);
00327         
00328         return;
00329     }
00330     
00331     ilen = strlen(icookie);
00332     rlen = strlen(rcookie);
00333 
00334     if((init[0]!='I' && init[0]!='R')
00335        || icookie[0]!='0' || icookie[1]!='x'
00336        || rcookie[0]!='0' || rcookie[1]!='x'
00337        || ilen!=18
00338        || rlen!=18) {
00339         (*ndo->ndo_warning)(ndo, "print_esp: line %s:%u improperly formatted.",
00340                     file, lineno);
00341 
00342         (*ndo->ndo_warning)(ndo, "init=%s icookie=%s(%u) rcookie=%s(%u)",
00343                     init, icookie, ilen, rcookie, rlen);
00344         
00345         return;
00346     }
00347 
00348     sa1.spi = 0;
00349     sa1.initiator = (init[0] == 'I');
00350     if(espprint_decode_hex(ndo, sa1.spii, sizeof(sa1.spii), icookie+2)!=8)
00351         return;
00352 
00353     if(espprint_decode_hex(ndo, sa1.spir, sizeof(sa1.spir), rcookie+2)!=8)
00354         return;
00355 
00356     if(!espprint_decode_encalgo(ndo, enckey, &sa1)) return;
00357 
00358     if(!espprint_decode_authalgo(ndo, authkey, &sa1)) return;
00359     
00360     esp_print_addsa(ndo, &sa1, FALSE);
00361 }

00374

00376     char *decode;
00377 
00378     spikey = strsep(&line, " \t");
00379     sa_def = 0;
00380     memset(&sa1, 0, sizeof(struct sa_list));
00381 
00382     /* if there is only one token, then it is an algo:key token */
00383     if (line == NULL) {
00384         decode = spikey;
00385         spikey = NULL;
00386         /* memset(&sa1.daddr, 0, sizeof(sa1.daddr)); */
00387         /* sa1.spi = 0; */
00388         sa_def    = 1;
00389     } else
00390         decode = line;
00391 
00392     if (spikey && strcasecmp(spikey, "file") == 0) {
00393         /* open file and read it */
00394         FILE *secretfile;
00395         char  fileline[1024];
00396         int   lineno=0;
00397         char  *nl;
00398         char *filename = line;
00399 
00400         secretfile = fopen(filename, FOPEN_READ_TXT);
00401         if (secretfile == NULL) {
00402             perror(filename);
00403             exit(3);
00404         }
00405 
00406         while (fgets(fileline, sizeof(fileline)-1, secretfile) != NULL) {
00407             lineno++;
00408             /* remove newline from the line */
00409             nl = strchr(fileline, '\n');
00410             if (nl)
00411                 *nl = '\0';
00412             if (fileline[0] == '#') continue;
00413             if (fileline[0] == '\0') continue;
00414 
00415             esp_print_decode_onesecret(ndo, fileline, filename, lineno);
00416         }
00417         fclose(secretfile);
00418 
00419         return;
00420     }
00421 
00422     if (spikey && strcasecmp(spikey, "ikev2") == 0) {
00423         esp_print_decode_ikeline(ndo, line, file, lineno);
00424         return;
00425     } 
00426 
00427     if (spikey) {
00428         
00429         char *spistr, *foo;
00430         u_int32_t spino;
00431         struct sockaddr_in *sin;
00432 #ifdef INET6
00433         struct sockaddr_in6 *sin6;
00434 #endif
00435         
00436         spistr = strsep(&spikey, "@");
00437         
00438         spino = strtoul(spistr, &foo, 0);
00439         if (spistr == foo || !spikey) {
00440             (*ndo->ndo_warning)(ndo, "print_esp: failed to decode spi# %s\n", foo);
00441             return;
00442         }
00443         
00444         sa1.spi = spino;
00445         
00446         sin = (struct sockaddr_in *)&sa1.daddr;
00447 #ifdef INET6
00448         sin6 = (struct sockaddr_in6 *)&sa1.daddr;

00450 #ifdef HAVE_SOCKADDR_SA_LEN
00451             sin6->sin6_len = sizeof(struct sockaddr_in6);
00452 #endif
00453             sin6->sin6_family = AF_INET6;
00454         } else
00455 #endif
00456             if (inet_pton(AF_INET, spikey, &sin->sin_addr) == 1) {
00457 #ifdef HAVE_SOCKADDR_SA_LEN
00458                 sin->sin_len = sizeof(struct sockaddr_in);
00459 #endif
00460                 sin->sin_family = AF_INET;
00461             } else {
00462                 (*ndo->ndo_warning)(ndo, "print_esp: can not decode IP# %s\n", spikey);
00463                 return;
00464             }
00465     }
00466 
00467     if (decode) {
00468         /* skip any blank spaces */
00469         while (isspace((unsigned char)*decode))
00470             decode++;
00471         
00472         if(!espprint_decode_encalgo(ndo, decode, &sa1)) {
00473             return;
00474         }
00475     }

00573         (*ndo->ndo_printf)(ndo, ", length %u", length);

00580         if (!ndo->ndo_espsecret)

00587         goto fail;

00591 #ifdef INET6

00608             }
00609         }

00611 #endif /*INET6*/

00613         /* nexthdr & padding are in the last fragment */
00614         if (EXTRACT_16BITS(&ip->ip_off) & IP_MF)
00615             goto fail;

00617 
00618         /* see if we can find the SA, and if so, decode it */
00619         for (sa = ndo->ndo_sa_list_head; sa != NULL; sa = sa->next) {
00620             struct sockaddr_in *sin = (struct sockaddr_in *)&sa->daddr;
00621             if (sa->spi == EXTRACT_32BITS(&esp->esp_spi) &&
00622                 sin->sin_family == AF_INET &&
00623                 sin->sin_addr.s_addr == ip->ip_dst.s_addr) {
00624                 break;
00625             }

00641

00648     }

00651     ivlen = sa->ivlen;
00652     secret = sa->secret;
00653     espsecret_keylen = sa->secretlen;
00654     ep = ep - sa->authlen;
00655
