00214 {
00215     struct e1000_phy_info *phy = &hw->phy;
00216     u32 i, mdic = 0;
00217     s32 ret_val = E1000_SUCCESS;
00218 
00219     DEBUGFUNC("e1000_read_phy_reg_mdic");
00220 
00221     if (offset > MAX_PHY_REG_ADDRESS) {
00222         DEBUGOUT1("PHY Address %d is out of range\n", offset);
00223         return -E1000_ERR_PARAM;
00224     }
00225 
00226     /*
00227      * Set up Op-code, Phy Address, and register offset in the MDI
00228      * Control register.  The MAC will take care of interfacing with the
00229      * PHY to retrieve the desired data.
00230      */
00231     mdic = ((offset << E1000_MDIC_REG_SHIFT) |
00232             (phy->addr << E1000_MDIC_PHY_SHIFT) |
00233             (E1000_MDIC_OP_READ));
00234 
00235     E1000_WRITE_REG(hw, E1000_MDIC, mdic);
00236 
00237     /*
00238      * Poll the ready bit to see if the MDI read completed
00239      * Increasing the time out as testing showed failures with
00240      * the lower time out
00241      */

00243         usec_delay(50);

00246             break;

00248     if (!(mdic & E1000_MDIC_READY)) {
00249         DEBUGOUT("MDI Read did not complete\n");
00250         ret_val = -E1000_ERR_PHY;
00251         goto out;
00252     }
00253     if (mdic & E1000_MDIC_ERROR) {
00254         DEBUGOUT("MDI Error\n");
00255         ret_val = -E1000_ERR_PHY;

00683 s32 e1000_copper_link_setup_m88(struct e1000_hw *hw)
00684 {
00685     struct e1000_phy_info *phy = &hw->phy;
00686     s32 ret_val;
00687     u16 phy_data;
00688 
00689     DEBUGFUNC("e1000_copper_link_setup_m88");
00690 
00691     if (phy->reset_disable) {
00692         ret_val = E1000_SUCCESS;
00693         goto out;
00694     }
00695 
00696     /* Enable CRS on Tx. This must be set for half-duplex operation. */
00697     ret_val = phy->ops.read_reg(hw, M88E1000_PHY_SPEC_CTRL, &phy_data);
00698     if (ret_val)
00699         goto out;
00700 
00701     phy_data |= M88E1000_PSCR_ASSERT_CRS_ON_TX;
00702 
00703     /*
00704      * Options:
00705      *   MDI/MDI-X = 0 (default)
00706      *   0 - Auto for all speeds
00707      *   1 - MDI mode
00708      *   2 - MDI-X mode
00709      *   3 - Auto for 1000Base-T only (MDI-X for 10/100Base-T modes)
00710      */
00711     phy_data &= ~M88E1000_PSCR_AUTO_X_MODE;
00712 
00713     switch (phy->mdix) {
00714     case 1:
00715         phy_data |= M88E1000_PSCR_MDI_MANUAL_MODE;
00716         break;
00717     case 2:
00718         phy_data |= M88E1000_PSCR_MDIX_MANUAL_MODE;
00719         break;
00720     case 3:
00721         phy_data |= M88E1000_PSCR_AUTO_X_1000T;
00722         break;
00723     case 0:
00724     default:
00725         phy_data |= M88E1000_PSCR_AUTO_X_MODE;
00726         break;
00727     }
00728 
00729     /*
00730      * Options:
00731      *   disable_polarity_correction = 0 (default)
00732      *       Automatic Correction for Reversed Cable Polarity
00733      *   0 - Disabled
00734      *   1 - Enabled
00735      */
00736     phy_data &= ~M88E1000_PSCR_POLARITY_REVERSAL;
00737     if (phy->disable_polarity_correction == 1)
00738         phy_data |= M88E1000_PSCR_POLARITY_REVERSAL;
00739 
00740     ret_val = phy->ops.write_reg(hw, M88E1000_PHY_SPEC_CTRL, phy_data);
00741     if (ret_val)
00742         goto out;
00743 
00744     if (phy->revision < E1000_REVISION_4) {
00745         /*
00746          * Force TX_CLK in the Extended PHY Specific Control Register
00747          * to 25MHz clock.
00748          */
00749         ret_val = phy->ops.read_reg(hw, M88E1000_EXT_PHY_SPEC_CTRL,
00750                                      &phy_data);
00751         if (ret_val)
00752             goto out;
00753 
00754         phy_data |= M88E1000_EPSCR_TX_CLK_25;
00755 
00756         if ((phy->revision == E1000_REVISION_2) &&
00757             (phy->id == M88E1111_I_PHY_ID)) {
00758             /* 82573L PHY - set the downshift counter to 5x. */
00759             phy_data &= ~M88EC018_EPSCR_DOWNSHIFT_COUNTER_MASK;
00760             phy_data |= M88EC018_EPSCR_DOWNSHIFT_COUNTER_5X;
00761         } else {
00762             /* Configure Master and Slave downshift values */
00763             phy_data &= ~(M88E1000_EPSCR_MASTER_DOWNSHIFT_MASK |
00764                          M88E1000_EPSCR_SLAVE_DOWNSHIFT_MASK);
00765             phy_data |= (M88E1000_EPSCR_MASTER_DOWNSHIFT_1X |
00766                          M88E1000_EPSCR_SLAVE_DOWNSHIFT_1X);
00767         }
00768         ret_val = phy->ops.write_reg(hw, M88E1000_EXT_PHY_SPEC_CTRL,
00769                                      phy_data);
00770         if (ret_val)
00771             goto out;
00772     }
00773 
00774     /* Commit the changes. */
00775     ret_val = phy->ops.commit(hw);
00776     if (ret_val) {
00777         DEBUGOUT("Error committing the PHY changes\n");
00778         goto out;
00779     }

00781 out:
00782     return ret_val;

01006 s32 e1000_phy_setup_autoneg(struct e1000_hw *hw)
01007 {
01008     struct e1000_phy_info *phy = &hw->phy;
01009     s32 ret_val;
01010     u16 mii_autoneg_adv_reg;
01011     u16 mii_1000t_ctrl_reg = 0;
01012 
01013     DEBUGFUNC("e1000_phy_setup_autoneg");
01014 
01015     phy->autoneg_advertised &= phy->autoneg_mask;
01016 
01017     /* Read the MII Auto-Neg Advertisement Register (Address 4). */
01018     ret_val = phy->ops.read_reg(hw, PHY_AUTONEG_ADV, &mii_autoneg_adv_reg);
01019     if (ret_val)
01020         goto out;
01021 
01022     if (phy->autoneg_mask & ADVERTISE_1000_FULL) {
01023         /* Read the MII 1000Base-T Control Register (Address 9). */
01024         ret_val = phy->ops.read_reg(hw, PHY_1000T_CTRL,
01025                                     &mii_1000t_ctrl_reg);
01026         if (ret_val)
01027             goto out;
01028     }
01029 
01030     /*
01031      * Need to parse both autoneg_advertised and fc and set up
01032      * the appropriate PHY registers.  First we will parse for
01033      * autoneg_advertised software override.  Since we can advertise
01034      * a plethora of combinations, we need to check each bit
01035      * individually.
01036      */
01037 
01038     /*
01039      * First we clear all the 10/100 mb speed bits in the Auto-Neg
01040      * Advertisement Register (Address 4) and the 1000 mb speed bits in
01041      * the  1000Base-T Control Register (Address 9).
01042      */
01043     mii_autoneg_adv_reg &= ~(NWAY_AR_100TX_FD_CAPS |
01044                              NWAY_AR_100TX_HD_CAPS |
01045                              NWAY_AR_10T_FD_CAPS   |
01046                              NWAY_AR_10T_HD_CAPS);
01047     mii_1000t_ctrl_reg &= ~(CR_1000T_HD_CAPS | CR_1000T_FD_CAPS);
01048 
01049     DEBUGOUT1("autoneg_advertised %x\n", phy->autoneg_advertised);
01050 
01051     /* Do we want to advertise 10 Mb Half Duplex? */
01052     if (phy->autoneg_advertised & ADVERTISE_10_HALF) {
01053         DEBUGOUT("Advertise 10mb Half duplex\n");
01054         mii_autoneg_adv_reg |= NWAY_AR_10T_HD_CAPS;
01055     }
01056 
01057     /* Do we want to advertise 10 Mb Full Duplex? */
01058     if (phy->autoneg_advertised & ADVERTISE_10_FULL) {
01059         DEBUGOUT("Advertise 10mb Full duplex\n");
01060         mii_autoneg_adv_reg |= NWAY_AR_10T_FD_CAPS;
01061     }
01062 
01063     /* Do we want to advertise 100 Mb Half Duplex? */
01064     if (phy->autoneg_advertised & ADVERTISE_100_HALF) {
01065         DEBUGOUT("Advertise 100mb Half duplex\n");
01066         mii_autoneg_adv_reg |= NWAY_AR_100TX_HD_CAPS;
01067     }
01068 
01069     /* Do we want to advertise 100 Mb Full Duplex? */
01070     if (phy->autoneg_advertised & ADVERTISE_100_FULL) {
01071         DEBUGOUT("Advertise 100mb Full duplex\n");
01072         mii_autoneg_adv_reg |= NWAY_AR_100TX_FD_CAPS;
01073     }
01074 
01075     /* We do not allow the Phy to advertise 1000 Mb Half Duplex */
01076     if (phy->autoneg_advertised & ADVERTISE_1000_HALF)
01077         DEBUGOUT("Advertise 1000mb Half duplex request denied!\n");
01078 
01079     /* Do we want to advertise 1000 Mb Full Duplex? */
01080     if (phy->autoneg_advertised & ADVERTISE_1000_FULL) {
01081         DEBUGOUT("Advertise 1000mb Full duplex\n");
01082         mii_1000t_ctrl_reg |= CR_1000T_FD_CAPS;
01083     }
01084 
01085     /*
01086      * Check for a software override of the flow control settings, and
01087      * setup the PHY advertisement registers accordingly.  If
01088      * auto-negotiation is enabled, then software will have to set the
01089      * "PAUSE" bits to the correct value in the Auto-Negotiation
01090      * Advertisement Register (PHY_AUTONEG_ADV) and re-start auto-
01091      * negotiation.
01092      *
01093      * The possible values of the "fc" parameter are:
01094      *      0:  Flow control is completely disabled
01095      *      1:  Rx flow control is enabled (we can receive pause frames
01096      *          but not send pause frames).
01097      *      2:  Tx flow control is enabled (we can send pause frames
01098      *          but we do not support receiving pause frames).
01099      *      3:  Both Rx and Tx flow control (symmetric) are enabled.
01100      *  other:  No software override.  The flow control configuration
01101      *          in the EEPROM is used.
01102      */
01103     switch (hw->fc.current_mode) {
01104     case e1000_fc_none:
01105         /*
01106          * Flow control (Rx & Tx) is completely disabled by a
01107          * software over-ride.
01108          */
01109         mii_autoneg_adv_reg &= ~(NWAY_AR_ASM_DIR | NWAY_AR_PAUSE);
01110         break;
01111     case e1000_fc_rx_pause:
01112         /*
01113          * Rx Flow control is enabled, and Tx Flow control is
01114          * disabled, by a software over-ride.
01115          *
01116          * Since there really isn't a way to advertise that we are
01117          * capable of Rx Pause ONLY, we will advertise that we
01118          * support both symmetric and asymmetric Rx PAUSE.  Later
01119          * (in e1000_config_fc_after_link_up) we will disable the
01120          * hw's ability to send PAUSE frames.
01121          */
01122         mii_autoneg_adv_reg |= (NWAY_AR_ASM_DIR | NWAY_AR_PAUSE);
01123         break;
01124     case e1000_fc_tx_pause:
01125         /*
01126          * Tx Flow control is enabled, and Rx Flow control is
01127          * disabled, by a software over-ride.
01128          */
01129         mii_autoneg_adv_reg |= NWAY_AR_ASM_DIR;
01130         mii_autoneg_adv_reg &= ~NWAY_AR_PAUSE;
01131         break;
01132     case e1000_fc_full:
01133         /*
01134          * Flow control (both Rx and Tx) is enabled by a software
01135          * over-ride.
01136          */
01137         mii_autoneg_adv_reg |= (NWAY_AR_ASM_DIR | NWAY_AR_PAUSE);
01138         break;
01139     default:
01140         DEBUGOUT("Flow control param set incorrectly\n");
01141         ret_val = -E1000_ERR_CONFIG;
01142         goto out;
01143     }
01144 
01145     ret_val = phy->ops.write_reg(hw, PHY_AUTONEG_ADV, mii_autoneg_adv_reg);
01146     if (ret_val)
01147         goto out;
01148 
01149     DEBUGOUT1("Auto-Neg Advertising %x\n", mii_autoneg_adv_reg);
01150 
01151     if (phy->autoneg_mask & ADVERTISE_1000_FULL) {
01152         ret_val = phy->ops.write_reg(hw,
01153                                       PHY_1000T_CTRL,
01154                                       mii_1000t_ctrl_reg);
01155         if (ret_val)
01156             goto out;
01157     }
01158 
01159 out:
01160     return ret_val;
01161 }
