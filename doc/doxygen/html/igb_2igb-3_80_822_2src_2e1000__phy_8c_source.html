01167      * individually.

01184     if (phy->autoneg_advertised & ADVERTISE_10_HALF) {
01185         DEBUGOUT("Advertise 10mb Half duplex\n");
01186         mii_autoneg_adv_reg |= NWAY_AR_10T_HD_CAPS;
01187     }
01188 
01189     /* Do we want to advertise 10 Mb Full Duplex? */
01190     if (phy->autoneg_advertised & ADVERTISE_10_FULL) {
01191         DEBUGOUT("Advertise 10mb Full duplex\n");
01192         mii_autoneg_adv_reg |= NWAY_AR_10T_FD_CAPS;
01193     }
01194 
01195     /* Do we want to advertise 100 Mb Half Duplex? */
01196     if (phy->autoneg_advertised & ADVERTISE_100_HALF) {
01197         DEBUGOUT("Advertise 100mb Half duplex\n");
01198         mii_autoneg_adv_reg |= NWAY_AR_100TX_HD_CAPS;
01199     }
01200 
01201     /* Do we want to advertise 100 Mb Full Duplex? */
01202     if (phy->autoneg_advertised & ADVERTISE_100_FULL) {
01203         DEBUGOUT("Advertise 100mb Full duplex\n");
01204         mii_autoneg_adv_reg |= NWAY_AR_100TX_FD_CAPS;
01205     }
01206 
01207     /* We do not allow the Phy to advertise 1000 Mb Half Duplex */
01208     if (phy->autoneg_advertised & ADVERTISE_1000_HALF)
01209         DEBUGOUT("Advertise 1000mb Half duplex request denied!\n");
01210 
01211     /* Do we want to advertise 1000 Mb Full Duplex? */
01212     if (phy->autoneg_advertised & ADVERTISE_1000_FULL) {
01213         DEBUGOUT("Advertise 1000mb Full duplex\n");
01214         mii_1000t_ctrl_reg |= CR_1000T_FD_CAPS;
01215     }
01216 
01217     /*
01218      * Check for a software override of the flow control settings, and
01219      * setup the PHY advertisement registers accordingly.  If
01220      * auto-negotiation is enabled, then software will have to set the
01221      * "PAUSE" bits to the correct value in the Auto-Negotiation
01222      * Advertisement Register (PHY_AUTONEG_ADV) and re-start auto-
01223      * negotiation.
01224      *
01225      * The possible values of the "fc" parameter are:
01226      *      0:  Flow control is completely disabled

01228      *          but not send pause frames).
01229      *      2:  Tx flow control is enabled (we can send pause frames
01230      *          but we do not support receiving pause frames).

01232      *  other:  No software override.  The flow control configuration
01233      *          in the EEPROM is used.
01234      */

01236     case e1000_fc_none:

01238          * Flow control (Rx & Tx) is completely disabled by a
01239          * software over-ride.

01241         mii_autoneg_adv_reg &= ~(NWAY_AR_ASM_DIR | NWAY_AR_PAUSE);
01242         break;
01243     case e1000_fc_rx_pause:

01245          * Rx Flow control is enabled, and Tx Flow control is
01246          * disabled, by a software over-ride.
01247          *
01248          * Since there really isn't a way to advertise that we are
01249          * capable of Rx Pause ONLY, we will advertise that we
01250          * support both symmetric and asymmetric Rx PAUSE.  Later
01251          * (in e1000_config_fc_after_link_up) we will disable the

01253          */
01254         mii_autoneg_adv_reg |= (NWAY_AR_ASM_DIR | NWAY_AR_PAUSE);
01255         break;
01256     case e1000_fc_tx_pause:

01258          * Tx Flow control is enabled, and Rx Flow control is
01259          * disabled, by a software over-ride.
01260          */
01261         mii_autoneg_adv_reg |= NWAY_AR_ASM_DIR;
01262         mii_autoneg_adv_reg &= ~NWAY_AR_PAUSE;
01263         break;
01264     case e1000_fc_full:
01265         /*
01266          * Flow control (both Rx and Tx) is enabled by a software
01267          * over-ride.
01268          */
01269         mii_autoneg_adv_reg |= (NWAY_AR_ASM_DIR | NWAY_AR_PAUSE);
01270         break;
01271     default:
01272         DEBUGOUT("Flow control param set incorrectly\n");
01273         ret_val = -E1000_ERR_CONFIG;
01274         goto out;
01275     }
01276

01278     if (ret_val)
01279         goto out;
01280 
01281     DEBUGOUT1("Auto-Neg Advertising %x\n", mii_autoneg_adv_reg);
01282

01284         ret_val = phy->ops.write_reg(hw,
01285                                       PHY_1000T_CTRL,
01286                                       mii_1000t_ctrl_reg);
01287         if (ret_val)
01288             goto out;
01289     }

01291 out:
01292     return ret_val;
