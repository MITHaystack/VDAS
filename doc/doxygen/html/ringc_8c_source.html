00089 /* Print out warning information in the event packet counter
00090  * maintained by the driver differ from those counted by the application,
00091  * then print out pkts/secs from the start of the program and in the latest time interval
00092  */
00093 static void statistics (void)
00094 {
00095   static unsigned hits = 0;
00096   static u_int64_t previous = 0;         /* # of packets received during last period */
00097   static struct timeval latest;
00098 
00099   struct timeval now;
00100   double delta;
00101 
00102   pfring_stat pstat;
00103 
00104   /* Print statistics information from PF_RING only they differ from those currently maintained by the application */
00105   if (pfring_stats (ring, & pstat) >= 0 && partial != pstat.recv)
00106     {
00107       if (pstat.drop)
00108     printf ("Warning: received by PF_RING #%llu received by the application #%llu [dropped #%llu (diff #%llu) (%%%.1f)]\n",
00109         (long long unsigned int)pstat.recv, partial, 
00110         (long long unsigned int)pstat.drop, 
00111         (long long unsigned int)(pstat.recv - pstat.drop),
00112         (!pstat.recv) ? 0 : (double) (pstat.drop * 100 / pstat.recv));
00113       else
00114     printf ("Warning: counted by PF_RING #%llu differ from those received by the application #%llu\n", 
00115         (long long unsigned int)pstat.recv, partial);
00116     }
00117 
00118   /* Calculate the time interval from the start of the capture */
00119   gettimeofday (& now, NULL);
00120   delta = delta_time_in_milliseconds (& now, & started);
00121 
00122   /* Show pkts/secs in the latest period */
00123   printf ("[%3u] pkts rcvd #%5llu => %.2f pkts/sec in %s ",
00124       ++ hits, partial, (double) (partial * 1000 / delta), elapsed_time (& started, & now));
00125 
00126   if (previous)
00127     delta = delta_time_in_milliseconds (& now, & latest),
00128       printf ("[+%llu => %.2f pkts/sec in %s]",
00129           partial - previous, (double) ((partial - previous) * 1000 / delta), elapsed_time (& latest, & now));
00130   printf ("\n");
00131 
00132   previous = partial;
00133   latest = now;
00134 }
