00046 /*
00047  * the frame relay header has a variable length
00048  *
00049  * the EA bit determines if there is another byte
00050  * in the header
00051  *
00052  * minimum header length is 2 bytes
00053  * maximum header length is 4 bytes
00054  *
00055  *      7    6    5    4    3    2    1    0
00056  *    +----+----+----+----+----+----+----+----+
00057  *    |        DLCI (6 bits)        | CR | EA |
00058  *    +----+----+----+----+----+----+----+----+
00059  *    |   DLCI (4 bits)   |FECN|BECN| DE | EA |
00060  *    +----+----+----+----+----+----+----+----+
00061  *    |           DLCI (7 bits)          | EA |
00062  *    +----+----+----+----+----+----+----+----+
00063  *    |        DLCI (6 bits)        |SDLC| EA |
00064  *    +----+----+----+----+----+----+----+----+
00065  */

00100 /* Finds out Q.922 address length, DLCI and flags. Returns 0 on success
00101  * save the flags dep. on address length
00102  */
00103 static int parse_q922_addr(const u_char *p, u_int *dlci,
00104                            u_int *addr_len, u_int8_t *flags)
00105 {
00106     if ((p[0] & FR_EA_BIT))
00107         return -1;
00108 
00109     *addr_len = 2;
00110     *dlci = ((p[0] & 0xFC) << 2) | ((p[1] & 0xF0) >> 4);
00111 
00112         flags[0] = p[0] & 0x02; /* populate the first flag fields */
00113         flags[1] = p[1] & 0x0c;
00114         flags[2] = 0;           /* clear the rest of the flags */
00115         flags[3] = 0;
00116 
00117     if (p[1] & FR_EA_BIT)
00118         return 0;   /* 2-byte Q.922 address */
00119 
00120     p += 2;
00121     (*addr_len)++;      /* 3- or 4-byte Q.922 address */
00122     if ((p[0] & FR_EA_BIT) == 0) {
00123         *dlci = (*dlci << 7) | (p[0] >> 1);
00124         (*addr_len)++;  /* 4-byte Q.922 address */
00125         p++;
00126     }
00127 
00128     if ((p[0] & FR_EA_BIT) == 0)
00129         return -1; /* more than 4 bytes of Q.922 address? */
00130 
00131         flags[3] = p[0] & 0x02;
00132 
00133         *dlci = (*dlci << 6) | (p[0] >> 2);
00134 
00135     return 0;
00136 }

00153 /* Frame Relay packet structure, with flags and CRC removed
00154 
00155                   +---------------------------+
00156                   |       Q.922 Address*      |
00157                   +--                       --+
00158                   |                           |
00159                   +---------------------------+
00160                   | Control (UI = 0x03)       |
00161                   +---------------------------+
00162                   | Optional Pad      (0x00)  |
00163                   +---------------------------+
00164                   | NLPID                     |
00165                   +---------------------------+
00166                   |             .             |
00167                   |             .             |
00168                   |             .             |
00169                   |           Data            |
00170                   |             .             |
00171                   |             .             |
00172                   +---------------------------+
00173 
00174            * Q.922 addresses, as presently defined, are two octets and
00175              contain a 10-bit DLCI.  In some networks Q.922 addresses
00176              may optionally be increased to three or four octets.
00177 */

00181 {
00182     if (!p[addr_len + 1] /* pad exist */)
00183         return addr_len + 1 /* UI */ + 1 /* pad */ + 1 /* NLPID */;
00184     else 
00185         return addr_len + 1 /* UI */ + 1 /* NLPID */;
00186 }

00232 u_int
00233 fr_print(register const u_char *p, u_int length)
00234 {
00235     u_int16_t extracted_ethertype;
00236     u_int dlci;
00237     u_int addr_len;
00238     u_int16_t nlpid;
00239     u_int hdr_len;
00240     u_int8_t flags[4];
00241 
00242     if (parse_q922_addr(p, &dlci, &addr_len, flags)) {
00243         printf("Q.922, invalid address");
00244         return 0;
00245     }
00246 
00247         TCHECK2(*p,addr_len+1+1);
00248     hdr_len = fr_hdrlen(p, addr_len);
00249         TCHECK2(*p,hdr_len);
00250 
00251     if (p[addr_len] != 0x03 && dlci != 0) {
00252 
00253                 /* lets figure out if we have cisco style encapsulation: */
00254                 extracted_ethertype = EXTRACT_16BITS(p+addr_len);
00255 
00256                 if (eflag)
00257                     fr_hdr_print(length, addr_len, dlci, flags, extracted_ethertype);
00258 
00259                 if (ethertype_print(extracted_ethertype,
00260                                       p+addr_len+ETHERTYPE_LEN,
00261                                       length-addr_len-ETHERTYPE_LEN,
00262                                       length-addr_len-ETHERTYPE_LEN) == 0)
00263                     /* ether_type not known, probably it wasn't one */
00264                     printf("UI %02x! ", p[addr_len]);
00265                 else
00266                     return hdr_len;
00267         }
00268 
00269     if (!p[addr_len + 1]) { /* pad byte should be used with 3-byte Q.922 */
00270         if (addr_len != 3)
00271             printf("Pad! ");
00272     } else if (addr_len == 3)
00273         printf("No pad! ");
00274 
00275     nlpid = p[hdr_len - 1];
00276 
00277     if (eflag)
00278         fr_hdr_print(length, addr_len, dlci, flags, nlpid);
00279     p += hdr_len;
00280     length -= hdr_len;
00281 
00282     switch (nlpid) {
00283     case NLPID_IP:
00284             ip_print(gndo, p, length);
00285         break;
00286 
00287 #ifdef INET6
00288     case NLPID_IP6:
00289         ip6_print(p, length);
00290         break;
00291 #endif
00292     case NLPID_CLNP:
00293     case NLPID_ESIS:
00294     case NLPID_ISIS:
00295                 isoclns_print(p-1, length+1, length+1); /* OSI printers need the NLPID field */
00296         break;
00297 
00298     case NLPID_SNAP:
00299         if (snap_print(p, length, length, 0) == 0) {
00300             /* ether_type not known, print raw packet */
00301                         if (!eflag)
00302                             fr_hdr_print(length + hdr_len, hdr_len,
00303                                          dlci, flags, nlpid);
00304             if (!suppress_default_print)
00305                             default_print(p - hdr_len, length + hdr_len);
00306         }
00307         break;
00308 
00309         case NLPID_Q933:
00310         q933_print(p, length);
00311         break;
00312 
00313         case NLPID_MFR:
00314                 frf15_print(p, length);
00315                 break;
00316 
00317         case NLPID_PPP:
00318                 ppp_print(p, length);
00319                 break;
00320 
00321     default:
00322         if (!eflag)
00323                     fr_hdr_print(length + hdr_len, addr_len,
00324                      dlci, flags, nlpid);
00325         if (!xflag)
00326             default_print(p, length);
00327     }
00328 
00329     return hdr_len;
00330 
00331  trunc:
00332         printf("[|fr]");
00333         return 0;
00334

00398 u_int
00399 mfr_print(register const u_char *p, u_int length)
00400 {
00401     u_int tlen,idx,hdr_len = 0;
00402     u_int16_t sequence_num;
00403     u_int8_t ie_type,ie_len;
00404     const u_int8_t *tptr;
00405 
00406 
00407 /*
00408  * FRF.16 Link Integrity Control Frame
00409  * 
00410  *      7    6    5    4    3    2    1    0
00411  *    +----+----+----+----+----+----+----+----+
00412  *    | B  | E  | C=1| 0    0    0    0  | EA |
00413  *    +----+----+----+----+----+----+----+----+
00414  *    | 0    0    0    0    0    0    0    0  |
00415  *    +----+----+----+----+----+----+----+----+
00416  *    |              message type             |
00417  *    +----+----+----+----+----+----+----+----+
00418  */
00419 
00420     TCHECK2(*p, 4); /* minimum frame header length */
00421 
00422     if ((p[0] & MFR_BEC_MASK) == MFR_CTRL_FRAME && p[1] == 0) {
00423         printf("FRF.16 Control, Flags [%s], %s, length %u",
00424                bittok2str(frf_flag_values,"none",(p[0] & MFR_BEC_MASK)),
00425                tok2str(mfr_ctrl_msg_values,"Unknown Message (0x%02x)",p[2]),
00426                length);
00427         tptr = p + 3;
00428         tlen = length -3;
00429         hdr_len = 3;
00430 
00431         if (!vflag)
00432             return hdr_len;
00433 
00434         while (tlen>sizeof(struct ie_tlv_header_t)) {
00435             TCHECK2(*tptr, sizeof(struct ie_tlv_header_t));
00436             ie_type=tptr[0];
00437             ie_len=tptr[1];
00438 
00439             printf("\n\tIE %s (%u), length %u: ",
00440                    tok2str(mfr_ctrl_ie_values,"Unknown",ie_type),
00441                    ie_type,
00442                    ie_len);
00443 
00444             /* infinite loop check */
00445             if (ie_type == 0 || ie_len <= sizeof(struct ie_tlv_header_t))
00446                 return hdr_len;
00447 
00448             TCHECK2(*tptr,ie_len);
00449             tptr+=sizeof(struct ie_tlv_header_t);
00450             /* tlv len includes header */
00451             ie_len-=sizeof(struct ie_tlv_header_t);
00452             tlen-=sizeof(struct ie_tlv_header_t);
00453 
00454             switch (ie_type) {
00455 
00456             case MFR_CTRL_IE_MAGIC_NUM:
00457                 printf("0x%08x",EXTRACT_32BITS(tptr));
00458                 break;
00459 
00460             case MFR_CTRL_IE_BUNDLE_ID: /* same message format */
00461             case MFR_CTRL_IE_LINK_ID:
00462                 for (idx = 0; idx < ie_len && idx < MFR_ID_STRING_MAXLEN; idx++) {
00463                     if (*(tptr+idx) != 0) /* don't print null termination */
00464                         safeputchar(*(tptr+idx));
00465                     else
00466                         break;
00467                 }
00468                 break;
00469 
00470             case MFR_CTRL_IE_TIMESTAMP:
00471                 if (ie_len == sizeof(struct timeval)) {
00472                     ts_print((const struct timeval *)tptr);
00473                     break;
00474                 }
00475                 /* fall through and hexdump if no unix timestamp */
00476 
00477                 /*
00478                  * FIXME those are the defined IEs that lack a decoder
00479                  * you are welcome to contribute code ;-)
00480                  */
00481 
00482             case MFR_CTRL_IE_VENDOR_EXT:
00483             case MFR_CTRL_IE_CAUSE:
00484 
00485             default:
00486                 if (vflag <= 1)
00487                     print_unknown_data(tptr,"\n\t  ",ie_len);
00488                 break;
00489             }
00490 
00491             /* do we want to see a hexdump of the IE ? */
00492             if (vflag > 1 )
00493                 print_unknown_data(tptr,"\n\t  ",ie_len);
00494             
00495             tlen-=ie_len;
00496             tptr+=ie_len;
00497         }
00498         return hdr_len;
00499     }
00500 /*
00501  * FRF.16 Fragmentation Frame
00502  * 
00503  *      7    6    5    4    3    2    1    0
00504  *    +----+----+----+----+----+----+----+----+
00505  *    | B  | E  | C=0|seq. (high 4 bits) | EA  |
00506  *    +----+----+----+----+----+----+----+----+
00507  *    |        sequence  (low 8 bits)         |
00508  *    +----+----+----+----+----+----+----+----+
00509  *    |        DLCI (6 bits)        | CR | EA  |
00510  *    +----+----+----+----+----+----+----+----+
00511  *    |   DLCI (4 bits)   |FECN|BECN| DE | EA |
00512  *    +----+----+----+----+----+----+----+----+
00513  */
00514 
00515     sequence_num = (p[0]&0x1e)<<7 | p[1];
00516     /* whole packet or first fragment ? */
00517     if ((p[0] & MFR_BEC_MASK) == MFR_FRAG_FRAME ||
00518         (p[0] & MFR_BEC_MASK) == MFR_B_BIT) {
00519         printf("FRF.16 Frag, seq %u, Flags [%s], ",
00520                sequence_num,
00521                bittok2str(frf_flag_values,"none",(p[0] & MFR_BEC_MASK)));
00522         hdr_len = 2;
00523         fr_print(p+hdr_len,length-hdr_len);
00524         return hdr_len;
00525     }
00526 
00527     /* must be a middle or the last fragment */
00528     printf("FRF.16 Frag, seq %u, Flags [%s]",
00529            sequence_num,
00530            bittok2str(frf_flag_values,"none",(p[0] & MFR_BEC_MASK)));
00531     print_unknown_data(p,"\n\t",length);
00532 
00533     return hdr_len;
00534 
00535  trunc:
00536     printf("[|mfr]");
00537     return length;
00538 }

00726 void
00727 q933_print(const u_char *p, u_int length)
00728 {
00729     const u_char *ptemp = p;
00730     struct ie_tlv_header_t  *ie_p;
00731         int olen;
00732     int is_ansi = 0;
00733         u_int codeset;
00734         u_int ie_is_known = 0;
00735 
00736     if (length < 9) {   /* shortest: Q.933a LINK VERIFY */
00737         printf("[|q.933]");
00738         return;
00739     }
00740 
00741         codeset = p[2]&0x0f;   /* extract the codeset */
00742 
00743     if (p[2] == MSG_ANSI_LOCKING_SHIFT) {
00744             is_ansi = 1;
00745     }
00746     
00747         printf("%s", eflag ? "" : "Q.933, ");
00748 
00749     /* printing out header part */
00750     printf("%s, codeset %u", is_ansi ? "ANSI" : "CCITT", codeset);
00751 
00752     if (p[0]) {
00753             printf(", Call Ref: 0x%02x", p[0]);
00754     }
00755         if (vflag) {
00756                 printf(", %s (0x%02x), length %u",
00757                tok2str(fr_q933_msg_values,
00758                    "unknown message", p[1]),
00759                p[1],
00760                length);
00761         } else {
00762                 printf(", %s",
00763                tok2str(fr_q933_msg_values,
00764                    "unknown message 0x%02x", p[1]));
00765     }
00766 
00767         olen = length; /* preserve the original length for non verbose mode */
00768 
00769     if (length < (u_int)(2 - is_ansi)) {
00770         printf("[|q.933]");
00771         return;
00772     }
00773     length -= 2 + is_ansi;
00774     ptemp += 2 + is_ansi;
00775     
00776     /* Loop through the rest of IE */
00777     while (length > sizeof(struct ie_tlv_header_t)) {
00778         ie_p = (struct ie_tlv_header_t  *)ptemp;
00779         if (length < sizeof(struct ie_tlv_header_t) ||
00780             length < sizeof(struct ie_tlv_header_t) + ie_p->ie_len) {
00781                     if (vflag) { /* not bark if there is just a trailer */
00782                         printf("\n[|q.933]");
00783                     } else {
00784                         printf(", length %u",olen);
00785             }
00786                     return;
00787         }
00788 
00789                 /* lets do the full IE parsing only in verbose mode
00790                  * however some IEs (DLCI Status, Link Verify)
00791                  * are also interestting in non-verbose mode */
00792                 if (vflag) {
00793                     printf("\n\t%s IE (0x%02x), length %u: ",
00794                            tok2str(fr_q933_ie_codesets[codeset],
00795                    "unknown", ie_p->ie_type),
00796                            ie_p->ie_type,
00797                            ie_p->ie_len);
00798         }
00799 
00800                 /* sanity check */
00801                 if (ie_p->ie_type == 0 || ie_p->ie_len == 0) {
00802                     return;
00803         }
00804 
00805                 if (fr_q933_print_ie_codeset[codeset] != NULL) {
00806                     ie_is_known = fr_q933_print_ie_codeset[codeset](ie_p, ptemp);
00807         }               
00808 
00809                 if (vflag >= 1 && !ie_is_known) {
00810                     print_unknown_data(ptemp+2,"\n\t",ie_p->ie_len);
00811         }
00812 
00813                 /* do we want to see a hexdump of the IE ? */
00814                 if (vflag> 1 && ie_is_known) {
00815                     print_unknown_data(ptemp+2,"\n\t  ",ie_p->ie_len);
00816         }
00817 
00818         length = length - ie_p->ie_len - 2;
00819         ptemp = ptemp + ie_p->ie_len + 2;
00820     }
00821         if (!vflag) {
00822             printf(", length %u",olen);
00823     }
00824 }

00826 static int
00827 fr_q933_print_ie_codeset5(const struct ie_tlv_header_t  *ie_p, const u_char *p)
00828 {
00829         u_int dlci;
00830 
00831         switch (ie_p->ie_type) {
00832 
00833         case FR_LMI_ANSI_REPORT_TYPE_IE: /* fall through */
00834         case FR_LMI_CCITT_REPORT_TYPE_IE:
00835             if (vflag) {
00836                 printf("%s (%u)",
00837                        tok2str(fr_lmi_report_type_ie_values,"unknown",p[2]),
00838                        p[2]);
00839         }
00840             return 1;
00841 
00842         case FR_LMI_ANSI_LINK_VERIFY_IE: /* fall through */
00843         case FR_LMI_CCITT_LINK_VERIFY_IE:
00844         case FR_LMI_ANSI_LINK_VERIFY_IE_91:
00845             if (!vflag) {
00846                 printf(", ");
00847         }
00848             printf("TX Seq: %3d, RX Seq: %3d", p[2], p[3]);
00849             return 1;
00850 
00851         case FR_LMI_ANSI_PVC_STATUS_IE: /* fall through */
00852         case FR_LMI_CCITT_PVC_STATUS_IE:
00853             if (!vflag) {
00854                 printf(", ");
00855         }
00856             /* now parse the DLCI information element. */                    
00857             if ((ie_p->ie_len < 3) ||
00858                 (p[2] & 0x80) ||
00859                 ((ie_p->ie_len == 3) && !(p[3] & 0x80)) ||
00860                 ((ie_p->ie_len == 4) && ((p[3] & 0x80) || !(p[4] & 0x80))) ||
00861                 ((ie_p->ie_len == 5) && ((p[3] & 0x80) || (p[4] & 0x80) ||
00862                                    !(p[5] & 0x80))) ||
00863                 (ie_p->ie_len > 5) ||
00864                 !(p[ie_p->ie_len + 1] & 0x80)) {
00865                 printf("Invalid DLCI IE");
00866         }
00867                     
00868             dlci = ((p[2] & 0x3F) << 4) | ((p[3] & 0x78) >> 3);
00869             if (ie_p->ie_len == 4) {
00870                 dlci = (dlci << 6) | ((p[4] & 0x7E) >> 1);
00871         }
00872             else if (ie_p->ie_len == 5) {
00873                 dlci = (dlci << 13) | (p[4] & 0x7F) | ((p[5] & 0x7E) >> 1);
00874         }
00875 
00876             printf("DLCI %u: status %s%s", dlci,
00877                     p[ie_p->ie_len + 1] & 0x8 ? "New, " : "",
00878                     p[ie_p->ie_len + 1] & 0x2 ? "Active" : "Inactive");
00879             return 1;
00880     }
00881 
00882         return 0;
