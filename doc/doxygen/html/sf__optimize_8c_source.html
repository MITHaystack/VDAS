00377 /*
00378  * Return the register number that is used by s.  If A and X are both
00379  * used, return AX_ATOM.  If no register is used, return -1.
00380  *
00381  * The implementation should probably change to an array access.
00382  */
00383 static int atomuse(s)
00384      struct stmt *s;
00385 {
00386     register int c = s->code;
00387 
00388     if (c == NOP)
00389         return -1;
00390 
00391     switch (BPF_CLASS(c))
00392     {
00393 
00394         case BPF_RET:
00395             return (BPF_RVAL(c) == BPF_A) ? A_ATOM : (BPF_RVAL(c) == BPF_X) ? X_ATOM : -1;
00396 
00397         case BPF_LD:
00398         case BPF_LDX:
00399             return (BPF_MODE(c) == BPF_IND) ? X_ATOM : (BPF_MODE(c) == BPF_MEM) ? s->k : -1;
00400 
00401         case BPF_ST:
00402             return A_ATOM;
00403 
00404         case BPF_STX:
00405             return X_ATOM;
00406 
00407         case BPF_JMP:
00408         case BPF_ALU:
00409             if (BPF_SRC(c) == BPF_X)
00410                 return AX_ATOM;
00411             return A_ATOM;
00412 
00413         case BPF_MISC:
00414             return BPF_MISCOP(c) == BPF_TXA ? X_ATOM : A_ATOM;
00415     }
00416     abort();
00417     /* NOTREACHED */
00418 }

00422  * a single stmt cannot define more than one register.  If no register

00425  * The implementation should probably change to an array access.

00431         return -1;

00440         case BPF_LDX:
00441             return X_ATOM;

00443         case BPF_ST:

00445             return s->k;
00446 
00447         case BPF_MISC:
00448             return BPF_MISCOP(s->code) == BPF_TAX ? X_ATOM : A_ATOM;

00453 /*
00454  * Compute the sets of registers used, defined, and killed by 'b'.
00455  *
00456  * "Used" means that a statement in 'b' uses the register before any
00457  * statement in 'b' defines it, i.e. it uses the value left in
00458  * that register by a predecessor block of this block.
00459  * "Defined" means that a statement in 'b' defines it.
00460  * "Killed" means that a statement in 'b' defines it before any
00461  * statement in 'b' uses it, i.e. it kills the value left in that
00462  * register by a predecessor block of this block.
00463  */
00464 static void compute_local_ud(b)
00465      struct block *b;
00466 {
00467     struct slist *s;
00468     atomset def = 0, use = 0, kill = 0;
00469     int atom;
00470 
00471     for (s = b->stmts; s; s = s->next)
00472     {
00473         if (s->s.code == NOP)
00474             continue;
00475         atom = atomuse(&s->s);
00476         if (atom >= 0)
00477         {
00478             if (atom == AX_ATOM)
00479             {
00480                 if (!ATOMELEM(def, X_ATOM))
00481                     use |= ATOMMASK(X_ATOM);
00482                 if (!ATOMELEM(def, A_ATOM))
00483                     use |= ATOMMASK(A_ATOM);
00484             }
00485             else if (atom < N_ATOMS)
00486             {
00487                 if (!ATOMELEM(def, atom))
00488                     use |= ATOMMASK(atom);
00489             }
00490             else
00491                 abort();
00492         }
00493         atom = atomdef(&s->s);
00494         if (atom >= 0)
00495         {
00496             if (!ATOMELEM(use, atom))
00497                 kill |= ATOMMASK(atom);
00498             def |= ATOMMASK(atom);
00499         }
00500     }
00501     if (BPF_CLASS(b->s.code) == BPF_JMP)
00502     {
00503         /*
00504          * XXX - what about RET?
00505          */
00506         atom = atomuse(&b->s);
00507         if (atom >= 0)
00508         {
00509             if (atom == AX_ATOM)
00510             {
00511                 if (!ATOMELEM(def, X_ATOM))
00512                     use |= ATOMMASK(X_ATOM);
00513                 if (!ATOMELEM(def, A_ATOM))
00514                     use |= ATOMMASK(A_ATOM);
00515             }
00516             else if (atom < N_ATOMS)
00517             {
00518                 if (!ATOMELEM(def, atom))
00519                     use |= ATOMMASK(atom);
00520             }
00521             else
00522                 abort();
00523         }
00524     }
00525 
00526     b->def = def;
00527     b->kill = kill;
00528     b->in_use = use;
00529 }

00535      struct block *root;

00557             p->in_use |= p->out_use & ~p->kill;

00602 /* Because we really don't have an IR, this stuff is a little messy. */
00603 static int F(code, v0, v1)
00604      int code;
00605      int v0, v1;
00606 {
00607     u_int hash;
00608     int val;
00609     struct valnode *p;
00610 
00611     hash = (u_int) code ^ (v0 << 4) ^ (v1 << 8);
00612     hash %= MODULUS;
00613 
00614     for (p = hashtbl[hash]; p; p = p->next)
00615         if (p->code == code && p->v0 == v0 && p->v1 == v1)
00616             return p->val;
00617 
00618     val = ++curval;
00619     if (BPF_MODE(code) == BPF_IMM && (BPF_CLASS(code) == BPF_LD || BPF_CLASS(code) == BPF_LDX))
00620     {
00621         vmap[val].const_val = v0;
00622         vmap[val].is_const = 1;
00623     }
00624     p = next_vnode++;
00625     p->val = val;
00626     p->code = code;
00627     p->v0 = v0;
00628     p->v1 = v1;
00629     p->next = hashtbl[hash];
00630     hashtbl[hash] = p;
00631 
00632     return val;

00721 static void opt_peep(b)
00722      struct block *b;
00723 {
00724     struct slist *s;
00725     struct slist *next, *last;
00726     int val;
00727 
00728     s = b->stmts;
00729     if (s == 0)
00730         return;
00731 
00732     last = s;
00733     for ( /*empty */ ; /*empty */ ; s = next)
00734     {
00735         /*
00736          * Skip over nops.
00737          */
00738         s = this_op(s);
00739         if (s == 0)
00740             break;              /* nothing left in the block */
00741 
00742         /*
00743          * Find the next real instruction after that one
00744          * (skipping nops).
00745          */
00746         next = this_op(s->next);
00747         if (next == 0)
00748             break;              /* no next instruction */
00749         last = next;
00750 
00751         /*
00752          * st  M[k] --> st  M[k]
00753          * ldx M[k]     tax
00754          */
00755         if (s->s.code == BPF_ST && next->s.code == (BPF_LDX | BPF_MEM) && s->s.k == next->s.k)
00756         {
00757             done = 0;
00758             next->s.code = BPF_MISC | BPF_TAX;
00759         }
00760         /*
00761          * ld  #k   --> ldx  #k
00762          * tax          txa
00763          */
00764         if (s->s.code == (BPF_LD | BPF_IMM) && next->s.code == (BPF_MISC | BPF_TAX))
00765         {
00766             s->s.code = BPF_LDX | BPF_IMM;
00767             next->s.code = BPF_MISC | BPF_TXA;
00768             done = 0;
00769         }
00770         /*
00771          * This is an ugly special case, but it happens
00772          * when you say tcp[k] or udp[k] where k is a constant.
00773          */
00774         if (s->s.code == (BPF_LD | BPF_IMM))
00775         {
00776             struct slist *add, *tax, *ild;
00777 
00778             /*
00779              * Check that X isn't used on exit from this
00780              * block (which the optimizer might cause).
00781              * We know the code generator won't generate
00782              * any local dependencies.
00783              */
00784             if (ATOMELEM(b->out_use, X_ATOM))
00785                 continue;
00786 
00787             /*
00788              * Check that the instruction following the ldi
00789              * is an addx, or it's an ldxms with an addx
00790              * following it (with 0 or more nops between the
00791              * ldxms and addx).
00792              */
00793             if (next->s.code != (BPF_LDX | BPF_MSH | BPF_B))
00794                 add = next;
00795             else
00796                 add = this_op(next->next);
00797             if (add == 0 || add->s.code != (BPF_ALU | BPF_ADD | BPF_X))
00798                 continue;
00799 
00800             /*
00801              * Check that a tax follows that (with 0 or more
00802              * nops between them).
00803              */
00804             tax = this_op(add->next);
00805             if (tax == 0 || tax->s.code != (BPF_MISC | BPF_TAX))
00806                 continue;
00807 
00808             /*
00809              * Check that an ild follows that (with 0 or more
00810              * nops between them).
00811              */
00812             ild = this_op(tax->next);
00813             if (ild == 0 || BPF_CLASS(ild->s.code) != BPF_LD || BPF_MODE(ild->s.code) != BPF_IND)
00814                 continue;
00815             /*
00816              * We want to turn this sequence:
00817              *
00818              * (004) ldi     #0x2       {s}
00819              * (005) ldxms   [14]       {next}  -- optional
00820              * (006) addx           {add}
00821              * (007) tax            {tax}
00822              * (008) ild     [x+0]      {ild}
00823              *
00824              * into this sequence:
00825              *
00826              * (004) nop
00827              * (005) ldxms   [14]
00828              * (006) nop
00829              * (007) nop
00830              * (008) ild     [x+2]
00831              *
00832              * XXX We need to check that X is not
00833              * subsequently used, because we want to change
00834              * what'll be in it after this sequence.
00835              *
00836              * We know we can eliminate the accumulator
00837              * modifications earlier in the sequence since
00838              * it is defined by the last stmt of this sequence
00839              * (i.e., the last statement of the sequence loads
00840              * a value into the accumulator, so we can eliminate
00841              * earlier operations on the accumulator).
00842              */
00843             ild->s.k += s->s.k;
00844             s->s.code = NOP;
00845             add->s.code = NOP;
00846             tax->s.code = NOP;
00847             done = 0;
00848         }
00849     }
00850     /*
00851      * If the comparison at the end of a block is an equality
00852      * comparison against a constant, and nobody uses the value
00853      * we leave in the A register at the end of a block, and
00854      * the operation preceding the comparison is an arithmetic
00855      * operation, we can sometime optimize it away.
00856      */
00857     if (b->s.code == (BPF_JMP | BPF_JEQ | BPF_K) && !ATOMELEM(b->out_use, A_ATOM))
00858     {
00859         /*
00860          * We can optimize away certain subtractions of the
00861          * X register.
00862          */
00863         if (last->s.code == (BPF_ALU | BPF_SUB | BPF_X))
00864         {
00865             val = b->val[X_ATOM];
00866             if (vmap[val].is_const)
00867             {
00868                 /*
00869                  * If we have a subtract to do a comparison,
00870                  * and the X register is a known constant,
00871                  * we can merge this value into the
00872                  * comparison:
00873                  *
00874                  * sub x  ->    nop
00875                  * jeq #y   jeq #(x+y)
00876                  */
00877                 b->s.k += vmap[val].const_val;
00878                 last->s.code = NOP;
00879                 done = 0;
00880             }
00881             else if (b->s.k == 0)
00882             {
00883                 /*
00884                  * If the X register isn't a constant,
00885                  * and the comparison in the test is
00886                  * against 0, we can compare with the
00887                  * X register, instead:
00888                  *
00889                  * sub x  ->    nop
00890                  * jeq #0   jeq x
00891                  */
00892                 last->s.code = NOP;
00893                 b->s.code = BPF_JMP | BPF_JEQ | BPF_X;
00894                 done = 0;
00895             }
00896         }
00897         /*
00898          * Likewise, a constant subtract can be simplified:
00899          *
00900          * sub #x ->    nop
00901          * jeq #y ->    jeq #(x+y)
00902          */
00903         else if (last->s.code == (BPF_ALU | BPF_SUB | BPF_K))
00904         {
00905             last->s.code = NOP;
00906             b->s.k += last->s.k;
00907             done = 0;
00908         }
00909         /*
00910          * And, similarly, a constant AND can be simplified
00911          * if we're testing against 0, i.e.:
00912          *
00913          * and #k   nop
00914          * jeq #0  ->   jset #k
00915          */
00916         else if (last->s.code == (BPF_ALU | BPF_AND | BPF_K) && b->s.k == 0)
00917         {
00918             b->s.k = last->s.k;
00919             b->s.code = BPF_JMP | BPF_K | BPF_JSET;
00920             last->s.code = NOP;
00921             done = 0;
00922             opt_not(b);
00923         }
00924     }
00925     /*
00926      * jset #0        ->   never
00927      * jset #ffffffff ->   always
00928      */
00929     if (b->s.code == (BPF_JMP | BPF_K | BPF_JSET))
00930     {
00931         if (b->s.k == 0)
00932             JT(b) = JF(b);
00933         if (b->s.k == 0xffffffff)
00934             JF(b) = JT(b);
00935     }
00936     /*
00937      * If we're comparing against the index register, and the index
00938      * register is a known constant, we can just compare against that
00939      * constant.
00940      */
00941     val = b->val[X_ATOM];
00942     if (vmap[val].is_const && BPF_SRC(b->s.code) == BPF_X)
00943     {
00944         bpf_int32 v = vmap[val].const_val;
00945         b->s.code &= ~BPF_X;
00946         b->s.k = v;
00947     }
00948     /*
00949      * If the accumulator is a known constant, we can compute the
00950      * comparison result.
00951      */
00952     val = b->val[A_ATOM];
00953     if (vmap[val].is_const && BPF_SRC(b->s.code) == BPF_K)
00954     {
00955         bpf_int32 v = vmap[val].const_val;
00956         switch (BPF_OP(b->s.code))
00957         {
00958 
00959             case BPF_JEQ:
00960                 v = v == b->s.k;
00961                 break;
00962 
00963             case BPF_JGT:
00964                 v = (unsigned) v > b->s.k;
00965                 break;
00966 
00967             case BPF_JGE:
00968                 v = (unsigned) v >= b->s.k;
00969                 break;
00970 
00971             case BPF_JSET:
00972                 v &= b->s.k;
00973                 break;
00974 
00975             default:
00976                 abort();
00977         }
00978         if (JF(b) != JT(b))
00979             done = 0;
00980         if (v)
00981             JF(b) = JT(b);
00982         else
00983             JT(b) = JF(b);
00984     }
00985 }

00987 /*
00988  * Compute the symbolic value of expression of 's', and update
00989  * anything it defines in the value table 'val'.  If 'alter' is true,
00990  * do various optimizations.  This code would be cleaner if symbolic
00991  * evaluation and code transformations weren't folded together.
00992  */
00993 static void opt_stmt(s, val, alter)
00994      struct stmt *s;
00995      int val[];
00996      int alter;
00997 {
00998     int op;
00999     int v;
01000 
01001     switch (s->code)
01002     {
01003 
01004         case BPF_LD | BPF_ABS | BPF_W:
01005         case BPF_LD | BPF_ABS | BPF_H:
01006         case BPF_LD | BPF_ABS | BPF_B:
01007             v = F(s->code, s->k, 0L);
01008             vstore(s, &val[A_ATOM], v, alter);
01009             break;
01010 
01011         case BPF_LD | BPF_IND | BPF_W:
01012         case BPF_LD | BPF_IND | BPF_H:
01013         case BPF_LD | BPF_IND | BPF_B:
01014             v = val[X_ATOM];
01015             if (alter && vmap[v].is_const)
01016             {
01017                 s->code = BPF_LD | BPF_ABS | BPF_SIZE(s->code);
01018                 s->k += vmap[v].const_val;
01019                 v = F(s->code, s->k, 0L);
01020                 done = 0;
01021             }
01022             else
01023                 v = F(s->code, s->k, v);
01024             vstore(s, &val[A_ATOM], v, alter);
01025             break;
01026 
01027         case BPF_LD | BPF_LEN:
01028             v = F(s->code, 0L, 0L);
01029             vstore(s, &val[A_ATOM], v, alter);
01030             break;
01031 
01032         case BPF_LD | BPF_IMM:
01033             v = K(s->k);
01034             vstore(s, &val[A_ATOM], v, alter);
01035             break;
01036 
01037         case BPF_LDX | BPF_IMM:
01038             v = K(s->k);
01039             vstore(s, &val[X_ATOM], v, alter);
01040             break;
01041 
01042         case BPF_LDX | BPF_MSH | BPF_B:
01043             v = F(s->code, s->k, 0L);
01044             vstore(s, &val[X_ATOM], v, alter);
01045             break;
01046 
01047         case BPF_ALU | BPF_NEG:
01048             if (alter && vmap[val[A_ATOM]].is_const)
01049             {
01050                 s->code = BPF_LD | BPF_IMM;
01051                 s->k = -vmap[val[A_ATOM]].const_val;
01052                 val[A_ATOM] = K(s->k);
01053             }
01054             else
01055                 val[A_ATOM] = F(s->code, val[A_ATOM], 0L);
01056             break;
01057 
01058         case BPF_ALU | BPF_ADD | BPF_K:
01059         case BPF_ALU | BPF_SUB | BPF_K:
01060         case BPF_ALU | BPF_MUL | BPF_K:
01061         case BPF_ALU | BPF_DIV | BPF_K:
01062         case BPF_ALU | BPF_AND | BPF_K:
01063         case BPF_ALU | BPF_OR | BPF_K:
01064         case BPF_ALU | BPF_LSH | BPF_K:
01065         case BPF_ALU | BPF_RSH | BPF_K:
01066             op = BPF_OP(s->code);
01067             if (alter)
01068             {
01069                 if (s->k == 0)
01070                 {
01071                     /* don't optimize away "sub #0"
01072                      * as it may be needed later to
01073                      * fixup the generated math code */
01074                     if (op == BPF_ADD || op == BPF_LSH || op == BPF_RSH || op == BPF_OR)
01075                     {
01076                         s->code = NOP;
01077                         break;
01078                     }
01079                     if (op == BPF_MUL || op == BPF_AND)
01080                     {
01081                         s->code = BPF_LD | BPF_IMM;
01082                         val[A_ATOM] = K(s->k);
01083                         break;
01084                     }
01085                 }
01086                 if (vmap[val[A_ATOM]].is_const)
01087                 {
01088                     fold_op(s, val[A_ATOM], K(s->k));
01089                     val[A_ATOM] = K(s->k);
01090                     break;
01091                 }
01092             }
01093             val[A_ATOM] = F(s->code, val[A_ATOM], K(s->k));
01094             break;
01095 
01096         case BPF_ALU | BPF_ADD | BPF_X:
01097         case BPF_ALU | BPF_SUB | BPF_X:
01098         case BPF_ALU | BPF_MUL | BPF_X:
01099         case BPF_ALU | BPF_DIV | BPF_X:
01100         case BPF_ALU | BPF_AND | BPF_X:
01101         case BPF_ALU | BPF_OR | BPF_X:
01102         case BPF_ALU | BPF_LSH | BPF_X:
01103         case BPF_ALU | BPF_RSH | BPF_X:
01104             op = BPF_OP(s->code);
01105             if (alter && vmap[val[X_ATOM]].is_const)
01106             {
01107                 if (vmap[val[A_ATOM]].is_const)
01108                 {
01109                     fold_op(s, val[A_ATOM], val[X_ATOM]);
01110                     val[A_ATOM] = K(s->k);
01111                 }
01112                 else
01113                 {
01114                     s->code = BPF_ALU | BPF_K | op;
01115                     s->k = vmap[val[X_ATOM]].const_val;
01116                     done = 0;
01117                     val[A_ATOM] = F(s->code, val[A_ATOM], K(s->k));
01118                 }
01119                 break;
01120             }
01121             /*
01122              * Check if we're doing something to an accumulator
01123              * that is 0, and simplify.  This may not seem like
01124              * much of a simplification but it could open up further
01125              * optimizations.
01126              * XXX We could also check for mul by 1, etc.
01127              */
01128             if (alter && vmap[val[A_ATOM]].is_const && vmap[val[A_ATOM]].const_val == 0)
01129             {
01130                 if (op == BPF_ADD || op == BPF_OR)
01131                 {
01132                     s->code = BPF_MISC | BPF_TXA;
01133                     vstore(s, &val[A_ATOM], val[X_ATOM], alter);
01134                     break;
01135                 }
01136                 else if (op == BPF_MUL || op == BPF_DIV || op == BPF_AND || op == BPF_LSH || op == BPF_RSH)
01137                 {
01138                     s->code = BPF_LD | BPF_IMM;
01139                     s->k = 0;
01140                     vstore(s, &val[A_ATOM], K(s->k), alter);
01141                     break;
01142                 }
01143                 else if (op == BPF_NEG)
01144                 {
01145                     s->code = NOP;
01146                     break;
01147                 }
01148             }
01149             val[A_ATOM] = F(s->code, val[A_ATOM], val[X_ATOM]);
01150             break;
01151 
01152         case BPF_MISC | BPF_TXA:
01153             vstore(s, &val[A_ATOM], val[X_ATOM], alter);
01154             break;
01155 
01156         case BPF_LD | BPF_MEM:
01157             v = val[s->k];
01158             if (alter && vmap[v].is_const)
01159             {
01160                 s->code = BPF_LD | BPF_IMM;
01161                 s->k = vmap[v].const_val;
01162                 done = 0;
01163             }
01164             vstore(s, &val[A_ATOM], v, alter);
01165             break;
01166 
01167         case BPF_MISC | BPF_TAX:
01168             vstore(s, &val[X_ATOM], val[A_ATOM], alter);
01169             break;
01170 
01171         case BPF_LDX | BPF_MEM:
01172             v = val[s->k];
01173             if (alter && vmap[v].is_const)
01174             {
01175                 s->code = BPF_LDX | BPF_IMM;
01176                 s->k = vmap[v].const_val;
01177                 done = 0;
01178             }
01179             vstore(s, &val[X_ATOM], v, alter);
01180             break;
01181 
01182         case BPF_ST:
01183             vstore(s, &val[s->k], val[A_ATOM], alter);
01184             break;
01185 
01186         case BPF_STX:
01187             vstore(s, &val[s->k], val[X_ATOM], alter);
01188             break;
01189     }
01190 }

01242 static void opt_blk(b, do_stmts)
01243      struct block *b;
01244      int do_stmts;
01245 {
01246     struct slist *s;
01247     struct edge *p;
01248     int i;
01249     bpf_int32 aval, xval;
01250 
01251 #if 0
01252     for (s = b->stmts; s && s->next; s = s->next)
01253         if (BPF_CLASS(s->s.code) == BPF_JMP)
01254         {

01256             break;
01257         }

01260     /*
01261      * Initialize the atom values.
01262      */

01264     if (p == 0)
01265     {
01266         /*
01267          * We have no predecessors, so everything is undefined
01268          * upon entry to this block.
01269          */
01270         memset((char *) b->val, 0, sizeof(b->val));
01271     }
01272     else
01273     {
01274         /*
01275          * Inherit values from our predecessors.
01276          *
01277          * First, get the values from the predecessor along the
01278          * first edge leading to this node.
01279          */

01281         /*
01282          * Now look at all the other nodes leading to this node.
01283          * If, for the predecessor along that edge, a register
01284          * has a different value from the one we have (i.e.,
01285          * control paths are merging, and the merging paths
01286          * assign different values to that register), give the
01287          * register the undefined value of 0.
01288          */

01290         {

01292                 if (b->val[i] != p->pred->val[i])
01293                     b->val[i] = 0;
01294         }

01297     xval = b->val[X_ATOM];

01299         opt_stmt(&s->s, b->val, do_stmts);
01300 
01301     /*
01302      * This is a special case: if we don't use anything from this
01303      * block, and we load the accumulator or index register with a
01304      * value that is already there, or if this block is a return,
01305      * eliminate all the statements.
01306      *
01307      * XXX - what if it does a store?
01308      *
01309      * XXX - why does it matter whether we use anything from this
01310      * block?  If the accumulator or index register doesn't change
01311      * its value, isn't that OK even if we use that value?
01312      *
01313      * XXX - if we load the accumulator with a different value,
01314      * and the block ends with a conditional branch, we obviously
01315      * can't eliminate it, as the branch depends on that value.
01316      * For the index register, the conditional branch only depends
01317      * on the index register value if the test is against the index
01318      * register value rather than a constant; if nothing uses the
01319      * value we put into the index register, and we're not testing
01320      * against the index register's value, and there aren't any
01321      * other problems that would keep us from eliminating this
01322      * block, can we eliminate it?
01323      */
01324     if (do_stmts &&
01325         ((b->out_use == 0 && aval != 0 && b->val[A_ATOM] == aval &&
01326           xval != 0 && b->val[X_ATOM] == xval) || BPF_CLASS(b->s.code) == BPF_RET))
01327     {
01328         if (b->stmts != 0)
01329         {
01330             b->stmts = 0;
01331             done = 0;
01332         }
01333     }
01334     else
01335     {
01336         opt_peep(b);
01337         opt_deadstores(b);
01338     }
01339     /*
01340      * Set up values for branch optimizer.
01341      */
01342     if (BPF_SRC(b->s.code) == BPF_K)
01343         b->oval = K(b->s.k);
01344     else
01345         b->oval = b->val[X_ATOM];

01347     b->ef.code = -b->s.code;

01350 /*
01351  * Return true if any register that is used on exit from 'succ', has
01352  * an exit value that is different from the corresponding exit value
01353  * from 'b'.
01354  */
01355 static int use_conflict(b, succ)
01356      struct block *b, *succ;
01357 {
01358     int atom;
01359     atomset use = succ->out_use;
01360 
01361     if (use == 0)
01362         return 0;
01363 
01364     for (atom = 0; atom < N_ATOMS; ++atom)
01365         if (ATOMELEM(use, atom))
01366             if (b->val[atom] != succ->val[atom])
01367                 return 1;
01368     return 0;
01369 }

01371 static struct block *fold_edge(child, ep)
01372      struct block *child;
01373      struct edge *ep;
01374 {
01375     int sense;
01376     int aval0, aval1, oval0, oval1;
01377     int code = ep->code;
01378 
01379     if (code < 0)
01380     {
01381         code = -code;
01382         sense = 0;
01383     }
01384     else
01385         sense = 1;
01386 
01387     if (child->s.code != code)
01388         return 0;
01389 
01390     aval0 = child->val[A_ATOM];
01391     oval0 = child->oval;
01392     aval1 = ep->pred->val[A_ATOM];
01393     oval1 = ep->pred->oval;
01394 
01395     if (aval0 != aval1)
01396         return 0;
01397 
01398     if (oval0 == oval1)
01399         /*
01400          * The operands of the branch instructions are
01401          * identical, so the result is true if a true
01402          * branch was taken to get here, otherwise false.
01403          */
01404         return sense ? JT(child) : JF(child);
01405 
01406     if (sense && code == (BPF_JMP | BPF_JEQ | BPF_K))
01407         /*
01408          * At this point, we only know the comparison if we
01409          * came down the true branch, and it was an equality
01410          * comparison with a constant.
01411          *
01412          * I.e., if we came down the true branch, and the branch
01413          * was an equality comparison with a constant, we know the
01414          * accumulator contains that constant.  If we came down
01415          * the false branch, or the comparison wasn't with a
01416          * constant, we don't know what was in the accumulator.
01417          *
01418          * We rely on the fact that distinct constants have distinct
01419          * value numbers.
01420          */
01421         return JF(child);
01422 
01423     return 0;
01424 }

01426 static void opt_j(ep)
01427      struct edge *ep;
01428 {
01429     register int i, k;
01430     register struct block *target;
01431 
01432     if (JT(ep->succ) == 0)
01433         return;
01434 
01435     if (JT(ep->succ) == JF(ep->succ))
01436     {
01437         /*
01438          * Common branch targets can be eliminated, provided
01439          * there is no data dependency.
01440          */
01441         if (!use_conflict(ep->pred, ep->succ->et.succ))
01442         {
01443             done = 0;
01444             ep->succ = JT(ep->succ);
01445         }
01446     }
01447     /*
01448      * For each edge dominator that matches the successor of this
01449      * edge, promote the edge successor to the its grandchild.
01450      *
01451      * XXX We violate the set abstraction here in favor a reasonably
01452      * efficient loop.
01453      */
01454   top:
01455     for (i = 0; i < edgewords; ++i)
01456     {
01457         register bpf_u_int32 x = ep->edom[i];
01458 
01459         while (x != 0)
01460         {
01461             k = ffs(x) - 1;
01462             x &= ~(1 << k);
01463             k += i * BITS_PER_WORD;
01464 
01465             target = fold_edge(ep->succ, edges[k]);
01466             /*
01467              * Check that there is no data dependency between
01468              * nodes that will be violated if we move the edge.
01469              */
01470             if (target != 0 && !use_conflict(ep->pred, target))
01471             {
01472                 done = 0;
01473                 ep->succ = target;
01474                 if (JT(target) != 0)
01475                     /*
01476                      * Start over unless we hit a leaf.
01477                      */
01478                     goto top;
01479                 return;
01480             }
01481         }
01482     }
01483 }

02302 /*
02303  * Convert flowgraph intermediate representation to the
02304  * BPF array representation.  Set *lenp to the number of instructions.
02305  *
02306  * This routine does *NOT* leak the memory pointed to by fp.  It *must
02307  * not* do free(fp) before returning fp; doing so would make no sense,
02308  * as the BPF array pointed to by the return value of icode_to_fcode()
02309  * must be valid - it's being returned for use in a bpf_program structure.
02310  *
02311  * If it appears that icode_to_fcode() is leaking, the problem is that
02312  * the program using pcap_compile() is failing to free the memory in
02313  * the BPF program when it's done - the leak is in the program, not in
02314  * the routine that happens to be allocating the memory.  (By analogy, if
02315  * a program calls fopen() without ever calling fclose() on the FILE *,
02316  * it will leak the FILE structure; the leak is not in fopen(), it's in
02317  * the program.)  Change the program to use pcap_freecode() when it's
02318  * done with the filter program.  See the pcap man page.
02319  */

02327     /*

02329      * with too-large offsets.
02330      */

02332     {
02333         unMarkAll();
02334         n = *lenp = count_stmts(root);
02335 
02336         fp = (struct bpf_insn *) malloc(sizeof(*fp) * n);
02337         if (fp == NULL)
02338             bpf_error("malloc");
02339         memset((char *) fp, 0, sizeof(*fp) * n);
02340         fstart = fp;
02341         ftail = fp + n;

02343         unMarkAll();
