00085 void
00086 udld_print (const u_char *pptr, u_int length)
00087 {
00088     int code, type, len;
00089     const u_char *tptr;
00090 
00091     if (length < UDLD_HEADER_LEN)
00092         goto trunc;
00093 
00094     tptr = pptr; 
00095 
00096     if (!TTEST2(*tptr, UDLD_HEADER_LEN))    
00097     goto trunc;
00098 
00099     code = UDLD_EXTRACT_OPCODE(*tptr);
00100 
00101     printf("UDLDv%u, Code %s (%x), Flags [%s] (0x%02x), length %u", 
00102            UDLD_EXTRACT_VERSION(*tptr),
00103            tok2str(udld_code_values, "Reserved", code),
00104            code,
00105            bittok2str(udld_flags_values, "none", *(tptr+1)),
00106            *(tptr+1),
00107            length);
00108 
00109     /*
00110      * In non-verbose mode, just print version and opcode type
00111      */
00112     if (vflag < 1) {
00113     return;
00114     }
00115 
00116     printf("\n\tChecksum 0x%04x (unverified)", EXTRACT_16BITS(tptr+2));
00117 
00118     tptr += UDLD_HEADER_LEN;
00119 
00120     while (tptr < (pptr+length)) {
00121 
00122         if (!TTEST2(*tptr, 4)) 
00123             goto trunc;
00124 
00125     type = EXTRACT_16BITS(tptr);
00126         len  = EXTRACT_16BITS(tptr+2); 
00127         len -= 4;
00128         tptr += 4;
00129 
00130         /* infinite loop check */
00131         if (type == 0 || len == 0) {
00132             return;
00133         }
00134 
00135         printf("\n\t%s (0x%04x) TLV, length %u",
00136                tok2str(udld_tlv_values, "Unknown", type),
00137                type, len);
00138 
00139         switch (type) {
00140         case UDLD_DEVICE_ID_TLV:
00141         case UDLD_PORT_ID_TLV:
00142         case UDLD_ECHO_TLV:
00143         case UDLD_DEVICE_NAME_TLV: 
00144             printf(", %s", tptr);
00145             break;
00146 
00147         case UDLD_MESSAGE_INTERVAL_TLV: 
00148         case UDLD_TIMEOUT_INTERVAL_TLV:
00149             printf(", %us", (*tptr));
00150             break;
00151 
00152         case UDLD_SEQ_NUMBER_TLV:
00153             printf(", %u", EXTRACT_32BITS(tptr));
00154             break;
00155 
00156         default:
00157             break;
00158         }   
00159         tptr += len;
00160     }
00161 
00162     return;
00163 
00164  trunc:
00165     printf("[|udld]");
00166 }
