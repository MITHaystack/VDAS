00101 hsrp_print(register const u_int8_t *bp, register u_int len)
00102 {
00103     struct hsrp *hp = (struct hsrp *) bp;
00104 
00105     TCHECK(hp->hsrp_version);
00106     printf("HSRPv%d", hp->hsrp_version);
00107     if (hp->hsrp_version != 0)
00108         return;
00109     TCHECK(hp->hsrp_op_code);
00110     printf("-");
00111     printf("%s ", tok2strary(op_code_str, "unknown (%d)", hp->hsrp_op_code));
00112     printf("%d: ", len);
00113     TCHECK(hp->hsrp_state);
00114     printf("state=%s ", tok2str(states, "Unknown (%d)", hp->hsrp_state));
00115     TCHECK(hp->hsrp_group);
00116     printf("group=%d ", hp->hsrp_group);
00117     TCHECK(hp->hsrp_reserved);
00118     if (hp->hsrp_reserved != 0) {
00119         printf("[reserved=%d!] ", hp->hsrp_reserved);
00120     }
00121     TCHECK(hp->hsrp_virtaddr);
00122     printf("addr=%s", ipaddr_string(&hp->hsrp_virtaddr));
00123     if (vflag) {
00124         printf(" hellotime=");
00125         relts_print(hp->hsrp_hellotime);
00126         printf(" holdtime=");
00127         relts_print(hp->hsrp_holdtime);
00128         printf(" priority=%d", hp->hsrp_priority);
00129         printf(" auth=\"");
00130         if (fn_printn(hp->hsrp_authdata, sizeof(hp->hsrp_authdata),
00131             snapend)) {
00132             printf("\"");
00133             goto trunc;
00134         }
00135         printf("\"");
00136     }
