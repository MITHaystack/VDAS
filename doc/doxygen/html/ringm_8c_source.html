00136 /* Thread main loop */
00137 static void * sniffer (void * _thread)
00138 {
00139   my_thread_t * t = _thread;
00140 
00141   struct pfring_pkthdr header;
00142   u_char * packet;             /* pointer to the packet */
00143 
00144   /* Get memory for packet capturing */
00145   packet = calloc (t -> snapshot, 1);
00146 
00147   gettimeofday (& t -> started, NULL);
00148 
00149   t -> partial = 0;
00150   while (t -> partial < t -> maxcount)
00151     {
00152       /* Please give me just a packet from the ring */
00153       if (pfring_recv (t -> ring, &packet, t -> snapshot, & header, 1) > 0)
00154     {
00155       t -> partial ++;
00156       if (t -> bar)
00157         showbar (t -> partial);
00158     }
00159       else
00160     t -> errors ++;
00161     }
00162 
00163   gettimeofday (& t -> stopped, NULL);
00164 
00165   t -> elapsed = delta_time_in_microseconds (& t -> stopped, & t -> started);
00166 
00167   free (packet);
00168 
00169   return NULL;
00170 }
