00264 static int ethtool_get_regs(struct net_device *dev, char *useraddr)
00265 {
00266     struct ethtool_regs regs;
00267     struct ethtool_ops *ops = ethtool_ops;
00268     void *regbuf;
00269     int reglen, ret;
00270 
00271     if (!ops->get_regs || !ops->get_regs_len)
00272         return -EOPNOTSUPP;
00273 
00274     if (copy_from_user(&regs, useraddr, sizeof(regs)))
00275         return -EFAULT;
00276 
00277     reglen = ops->get_regs_len(dev);
00278     if (regs.len > reglen)
00279         regs.len = reglen;
00280 
00281     regbuf = kmalloc(reglen, GFP_USER);
00282     if (!regbuf)
00283         return -ENOMEM;
00284 
00285     ops->get_regs(dev, &regs, regbuf);
00286 
00287     ret = -EFAULT;
00288     if (copy_to_user(useraddr, &regs, sizeof(regs)))
00289         goto out;
00290     useraddr += offsetof(struct ethtool_regs, data);
00291     if (copy_to_user(useraddr, regbuf, reglen))
00292         goto out;
00293     ret = 0;
00294 
00295 out:
00296     kfree(regbuf);
00297     return ret;
00298 }

00377 static int ethtool_get_eeprom(struct net_device *dev, void *useraddr)
00378 {
00379     struct ethtool_eeprom eeprom;
00380     struct ethtool_ops *ops = ethtool_ops;
00381     u8 *data;
00382     int ret;
00383 
00384     if (!ops->get_eeprom || !ops->get_eeprom_len)
00385         return -EOPNOTSUPP;
00386 
00387     if (copy_from_user(&eeprom, useraddr, sizeof(eeprom)))
00388         return -EFAULT;
00389 
00390     /* Check for wrap and zero */
00391     if (eeprom.offset + eeprom.len <= eeprom.offset)
00392         return -EINVAL;
00393 
00394     /* Check for exceeding total eeprom len */
00395     if (eeprom.offset + eeprom.len > ops->get_eeprom_len(dev))
00396         return -EINVAL;
00397 
00398     data = kmalloc(eeprom.len, GFP_USER);
00399     if (!data)
00400         return -ENOMEM;
00401 
00402     ret = -EFAULT;
00403     if (copy_from_user(data, useraddr + sizeof(eeprom), eeprom.len))
00404         goto out;
00405 
00406     ret = ops->get_eeprom(dev, &eeprom, data);
00407     if (ret)
00408         goto out;
00409 
00410     ret = -EFAULT;
00411     if (copy_to_user(useraddr, &eeprom, sizeof(eeprom)))
00412         goto out;
00413     if (copy_to_user(useraddr + sizeof(eeprom), data, eeprom.len))
00414         goto out;
00415     ret = 0;
00416 
00417 out:
00418     kfree(data);
00419     return ret;
00420 }

00941     /* only supports internal transceiver */

00945     ecmd->phy_address = mii->phy_id;
00946

00948     advert = mii->mdio_read(dev, mii->phy_id, MII_ADVERTISE);
00949     if (advert & ADVERTISE_10HALF)
00950         ecmd->advertising |= ADVERTISED_10baseT_Half;
00951     if (advert & ADVERTISE_10FULL)
00952         ecmd->advertising |= ADVERTISED_10baseT_Full;
00953     if (advert & ADVERTISE_100HALF)
00954         ecmd->advertising |= ADVERTISED_100baseT_Half;
00955     if (advert & ADVERTISE_100FULL)
00956         ecmd->advertising |= ADVERTISED_100baseT_Full;
00957 
00958     bmcr = mii->mdio_read(dev, mii->phy_id, MII_BMCR);
00959     lpa = mii->mdio_read(dev, mii->phy_id, MII_LPA);
00960     if (bmcr & BMCR_ANENABLE) {
00961         ecmd->advertising |= ADVERTISED_Autoneg;
00962         ecmd->autoneg = AUTONEG_ENABLE;
00963 
00964         nego = mii_nway_result(advert & lpa);
00965         if (nego == LPA_100FULL || nego == LPA_100HALF)
00966             ecmd->speed = SPEED_100;
00967         else
00968             ecmd->speed = SPEED_10;
00969         if (nego == LPA_100FULL || nego == LPA_10FULL) {
00970             ecmd->duplex = DUPLEX_FULL;
00971             mii->full_duplex = 1;
00972         } else {
00973             ecmd->duplex = DUPLEX_HALF;
00974             mii->full_duplex = 0;
00975         }
00976     } else {
00977         ecmd->autoneg = AUTONEG_DISABLE;

00979         ecmd->speed = (bmcr & BMCR_SPEED100) ? SPEED_100 : SPEED_10;
00980         ecmd->duplex = (bmcr & BMCR_FULLDPLX) ? DUPLEX_FULL : DUPLEX_HALF;
00981     }

00988 int _kc_mii_ethtool_sset(struct mii_if_info *mii, struct ethtool_cmd *ecmd)
00989 {
00990     struct net_device *dev = mii->dev;
00991 
00992     if (ecmd->speed != SPEED_10 && ecmd->speed != SPEED_100)
00993         return -EINVAL;
00994     if (ecmd->duplex != DUPLEX_HALF && ecmd->duplex != DUPLEX_FULL)
00995         return -EINVAL;
00996     if (ecmd->port != PORT_MII)
00997         return -EINVAL;
00998     if (ecmd->transceiver != XCVR_INTERNAL)
00999         return -EINVAL;
01000     if (ecmd->phy_address != mii->phy_id)
01001         return -EINVAL;
01002     if (ecmd->autoneg != AUTONEG_DISABLE && ecmd->autoneg != AUTONEG_ENABLE)
01003         return -EINVAL;
01004 
01005     /* ignore supported, maxtxpkt, maxrxpkt */
01006 
01007     if (ecmd->autoneg == AUTONEG_ENABLE) {
01008         u32 bmcr, advert, tmp;
01009 
01010         if ((ecmd->advertising & (ADVERTISED_10baseT_Half |
01011                       ADVERTISED_10baseT_Full |
01012                       ADVERTISED_100baseT_Half |
01013                       ADVERTISED_100baseT_Full)) == 0)
01014             return -EINVAL;
01015 
01016         /* advertise only what has been requested */
01017         advert = mii->mdio_read(dev, mii->phy_id, MII_ADVERTISE);
01018         tmp = advert & ~(ADVERTISE_ALL | ADVERTISE_100BASE4);
01019         if (ADVERTISED_10baseT_Half)
01020             tmp |= ADVERTISE_10HALF;
01021         if (ADVERTISED_10baseT_Full)
01022             tmp |= ADVERTISE_10FULL;
01023         if (ADVERTISED_100baseT_Half)
01024             tmp |= ADVERTISE_100HALF;
01025         if (ADVERTISED_100baseT_Full)
01026             tmp |= ADVERTISE_100FULL;
01027         if (advert != tmp) {
01028             mii->mdio_write(dev, mii->phy_id, MII_ADVERTISE, tmp);
01029             mii->advertising = tmp;
01030         }
01031 
01032         /* turn on autonegotiation, and force a renegotiate */
01033         bmcr = mii->mdio_read(dev, mii->phy_id, MII_BMCR);
01034         bmcr |= (BMCR_ANENABLE | BMCR_ANRESTART);
01035         mii->mdio_write(dev, mii->phy_id, MII_BMCR, bmcr);
01036 
01037         mii->force_media = 0;
01038     } else {
01039         u32 bmcr, tmp;
01040 
01041         /* turn off auto negotiation, set speed and duplexity */
01042         bmcr = mii->mdio_read(dev, mii->phy_id, MII_BMCR);
01043         tmp = bmcr & ~(BMCR_ANENABLE | BMCR_SPEED100 | BMCR_FULLDPLX);

01045             tmp |= BMCR_SPEED100;
01046         if (ecmd->duplex == DUPLEX_FULL) {
01047             tmp |= BMCR_FULLDPLX;
01048             mii->full_duplex = 1;
01049         } else
01050             mii->full_duplex = 0;
01051         if (bmcr != tmp)
01052             mii->mdio_write(dev, mii->phy_id, MII_BMCR, tmp);
01053
