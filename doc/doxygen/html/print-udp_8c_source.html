00109 static void
00110 vat_print(const void *hdr, register const struct udphdr *up)
00111 {
00112     /* vat/vt audio */
00113     u_int ts = *(u_int16_t *)hdr;
00114     if ((ts & 0xf060) != 0) {
00115         /* probably vt */
00116         (void)printf("udp/vt %u %d / %d",
00117                  (u_int32_t)(EXTRACT_16BITS(&up->uh_ulen) - sizeof(*up)),
00118                  ts & 0x3ff, ts >> 10);
00119     } else {
00120         /* probably vat */
00121         u_int32_t i0 = EXTRACT_32BITS(&((u_int *)hdr)[0]);
00122         u_int32_t i1 = EXTRACT_32BITS(&((u_int *)hdr)[1]);
00123         printf("udp/vat %u c%d %u%s",
00124             (u_int32_t)(EXTRACT_16BITS(&up->uh_ulen) - sizeof(*up) - 8),
00125             i0 & 0xffff,
00126             i1, i0 & 0x800000? "*" : "");
00127         /* audio format */
00128         if (i0 & 0x1f0000)
00129             printf(" f%d", (i0 >> 16) & 0x1f);
00130         if (i0 & 0x3f000000)
00131             printf(" s%d", (i0 >> 24) & 0x3f);
00132     }
00133 }

00135 static void
00136 rtp_print(const void *hdr, u_int len, register const struct udphdr *up)
00137 {
00138     /* rtp v1 or v2 */
00139     u_int *ip = (u_int *)hdr;
00140     u_int hasopt, hasext, contype, hasmarker;
00141     u_int32_t i0 = EXTRACT_32BITS(&((u_int *)hdr)[0]);
00142     u_int32_t i1 = EXTRACT_32BITS(&((u_int *)hdr)[1]);
00143     u_int dlen = EXTRACT_16BITS(&up->uh_ulen) - sizeof(*up) - 8;
00144     const char * ptype;
00145 
00146     ip += 2;
00147     len >>= 2;
00148     len -= 2;
00149     hasopt = 0;
00150     hasext = 0;
00151     if ((i0 >> 30) == 1) {
00152         /* rtp v1 */
00153         hasopt = i0 & 0x800000;
00154         contype = (i0 >> 16) & 0x3f;
00155         hasmarker = i0 & 0x400000;
00156         ptype = "rtpv1";
00157     } else {
00158         /* rtp v2 */
00159         hasext = i0 & 0x10000000;
00160         contype = (i0 >> 16) & 0x7f;
00161         hasmarker = i0 & 0x800000;
00162         dlen -= 4;
00163         ptype = "rtp";
00164         ip += 1;
00165         len -= 1;
00166     }
00167     printf("udp/%s %d c%d %s%s %d %u",
00168         ptype,
00169         dlen,
00170         contype,
00171         (hasopt || hasext)? "+" : "",
00172         hasmarker? "*" : "",
00173         i0 & 0xffff,
00174         i1);
00175     if (vflag) {
00176         printf(" %u", EXTRACT_32BITS(&((u_int *)hdr)[2]));
00177         if (hasopt) {
00178             u_int i2, optlen;
00179             do {
00180                 i2 = ip[0];
00181                 optlen = (i2 >> 16) & 0xff;
00182                 if (optlen == 0 || optlen > len) {
00183                     printf(" !opt");
00184                     return;
00185                 }
00186                 ip += optlen;
00187                 len -= optlen;
00188             } while ((int)i2 >= 0);
00189         }
00190         if (hasext) {
00191             u_int i2, extlen;
00192             i2 = ip[0];
00193             extlen = (i2 & 0xffff) + 1;
00194             if (extlen > len) {
00195                 printf(" !ext");
00196                 return;
00197             }
00198             ip += extlen;
00199         }
00200         if (contype == 0x1f) /*XXX H.261 */
00201             printf(" 0x%04x", ip[0] >> 16);
00202     }
00203 }

00205 static const u_char *
00206 rtcp_print(const u_char *hdr, const u_char *ep)
00207 {
00208     /* rtp v2 control (rtcp) */
00209     struct rtcp_rr *rr = 0;
00210     struct rtcp_sr *sr;
00211     struct rtcphdr *rh = (struct rtcphdr *)hdr;
00212     u_int len;
00213     u_int16_t flags;
00214     int cnt;
00215     double ts, dts;
00216     if ((u_char *)(rh + 1) > ep) {
00217         printf(" [|rtcp]");
00218         return (ep);
00219     }
00220     len = (EXTRACT_16BITS(&rh->rh_len) + 1) * 4;
00221     flags = EXTRACT_16BITS(&rh->rh_flags);
00222     cnt = (flags >> 8) & 0x1f;
00223     switch (flags & 0xff) {
00224     case RTCP_PT_SR:
00225         sr = (struct rtcp_sr *)(rh + 1);
00226         printf(" sr");
00227         if (len != cnt * sizeof(*rr) + sizeof(*sr) + sizeof(*rh))
00228             printf(" [%d]", len);
00229         if (vflag)
00230             printf(" %u", EXTRACT_32BITS(&rh->rh_ssrc));
00231         if ((u_char *)(sr + 1) > ep) {
00232             printf(" [|rtcp]");
00233             return (ep);
00234         }
00235         ts = (double)(EXTRACT_32BITS(&sr->sr_ntp.upper)) +
00236             ((double)(EXTRACT_32BITS(&sr->sr_ntp.lower)) /
00237             4294967296.0);
00238         printf(" @%.2f %u %up %ub", ts, EXTRACT_32BITS(&sr->sr_ts),
00239             EXTRACT_32BITS(&sr->sr_np), EXTRACT_32BITS(&sr->sr_nb));
00240         rr = (struct rtcp_rr *)(sr + 1);
00241         break;
00242     case RTCP_PT_RR:
00243         printf(" rr");
00244         if (len != cnt * sizeof(*rr) + sizeof(*rh))
00245             printf(" [%d]", len);
00246         rr = (struct rtcp_rr *)(rh + 1);
00247         if (vflag)
00248             printf(" %u", EXTRACT_32BITS(&rh->rh_ssrc));
00249         break;
00250     case RTCP_PT_SDES:
00251         printf(" sdes %d", len);
00252         if (vflag)
00253             printf(" %u", EXTRACT_32BITS(&rh->rh_ssrc));
00254         cnt = 0;
00255         break;
00256     case RTCP_PT_BYE:
00257         printf(" bye %d", len);
00258         if (vflag)
00259             printf(" %u", EXTRACT_32BITS(&rh->rh_ssrc));
00260         cnt = 0;
00261         break;
00262     default:
00263         printf(" type-0x%x %d", flags & 0xff, len);
00264         cnt = 0;
00265         break;
00266     }
00267     if (cnt > 1)
00268         printf(" c%d", cnt);
00269     while (--cnt >= 0) {
00270         if ((u_char *)(rr + 1) > ep) {
00271             printf(" [|rtcp]");
00272             return (ep);
00273         }
00274         if (vflag)
00275             printf(" %u", EXTRACT_32BITS(&rr->rr_srcid));
00276         ts = (double)(EXTRACT_32BITS(&rr->rr_lsr)) / 65536.;
00277         dts = (double)(EXTRACT_32BITS(&rr->rr_dlsr)) / 65536.;
00278         printf(" %ul %us %uj @%.2f+%.2f",
00279             EXTRACT_32BITS(&rr->rr_nl) & 0x00ffffff,
00280             EXTRACT_32BITS(&rr->rr_ls),
00281             EXTRACT_32BITS(&rr->rr_dv), ts, dts);
00282     }
00283     return (hdr + len);
00284 }

00286 static int udp_cksum(register const struct ip *ip,
00287              register const struct udphdr *up,
00288              register u_int len)
00289 {
00290     union phu {
00291         struct phdr {
00292             u_int32_t src;
00293             u_int32_t dst;
00294             u_char mbz;
00295             u_char proto;
00296             u_int16_t len;
00297         } ph;
00298         u_int16_t pa[6];
00299     } phu;
00300     register const u_int16_t *sp;
00301 
00302     /* pseudo-header.. */
00303     phu.ph.len = htons((u_int16_t)len);
00304     phu.ph.mbz = 0;
00305     phu.ph.proto = IPPROTO_UDP;
00306     memcpy(&phu.ph.src, &ip->ip_src.s_addr, sizeof(u_int32_t));
00307     if (IP_HL(ip) == 5)
00308         memcpy(&phu.ph.dst, &ip->ip_dst.s_addr, sizeof(u_int32_t));
00309     else
00310         phu.ph.dst = ip_finddst(ip);
00311 
00312     sp = &phu.pa[0];
00313     return in_cksum((u_short *)up, len,
00314             sp[0]+sp[1]+sp[2]+sp[3]+sp[4]+sp[5]);
00315 }

00318 static int udp6_cksum(const struct ip6_hdr *ip6, const struct udphdr *up,
00319     u_int len)
00320 {
00321     size_t i;
00322     register const u_int16_t *sp;
00323     u_int32_t sum;
00324     union {
00325         struct {
00326             struct in6_addr ph_src;
00327             struct in6_addr ph_dst;
00328             u_int32_t   ph_len;
00329             u_int8_t    ph_zero[3];
00330             u_int8_t    ph_nxt;
00331         } ph;
00332         u_int16_t pa[20];
00333     } phu;
00334 
00335     /* pseudo-header */
00336     memset(&phu, 0, sizeof(phu));
00337     phu.ph.ph_src = ip6->ip6_src;
00338     phu.ph.ph_dst = ip6->ip6_dst;
00339     phu.ph.ph_len = htonl(len);
00340     phu.ph.ph_nxt = IPPROTO_UDP;
00341 
00342     sum = 0;
00343     for (i = 0; i < sizeof(phu.pa) / sizeof(phu.pa[0]); i++)
00344         sum += phu.pa[i];
00345 
00346     sp = (const u_int16_t *)up;
00347 
00348     for (i = 0; i < (len & ~1); i += 2)
00349         sum += *sp++;
00350 
00351     if (len & 1)
00352         sum += htons((*(const u_int8_t *)sp) << 8);
00353 
00354     while (sum > 0xffff)
00355         sum = (sum & 0xffff) + (sum >> 16);
00356     sum = ~sum & 0xffff;
00357 
00358     return (sum);
00359 }
