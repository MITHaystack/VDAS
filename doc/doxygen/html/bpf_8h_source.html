00093 /*
00094  * Struct return by BIOCVERSION.  This represents the version number of 
00095  * the filter language described by the instruction encodings below.
00096  * bpf understands a program iff kernel_major == filter_major &&
00097  * kernel_minor >= filter_minor, that is, if the value returned by the
00098  * running kernel has the same major number and a minor number equal
00099  * equal to or less than the filter being downloaded.  Otherwise, the
00100  * results are undefined, meaning an error may be returned or packets
00101  * may be accepted haphazardly.
00102  * It has nothing to do with the source code version.
00103  */
00104 struct bpf_version {
00105     u_short bv_major;
00106     u_short bv_minor;
00107 };
00108 /* Current version number of filter architecture. */
00109 #define BPF_MAJOR_VERSION 1
00110 #define BPF_MINOR_VERSION 1

00900  * Solaris ipnet pseudo-header; requested by Darren Reed <Darren.Reed@Sun.COM>.
00901  *
00902  * The pseudo-header starts with a one-byte version number; for version 2,
00903  * the pseudo-header is:
00904  *
00905  * struct dl_ipnetinfo {
00906  *     u_int8_t   dli_version;
00907  *     u_int8_t   dli_family;
00908  *     u_int16_t  dli_htype;
00909  *     u_int32_t  dli_pktlen;
00910  *     u_int32_t  dli_ifindex;
00911  *     u_int32_t  dli_grifindex;
00912  *     u_int32_t  dli_zsrc;
00913  *     u_int32_t  dli_zdst;
00914  * };
00915  *
00916  * dli_version is 2 for the current version of the pseudo-header.
00917  *
00918  * dli_family is a Solaris address family value, so it's 2 for IPv4
00919  * and 26 for IPv6.
00920  *
00921  * dli_htype is a "hook type" - 0 for incoming packets, 1 for outgoing
00922  * packets, and 2 for packets arriving from another zone on the same
00923  * machine.
00924  *
00925  * dli_pktlen is the length of the packet data following the pseudo-header
00926  * (so the captured length minus dli_pktlen is the length of the
00927  * pseudo-header, assuming the entire pseudo-header was captured).
00928  *
00929  * dli_ifindex is the interface index of the interface on which the
00930  * packet arrived.
00931  *
00932  * dli_grifindex is the group interface index number (for IPMP interfaces).
00933  *
00934  * dli_zsrc is the zone identifier for the source of the packet.
00935  *
00936  * dli_zdst is the zone identifier for the destination of the packet.
00937  *
00938  * A zone number of 0 is the global zone; a zone number of 0xffffffff
00939  * means that the packet arrived from another host on the network, not
00940  * from another zone on the same machine.

00943  * which of those it is.
