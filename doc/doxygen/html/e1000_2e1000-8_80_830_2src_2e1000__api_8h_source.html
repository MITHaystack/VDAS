00105 /*
00106  * TBI_ACCEPT macro definition:
00107  *
00108  * This macro requires:
00109  *      adapter = a pointer to struct e1000_hw
00110  *      status = the 8 bit status field of the Rx descriptor with EOP set
00111  *      error = the 8 bit error field of the Rx descriptor with EOP set
00112  *      length = the sum of all the length fields of the Rx descriptors that
00113  *               make up the current frame
00114  *      last_byte = the last byte of the frame DMAed by the hardware
00115  *      max_frame_length = the maximum frame length we want to accept.
00116  *      min_frame_length = the minimum frame length we want to accept.
00117  *
00118  * This macro is a conditional that should be used in the interrupt
00119  * handler's Rx processing routine when RxErrors have been detected.
00120  *
00121  * Typical use:
00122  *  ...
00123  *  if (TBI_ACCEPT) {
00124  *      accept_frame = true;
00125  *      e1000_tbi_adjust_stats(adapter, MacAddress);
00126  *      frame_length--;
00127  *  } else {
00128  *      accept_frame = false;
00129  *  }
00130  *  ...
00131  */
00132 
00133 /* The carrier extension symbol, as received by the NIC. */
00134 #define CARRIER_EXTENSION   0x0F
00135 
00136 #define TBI_ACCEPT(a, status, errors, length, last_byte, min_frame_size, max_frame_size) \
00137     (e1000_tbi_sbp_enabled_82543(a) && \
00138      (((errors) & E1000_RXD_ERR_FRAME_ERR_MASK) == E1000_RXD_ERR_CE) && \
00139      ((last_byte) == CARRIER_EXTENSION) && \
00140      (((status) & E1000_RXD_STAT_VP) ? \
00141           (((length) > (min_frame_size - VLAN_TAG_SIZE)) && \
00142            ((length) <= (max_frame_size + 1))) : \
00143           (((length) > min_frame_size) && \
00144            ((length) <= (max_frame_size + VLAN_TAG_SIZE + 1)))))
00145 
00146 #endif
