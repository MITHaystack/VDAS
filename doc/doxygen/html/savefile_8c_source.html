00237 pcap_t *
00238 pcap_fopen_offline(FILE *fp, char *errbuf)
00239 {
00240     register pcap_t *p;
00241     bpf_u_int32 magic;
00242     size_t amt_read;
00243     u_int i;
00244 
00245     p = pcap_create_common("(savefile)", errbuf);
00246     if (p == NULL)
00247         return (NULL);
00248 
00249     /*
00250      * Read the first 4 bytes of the file; the network analyzer dump
00251      * file formats we support (pcap and pcap-ng), and several other
00252      * formats we might support in the future (such as snoop, DOS and
00253      * Windows Sniffer, and Microsoft Network Monitor) all have magic
00254      * numbers that are unique in their first 4 bytes.
00255      */
00256     amt_read = fread((char *)&magic, 1, sizeof(magic), fp);
00257     if (amt_read != sizeof(magic)) {
00258         if (ferror(fp)) {
00259             snprintf(errbuf, PCAP_ERRBUF_SIZE,
00260                 "error reading dump file: %s",
00261                 pcap_strerror(errno));
00262         } else {
00263             snprintf(errbuf, PCAP_ERRBUF_SIZE,
00264                 "truncated dump file; tried to read %lu file header bytes, only got %lu",
00265                 (unsigned long)sizeof(magic),
00266                 (unsigned long)amt_read);
00267         }
00268         goto bad;
00269     }
00270 
00271     /*
00272      * Try all file types.
00273      */
00274     for (i = 0; i < N_FILE_TYPES; i++) {
00275         switch ((*check_headers[i])(p, magic, fp, errbuf)) {
00276 
00277         case -1:
00278             /*
00279              * Error trying to read the header.
00280              */
00281             goto bad;
00282 
00283         case 1:
00284             /*
00285              * Yup, that's it.
00286              */
00287             goto found;
00288         }
00289     }
00290 
00291     /*
00292      * Well, who knows what this mess is....
00293      */
00294     snprintf(errbuf, PCAP_ERRBUF_SIZE, "unknown file format");
00295     goto bad;
00296 
00297 found:
00298     p->sf.rfile = fp;
00299 
00300 #ifdef PCAP_FDDIPAD
00301     /* Padding only needed for live capture fcode */
00302     p->fddipad = 0;
00303 #endif
00304 
00305 #if !defined(WIN32) && !defined(MSDOS)
00306     /*
00307      * You can do "select()" and "poll()" on plain files on most
00308      * platforms, and should be able to do so on pipes.
00309      *
00310      * You can't do "select()" on anything other than sockets in
00311      * Windows, so, on Win32 systems, we don't have "selectable_fd".
00312      */
00313     p->selectable_fd = fileno(fp);
00314 #endif
00315 
00316     p->read_op = pcap_offline_read;
00317     p->inject_op = sf_inject;
00318     p->setfilter_op = install_bpf_program;
00319     p->setdirection_op = sf_setdirection;
00320     p->set_datalink_op = NULL;  /* we don't support munging link-layer headers */
00321     p->getnonblock_op = sf_getnonblock;
00322     p->setnonblock_op = sf_setnonblock;
00323     p->stats_op = sf_stats;
00324 #ifdef WIN32
00325     p->setbuff_op = sf_setbuff;
00326     p->setmode_op = sf_setmode;
00327     p->setmintocopy_op = sf_setmintocopy;
00328 #endif
00329     p->cleanup_op = sf_cleanup;
00330     p->activated = 1;
00331 
00332     return (p);
00333  bad:
00334     free(p);

00338 /*
00339  * Read packets from a capture file, and call the callback for each
00340  * packet.
00341  * If cnt > 0, return after 'cnt' packets, otherwise continue until eof.
00342  */
00343 int
00344 pcap_offline_read(pcap_t *p, int cnt, pcap_handler callback, u_char *user)
00345 {
00346     struct bpf_insn *fcode;
00347     int status = 0;
00348     int n = 0;
00349     u_char *data;
00350 
00351     while (status == 0) {
00352         struct pcap_pkthdr h;
00353 
00354         /*
00355          * Has "pcap_breakloop()" been called?
00356          * If so, return immediately - if we haven't read any
00357          * packets, clear the flag and return -2 to indicate
00358          * that we were told to break out of the loop, otherwise
00359          * leave the flag set, so that the *next* call will break
00360          * out of the loop without having read any packets, and
00361          * return the number of packets we've processed so far.
00362          */
00363         if (p->break_loop) {
00364             if (n == 0) {
00365                 p->break_loop = 0;
00366                 return (-2);
00367             } else
00368                 return (n);
00369         }
00370 
00371         status = p->sf.next_packet_op(p, &h, &data);
00372         if (status) {
00373             if (status == 1)
00374                 return (0);
00375             return (status);
00376         }
00377 
00378         if ((fcode = p->fcode.bf_insns) == NULL ||
00379             bpf_filter(fcode, p->buffer, h.len, h.caplen)) {
00380             (*callback)(user, &h, data);
00381             if (++n >= cnt && cnt > 0)
00382                 break;
00383         }
00384     }
00385     /*XXX this breaks semantics tcpslice expects */
00386     return (n);
00387 }
