00041 s32 ixgbe_dcb_get_tc_stats_82599(struct ixgbe_hw *hw,
00042                                  struct ixgbe_hw_stats *stats,
00043                                  u8 tc_count)
00044 {
00045     int tc;
00046 
00047     if (tc_count > MAX_TRAFFIC_CLASS)
00048         return DCB_ERR_PARAM;
00049     /* Statistics pertaining to each traffic class */
00050     for (tc = 0; tc < tc_count; tc++) {
00051         /* Transmitted Packets */
00052         stats->qptc[tc] += IXGBE_READ_REG(hw, IXGBE_QPTC(tc));
00053         /* Transmitted Bytes (read low first to prevent missed carry) */
00054         stats->qbtc[tc] += IXGBE_READ_REG(hw, IXGBE_QBTC_L(tc));
00055         stats->qbtc[tc] +=
00056             (((u64)(IXGBE_READ_REG(hw, IXGBE_QBTC_H(tc)))) << 32);
00057         /* Received Packets */
00058         stats->qprc[tc] += IXGBE_READ_REG(hw, IXGBE_QPRC(tc));
00059         /* Received Bytes (read low first to prevent missed carry) */
00060         stats->qbrc[tc] += IXGBE_READ_REG(hw, IXGBE_QBRC_L(tc));
00061         stats->qbrc[tc] +=
00062             (((u64)(IXGBE_READ_REG(hw, IXGBE_QBRC_H(tc)))) << 32);
00063 
00064         /* Received Dropped Packet */
00065         stats->qprdc[tc] += IXGBE_READ_REG(hw, IXGBE_QPRDC(tc));
00066     }
00067 
00068     return 0;

00104 s32 ixgbe_dcb_config_packet_buffers_82599(struct ixgbe_hw *hw,
00105                                           struct ixgbe_dcb_config *dcb_config)
00106 {
00107     s32 ret_val = 0;
00108     u32 rxpktsize;
00109     u32 maxtxpktsize = IXGBE_TXPBSIZE_MAX;
00110     u32 txpktsize;
00111     int num_tcs;
00112     u8  i = 0;
00113 
00114     num_tcs = dcb_config->num_tcs.pg_tcs;
00115     /* Setup Rx packet buffer sizes */
00116     if (dcb_config->rx_pba_cfg == pba_80_48) {
00117         /*
00118          * This really means configure the first half of the TCs
00119          * (Traffic Classes) to use 5/8 of the Rx packet buffer
00120          * space.  To determine the size of the buffer for each TC,
00121          * multiply the size of the entire packet buffer by 5/8
00122          * then divide by half of the number of TCs.
00123          */
00124         rxpktsize = (hw->mac.rx_pb_size * 5 / 8) / (num_tcs / 2);
00125         for (i = 0; i < (num_tcs / 2); i++)
00126             IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(i),
00127                             rxpktsize << IXGBE_RXPBSIZE_SHIFT);
00128 
00129         /*
00130          * The second half of the TCs use the remaining 3/8
00131          * of the Rx packet buffer space.
00132          */
00133         rxpktsize = (hw->mac.rx_pb_size * 3 / 8) / (num_tcs / 2);
00134         for (; i < num_tcs; i++)
00135             IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(i),
00136                             rxpktsize << IXGBE_RXPBSIZE_SHIFT);
00137     } else {
00138         /* Divide the Rx packet buffer evenly among the TCs */
00139         rxpktsize = hw->mac.rx_pb_size / num_tcs;
00140         for (i = 0; i < num_tcs; i++)
00141             IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(i),
00142                             rxpktsize << IXGBE_RXPBSIZE_SHIFT);
00143     }
00144     /* Setup remainig TCs, if any, to zero buffer size*/
00145     for (; i < MAX_TRAFFIC_CLASS; i++)
00146         IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(i), 0);
00147 
00148     /* Setup Tx packet buffer and threshold equally for all TCs */
00149     txpktsize = maxtxpktsize/num_tcs;
00150     for (i = 0; i < num_tcs; i++) {
00151         IXGBE_WRITE_REG(hw, IXGBE_TXPBSIZE(i), txpktsize);
00152         IXGBE_WRITE_REG(hw, IXGBE_TXPBTHRESH(i),
00153                         ((txpktsize  / 1024) - IXGBE_TXPKT_SIZE_MAX));
00154     }
00155 
00156     /* Setup remainig TCs, if any, to zero buffer size*/
00157     for (; i < MAX_TRAFFIC_CLASS; i++) {
00158         IXGBE_WRITE_REG(hw, IXGBE_TXPBSIZE(i), 0);
00159         IXGBE_WRITE_REG(hw, IXGBE_TXPBTHRESH(i), 0);
00160     }
00161 
00162     return ret_val;
00163 }

00172 s32 ixgbe_dcb_config_rx_arbiter_82599(struct ixgbe_hw *hw,
00173                                       struct ixgbe_dcb_config *dcb_config)
00174 {
00175     struct tc_bw_alloc    *p;
00176     u32    reg           = 0;
00177     u32    credit_refill = 0;
00178     u32    credit_max    = 0;
00179     u8     i             = 0;
00180     u8     j;
00181 
00182     /*
00183      * Disable the arbiter before changing parameters
00184      * (always enable recycle mode; WSP)
00185      */
00186     reg = IXGBE_RTRPCS_RRM | IXGBE_RTRPCS_RAC | IXGBE_RTRPCS_ARBDIS;
00187     IXGBE_WRITE_REG(hw, IXGBE_RTRPCS, reg);
00188 
00189     /*
00190      * map all UPs to TCs. up_to_tc_bitmap for each TC has corresponding
00191      * bits sets for the UPs that needs to be mappped to that TC.
00192      * e.g if priorities 6 and 7 are to be mapped to a TC then the
00193      * up_to_tc_bitmap value for that TC will be 11000000 in binary.
00194      */
00195     reg = 0;
00196     for (i = 0; i < MAX_TRAFFIC_CLASS; i++) {
00197         p = &dcb_config->tc_config[i].path[DCB_RX_CONFIG];
00198         for (j = 0; j < MAX_USER_PRIORITY; j++) {
00199             if (p->up_to_tc_bitmap & (1 << j))
00200                 reg |= (i << (j * IXGBE_RTRUP2TC_UP_SHIFT));
00201         }
00202     }
00203     IXGBE_WRITE_REG(hw, IXGBE_RTRUP2TC, reg);
00204 
00205     /* Configure traffic class credits and priority */
00206     for (i = 0; i < dcb_config->num_tcs.pg_tcs; i++) {
00207         p = &dcb_config->tc_config[i].path[DCB_RX_CONFIG];
00208 
00209         credit_refill = p->data_credits_refill;
00210         credit_max    = p->data_credits_max;
00211         reg = credit_refill | (credit_max << IXGBE_RTRPT4C_MCL_SHIFT);
00212 
00213         reg |= (u32)(p->bwg_id) << IXGBE_RTRPT4C_BWG_SHIFT;
00214 
00215         if (p->prio_type == prio_link)
00216             reg |= IXGBE_RTRPT4C_LSP;
00217 
00218         IXGBE_WRITE_REG(hw, IXGBE_RTRPT4C(i), reg);
00219     }
00220 
00221     /*
00222      * Configure Rx packet plane (recycle mode; WSP) and
00223      * enable arbiter
00224      */
00225     reg = IXGBE_RTRPCS_RRM | IXGBE_RTRPCS_RAC;
00226     IXGBE_WRITE_REG(hw, IXGBE_RTRPCS, reg);
00227 
00228     return 0;
00229 }

00350 s32 ixgbe_dcb_config_pfc_82599(struct ixgbe_hw *hw,
00351                                struct ixgbe_dcb_config *dcb_config)
00352 {
00353     u32 i, reg, rx_pba_size;
00354 
00355     /* If PFC is disabled globally then fall back to LFC. */
00356     if (!dcb_config->pfc_mode_enable) {
00357         for (i = 0; i < dcb_config->num_tcs.pg_tcs; i++)
00358             hw->mac.ops.fc_enable(hw, i);

00360     }
00361

00363     for (i = 0; i < dcb_config->num_tcs.pg_tcs; i++) {
00364         rx_pba_size = IXGBE_READ_REG(hw, IXGBE_RXPBSIZE(i));
00365         rx_pba_size >>= IXGBE_RXPBSIZE_SHIFT;

00368

00370             dcb_config->tc_config[i].dcb_pfc == pfc_enabled_tx)
00371             reg |= IXGBE_FCRTL_XONE;
00372         IXGBE_WRITE_REG(hw, IXGBE_FCRTL_82599(i), reg);
00373 
00374         reg = (rx_pba_size - hw->fc.high_water) << 10;
00375         if (dcb_config->tc_config[i].dcb_pfc == pfc_enabled_full ||
00376             dcb_config->tc_config[i].dcb_pfc == pfc_enabled_tx)
00377             reg |= IXGBE_FCRTH_FCEN;
00378         IXGBE_WRITE_REG(hw, IXGBE_FCRTH_82599(i), reg);
00379     }
00380 
00381     /* Configure pause time (2 TCs per register) */
00382     reg = hw->fc.pause_time | (hw->fc.pause_time << 16);
00383     for (i = 0; i < (MAX_TRAFFIC_CLASS / 2); i++)
00384         IXGBE_WRITE_REG(hw, IXGBE_FCTTV(i), reg);
00385 
00386     /* Configure flow control refresh threshold value */
00387     IXGBE_WRITE_REG(hw, IXGBE_FCRTV, hw->fc.pause_time / 2);
00388 
00389     /* Enable Transmit PFC */
00390     reg = IXGBE_FCCFG_TFCE_PRIORITY;
00391     IXGBE_WRITE_REG(hw, IXGBE_FCCFG, reg);

00413 s32 ixgbe_dcb_config_tc_stats_82599(struct ixgbe_hw *hw)
00414 {
00415     u32 reg = 0;
00416     u8  i   = 0;
00417 
00418     /*
00419      * Receive Queues stats setting
00420      * 32 RQSMR registers, each configuring 4 queues.
00421      * Set all 16 queues of each TC to the same stat
00422      * with TC 'n' going to stat 'n'.
00423      */
00424     for (i = 0; i < 32; i++) {
00425         reg = 0x01010101 * (i / 4);
00426         IXGBE_WRITE_REG(hw, IXGBE_RQSMR(i), reg);
00427     }
00428     /*
00429      * Transmit Queues stats setting
00430      * 32 TQSM registers, each controlling 4 queues.
00431      * Set all queues of each TC to the same stat
00432      * with TC 'n' going to stat 'n'.
00433      * Tx queues are allocated non-uniformly to TCs:
00434      * 32, 32, 16, 16, 8, 8, 8, 8.
00435      */
00436     for (i = 0; i < 32; i++) {
00437         if (i < 8)
00438             reg = 0x00000000;
00439         else if (i < 16)
00440             reg = 0x01010101;
00441         else if (i < 20)
00442             reg = 0x02020202;
00443         else if (i < 24)
00444             reg = 0x03030303;
00445         else if (i < 26)
00446             reg = 0x04040404;
00447         else if (i < 28)
00448             reg = 0x05050505;
00449         else if (i < 30)
00450             reg = 0x06060606;
00451         else
00452             reg = 0x07070707;
00453         IXGBE_WRITE_REG(hw, IXGBE_TQSM(i), reg);
00454     }

00466 s32 ixgbe_dcb_config_82599(struct ixgbe_hw *hw,
00467                            struct ixgbe_dcb_config *dcb_config)
00468 {
00469     u32 reg;
00470     u32 q;
00471 
00472     /* Disable the Tx desc arbiter so that MTQC can be changed */
00473     reg = IXGBE_READ_REG(hw, IXGBE_RTTDCS);
00474     reg |= IXGBE_RTTDCS_ARBDIS;
00475     IXGBE_WRITE_REG(hw, IXGBE_RTTDCS, reg);

00477     /* Enable DCB for Rx with 8 TCs */

00480     case 0:

00484         break;
00485     case IXGBE_MRQC_RSSEN:
00486     case IXGBE_MRQC_RTRSS4TCEN:
00487         /* RSS enabled cases */

00489         break;
00490     default:
00491         /* Unsupported value, assume stale data, overwrite no RSS */
00492         reg = (reg & ~IXGBE_MRQC_MRQE_MASK) | IXGBE_MRQC_RT8TCEN;
00493     }
00494     if (dcb_config->num_tcs.pg_tcs == 4) {
00495         /* Enable DCB for Rx with 4 TCs and VT Mode*/
00496         reg = (reg & ~IXGBE_MRQC_MRQE_MASK) | IXGBE_MRQC_VMDQRT4TCEN;
00497     }
00498     IXGBE_WRITE_REG(hw, IXGBE_MRQC, reg);
00499 
00500     /* Enable DCB for Tx with 8 TCs */
00501     if (dcb_config->num_tcs.pg_tcs == 8)
00502         reg = IXGBE_MTQC_RT_ENA | IXGBE_MTQC_8TC_8TQ;
00503     else /* Enable DCB for Tx with 4 TCs and VT Mode*/
00504         reg = IXGBE_MTQC_RT_ENA | IXGBE_MTQC_VT_ENA
00505                     | IXGBE_MTQC_4TC_4TQ;
00506     IXGBE_WRITE_REG(hw, IXGBE_MTQC, reg);
00507 
00508     /* Disable drop for all queues */
00509     for (q=0; q < 128; q++) {
00510         IXGBE_WRITE_REG(hw, IXGBE_QDE, q << IXGBE_QDE_IDX_SHIFT);
00511     }
00512 
00513     /* Enable the Tx desc arbiter */
00514     reg = IXGBE_READ_REG(hw, IXGBE_RTTDCS);
00515     reg &= ~IXGBE_RTTDCS_ARBDIS;
00516     IXGBE_WRITE_REG(hw, IXGBE_RTTDCS, reg);
00517 
00518     /* Enable Security TX Buffer IFG for DCB */

00520     reg |= IXGBE_SECTX_DCB;
00521     IXGBE_WRITE_REG(hw, IXGBE_SECTXMINIFG, reg);
