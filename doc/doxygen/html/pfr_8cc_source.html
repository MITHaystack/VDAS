00060 int PFR::add_bpf_filter(char *the_filter) {
00061   struct bpf_program fcode;
00062 
00063   if(pcapPtr == NULL) return(-1);
00064   if(the_filter == NULL) return(0);
00065 
00066   if(the_filter != NULL) {
00067     if(pcap_compile(pcapPtr, &fcode, the_filter, 1, 0xFFFFFF00) < 0) {
00068       return(-2);
00069     } else {
00070       if(pcap_setfilter(pcapPtr, &fcode) < 0) {
00071     pcap_freecode(&fcode);
00072     return(-3);
00073       } else
00074     pcap_freecode(&fcode);
00075     }    
00076   } else
00077     return(0);
00078 }

00082 int PFR::get_next_packet(struct pfring_pkthdr *hdr, const u_char *pkt, u_int pkt_len) {
00083   if((!pcapPtr) || (!hdr) || (pkt_len < snaplen)) return(-1);
00084   
00085   if(pcapPtr->ring) {
00086     return(pfring_recv(pcapPtr->ring, (u_char**)&pkt, pkt_len, hdr, 1 /* wait_for_incoming_packet */));
00087   } else {
00088     pcap_pkthdr *_hdr;
00089     int rc;
00090 
00091     rc = pcap_next_ex(pcapPtr, &_hdr, &pkt);
00092     memcpy(hdr, _hdr, sizeof(struct pcap_pkthdr));
00093     return(rc);
00094   }
00095 }

00099 bool PFR::wait_for_packets(int msec) {
00100   struct pollfd pfd;
00101   int rc;
00102 
00103   /* Sleep when nothing is happening */
00104   pfd.fd      = get_socket_id();
00105   pfd.events  = POLLIN|POLLERR;
00106   pfd.revents = 0;
00107 
00108   errno = 0;
00109   rc = poll(&pfd, 1, msec);
00110     
00111   if(rc == -1)
00112     return(false);
00113   else
00114     return((rc > 0) ? true : false);
