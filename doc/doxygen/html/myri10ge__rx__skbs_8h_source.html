00046 /*
00047  * Set of routines to get a new receive buffer.  Any buffer which
00048  * crosses a 4KB boundary must start on a 4KB boundary due to PCIe
00049  * wdma restrictions. We also try to align any smaller allocation to
00050  * at least a 16 byte boundary for efficiency.  We assume the linux
00051  * memory allocator works by powers of 2, and will not return memory
00052  * smaller than 2KB which crosses a 4KB boundary.  If it does, we fall
00053  * back to allocating 2x as much space as required.
00054  *
00055  * We intend to replace large (>4KB) skb allocations by using
00056  * pages directly and building a fraglist in the near future.
00057  */

00096    does not cross a 4KB boundary */
00097 static inline struct sk_buff *
00098 myri10ge_alloc_small_safe(struct net_device *dev, unsigned int bytes)
00099 {
00100     struct sk_buff *skb;
00101     unsigned long data, boundary;
00102 
00103     skb = myri10ge_netdev_alloc_skb(dev, 2 * (bytes + MXGEFW_PAD) - 1);
00104     if (unlikely(skb == NULL))
00105         return NULL;
00106 
00107     /* Correct skb->truesize so that socket buffer
00108      * accounting is not confused the rounding we must
00109      * do to satisfy alignment constraints.
00110      */
00111     myri10ge_reduce_truesize(skb, bytes + MXGEFW_PAD);
00112 
00113     data = (unsigned long)(skb->data);
00114     boundary = (data + 4095UL) & ~4095UL;
00115     if ((boundary - data) >= (bytes + MXGEFW_PAD))
00116         return skb;
00117 
00118     skb_reserve(skb, boundary - data);
00119     return skb;

00125    space does not cross a 4KB boundary */
00126 static inline struct sk_buff *
00127 myri10ge_alloc_small(struct net_device *dev, int bytes)
00128 {
00129     struct sk_buff *skb;
00130     unsigned long roundup, data, end;
00131 
00132     skb = myri10ge_netdev_alloc_skb(dev, bytes + MYRI10GE_RX_ALIGN + MXGEFW_PAD);
00133     if (unlikely(skb == NULL))
00134         return NULL;
00135 
00136     /* Round allocated buffer to 16 byte boundary */
00137     data = (unsigned long)(skb->data);
00138     roundup = (-data) & (MYRI10GE_RX_ALIGN - 1UL);
00139     skb_reserve(skb, roundup);
00140     /* Verify that the data buffer does not cross a page boundary */
00141     data = (unsigned long)(skb->data);
00142     end = data + bytes + MXGEFW_PAD - 1;
00143 #ifndef MYRI10GE_RELAX_RX_ALIGN
00144     if (unlikely (((end >> 12) != (data >> 12)) && (data & 4095UL))) {
00145         printk(KERN_NOTICE "myri10ge_alloc_small: small skb crossed 4KB boundary\n");
00146         myri10ge_skb_cross_4k = 1;
00147         dev_kfree_skb_any(skb);
00148         skb = myri10ge_alloc_small_safe(dev, bytes);
00149     }
00150 #endif
00151     return skb;
00152 }
