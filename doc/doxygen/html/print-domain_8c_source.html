00056 static const u_char *
00057 ns_nskip(register const u_char *cp)
00058 {
00059     register u_char i;
00060 
00061     if (!TTEST2(*cp, 1))
00062         return (NULL);
00063     i = *cp++;
00064     while (i) {
00065         if ((i & INDIR_MASK) == INDIR_MASK)
00066             return (cp + 1);
00067         if ((i & INDIR_MASK) == EDNS0_MASK) {
00068             int bitlen, bytelen;
00069 
00070             if ((i & ~INDIR_MASK) != EDNS0_ELT_BITLABEL)
00071                 return(NULL); /* unknown ELT */
00072             if (!TTEST2(*cp, 1))
00073                 return (NULL);
00074             if ((bitlen = *cp++) == 0)
00075                 bitlen = 256;
00076             bytelen = (bitlen + 7) / 8;
00077             cp += bytelen;
00078         } else
00079             cp += i;
00080         if (!TTEST2(*cp, 1))
00081             return (NULL);
00082         i = *cp++;
00083     }
00084     return (cp);
00085 }

00088 static const u_char *
00089 blabel_print(const u_char *cp)
00090 {
00091     int bitlen, slen, b;
00092     const u_char *bitp, *lim;
00093     char tc;
00094 
00095     if (!TTEST2(*cp, 1))
00096         return(NULL);
00097     if ((bitlen = *cp) == 0)
00098         bitlen = 256;
00099     slen = (bitlen + 3) / 4;
00100     lim = cp + 1 + slen;
00101 
00102     /* print the bit string as a hex string */
00103     printf("\\[x");
00104     for (bitp = cp + 1, b = bitlen; bitp < lim && b > 7; b -= 8, bitp++) {
00105         TCHECK(*bitp);
00106         printf("%02x", *bitp);
00107     }
00108     if (b > 4) {
00109         TCHECK(*bitp);
00110         tc = *bitp++;
00111         printf("%02x", tc & (0xff << (8 - b)));
00112     } else if (b > 0) {
00113         TCHECK(*bitp);
00114         tc = *bitp++;
00115         printf("%1x", ((tc >> 4) & 0x0f) & (0x0f << (4 - b)));
00116     }
00117     printf("/%d]", bitlen);
00118     return lim;
00119 trunc:
00120     printf(".../%d]", bitlen);
00121     return NULL;
00122 }

00147 static const u_char *
00148 ns_nprint(register const u_char *cp, register const u_char *bp)
00149 {
00150     register u_int i, l;
00151     register const u_char *rp = NULL;
00152     register int compress = 0;
00153     int chars_processed;
00154     int elt;
00155     int data_size = snapend - bp;
00156 
00157     if ((l = labellen(cp)) == (u_int)-1)
00158         return(NULL);
00159     if (!TTEST2(*cp, 1))
00160         return(NULL);
00161     chars_processed = 1;
00162     if (((i = *cp++) & INDIR_MASK) != INDIR_MASK) {
00163         compress = 0;
00164         rp = cp + l;
00165     }
00166 
00167     if (i != 0)
00168         while (i && cp < snapend) {
00169             if ((i & INDIR_MASK) == INDIR_MASK) {
00170                 if (!compress) {
00171                     rp = cp + 1;
00172                     compress = 1;
00173                 }
00174                 if (!TTEST2(*cp, 1))
00175                     return(NULL);
00176                 cp = bp + (((i << 8) | *cp) & 0x3fff);
00177                 if ((l = labellen(cp)) == (u_int)-1)
00178                     return(NULL);
00179                 if (!TTEST2(*cp, 1))
00180                     return(NULL);
00181                 i = *cp++;
00182                 chars_processed++;
00183 
00184                 /*
00185                  * If we've looked at every character in
00186                  * the message, this pointer will make
00187                  * us look at some character again,
00188                  * which means we're looping.
00189                  */
00190                 if (chars_processed >= data_size) {
00191                     printf("<LOOP>");
00192                     return (NULL);
00193                 }
00194                 continue;
00195             }
00196             if ((i & INDIR_MASK) == EDNS0_MASK) {
00197                 elt = (i & ~INDIR_MASK);
00198                 switch(elt) {
00199                 case EDNS0_ELT_BITLABEL:
00200                     if (blabel_print(cp) == NULL)
00201                         return (NULL);
00202                     break;
00203                 default:
00204                     /* unknown ELT */
00205                     printf("<ELT %d>", elt);
00206                     return(NULL);
00207                 }
00208             } else {
00209                 if (fn_printn(cp, l, snapend))
00210                     return(NULL);
00211             }
00212 
00213             cp += l;
00214             chars_processed += l;
00215             putchar('.');
00216             if ((l = labellen(cp)) == (u_int)-1)
00217                 return(NULL);
00218             if (!TTEST2(*cp, 1))
00219                 return(NULL);
00220             i = *cp++;
00221             chars_processed++;
00222             if (!compress)
00223                 rp += l + 1;
00224         }
00225     else
00226         putchar('.');
00227     return (rp);
00228 }

00319 static const u_char *
00320 ns_qprint(register const u_char *cp, register const u_char *bp, int is_mdns)
00321 {
00322     register const u_char *np = cp;
00323     register u_int i, class;
00324 
00325     cp = ns_nskip(cp);
00326 
00327     if (cp == NULL || !TTEST2(*cp, 4))
00328         return(NULL);
00329 
00330     /* print the qtype */
00331     i = EXTRACT_16BITS(cp);
00332     cp += 2;
00333     printf(" %s", tok2str(ns_type2str, "Type%d", i));
00334     /* print the qclass (if it's not IN) */
00335     i = EXTRACT_16BITS(cp);
00336     cp += 2;
00337     if (is_mdns)
00338         class = (i & ~C_QU);

00340         class = i;

00342         printf(" %s", tok2str(ns_class2str, "(Class %d)", class));
00343     if (is_mdns) {
00344         if (i & C_QU)
00345             printf(" (QU)");
00346         else
00347             printf(" (QM)");

00350     fputs("? ", stdout);

00352     return(cp ? cp + 4 : NULL);
