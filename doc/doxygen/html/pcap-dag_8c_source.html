00188 static unsigned int
00189 dag_erf_ext_header_count(uint8_t * erf, size_t len)
00190 {
00191     uint32_t hdr_num = 0;
00192     uint8_t  hdr_type;
00193 
00194     /* basic sanity checks */
00195     if ( erf == NULL )
00196         return 0;
00197     if ( len < 16 )
00198         return 0;
00199 
00200     /* check if we have any extension headers */
00201     if ( (erf[8] & 0x80) == 0x00 )
00202         return 0;
00203 
00204     /* loop over the extension headers */
00205     do {
00206     
00207         /* sanity check we have enough bytes */
00208         if ( len <= (24 + (hdr_num * 8)) )
00209             return hdr_num;
00210 
00211         /* get the header type */
00212         hdr_type = erf[(16 + (hdr_num * 8))];
00213         hdr_num++;
00214 
00215     } while ( hdr_type & 0x80 );
00216 
00217     return hdr_num;
00218 }

00220 /*
00221  *  Read at most max_packets from the capture stream and call the callback
00222  *  for each of them. Returns the number of packets handled, -1 if an
00223  *  error occured, or -2 if we were told to break out of the loop.
00224  */
00225 static int
00226 dag_read(pcap_t *p, int cnt, pcap_handler callback, u_char *user)
00227 {
00228     unsigned int processed = 0;
00229     int flags = p->md.dag_offset_flags;
00230     unsigned int nonblocking = flags & DAGF_NONBLOCK;
00231     unsigned int num_ext_hdr = 0;
00232 
00233     /* Get the next bufferful of packets (if necessary). */
00234     while (p->md.dag_mem_top - p->md.dag_mem_bottom < dag_record_size) {
00235  
00236         /*
00237          * Has "pcap_breakloop()" been called?
00238          */
00239         if (p->break_loop) {
00240             /*
00241              * Yes - clear the flag that indicates that
00242              * it has, and return -2 to indicate that
00243              * we were told to break out of the loop.
00244              */
00245             p->break_loop = 0;
00246             return -2;
00247         }
00248 
00249 #ifdef HAVE_DAG_STREAMS_API
00250         /* dag_advance_stream() will block (unless nonblock is called)
00251          * until 64kB of data has accumulated.
00252          * If to_ms is set, it will timeout before 64kB has accumulated.
00253          * We wait for 64kB because processing a few packets at a time
00254          * can cause problems at high packet rates (>200kpps) due
00255          * to inefficiencies.
00256          * This does mean if to_ms is not specified the capture may 'hang'
00257          * for long periods if the data rate is extremely slow (<64kB/sec)
00258          * If non-block is specified it will return immediately. The user
00259          * is then responsible for efficiency.
00260          */
00261         if ( NULL == (p->md.dag_mem_top = dag_advance_stream(p->fd, p->md.dag_stream, &(p->md.dag_mem_bottom))) ) {
00262              return -1;
00263         }
00264 #else
00265         /* dag_offset does not support timeouts */
00266         p->md.dag_mem_top = dag_offset(p->fd, &(p->md.dag_mem_bottom), flags);
00267 #endif /* HAVE_DAG_STREAMS_API */
00268 
00269         if (nonblocking && (p->md.dag_mem_top - p->md.dag_mem_bottom < dag_record_size))
00270         {
00271             /* Pcap is configured to process only available packets, and there aren't any, return immediately. */
00272             return 0;
00273         }
00274         
00275         if(!nonblocking &&
00276            p->md.dag_timeout &&
00277            (p->md.dag_mem_top - p->md.dag_mem_bottom < dag_record_size))
00278         {
00279             /* Blocking mode, but timeout set and no data has arrived, return anyway.*/
00280             return 0;
00281         }
00282 
00283     }
00284     
00285     /* Process the packets. */
00286     while (p->md.dag_mem_top - p->md.dag_mem_bottom >= dag_record_size) {
00287         
00288         unsigned short packet_len = 0;
00289         int caplen = 0;
00290         struct pcap_pkthdr  pcap_header;
00291         
00292 #ifdef HAVE_DAG_STREAMS_API
00293         dag_record_t *header = (dag_record_t *)(p->md.dag_mem_bottom);
00294 #else
00295         dag_record_t *header = (dag_record_t *)(p->md.dag_mem_base + p->md.dag_mem_bottom);
00296 #endif /* HAVE_DAG_STREAMS_API */
00297 
00298         u_char *dp = ((u_char *)header); /* + dag_record_size; */
00299         unsigned short rlen;
00300  
00301         /*
00302          * Has "pcap_breakloop()" been called?
00303          */
00304         if (p->break_loop) {
00305             /*
00306              * Yes - clear the flag that indicates that
00307              * it has, and return -2 to indicate that
00308              * we were told to break out of the loop.
00309              */
00310             p->break_loop = 0;
00311             return -2;
00312         }
00313  
00314         rlen = ntohs(header->rlen);
00315         if (rlen < dag_record_size)
00316         {
00317             strncpy(p->errbuf, "dag_read: record too small", PCAP_ERRBUF_SIZE);
00318             return -1;
00319         }
00320         p->md.dag_mem_bottom += rlen;
00321 
00322         /* Count lost packets. */
00323         switch((header->type & 0x7f)) {
00324             /* in these types the color value overwrites the lctr */
00325         case TYPE_COLOR_HDLC_POS:
00326         case TYPE_COLOR_ETH:
00327         case TYPE_DSM_COLOR_HDLC_POS:
00328         case TYPE_DSM_COLOR_ETH:
00329         case TYPE_COLOR_MC_HDLC_POS:
00330         case TYPE_COLOR_HASH_ETH:
00331         case TYPE_COLOR_HASH_POS:
00332             break;
00333 
00334         default:
00335             if (header->lctr) {
00336                 if (p->md.stat.ps_drop > (UINT_MAX - ntohs(header->lctr))) {
00337                     p->md.stat.ps_drop = UINT_MAX;
00338                 } else {
00339                     p->md.stat.ps_drop += ntohs(header->lctr);
00340                 }
00341             }
00342         }
00343         
00344         if ((header->type & 0x7f) == TYPE_PAD) {
00345             continue;
00346         }
00347 
00348         num_ext_hdr = dag_erf_ext_header_count(dp, rlen);
00349 
00350         /* ERF encapsulation */
00351         /* The Extensible Record Format is not dropped for this kind of encapsulation, 
00352          * and will be handled as a pseudo header by the decoding application.
00353          * The information carried in the ERF header and in the optional subheader (if present)
00354          * could be merged with the libpcap information, to offer a better decoding.
00355          * The packet length is
00356          * o the length of the packet on the link (header->wlen),
00357          * o plus the length of the ERF header (dag_record_size), as the length of the 
00358          *   pseudo header will be adjusted during the decoding,
00359          * o plus the length of the optional subheader (if present).
00360          *
00361          * The capture length is header.rlen and the byte stuffing for alignment will be dropped
00362          * if the capture length is greater than the packet length.
00363          */
00364         if (p->linktype == DLT_ERF) {
00365             packet_len = ntohs(header->wlen) + dag_record_size;
00366             caplen = rlen;
00367             switch ((header->type & 0x7f)) {
00368             case TYPE_MC_AAL5:
00369             case TYPE_MC_ATM:
00370             case TYPE_MC_HDLC:
00371             case TYPE_MC_RAW_CHANNEL:
00372             case TYPE_MC_RAW:
00373             case TYPE_MC_AAL2:
00374             case TYPE_COLOR_MC_HDLC_POS:
00375                 packet_len += 4; /* MC header */
00376                 break;
00377 
00378             case TYPE_COLOR_HASH_ETH:
00379             case TYPE_DSM_COLOR_ETH:
00380             case TYPE_COLOR_ETH:
00381             case TYPE_ETH:
00382                 packet_len += 2; /* ETH header */
00383                 break;
00384             } /* switch type */
00385 
00386             /* Include ERF extension headers */
00387             packet_len += (8 * num_ext_hdr);
00388 
00389             if (caplen > packet_len) {
00390                 caplen = packet_len;
00391             }
00392         } else {
00393             /* Other kind of encapsulation according to the header Type */
00394 
00395             /* Skip over generic ERF header */
00396             dp += dag_record_size;
00397             /* Skip over extension headers */
00398             dp += 8 * num_ext_hdr;
00399             
00400             switch((header->type & 0x7f)) {
00401             case TYPE_ATM:
00402             case TYPE_AAL5:
00403                 if (header->type == TYPE_AAL5) {
00404                     packet_len = ntohs(header->wlen);
00405                     caplen = rlen - dag_record_size;
00406                 }
00407             case TYPE_MC_ATM:
00408                 if (header->type == TYPE_MC_ATM) {
00409                     caplen = packet_len = ATM_CELL_SIZE;
00410                     dp+=4;
00411                 }
00412             case TYPE_MC_AAL5:
00413                 if (header->type == TYPE_MC_AAL5) {
00414                     packet_len = ntohs(header->wlen);
00415                     caplen = rlen - dag_record_size - 4;
00416                     dp+=4;
00417                 }
00418                 if (header->type == TYPE_ATM) {
00419                     caplen = packet_len = ATM_CELL_SIZE;
00420                 }
00421                 if (p->linktype == DLT_SUNATM) {
00422                     struct sunatm_hdr *sunatm = (struct sunatm_hdr *)dp;
00423                     unsigned long rawatm;
00424                     
00425                     rawatm = ntohl(*((unsigned long *)dp));
00426                     sunatm->vci = htons((rawatm >>  4) & 0xffff);
00427                     sunatm->vpi = (rawatm >> 20) & 0x00ff;
00428                     sunatm->flags = ((header->flags.iface & 1) ? 0x80 : 0x00) | 
00429                         ((sunatm->vpi == 0 && sunatm->vci == htons(5)) ? 6 :
00430                          ((sunatm->vpi == 0 && sunatm->vci == htons(16)) ? 5 : 
00431                           ((dp[ATM_HDR_SIZE] == 0xaa &&
00432                             dp[ATM_HDR_SIZE+1] == 0xaa &&
00433                             dp[ATM_HDR_SIZE+2] == 0x03) ? 2 : 1)));
00434 
00435                 } else {
00436                     packet_len -= ATM_HDR_SIZE;
00437                     caplen -= ATM_HDR_SIZE;
00438                     dp += ATM_HDR_SIZE;
00439                 }
00440                 break;
00441 
00442             case TYPE_COLOR_HASH_ETH:
00443             case TYPE_DSM_COLOR_ETH:
00444             case TYPE_COLOR_ETH:
00445             case TYPE_ETH:
00446                 packet_len = ntohs(header->wlen);
00447                 packet_len -= (p->md.dag_fcs_bits >> 3);
00448                 caplen = rlen - dag_record_size - 2;
00449                 if (caplen > packet_len) {
00450                     caplen = packet_len;
00451                 }
00452                 dp += 2;
00453                 break;
00454 
00455             case TYPE_COLOR_HASH_POS:
00456             case TYPE_DSM_COLOR_HDLC_POS:
00457             case TYPE_COLOR_HDLC_POS:
00458             case TYPE_HDLC_POS:
00459                 packet_len = ntohs(header->wlen);
00460                 packet_len -= (p->md.dag_fcs_bits >> 3);
00461                 caplen = rlen - dag_record_size;
00462                 if (caplen > packet_len) {
00463                     caplen = packet_len;
00464                 }
00465                 break;
00466 
00467             case TYPE_COLOR_MC_HDLC_POS:
00468             case TYPE_MC_HDLC:
00469                 packet_len = ntohs(header->wlen);
00470                 packet_len -= (p->md.dag_fcs_bits >> 3);
00471                 caplen = rlen - dag_record_size - 4;
00472                 if (caplen > packet_len) {
00473                     caplen = packet_len;
00474                 }
00475                 /* jump the MC_HDLC_HEADER */
00476                 dp += 4;
00477 #ifdef DLT_MTP2_WITH_PHDR
00478                 if (p->linktype == DLT_MTP2_WITH_PHDR) {
00479                     /* Add the MTP2 Pseudo Header */
00480                     caplen += MTP2_HDR_LEN;
00481                     packet_len += MTP2_HDR_LEN;
00482                     
00483                     TempPkt[MTP2_SENT_OFFSET] = 0;
00484                     TempPkt[MTP2_ANNEX_A_USED_OFFSET] = MTP2_ANNEX_A_USED_UNKNOWN;
00485                     *(TempPkt+MTP2_LINK_NUMBER_OFFSET) = ((header->rec.mc_hdlc.mc_header>>16)&0x01);
00486                     *(TempPkt+MTP2_LINK_NUMBER_OFFSET+1) = ((header->rec.mc_hdlc.mc_header>>24)&0xff);
00487                     memcpy(TempPkt+MTP2_HDR_LEN, dp, caplen);
00488                     dp = TempPkt;
00489                 }
00490 #endif
00491                 break;
00492 
00493             case TYPE_IPV4:
00494             case TYPE_IPV6:
00495                 packet_len = ntohs(header->wlen);
00496                 caplen = rlen - dag_record_size;
00497                 if (caplen > packet_len) {
00498                     caplen = packet_len;
00499                 }
00500                 break;
00501 
00502             /* These types have no matching 'native' DLT, but can be used with DLT_ERF above */
00503             case TYPE_MC_RAW:
00504             case TYPE_MC_RAW_CHANNEL:
00505             case TYPE_IP_COUNTER:
00506             case TYPE_TCP_FLOW_COUNTER:
00507             case TYPE_INFINIBAND:
00508             case TYPE_RAW_LINK:
00509             case TYPE_INFINIBAND_LINK:
00510             default:
00511                 /* Unhandled ERF type.
00512                  * Ignore rather than generating error
00513                  */
00514                 continue;
00515             } /* switch type */
00516 
00517             /* Skip over extension headers */
00518             caplen -= (8 * num_ext_hdr);
00519 
00520         } /* ERF encapsulation */
00521         
00522         if (caplen > p->snapshot)
00523             caplen = p->snapshot;
00524 
00525         /* Run the packet filter if there is one. */
00526         if ((p->fcode.bf_insns == NULL) || bpf_filter(p->fcode.bf_insns, dp, packet_len, caplen)) {
00527             
00528             /* convert between timestamp formats */
00529             register unsigned long long ts;
00530                 
00531             if (IS_BIGENDIAN()) {
00532                 ts = SWAPLL(header->ts);
00533             } else {
00534                 ts = header->ts;
00535             }
00536 
00537             pcap_header.ts.tv_sec = ts >> 32;
00538             ts = (ts & 0xffffffffULL) * 1000000;
00539             ts += 0x80000000; /* rounding */
00540             pcap_header.ts.tv_usec = ts >> 32;      
00541             if (pcap_header.ts.tv_usec >= 1000000) {
00542                 pcap_header.ts.tv_usec -= 1000000;
00543                 pcap_header.ts.tv_sec++;
00544             }
00545 
00546             /* Fill in our own header data */
00547             pcap_header.caplen = caplen;
00548             pcap_header.len = packet_len;
00549     
00550             /* Count the packet. */
00551             p->md.stat.ps_recv++;
00552     
00553             /* Call the user supplied callback function */
00554             callback(user, &pcap_header, dp);
00555     
00556             /* Only count packets that pass the filter, for consistency with standard Linux behaviour. */
00557             processed++;
00558             if (processed == cnt && cnt > 0)
00559             {
00560                 /* Reached the user-specified limit. */
00561                 return cnt;
00562             }
00563         }
00564     }
00565 
00566     return processed;
00567 }
