00096 /*
00097  * Header prepended by linux kernel to each event for the 2.6.31
00098  * and later kernels; for the 2.6.21 through 2.6.30 kernels, the
00099  * "iso_rec" information, and the fields starting with "interval"
00100  * are zeroed-out padding fields.
00101  *
00102  * Appears at the front of each packet in DLT_USB_LINUX_MMAPPED captures.
00103  */
00104 typedef struct _usb_header_mmapped {
00105     u_int64_t id;
00106     u_int8_t event_type;
00107     u_int8_t transfer_type;
00108     u_int8_t endpoint_number;
00109     u_int8_t device_address;
00110     u_int16_t bus_id;
00111     char setup_flag;/*if !=0 the urb setup header is not present*/
00112     char data_flag; /*if !=0 no urb data is present*/
00113     int64_t ts_sec;
00114     int32_t ts_usec;
00115     int32_t status;
00116     u_int32_t urb_len;
00117     u_int32_t data_len; /* amount of urb data really present in this event*/
00118     union {
00119         pcap_usb_setup setup;
00120         iso_rec iso;
00121     } s;
00122     int32_t interval;   /* for Interrupt and Isochronous events */
00123     int32_t start_frame;    /* for Isochronous events */
00124     u_int32_t xfer_flags;   /* copy of URB's transfer flags */
00125     u_int32_t ndesc;    /* number of isochronous descriptors */
00126 } pcap_usb_header_mmapped;

00128 /*
00129  * Isochronous descriptors; for isochronous transfers there might be
00130  * one or more of these at the beginning of the packet data.  The
00131  * number of descriptors is given by the "ndesc" field in the header;
00132  * as indicated, in older kernels that don't put the descriptors at

00136 typedef struct _usb_isodesc {
