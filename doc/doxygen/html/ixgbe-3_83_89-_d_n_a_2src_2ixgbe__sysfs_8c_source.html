00035 /* Ethernet payload size for FCoE to be able to carry full sized FC Frames
00036  * 14 byte FCoE header + 24 byte FC header + 2112 max payload + 4 byte CRC
00037  *  + 4 byte FCoE trailing encapsulation = 2158
00038  * This is the Ethernet payload, replacing the default of 1500, and does
00039  * not include Ethernet headers, VLAN tags, or Ethernet CRC.
00040  */

00052 int ixgbe_sysfs_create(struct ixgbe_adapter *adapter)
00053 {
00054     struct net_device *netdev = adapter->netdev;
00055     int err;
00056     int i;
00057 
00058     for (i = 0 ; i < ARRAY_SIZE(ixgbe_attrs); i++) {
00059         err = device_create_file(&netdev->dev, &ixgbe_attrs[i]);
00060         if (err)
00061             goto fail;
00062     }
00063     return 0;
00064 
00065 fail:
00066     while (i-- >= 0)
00067         device_remove_file(&netdev->dev, &ixgbe_attrs[i]);
00068     return err;
00069 }
