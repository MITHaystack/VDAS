00524 /*
00525  * Return true if the 'fcode' is a valid filter program.
00526  * The constraints are that each jump be forward and to a valid
00527  * code, that memory accesses are within valid ranges (to the
00528  * extent that this can be checked statically; loads of packet
00529  * data have to be, and are, also checked at run time), and that
00530  * the code terminates with either an accept or reject.
00531  *
00532  * The kernel needs to be able to verify an application's filter code.
00533  * Otherwise, a bogus program could easily crash the system.
00534  */
00535 int
00536 bpf_validate(f, len)
00537     const struct bpf_insn *f;
00538     int len;
00539 {
00540     u_int i, from;
00541     const struct bpf_insn *p;
00542 
00543     if (len < 1)
00544         return 0;
00545     /*
00546      * There's no maximum program length in userland.
00547      */
00548 #if defined(KERNEL) || defined(_KERNEL)
00549     if (len > BPF_MAXINSNS)
00550         return 0;
00551 #endif
00552 
00553     for (i = 0; i < len; ++i) {
00554         p = &f[i];
00555         switch (BPF_CLASS(p->code)) {
00556         /*
00557          * Check that memory operations use valid addresses.
00558          */
00559         case BPF_LD:
00560         case BPF_LDX:
00561             switch (BPF_MODE(p->code)) {
00562             case BPF_IMM:
00563                 break;
00564             case BPF_ABS:
00565             case BPF_IND:
00566             case BPF_MSH:
00567                 /*
00568                  * There's no maximum packet data size
00569                  * in userland.  The runtime packet length
00570                  * check suffices.
00571                  */
00572 #if defined(KERNEL) || defined(_KERNEL)
00573                 /*
00574                  * More strict check with actual packet length
00575                  * is done runtime.
00576                  */
00577                 if (p->k >= bpf_maxbufsize)
00578                     return 0;
00579 #endif
00580                 break;
00581             case BPF_MEM:
00582                 if (p->k >= BPF_MEMWORDS)
00583                     return 0;
00584                 break;
00585             case BPF_LEN:
00586                 break;
00587             default:
00588                 return 0;
00589             }
00590             break;
00591         case BPF_ST:
00592         case BPF_STX:
00593             if (p->k >= BPF_MEMWORDS)
00594                 return 0;
00595             break;
00596         case BPF_ALU:
00597             switch (BPF_OP(p->code)) {
00598             case BPF_ADD:
00599             case BPF_SUB:
00600             case BPF_MUL:
00601             case BPF_OR:
00602             case BPF_AND:
00603             case BPF_LSH:
00604             case BPF_RSH:
00605             case BPF_NEG:
00606                 break;
00607             case BPF_DIV:
00608                 /*
00609                  * Check for constant division by 0.
00610                  */
00611                 if (BPF_RVAL(p->code) == BPF_K && p->k == 0)
00612                     return 0;
00613                 break;
00614             default:
00615                 return 0;
00616             }
00617             break;
00618         case BPF_JMP:
00619             /*
00620              * Check that jumps are within the code block,
00621              * and that unconditional branches don't go
00622              * backwards as a result of an overflow.
00623              * Unconditional branches have a 32-bit offset,
00624              * so they could overflow; we check to make
00625              * sure they don't.  Conditional branches have
00626              * an 8-bit offset, and the from address is <=
00627              * BPF_MAXINSNS, and we assume that BPF_MAXINSNS
00628              * is sufficiently small that adding 255 to it
00629              * won't overflow.
00630              *
00631              * We know that len is <= BPF_MAXINSNS, and we
00632              * assume that BPF_MAXINSNS is < the maximum size
00633              * of a u_int, so that i + 1 doesn't overflow.
00634              *
00635              * For userland, we don't know that the from
00636              * or len are <= BPF_MAXINSNS, but we know that
00637              * from <= len, and, except on a 64-bit system,
00638              * it's unlikely that len, if it truly reflects
00639              * the size of the program we've been handed,
00640              * will be anywhere near the maximum size of
00641              * a u_int.  We also don't check for backward
00642              * branches, as we currently support them in
00643              * userland for the protochain operation.
00644              */
00645             from = i + 1;
00646             switch (BPF_OP(p->code)) {
00647             case BPF_JA:
00648 #if defined(KERNEL) || defined(_KERNEL)
00649                 if (from + p->k < from || from + p->k >= len)
00650 #else
00651                 if (from + p->k >= len)
00652 #endif
00653                     return 0;
00654                 break;
00655             case BPF_JEQ:
00656             case BPF_JGT:
00657             case BPF_JGE:
00658             case BPF_JSET:
00659                 if (from + p->jt >= len || from + p->jf >= len)
00660                     return 0;
00661                 break;
00662             default:
00663                 return 0;
00664             }
00665             break;
00666         case BPF_RET:
00667             break;
00668         case BPF_MISC:
00669             break;
00670         default:
00671             return 0;
00672         }
00673     }
00674     return BPF_CLASS(f[len - 1].code) == BPF_RET;
