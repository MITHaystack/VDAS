00206 s32 e1000_setup_init_funcs(struct e1000_hw *hw, bool init_device)
00207 {
00208     s32 ret_val;
00209 
00210     /* Can't do much good without knowing the MAC type. */
00211     ret_val = e1000_set_mac_type(hw);
00212     if (ret_val) {
00213         DEBUGOUT("ERROR: MAC type could not be set properly.\n");
00214         goto out;
00215     }
00216 
00217     if (!hw->hw_addr) {
00218         DEBUGOUT("ERROR: Registers not mapped\n");
00219         ret_val = -E1000_ERR_CONFIG;
00220         goto out;
00221     }
00222 
00223     /*
00224      * Init function pointers to generic implementations. We do this first
00225      * allowing a driver module to override it afterward.
00226      */
00227     e1000_init_mac_ops_generic(hw);
00228     e1000_init_nvm_ops_generic(hw);
00229     e1000_init_mbx_ops_generic(hw);
00230 
00231     /*
00232      * Set up the init function pointers. These are functions within the
00233      * adapter family file that sets up function pointers for the rest of
00234      * the functions in that family.
00235      */
00236     switch (hw->mac.type) {
00237     case e1000_82575:
00238     case e1000_82576:
00239     case e1000_82580:
00240     case e1000_i350:
00241         e1000_init_function_pointers_82575(hw);
00242         break;
00243     default:
00244         DEBUGOUT("Hardware not supported\n");
00245         ret_val = -E1000_ERR_CONFIG;
00246         break;
00247     }
00248 
00249     /*
00250      * Initialize the rest of the function pointers. These require some
00251      * register reads/writes in some cases.
00252      */
00253     if (!(ret_val) && init_device) {
00254         ret_val = e1000_init_mac_params(hw);
00255         if (ret_val)
00256             goto out;
00257 
00258         ret_val = e1000_init_nvm_params(hw);
00259         if (ret_val)
00260             goto out;
00261 
00262         ret_val = e1000_init_phy_params(hw);
00263         if (ret_val)
00264             goto out;
00265 
00266         ret_val = e1000_init_mbx_params(hw);
00267         if (ret_val)
00268             goto out;
00269     }
00270 
00271 out:
00272     return ret_val;
00273 }
