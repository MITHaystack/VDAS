00036 /* A generic radio capture format is desirable. It must be
00037  * rigidly defined (e.g., units for fields should be given),
00038  * and easily extensible.
00039  *
00040  * The following is an extensible radio capture format. It is
00041  * based on a bitmap indicating which fields are present.
00042  *
00043  * I am trying to describe precisely what the application programmer
00044  * should expect in the following, and for that reason I tell the
00045  * units and origin of each measurement (where it applies), or else I
00046  * use sufficiently weaselly language ("is a monotonically nondecreasing
00047  * function of...") that I cannot set false expectations for lawyerly
00048  * readers.
00049  */
00050 
00051 /*
00052  * The radio capture header precedes the 802.11 header.
00053  *
00054  * Note well: all radiotap fields are little-endian.
00055  */
00056 struct ieee80211_radiotap_header {
00057     u_int8_t    it_version; /* Version 0. Only increases
00058                      * for drastic changes,
00059                      * introduction of compatible
00060                      * new fields does not count.
00061                      */
00062     u_int8_t    it_pad;
00063     u_int16_t       it_len;         /* length of the whole
00064                      * header in bytes, including
00065                      * it_version, it_pad,
00066                      * it_len, and data fields.
00067                      */
00068     u_int32_t       it_present;     /* A bitmap telling which
00069                      * fields are present. Set bit 31
00070                      * (0x80000000) to extend the
00071                      * bitmap by another 32 bits.
00072                      * Additional extensions are made
00073                      * by setting bit 31.
00074                      */
00075 };

00077 /* Name                                 Data type       Units
00078  * ----                                 ---------       -----
00079  *
00080  * IEEE80211_RADIOTAP_TSFT              u_int64_t       microseconds
00081  *
00082  *      Value in microseconds of the MAC's 64-bit 802.11 Time
00083  *      Synchronization Function timer when the first bit of the
00084  *      MPDU arrived at the MAC. For received frames, only.
00085  *
00086  * IEEE80211_RADIOTAP_CHANNEL           2 x u_int16_t   MHz, bitmap
00087  *
00088  *      Tx/Rx frequency in MHz, followed by flags (see below).
00089  *  Note that IEEE80211_RADIOTAP_XCHANNEL must be used to
00090  *  represent an HT channel as there is not enough room in
00091  *  the flags word.
00092  *
00093  * IEEE80211_RADIOTAP_FHSS              u_int16_t       see below
00094  *
00095  *      For frequency-hopping radios, the hop set (first byte)
00096  *      and pattern (second byte).
00097  *
00098  * IEEE80211_RADIOTAP_RATE              u_int8_t        500kb/s or index
00099  *
00100  *      Tx/Rx data rate.  If bit 0x80 is set then it represents an
00101  *  an MCS index and not an IEEE rate.
00102  *
00103  * IEEE80211_RADIOTAP_DBM_ANTSIGNAL     int8_t          decibels from
00104  *                                                      one milliwatt (dBm)
00105  *
00106  *      RF signal power at the antenna, decibel difference from
00107  *      one milliwatt.
00108  *
00109  * IEEE80211_RADIOTAP_DBM_ANTNOISE      int8_t          decibels from
00110  *                                                      one milliwatt (dBm)
00111  *
00112  *      RF noise power at the antenna, decibel difference from one
00113  *      milliwatt.
00114  *
00115  * IEEE80211_RADIOTAP_DB_ANTSIGNAL      u_int8_t        decibel (dB)
00116  *
00117  *      RF signal power at the antenna, decibel difference from an
00118  *      arbitrary, fixed reference.
00119  *
00120  * IEEE80211_RADIOTAP_DB_ANTNOISE       u_int8_t        decibel (dB)
00121  *
00122  *      RF noise power at the antenna, decibel difference from an
00123  *      arbitrary, fixed reference point.
00124  *
00125  * IEEE80211_RADIOTAP_LOCK_QUALITY      u_int16_t       unitless
00126  *
00127  *      Quality of Barker code lock. Unitless. Monotonically
00128  *      nondecreasing with "better" lock strength. Called "Signal
00129  *      Quality" in datasheets.  (Is there a standard way to measure
00130  *      this?)
00131  *
00132  * IEEE80211_RADIOTAP_TX_ATTENUATION    u_int16_t       unitless
00133  *
00134  *      Transmit power expressed as unitless distance from max
00135  *      power set at factory calibration.  0 is max power.
00136  *      Monotonically nondecreasing with lower power levels.
00137  *
00138  * IEEE80211_RADIOTAP_DB_TX_ATTENUATION u_int16_t       decibels (dB)
00139  *
00140  *      Transmit power expressed as decibel distance from max power
00141  *      set at factory calibration.  0 is max power.  Monotonically
00142  *      nondecreasing with lower power levels.
00143  *
00144  * IEEE80211_RADIOTAP_DBM_TX_POWER      int8_t          decibels from
00145  *                                                      one milliwatt (dBm)
00146  *
00147  *      Transmit power expressed as dBm (decibels from a 1 milliwatt
00148  *      reference). This is the absolute power level measured at
00149  *      the antenna port.
00150  *
00151  * IEEE80211_RADIOTAP_FLAGS             u_int8_t        bitmap
00152  *
00153  *      Properties of transmitted and received frames. See flags
00154  *      defined below.
00155  *
00156  * IEEE80211_RADIOTAP_ANTENNA           u_int8_t        antenna index
00157  *
00158  *      Unitless indication of the Rx/Tx antenna for this packet.
00159  *      The first antenna is antenna 0.
00160  *
00161  * IEEE80211_RADIOTAP_XCHANNEL          u_int32_t   bitmap
00162  *                  u_int16_t   MHz
00163  *                  u_int8_t    channel number
00164  *                  u_int8_t    .5 dBm
00165  *
00166  *  Extended channel specification: flags (see below) followed by
00167  *  frequency in MHz, the corresponding IEEE channel number, and
00168  *  finally the maximum regulatory transmit power cap in .5 dBm
00169  *  units.  This property supersedes IEEE80211_RADIOTAP_CHANNEL
00170  *  and only one of the two should be present.
00171  */
