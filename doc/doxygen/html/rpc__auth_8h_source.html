00045 /*
00046  * Status returned from authentication check
00047  */
00048 enum sunrpc_auth_stat {
00049     SUNRPC_AUTH_OK=0,
00050     /*
00051      * failed at remote end
00052      */
00053     SUNRPC_AUTH_BADCRED=1,      /* bogus credentials (seal broken) */
00054     SUNRPC_AUTH_REJECTEDCRED=2, /* client should begin new session */
00055     SUNRPC_AUTH_BADVERF=3,      /* bogus verifier (seal broken) */
00056     SUNRPC_AUTH_REJECTEDVERF=4, /* verifier expired or was replayed */
00057     SUNRPC_AUTH_TOOWEAK=5,      /* rejected due to security reasons */
00058     /*
00059      * failed locally
00060     */
00061     SUNRPC_AUTH_INVALIDRESP=6,  /* bogus response verifier */
00062     SUNRPC_AUTH_FAILED=7        /* some unknown reason */
00063 };
