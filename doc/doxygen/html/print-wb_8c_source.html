00417             return;

00419 
00420     case PT_RREP:
00421         if (wb_rrep((struct pkt_rrep *)(ph + 1), len) >= 0)
00422             return;
00423         break;
00424 
00425     case PT_DRAWOP:
00426         if (wb_drawop((struct pkt_dop *)(ph + 1), len) >= 0)
00427             return;
00428         break;
00429 
00430     case PT_PREQ:
00431         if (wb_preq((struct pkt_preq *)(ph + 1), len) >= 0)
00432             return;
00433         break;
00434 
00435     case PT_PREP:
00436         if (wb_prep((struct pkt_prep *)(ph + 1), len) >= 0)
00437             return;
