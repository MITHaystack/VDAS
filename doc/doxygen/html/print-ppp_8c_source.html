00422 /* generic Control Protocol (e.g. LCP, IPCP, CCP, etc.) handler */
00423 static void
00424 handle_ctrl_proto(u_int proto, const u_char *pptr, int length)
00425 {
00426     const char *typestr;
00427     u_int code, len;
00428     int (*pfunc)(const u_char *, int);
00429     int x, j;
00430         const u_char *tptr;
00431 
00432         tptr=pptr;
00433 
00434         typestr = tok2str(ppptype2str, "unknown ctrl-proto (0x%04x)", proto);
00435         printf("%s, ",typestr);
00436 
00437     if (length < 4) /* FIXME weak boundary checking */
00438         goto trunc;
00439     TCHECK2(*tptr, 2);
00440 
00441     code = *tptr++;
00442     
00443         printf("%s (0x%02x), id %u, length %u",
00444                tok2str(cpcodes, "Unknown Opcode",code),
00445                code,
00446                *tptr++, /* ID */
00447                length+2);
00448 
00449         if (!vflag)
00450                 return;
00451 
00452     if (length <= 4)
00453         return;    /* there may be a NULL confreq etc. */
00454 
00455     TCHECK2(*tptr, 2);
00456     len = EXTRACT_16BITS(tptr);
00457     tptr += 2;
00458 
00459         printf("\n\tencoded length %u (=Option(s) length %u)",len,len-4);
00460 
00461         if (vflag>1)
00462             print_unknown_data(pptr-2,"\n\t",6);
00463 
00464 
00465     switch (code) {
00466     case CPCODES_VEXT:
00467         if (length < 11)
00468             break;
00469         TCHECK2(*tptr, 4);
00470         printf("\n\t  Magic-Num 0x%08x", EXTRACT_32BITS(tptr));
00471         tptr += 4;
00472         TCHECK2(*tptr, 3);
00473         printf(" Vendor: %s (%u)",
00474                        tok2str(oui_values,"Unknown",EXTRACT_24BITS(tptr)),
00475                        EXTRACT_24BITS(tptr));
00476         /* XXX: need to decode Kind and Value(s)? */
00477         break;
00478     case CPCODES_CONF_REQ:
00479     case CPCODES_CONF_ACK:
00480     case CPCODES_CONF_NAK:
00481     case CPCODES_CONF_REJ:
00482         x = len - 4;    /* Code(1), Identifier(1) and Length(2) */
00483         do {
00484             switch (proto) {
00485             case PPP_LCP:
00486                 pfunc = print_lcp_config_options;
00487                 break;
00488             case PPP_IPCP:
00489                 pfunc = print_ipcp_config_options;
00490                 break;
00491             case PPP_IPV6CP:
00492                 pfunc = print_ip6cp_config_options;
00493                 break;
00494             case PPP_CCP:
00495                 pfunc = print_ccp_config_options;
00496                 break;
00497             case PPP_BACP:
00498                 pfunc = print_bacp_config_options;
00499                 break;
00500             default:
00501                 /*
00502                  * No print routine for the options for
00503                  * this protocol.
00504                  */
00505                 pfunc = NULL;
00506                 break;
00507             }
00508 
00509             if (pfunc == NULL) /* catch the above null pointer if unknown CP */
00510                 break;
00511  
00512             if ((j = (*pfunc)(tptr, len)) == 0)
00513                 break;
00514             x -= j;
00515             tptr += j;
00516         } while (x > 0);
00517         break;
00518 
00519     case CPCODES_TERM_REQ:
00520     case CPCODES_TERM_ACK:
00521         /* XXX: need to decode Data? */
00522         break;
00523     case CPCODES_CODE_REJ:
00524         /* XXX: need to decode Rejected-Packet? */
00525         break;
00526     case CPCODES_PROT_REJ:
00527         if (length < 6)
00528             break;
00529         TCHECK2(*tptr, 2);
00530         printf("\n\t  Rejected %s Protocol (0x%04x)",
00531                tok2str(ppptype2str,"unknown", EXTRACT_16BITS(tptr)),
00532                EXTRACT_16BITS(tptr));
00533         /* XXX: need to decode Rejected-Information? - hexdump for now */
00534                 if (len > 6) {
00535                         printf("\n\t  Rejected Packet");
00536                         print_unknown_data(tptr+2,"\n\t    ",len-2);
00537                 }
00538         break;
00539     case CPCODES_ECHO_REQ:
00540     case CPCODES_ECHO_RPL:
00541     case CPCODES_DISC_REQ:
00542         if (length < 8)
00543             break;
00544         TCHECK2(*tptr, 4);
00545         printf("\n\t  Magic-Num 0x%08x", EXTRACT_32BITS(tptr));
00546         /* XXX: need to decode Data? - hexdump for now */
