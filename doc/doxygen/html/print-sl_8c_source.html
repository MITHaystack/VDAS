00112 static void
00113 sliplink_print(register const u_char *p, register const struct ip *ip,
00114            register u_int length)
00115 {
00116     int dir;
00117     u_int hlen;
00118 
00119     dir = p[SLX_DIR];
00120     putchar(dir == SLIPDIR_IN ? 'I' : 'O');
00121     putchar(' ');
00122 
00123     if (nflag) {
00124         /* XXX just dump the header */
00125         register int i;
00126 
00127         for (i = SLX_CHDR; i < SLX_CHDR + CHDR_LEN - 1; ++i)
00128             printf("%02x.", p[i]);
00129         printf("%02x: ", p[SLX_CHDR + CHDR_LEN - 1]);
00130         return;
00131     }
00132     switch (p[SLX_CHDR] & 0xf0) {
00133 
00134     case TYPE_IP:
00135         printf("ip %d: ", length + SLIP_HDRLEN);
00136         break;
00137 
00138     case TYPE_UNCOMPRESSED_TCP:
00139         /*
00140          * The connection id is stored in the IP protocol field.
00141          * Get it from the link layer since sl_uncompress_tcp()
00142          * has restored the IP header copy to IPPROTO_TCP.
00143          */
00144         lastconn = ((struct ip *)&p[SLX_CHDR])->ip_p;
00145         hlen = IP_HL(ip);
00146         hlen += TH_OFF((struct tcphdr *)&((int *)ip)[hlen]);
00147         lastlen[dir][lastconn] = length - (hlen << 2);
00148         printf("utcp %d: ", lastconn);
00149         break;
00150 
00151     default:
00152         if (p[SLX_CHDR] & TYPE_COMPRESSED_TCP) {
00153             compressed_sl_print(&p[SLX_CHDR], ip,
00154                 length, dir);
00155             printf(": ");
00156         } else
00157             printf("slip-%d!: ", p[SLX_CHDR]);
00158     }
00159 }

00190 static void
00191 compressed_sl_print(const u_char *chdr, const struct ip *ip,
00192             u_int length, int dir)
00193 {
00194     register const u_char *cp = chdr;
00195     register u_int flags, hlen;
00196 
00197     flags = *cp++;
00198     if (flags & NEW_C) {
00199         lastconn = *cp++;
00200         printf("ctcp %d", lastconn);
00201     } else
00202         printf("ctcp *");
00203 
00204     /* skip tcp checksum */
00205     cp += 2;
00206 
00207     switch (flags & SPECIALS_MASK) {
00208     case SPECIAL_I:
00209         printf(" *SA+%d", lastlen[dir][lastconn]);
00210         break;
00211 
00212     case SPECIAL_D:
00213         printf(" *S+%d", lastlen[dir][lastconn]);
00214         break;
00215 
00216     default:
00217         if (flags & NEW_U)
00218             cp = print_sl_change("U=", cp);
00219         if (flags & NEW_W)
00220             cp = print_sl_winchange(cp);
00221         if (flags & NEW_A)
00222             cp = print_sl_change("A+", cp);
00223         if (flags & NEW_S)
00224             cp = print_sl_change("S+", cp);
00225         break;
00226     }
00227     if (flags & NEW_I)
00228         cp = print_sl_change("I+", cp);
00229 
00230     /*
00231      * 'hlen' is the length of the uncompressed TCP/IP header (in words).
00232      * 'cp - chdr' is the length of the compressed header.
00233      * 'length - hlen' is the amount of data in the packet.
00234      */
00235     hlen = IP_HL(ip);
00236     hlen += TH_OFF((struct tcphdr *)&((int32_t *)ip)[hlen]);
00237     lastlen[dir][lastconn] = length - (hlen << 2);
00238     printf(" %d (%ld)", lastlen[dir][lastconn], (long)(cp - chdr));
00239 }
