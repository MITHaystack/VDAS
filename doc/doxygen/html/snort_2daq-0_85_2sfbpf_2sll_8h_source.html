00041 /*
00042  * For captures on Linux cooked sockets, we construct a fake header
00043  * that includes:
00044  *
00045  *  a 2-byte "packet type" which is one of:
00046  *
00047  *      LINUX_SLL_HOST      packet was sent to us
00048  *      LINUX_SLL_BROADCAST packet was broadcast
00049  *      LINUX_SLL_MULTICAST packet was multicast
00050  *      LINUX_SLL_OTHERHOST packet was sent to somebody else
00051  *      LINUX_SLL_OUTGOING  packet was sent *by* us;
00052  *
00053  *  a 2-byte Ethernet protocol field;
00054  *
00055  *  a 2-byte link-layer type;
00056  *
00057  *  a 2-byte link-layer address length;
00058  *
00059  *  an 8-byte source link-layer address, whose actual length is
00060  *  specified by the previous value.
00061  *
00062  * All fields except for the link-layer address are in network byte order.
00063  *
00064  * DO NOT change the layout of this structure, or change any of the
00065  * LINUX_SLL_ values below.  If you must change the link-layer header
00066  * for a "cooked" Linux capture, introduce a new DLT_ type (ask
00067  * "tcpdump-workers@lists.tcpdump.org" for one, so that you don't give it
00068  * a value that collides with a value already being used), and use the
00069  * new header in captures of that type, so that programs that can
00070  * handle DLT_LINUX_SLL captures will continue to handle them correctly
00071  * without any change, and so that capture files with different headers
00072  * can be told apart and programs that read them can dissect the
00073  * packets in them.
00074  */
00075 
00076 #ifndef lib_pcap_sll_h
00077 #define lib_pcap_sll_h
00078 
00079 /*
00080  * A DLT_LINUX_SLL fake link-layer header.
00081  */
00082 #define SLL_HDR_LEN 16          /* total header length */
00083 #define SLL_ADDRLEN 8           /* length of address field */
00084 
00085 struct sll_header
00086 {
00087     u_int16_t sll_pkttype;      /* packet type */
00088     u_int16_t sll_hatype;       /* link-layer address type */
00089     u_int16_t sll_halen;        /* link-layer address length */
00090     u_int8_t sll_addr[SLL_ADDRLEN]; /* link-layer address */
00091     u_int16_t sll_protocol;     /* protocol */
00092 };
00093 
00094 /*
00095  * The LINUX_SLL_ values for "sll_pkttype"; these correspond to the
00096  * PACKET_ values on Linux, but are defined here so that they're
00097  * available even on systems other than Linux, and so that they
00098  * don't change even if the PACKET_ values change.
00099  */
00100 #define LINUX_SLL_HOST      0
00101 #define LINUX_SLL_BROADCAST 1
00102 #define LINUX_SLL_MULTICAST 2
00103 #define LINUX_SLL_OTHERHOST 3
00104 #define LINUX_SLL_OUTGOING  4
00105 
00106 /*
00107  * The LINUX_SLL_ values for "sll_protocol"; these correspond to the
00108  * ETH_P_ values on Linux, but are defined here so that they're
00109  * available even on systems other than Linux.  We assume, for now,
00110  * that the ETH_P_ values won't change in Linux; if they do, then:
00111  *
00112  *  if we don't translate them in "pcap-linux.c", capture files
00113  *  won't necessarily be readable if captured on a system that
00114  *  defines ETH_P_ values that don't match these values;
00115  *
00116  *  if we do translate them in "pcap-linux.c", that makes life
00117  *  unpleasant for the BPF code generator, as the values you test
00118  *  for in the kernel aren't the values that you test for when
00119  *  reading a capture file, so the fixup code run on BPF programs
00120  *  handed to the kernel ends up having to do more work.
00121  *
00122  * Add other values here as necessary, for handling packet types that
00123  * might show up on non-Ethernet, non-802.x networks.  (Not all the ones
00124  * in the Linux "if_ether.h" will, I suspect, actually show up in
00125  * captures.)
00126  */
00127 #define LINUX_SLL_P_802_3   0x0001  /* Novell 802.3 frames without 802.2 LLC header */
00128 #define LINUX_SLL_P_802_2   0x0004  /* 802.2 frames (not D/I/X Ethernet) */
00129 
00130 #endif
