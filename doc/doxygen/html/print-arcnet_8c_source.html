00068     if (qflag) {

00100             ap->arc_shost, ap->arc_dhost, arctypename, seqid,

00104 /*
00105  * This is the top level routine of the printer.  'p' points
00106  * to the ARCNET header of the packet, 'h->ts' is the timestamp,
00107  * 'h->len' is the length of the packet off the wire, and 'h->caplen'
00108  * is the number of bytes actually captured.
00109  */
00110 u_int
00111 arcnet_if_print(const struct pcap_pkthdr *h, const u_char *p)
00112 {
00113     u_int caplen = h->caplen;
00114     u_int length = h->len;
00115     const struct arc_header *ap;
00116 
00117     int phds, flag = 0, archdrlen = 0;
00118     u_int seqid = 0;
00119     u_char arc_type;
00120 
00121     if (caplen < ARC_HDRLEN) {
00122         printf("[|arcnet]");
00123         return (caplen);
00124     }
00125 
00126     ap = (const struct arc_header *)p;
00127     arc_type = ap->arc_type;
00128 
00129     switch (arc_type) {
00130     default:
00131         phds = 1;
00132         break;
00133     case ARCTYPE_IP_OLD:
00134     case ARCTYPE_ARP_OLD:
00135     case ARCTYPE_DIAGNOSE:
00136         phds = 0;
00137         archdrlen = ARC_HDRLEN;
00138         break;
00139     }
00140 
00141     if (phds) {
00142         if (caplen < ARC_HDRNEWLEN) {
00143             arcnet_print(p, length, 0, 0, 0);
00144             printf("[|phds]");
00145             return (caplen);
00146         }
00147 
00148         if (ap->arc_flag == 0xff) {
00149             if (caplen < ARC_HDRNEWLEN_EXC) {
00150                 arcnet_print(p, length, 0, 0, 0);
00151                 printf("[|phds extended]");
00152                 return (caplen);
00153             }
00154             flag = ap->arc_flag2;
00155             seqid = EXTRACT_16BITS(&ap->arc_seqid2);
00156             archdrlen = ARC_HDRNEWLEN_EXC;
00157         } else {
00158             flag = ap->arc_flag;
00159             seqid = EXTRACT_16BITS(&ap->arc_seqid);
00160             archdrlen = ARC_HDRNEWLEN;
00161         }
00162     }
00163 
00164 
00165     if (eflag)
00166         arcnet_print(p, length, phds, flag, seqid);
00167 
00168     /*
00169      * Go past the ARCNET header.
00170      */
00171     length -= archdrlen;
00172     caplen -= archdrlen;
00173     p += archdrlen;
00174 
00175     if (phds && flag && (flag & 1) == 0) {
00176         /*
00177          * This is a middle fragment.
00178          */
00179         return (archdrlen);
00180     }
00181 
00182     if (!arcnet_encap_print(arc_type, p, length, caplen))
00183         default_print(p, caplen);
00184

00188 /*
00189  * This is the top level routine of the printer.  'p' points
00190  * to the ARCNET header of the packet, 'h->ts' is the timestamp,
00191  * 'h->len' is the length of the packet off the wire, and 'h->caplen'
00192  * is the number of bytes actually captured.  It is quite similar
00193  * to the non-Linux style printer except that Linux doesn't ever
00194  * supply packets that look like exception frames, it always supplies
00195  * reassembled packets rather than raw frames, and headers have an
00196  * extra "offset" field between the src/dest and packet type.
00197  */
00198 u_int
00199 arcnet_linux_if_print(const struct pcap_pkthdr *h, const u_char *p)
00200 {
00201     u_int caplen = h->caplen;
00202     u_int length = h->len;
00203     const struct arc_linux_header *ap;
00204 
00205     int archdrlen = 0;
00206     u_char arc_type;
00207 
00208     if (caplen < ARC_LINUX_HDRLEN) {
00209         printf("[|arcnet]");
00210         return (caplen);
00211     }
00212 
00213     ap = (const struct arc_linux_header *)p;
00214     arc_type = ap->arc_type;
00215 
00216     switch (arc_type) {
00217     default:
00218         archdrlen = ARC_LINUX_HDRNEWLEN;
00219         if (caplen < ARC_LINUX_HDRNEWLEN) {
00220             printf("[|arcnet]");
00221             return (caplen);
00222         }
00223         break;
00224     case ARCTYPE_IP_OLD:
00225     case ARCTYPE_ARP_OLD:
00226     case ARCTYPE_DIAGNOSE:
00227         archdrlen = ARC_LINUX_HDRLEN;
00228         break;
00229     }
00230 
00231     if (eflag)
00232         arcnet_print(p, length, 0, 0, 0);
00233 
00234     /*
00235      * Go past the ARCNET header.
00236      */
00237     length -= archdrlen;
00238     caplen -= archdrlen;
00239     p += archdrlen;
00240 
00241     if (!arcnet_encap_print(arc_type, p, length, caplen))
00242         default_print(p, caplen);
00243 
00244     return (archdrlen);
