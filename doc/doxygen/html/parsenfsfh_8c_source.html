00060 /*
00061  * This routine attempts to parse a file handle (in network byte order),
00062  * using heuristics to guess what kind of format it is in.  See the
00063  * file "fhandle_layouts" for a detailed description of the various
00064  * patterns we know about.
00065  *
00066  * The file handle is parsed into our internal representation of a
00067  * file-system id, and an internal representation of an inode-number.
00068  */

00074 #define FHT_IRIX5   4
00075 #define FHT_SUNOS3  5
00076 #define FHT_SUNOS4  6
00077 #define FHT_ULTRIX  7
00078 #define FHT_VMSUCX  8

00091 #define make_uint32(msb,b,c,lsb)\
00092     (XFF(lsb) + (XFF(c)<<8) + (XFF(b)<<16) + (XFF(msb)<<24))

00123     u_int32_t temp;
00124     int fhtype = FHT_UNKNOWN;
00125     int i;
00126 
00127     if (ourself) {
00128         /* File handle generated on this host, no need for guessing */
00129 #if defined(IRIX40)
00130         fhtype = FHT_IRIX4;
00131 #endif
00132 #if defined(IRIX50)
00133         fhtype = FHT_IRIX5;
00134 #endif
00135 #if defined(IRIX51)
00136         fhtype = FHT_IRIX5;
00137 #endif
00138 #if defined(SUNOS4)
00139         fhtype = FHT_SUNOS4;
00140 #endif
00141 #if defined(SUNOS5)
00142         fhtype = FHT_SUNOS5;
00143 #endif
00144 #if defined(ultrix)
00145         fhtype = FHT_ULTRIX;
00146 #endif
00147 #if defined(__osf__)
00148         fhtype = FHT_DECOSF;
00149 #endif
00150 #if defined(__NetBSD__) || defined(__FreeBSD__) || defined(__DragonFly__) \
00151      || defined(__OpenBSD__)
00152         fhtype = FHT_BSD44;
00153 #endif
00154     }
00155     /*
00156      * This is basically a big decision tree
00157      */
00158     else if ((fhp[0] == 0) && (fhp[1] == 0)) {
00159         /* bytes[0,1] == (0,0); rules out Ultrix, IRIX5, SUNOS5 */
00160         /* probably rules out HP-UX, AIX unless they allow major=0 */
00161         if ((fhp[2] == 0) && (fhp[3] == 0)) {
00162         /* bytes[2,3] == (0,0); must be Auspex */
00163         /* XXX or could be Ultrix+MASSBUS "hp" disk? */
00164         fhtype = FHT_AUSPEX;
00165         }
00166         else {
00167         /*
00168          * bytes[2,3] != (0,0); rules out Auspex, could be
00169          * DECOSF, SUNOS4, or IRIX4
00170          */
00171         if ((fhp[4] != 0) && (fhp[5] == 0) &&
00172             (fhp[8] == 12) && (fhp[9] == 0)) {
00173             /* seems to be DECOSF, with minor == 0 */
00174             fhtype = FHT_DECOSF;
00175         }
00176         else {
00177             /* could be SUNOS4 or IRIX4 */
00178             /* XXX the test of fhp[5] == 8 could be wrong */
00179             if ((fhp[4] == 0) && (fhp[5] == 8) && (fhp[6] == 0) &&
00180             (fhp[7] == 0)) {
00181             /* looks like a length, not a file system typecode */
00182             fhtype = FHT_IRIX4;
00183             }
00184             else {
00185             /* by elimination */
00186             fhtype = FHT_SUNOS4;
00187             }
00188         }
00189         }
00190     }
00191     else {
00192         /*
00193          * bytes[0,1] != (0,0); rules out Auspex, IRIX4, SUNOS4
00194          * could be IRIX5, DECOSF, UCX, Ultrix, SUNOS5
00195          * could be AIX, HP-UX
00196          */
00197         if ((fhp[2] == 0) && (fhp[3] == 0)) {
00198         /*
00199          * bytes[2,3] == (0,0); rules out OSF, probably not UCX
00200          * (unless the exported device name is just one letter!),
00201          * could be Ultrix, IRIX5, AIX, or SUNOS5
00202          * might be HP-UX (depends on their values for minor devs)
00203          */
00204         if ((fhp[6] == 0) && (fhp[7] == 0)) {
00205             fhtype = FHT_BSD44;
00206         }
00207         /*XXX we probably only need to test of these two bytes */
00208         else if ((fhp[21] == 0) && (fhp[23] == 0)) {
00209             fhtype = FHT_ULTRIX;
00210         }
00211         else {
00212             /* Could be SUNOS5/IRIX5, maybe AIX */
00213             /* XXX no obvious difference between SUNOS5 and IRIX5 */
00214             if (fhp[9] == 10)
00215             fhtype = FHT_SUNOS5;
00216             /* XXX what about AIX? */
00217         }
00218         }
00219         else {
00220         /*
00221          * bytes[2,3] != (0,0); rules out Ultrix, could be
00222          * DECOSF, SUNOS5, IRIX5, AIX, HP-UX, or UCX
00223          */
00224         if ((fhp[8] == 12) && (fhp[9] == 0)) {
00225             fhtype = FHT_DECOSF;
00226         }
00227         else if ((fhp[8] == 0) && (fhp[9] == 10)) {
00228             /* could be SUNOS5/IRIX5, AIX, HP-UX */
00229             if ((fhp[7] == 0) && (fhp[6] == 0) &&
00230             (fhp[5] == 0) && (fhp[4] == 0)) {
00231             /* XXX is this always true of HP-UX? */
00232             fhtype = FHT_HPUX9;
00233             }
00234             else if (fhp[7] == 2) {
00235             /* This would be MNT_NFS on AIX, which is impossible */
00236             fhtype = FHT_SUNOS5;    /* or maybe IRIX5 */
00237             }
00238             else {
00239             /*
00240              * XXX Could be SUNOS5/IRIX5 or AIX.  I don't
00241              * XXX see any way to disambiguate these, so
00242              * XXX I'm going with the more likely guess.
00243              * XXX Sorry, Big Blue.
00244              */
00245             fhtype = FHT_SUNOS5;    /* or maybe IRIX5 */
00246             }
00247             }
00248         else {
00249             if (is_UCX(fhp)) {
00250             fhtype = FHT_VMSUCX;
00251             }
00252             else {
00253             fhtype = FHT_UNKNOWN;
00254             }
00255         }
00256         }
00257     }
00258 
00259     /* XXX still needs to handle SUNOS3 */
00260 
00261     switch (fhtype) {
00262     case FHT_AUSPEX:
00263         fsidp->Fsid_dev.Minor = fhp[7];
00264         fsidp->Fsid_dev.Major = fhp[6];
00265         fsidp->fsid_code = 0;
00266 
00267         temp = make_uint32(fhp[12], fhp[13], fhp[14], fhp[15]);
00268         *inop = temp;
00269 
00270         if (osnamep)
00271         *osnamep = "Auspex";
00272         break;
00273 
00274     case FHT_BSD44:
00275         fsidp->Fsid_dev.Minor = fhp[0];
00276         fsidp->Fsid_dev.Major = fhp[1];
00277         fsidp->fsid_code = 0;
00278 
00279         temp = make_uint32(fhp[15], fhp[14], fhp[13], fhp[12]);
00280         *inop = temp;
00281 
00282         if (osnamep)
00283         *osnamep = "BSD 4.4";
00284         break;
00285 
00286     case FHT_DECOSF:
00287         fsidp->fsid_code = make_uint32(fhp[7], fhp[6], fhp[5], fhp[4]);
00288             /* XXX could ignore 3 high-order bytes */
00289 
00290         temp = make_uint32(fhp[3], fhp[2], fhp[1], fhp[0]);
00291         fsidp->Fsid_dev.Minor = temp & 0xFFFFF;
00292         fsidp->Fsid_dev.Major = (temp>>20) & 0xFFF;
00293 
00294         temp = make_uint32(fhp[15], fhp[14], fhp[13], fhp[12]);
00295         *inop = temp;
00296         if (osnamep)
00297         *osnamep = "OSF";
00298         break;
00299 
00300     case FHT_IRIX4:
00301         fsidp->Fsid_dev.Minor = fhp[3];
00302         fsidp->Fsid_dev.Major = fhp[2];
00303         fsidp->fsid_code = 0;
00304 
00305         temp = make_uint32(fhp[8], fhp[9], fhp[10], fhp[11]);
00306         *inop = temp;
00307 
00308         if (osnamep)
00309         *osnamep = "IRIX4";
00310         break;
00311 
00312     case FHT_IRIX5:
00313         fsidp->Fsid_dev.Minor = make_uint16(fhp[2], fhp[3]);
00314         fsidp->Fsid_dev.Major = make_uint16(fhp[0], fhp[1]);
00315         fsidp->fsid_code = make_uint32(fhp[4], fhp[5], fhp[6], fhp[7]);
00316 
00317         temp = make_uint32(fhp[12], fhp[13], fhp[14], fhp[15]);
00318         *inop = temp;
00319 
00320         if (osnamep)
00321         *osnamep = "IRIX5";
00322         break;
00323 
00324 #ifdef notdef
00325     case FHT_SUNOS3:
00326         /*
00327          * XXX - none of the heuristics above return this.
00328          * Are there any SunOS 3.x systems around to care about?
00329          */
00330         if (osnamep)
00331         *osnamep = "SUNOS3";
00332         break;
00333 #endif
00334 
00335     case FHT_SUNOS4:
00336         fsidp->Fsid_dev.Minor = fhp[3];
00337         fsidp->Fsid_dev.Major = fhp[2];
00338         fsidp->fsid_code = make_uint32(fhp[4], fhp[5], fhp[6], fhp[7]);
00339 
00340         temp = make_uint32(fhp[12], fhp[13], fhp[14], fhp[15]);
00341         *inop = temp;
00342 
00343         if (osnamep)
00344         *osnamep = "SUNOS4";
00345         break;
00346 
00347     case FHT_SUNOS5:
00348         temp = make_uint16(fhp[0], fhp[1]);
00349         fsidp->Fsid_dev.Major = (temp>>2) &  0x3FFF;
00350         temp = make_uint24(fhp[1], fhp[2], fhp[3]);
00351         fsidp->Fsid_dev.Minor = temp & 0x3FFFF;
00352         fsidp->fsid_code = make_uint32(fhp[4], fhp[5], fhp[6], fhp[7]);
00353 
00354         temp = make_uint32(fhp[12], fhp[13], fhp[14], fhp[15]);
00355         *inop = temp;

00361     case FHT_ULTRIX:
00362         fsidp->fsid_code = 0;
00363         fsidp->Fsid_dev.Minor = fhp[0];
00364         fsidp->Fsid_dev.Major = fhp[1];
00365 
00366         temp = make_uint32(fhp[7], fhp[6], fhp[5], fhp[4]);
