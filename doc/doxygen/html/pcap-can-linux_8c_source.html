00108     /* Create socket */
00109     handle->fd = socket(PF_CAN, SOCK_RAW, CAN_RAW);
00110     if (handle->fd < 0)
00111     {
00112         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE, "Can't create raw socket %d:%s",
00113             errno, strerror(errno));
00114         return PCAP_ERROR;
00115     }
00116 
00117     /* get interface index */
00118     memset(&ifr, 0, sizeof(ifr));
00119     strncpy(ifr.ifr_name, handle->opt.source, sizeof(ifr.ifr_name));
00120     if (ioctl(handle->fd, SIOCGIFINDEX, &ifr) < 0)
00121     {
00122         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE,
00123                 "Unable to get interface index: %s",

00126         return PCAP_ERROR;

00128     handle->md.ifindex = ifr.ifr_ifindex;
00129 
00130     /* allocate butter */
00131     handle->buffer = malloc(handle->bufsize);
00132     if (!handle->buffer)

00134         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE, "Can't allocate dump buffer: %s",

00137         return PCAP_ERROR;

00139 
00140     /* Bind to the socket */
00141     addr.can_family = AF_CAN;

00165 can_read_linux(pcap_t *handle, int max_packets, pcap_handler callback, u_char *user)
00166 {
00167     struct msghdr msg;
00168     struct pcap_pkthdr pkth;
00169     struct iovec iv;
00170     struct can_frame* cf;
00171 
00172     iv.iov_base = &handle->buffer[handle->offset];
00173     iv.iov_len = handle->snapshot;
00174 
00175     memset(&msg, 0, sizeof(msg));
00176     msg.msg_iov = &iv;
00177     msg.msg_iovlen = 1;
00178     msg.msg_control = handle->buffer;
00179     msg.msg_controllen = handle->offset;
00180 
00181     do
00182     {
00183         pkth.caplen = recvmsg(handle->fd, &msg, 0);
00184         if (handle->break_loop)
00185         {
00186             handle->break_loop = 0;
00187             return -2;
00188         }
00189     } while ((pkth.caplen == -1) && (errno == EINTR));
00190 
00191     if (pkth.caplen < 0)
00192     {
00193         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE, "Can't receive packet %d:%s",
00194             errno, strerror(errno));
00195         return -1;
00196     }
00197 
00198     /* adjust capture len according to frame len */
00199     cf = (struct can_frame*)&handle->buffer[8];
00200     pkth.caplen -= 8 - cf->can_dlc;
00201     pkth.len = pkth.caplen;
00202 
00203     cf->can_id = htonl( cf->can_id );
00204 
00205     if( -1 == gettimeofday(&pkth.ts, NULL) )
00206     {
00207         snprintf(handle->errbuf, PCAP_ERRBUF_SIZE, "Can't get time of day %d:%s",
00208             errno, strerror(errno));
00209         return -1;
00210     }
00211 
00212     callback(user, &pkth, &handle->buffer[8]);
00213 
00214     return 1;
