00177 s32 ixgbe_reset_hw_X540(struct ixgbe_hw *hw)
00178 {
00179     ixgbe_link_speed link_speed;
00180     s32 status = 0;
00181     u32 ctrl, ctrl_ext, reset_bit;
00182     u32 i;
00183     bool link_up = false;
00184 
00185     /* Call adapter stop to disable tx/rx and clear interrupts */
00186     hw->mac.ops.stop_adapter(hw);
00187 
00188     /*
00189      * Prevent the PCI-E bus from from hanging by disabling PCI-E master
00190      * access and verify no pending requests before reset
00191      */
00192     ixgbe_disable_pcie_master(hw);
00193 
00194 mac_reset_top:
00195     /*
00196      * Issue global reset to the MAC. Needs to be SW reset if link is up.
00197      * If link reset is used when link is up, it might reset the PHY when
00198      * mng is using it.  If link is down or the flag to force full link
00199      * reset is set, then perform link reset.
00200      */
00201     if (hw->force_full_reset) {
00202         reset_bit = IXGBE_CTRL_LNK_RST;
00203     } else {
00204         hw->mac.ops.check_link(hw, &link_speed, &link_up, false);
00205         if (!link_up)
00206             reset_bit = IXGBE_CTRL_LNK_RST;
00207         else
00208             reset_bit = IXGBE_CTRL_RST;
00209     }
00210 
00211     ctrl = IXGBE_READ_REG(hw, IXGBE_CTRL);
00212     IXGBE_WRITE_REG(hw, IXGBE_CTRL, (ctrl | reset_bit));
00213     IXGBE_WRITE_FLUSH(hw);
00214 
00215     /* Poll for reset bit to self-clear indicating reset is complete */
00216     for (i = 0; i < 10; i++) {
00217         udelay(1);
00218         ctrl = IXGBE_READ_REG(hw, IXGBE_CTRL);
00219         if (!(ctrl & reset_bit))
00220             break;
00221     }
00222 
00223 
00224     if (ctrl & reset_bit) {
00225         status = IXGBE_ERR_RESET_FAILED;
00226         hw_dbg(hw, "Reset polling failed to complete.\n");
00227     }
00228 
00229     /*
00230      * Double resets are required for recovery from certain error
00231      * conditions.  Between resets, it is necessary to stall to allow time
00232      * for any pending HW events to complete.  We use 1usec since that is
00233      * what is needed for ixgbe_disable_pcie_master().  The second reset
00234      * then clears out any effects of those events.
00235      */
00236     if (hw->mac.flags & IXGBE_FLAGS_DOUBLE_RESET_REQUIRED) {
00237         hw->mac.flags &= ~IXGBE_FLAGS_DOUBLE_RESET_REQUIRED;
00238         udelay(1);
00239         goto mac_reset_top;
00240     }
00241 
00242     /* Clear PF Reset Done bit so PF/VF Mail Ops can work */
00243     ctrl_ext = IXGBE_READ_REG(hw, IXGBE_CTRL_EXT);
00244     ctrl_ext |= IXGBE_CTRL_EXT_PFRSTD;
00245     IXGBE_WRITE_REG(hw, IXGBE_CTRL_EXT, ctrl_ext);
00246 
00247     msleep(50);
00248 
00249     /* Set the Rx packet buffer size. */
00250     IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(0), 384 << IXGBE_RXPBSIZE_SHIFT);
00251 
00252     /* Store the permanent mac address */
00253     hw->mac.ops.get_mac_addr(hw, hw->mac.perm_addr);
00254 
00255     /*
00256      * Store MAC address from RAR0, clear receive address registers, and
00257      * clear the multicast table.  Also reset num_rar_entries to 128,
00258      * since we modify this value when programming the SAN MAC address.
00259      */
00260     hw->mac.num_rar_entries = 128;
00261     hw->mac.ops.init_rx_addrs(hw);
00262 
00263     /* Store the permanent SAN mac address */
00264     hw->mac.ops.get_san_mac_addr(hw, hw->mac.san_addr);
00265 
00266     /* Add the SAN MAC address to the RAR only if it's a valid address */
00267     if (ixgbe_validate_mac_addr(hw->mac.san_addr) == 0) {
00268         hw->mac.ops.set_rar(hw, hw->mac.num_rar_entries - 1,
00269                             hw->mac.san_addr, 0, IXGBE_RAH_AV);
00270 
00271         /* Reserve the last RAR for the SAN MAC address */
00272         hw->mac.num_rar_entries--;
00273     }
00274 
00275     return status;
00276 }

00363     s32 status;
00364

00403 u16 ixgbe_calc_eeprom_checksum_X540(struct ixgbe_hw *hw)
00404 {
00405     u16 i;
00406     u16 j;
00407     u16 checksum = 0;
00408     u16 length = 0;
00409     u16 pointer = 0;
00410     u16 word = 0;
00411 
00412     /* Include 0x0-0x3F in the checksum */
00413     for (i = 0; i < IXGBE_EEPROM_CHECKSUM; i++) {
00414         if (hw->eeprom.ops.read(hw, i, &word) != 0) {
00415             hw_dbg(hw, "EEPROM read failed\n");
00416             break;
00417         }
00418         checksum += word;
00419     }
00420 
00421     /*
00422      * Include all data from pointers 0x3, 0x6-0xE.  This excludes the
00423      * FW, PHY module, and PCIe Expansion/Option ROM pointers.
00424      */
00425     for (i = IXGBE_PCIE_ANALOG_PTR; i < IXGBE_FW_PTR; i++) {
00426         if (i == IXGBE_PHY_PTR || i == IXGBE_OPTION_ROM_PTR)
00427             continue;
00428 
00429         if (hw->eeprom.ops.read(hw, i, &pointer) != 0) {
00430             hw_dbg(hw, "EEPROM read failed\n");
00431             break;
00432         }
00433 
00434         /* Skip pointer section if the pointer is invalid. */
00435         if (pointer == 0xFFFF || pointer == 0 ||
00436             pointer >= hw->eeprom.word_size)
00437             continue;
00438 
00439         if (hw->eeprom.ops.read(hw, pointer, &length)
00440             != 0) {
00441             hw_dbg(hw, "EEPROM read failed\n");
00442             break;
00443         }
00444 
00445         /* Skip pointer section if length is invalid. */
00446         if (length == 0xFFFF || length == 0 ||
00447             (pointer + length) >= hw->eeprom.word_size)
00448             continue;
00449 
00450         for (j = pointer+1; j <= pointer+length; j++) {
00451             if (hw->eeprom.ops.read(hw, j, &word)
00452                 != 0) {
00453                 hw_dbg(hw, "EEPROM read failed\n");
00454                 break;
00455             }
00456             checksum += word;
00457         }
00458     }
00459 
00460     checksum = (u16)IXGBE_EEPROM_SUM - checksum;
00461 
00462     return checksum;
00463 }

00562 s32 ixgbe_acquire_swfw_sync_X540(struct ixgbe_hw *hw, u16 mask)
00563 {
00564     u32 swfw_sync;
00565     u32 swmask = mask;
00566     u32 fwmask = mask << 5;
00567     u32 hwmask = 0;
00568     u32 timeout = 200;
00569     u32 i;
00570 
00571     if (swmask == IXGBE_GSSR_EEP_SM)
00572         hwmask = IXGBE_GSSR_FLASH_SM;
00573 
00574     for (i = 0; i < timeout; i++) {
00575         /*
00576          * SW NVM semaphore bit is used for access to all
00577          * SW_FW_SYNC bits (not just NVM)
00578          */
00579         if (ixgbe_get_swfw_sync_semaphore(hw))
00580             return IXGBE_ERR_SWFW_SYNC;
00581 
00582         swfw_sync = IXGBE_READ_REG(hw, IXGBE_SWFW_SYNC);
00583         if (!(swfw_sync & (fwmask | swmask | hwmask))) {
00584             swfw_sync |= swmask;
00585             IXGBE_WRITE_REG(hw, IXGBE_SWFW_SYNC, swfw_sync);
00586             ixgbe_release_swfw_sync_semaphore(hw);
00587             break;
00588         } else {
00589             /*
00590              * Firmware currently using resource (fwmask), hardware currently
00591              * using resource (hwmask), or other software thread currently
00592              * using resource (swmask)
00593              */
00594             ixgbe_release_swfw_sync_semaphore(hw);
00595             msleep(5);
00596         }
00597     }
00598 
00599     /* If the resource is not released by the FW/HW the SW can assume that
00600      * the FW/HW malfunctions. In that case the SW should sets the SW bit(s)
00601      * of the requested resource(s) while ignoring the corresponding FW/HW
00602      * bits in the SW_FW_SYNC register.
00603      */
00604     if (i >= timeout) {
00605         swfw_sync = IXGBE_READ_REG(hw, IXGBE_SWFW_SYNC);
00606         if (swfw_sync & (fwmask| hwmask)) {
00607             if (ixgbe_get_swfw_sync_semaphore(hw))
00608                 return IXGBE_ERR_SWFW_SYNC;
00609 
00610             swfw_sync |= swmask;
00611             IXGBE_WRITE_REG(hw, IXGBE_SWFW_SYNC, swfw_sync);
00612             ixgbe_release_swfw_sync_semaphore(hw);
00613         }
00614     }
00615 
00616     msleep(5);
00617     return 0;
00618 }
