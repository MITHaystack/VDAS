00126 #ifndef BCM_HAS_MSECS_TO_JIFFIES
00127 static unsigned long msecs_to_jiffies(const unsigned int m)
00128 {
00129 #if HZ <= MSEC_PER_SEC && !(MSEC_PER_SEC % HZ)
00130     /*
00131      * HZ is equal to or smaller than 1000, and 1000 is a nice
00132      * round multiple of HZ, divide with the factor between them,
00133      * but round upwards:
00134      */
00135     return (m + (MSEC_PER_SEC / HZ) - 1) / (MSEC_PER_SEC / HZ);
00136 #elif HZ > MSEC_PER_SEC && !(HZ % MSEC_PER_SEC)
00137     /*
00138      * HZ is larger than 1000, and HZ is a nice round multiple of
00139      * 1000 - simply multiply with the factor between them.
00140      *
00141      * But first make sure the multiplication result cannot
00142      * overflow:
00143      */
00144     if (m > jiffies_to_msecs(MAX_JIFFY_OFFSET))
00145         return MAX_JIFFY_OFFSET;
00146 
00147     return m * (HZ / MSEC_PER_SEC);
00148 #else
00149     /*
00150      * Generic case - multiply, round and divide. But first
00151      * check that if we are doing a net multiplication, that
00152      * we wouldn't overflow:
00153      */
00154     if (HZ > MSEC_PER_SEC && m > jiffies_to_msecs(MAX_JIFFY_OFFSET))
00155         return MAX_JIFFY_OFFSET;
00156 
00157     return (m * HZ + MSEC_PER_SEC - 1) / MSEC_PER_SEC;
00158 #endif
00159 }
00160 #endif /* BCM_HAS_MSECS_TO_JIFFIES */
