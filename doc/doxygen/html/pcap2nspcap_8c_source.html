00071 int run() {
00072   char errbuf[PCAP_ERRBUF_SIZE];
00073   int rc, promisc = 1;
00074 
00075   if((rc = unlink(out_pipe_filename)) != 0) { /* Just to be safe */
00076     if(verbose) printf("Unlink failed: %d\n", rc);
00077   } else
00078     if(verbose) printf("Deleted named pipe %s\n", out_pipe_filename);
00079 
00080   if(verbose) printf("Creating named pipe %s...\n", out_pipe_filename);
00081 
00082   /* read/write permissions for owner, and with read permissions for group and others. */
00083   if(mkfifo(out_pipe_filename, S_IWUSR | S_IRUSR | S_IRGRP | S_IROTH) != 0) {
00084     printf("ERROR: Unable to create named pipe %s: file already existing ?\n",
00085        out_pipe_filename);
00086     exit(-1);
00087   }
00088 
00089   if(verbose) printf("Waiting for process to attach to named pipe %s...\n",
00090              out_pipe_filename);
00091   if((out_fd = fopen(out_pipe_filename, "w")) == NULL) {
00092     printf("ERROR: Unable to write into file %s\n", out_pipe_filename);
00093     cleanup(-1);
00094   } else
00095     out_fd_id = fileno(out_fd);
00096 
00097   pd = pcap_open_live(in_device, snaplen, promisc, 500, errbuf);
00098   if(pd == NULL) pd = pcap_open_offline(in_device, errbuf);
00099 
00100   if(pd == NULL) {
00101     printf("ERROR: unable to open pcap device/file %s: %s\n",
00102        in_device, errbuf);
00103     return(-1);
00104   }
00105 
00106   write_pcap_header();
00107   pcap_loop(pd, -1, processsPacket, NULL);
00108 
00109   cleanup(-1);
00110 
00111   return(0);
00112 }

00137 void append(char *ptr, int len) {
00138   int writesize = len;
00139 
00140   while(writesize > 0) {
00141     int rc = write(out_fd_id, &ptr[len-writesize], writesize);

00144       printf("ERROR: Fatal write error: %s\n", strerror(errno));
