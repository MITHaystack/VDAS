00061  * The time difference in millisecond
00062  */
00063 double delta_time (struct timeval * now,
00064            struct timeval * before) {
00065   time_t delta_seconds;
00066   time_t delta_microseconds;
00067 
00068   /*
00069    * compute delta in second, 1/10's and 1/1000's second units
00070    */
00071   delta_seconds      = now -> tv_sec  - before -> tv_sec;
00072   delta_microseconds = now -> tv_usec - before -> tv_usec;
00073 
00074   if(delta_microseconds < 0) {
00075     /* manually carry a one from the seconds field */
00076     delta_microseconds += 1000000;  /* 1e6 */
00077     -- delta_seconds;
00078   }
00079   return((double)(delta_seconds * 1000) + (double)delta_microseconds/1000);
00080 }

00209   int n;
00210 
00211   cp = &buf[bufLen];
00212   *--cp = '\0';
00213

00215   do {
00216     byte = addr & 0xff;
00217     *--cp = byte % 10 + '0';
00218     byte /= 10;
00219     if (byte > 0) {
00220       *--cp = byte % 10 + '0';
00221       byte /= 10;
00222       if (byte > 0)
00223     *--cp = byte + '0';
00224     }
00225     *--cp = '.';
00226     addr >>= 8;
00227   } while (--n > 0);
00228 
00229   /* Convert the string to lowercase */
00230   retStr = (char*)(cp+1);
00231 
00232   return(retStr);

00364 int32_t gmt2local(time_t t) {
00365   int dt, dir;
00366   struct tm *gmt, *loc;
00367   struct tm sgmt;
00368 
00369   if (t == 0)
00370     t = time(NULL);
00371   gmt = &sgmt;
00372   *gmt = *gmtime(&t);
00373   loc = localtime(&t);
00374   dt = (loc->tm_hour - gmt->tm_hour) * 60 * 60 +
00375     (loc->tm_min - gmt->tm_min) * 60;
00376 
00377   /*
00378    * If the year or julian day is different, we span 00:00 GMT
00379    * and must add or subtract a day. Check the year first to
00380    * avoid problems when the julian day wraps.
00381    */
00382   dir = loc->tm_year - gmt->tm_year;
00383   if (dir == 0)
00384     dir = loc->tm_yday - gmt->tm_yday;
00385   dt += dir * 24 * 60 * 60;
00386 
00387   return (dt);
