00052 /*
00053  * getopt --
00054  *  Parse argc/argv argument vector.
00055  */
00056 int
00057 getopt(nargc, nargv, ostr)
00058     int nargc;
00059     char * const *nargv;
00060     const char *ostr;
00061 {
00062     extern char *program_name;
00063     static char *place = EMSG;      /* option letter processing */
00064     char *oli;              /* option letter list index */
00065 
00066     if (optreset || !*place) {      /* update scanning pointer */
00067         optreset = 0;
00068         if (optind >= nargc || *(place = nargv[optind]) != '-') {
00069             place = EMSG;
00070             return (-1);
00071         }
00072         if (place[1] && *++place == '-') {  /* found "--" */
00073             ++optind;
00074             place = EMSG;
00075             return (-1);
00076         }
00077     }                   /* option letter okay? */
00078     if ((optopt = (int)*place++) == (int)':' ||
00079         !(oli = strchr(ostr, optopt))) {
00080         /*
00081          * if the user didn't specify '-' as an option,
00082          * assume it means -1.
00083          */
00084         if (optopt == (int)'-')
00085             return (-1);
00086         if (!*place)
00087             ++optind;
00088         if (opterr && *ostr != ':')
00089             (void)fprintf(stderr,
00090                 "%s: illegal option -- %c\n", program_name, optopt);
00091         return (BADCH);
00092     }
00093     if (*++oli != ':') {            /* don't need argument */
00094         optarg = NULL;
00095         if (!*place)
00096             ++optind;
00097     }
00098     else {                  /* need an argument */
00099         if (*place)         /* no white space */
00100             optarg = place;
00101         else if (nargc <= ++optind) {   /* no arg */
00102             place = EMSG;
00103             if (*ostr == ':')
00104                 return (BADARG);
00105             if (opterr)
00106                 (void)fprintf(stderr,
00107                     "%s: option requires an argument -- %c\n",
00108                     program_name, optopt);
00109             return (BADCH);
00110         }
00111         else                /* white space */
00112             optarg = nargv[optind];
00113         place = EMSG;
00114         ++optind;
00115     }
00116     return (optopt);            /* dump back option letter */
00117 }
