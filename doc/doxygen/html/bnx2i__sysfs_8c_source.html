00054 static ssize_t bnx2i_set_sq_info(struct device *dev,
00055                  struct device_attribute *attr,
00056                  const char *buf, size_t count)
00057 {
00058     struct bnx2i_hba *hba = bnx2i_dev_to_hba(dev);
00059     u32 val;
00060     int max_sq_size;
00061 
00062     if (hba->ofld_conns_active)
00063         goto skip_config;
00064 
00065     if (test_bit(BNX2I_NX2_DEV_57710, &hba->cnic_dev_type))
00066         max_sq_size = BNX2I_5770X_SQ_WQES_MAX;
00067     else
00068         max_sq_size = BNX2I_570X_SQ_WQES_MAX;
00069 
00070     if (sscanf(buf, " 0x%x ", &val) > 0) {
00071         if ((val >= BNX2I_SQ_WQES_MIN) && (val <= max_sq_size) &&
00072             (is_power_of_2(val)))
00073             hba->max_sqes = val;
00074     }
00075 
00076     return count;
00077

00079     printk(KERN_ERR "bnx2i: device busy, cannot change SQ size\n");
00080     return 0;
