00021 int parse_pkt(u_char *pkt, struct pfring_pkthdr *hdr)
00022 {
00023   struct iphdr *ip;
00024   struct eth_hdr *eh = (struct eth_hdr*)pkt;
00025   u_int16_t displ;
00026 
00027   memset(&hdr->extended_hdr.parsed_pkt, 0, sizeof(struct pkt_parsing_info));
00028 
00029   hdr->extended_hdr.parsed_header_len = 0;
00030   hdr->extended_hdr.parsed_pkt.eth_type = ntohs(eh->h_proto);
00031   hdr->extended_hdr.parsed_pkt.offset.eth_offset = 0;
00032 
00033   if(hdr->extended_hdr.parsed_pkt.eth_type == 0x8100 /* 802.1q (VLAN) */)
00034     {
00035       hdr->extended_hdr.parsed_pkt.offset.vlan_offset = hdr->extended_hdr.parsed_pkt.offset.eth_offset + sizeof(struct eth_hdr);
00036       hdr->extended_hdr.parsed_pkt.vlan_id = (pkt[hdr->extended_hdr.parsed_pkt.offset.eth_offset + 14] & 15) * 256
00037     + pkt[hdr->extended_hdr.parsed_pkt.offset.eth_offset + 15];
00038       hdr->extended_hdr.parsed_pkt.eth_type = (pkt[hdr->extended_hdr.parsed_pkt.offset.eth_offset + 16]) * 256
00039     + pkt[hdr->extended_hdr.parsed_pkt.offset.eth_offset + 17];
00040       displ = 4;
00041     }
00042   else
00043     {
00044       displ = 0;
00045       hdr->extended_hdr.parsed_pkt.vlan_id = 0; /* Any VLAN */
00046     }
00047 
00048   if(hdr->extended_hdr.parsed_pkt.eth_type == 0x0800 /* IP */) {
00049     hdr->extended_hdr.parsed_pkt.offset.l3_offset = hdr->extended_hdr.parsed_pkt.offset.eth_offset+displ+sizeof(struct eth_hdr);
00050     ip = (struct iphdr*)(pkt+hdr->extended_hdr.parsed_pkt.offset.l3_offset);
00051 
00052     hdr->extended_hdr.parsed_pkt.ipv4_src = ntohl(ip->saddr), hdr->extended_hdr.parsed_pkt.ipv4_dst = ntohl(ip->daddr), hdr->extended_hdr.parsed_pkt.l3_proto = ip->protocol;
00053     hdr->extended_hdr.parsed_pkt.ipv4_tos = ip->tos;
00054 
00055     if((ip->protocol == IPPROTO_TCP) || (ip->protocol == IPPROTO_UDP)) {
00056       u_int16_t ip_len = ip->ihl*4;
00057 
00058       hdr->extended_hdr.parsed_pkt.offset.l4_offset = hdr->extended_hdr.parsed_pkt.offset.l3_offset+ip_len;
00059 
00060       if(ip->protocol == IPPROTO_TCP) {
00061     struct tcphdr *tcp = (struct tcphdr*)(pkt+hdr->extended_hdr.parsed_pkt.offset.l4_offset);
00062     hdr->extended_hdr.parsed_pkt.l4_src_port = ntohs(tcp->source), hdr->extended_hdr.parsed_pkt.l4_dst_port = ntohs(tcp->dest);
00063     hdr->extended_hdr.parsed_pkt.offset.payload_offset = hdr->extended_hdr.parsed_pkt.offset.l4_offset+(tcp->doff * 4);
00064     hdr->extended_hdr.parsed_pkt.tcp.seq_num = ntohl(tcp->seq);
00065     hdr->extended_hdr.parsed_pkt.tcp.ack_num = ntohl(tcp->ack_seq);
00066     hdr->extended_hdr.parsed_pkt.tcp.flags = (tcp->fin * TH_FIN_MULTIPLIER) + (tcp->syn * TH_SYN_MULTIPLIER) + (tcp->rst * TH_RST_MULTIPLIER) +
00067       (tcp->psh * TH_PUSH_MULTIPLIER) + (tcp->ack * TH_ACK_MULTIPLIER) + (tcp->urg * TH_URG_MULTIPLIER);
00068       } else if(ip->protocol == IPPROTO_UDP) {
00069     struct udphdr *udp = (struct udphdr*)(pkt+hdr->extended_hdr.parsed_pkt.offset.l4_offset);
00070     hdr->extended_hdr.parsed_pkt.l4_src_port = ntohs(udp->source), hdr->extended_hdr.parsed_pkt.l4_dst_port = ntohs(udp->dest);
00071     hdr->extended_hdr.parsed_pkt.offset.payload_offset = hdr->extended_hdr.parsed_pkt.offset.l4_offset+sizeof(struct udphdr);
00072       } else
00073     hdr->extended_hdr.parsed_pkt.offset.payload_offset = hdr->extended_hdr.parsed_pkt.offset.l4_offset;
00074     } else
00075       hdr->extended_hdr.parsed_pkt.l4_src_port = hdr->extended_hdr.parsed_pkt.l4_dst_port = 0;
00076 
00077     return(1); /* IP */
00078   } /* TODO: handle IPv6 */
00079 
00080   return(0); /* No IP */
00081 }

00085 int set_if_promisc(const char *device, int set_promisc) {
00086   int sock_fd, ret = 0;
00087   struct ifreq ifr;
00088 
00089   if(device == NULL) return(-3);
00090 
00091   sock_fd = socket(PF_PACKET, SOCK_RAW, htons(ETH_P_ALL));
00092   if(sock_fd <= 0) return(-1);
00093 
00094   memset(&ifr, 0, sizeof(ifr));
00095   strncpy(ifr.ifr_name, device, sizeof(ifr.ifr_name));
00096   if(ioctl(sock_fd, SIOCGIFFLAGS, &ifr) == -1) {
00097     close(sock_fd);
00098     return(-2);
00099   }
00100 
00101   ret = ifr.ifr_flags & IFF_PROMISC;
00102   if(set_promisc) {
00103     if(ret == 0) ifr.ifr_flags |= IFF_PROMISC;
00104   } else {
00105     /* Remove promisc */
00106     if(ret != 0) ifr.ifr_flags &= ~IFF_PROMISC;
00107   }
00108 
00109   if(ioctl(sock_fd, SIOCSIFFLAGS, &ifr) == -1)
00110     return(-1);
00111 
00112   close(sock_fd);
00113   return(ret);
00114 }

00118 char* format_numbers(double val, char *buf, u_int buf_len, u_int8_t add_decimals) {
00119   u_int a1 = ((u_long)val / 1000000000) % 1000;
00120   u_int a = ((u_long)val / 1000000) % 1000;
00121   u_int b = ((u_long)val / 1000) % 1000;
00122   u_int c = (u_long)val % 1000;
00123   u_int d = (u_int)((val - (u_long)val)*100) % 100;  
00124 
00125   if(add_decimals) {
00126     if(val >= 1000000000) {
00127       snprintf(buf, buf_len, "%u'%03u'%03u'%03u.%02d", a1, a, b, c, d);
00128     } else if(val >= 1000000) {
00129       snprintf(buf, buf_len, "%u'%03u'%03u.%02d", a, b, c, d);
00130     } else if(val >= 100000) {
00131       snprintf(buf, buf_len, "%u'%03u.%02d", b, c, d);
00132     } else if(val >= 1000) {
00133       snprintf(buf, buf_len, "%u'%03u.%02d", b, c, d);
00134     } else
00135       snprintf(buf, buf_len, "%.2f", val);
00136   } else {
00137     if(val >= 1000000000) {
00138       snprintf(buf, buf_len, "%u'%03u'%03u'%03u", a1, a, b, c);
00139     } else if(val >= 1000000) {
00140       snprintf(buf, buf_len, "%u'%03u'%03u", a, b, c);
00141     } else if(val >= 100000) {
00142       snprintf(buf, buf_len, "%u'%03u", b, c);
00143     } else if(val >= 1000) {
00144       snprintf(buf, buf_len, "%u'%03u", b, c);
00145     } else
00146       snprintf(buf, buf_len, "%u", (unsigned int)val);
00147   }
00148 
00149   return(buf);
00150 }
