00118 {
00119     register const u_char *cm = charmap, *us1 = (const u_char *) s1, *us2 = (const u_char *) s2;
00120 
00121     while (cm[*us1] == cm[*us2++])
00122         if (*us1++ == '\0')
00123             return (0);
00124     return (cm[*us1] - cm[*--us2]);

00164 struct pcap_etherent *pcap_next_etherent(FILE * fp)
00165 {
00166     register int c, d, i;
00167     char *bp;
00168     static struct pcap_etherent e;
00169 
00170     memset((char *) &e, 0, sizeof(e));
00171     do
00172     {
00173         /* Find addr */
00174         c = skip_space(fp);
00175         if (c == '\n')
00176             continue;
00177 
00178         /* If this is a comment, or first thing on line
00179            cannot be etehrnet address, skip the line. */
00180         if (!isxdigit(c))
00181         {
00182             c = skip_line(fp);
00183             continue;
00184         }
00185 
00186         /* must be the start of an address */
00187         for (i = 0; i < 6; i += 1)
00188         {
00189             d = xdtoi(c);
00190             c = getc(fp);
00191             if (isxdigit(c))
00192             {
00193                 d <<= 4;
00194                 d |= xdtoi(c);
00195                 c = getc(fp);
00196             }
00197             e.addr[i] = d;
00198             if (c != ':')
00199                 break;
00200             c = getc(fp);
00201         }
00202         if (c == EOF)
00203             break;
00204 
00205         /* Must be whitespace */
00206         if (!isspace(c))
00207         {
00208             c = skip_line(fp);
00209             continue;
00210         }
00211         c = skip_space(fp);
00212 
00213         /* hit end of line... */
00214         if (c == '\n')
00215             continue;
00216 
00217         if (c == '#')
00218         {
00219             c = skip_line(fp);
00220             continue;
00221         }
00222 
00223         /* pick up name */
00224         bp = e.name;
00225         /* Use 'd' to prevent buffer overflow. */
00226         d = sizeof(e.name) - 1;
00227         do
00228         {
00229             *bp++ = c;
00230             c = getc(fp);
00231         } while (!isspace(c) && c != EOF && --d > 0);
00232         *bp = '\0';
00233 
00234         /* Eat trailing junk */
00235         if (c != '\n')
00236             (void) skip_line(fp);
00237 
00238         return &e;

00240     } while (c != EOF);
