00044 int ixgbe_set_vf_multicasts(struct ixgbe_adapter *adapter,
00045                 int entries, u16 *hash_list, u32 vf)
00046 {
00047     struct vf_data_storage *vfinfo = &adapter->vfinfo[vf];
00048     struct ixgbe_hw *hw = &adapter->hw;
00049     int i;
00050     u32 vector_bit;
00051     u32 vector_reg;
00052     u32 mta_reg;
00053     u32 vmolr = IXGBE_READ_REG(hw, IXGBE_VMOLR(vf));
00054 
00055     /* only so many hash values supported */
00056     entries = min(entries, IXGBE_MAX_VF_MC_ENTRIES);
00057 
00058     /* salt away the number of multi cast addresses assigned
00059      * to this VF for later use to restore when the PF multi cast
00060      * list changes
00061      */
00062     vfinfo->num_vf_mc_hashes = entries;
00063 
00064     /* VFs are limited to using the MTA hash table for their multicast
00065      * addresses */
00066     for (i = 0; i < entries; i++) {
00067         vfinfo->vf_mc_hashes[i] = hash_list[i];;
00068     }
00069 
00070     for (i = 0; i < vfinfo->num_vf_mc_hashes; i++) {
00071         vector_reg = (vfinfo->vf_mc_hashes[i] >> 5) & 0x7F;
00072         vector_bit = vfinfo->vf_mc_hashes[i] & 0x1F;
00073         mta_reg = IXGBE_READ_REG(hw, IXGBE_MTA(vector_reg));
00074         mta_reg |= (1 << vector_bit);
00075         IXGBE_WRITE_REG(hw, IXGBE_MTA(vector_reg), mta_reg);
00076     }
00077     vmolr |= IXGBE_VMOLR_ROMPE;
00078     IXGBE_WRITE_REG(hw, IXGBE_VMOLR(vf), vmolr);
00079 
00080     return 0;
00081 }

00098             vector_bit = vfinfo->vf_mc_hashes[j] & 0x1F;

00100             mta_reg |= (1 << vector_bit);

00102         }
00103         if (vfinfo->num_vf_mc_hashes)
00104             vmolr |= IXGBE_VMOLR_ROMPE;

00143 {

00145     int rar_entry = hw->mac.num_rar_entries - (vf + 1);
00146 
00147     /* reset offloads to defaults */

00150                   adapter->vfinfo[vf].pf_vlan, vf);

00210     u32 reg;

00232 static int ixgbe_rcv_msg_from_vf(struct ixgbe_adapter *adapter, u32 vf)
00233 {
00234     u32 mbx_size = IXGBE_VFMAILBOX_SIZE;
00235     u32 msgbuf[mbx_size];
00236     struct ixgbe_hw *hw = &adapter->hw;
00237     s32 retval;
00238     int entries;
00239     u16 *hash_list;
00240     int add, vid;
00241     u8 *new_mac;
00242 
00243     retval = ixgbe_read_mbx(hw, msgbuf, mbx_size, vf);
00244 
00245     if (retval)
00246         printk(KERN_ERR "Error receiving message from VF\n");
00247 
00248     /* this is a message we already processed, do nothing */
00249     if (msgbuf[0] & (IXGBE_VT_MSGTYPE_ACK | IXGBE_VT_MSGTYPE_NACK))
00250         return retval;
00251 
00252     /*
00253      * until the vf completes a virtual function reset it should not be
00254      * allowed to start any configuration.
00255      */
00256 
00257     if (msgbuf[0] == IXGBE_VF_RESET) {
00258         unsigned char *vf_mac = adapter->vfinfo[vf].vf_mac_addresses;
00259         new_mac = (u8 *)(&msgbuf[1]);
00260         adapter->vfinfo[vf].clear_to_send = false;
00261         ixgbe_vf_reset_msg(adapter, vf);
00262         adapter->vfinfo[vf].clear_to_send = true;
00263 
00264         if (is_valid_ether_addr(new_mac) &&
00265             !adapter->vfinfo[vf].pf_set_mac)
00266             ixgbe_set_vf_mac(adapter, vf, vf_mac);
00267         else
00268             ixgbe_set_vf_mac(adapter,
00269                  vf, adapter->vfinfo[vf].vf_mac_addresses);
00270 
00271         /* reply to reset with ack and vf mac address */
00272         msgbuf[0] = IXGBE_VF_RESET | IXGBE_VT_MSGTYPE_ACK;
00273         memcpy(new_mac, vf_mac, IXGBE_ETH_LENGTH_OF_ADDRESS);
00274         /* Piggyback the multicast filter type so VF can compute the
00275          * correct vectors */
00276         msgbuf[3] = hw->mac.mc_filter_type;
00277         ixgbe_write_mbx(hw, msgbuf, IXGBE_VF_PERMADDR_MSG_LEN, vf);
00278 
00279         return retval;
00280     }
00281 
00282     if (!adapter->vfinfo[vf].clear_to_send) {
00283         msgbuf[0] |= IXGBE_VT_MSGTYPE_NACK;
00284         ixgbe_write_mbx(hw, msgbuf, 1, vf);
00285         return retval;
00286     }
00287 
00288     switch ((msgbuf[0] & 0xFFFF)) {
00289     case IXGBE_VF_SET_MAC_ADDR:
00290         new_mac = ((u8 *)(&msgbuf[1]));
00291         if (is_valid_ether_addr(new_mac) &&
00292             !adapter->vfinfo[vf].pf_set_mac) {
00293             ixgbe_set_vf_mac(adapter, vf, new_mac);
00294             DPRINTK(PROBE, INFO,
00295                 "Set MAC msg received from VF %d\n", vf);
00296         } else if (memcmp(adapter->vfinfo[vf].vf_mac_addresses,
00297                   new_mac, ETH_ALEN)) {
00298             DPRINTK(PROBE, INFO,
00299                 "VF %d attempted to override administratively "
00300                 "set MAC address\nReload the VF driver to "
00301                 "resume operations\n", vf);
00302             retval = -1;
00303         }
00304         break;
00305     case IXGBE_VF_SET_MULTICAST:
00306         entries = (msgbuf[0] & IXGBE_VT_MSGINFO_MASK)
00307                     >> IXGBE_VT_MSGINFO_SHIFT;
00308         hash_list = (u16 *)&msgbuf[1];
00309         retval = ixgbe_set_vf_multicasts(adapter, entries,
00310                          hash_list, vf);
00311         break;
00312     case IXGBE_VF_SET_LPE:
00313         DPRINTK(PROBE, INFO, "Set LPE msg received from vf %d\n", vf);
00314         ixgbe_set_vf_lpe(adapter, msgbuf);
00315         break;
00316     case IXGBE_VF_SET_VLAN:
00317         add = (msgbuf[0] & IXGBE_VT_MSGINFO_MASK)
00318                     >> IXGBE_VT_MSGINFO_SHIFT;
00319         vid = (msgbuf[1] & IXGBE_VLVF_VLANID_MASK);
00320         if (adapter->vfinfo[vf].pf_vlan) {
00321             DPRINTK(PROBE, INFO,
00322                 "VF %d attempted to override administratively "
00323                 "set VLAN configuration\nReload the VF driver "
00324                 "to resume operations\n", vf);
00325             retval = -1;
00326         } else {
00327             retval = ixgbe_set_vf_vlan(adapter, add, vid, vf);
00328         }
00329         break;
00330     default:
00331         DPRINTK(PROBE, ERR, "Unhandled Msg %8.8x\n", msgbuf[0]);
00332         retval = IXGBE_ERR_MBX;
00333         break;
00334     }
00335 
00336     /* notify the VF of the results of what it sent us */
00337     if (retval)
00338         msgbuf[0] |= IXGBE_VT_MSGTYPE_NACK;
00339     else
00340         msgbuf[0] |= IXGBE_VT_MSGTYPE_ACK;
00341 
00342     msgbuf[0] |= IXGBE_VT_MSGTYPE_CTS;
00343 
00344     ixgbe_write_mbx(hw, msgbuf, 1, vf);
00345 
00346     return retval;
00347 }
