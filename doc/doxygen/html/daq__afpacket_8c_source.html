00182 static int set_up_rx_ring(AFPacket_Context_t *afpc, AFPacketInstance *instance)
00183 {
00184     unsigned idx, block, frame, ringsize;
00185 
00186     /* Memory map the RX ring. */
00187     ringsize = instance->layout.tp_block_nr * instance->layout.tp_block_size;
00188     instance->buffer = mmap(0, ringsize, PROT_READ | PROT_WRITE, MAP_SHARED, instance->fd, 0);
00189     if (instance->buffer == MAP_FAILED)
00190     {
00191         DPE(afpc->errbuf, "%s: Couldn't MMAP the RX ring: %s", __FUNCTION__, strerror(errno));
00192 
00193         /* Destroy the kernel RX ring on error. */
00194         destroy_rx_ring(instance);
00195         return DAQ_ERROR;
00196     }
00197 
00198     /* Allocate a ring to hold packet pointers. */
00199     instance->ring = calloc(instance->layout.tp_frame_nr, sizeof(AFPacketEntry));
00200     if (!instance->ring)
00201     {
00202         DPE(afpc->errbuf, "%s: Could not allocate entry ring for device %s", __FUNCTION__, instance->name);
00203         destroy_rx_ring(instance);
00204         return DAQ_ERROR_NOMEM;
00205     }
00206 
00207     /* Set up the buffer entry pointers in the ring. */
00208     idx = 0;
00209     for (block = 0; block < instance->layout.tp_block_nr; block++)
00210     {
00211         for (frame = 0; frame < instance->frames_per_block && idx < instance->layout.tp_frame_nr; frame++)
00212         {
00213             instance->ring[idx].begin = (uint8_t *) instance->buffer + (block * instance->layout.tp_block_size) + (frame * instance->layout.tp_frame_size);
00214             instance->ring[idx].hdr.raw = instance->ring[idx].begin;
00215             instance->ring[idx].next = &instance->ring[idx + 1];
00216             idx++;
00217         }
00218     }
00219     /* Make this a circular buffer ... a RING if you will! */
00220     instance->ring[instance->layout.tp_frame_nr - 1].next = &instance->ring[0];
00221     /* Initialize our entry point into the ring as the first buffer entry. */
00222     instance->entry = &instance->ring[0];
00223 
00224     return DAQ_SUCCESS;
00225 }

00341 static int calculate_layout(AFPacket_Context_t *afpc, AFPacketInstance *instance, int order)
00342 {
00343     /* Calculate the frame size and minimum block size required. */
00344     instance->layout.tp_frame_size = TPACKET_ALIGN(afpc->snaplen + TPACKET_ALIGN(TPACKET_ALIGN(instance->tp_hdrlen) + sizeof(struct sockaddr_ll) + ETH_HLEN) - ETH_HLEN);
00345     instance->layout.tp_block_size = getpagesize() << order;
00346     while (instance->layout.tp_block_size < instance->layout.tp_frame_size)
00347         instance->layout.tp_block_size <<= 1;
00348     instance->frames_per_block = instance->layout.tp_block_size / instance->layout.tp_frame_size;
00349     if (instance->frames_per_block == 0)
00350     {
00351         DPE(afpc->errbuf, "%s: Invalid frames per block (%u/%u) for %s",
00352                 __FUNCTION__, instance->layout.tp_block_size, instance->layout.tp_frame_size, afpc->device);
00353         return DAQ_ERROR;
00354     }
00355 
00356     /* Find the total number of frames required to amount to the requested per-interface memory.
00357         Then find the number of blocks required to hold those packet buffer frames. */
00358     instance->layout.tp_frame_nr = afpc->size / instance->layout.tp_frame_size;
00359     instance->layout.tp_block_nr = instance->layout.tp_frame_nr / instance->frames_per_block;
00360     /* afpc->layout.tp_frame_nr is requested to match frames_per_block*n_blocks */
00361     instance->layout.tp_frame_nr = instance->layout.tp_block_nr * instance->frames_per_block;
00362     if (afpc->debug)
00363     {
00364         printf("AFPacket Layout:\n");
00365         printf("  Frame Size: %u\n", instance->layout.tp_frame_size);
00366         printf("  Frames:     %u\n", instance->layout.tp_frame_nr);
00367         printf("  Block Size: %u (Order %d)\n", instance->layout.tp_block_size, order);
00368         printf("  Blocks:     %u\n", instance->layout.tp_block_nr);
00369     }
00370 
00371     return DAQ_SUCCESS;
00372 }

00374 #define DEFAULT_ORDER 3
00375 static int create_rx_ring(AFPacket_Context_t *afpc, AFPacketInstance *instance)
00376 {
00377     int rc, order;
00378 
00379     /* Starting with page allocations of order 3, try to allocate an RX ring in the kernel. */
00380     for (order = DEFAULT_ORDER; order >= 0; order--)
00381     {
00382         if (calculate_layout(afpc, instance, order))
00383             return DAQ_ERROR;
00384 
00385         /* Ask the kernel to create the ring. */
00386         rc = setsockopt(instance->fd, SOL_PACKET, PACKET_RX_RING, (void*) &instance->layout, sizeof(struct tpacket_req));
00387         if (rc)
00388         {
00389             if (errno == ENOMEM)
00390             {
00391                 if (afpc->debug)
00392                     printf("%s: Allocation of kernel packet RX ring failed with order %d, retrying...\n", instance->name, order);
00393                 continue;
00394             }
00395             DPE(afpc->errbuf, "%s: Couldn't create kernel RX ring on packet socket: %s",
00396                     __FUNCTION__, strerror(errno));
00397             return DAQ_ERROR;
00398         }
00399         return DAQ_SUCCESS;
00400     }
00401 
00402     /* If we got here, it means we failed allocation on order 0. */
00403     DPE(afpc->errbuf, "%s: Couldn't allocate enough memory for the kernel RX ring!", instance->name);
00404     return DAQ_ERROR;
00405 }

00407 static int start_instance(AFPacket_Context_t *afpc, AFPacketInstance *instance)
00408 {
00409     struct packet_mreq mr;
00410     int arptype;
00411 
00412     /* Bind to the specified device so we only see packets from it. */
00413     if (bind_interface(afpc, instance) != 0)
00414         return -1;
00415 
00416     /* Turn on promiscuous mode for the device. */
00417     memset(&mr, 0, sizeof(mr));
00418     mr.mr_ifindex = instance->index;
00419     mr.mr_type = PACKET_MR_PROMISC;
00420     if (setsockopt(instance->fd, SOL_PACKET, PACKET_ADD_MEMBERSHIP, &mr, sizeof(mr)) == -1)
00421     {
00422         DPE(afpc->errbuf, "%s: setsockopt: %s", __FUNCTION__, strerror(errno));
00423         return -1;
00424     }
00425 
00426     /* Get the link-layer type. */
00427     arptype = iface_get_arptype(instance);
00428     if (arptype < 0)
00429     {
00430         DPE(afpc->errbuf, "%s: failed to get interface type for device %s: (%d) %s",
00431                 __FUNCTION__, instance->name, errno, strerror(errno));
00432         return -1;
00433     }
00434 
00435     if (arptype != ARPHRD_ETHER)
00436     {
00437         DPE(afpc->errbuf, "%s: invalid interface type for device %s: %d != %d",
00438                 __FUNCTION__, instance->name, arptype, ARPHRD_ETHER);
00439         return -1;
00440     }
00441 
00442     /* Determine which versions of TPACKET the socket supports. */
00443     if (determine_version(afpc, instance) != DAQ_SUCCESS)
00444         return -1;
00445 
00446     if (afpc->debug)
00447     {
00448         printf("Version: %u\n", instance->tp_version);
00449         printf("Header Length: %u\n", instance->tp_hdrlen);
00450     }
00451 
00452     if (create_rx_ring(afpc, instance) != DAQ_SUCCESS)
00453         return -1;
00454 
00455     if (set_up_rx_ring(afpc, instance) != DAQ_SUCCESS)
00456         return -1;
00457 
00458     return 0;
00459 }

00552     if (!afpc)

00602             {

00604                 {
00605                     name1 = afpc->instances->next->name;
