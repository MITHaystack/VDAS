00137 #define UPDATE_VF_COUNTER_32bit(reg, last_counter, counter) \
00138     {                           \
00139         u32 current_counter = IXGBE_READ_REG(hw, reg);  \
00140         if (current_counter < last_counter)     \
00141             counter += 0x100000000LL;       \
00142         last_counter = current_counter;         \
00143         counter &= 0xFFFFFFFF00000000LL;        \
00144         counter |= current_counter;         \
00145     }
00146 
00147 #define UPDATE_VF_COUNTER_36bit(reg_lsb, reg_msb, last_counter, counter) \
00148     {                                \
00149         u64 current_counter_lsb = IXGBE_READ_REG(hw, reg_lsb);   \
00150         u64 current_counter_msb = IXGBE_READ_REG(hw, reg_msb);   \
00151         u64 current_counter = (current_counter_msb << 32) |      \
00152             current_counter_lsb;                             \
00153         if (current_counter < last_counter)          \
00154             counter += 0x1000000000LL;           \
00155         last_counter = current_counter;              \
00156         counter &= 0xFFFFFFF000000000LL;             \
00157         counter |= current_counter;              \
00158     }
