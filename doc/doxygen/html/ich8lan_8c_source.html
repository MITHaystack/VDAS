00394     struct e1000_dev_spec_ich8lan *dev_spec = &hw->dev_spec.ich8lan;
00395     u32 gfpreg, sector_base_addr, sector_end_addr;
00396     s32 ret_val = E1000_SUCCESS;
00397     u16 i;
00398 
00399     /* Can't read flash registers if the register set isn't mapped. */
00400     if (!hw->flash_address) {
00401         e_dbg("ERROR: Flash registers not mapped\n");
00402         ret_val = -E1000_ERR_CONFIG;
00403         goto out;
00404     }
00405 
00406     nvm->type = e1000_nvm_flash_sw;
00407 
00408     gfpreg = er32flash(ICH_FLASH_GFPREG);
00409 
00410     /*
00411      * sector_X_addr is a "sector"-aligned address (4096 bytes)
00412      * Add 1 to sector_end_addr since this sector is included in
00413      * the overall size.
00414      */
00415     sector_base_addr = gfpreg & FLASH_GFPREG_BASE_MASK;
00416     sector_end_addr = ((gfpreg >> 16) & FLASH_GFPREG_BASE_MASK) + 1;
00417 
00418     /* flash_base_addr is byte-aligned */
00419     nvm->flash_base_addr = sector_base_addr << FLASH_SECTOR_ADDR_SHIFT;
00420 
00421     /*
00422      * find total size of the NVM, then cut in half since the total
00423      * size represents two separate NVM banks.
00424      */
00425     nvm->flash_bank_size = (sector_end_addr - sector_base_addr)
00426                               << FLASH_SECTOR_ADDR_SHIFT;
00427     nvm->flash_bank_size /= 2;
00428     /* Adjust to word count */
00429     nvm->flash_bank_size /= sizeof(u16);
00430 
00431     nvm->word_size = E1000_ICH8_SHADOW_RAM_WORDS;
00432 
00433     /* Clear shadow ram */
00434     for (i = 0; i < nvm->word_size; i++) {
00435         dev_spec->shadow_ram[i].modified = false;
00436         dev_spec->shadow_ram[i].value    = 0xFFFF;
00437     }
00438 
00439     /* Function Pointers */
00440     nvm->ops.acquire       = e1000_acquire_nvm_ich8lan;
00441     nvm->ops.release       = e1000_release_nvm_ich8lan;
00442     nvm->ops.read          = e1000_read_nvm_ich8lan;
00443     nvm->ops.update        = e1000_update_nvm_checksum_ich8lan;
00444     nvm->ops.valid_led_default = e1000_valid_led_default_ich8lan;
00445     nvm->ops.validate      = e1000_validate_nvm_checksum_ich8lan;
00446     nvm->ops.write         = e1000_write_nvm_ich8lan;

00448 out:

00954 static s32 e1000_sw_lcd_config_ich8lan(struct e1000_hw *hw)
00955 {
00956     struct e1000_phy_info *phy = &hw->phy;
00957     u32 i, data, cnf_size, cnf_base_addr, sw_cfg_mask;
00958     s32 ret_val = E1000_SUCCESS;
00959     u16 word_addr, reg_data, reg_addr, phy_page = 0;
00960 
00961     /*
00962      * Initialize the PHY from the NVM on ICH platforms.  This
00963      * is needed due to an issue where the NVM configuration is
00964      * not properly autoloaded after power transitions.
00965      * Therefore, after each PHY reset, we will load the
00966      * configuration data out of the NVM manually.
00967      */
00968     switch (hw->mac.type) {
00969     case e1000_ich8lan:
00970         if (phy->type != e1000_phy_igp_3)
00971             return ret_val;
00972 
00973         if ((hw->adapter->pdev->device == E1000_DEV_ID_ICH8_IGP_AMT) ||
00974             (hw->adapter->pdev->device == E1000_DEV_ID_ICH8_IGP_C)) {
00975             sw_cfg_mask = E1000_FEXTNVM_SW_CONFIG;
00976             break;
00977         }
00978         /* Fall-thru */
00979     case e1000_pchlan:
00980     case e1000_pch2lan:
00981         sw_cfg_mask = E1000_FEXTNVM_SW_CONFIG_ICH8M;
00982         break;
00983     default:
00984         return ret_val;
00985     }
00986 
00987     ret_val = hw->phy.ops.acquire(hw);
00988     if (ret_val)
00989         return ret_val;
00990 
00991     data = er32(FEXTNVM);
00992     if (!(data & sw_cfg_mask))
00993         goto out;
00994 
00995     /*
00996      * Make sure HW does not configure LCD from PHY
00997      * extended configuration before SW configuration
00998      */
00999     data = er32(EXTCNF_CTRL);
01000     if (!(hw->mac.type == e1000_pch2lan)) {
01001         if (data & E1000_EXTCNF_CTRL_LCD_WRITE_ENABLE)
01002             goto out;
01003     }
01004 
01005     cnf_size = er32(EXTCNF_SIZE);
01006     cnf_size &= E1000_EXTCNF_SIZE_EXT_PCIE_LENGTH_MASK;
01007     cnf_size >>= E1000_EXTCNF_SIZE_EXT_PCIE_LENGTH_SHIFT;
01008     if (!cnf_size)
01009         goto out;
01010 
01011     cnf_base_addr = data & E1000_EXTCNF_CTRL_EXT_CNF_POINTER_MASK;
01012     cnf_base_addr >>= E1000_EXTCNF_CTRL_EXT_CNF_POINTER_SHIFT;
01013 
01014     if ((!(data & E1000_EXTCNF_CTRL_OEM_WRITE_ENABLE) &&
01015         (hw->mac.type == e1000_pchlan)) ||
01016          (hw->mac.type == e1000_pch2lan)) {
01017         /*
01018          * HW configures the SMBus address and LEDs when the
01019          * OEM and LCD Write Enable bits are set in the NVM.
01020          * When both NVM bits are cleared, SW will configure
01021          * them instead.
01022          */
01023         ret_val = e1000_write_smbus_addr(hw);
01024         if (ret_val)
01025             goto out;
01026 
01027         data = er32(LEDCTL);
01028         ret_val = e1000_write_phy_reg_hv_locked(hw, HV_LED_CONFIG,
01029                             (u16)data);
01030         if (ret_val)

01033 
01034     /* Configure LCD from extended configuration region. */
01035 
01036     /* cnf_base_addr is in DWORD */

01038

01040         ret_val = e1000_read_nvm(hw, (word_addr + i * 2), 1,

01042         if (ret_val)
01043             goto out;
01044

01046                        1, &reg_addr);
01047         if (ret_val)
01048             goto out;
01049 
01050         /* Save off the PHY page for future writes. */
01051         if (reg_addr == IGP01E1000_PHY_PAGE_SELECT) {
01052             phy_page = reg_data;
01053             continue;
01054         }
01055 
01056         reg_addr &= PHY_REG_MASK;
01057         reg_addr |= phy_page;
01058 
01059         ret_val = phy->ops.write_reg_locked(hw, (u32)reg_addr,
01060                             reg_data);
01061         if (ret_val)
01062             goto out;
01063     }
01064 
01065 out:
01066     hw->phy.ops.release(hw);
01067     return ret_val;
01068 }

02128 static s32 e1000_flash_cycle_init_ich8lan(struct e1000_hw *hw)
02129 {
02130     union ich8_hws_flash_status hsfsts;
02131     s32 ret_val = -E1000_ERR_NVM;
02132 
02133     hsfsts.regval = er16flash(ICH_FLASH_HSFSTS);
02134 
02135     /* Check if the flash descriptor is valid */
02136     if (hsfsts.hsf_status.fldesvalid == 0) {
02137         e_dbg("Flash descriptor invalid.  "
02138                  "SW Sequencing must be used.");
02139         goto out;
02140     }
02141 
02142     /* Clear FCERR and DAEL in hw status by writing 1 */
02143     hsfsts.hsf_status.flcerr = 1;
02144     hsfsts.hsf_status.dael = 1;
02145 
02146     ew16flash(ICH_FLASH_HSFSTS, hsfsts.regval);
02147 
02148     /*
02149      * Either we should have a hardware SPI cycle in progress
02150      * bit to check against, in order to start a new cycle or
02151      * FDONE bit should be changed in the hardware so that it
02152      * is 1 after hardware reset, which can then be used as an
02153      * indication whether a cycle is in progress or has been
02154      * completed.
02155      */
02156 
02157     if (hsfsts.hsf_status.flcinprog == 0) {
02158         /*
02159          * There is no cycle running at present,
02160          * so we can start a cycle.
02161          * Begin by setting Flash Cycle Done.
02162          */
02163         hsfsts.hsf_status.flcdone = 1;
02164         ew16flash(ICH_FLASH_HSFSTS, hsfsts.regval);
02165         ret_val = E1000_SUCCESS;
02166     } else {
02167         s32 i;
02168 
02169         /*
02170          * Otherwise poll for sometime so the current
02171          * cycle has a chance to end before giving up.
02172          */
02173         for (i = 0; i < ICH_FLASH_READ_COMMAND_TIMEOUT; i++) {
02174             hsfsts.regval = er16flash(
02175                                                   ICH_FLASH_HSFSTS);
02176             if (hsfsts.hsf_status.flcinprog == 0) {
02177                 ret_val = E1000_SUCCESS;
02178                 break;
02179             }
02180             udelay(1);
02181         }
02182         if (!ret_val) {
02183             /*
02184              * Successful in waiting for previous cycle to timeout,
02185              * now set the Flash Cycle Done.
02186              */
02187             hsfsts.hsf_status.flcdone = 1;
02188             ew16flash(ICH_FLASH_HSFSTS,
02189                                     hsfsts.regval);
02190         } else {
02191             e_dbg("Flash controller busy, cannot get access");
02192         }
02193     }
02194 
02195 out:
02196     return ret_val;
02197 }

02286 static s32 e1000_read_flash_data_ich8lan(struct e1000_hw *hw, u32 offset,
02287                                          u8 size, u16 *data)
02288 {
02289     union ich8_hws_flash_status hsfsts;
02290     union ich8_hws_flash_ctrl hsflctl;
02291     u32 flash_linear_addr;
02292     u32 flash_data = 0;
02293     s32 ret_val = -E1000_ERR_NVM;
02294     u8 count = 0;
02295 
02296     if (size < 1  || size > 2 || offset > ICH_FLASH_LINEAR_ADDR_MASK)
02297         goto out;
02298     flash_linear_addr = (ICH_FLASH_LINEAR_ADDR_MASK & offset) +
02299                         hw->nvm.flash_base_addr;
02300 
02301     do {
02302         udelay(1);
02303         /* Steps */
02304         ret_val = e1000_flash_cycle_init_ich8lan(hw);
02305         if (ret_val)
02306             break;
02307 
02308         hsflctl.regval = er16flash(ICH_FLASH_HSFCTL);
02309         /* 0b/1b corresponds to 1 or 2 byte size, respectively. */
02310         hsflctl.hsf_ctrl.fldbcount = size - 1;
02311         hsflctl.hsf_ctrl.flcycle = ICH_CYCLE_READ;
02312         ew16flash(ICH_FLASH_HSFCTL, hsflctl.regval);
02313 
02314         ew32flash(ICH_FLASH_FADDR, flash_linear_addr);
02315 
02316         ret_val = e1000_flash_cycle_ich8lan(hw,
02317                                         ICH_FLASH_READ_COMMAND_TIMEOUT);
02318 
02319         /*
02320          * Check if FCERR is set to 1, if set to 1, clear it
02321          * and try the whole sequence a few more times, else
02322          * read in (shift in) the Flash Data0, the order is
02323          * least significant byte first msb to lsb
02324          */
02325         if (!ret_val) {
02326             flash_data = er32flash(ICH_FLASH_FDATA0);
02327             if (size == 1)
02328                 *data = (u8)(flash_data & 0x000000FF);
02329             else if (size == 2)
02330                 *data = (u16)(flash_data & 0x0000FFFF);
02331             break;
02332         } else {
02333             /*
02334              * If we've gotten here, then things are probably
02335              * completely hosed, but if the error condition is
02336              * detected, it won't hurt to give it another try...
02337              * ICH_FLASH_CYCLE_REPEAT_COUNT times.
02338              */
02339             hsfsts.regval = er16flash(
02340                                                   ICH_FLASH_HSFSTS);
02341             if (hsfsts.hsf_status.flcerr == 1) {
02342                 /* Repeat for some time before giving up. */
02343                 continue;
02344             } else if (hsfsts.hsf_status.flcdone == 0) {
02345                 e_dbg("Timeout error - flash cycle "
02346                          "did not complete.");
02347                 break;
02348             }
02349         }
02350     } while (count++ < ICH_FLASH_CYCLE_REPEAT_COUNT);
02351 
02352 out:
02353     return ret_val;
02354 }

02404 static s32 e1000_update_nvm_checksum_ich8lan(struct e1000_hw *hw)
02405 {
02406     struct e1000_nvm_info *nvm = &hw->nvm;
02407     struct e1000_dev_spec_ich8lan *dev_spec = &hw->dev_spec.ich8lan;
02408     u32 i, act_offset, new_bank_offset, old_bank_offset, bank;
02409     s32 ret_val;
02410     u16 data;
02411 
02412     ret_val = e1000e_update_nvm_checksum_generic(hw);
02413     if (ret_val)
02414         goto out;
02415 
02416     if (nvm->type != e1000_nvm_flash_sw)
02417         goto out;
02418 
02419     nvm->ops.acquire(hw);
02420 
02421     /*
02422      * We're writing to the opposite bank so if we're on bank 1,
02423      * write to bank 0 etc.  We also need to erase the segment that
02424      * is going to be written
02425      */
02426     ret_val =  e1000_valid_nvm_bank_detect_ich8lan(hw, &bank);
02427     if (ret_val) {
02428         e_dbg("Could not detect valid bank, assuming bank 0\n");
02429         bank = 0;
02430     }
02431 
02432     if (bank == 0) {
02433         new_bank_offset = nvm->flash_bank_size;
02434         old_bank_offset = 0;
02435         ret_val = e1000_erase_flash_bank_ich8lan(hw, 1);
02436         if (ret_val)
02437             goto release;
02438     } else {
02439         old_bank_offset = nvm->flash_bank_size;
02440         new_bank_offset = 0;
02441         ret_val = e1000_erase_flash_bank_ich8lan(hw, 0);
02442         if (ret_val)
02443             goto release;
02444     }
02445 
02446     for (i = 0; i < E1000_ICH8_SHADOW_RAM_WORDS; i++) {
02447         /*
02448          * Determine whether to write the value stored
02449          * in the other NVM bank or a modified value stored
02450          * in the shadow RAM
02451          */
02452         if (dev_spec->shadow_ram[i].modified) {
02453             data = dev_spec->shadow_ram[i].value;
02454         } else {
02455             ret_val = e1000_read_flash_word_ich8lan(hw, i +
02456                                                     old_bank_offset,
02457                                                     &data);
02458             if (ret_val)
02459                 break;
02460         }
02461 
02462         /*
02463          * If the word is 0x13, then make sure the signature bits
02464          * (15:14) are 11b until the commit has completed.
02465          * This will allow us to write 10b which indicates the
02466          * signature is valid.  We want to do this after the write
02467          * has completed so that we don't mark the segment valid
02468          * while the write is still in progress
02469          */
02470         if (i == E1000_ICH_NVM_SIG_WORD)
02471             data |= E1000_ICH_NVM_SIG_MASK;
02472 
02473         /* Convert offset to bytes. */
02474         act_offset = (i + new_bank_offset) << 1;
02475 
02476         udelay(100);
02477         /* Write the bytes to the new bank. */
02478         ret_val = e1000_retry_write_flash_byte_ich8lan(hw,
02479                                                        act_offset,
02480                                                        (u8)data);
02481         if (ret_val)
02482             break;
02483 
02484         udelay(100);
02485         ret_val = e1000_retry_write_flash_byte_ich8lan(hw,
02486                                                   act_offset + 1,
02487                                                   (u8)(data >> 8));
02488         if (ret_val)
02489             break;
02490     }
02491 
02492     /*
02493      * Don't bother writing the segment valid bits if sector
02494      * programming failed.
02495      */
02496     if (ret_val) {
02497         e_dbg("Flash commit failed.\n");
02498         goto release;
02499     }
02500 
02501     /*
02502      * Finally validate the new segment by setting bit 15:14
02503      * to 10b in word 0x13 , this can be done without an
02504      * erase as well since these bits are 11 to start with
02505      * and we need to change bit 14 to 0b
02506      */
02507     act_offset = new_bank_offset + E1000_ICH_NVM_SIG_WORD;
02508     ret_val = e1000_read_flash_word_ich8lan(hw, act_offset, &data);
02509     if (ret_val)
02510         goto release;
02511 
02512     data &= 0xBFFF;
02513     ret_val = e1000_retry_write_flash_byte_ich8lan(hw,
02514                                                    act_offset * 2 + 1,
02515                                                    (u8)(data >> 8));
02516     if (ret_val)
02517         goto release;
02518 
02519     /*
02520      * And invalidate the previously valid segment by setting
02521      * its signature word (0x13) high_byte to 0b. This can be
02522      * done without an erase because flash erase sets all bits
02523      * to 1's. We can write 1's to 0's without an erase
02524      */
02525     act_offset = (old_bank_offset + E1000_ICH_NVM_SIG_WORD) * 2 + 1;
02526     ret_val = e1000_retry_write_flash_byte_ich8lan(hw, act_offset, 0);
02527     if (ret_val)
02528         goto release;
02529 
02530     /* Great!  Everything worked, we can now clear the cached entries. */
02531     for (i = 0; i < E1000_ICH8_SHADOW_RAM_WORDS; i++) {
02532         dev_spec->shadow_ram[i].modified = false;
02533         dev_spec->shadow_ram[i].value = 0xFFFF;
02534     }
02535 
02536 release:
02537     nvm->ops.release(hw);
02538 
02539     /*
02540      * Reload the EEPROM, or else modifications will not appear
02541      * until after the next adapter reset.
02542      */
02543     if (!ret_val) {
02544         nvm->ops.reload(hw);
02545         msleep(10);
02546     }
02547 
02548 out:
02549     if (ret_val)
02550         e_dbg("NVM update error: %d\n", ret_val);
02551 
02552     return ret_val;
02553 }

02635         if (size == 1)
02636             flash_data = (u32)data & 0x00FF;
02637         else
02638             flash_data = (u32)data;
02639 
02640         ew32flash(ICH_FLASH_FDATA0, flash_data);
02641 
02642         /*
02643          * check if FCERR is set to 1 , if set to 1, clear it
02644          * and try the whole sequence a few more times else done
02645          */
02646         ret_val = e1000_flash_cycle_ich8lan(hw,
02647                                        ICH_FLASH_WRITE_COMMAND_TIMEOUT);
02648         if (!ret_val)
02649             break;
02650 
02651         /*
02652          * If we're here, then things are most likely
02653          * completely hosed, but if the error condition
02654          * is detected, it won't hurt to give it another
02655          * try...ICH_FLASH_CYCLE_REPEAT_COUNT times.
02656          */
02657         hsfsts.regval = er16flash(ICH_FLASH_HSFSTS);
02658         if (hsfsts.hsf_status.flcerr == 1)
02659             /* Repeat for some time before giving up. */
02660             continue;
02661         if (hsfsts.hsf_status.flcdone == 0) {
02662             e_dbg("Timeout error - flash cycle "
02663                  "did not complete.");
02664             break;
02665         }
02666     } while (count++ < ICH_FLASH_CYCLE_REPEAT_COUNT);
02667 
02668 out:
02669     return ret_val;

02731 static s32 e1000_erase_flash_bank_ich8lan(struct e1000_hw *hw, u32 bank)
02732 {
02733     struct e1000_nvm_info *nvm = &hw->nvm;
02734     union ich8_hws_flash_status hsfsts;
02735     union ich8_hws_flash_ctrl hsflctl;
02736     u32 flash_linear_addr;
02737     /* bank size is in 16bit words - adjust to bytes */
02738     u32 flash_bank_size = nvm->flash_bank_size * 2;
02739     s32 ret_val = E1000_SUCCESS;
02740     s32 count = 0;
02741     s32 j, iteration, sector_size;
02742 
02743     hsfsts.regval = er16flash(ICH_FLASH_HSFSTS);
02744 
02745     /*
02746      * Determine HW Sector size: Read BERASE bits of hw flash status
02747      * register
02748      * 00: The Hw sector is 256 bytes, hence we need to erase 16
02749      *     consecutive sectors.  The start index for the nth Hw sector
02750      *     can be calculated as = bank * 4096 + n * 256
02751      * 01: The Hw sector is 4K bytes, hence we need to erase 1 sector.
02752      *     The start index for the nth Hw sector can be calculated
02753      *     as = bank * 4096
02754      * 10: The Hw sector is 8K bytes, nth sector = bank * 8192
02755      *     (ich9 only, otherwise error condition)
02756      * 11: The Hw sector is 64K bytes, nth sector = bank * 65536
02757      */
02758     switch (hsfsts.hsf_status.berasesz) {
02759     case 0:
02760         /* Hw sector size 256 */
02761         sector_size = ICH_FLASH_SEG_SIZE_256;
02762         iteration = flash_bank_size / ICH_FLASH_SEG_SIZE_256;
02763         break;
02764     case 1:
02765         sector_size = ICH_FLASH_SEG_SIZE_4K;
02766         iteration = 1;
02767         break;
02768     case 2:
02769         sector_size = ICH_FLASH_SEG_SIZE_8K;
02770         iteration = 1;
02771         break;
02772     case 3:
02773         sector_size = ICH_FLASH_SEG_SIZE_64K;
02774         iteration = 1;
02775         break;
02776     default:
02777         ret_val = -E1000_ERR_NVM;
02778         goto out;
02779     }
02780 
02781     /* Start with the base address, then add the sector offset. */
02782     flash_linear_addr = hw->nvm.flash_base_addr;
02783     flash_linear_addr += (bank) ? flash_bank_size : 0;
02784 
02785     for (j = 0; j < iteration ; j++) {
02786         do {
02787             /* Steps */
02788             ret_val = e1000_flash_cycle_init_ich8lan(hw);
02789             if (ret_val)
02790                 goto out;
02791 
02792             /*
02793              * Write a value 11 (block Erase) in Flash
02794              * Cycle field in hw flash control
02795              */
02796             hsflctl.regval = er16flash(
02797                                                   ICH_FLASH_HSFCTL);
02798             hsflctl.hsf_ctrl.flcycle = ICH_CYCLE_ERASE;
02799             ew16flash(ICH_FLASH_HSFCTL,
02800                                     hsflctl.regval);
02801 
02802             /*
02803              * Write the last 24 bits of an index within the
02804              * block into Flash Linear address field in Flash
02805              * Address.
02806              */
02807             flash_linear_addr += (j * sector_size);
02808             ew32flash(ICH_FLASH_FADDR,
02809                                   flash_linear_addr);
02810 
02811             ret_val = e1000_flash_cycle_ich8lan(hw,
02812                                    ICH_FLASH_ERASE_COMMAND_TIMEOUT);
02813             if (!ret_val)
02814                 break;
02815 
02816             /*
02817              * Check if FCERR is set to 1.  If 1,
02818              * clear it and try the whole sequence
02819              * a few more times else Done
02820              */
02821             hsfsts.regval = er16flash(
02822                               ICH_FLASH_HSFSTS);
02823             if (hsfsts.hsf_status.flcerr == 1)
02824                 /* repeat for some time before giving up */
02825                 continue;
02826             else if (hsfsts.hsf_status.flcdone == 0)
02827                 goto out;
02828         } while (++count < ICH_FLASH_CYCLE_REPEAT_COUNT);
02829     }
02830 
02831 out:
02832     return ret_val;
02833 }

02968 static s32 e1000_reset_hw_ich8lan(struct e1000_hw *hw)
02969 {
02970     struct e1000_dev_spec_ich8lan *dev_spec = &hw->dev_spec.ich8lan;
02971     u16 reg;
02972     u32 ctrl, kab;
02973     s32 ret_val;
02974 
02975     /*
02976      * Prevent the PCI-E bus from sticking if there is no TLP connection
02977      * on the last TLP read/write transaction when MAC is reset.
02978      */
02979     ret_val = e1000e_disable_pcie_master(hw);
02980     if (ret_val)
02981         e_dbg("PCI-E Master disable polling has failed.\n");
02982 
02983     e_dbg("Masking off all interrupts\n");
02984     ew32(IMC, 0xffffffff);
02985 
02986     /*
02987      * Disable the Transmit and Receive units.  Then delay to allow
02988      * any pending transactions to complete before we hit the MAC
02989      * with the global reset.
02990      */
02991     ew32(RCTL, 0);
02992     ew32(TCTL, E1000_TCTL_PSP);
02993     e1e_flush();
02994 
02995     msleep(10);
02996 
02997     /* Workaround for ICH8 bit corruption issue in FIFO memory */
02998     if (hw->mac.type == e1000_ich8lan) {
02999         /* Set Tx and Rx buffer allocation to 8k apiece. */
03000         ew32(PBA, E1000_PBA_8K);
03001         /* Set Packet Buffer Size to 16k. */
03002         ew32(PBS, E1000_PBS_16K);
03003     }
03004 
03005     if (hw->mac.type == e1000_pchlan) {
03006         /* Save the NVM K1 bit setting*/
03007         ret_val = e1000_read_nvm(hw, E1000_NVM_K1_CONFIG, 1, &reg);
03008         if (ret_val)
03009             return ret_val;
03010 
03011         if (reg & E1000_NVM_K1_ENABLE)
03012             dev_spec->nvm_k1_enabled = true;
03013         else
03014             dev_spec->nvm_k1_enabled = false;
03015     }
03016 
03017     ctrl = er32(CTRL);
03018 
03019     if (!e1000_check_reset_block(hw)) {
03020         /*
03021          * Full-chip reset requires MAC and PHY reset at the same
03022          * time to make sure the interface between MAC and the
03023          * external PHY is reset.
03024          */
03025         ctrl |= E1000_CTRL_PHY_RST;
03026 
03027         /*
03028          * Gate automatic PHY configuration by hardware on
03029          * non-managed 82579
03030          */
03031         if ((hw->mac.type == e1000_pch2lan) &&
03032             !(er32(FWSM) & E1000_ICH_FWSM_FW_VALID))
03033             e1000_gate_hw_phy_config_ich8lan(hw, true);
03034     }
03035     ret_val = e1000_acquire_swflag_ich8lan(hw);
03036     e_dbg("Issuing a global reset to ich8lan\n");
03037     ew32(CTRL, (ctrl | E1000_CTRL_RST));
03038     msleep(20);
03039 
03040     if (!ret_val)
03041         e1000_release_swflag_ich8lan(hw);
03042 
03043     if (ctrl & E1000_CTRL_PHY_RST) {
03044         ret_val = hw->phy.ops.get_cfg_done(hw);
03045         if (ret_val)
03046             goto out;
03047 
03048         ret_val = e1000_post_phy_reset_ich8lan(hw);
03049         if (ret_val)
03050             goto out;
03051     }
03052 
03053     /*
03054      * For PCH, this write will make sure that any noise
03055      * will be detected as a CRC error and be dropped rather than show up
03056      * as a bad packet to the DMA engine.
03057      */
03058     if (hw->mac.type == e1000_pchlan)
03059         ew32(CRC_OFFSET, 0x65656565);
03060 
03061     ew32(IMC, 0xffffffff);
03062     er32(ICR);
03063 
03064     kab = er32(KABGTXD);
03065     kab |= E1000_KABGTXD_BGSQLBIAS;
03066     ew32(KABGTXD, kab);
03067 
03068 out:
03069     return ret_val;
03070 }

03084 static s32 e1000_init_hw_ich8lan(struct e1000_hw *hw)
03085 {
03086     struct e1000_mac_info *mac = &hw->mac;
03087     u32 ctrl_ext, txdctl, snoop;
03088     s32 ret_val;
03089     u16 i;
03090 
03091     e1000_initialize_hw_bits_ich8lan(hw);
03092 
03093     /* Initialize identification LED */
03094     ret_val = mac->ops.id_led_init(hw);
03095     if (ret_val)
03096         e_dbg("Error initializing identification LED\n");
03097         /* This is not fatal and we should not stop init due to this */
03098 
03099     /* Setup the receive address. */
03100     e1000e_init_rx_addrs(hw, mac->rar_entry_count);
03101 
03102     /* Zero out the Multicast HASH table */
03103     e_dbg("Zeroing the MTA\n");
03104     for (i = 0; i < mac->mta_reg_count; i++)
03105         E1000_WRITE_REG_ARRAY(hw, E1000_MTA, i, 0);
03106 
03107     /*
03108      * The 82578 Rx buffer will stall if wakeup is enabled in host and
03109      * the ME.  Reading the BM_WUC register will clear the host wakeup bit.
03110      * Reset the phy after disabling host wakeup to reset the Rx buffer.
03111      */
03112     if (hw->phy.type == e1000_phy_82578) {
03113         e1e_rphy(hw, BM_WUC, &i);
03114         ret_val = e1000_phy_hw_reset_ich8lan(hw);
03115         if (ret_val)
03116             return ret_val;
03117     }
03118 
03119     /* Setup link and flow control */
03120     ret_val = mac->ops.setup_link(hw);
03121 
03122     /* Set the transmit descriptor write-back policy for both queues */
03123     txdctl = er32(TXDCTL(0));
03124     txdctl = (txdctl & ~E1000_TXDCTL_WTHRESH) |
03125          E1000_TXDCTL_FULL_TX_DESC_WB;
03126     txdctl = (txdctl & ~E1000_TXDCTL_PTHRESH) |
03127              E1000_TXDCTL_MAX_TX_DESC_PREFETCH;
03128     ew32(TXDCTL(0), txdctl);
03129     txdctl = er32(TXDCTL(1));
03130     txdctl = (txdctl & ~E1000_TXDCTL_WTHRESH) |
03131          E1000_TXDCTL_FULL_TX_DESC_WB;
03132     txdctl = (txdctl & ~E1000_TXDCTL_PTHRESH) |
03133              E1000_TXDCTL_MAX_TX_DESC_PREFETCH;
03134     ew32(TXDCTL(1), txdctl);
03135 
03136     /*
03137      * ICH8 has opposite polarity of no_snoop bits.
03138      * By default, we should use snoop behavior.
03139      */
03140     if (mac->type == e1000_ich8lan)
03141         snoop = PCIE_ICH8_SNOOP_ALL;
03142     else
03143         snoop = (u32) ~(PCIE_NO_SNOOP_ALL);
03144     e1000e_set_pcie_no_snoop(hw, snoop);
03145 
03146     ctrl_ext = er32(CTRL_EXT);
03147     ctrl_ext |= E1000_CTRL_EXT_RO_DIS;
03148     ew32(CTRL_EXT, ctrl_ext);
03149 
03150     /*
03151      * Clear all of the statistics registers (clear on read).  It is
03152      * important that we do this after we have tried to establish link
03153      * because the symbol error count will increment wildly if there
03154      * is no link.
03155      */
03156     e1000_clear_hw_cntrs_ich8lan(hw);
03157 
03158     return ret_val;

03424

03462                  E1000_PHY_CTRL_NOND0A_GBE_DISABLE);
