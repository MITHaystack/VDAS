00044 /*
00045  * Appends src to string dst of size siz (unlike strncat, siz is the
00046  * full size of dst, not space left).  At most siz-1 characters
00047  * will be copied.  Always NUL terminates (unless siz == 0).
00048  * Returns strlen(src); if retval >= siz, truncation occurred.
00049  */
00050 size_t
00051 strlcat(char *dst, const char *src, size_t siz)
00052 {
00053     register char *d = dst;
00054     register const char *s = src;
00055     register size_t n = siz;
00056     size_t dlen;
00057 
00058     /* Find the end of dst and adjust bytes left but don't go past end */
00059     while (*d != '\0' && n-- != 0)
00060         d++;
00061     dlen = d - dst;
00062     n = siz - dlen;
00063 
00064     if (n == 0)
00065         return(dlen + strlen(s));
00066     while (*s != '\0') {
00067         if (n != 1) {
00068             *d++ = *s;
00069             n--;
00070         }
00071         s++;
00072     }
00073     *d = '\0';
00074 
00075     return(dlen + (s - src));   /* count does not include NUL */
00076 }
