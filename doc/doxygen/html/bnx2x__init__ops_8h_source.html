00047 static void bnx2x_init_fill(struct bnx2x *bp, u32 addr, int fill, u32 len)
00048 {
00049     u32 buf_len = (((len*4) > FW_BUF_SIZE) ? FW_BUF_SIZE : (len*4));
00050     u32 buf_len32 = buf_len/4;
00051     u32 i;
00052 
00053     memset(GUNZIP_BUF(bp), (u8)fill, buf_len);

00055     for (i = 0; i < len; i += buf_len32) {
00056         u32 cur_len = min(buf_len32, len - i);

00058         bnx2x_write_big_buf(bp, addr + i*4, cur_len);

00062 static void bnx2x_init_wr_64(struct bnx2x *bp, u32 addr, const u32 *data,
00063                  u32 len64)
00064 {
00065     u32 buf_len32 = FW_BUF_SIZE/4;
00066     u32 len = len64*2;
00067     u64 data64 = 0;
00068     u32 i;
00069 
00070     /* 64 bit value is in a blob: first low DWORD, then high DWORD */
00071     data64 = HILO_U64((*(data + 1)), (*data));
00072 
00073     len64 = min((u32)(FW_BUF_SIZE/8), len64);
00074     for (i = 0; i < len64; i++) {
00075         u64 *pdata = ((u64 *)(GUNZIP_BUF(bp))) + i;
00076 
00077         *pdata = data64;
00078     }
00079 
00080     for (i = 0; i < len; i += buf_len32) {
00081         u32 cur_len = min(buf_len32, len - i);
00082 
00083         bnx2x_write_big_buf(bp, addr + i*4, cur_len);
00084     }

00087 /*********************************************************
00088    There are different blobs for each PRAM section.
00089    In addition, each blob write operation is divided into a few operations
00090    in order to decrease the amount of phys. contiguous buffer needed.
00091    Thus, when we select a blob the address may be with some offset
00092    from the beginning of PRAM section.
00093    The same holds for the INT_TABLE sections.
00094 **********************************************************/
00095 #define IF_IS_INT_TABLE_ADDR(base, addr) \
00096             if (((base) <= (addr)) && ((base) + 0x400 >= (addr)))
00097 
00098 #define IF_IS_PRAM_ADDR(base, addr) \
00099             if (((base) <= (addr)) && ((base) + 0x40000 >= (addr)))
00100 
00101 static const u8 *bnx2x_sel_blob(struct bnx2x *bp, u32 addr, const u8 *data)
00102 {
00103     IF_IS_INT_TABLE_ADDR(TSEM_REG_INT_TABLE, addr)
00104         data = INIT_TSEM_INT_TABLE_DATA(bp);
00105     else
00106         IF_IS_INT_TABLE_ADDR(CSEM_REG_INT_TABLE, addr)
00107             data = INIT_CSEM_INT_TABLE_DATA(bp);
00108     else
00109         IF_IS_INT_TABLE_ADDR(USEM_REG_INT_TABLE, addr)
00110             data = INIT_USEM_INT_TABLE_DATA(bp);
00111     else
00112         IF_IS_INT_TABLE_ADDR(XSEM_REG_INT_TABLE, addr)
00113             data = INIT_XSEM_INT_TABLE_DATA(bp);
00114     else
00115         IF_IS_PRAM_ADDR(TSEM_REG_PRAM, addr)
00116             data = INIT_TSEM_PRAM_DATA(bp);
00117     else
00118         IF_IS_PRAM_ADDR(CSEM_REG_PRAM, addr)
00119             data = INIT_CSEM_PRAM_DATA(bp);
00120     else
00121         IF_IS_PRAM_ADDR(USEM_REG_PRAM, addr)
00122             data = INIT_USEM_PRAM_DATA(bp);
00123     else
00124         IF_IS_PRAM_ADDR(XSEM_REG_PRAM, addr)
00125             data = INIT_XSEM_PRAM_DATA(bp);
00126 
00127     return data;
00128 }

00138 static void bnx2x_init_wr_wb(struct bnx2x *bp, u32 addr, const u32 *data,
00139                  u32 len)
00140 {
00141     const u32 *old_data = data;
00142 
00143     data = (const u32 *)bnx2x_sel_blob(bp, addr, (const u8 *)data);
00144 
00145     if (bp->dmae_ready) {
00146         if (old_data != data)
00147             VIRT_WR_DMAE_LEN(bp, data, addr, len, 1);
00148         else
00149             VIRT_WR_DMAE_LEN(bp, data, addr, len, 0);
00150     } else
00151         bnx2x_init_ind_wr(bp, addr, data, len);
00152 }

00154 static void bnx2x_init_wr_zp(struct bnx2x *bp, u32 addr, u32 len, u32 blob_off)
00155 {
00156     const u8 *data = NULL;
00157     int rc;
00158     u32 i;
00159 
00160     data = bnx2x_sel_blob(bp, addr, data) + blob_off*4;
00161 
00162     rc = bnx2x_gunzip(bp, data, len);
00163     if (rc)
00164         return;
00165 
00166     /* gunzip_outlen is in dwords */
00167     len = GUNZIP_OUTLEN(bp);
00168     for (i = 0; i < len; i++)
00169         ((u32 *)GUNZIP_BUF(bp))[i] =
00170                 cpu_to_le32(((u32 *)GUNZIP_BUF(bp))[i]);
00171 
00172     bnx2x_write_big_buf_wb(bp, addr, len);
00173 }

00175 static void bnx2x_init_block(struct bnx2x *bp, u32 block, u32 stage)
00176 {
00177     u16 op_start =
00178         INIT_OPS_OFFSETS(bp)[BLOCK_OPS_IDX(block, stage, STAGE_START)];
00179     u16 op_end =
00180         INIT_OPS_OFFSETS(bp)[BLOCK_OPS_IDX(block, stage, STAGE_END)];
00181     union init_op *op;
00182     int hw_wr;
00183     u32 i, op_type, addr, len;
00184     const u32 *data, *data_base;
00185 
00186     /* If empty block */
00187     if (op_start == op_end)
00188         return;
00189 
00190     if (CHIP_REV_IS_FPGA(bp))
00191         hw_wr = OP_WR_FPGA;
00192     else if (CHIP_REV_IS_EMUL(bp))
00193         hw_wr = OP_WR_EMUL;
00194     else
00195         hw_wr = OP_WR_ASIC;
00196 
00197     data_base = INIT_DATA(bp);
00198 
00199     for (i = op_start; i < op_end; i++) {
00200 
00201         op = (union init_op *)&(INIT_OPS(bp)[i]);
00202 
00203         op_type = op->str_wr.op;
00204         addr = op->str_wr.offset;
00205         len = op->str_wr.data_len;
00206         data = data_base + op->str_wr.data_off;
00207 
00208         /* HW/EMUL specific */
00209         if ((op_type > OP_WB) && (op_type == hw_wr))
00210             op_type = OP_WR;
00211 
00212         switch (op_type) {
00213         case OP_RD:
00214             REG_RD(bp, addr);
00215             break;
00216         case OP_WR:
00217             REG_WR(bp, addr, op->write.val);
00218             break;
00219         case OP_SW:
00220             bnx2x_init_str_wr(bp, addr, data, len);
00221             break;
00222         case OP_WB:
00223             bnx2x_init_wr_wb(bp, addr, data, len);
00224             break;
00225         case OP_SI:
00226             bnx2x_init_ind_wr(bp, addr, data, len);
00227             break;
00228         case OP_ZR:
00229             bnx2x_init_fill(bp, addr, 0, op->zero.len);
00230             break;
00231         case OP_ZP:
00232             bnx2x_init_wr_zp(bp, addr, len,
00233                      op->str_wr.data_off);
00234             break;
00235         case OP_WR_64:
00236             bnx2x_init_wr_64(bp, addr, data, len);
00237             break;
00238         default:
00239             /* happens whenever an op is of a diff HW */
00240             break;
00241         }
00242     }
00243 }

00409 static void bnx2x_init_pxp_arb(struct bnx2x *bp, int r_order, int w_order)
00410 {
00411     u32 val, i;
00412 
00413     if (r_order > MAX_RD_ORD) {
00414         DP(NETIF_MSG_HW, "read order of %d  order adjusted to %d\n",
00415            r_order, MAX_RD_ORD);
00416         r_order = MAX_RD_ORD;
00417     }
00418     if (w_order > MAX_WR_ORD) {
00419         DP(NETIF_MSG_HW, "write order of %d  order adjusted to %d\n",
00420            w_order, MAX_WR_ORD);
00421         w_order = MAX_WR_ORD;
00422     }
00423     if (CHIP_REV_IS_FPGA(bp)) {
00424         DP(NETIF_MSG_HW, "write order adjusted to 1 for FPGA\n");
