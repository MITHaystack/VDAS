00047 #define RRCP_OPCODE_MASK    0x7F    /* 0x00 = hello, 0x01 = get, 0x02 = set */
00048 #define RRCP_ISREPLY        0x80    /* 0 = request to switch, 0x80 = reply from switch */
00049 
00050 #define RRCP_PROTO_OFFSET       0   /* proto - 1 byte, must be 1 */
00051 #define RRCP_OPCODE_ISREPLY_OFFSET  1   /* opcode and isreply flag - 1 byte */
00052 #define RRCP_AUTHKEY_OFFSET     2   /* authorization key - 2 bytes, 0x2379 by default */
00053 
00054 /* most packets */
00055 #define RRCP_REG_ADDR_OFFSET        4   /* register address - 2 bytes */
00056 #define RRCP_REG_DATA_OFFSET        6   /* register data - 4 bytes */
00057 #define RRCP_COOKIE1_OFFSET     10  /* 4 bytes */
00058 #define RRCP_COOKIE2_OFFSET     14  /* 4 bytes */

00064 #define RRCP_CHIP_ID_OFFSET     12  /* 2 bytes */

00081  * Print RRCP requests

00084 rrcp_print(netdissect_options *ndo,
00085       register const u_char *cp,
00086       u_int length _U_)
00087 {
00088     const u_char *rrcp;
00089     u_int8_t rrcp_proto;
00090     u_int8_t rrcp_opcode;
00091     register const struct ether_header *ep;
00092     char proto_str[16];
00093     char opcode_str[32];
00094 
00095     ep = (const struct ether_header *)cp;
00096     rrcp = cp + ETHER_HDRLEN;
00097 
00098     ND_TCHECK(*(rrcp + RRCP_PROTO_OFFSET));
00099     rrcp_proto = *(rrcp + RRCP_PROTO_OFFSET);
00100     ND_TCHECK(*(rrcp + RRCP_OPCODE_ISREPLY_OFFSET));
00101     rrcp_opcode = (*(rrcp + RRCP_OPCODE_ISREPLY_OFFSET)) & RRCP_OPCODE_MASK;
00102         ND_PRINT((ndo, "%s > %s, %s %s",
00103         etheraddr_string(ESRC(ep)),
00104         etheraddr_string(EDST(ep)),
00105         tok2strbuf(proto_values,"RRCP-0x%02d",rrcp_proto,proto_str,sizeof(proto_str)),
00106         ((*(rrcp + RRCP_OPCODE_ISREPLY_OFFSET)) & RRCP_ISREPLY) ? "reply" : "query"));
00107     if (rrcp_proto==1){
00108             ND_PRINT((ndo, ": %s",
00109              tok2strbuf(opcode_values,"unknown opcode (0x%02d)",rrcp_opcode,opcode_str,sizeof(opcode_str))));
00110     }
00111     if (rrcp_opcode==1 || rrcp_opcode==2){
00112         ND_TCHECK2(*(rrcp + RRCP_REG_ADDR_OFFSET), 6);
00113             ND_PRINT((ndo, " addr=0x%04x, data=0x%08x",
00114              EXTRACT_16BITS(rrcp + RRCP_REG_ADDR_OFFSET),
00115              EXTRACT_32BITS(rrcp + RRCP_REG_DATA_OFFSET)));
00116     }
00117     if (rrcp_proto==1){
00118         ND_TCHECK2(*(rrcp + RRCP_AUTHKEY_OFFSET), 2);
00119             ND_PRINT((ndo, ", auth=0x%04x",
00120           EXTRACT_16BITS(rrcp + RRCP_AUTHKEY_OFFSET)));
00121     }
00122     if (rrcp_proto==1 && rrcp_opcode==0 &&
00123          ((*(rrcp + RRCP_OPCODE_ISREPLY_OFFSET)) & RRCP_ISREPLY)){
00124         ND_TCHECK2(*(rrcp + RRCP_VENDOR_ID_OFFSET), 4);
00125         ND_PRINT((ndo, " downlink_port=%d, uplink_port=%d, uplink_mac=%s, vendor_id=%08x ,chip_id=%04x ",
00126              *(rrcp + RRCP_DOWNLINK_PORT_OFFSET),
00127              *(rrcp + RRCP_UPLINK_PORT_OFFSET),
00128              etheraddr_string(rrcp + RRCP_UPLINK_MAC_OFFSET),
00129              EXTRACT_32BITS(rrcp + RRCP_VENDOR_ID_OFFSET),
00130              EXTRACT_16BITS(rrcp + RRCP_CHIP_ID_OFFSET)));
00131     }else if (rrcp_opcode==1 || rrcp_opcode==2 || rrcp_proto==2){
00132         ND_TCHECK2(*(rrcp + RRCP_COOKIE2_OFFSET), 4);
00133         ND_PRINT((ndo, ", cookie=0x%08x%08x ",
00134             EXTRACT_32BITS(rrcp + RRCP_COOKIE2_OFFSET),
00135             EXTRACT_32BITS(rrcp + RRCP_COOKIE1_OFFSET)));
00136     }
00137         if (!ndo->ndo_vflag)
00138             return;
00139     return;
00140

00143 }
