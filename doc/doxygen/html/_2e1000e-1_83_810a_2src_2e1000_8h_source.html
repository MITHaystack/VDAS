00107  * in the case of WTHRESH, it appears at least the 82571/2 hardware
00108  * writes back 4 descriptors when WTHRESH=5, and 3 descriptors when
00109  * WTHRESH=4, and since we want 64 bytes at a time written back, set
00110  * it to 5
00111  */
00112 #define E1000_TXDCTL_DMA_BURST_ENABLE                          \
00113     (E1000_TXDCTL_GRAN | /* set descriptor granularity */  \
00114      E1000_TXDCTL_COUNT_DESC |                             \

00116      (1 << 8)  | /* hthresh */                             \
