00208 s32 e1000_setup_init_funcs(struct e1000_hw *hw, bool init_device)
00209 {
00210     s32 ret_val;
00211 
00212     /* Can't do much good without knowing the MAC type. */
00213     ret_val = e1000_set_mac_type(hw);
00214     if (ret_val) {
00215         DEBUGOUT("ERROR: MAC type could not be set properly.\n");
00216         goto out;
00217     }
00218 
00219     if (!hw->hw_addr) {
00220         DEBUGOUT("ERROR: Registers not mapped\n");
00221         ret_val = -E1000_ERR_CONFIG;
00222         goto out;
00223     }
00224 
00225     /*
00226      * Init function pointers to generic implementations. We do this first
00227      * allowing a driver module to override it afterward.
00228      */
00229     e1000_init_mac_ops_generic(hw);
00230     e1000_init_phy_ops_generic(hw);
00231     e1000_init_nvm_ops_generic(hw);
00232 
00233     /*
00234      * Set up the init function pointers. These are functions within the
00235      * adapter family file that sets up function pointers for the rest of
00236      * the functions in that family.
00237      */
00238     switch (hw->mac.type) {
00239     case e1000_82542:
00240         e1000_init_function_pointers_82542(hw);
00241         break;
00242     case e1000_82543:
00243     case e1000_82544:
00244         e1000_init_function_pointers_82543(hw);
00245         break;
00246     case e1000_82540:
00247     case e1000_82545:
00248     case e1000_82545_rev_3:
00249     case e1000_82546:
00250     case e1000_82546_rev_3:
00251         e1000_init_function_pointers_82540(hw);
00252         break;
00253     case e1000_82541:
00254     case e1000_82541_rev_2:
00255     case e1000_82547:
00256     case e1000_82547_rev_2:
00257         e1000_init_function_pointers_82541(hw);
00258         break;
00259     default:
00260         DEBUGOUT("Hardware not supported\n");
00261         ret_val = -E1000_ERR_CONFIG;
00262         break;
00263     }
00264 
00265     /*
00266      * Initialize the rest of the function pointers. These require some
00267      * register reads/writes in some cases.
00268      */
00269     if (!(ret_val) && init_device) {
00270         ret_val = e1000_init_mac_params(hw);
00271         if (ret_val)
00272             goto out;
00273 
00274         ret_val = e1000_init_nvm_params(hw);
00275         if (ret_val)
00276             goto out;
00277 
00278         ret_val = e1000_init_phy_params(hw);
00279         if (ret_val)
00280             goto out;
00281     }
00282 
00283 out:
00284     return ret_val;
00285 }
