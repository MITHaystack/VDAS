00047 s32 ixgbe_init_shared_code(struct ixgbe_hw *hw)
00048 {
00049     s32 status;
00050 
00051     /*
00052      * Set the mac type
00053      */
00054     ixgbe_set_mac_type(hw);
00055 
00056     switch (hw->mac.type) {
00057     case ixgbe_mac_82598EB:
00058         status = ixgbe_init_ops_82598(hw);
00059         break;
00060     case ixgbe_mac_82599EB:
00061         status = ixgbe_init_ops_82599(hw);

00065         break;
00066     default:
00067         status = IXGBE_ERR_DEVICE_NOT_SUPPORTED;
00068         break;
00069     }
00070 
00071     return status;

00081 s32 ixgbe_set_mac_type(struct ixgbe_hw *hw)
00082 {
00083     s32 ret_val = 0;
00084 
00085     if (hw->vendor_id == IXGBE_INTEL_VENDOR_ID) {
00086         switch (hw->device_id) {
00087         case IXGBE_DEV_ID_82598:
00088         case IXGBE_DEV_ID_82598_BX:
00089         case IXGBE_DEV_ID_82598AF_SINGLE_PORT:
00090         case IXGBE_DEV_ID_82598AF_DUAL_PORT:
00091         case IXGBE_DEV_ID_82598AT:
00092         case IXGBE_DEV_ID_82598AT2:
00093         case IXGBE_DEV_ID_82598EB_CX4:
00094         case IXGBE_DEV_ID_82598_CX4_DUAL_PORT:
00095         case IXGBE_DEV_ID_82598_DA_DUAL_PORT:
00096         case IXGBE_DEV_ID_82598_SR_DUAL_PORT_EM:
00097         case IXGBE_DEV_ID_82598EB_XF_LR:
00098         case IXGBE_DEV_ID_82598EB_SFP_LOM:
00099             hw->mac.type = ixgbe_mac_82598EB;
00100             break;
00101         case IXGBE_DEV_ID_82599_KX4:
00102         case IXGBE_DEV_ID_82599_KX4_MEZZ:
00103         case IXGBE_DEV_ID_82599_XAUI_LOM:
00104         case IXGBE_DEV_ID_82599_COMBO_BACKPLANE:
00105         case IXGBE_DEV_ID_82599_KR:
00106         case IXGBE_DEV_ID_82599_SFP:
00107         case IXGBE_DEV_ID_82599_BACKPLANE_FCOE:
00108         case IXGBE_DEV_ID_82599_SFP_FCOE:
00109         case IXGBE_DEV_ID_82599_SFP_EM:
00110         case IXGBE_DEV_ID_82599_CX4:
00111         case IXGBE_DEV_ID_82599_T3_LOM:
00112             hw->mac.type = ixgbe_mac_82599EB;
00113             break;
00114         case IXGBE_DEV_ID_X540T:
00115             hw->mac.type = ixgbe_mac_X540;
00116             break;
00117         default:
00118             ret_val = IXGBE_ERR_DEVICE_NOT_SUPPORTED;
00119             break;
00120         }
00121     } else {
00122         ret_val = IXGBE_ERR_DEVICE_NOT_SUPPORTED;
00123     }
00124 
00125     hw_dbg(hw, "ixgbe_set_mac_type found mac: %d, returns: %d\n",
00126               hw->mac.type, ret_val);
00127     return ret_val;
00128 }
