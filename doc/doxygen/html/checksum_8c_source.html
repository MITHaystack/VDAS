00094 static void
00095 init_crc10_table(void)
00096 {   
00097 #define CRC10_POLYNOMIAL 0x633
00098     register int i, j;
00099     register u_int16_t accum;
00100     u_int16_t verify_crc10_table[256];
00101     
00102     for ( i = 0;  i < 256;  i++ )
00103     {
00104         accum = ((unsigned short) i << 2);
00105         for ( j = 0;  j < 8;  j++ )
00106         {
00107             if ((accum <<= 1) & 0x400) accum ^= CRC10_POLYNOMIAL;
00108         }
00109         verify_crc10_table[i] = accum;
00110     }
00111     assert(memcmp(verify_crc10_table,
00112                   crc10_table,
00113                   sizeof(verify_crc10_table)) == 0);
00114 #undef CRC10_POLYNOMIAL
00115 }

00139 /*
00140  * Creates the OSI Fletcher checksum. See 8473-1, Appendix C, section C.3.
00141  * The checksum field of the passed PDU does not need to be reset to zero.
00142  */
00143 u_int16_t
00144 create_osi_cksum (const u_int8_t *pptr, int checksum_offset, int length)
00145 {
00146 
00147     int x;
00148     int y;
00149     u_int32_t mul;
00150     u_int32_t c0;
00151     u_int32_t c1;
00152     u_int16_t checksum;
00153     int index;
00154 
00155     checksum = 0;
00156 
00157     c0 = 0;
00158     c1 = 0;
00159 
00160     for (index = 0; index < length; index++) {
00161         /*
00162          * Ignore the contents of the checksum field.
00163          */
00164         if (index == checksum_offset ||
00165             index == checksum_offset+1) {
00166             c1 += c0;
00167             pptr++;
00168         } else {
00169             c0 = c0 + *(pptr++);
00170             c1 += c0;
00171         } 
00172     }
00173 
00174     c0 = c0 % 255;
00175     c1 = c1 % 255;
00176 
00177     mul = (length - checksum_offset)*(c0);
00178   
00179     x = mul - c0 - c1;
00180     y = c1 - mul - 1;
00181 
00182     if ( y >= 0 ) y++;
00183     if ( x < 0 ) x--;
00184 
00185     x %= 255;
00186     y %= 255;
00187 
00188 
00189     if (x == 0) x = 255;
00190     if (y == 0) y = 255;
00191 
00192     y &= 0x00FF;
00193     checksum = ((x << 8) | y);
00194   
00195     return checksum;
00196 }
