00154 /*
00155  * Given a SAP value, generate the LLC header value for a UI packet
00156  * with that SAP as the source and destination SAP.
00157  */
00158 #define LLC_UI_HDR(sap) ((sap)<<16 | (sap<<8) | 0x03)
00159 
00160 /*
00161  * This is the top level routine of the printer.  'p' points
00162  * to the LLC/SNAP header of the packet, 'h->ts' is the timestamp,
00163  * 'h->len' is the length of the packet off the wire, and 'h->caplen'
00164  * is the number of bytes actually captured.
00165  */
00166 u_int
00167 atm_if_print(const struct pcap_pkthdr *h, const u_char *p)
00168 {
00169     u_int caplen = h->caplen;
00170     u_int length = h->len;
00171     u_int32_t llchdr;
00172     u_int hdrlen = 0;
00173 
00174     if (caplen < 8) {
00175         printf("[|atm]");
00176         return (caplen);
00177     }
00178 
00179         /* Cisco Style NLPID ? */
00180         if (*p == LLC_UI) {
00181             if (eflag)
00182                 printf("CNLPID ");
00183             isoclns_print(p+1, length-1, caplen-1);
00184             return hdrlen;
00185         }
00186 
00187     /*
00188      * Extract the presumed LLC header into a variable, for quick
00189      * testing.
00190      * Then check for a header that's neither a header for a SNAP
00191      * packet nor an RFC 2684 routed NLPID-formatted PDU nor
00192      * an 802.2-but-no-SNAP IP packet.
00193      */
00194     llchdr = EXTRACT_24BITS(p);
00195     if (llchdr != LLC_UI_HDR(LLCSAP_SNAP) &&
00196         llchdr != LLC_UI_HDR(LLCSAP_ISONS) &&
00197         llchdr != LLC_UI_HDR(LLCSAP_IP)) {
00198         /*
00199          * XXX - assume 802.6 MAC header from Fore driver.
00200          *
00201          * Unfortunately, the above list doesn't check for
00202          * all known SAPs, doesn't check for headers where
00203          * the source and destination SAP aren't the same,
00204          * and doesn't check for non-UI frames.  It also
00205          * runs the risk of an 802.6 MAC header that happens
00206          * to begin with one of those values being
00207          * incorrectly treated as an 802.2 header.
00208          *
00209          * So is that Fore driver still around?  And, if so,
00210          * is it still putting 802.6 MAC headers on ATM
00211          * packets?  If so, could it be changed to use a
00212          * new DLT_IEEE802_6 value if we added it?
00213          */
00214         if (eflag)
00215             printf("%08x%08x %08x%08x ",
00216                    EXTRACT_32BITS(p),
00217                    EXTRACT_32BITS(p+4),
00218                    EXTRACT_32BITS(p+8),
00219                    EXTRACT_32BITS(p+12));
00220         p += 20;
00221         length -= 20;
00222         caplen -= 20;
00223         hdrlen += 20;
00224     }
00225     atm_llc_print(p, length, caplen);
00226     return (hdrlen);
00227 }

00253 {
00254     bpf_u_int32 call_ref;
00255 
00256     if (caplen < PROTO_POS) {
00257         printf("[|atm]");

00260     if (p[PROTO_POS] == Q2931) {
00261         /*
00262          * protocol:Q.2931 for User to Network Interface

00264          */
00265         printf("Q.2931");
00266         if (caplen < MSG_TYPE_POS) {
00267             printf(" [|atm]");

00270         printf(":%s ",
00271             tok2str(msgtype2str, "msgtype#%d", p[MSG_TYPE_POS]));
00272 
00273         if (caplen < CALL_REF_POS+3) {
00274             printf("[|atm]");
00275             return;
00276         }
00277         call_ref = EXTRACT_24BITS(&p[CALL_REF_POS]);
00278         printf("CALL_REF:0x%06x", call_ref);
00279     } else {
00280         /* SCCOP with some unknown protocol atop it */
00281         printf("SSCOP, proto %d ", p[PROTO_POS]);

00352 int 
00353 oam_print (const u_char *p, u_int length, u_int hec) {
00354 
00355     u_int32_t cell_header;
00356     u_int16_t vpi, vci, cksum, cksum_shouldbe, idx;
00357     u_int8_t  cell_type, func_type, payload, clp;
00358 
00359     union {
00360         const struct oam_fm_loopback_t *oam_fm_loopback;
00361         const struct oam_fm_ais_rdi_t *oam_fm_ais_rdi;
00362     } oam_ptr;
00363 
00364 
00365     cell_header = EXTRACT_32BITS(p+hec);
00366     cell_type = ((*(p+ATM_HDR_LEN_NOHEC+hec))>>4) & 0x0f;
00367     func_type = (*(p+ATM_HDR_LEN_NOHEC+hec)) & 0x0f;
00368 
00369     vpi = (cell_header>>20)&0xff;
00370     vci = (cell_header>>4)&0xffff;
00371     payload = (cell_header>>1)&0x7;
00372     clp = cell_header&0x1;
00373 
00374     printf("%s, vpi %u, vci %u, payload [ %s ], clp %u, length %u",
00375            tok2str(oam_f_values, "OAM F5", vci),
00376            vpi, vci,
00377            tok2str(atm_pty_values, "Unknown", payload),
00378            clp, length);
00379 
00380     if (!vflag) {
00381         return 1;
00382     }
00383 
00384     printf("\n\tcell-type %s (%u)",
00385            tok2str(oam_celltype_values, "unknown", cell_type),
00386            cell_type);
00387 
00388     if (oam_functype_values[cell_type] == NULL)
00389         printf(", func-type unknown (%u)", func_type);
00390     else
00391         printf(", func-type %s (%u)",
00392                tok2str(oam_functype_values[cell_type],"none",func_type),
00393                func_type);
00394 
00395     p += ATM_HDR_LEN_NOHEC + hec;
00396 
00397     switch (cell_type << 4 | func_type) {
00398     case (OAM_CELLTYPE_FM << 4 | OAM_FM_FUNCTYPE_LOOPBACK):
00399         oam_ptr.oam_fm_loopback = (const struct oam_fm_loopback_t *)(p + OAM_CELLTYPE_FUNCTYPE_LEN);
00400         printf("\n\tLoopback-Indicator %s, Correlation-Tag 0x%08x",
00401                tok2str(oam_fm_loopback_indicator_values,
00402                        "Unknown",
00403                        oam_ptr.oam_fm_loopback->loopback_indicator & OAM_FM_LOOPBACK_INDICATOR_MASK),
00404                EXTRACT_32BITS(&oam_ptr.oam_fm_loopback->correlation_tag));
00405         printf("\n\tLocation-ID ");
00406         for (idx = 0; idx < sizeof(oam_ptr.oam_fm_loopback->loopback_id); idx++) {
00407             if (idx % 2) {
00408                 printf("%04x ", EXTRACT_16BITS(&oam_ptr.oam_fm_loopback->loopback_id[idx]));
00409             }
00410         }
00411         printf("\n\tSource-ID   ");
00412         for (idx = 0; idx < sizeof(oam_ptr.oam_fm_loopback->source_id); idx++) {
00413             if (idx % 2) {
00414                 printf("%04x ", EXTRACT_16BITS(&oam_ptr.oam_fm_loopback->source_id[idx]));
00415             }
00416         }
00417         break;
00418 
00419     case (OAM_CELLTYPE_FM << 4 | OAM_FM_FUNCTYPE_AIS):
00420     case (OAM_CELLTYPE_FM << 4 | OAM_FM_FUNCTYPE_RDI):
00421         oam_ptr.oam_fm_ais_rdi = (const struct oam_fm_ais_rdi_t *)(p + OAM_CELLTYPE_FUNCTYPE_LEN);
00422         printf("\n\tFailure-type 0x%02x", oam_ptr.oam_fm_ais_rdi->failure_type);
00423         printf("\n\tLocation-ID ");
00424         for (idx = 0; idx < sizeof(oam_ptr.oam_fm_ais_rdi->failure_location); idx++) {
00425             if (idx % 2) {
00426                 printf("%04x ", EXTRACT_16BITS(&oam_ptr.oam_fm_ais_rdi->failure_location[idx]));
00427             }
00428         }
00429         break;
00430 
00431     case (OAM_CELLTYPE_FM << 4 | OAM_FM_FUNCTYPE_CONTCHECK):
00432         /* FIXME */
00433         break;
00434 
00435     default:
00436         break;
00437     }
00438 
00439     /* crc10 checksum verification */
00440     cksum = EXTRACT_16BITS(p + OAM_CELLTYPE_FUNCTYPE_LEN + OAM_FUNCTION_SPECIFIC_LEN)
00441         & OAM_CRC10_MASK;
00442     cksum_shouldbe = verify_crc10_cksum(0, p, OAM_PAYLOAD_LEN);
00443     
00444     printf("\n\tcksum 0x%03x (%scorrect)",
00445            cksum,
00446            cksum_shouldbe == 0 ? "" : "in");
