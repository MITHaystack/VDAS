00090 void StatsWriter::run() {
00091   LOG4CXX_INFO(logger, "StatsWriter Running...");
00092 
00093   Timer run_timer;
00094   Timer command_timer;
00095   
00096   try {
00097     // Main processing loop.
00098     while (_running) {
00099 
00100       // State machine.
00101       switch (_state) {
00102       case WRITE_TO_DISK:
00103     if (command_timer.elapsed() > _command_interval) {
00104       command_timer.restart();
00105       continue;
00106     }
00107     handle_write_to_disk();
00108     break;
00109 
00110       case IDLE:
00111     if (command_timer.elapsed() > _command_interval) {
00112       command_timer.restart();
00113       continue;
00114     }
00115     handle_idle();
00116     break;
00117 
00118       case STOP:
00119     handle_stop();
00120     break;
00121 
00122       default:
00123     LOG4CXX_ERROR(logger, "Unknown state.");
00124     break;
00125       }
00126     }
00127     LOG4CXX_DEBUG(logger, "elapsed run time: " << run_timer.elapsed());
00128   } catch(std::exception &ex) {
00129     LOG4CXX_ERROR(logger, "error: " << ex.what());
00130   }
00131 }

00151 void StatsWriter::handle_write_to_disk() {
00152   struct timeval end_time;
00153   struct timeval diff, instant_diff;
00154 
00155   // Cache stats variables.
00156   boost::uint64_t num_packets;
00157   boost::uint64_t num_bytes;
00158   {
00159     boost::mutex::scoped_lock lock(_mutex);
00160     num_packets = _num_packets;
00161     num_bytes = _num_bytes;
00162   }
00163 
00164   // Update time variables.
00165   if (_start_time.tv_sec == 0) {
00166     gettimeofday(&_start_time, NULL);
00167   }
00168   gettimeofday(&end_time, NULL);
00169 
00170   // Elapsed _interval.
00171   timersub(&end_time, &_start_time, &diff);
00172   timersub(&end_time, &_last_time, &instant_diff);
00173 
00174   const double delta_seconds = diff.tv_sec + diff.tv_usec/1000000.0;
00175   const double instant_delta_seconds = instant_diff.tv_sec
00176     + instant_diff.tv_usec/1000000.0;
00177   const double instant_packet_rate = (num_packets - _last_num_packets)
00178     /instant_delta_seconds;
00179   const double instant_byte_rate = _BPS_TO_MBPS*(num_bytes - _last_num_bytes)
00180     /instant_delta_seconds;
00181   const double lifetime_packet_rate = num_packets/delta_seconds;
00182   const double lifetime_byte_rate = _BPS_TO_MBPS*num_bytes/delta_seconds;
00183   const double average_buffer_size = _buffer_size/delta_seconds;
00184   const double instant_drop_rate = _dropped_packets/delta_seconds;
00185 
00186   _average_packet_rate = (1.0-_ALPHA)*_average_packet_rate
00187     + _ALPHA*instant_packet_rate;
00188   _average_byte_rate = (1.0-_ALPHA)*_average_byte_rate
00189     + _ALPHA*instant_byte_rate;
00190 
00191   if (_interval % _PAGE_LENGTH == 0) {
00192     _stats_stream
00193       << HLINE
00194       << std::endl
00195       << "|"

00202       << std::setw(10) << "mbps"   << " "
00203       << std::setw(10) << "mbps"  << " "
00204       << "|\n"
00205       << "|"
00206       << std::setw(10) << "(s)" << " "
00207       << std::setw(10) << "(inst)" << " "
00208       << std::setw(10) << "(lifetime)"   << " "
00209       << std::setw(10) << "(average)"  << " "
00210       << "|"
00211       << std::setw(10) << "(inst)" << " "
00212       << std::setw(10) << "(lifetime)"   << " "
00213       << std::setw(10) << "(average)"  << " "
00214       << "|\n"
00215       << HLINE
00216       << std::endl;
00217   }
00218   ++_interval;
00219 
00220   _stats_stream
00221     << "|"
00222     << std::setw(10) << delta_seconds << " "
00223     << std::setw(10) << instant_packet_rate << " "
00224     << std::setw(10) << lifetime_packet_rate << " "
00225     << std::setw(10) << _average_packet_rate << " "
00226     << "|"
00227     << std::setw(10) << instant_byte_rate << " "
00228     << std::setw(10) << lifetime_byte_rate << " "
00229     << std::setw(10) << _average_byte_rate << " "
00230     << "|\n";
00231   
00232   _csv_stream
00233     <<  delta_seconds << ","
00234     <<  instant_packet_rate << ","
00235     <<  lifetime_packet_rate << ","
00236     <<  _average_packet_rate << ","
00237     <<  instant_byte_rate << ","
00238     <<  lifetime_byte_rate << ","
00239     <<  _average_byte_rate << ","
00240     << instant_drop_rate << ","
00241     << average_buffer_size
00242     << std::endl;
00243   
00244   _dropped_packets = 0;
00245   _buffer_size = 0;
00246 
00247   // Update state.
00248   gettimeofday(&_last_time, NULL);
00249   _last_num_packets = num_packets;
00250   _last_num_bytes = num_bytes;
