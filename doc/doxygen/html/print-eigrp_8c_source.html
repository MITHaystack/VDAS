00221

00223         const struct eigrp_tlv_general_parm_t *eigrp_tlv_general_parm;
00224         const struct eigrp_tlv_sw_version_t *eigrp_tlv_sw_version;

00228         const struct eigrp_tlv_at_int_t *eigrp_tlv_at_int;
00229         const struct eigrp_tlv_at_ext_t *eigrp_tlv_at_ext;
00230     } tlv_ptr;

00232     tptr=pptr;
00233     eigrp_com_header = (const struct eigrp_common_header *)pptr;
00234     TCHECK(*eigrp_com_header);
00235 
00236     /*
00237      * Sanity checking of the header.
00238      */
00239     if (eigrp_com_header->version != EIGRP_VERSION) {
00240     printf("EIGRP version %u packet not supported",eigrp_com_header->version);
00241     return;
00242     }
00243 
00244     /* in non-verbose mode just lets print the basic Message Type*/
00245     if (vflag < 1) {
00246         printf("EIGRP %s, length: %u",
00247                tok2str(eigrp_opcode_values, "unknown (%u)",eigrp_com_header->opcode),
00248                len);
00249         return;
00250     }
00251 
00252     /* ok they seem to want to know everything - lets fully decode it */
00253 
00254     tlen=len-sizeof(struct eigrp_common_header);
00255 
00256     /* FIXME print other header info */
00257     printf("\n\tEIGRP v%u, opcode: %s (%u), chksum: 0x%04x, Flags: [%s]\n\tseq: 0x%08x, ack: 0x%08x, AS: %u, length: %u",
00258            eigrp_com_header->version,
00259            tok2str(eigrp_opcode_values, "unknown, type: %u",eigrp_com_header->opcode),
00260            eigrp_com_header->opcode,
00261            EXTRACT_16BITS(&eigrp_com_header->checksum),
00262            tok2str(eigrp_common_header_flag_values,
00263                    "none",
00264                    EXTRACT_32BITS(&eigrp_com_header->flags)),
00265            EXTRACT_32BITS(&eigrp_com_header->seq),
00266            EXTRACT_32BITS(&eigrp_com_header->ack),
00267            EXTRACT_32BITS(&eigrp_com_header->asn),
00268            tlen);
00269 
00270     tptr+=sizeof(const struct eigrp_common_header);
00271 
00272     while(tlen>0) {
00273         /* did we capture enough for fully decoding the object header ? */
00274         TCHECK2(*tptr, sizeof(struct eigrp_tlv_header));
00275 
00276         eigrp_tlv_header = (const struct eigrp_tlv_header *)tptr;
00277         eigrp_tlv_len=EXTRACT_16BITS(&eigrp_tlv_header->length);
00278         eigrp_tlv_type=EXTRACT_16BITS(&eigrp_tlv_header->type);
00279 
00280 
00281         if (eigrp_tlv_len < sizeof(struct eigrp_tlv_header) ||
00282             eigrp_tlv_len > tlen) {
00283             print_unknown_data(tptr+sizeof(sizeof(struct eigrp_tlv_header)),"\n\t    ",tlen);
00284             return;
00285         }
00286 
00287         printf("\n\t  %s TLV (0x%04x), length: %u",
00288                tok2str(eigrp_tlv_values,
00289                        "Unknown",
00290                        eigrp_tlv_type),
00291                eigrp_tlv_type,
00292                eigrp_tlv_len);
00293 
00294         tlv_tptr=tptr+sizeof(struct eigrp_tlv_header);
00295         tlv_tlen=eigrp_tlv_len-sizeof(struct eigrp_tlv_header);
00296 
00297         /* did we capture enough for fully decoding the object ? */
00298         TCHECK2(*tptr, eigrp_tlv_len);
00299 
00300         switch(eigrp_tlv_type) {
00301 
00302         case EIGRP_TLV_GENERAL_PARM:
00303             tlv_ptr.eigrp_tlv_general_parm = (const struct eigrp_tlv_general_parm_t *)tlv_tptr;
00304 
00305             printf("\n\t    holdtime: %us, k1 %u, k2 %u, k3 %u, k4 %u, k5 %u",
00306                    EXTRACT_16BITS(tlv_ptr.eigrp_tlv_general_parm->holdtime),
00307                    tlv_ptr.eigrp_tlv_general_parm->k1,
00308                    tlv_ptr.eigrp_tlv_general_parm->k2,
00309                    tlv_ptr.eigrp_tlv_general_parm->k3,
00310                    tlv_ptr.eigrp_tlv_general_parm->k4,
00311                    tlv_ptr.eigrp_tlv_general_parm->k5);
00312             break;

00315             tlv_ptr.eigrp_tlv_sw_version = (const struct eigrp_tlv_sw_version_t *)tlv_tptr;

00318                    tlv_ptr.eigrp_tlv_sw_version->ios_major,
00319                    tlv_ptr.eigrp_tlv_sw_version->ios_minor,
00320                    tlv_ptr.eigrp_tlv_sw_version->eigrp_major,
00321                    tlv_ptr.eigrp_tlv_sw_version->eigrp_minor);
00322             break;

00325             tlv_ptr.eigrp_tlv_ip_int = (const struct eigrp_tlv_ip_int_t *)tlv_tptr;

00328             if (bit_length > 32) {
00329                 printf("\n\t    illegal prefix length %u",bit_length);
00330                 break;
00331             }
00332             byte_length = (bit_length + 7) / 8; /* variable length encoding */
00333             memset(prefix, 0, 4);
00334             memcpy(prefix,&tlv_ptr.eigrp_tlv_ip_int->destination,byte_length);
