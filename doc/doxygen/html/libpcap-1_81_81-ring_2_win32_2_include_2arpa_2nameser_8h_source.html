00245 /*
00246  * Structure for query header.  The order of the fields is machine- and
00247  * compiler-dependent, depending on the byte/bit order and the layout
00248  * of bit fields.  We use bit fields only in int variables, as this
00249  * is all ANSI requires.  This requires a somewhat confusing rearrangement.
00250  */
00251 
00252 typedef struct {
00253     unsigned    id :16;     /* query identification number */
00254 #if __BYTE_ORDER == __BIG_ENDIAN
00255             /* fields in third byte */
00256     unsigned    qr: 1;      /* response flag */
00257     unsigned    opcode: 4;  /* purpose of message */
00258     unsigned    aa: 1;      /* authoritive answer */
00259     unsigned    tc: 1;      /* truncated message */
00260     unsigned    rd: 1;      /* recursion desired */
00261             /* fields in fourth byte */
00262     unsigned    ra: 1;      /* recursion available */
00263     unsigned    pr: 1;      /* primary server req'd (!standard) */
00264     unsigned    unused :2;  /* unused bits (MBZ as of 4.9.3a3) */
00265     unsigned    rcode :4;   /* response code */
00266 #endif
00267 #if __BYTE_ORDER == __LITTLE_ENDIAN || __BYTE_ORDER == __PDP_ENDIAN
00268             /* fields in third byte */
00269     unsigned    rd :1;      /* recursion desired */
00270     unsigned    tc :1;      /* truncated message */
00271     unsigned    aa :1;      /* authoritive answer */
00272     unsigned    opcode :4;  /* purpose of message */
00273     unsigned    qr :1;      /* response flag */
00274             /* fields in fourth byte */
00275     unsigned    rcode :4;   /* response code */
00276     unsigned    unused :2;  /* unused bits (MBZ as of 4.9.3a3) */
00277     unsigned    pr :1;      /* primary server req'd (!standard) */
00278     unsigned    ra :1;      /* recursion available */
00279 #endif
00280             /* remaining bytes */
00281     unsigned    qdcount :16;    /* number of question entries */
00282     unsigned    ancount :16;    /* number of answer entries */
00283     unsigned    nscount :16;    /* number of authority entries */
00284     unsigned    arcount :16;    /* number of resource entries */
00285 } HEADER;

00307 /*
00308  * Inline versions of get/put short/long.  Pointer is advanced.
00309  *
00310  * These macros demonstrate the property of C whereby it can be
00311  * portable or it can be elegant but rarely both.
00312  */
00313 #define GETSHORT(s, cp) { \
00314     register u_char *t_cp = (u_char *)(cp); \
00315     (s) = ((u_int16_t)t_cp[0] << 8) \
00316         | ((u_int16_t)t_cp[1]) \
00317         ; \
00318     (cp) += INT16SZ; \
00319 }
00320 
00321 #define GETLONG(l, cp) { \
00322     register u_char *t_cp = (u_char *)(cp); \
00323     (l) = ((u_int32_t)t_cp[0] << 24) \
00324         | ((u_int32_t)t_cp[1] << 16) \
00325         | ((u_int32_t)t_cp[2] << 8) \
00326         | ((u_int32_t)t_cp[3]) \
00327         ; \
00328     (cp) += INT32SZ; \
00329 }
00330 
00331 #define PUTSHORT(s, cp) { \
00332     register u_int16_t t_s = (u_int16_t)(s); \
00333     register u_char *t_cp = (u_char *)(cp); \
00334     *t_cp++ = t_s >> 8; \
00335     *t_cp   = t_s; \
00336     (cp) += INT16SZ; \
00337 }
00338 
00339 #define PUTLONG(l, cp) { \
00340     register u_int32_t t_l = (u_int32_t)(l); \
00341     register u_char *t_cp = (u_char *)(cp); \
00342     *t_cp++ = t_l >> 24; \
00343     *t_cp++ = t_l >> 16; \
