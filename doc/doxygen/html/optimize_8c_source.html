00273  * Find dominator relationships.
00274  * Assumes graph has been leveled.
00275  */
00276 static void
00277 find_dom(root)
00278     struct block *root;
00279 {
00280     int i;
00281     struct block *b;
00282     bpf_u_int32 *x;
00283 
00284     /*
00285      * Initialize sets to contain all nodes.
00286      */
00287     x = all_dom_sets;

00302             SET_INTERSECT(JF(b)->dom, b->dom, nodewords);

00377 /*
00378  * Return the register number that is used by s.  If A and X are both
00379  * used, return AX_ATOM.  If no register is used, return -1.
00380  *
00381  * The implementation should probably change to an array access.
00382  */
00383 static int
00384 atomuse(s)
00385     struct stmt *s;
00386 {
00387     register int c = s->code;
00388 
00389     if (c == NOP)
00390         return -1;
00391 
00392     switch (BPF_CLASS(c)) {
00393 
00394     case BPF_RET:
00395         return (BPF_RVAL(c) == BPF_A) ? A_ATOM :
00396             (BPF_RVAL(c) == BPF_X) ? X_ATOM : -1;
00397 
00398     case BPF_LD:
00399     case BPF_LDX:
00400         return (BPF_MODE(c) == BPF_IND) ? X_ATOM :
00401             (BPF_MODE(c) == BPF_MEM) ? s->k : -1;
00402 
00403     case BPF_ST:
00404         return A_ATOM;
00405 
00406     case BPF_STX:
00407         return X_ATOM;
00408 
00409     case BPF_JMP:
00410     case BPF_ALU:
00411         if (BPF_SRC(c) == BPF_X)
00412             return AX_ATOM;
00413         return A_ATOM;
00414 
00415     case BPF_MISC:
00416         return BPF_MISCOP(c) == BPF_TXA ? X_ATOM : A_ATOM;
00417     }
00418     abort();
00419     /* NOTREACHED */
00420 }

00423  * Return the register number that is defined by 's'.  We assume that
00424  * a single stmt cannot define more than one register.  If no register

00427  * The implementation should probably change to an array access.
00428  */

00430 atomdef(s)

00432 {
00433     if (s->code == NOP)
00434         return -1;

00439     case BPF_ALU:

00441 
00442     case BPF_LDX:
00443         return X_ATOM;

00445     case BPF_ST:
00446     case BPF_STX:
00447         return s->k;

00449     case BPF_MISC:
00450         return BPF_MISCOP(s->code) == BPF_TAX ? X_ATOM : A_ATOM;

00455 /*
00456  * Compute the sets of registers used, defined, and killed by 'b'.
00457  *
00458  * "Used" means that a statement in 'b' uses the register before any
00459  * statement in 'b' defines it, i.e. it uses the value left in
00460  * that register by a predecessor block of this block.
00461  * "Defined" means that a statement in 'b' defines it.
00462  * "Killed" means that a statement in 'b' defines it before any
00463  * statement in 'b' uses it, i.e. it kills the value left in that
00464  * register by a predecessor block of this block.
00465  */
00466 static void
00467 compute_local_ud(b)
00468     struct block *b;
00469 {
00470     struct slist *s;
00471     atomset def = 0, use = 0, kill = 0;
00472     int atom;
00473 
00474     for (s = b->stmts; s; s = s->next) {
00475         if (s->s.code == NOP)
00476             continue;
00477         atom = atomuse(&s->s);
00478         if (atom >= 0) {
00479             if (atom == AX_ATOM) {
00480                 if (!ATOMELEM(def, X_ATOM))
00481                     use |= ATOMMASK(X_ATOM);
00482                 if (!ATOMELEM(def, A_ATOM))
00483                     use |= ATOMMASK(A_ATOM);
00484             }
00485             else if (atom < N_ATOMS) {
00486                 if (!ATOMELEM(def, atom))
00487                     use |= ATOMMASK(atom);
00488             }
00489             else
00490                 abort();
00491         }
00492         atom = atomdef(&s->s);
00493         if (atom >= 0) {
00494             if (!ATOMELEM(use, atom))
00495                 kill |= ATOMMASK(atom);
00496             def |= ATOMMASK(atom);
00497         }
00498     }
00499     if (BPF_CLASS(b->s.code) == BPF_JMP) {
00500         /*
00501          * XXX - what about RET?
00502          */
00503         atom = atomuse(&b->s);
00504         if (atom >= 0) {
00505             if (atom == AX_ATOM) {
00506                 if (!ATOMELEM(def, X_ATOM))
00507                     use |= ATOMMASK(X_ATOM);
00508                 if (!ATOMELEM(def, A_ATOM))
00509                     use |= ATOMMASK(A_ATOM);
00510             }
00511             else if (atom < N_ATOMS) {
00512                 if (!ATOMELEM(def, atom))
00513                     use |= ATOMMASK(atom);
00514             }
00515             else
00516                 abort();
00517         }
00518     }
00519 
00520     b->def = def;
00521     b->kill = kill;
00522     b->in_use = use;
00523 }

00525 /*
00526  * Assume graph is already leveled.
00527  */
00528 static void
00529 find_ud(root)
00530     struct block *root;
00531 {
00532     int i, maxlevel;
00533     struct block *p;
00534 
00535     /*
00536      * root->level is the highest level no found;
00537      * count down from there.
00538      */
00539     maxlevel = root->level;
00540     for (i = maxlevel; i >= 0; --i)
00541         for (p = levels[i]; p; p = p->link) {
00542             compute_local_ud(p);
00543             p->out_use = 0;
00544         }

00546     for (i = 1; i <= maxlevel; ++i) {
00547         for (p = levels[i]; p; p = p->link) {
00548             p->out_use |= JT(p)->in_use | JF(p)->in_use;
00549             p->in_use |= p->out_use &~ p->kill;
00550         }
00551     }

00593 /* Because we really don't have an IR, this stuff is a little messy. */
00594 static int
00595 F(code, v0, v1)
00596     int code;
00597     int v0, v1;
00598 {
00599     u_int hash;
00600     int val;
00601     struct valnode *p;
00602 
00603     hash = (u_int)code ^ (v0 << 4) ^ (v1 << 8);
00604     hash %= MODULUS;
00605 
00606     for (p = hashtbl[hash]; p; p = p->next)
00607         if (p->code == code && p->v0 == v0 && p->v1 == v1)
00608             return p->val;
00609 
00610     val = ++curval;
00611     if (BPF_MODE(code) == BPF_IMM &&
00612         (BPF_CLASS(code) == BPF_LD || BPF_CLASS(code) == BPF_LDX)) {
00613         vmap[val].const_val = v0;
00614         vmap[val].is_const = 1;
00615     }
00616     p = next_vnode++;
00617     p->val = val;
00618     p->code = code;
00619     p->v0 = v0;
00620     p->v1 = v1;
00621     p->next = hashtbl[hash];
00622     hashtbl[hash] = p;
00623 
00624     return val;
00625 }

00716 static void
00717 opt_peep(b)
00718     struct block *b;
00719 {
00720     struct slist *s;
00721     struct slist *next, *last;
00722     int val;
00723 
00724     s = b->stmts;
00725     if (s == 0)
00726         return;
00727 
00728     last = s;
00729     for (/*empty*/; /*empty*/; s = next) {
00730         /*
00731          * Skip over nops.
00732          */
00733         s = this_op(s);
00734         if (s == 0)
00735             break;  /* nothing left in the block */
00736 
00737         /*
00738          * Find the next real instruction after that one
00739          * (skipping nops).
00740          */
00741         next = this_op(s->next);
00742         if (next == 0)
00743             break;  /* no next instruction */
00744         last = next;
00745 
00746         /*
00747          * st  M[k] --> st  M[k]
00748          * ldx M[k]     tax
00749          */
00750         if (s->s.code == BPF_ST &&
00751             next->s.code == (BPF_LDX|BPF_MEM) &&
00752             s->s.k == next->s.k) {
00753             done = 0;
00754             next->s.code = BPF_MISC|BPF_TAX;
00755         }
00756         /*
00757          * ld  #k   --> ldx  #k
00758          * tax          txa
00759          */
00760         if (s->s.code == (BPF_LD|BPF_IMM) &&
00761             next->s.code == (BPF_MISC|BPF_TAX)) {
00762             s->s.code = BPF_LDX|BPF_IMM;
00763             next->s.code = BPF_MISC|BPF_TXA;
00764             done = 0;
00765         }
00766         /*
00767          * This is an ugly special case, but it happens
00768          * when you say tcp[k] or udp[k] where k is a constant.
00769          */
00770         if (s->s.code == (BPF_LD|BPF_IMM)) {
00771             struct slist *add, *tax, *ild;
00772 
00773             /*
00774              * Check that X isn't used on exit from this
00775              * block (which the optimizer might cause).
00776              * We know the code generator won't generate
00777              * any local dependencies.
00778              */
00779             if (ATOMELEM(b->out_use, X_ATOM))
00780                 continue;
00781 
00782             /*
00783              * Check that the instruction following the ldi
00784              * is an addx, or it's an ldxms with an addx
00785              * following it (with 0 or more nops between the
00786              * ldxms and addx).
00787              */
00788             if (next->s.code != (BPF_LDX|BPF_MSH|BPF_B))
00789                 add = next;
00790             else
00791                 add = this_op(next->next);
00792             if (add == 0 || add->s.code != (BPF_ALU|BPF_ADD|BPF_X))
00793                 continue;
00794 
00795             /*
00796              * Check that a tax follows that (with 0 or more
00797              * nops between them).
00798              */
00799             tax = this_op(add->next);
00800             if (tax == 0 || tax->s.code != (BPF_MISC|BPF_TAX))
00801                 continue;
00802 
00803             /*
00804              * Check that an ild follows that (with 0 or more
00805              * nops between them).
00806              */
00807             ild = this_op(tax->next);
00808             if (ild == 0 || BPF_CLASS(ild->s.code) != BPF_LD ||
00809                 BPF_MODE(ild->s.code) != BPF_IND)
00810                 continue;
00811             /*
00812              * We want to turn this sequence:
00813              *
00814              * (004) ldi     #0x2       {s}
00815              * (005) ldxms   [14]       {next}  -- optional
00816              * (006) addx           {add}
00817              * (007) tax            {tax}
00818              * (008) ild     [x+0]      {ild}
00819              *
00820              * into this sequence:
00821              *
00822              * (004) nop
00823              * (005) ldxms   [14]
00824              * (006) nop
00825              * (007) nop
00826              * (008) ild     [x+2]
00827              *
00828              * XXX We need to check that X is not
00829              * subsequently used, because we want to change
00830              * what'll be in it after this sequence.
00831              *
00832              * We know we can eliminate the accumulator
00833              * modifications earlier in the sequence since
00834              * it is defined by the last stmt of this sequence
00835              * (i.e., the last statement of the sequence loads
00836              * a value into the accumulator, so we can eliminate
00837              * earlier operations on the accumulator).
00838              */
00839             ild->s.k += s->s.k;
00840             s->s.code = NOP;
00841             add->s.code = NOP;
00842             tax->s.code = NOP;
00843             done = 0;
00844         }
00845     }
00846     /*
00847      * If the comparison at the end of a block is an equality
00848      * comparison against a constant, and nobody uses the value
00849      * we leave in the A register at the end of a block, and
00850      * the operation preceding the comparison is an arithmetic
00851      * operation, we can sometime optimize it away.
00852      */
00853     if (b->s.code == (BPF_JMP|BPF_JEQ|BPF_K) &&
00854         !ATOMELEM(b->out_use, A_ATOM)) {
00855             /*
00856              * We can optimize away certain subtractions of the
00857              * X register.
00858              */
00859         if (last->s.code == (BPF_ALU|BPF_SUB|BPF_X)) {
00860             val = b->val[X_ATOM];
00861             if (vmap[val].is_const) {
00862                 /*
00863                  * If we have a subtract to do a comparison,
00864                  * and the X register is a known constant,
00865                  * we can merge this value into the
00866                  * comparison:
00867                  *
00868                  * sub x  ->    nop
00869                  * jeq #y   jeq #(x+y)
00870                  */
00871                 b->s.k += vmap[val].const_val;
00872                 last->s.code = NOP;
00873                 done = 0;
00874             } else if (b->s.k == 0) {
00875                 /*
00876                  * If the X register isn't a constant,
00877                  * and the comparison in the test is
00878                  * against 0, we can compare with the
00879                  * X register, instead:
00880                  *
00881                  * sub x  ->    nop
00882                  * jeq #0   jeq x
00883                  */
00884                 last->s.code = NOP;
00885                 b->s.code = BPF_JMP|BPF_JEQ|BPF_X;
00886                 done = 0;
00887             }
00888         }
00889         /*
00890          * Likewise, a constant subtract can be simplified:
00891          *
00892          * sub #x ->    nop
00893          * jeq #y ->    jeq #(x+y)
00894          */
00895         else if (last->s.code == (BPF_ALU|BPF_SUB|BPF_K)) {
00896             last->s.code = NOP;
00897             b->s.k += last->s.k;
00898             done = 0;
00899         }
00900         /*
00901          * And, similarly, a constant AND can be simplified
00902          * if we're testing against 0, i.e.:
00903          *
00904          * and #k   nop
00905          * jeq #0  ->   jset #k
00906          */
00907         else if (last->s.code == (BPF_ALU|BPF_AND|BPF_K) &&
00908             b->s.k == 0) {
00909             b->s.k = last->s.k;
00910             b->s.code = BPF_JMP|BPF_K|BPF_JSET;
00911             last->s.code = NOP;
00912             done = 0;
00913             opt_not(b);
00914         }
00915     }
00916     /*
00917      * jset #0        ->   never
00918      * jset #ffffffff ->   always
00919      */
00920     if (b->s.code == (BPF_JMP|BPF_K|BPF_JSET)) {
00921         if (b->s.k == 0)
00922             JT(b) = JF(b);
00923         if (b->s.k == 0xffffffff)
00924             JF(b) = JT(b);
00925     }
00926     /*
00927      * If we're comparing against the index register, and the index
00928      * register is a known constant, we can just compare against that
00929      * constant.
00930      */
00931     val = b->val[X_ATOM];
00932     if (vmap[val].is_const && BPF_SRC(b->s.code) == BPF_X) {
00933         bpf_int32 v = vmap[val].const_val;
00934         b->s.code &= ~BPF_X;
00935         b->s.k = v;
00936     }
00937     /*
00938      * If the accumulator is a known constant, we can compute the
00939      * comparison result.
00940      */
00941     val = b->val[A_ATOM];
00942     if (vmap[val].is_const && BPF_SRC(b->s.code) == BPF_K) {
00943         bpf_int32 v = vmap[val].const_val;
00944         switch (BPF_OP(b->s.code)) {
00945 
00946         case BPF_JEQ:
00947             v = v == b->s.k;
00948             break;
00949 
00950         case BPF_JGT:
00951             v = (unsigned)v > b->s.k;
00952             break;
00953 
00954         case BPF_JGE:
00955             v = (unsigned)v >= b->s.k;
00956             break;
00957 
00958         case BPF_JSET:
00959             v &= b->s.k;
00960             break;
00961 
00962         default:
00963             abort();
00964         }
00965         if (JF(b) != JT(b))
00966             done = 0;
00967         if (v)
00968             JF(b) = JT(b);
00969         else
00970             JT(b) = JF(b);
00971     }
00972 }

00974 /*
00975  * Compute the symbolic value of expression of 's', and update
00976  * anything it defines in the value table 'val'.  If 'alter' is true,
00977  * do various optimizations.  This code would be cleaner if symbolic
00978  * evaluation and code transformations weren't folded together.
00979  */
00980 static void
00981 opt_stmt(s, val, alter)
00982     struct stmt *s;
00983     int val[];
00984     int alter;
00985 {
00986     int op;
00987     int v;
00988 
00989     switch (s->code) {
00990 
00991     case BPF_LD|BPF_ABS|BPF_W:
00992     case BPF_LD|BPF_ABS|BPF_H:
00993     case BPF_LD|BPF_ABS|BPF_B:
00994         v = F(s->code, s->k, 0L);
00995         vstore(s, &val[A_ATOM], v, alter);
00996         break;
00997 
00998     case BPF_LD|BPF_IND|BPF_W:
00999     case BPF_LD|BPF_IND|BPF_H:
01000     case BPF_LD|BPF_IND|BPF_B:
01001         v = val[X_ATOM];
01002         if (alter && vmap[v].is_const) {
01003             s->code = BPF_LD|BPF_ABS|BPF_SIZE(s->code);
01004             s->k += vmap[v].const_val;
01005             v = F(s->code, s->k, 0L);
01006             done = 0;
01007         }
01008         else
01009             v = F(s->code, s->k, v);
01010         vstore(s, &val[A_ATOM], v, alter);
01011         break;
01012 
01013     case BPF_LD|BPF_LEN:
01014         v = F(s->code, 0L, 0L);
01015         vstore(s, &val[A_ATOM], v, alter);
01016         break;
01017 
01018     case BPF_LD|BPF_IMM:
01019         v = K(s->k);
01020         vstore(s, &val[A_ATOM], v, alter);
01021         break;
01022 
01023     case BPF_LDX|BPF_IMM:
01024         v = K(s->k);
01025         vstore(s, &val[X_ATOM], v, alter);
01026         break;
01027 
01028     case BPF_LDX|BPF_MSH|BPF_B:
01029         v = F(s->code, s->k, 0L);
01030         vstore(s, &val[X_ATOM], v, alter);
01031         break;
01032 
01033     case BPF_ALU|BPF_NEG:
01034         if (alter && vmap[val[A_ATOM]].is_const) {
01035             s->code = BPF_LD|BPF_IMM;
01036             s->k = -vmap[val[A_ATOM]].const_val;
01037             val[A_ATOM] = K(s->k);
01038         }
01039         else
01040             val[A_ATOM] = F(s->code, val[A_ATOM], 0L);
01041         break;
01042 
01043     case BPF_ALU|BPF_ADD|BPF_K:
01044     case BPF_ALU|BPF_SUB|BPF_K:
01045     case BPF_ALU|BPF_MUL|BPF_K:
01046     case BPF_ALU|BPF_DIV|BPF_K:
01047     case BPF_ALU|BPF_AND|BPF_K:
01048     case BPF_ALU|BPF_OR|BPF_K:
01049     case BPF_ALU|BPF_LSH|BPF_K:
01050     case BPF_ALU|BPF_RSH|BPF_K:
01051         op = BPF_OP(s->code);
01052         if (alter) {
01053             if (s->k == 0) {
01054                 /* don't optimize away "sub #0"
01055                  * as it may be needed later to
01056                  * fixup the generated math code */
01057                 if (op == BPF_ADD ||
01058                     op == BPF_LSH || op == BPF_RSH ||
01059                     op == BPF_OR) {
01060                     s->code = NOP;
01061                     break;
01062                 }
01063                 if (op == BPF_MUL || op == BPF_AND) {
01064                     s->code = BPF_LD|BPF_IMM;
01065                     val[A_ATOM] = K(s->k);
01066                     break;
01067                 }
01068             }
01069             if (vmap[val[A_ATOM]].is_const) {
01070                 fold_op(s, val[A_ATOM], K(s->k));
01071                 val[A_ATOM] = K(s->k);
01072                 break;
01073             }
01074         }
01075         val[A_ATOM] = F(s->code, val[A_ATOM], K(s->k));
01076         break;
01077 
01078     case BPF_ALU|BPF_ADD|BPF_X:
01079     case BPF_ALU|BPF_SUB|BPF_X:
01080     case BPF_ALU|BPF_MUL|BPF_X:
01081     case BPF_ALU|BPF_DIV|BPF_X:
01082     case BPF_ALU|BPF_AND|BPF_X:
01083     case BPF_ALU|BPF_OR|BPF_X:
01084     case BPF_ALU|BPF_LSH|BPF_X:
01085     case BPF_ALU|BPF_RSH|BPF_X:
01086         op = BPF_OP(s->code);
01087         if (alter && vmap[val[X_ATOM]].is_const) {
01088             if (vmap[val[A_ATOM]].is_const) {
01089                 fold_op(s, val[A_ATOM], val[X_ATOM]);
01090                 val[A_ATOM] = K(s->k);
01091             }
01092             else {
01093                 s->code = BPF_ALU|BPF_K|op;
01094                 s->k = vmap[val[X_ATOM]].const_val;
01095                 done = 0;
01096                 val[A_ATOM] =
01097                     F(s->code, val[A_ATOM], K(s->k));
01098             }
01099             break;
01100         }
01101         /*
01102          * Check if we're doing something to an accumulator
01103          * that is 0, and simplify.  This may not seem like
01104          * much of a simplification but it could open up further
01105          * optimizations.
01106          * XXX We could also check for mul by 1, etc.
01107          */
01108         if (alter && vmap[val[A_ATOM]].is_const
01109             && vmap[val[A_ATOM]].const_val == 0) {
01110             if (op == BPF_ADD || op == BPF_OR) {
01111                 s->code = BPF_MISC|BPF_TXA;
01112                 vstore(s, &val[A_ATOM], val[X_ATOM], alter);
01113                 break;
01114             }
01115             else if (op == BPF_MUL || op == BPF_DIV ||
01116                  op == BPF_AND || op == BPF_LSH || op == BPF_RSH) {
01117                 s->code = BPF_LD|BPF_IMM;
01118                 s->k = 0;
01119                 vstore(s, &val[A_ATOM], K(s->k), alter);
01120                 break;
01121             }
01122             else if (op == BPF_NEG) {
01123                 s->code = NOP;
01124                 break;
01125             }
01126         }
01127         val[A_ATOM] = F(s->code, val[A_ATOM], val[X_ATOM]);
01128         break;
01129 
01130     case BPF_MISC|BPF_TXA:
01131         vstore(s, &val[A_ATOM], val[X_ATOM], alter);
01132         break;
01133 
01134     case BPF_LD|BPF_MEM:
01135         v = val[s->k];
01136         if (alter && vmap[v].is_const) {
01137             s->code = BPF_LD|BPF_IMM;
01138             s->k = vmap[v].const_val;
01139             done = 0;
01140         }
01141         vstore(s, &val[A_ATOM], v, alter);
01142         break;
01143 
01144     case BPF_MISC|BPF_TAX:
01145         vstore(s, &val[X_ATOM], val[A_ATOM], alter);
01146         break;
01147 
01148     case BPF_LDX|BPF_MEM:
01149         v = val[s->k];
01150         if (alter && vmap[v].is_const) {
01151             s->code = BPF_LDX|BPF_IMM;
01152             s->k = vmap[v].const_val;
01153             done = 0;
01154         }
01155         vstore(s, &val[X_ATOM], v, alter);
01156         break;
01157 
01158     case BPF_ST:
01159         vstore(s, &val[s->k], val[A_ATOM], alter);
01160         break;
01161 
01162     case BPF_STX:
01163         vstore(s, &val[s->k], val[X_ATOM], alter);
01164         break;
01165     }
01166 }

01349         sense = 0;
01350     } else
01351         sense = 1;
01352 
01353     if (child->s.code != code)
01354         return 0;
01355 
01356     aval0 = child->val[A_ATOM];
01357     oval0 = child->oval;
01358     aval1 = ep->pred->val[A_ATOM];
01359     oval1 = ep->pred->oval;
01360 
01361     if (aval0 != aval1)
01362         return 0;
01363 
01364     if (oval0 == oval1)
01365         /*
01366          * The operands of the branch instructions are
01367          * identical, so the result is true if a true
01368          * branch was taken to get here, otherwise false.
01369          */
01370         return sense ? JT(child) : JF(child);
01371 
01372     if (sense && code == (BPF_JMP|BPF_JEQ|BPF_K))
01373         /*
01374          * At this point, we only know the comparison if we
01375          * came down the true branch, and it was an equality
01376          * comparison with a constant.
01377          *
01378          * I.e., if we came down the true branch, and the branch
01379          * was an equality comparison with a constant, we know the
01380          * accumulator contains that constant.  If we came down
01381          * the false branch, or the comparison wasn't with a
01382          * constant, we don't know what was in the accumulator.
01383          *
01384          * We rely on the fact that distinct constants have distinct
01385          * value numbers.
01386          */
01387         return JF(child);
01388 
01389     return 0;
01390 }

01392 static void
01393 opt_j(ep)
01394     struct edge *ep;
01395 {
01396     register int i, k;
01397     register struct block *target;
01398 
01399     if (JT(ep->succ) == 0)
01400         return;
01401 
01402     if (JT(ep->succ) == JF(ep->succ)) {
01403         /*
01404          * Common branch targets can be eliminated, provided
01405          * there is no data dependency.
01406          */
01407         if (!use_conflict(ep->pred, ep->succ->et.succ)) {
01408             done = 0;
01409             ep->succ = JT(ep->succ);
01410         }
01411     }
01412     /*
01413      * For each edge dominator that matches the successor of this
01414      * edge, promote the edge successor to the its grandchild.
01415      *
01416      * XXX We violate the set abstraction here in favor a reasonably
01417      * efficient loop.
01418      */
01419  top:
01420     for (i = 0; i < edgewords; ++i) {
01421         register bpf_u_int32 x = ep->edom[i];
01422 
01423         while (x != 0) {
01424             k = ffs(x) - 1;
01425             x &=~ (1 << k);
01426             k += i * BITS_PER_WORD;
01427 
01428             target = fold_edge(ep->succ, edges[k]);
01429             /*
01430              * Check that there is no data dependency between
01431              * nodes that will be violated if we move the edge.
01432              */
01433             if (target != 0 && !use_conflict(ep->pred, target)) {
01434                 done = 0;
01435                 ep->succ = target;
01436                 if (JT(target) != 0)
01437                     /*
01438                      * Start over unless we hit a leaf.
01439                      */
01440                     goto top;
01441                 return;
01442             }
01443         }
01444     }
01445 }

01635     struct block *root;

01813 /*
01814  * True iff the two stmt lists load the same value from the packet into
01815  * the accumulator.
01816  */
01817 static int
01818 eq_slist(x, y)
01819     struct slist *x, *y;
01820 {
01821     while (1) {
01822         while (x && x->s.code == NOP)
01823             x = x->next;
01824         while (y && y->s.code == NOP)
01825             y = y->next;
01826         if (x == 0)
01827             return y == 0;
01828         if (y == 0)
01829             return x == 0;
01830         if (x->s.code != y->s.code || x->s.k != y->s.k)
01831             return 0;
01832         x = x->next;
01833         y = y->next;
01834     }
01835 }

01851     struct block *root;
01852 {

01859         blocks[i]->link = 0;

01867             if (!isMarked(blocks[j]))
01868                 continue;
01869             if (eq_blk(blocks[i], blocks[j])) {
01870                 blocks[i]->link = blocks[j]->link ?
01871                     blocks[j]->link : blocks[j];
