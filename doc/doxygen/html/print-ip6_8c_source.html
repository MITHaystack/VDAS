00047 /*
00048  * Compute a V6-style checksum by building a pseudoheader.
00049  */
00050 int
00051 nextproto6_cksum(const struct ip6_hdr *ip6, const u_short *data,
00052          u_int len, u_int next_proto)
00053 {
00054         size_t i;
00055         u_int32_t sum = 0;
00056     union ip6_pseudo_hdr phu;
00057 
00058         /* pseudo-header */
00059         memset(&phu, 0, sizeof(phu));
00060         phu.ph.ph_src = ip6->ip6_src;
00061         phu.ph.ph_dst = ip6->ip6_dst;
00062         phu.ph.ph_len = htonl(len);
00063         phu.ph.ph_nxt = next_proto;
00064 
00065         for (i = 0; i < sizeof(phu.pa) / sizeof(phu.pa[0]); i++) {
00066                 sum += phu.pa[i];
00067     }
00068 
00069         return in_cksum(data, len, sum);
00070 }

00073  * print an IP6 datagram.
00074  */
00075 void
00076 ip6_print(register const u_char *bp, register u_int length)
00077 {
00078     register const struct ip6_hdr *ip6;
00079     register int advance;
00080     u_int len;
00081     const u_char *ipend;
00082     register const u_char *cp;
00083     register u_int payload_len;
00084     int nh;
00085     int fragmented = 0;
00086     u_int flow;
00087 
00088     ip6 = (const struct ip6_hdr *)bp;
00089 
00090     TCHECK(*ip6);
00091     if (length < sizeof (struct ip6_hdr)) {
00092         (void)printf("truncated-ip6 %u", length);
00093         return;
00094     }
00095 
00096         if (!eflag)
00097             printf("IP6 ");
00098 
00099     payload_len = EXTRACT_16BITS(&ip6->ip6_plen);
00100     len = payload_len + sizeof(struct ip6_hdr);
00101     if (length < len)
00102         (void)printf("truncated-ip6 - %u bytes missing!",
00103             len - length);
00104 
00105         if (vflag) {
00106             flow = EXTRACT_32BITS(&ip6->ip6_flow);
00107             printf("(");
00108 #if 0
00109             /* rfc1883 */
00110             if (flow & 0x0f000000)
00111         (void)printf("pri 0x%02x, ", (flow & 0x0f000000) >> 24);
00112             if (flow & 0x00ffffff)
00113         (void)printf("flowlabel 0x%06x, ", flow & 0x00ffffff);
00114 #else
00115             /* RFC 2460 */
00116             if (flow & 0x0ff00000)
00117         (void)printf("class 0x%02x, ", (flow & 0x0ff00000) >> 20);
00118             if (flow & 0x000fffff)
00119         (void)printf("flowlabel 0x%05x, ", flow & 0x000fffff);
00120 #endif
00121 
00122             (void)printf("hlim %u, next-header %s (%u) payload length: %u) ",
00123                          ip6->ip6_hlim,
00124                          tok2str(ipproto_values,"unknown",ip6->ip6_nxt),
00125                          ip6->ip6_nxt,
00126                          payload_len);
00127         }
00128 
00129     /*
00130      * Cut off the snapshot length to the end of the IP payload.
00131      */
00132     ipend = bp + len;
00133     if (ipend < snapend)
00134         snapend = ipend;
00135 
00136     cp = (const u_char *)ip6;
00137     advance = sizeof(struct ip6_hdr);
00138     nh = ip6->ip6_nxt;
00139     while (cp < snapend && advance > 0) {
00140         cp += advance;
00141         len -= advance;
00142 
00143         if (cp == (const u_char *)(ip6 + 1) &&
00144             nh != IPPROTO_TCP && nh != IPPROTO_UDP &&
00145             nh != IPPROTO_DCCP && nh != IPPROTO_SCTP) {
00146             (void)printf("%s > %s: ", ip6addr_string(&ip6->ip6_src),
00147                      ip6addr_string(&ip6->ip6_dst));
00148         }
00149 
00150         switch (nh) {
00151         case IPPROTO_HOPOPTS:
00152             advance = hbhopt_print(cp);
00153             nh = *cp;
00154             break;
00155         case IPPROTO_DSTOPTS:
00156             advance = dstopt_print(cp);
00157             nh = *cp;
00158             break;
00159         case IPPROTO_FRAGMENT:
00160             advance = frag6_print(cp, (const u_char *)ip6);
00161             if (snapend <= cp + advance)
00162                 return;
00163             nh = *cp;
00164             fragmented = 1;
00165             break;
00166 
00167         case IPPROTO_MOBILITY_OLD:
00168         case IPPROTO_MOBILITY:
00169             /*
00170              * XXX - we don't use "advance"; the current
00171              * "Mobility Support in IPv6" draft
00172              * (draft-ietf-mobileip-ipv6-24) says that
00173              * the next header field in a mobility header
00174              * should be IPPROTO_NONE, but speaks of
00175              * the possiblity of a future extension in
00176              * which payload can be piggybacked atop a
00177              * mobility header.
00178              */
00179             advance = mobility_print(cp, (const u_char *)ip6);
00180             nh = *cp;
00181             return;
00182         case IPPROTO_ROUTING:
00183             advance = rt6_print(cp, (const u_char *)ip6);
00184             nh = *cp;
00185             break;
00186         case IPPROTO_SCTP:
00187             sctp_print(cp, (const u_char *)ip6, len);
00188             return;
00189         case IPPROTO_DCCP:
00190             dccp_print(cp, (const u_char *)ip6, len);
00191             return;
00192         case IPPROTO_TCP:
00193             tcp_print(cp, len, (const u_char *)ip6, fragmented);
00194             return;
00195         case IPPROTO_UDP:
00196             udp_print(cp, len, (const u_char *)ip6, fragmented);
00197             return;
00198         case IPPROTO_ICMPV6:
00199             icmp6_print(gndo, cp, len, (const u_char *)ip6, fragmented);
00200             return;
00201         case IPPROTO_AH:
00202             advance = ah_print(cp);
00203             nh = *cp;
00204             break;
00205         case IPPROTO_ESP:
00206             {
00207             int enh, padlen;
00208             advance = esp_print(gndo, cp, len, (const u_char *)ip6, &enh, &padlen);
00209             nh = enh & 0xff;
00210             len -= padlen;
00211             break;
00212             }
00213         case IPPROTO_IPCOMP:
00214             {
00215             int enh;
00216             advance = ipcomp_print(cp, &enh);
00217             nh = enh & 0xff;
00218             break;
00219             }
00220 
00221         case IPPROTO_PIM:
00222             pim_print(cp, len, nextproto6_cksum(ip6, (u_short *)cp, len,
00223                                 IPPROTO_PIM));
00224             return;
00225 
00226         case IPPROTO_OSPF:
00227             ospf6_print(cp, len);
00228             return;
00229 
00230         case IPPROTO_IPV6:
00231             ip6_print(cp, len);
00232             return;
00233 
00234         case IPPROTO_IPV4:
00235                 ip_print(gndo, cp, len);
00236             return;
00237 
00238                 case IPPROTO_PGM:
00239                         pgm_print(cp, len, (const u_char *)ip6);
00240                         return;
00241 
00242         case IPPROTO_GRE:
00243             gre_print(cp, len);
00244             return;
00245 
00246         case IPPROTO_RSVP:
00247             rsvp_print(cp, len);
00248             return;
00249 
00250         case IPPROTO_NONE:
00251             (void)printf("no next header");
00252             return;
00253 
00254         default:
00255             (void)printf("ip-proto-%d %d", nh, len);
00256             return;
00257         }
00258     }
00259 
00260     return;
00261 trunc:
00262     (void)printf("[|ip6]");
00263 }
