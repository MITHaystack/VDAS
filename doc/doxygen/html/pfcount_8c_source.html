00063 /*
00064  * The time difference in millisecond
00065  */
00066 double delta_time (struct timeval * now,
00067            struct timeval * before) {
00068   time_t delta_seconds;
00069   time_t delta_microseconds;
00070 
00071   /*
00072    * compute delta in second, 1/10's and 1/1000's second units
00073    */
00074   delta_seconds      = now -> tv_sec  - before -> tv_sec;
00075   delta_microseconds = now -> tv_usec - before -> tv_usec;
00076 
00077   if(delta_microseconds < 0) {
00078     /* manually carry a one from the seconds field */
00079     delta_microseconds += 1000000;  /* 1e6 */
00080     -- delta_seconds;
00081   }
00082   return((double)(delta_seconds * 1000) + (double)delta_microseconds/1000);
00083 }

00254  */
00255 char* _intoa(unsigned int addr, char* buf, u_short bufLen) {
00256   char *cp, *retStr;
00257   u_int byte;
00258   int n;
00259 
00260   cp = &buf[bufLen];
00261   *--cp = '\0';
00262 
00263   n = 4;
00264   do {
00265     byte = addr & 0xff;
00266     *--cp = byte % 10 + '0';
00267     byte /= 10;
00268     if (byte > 0) {
00269       *--cp = byte % 10 + '0';
00270       byte /= 10;
00271       if (byte > 0)
00272     *--cp = byte + '0';
00273     }
00274     *--cp = '.';
00275     addr >>= 8;
00276   } while (--n > 0);
00277 
00278   /* Convert the string to lowercase */
00279   retStr = (char*)(cp+1);
00280 
00281   return(retStr);
00282 }

00422 int32_t gmt2local(time_t t) {
00423   int dt, dir;
00424   struct tm *gmt, *loc;
00425   struct tm sgmt;
00426 
00427   if (t == 0)
00428     t = time(NULL);
00429   gmt = &sgmt;
00430   *gmt = *gmtime(&t);
00431   loc = localtime(&t);
00432   dt = (loc->tm_hour - gmt->tm_hour) * 60 * 60 +
00433     (loc->tm_min - gmt->tm_min) * 60;
00434 
00435   /*
00436    * If the year or julian day is different, we span 00:00 GMT
00437    * and must add or subtract a day. Check the year first to
00438    * avoid problems when the julian day wraps.
00439    */
00440   dir = loc->tm_year - gmt->tm_year;
00441   if (dir == 0)
00442     dir = loc->tm_yday - gmt->tm_yday;
00443   dt += dir * 24 * 60 * 60;
00444 
00445   return (dt);
00446 }
