00031 /*
00032  * interpret a 32 bit dos packed date/time to some parameters
00033  */
00034 static void
00035 interpret_dos_date(u_int32_t date, struct tm *tp)
00036 {
00037     u_int32_t p0, p1, p2, p3;
00038 
00039     p0 = date & 0xFF;
00040     p1 = ((date & 0xFF00) >> 8) & 0xFF;
00041     p2 = ((date & 0xFF0000) >> 16) & 0xFF;
00042     p3 = ((date & 0xFF000000) >> 24) & 0xFF;
00043 
00044     tp->tm_sec = 2 * (p0 & 0x1F);
00045     tp->tm_min = ((p0 >> 5) & 0xFF) + ((p1 & 0x7) << 3);
00046     tp->tm_hour = (p1 >> 3) & 0xFF;
00047     tp->tm_mday = (p2 & 0x1F);
00048     tp->tm_mon = ((p2 >> 5) & 0xFF) + ((p3 & 0x1) << 3) - 1;
00049     tp->tm_year = ((p3 >> 1) & 0xFF) + 80;
00050 }

00339 static const char *
00340 unistr(const u_char *s, u_int32_t *len, int use_unicode)
00341 {
00342     static char buf[MAX_UNISTR_SIZE+1];
00343     size_t l = 0;
00344     u_int32_t strsize;
00345     const u_char *sp;
00346 
00347     if (use_unicode) {
00348     /*
00349      * Skip padding that puts the string on an even boundary.
00350      */
00351     if (((s - startbuf) % 2) != 0) {
00352         TCHECK(s[0]);
00353         s++;
00354     }
00355     }
00356     if (*len == 0) {
00357     /*
00358      * Null-terminated string.
00359      */
00360     strsize = 0;
00361     sp = s;
00362     if (!use_unicode) {
00363         for (;;) {
00364         TCHECK(sp[0]);
00365         *len += 1;
00366         if (sp[0] == 0)
00367             break;
00368         sp++;
00369         }
00370         strsize = *len - 1;
00371     } else {
00372         for (;;) {
00373         TCHECK2(sp[0], 2);
00374         *len += 2;
00375         if (sp[0] == 0 && sp[1] == 0)
00376             break;
00377         sp += 2;
00378         }
00379         strsize = *len - 2;
00380     }
00381     } else {
00382     /*
00383      * Counted string.
00384      */
00385     strsize = *len;
00386     }
00387     if (!use_unicode) {
00388         while (strsize != 0) {
00389             TCHECK(s[0]);
00390         if (l >= MAX_UNISTR_SIZE)
00391         break;
00392         if (isprint(s[0]))
00393         buf[l] = s[0];
00394         else {
00395         if (s[0] == 0)
00396             break;
00397         buf[l] = '.';
00398         }
00399         l++;
00400         s++;
00401         strsize--;
00402     }
00403     } else {
00404     while (strsize != 0) {
00405         TCHECK2(s[0], 2);
00406         if (l >= MAX_UNISTR_SIZE)
00407         break;
00408         if (s[1] == 0 && isprint(s[0])) {
00409         /* It's a printable ASCII character */
00410         buf[l] = s[0];
00411         } else {
00412         /* It's a non-ASCII character or a non-printable ASCII character */
00413         if (s[0] == 0 && s[1] == 0)
00414             break;
00415         buf[l] = '.';
00416         }
00417         l++;
00418         s += 2;
00419         if (strsize == 1)
00420         break;
00421         strsize -= 2;
00422     }
00423     }
00424     buf[l] = 0;
00425     return buf;
00426 
00427 trunc:
00428     return NULL;

00790 const u_char *
00791 smb_fdata(const u_char *buf, const char *fmt, const u_char *maxbuf,
00792     int unicodestr)
00793 {
00794     static int depth = 0;
00795     char s[128];
00796     char *p;
00797 
00798     while (*fmt) {
00799     switch (*fmt) {
00800     case '*':
00801         fmt++;
00802         while (buf < maxbuf) {
00803         const u_char *buf2;
00804         depth++;
00805         buf2 = smb_fdata(buf, fmt, maxbuf, unicodestr);
00806         depth--;
00807         if (buf2 == NULL)
00808             return(NULL);
00809         if (buf2 == buf)
00810             return(buf);
00811         buf = buf2;
00812         }
00813         return(buf);
00814 
00815     case '|':
00816         fmt++;
00817         if (buf >= maxbuf)
00818         return(buf);
00819         break;
00820 
00821     case '%':
00822         fmt++;
00823         buf = maxbuf;
00824         break;
00825 
00826     case '#':
00827         fmt++;
00828         return(buf);
00829         break;
00830 
00831     case '[':
00832         fmt++;
00833         if (buf >= maxbuf)
00834         return(buf);
00835         memset(s, 0, sizeof(s));
00836         p = strchr(fmt, ']');
00837         if ((size_t)(p - fmt + 1) > sizeof(s)) {
00838         /* overrun */
00839         return(buf);
00840         }
00841         strncpy(s, fmt, p - fmt);
00842         s[p - fmt] = '\0';
00843         fmt = p + 1;
00844         buf = smb_fdata1(buf, s, maxbuf, unicodestr);
00845         if (buf == NULL)
00846         return(NULL);
00847         break;
00848 
00849     default:
00850         putchar(*fmt);
00851         fmt++;
00852         fflush(stdout);
00853         break;
00854     }
00855     }
00856     if (!depth && buf < maxbuf) {
00857     size_t len = PTR_DIFF(maxbuf, buf);
00858     printf("Data: (%lu bytes)\n", (unsigned long)len);
00859     print_data(buf, len);
00860     return(buf + len);
00861     }
00862     return(buf);
00863 }
