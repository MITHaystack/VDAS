00260 /*
00261  * Zero-copy BPF buffer routines to check for and acknowledge BPF data in
00262  * shared memory buffers.
00263  *
00264  * pcap_next_zbuf_shm(): Check for a newly available shared memory buffer,
00265  * and set up p->buffer and cc to reflect one if available.  Notice that if
00266  * there was no prior buffer, we select zbuf1 as this will be the first
00267  * buffer filled for a fresh BPF session.
00268  */
00269 static int
00270 pcap_next_zbuf_shm(pcap_t *p, int *cc)
00271 {
00272     struct bpf_zbuf_header *bzh;
00273 
00274     if (p->md.zbuffer == p->md.zbuf2 || p->md.zbuffer == NULL) {
00275         bzh = (struct bpf_zbuf_header *)p->md.zbuf1;
00276         if (bzh->bzh_user_gen !=
00277             atomic_load_acq_int(&bzh->bzh_kernel_gen)) {
00278             p->md.bzh = bzh;
00279             p->md.zbuffer = (u_char *)p->md.zbuf1;
00280             p->buffer = p->md.zbuffer + sizeof(*bzh);
00281             *cc = bzh->bzh_kernel_len;
00282             return (1);
00283         }
00284     } else if (p->md.zbuffer == p->md.zbuf1) {
00285         bzh = (struct bpf_zbuf_header *)p->md.zbuf2;
00286         if (bzh->bzh_user_gen !=
00287             atomic_load_acq_int(&bzh->bzh_kernel_gen)) {
00288             p->md.bzh = bzh;
00289             p->md.zbuffer = (u_char *)p->md.zbuf2;
00290             p->buffer = p->md.zbuffer + sizeof(*bzh);
00291             *cc = bzh->bzh_kernel_len;
00292             return (1);
00293         }
00294     }
00295     *cc = 0;
00296     return (0);

00299 /*
00300  * pcap_next_zbuf() -- Similar to pcap_next_zbuf_shm(), except wait using
00301  * select() for data or a timeout, and possibly force rotation of the buffer
00302  * in the event we time out or are in immediate mode.  Invoke the shared
00303  * memory check before doing system calls in order to avoid doing avoidable
00304  * work.
00305  */
00306 static int
00307 pcap_next_zbuf(pcap_t *p, int *cc)
00308 {
00309     struct bpf_zbuf bz;
00310     struct timeval tv;
00311     struct timespec cur;
00312     fd_set r_set;
00313     int data, r;
00314     int expire, tmout;
00315 
00316 #define TSTOMILLI(ts) (((ts)->tv_sec * 1000) + ((ts)->tv_nsec / 1000000))
00317     /*
00318      * Start out by seeing whether anything is waiting by checking the
00319      * next shared memory buffer for data.
00320      */
00321     data = pcap_next_zbuf_shm(p, cc);
00322     if (data)
00323         return (data);
00324     /*
00325      * If a previous sleep was interrupted due to signal delivery, make
00326      * sure that the timeout gets adjusted accordingly.  This requires
00327      * that we analyze when the timeout should be been expired, and
00328      * subtract the current time from that.  If after this operation,
00329      * our timeout is less then or equal to zero, handle it like a
00330      * regular timeout.
00331      */
00332     tmout = p->md.timeout;
00333     if (tmout)
00334         (void) clock_gettime(CLOCK_MONOTONIC, &cur);
00335     if (p->md.interrupted && p->md.timeout) {
00336         expire = TSTOMILLI(&p->md.firstsel) + p->md.timeout;
00337         tmout = expire - TSTOMILLI(&cur);
00338 #undef TSTOMILLI
00339         if (tmout <= 0) {
00340             p->md.interrupted = 0;
00341             data = pcap_next_zbuf_shm(p, cc);
00342             if (data)
00343                 return (data);
00344             if (ioctl(p->fd, BIOCROTZBUF, &bz) < 0) {
00345                 (void) snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00346                     "BIOCROTZBUF: %s", strerror(errno));
00347                 return (PCAP_ERROR);
00348             }
00349             return (pcap_next_zbuf_shm(p, cc));
00350         }
00351     }
00352     /*
00353      * No data in the buffer, so must use select() to wait for data or
00354      * the next timeout.  Note that we only call select if the handle
00355      * is in blocking mode.
00356      */
00357     if (p->md.timeout >= 0) {
00358         FD_ZERO(&r_set);
00359         FD_SET(p->fd, &r_set);
00360         if (tmout != 0) {
00361             tv.tv_sec = tmout / 1000;
00362             tv.tv_usec = (tmout * 1000) % 1000000;
00363         }
00364         r = select(p->fd + 1, &r_set, NULL, NULL,
00365             p->md.timeout != 0 ? &tv : NULL);
00366         if (r < 0 && errno == EINTR) {
00367             if (!p->md.interrupted && p->md.timeout) {
00368                 p->md.interrupted = 1;
00369                 p->md.firstsel = cur;
00370             }
00371             return (0);
00372         } else if (r < 0) {
00373             (void) snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00374                 "select: %s", strerror(errno));
00375             return (PCAP_ERROR);
00376         }
00377     }
00378     p->md.interrupted = 0;
00379     /*
00380      * Check again for data, which may exist now that we've either been
00381      * woken up as a result of data or timed out.  Try the "there's data"
00382      * case first since it doesn't require a system call.
00383      */
00384     data = pcap_next_zbuf_shm(p, cc);
00385     if (data)
00386         return (data);
00387     /*
00388      * Try forcing a buffer rotation to dislodge timed out or immediate
00389      * data.
00390      */
00391     if (ioctl(p->fd, BIOCROTZBUF, &bz) < 0) {
00392         (void) snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00393             "BIOCROTZBUF: %s", strerror(errno));
00394         return (PCAP_ERROR);
00395     }
00396     return (pcap_next_zbuf_shm(p, cc));
00397 }

00607      * In 10.4, if adapter enN supports monitor mode, there's a
00608      * wltN adapter corresponding to it; you open it, instead of
00609      * enN, to get monitor mode.  You get whatever link-layer
00610      * headers it supplies.
00611      *
00612      * In 10.5, and, we assume, later releases, if adapter enN
00613      * supports monitor mode, it offers, among its selectable
00614      * DLT_ values, values that let you get the 802.11 header;
00615      * selecting one of those values puts the adapter into monitor
00616      * mode (i.e., you can't get 802.11 headers except in monitor
00617      * mode, and you can't get Ethernet headers in monitor mode).
00618      */
00619     if (uname(&osinfo) == -1) {
00620         /*
00621          * Can't get the OS version; just say "no".
00622          */
00623         return (0);
00624     }
00625     /*
00626      * We assume osinfo.sysname is "Darwin", because
00627      * __APPLE__ is defined.  We just check the version.
00628      */
00629     if (osinfo.release[0] < '8' && osinfo.release[1] == '.') {
00630         /*
00631          * 10.3 (Darwin 7.x) or earlier.
00632          * Monitor mode not supported.
00633          */
00634         return (0);
00635     }
00636     if (osinfo.release[0] == '8' && osinfo.release[1] == '.') {
00637         /*
00638          * 10.4 (Darwin 8.x).  s/en/wlt/, and check
00639          * whether the device exists.
00640          */
00641         if (strncmp(p->opt.source, "en", 2) != 0) {
00642             /*
00643              * Not an enN device; no monitor mode.
00644              */
00645             return (0);
00646         }
00647         fd = socket(AF_INET, SOCK_DGRAM, 0);
00648         if (fd == -1) {
00649             (void)snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00650                 "socket: %s", pcap_strerror(errno));
00651             return (PCAP_ERROR);
00652         }
00653         strlcpy(ifr.ifr_name, "wlt", sizeof(ifr.ifr_name));
00654         strlcat(ifr.ifr_name, p->opt.source + 2, sizeof(ifr.ifr_name));
00655         if (ioctl(fd, SIOCGIFFLAGS, (char *)&ifr) < 0) {
00656             /*
00657              * No such device?
00658              */
00659             close(fd);
00660             return (0);
00661         }
00662         close(fd);
00663         return (1);
00664     }
00665 
00666 #ifdef BIOCGDLTLIST
00667     /*
00668      * Everything else is 10.5 or later; for those,
00669      * we just open the enN device, and check whether
00670      * we have any 802.11 devices.
00671      *
00672      * First, open a BPF device.
00673      */
00674     fd = bpf_open(p);
00675     if (fd < 0)
00676         return (fd);
00677 
00678     /*
00679      * Now bind to the device.
00680      */
00681     (void)strncpy(ifr.ifr_name, p->opt.source, sizeof(ifr.ifr_name));
00682     if (ioctl(fd, BIOCSETIF, (caddr_t)&ifr) < 0) {
00683         if (errno == ENETDOWN) {
00684             /*
00685              * Return a "network down" indication, so that
00686              * the application can report that rather than
00687              * saying we had a mysterious failure and
00688              * suggest that they report a problem to the
00689              * libpcap developers.
00690              */
00691             close(fd);
00692             return (PCAP_ERROR_IFACE_NOT_UP);
00693         } else {
00694             snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00695                 "BIOCSETIF: %s: %s",
00696                 p->opt.source, pcap_strerror(errno));
00697             close(fd);
00698             return (PCAP_ERROR);
00699         }
00700     }
00701 
00702     /*
00703      * We know the default link type -- now determine all the DLTs
00704      * this interface supports.  If this fails with EINVAL, it's
00705      * not fatal; we just don't get to use the feature later.
00706      * (We don't care about DLT_DOCSIS, so we pass DLT_NULL
00707      * as the default DLT for this adapter.)
00708      */
00709     if (get_dlt_list(fd, DLT_NULL, &bdl, p->errbuf) == PCAP_ERROR) {
00710         close(fd);
00711         return (PCAP_ERROR);
00712     }
00713     if (find_802_11(&bdl) != -1) {
00714         /*
00715          * We have an 802.11 DLT, so we can set monitor mode.
00716          */
00717         free(bdl.bfl_list);
00718         close(fd);
00719         return (1);
00720     }
00721     free(bdl.bfl_list);
00722 #endif /* BIOCGDLTLIST */
00723     return (0);
00724 #elif defined(HAVE_BSD_IEEE80211)
00725     int ret;
00726 
00727     ret = monitor_mode(p, 0);
00728     if (ret == PCAP_ERROR_RFMON_NOTSUP)
00729         return (0); /* not an error, just a "can't do" */
00730     if (ret == 0)
00731         return (1); /* success */
00732     return (ret);
00733 #else
00734     return (0);
00735 #endif

00768 static int
00769 pcap_read_bpf(pcap_t *p, int cnt, pcap_handler callback, u_char *user)
00770 {
00771     int cc;
00772     int n = 0;
00773     register u_char *bp, *ep;
00774     u_char *datap;
00775 #ifdef PCAP_FDDIPAD
00776     register int pad;
00777 #endif
00778 #ifdef HAVE_ZEROCOPY_BPF
00779     int i;
00780 #endif
00781 
00782  again:
00783     /*
00784      * Has "pcap_breakloop()" been called?
00785      */
00786     if (p->break_loop) {
00787         /*
00788          * Yes - clear the flag that indicates that it
00789          * has, and return PCAP_ERROR_BREAK to indicate
00790          * that we were told to break out of the loop.
00791          */
00792         p->break_loop = 0;
00793         return (PCAP_ERROR_BREAK);
00794     }
00795     cc = p->cc;
00796     if (p->cc == 0) {
00797         /*
00798          * When reading without zero-copy from a file descriptor, we
00799          * use a single buffer and return a length of data in the
00800          * buffer.  With zero-copy, we update the p->buffer pointer
00801          * to point at whatever underlying buffer contains the next
00802          * data and update cc to reflect the data found in the
00803          * buffer.
00804          */
00805 #ifdef HAVE_ZEROCOPY_BPF
00806         if (p->md.zerocopy) {
00807             if (p->buffer != NULL)
00808                 pcap_ack_zbuf(p);
00809             i = pcap_next_zbuf(p, &cc);
00810             if (i == 0)
00811                 goto again;
00812             if (i < 0)
00813                 return (PCAP_ERROR);
00814         } else
00815 #endif
00816         {
00817             cc = read(p->fd, (char *)p->buffer, p->bufsize);
00818         }
00819         if (cc < 0) {
00820             /* Don't choke when we get ptraced */
00821             switch (errno) {
00822 
00823             case EINTR:
00824                 goto again;
00825 
00826 #ifdef _AIX
00827             case EFAULT:
00828                 /*
00829                  * Sigh.  More AIX wonderfulness.
00830                  *
00831                  * For some unknown reason the uiomove()
00832                  * operation in the bpf kernel extension
00833                  * used to copy the buffer into user
00834                  * space sometimes returns EFAULT. I have
00835                  * no idea why this is the case given that
00836                  * a kernel debugger shows the user buffer
00837                  * is correct. This problem appears to
00838                  * be mostly mitigated by the memset of
00839                  * the buffer before it is first used.
00840                  * Very strange.... Shaun Clowes
00841                  *
00842                  * In any case this means that we shouldn't
00843                  * treat EFAULT as a fatal error; as we
00844                  * don't have an API for returning
00845                  * a "some packets were dropped since
00846                  * the last packet you saw" indication,
00847                  * we just ignore EFAULT and keep reading.
00848                  */
00849                 goto again;
00850 #endif
00851 
00852             case EWOULDBLOCK:
00853                 return (0);
00854 
00855             case ENXIO:
00856                 /*
00857                  * The device on which we're capturing
00858                  * went away.
00859                  *
00860                  * XXX - we should really return
00861                  * PCAP_ERROR_IFACE_NOT_UP, but
00862                  * pcap_dispatch() etc. aren't
00863                  * defined to retur that.
00864                  */
00865                 snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00866                     "The interface went down");
00867                 return (PCAP_ERROR);
00868 
00869 #if defined(sun) && !defined(BSD) && !defined(__svr4__) && !defined(__SVR4)
00870             /*
00871              * Due to a SunOS bug, after 2^31 bytes, the kernel
00872              * file offset overflows and read fails with EINVAL.
00873              * The lseek() to 0 will fix things.
00874              */
00875             case EINVAL:
00876                 if (lseek(p->fd, 0L, SEEK_CUR) +
00877                     p->bufsize < 0) {
00878                     (void)lseek(p->fd, 0L, SEEK_SET);
00879                     goto again;
00880                 }
00881                 /* fall through */
00882 #endif
00883             }
00884             snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "read: %s",
00885                 pcap_strerror(errno));
00886             return (PCAP_ERROR);
00887         }
00888         bp = p->buffer;
00889     } else
00890         bp = p->bp;
00891 
00892     /*
00893      * Loop through each packet.
00894      */
00895 #define bhp ((struct bpf_hdr *)bp)
00896     ep = bp + cc;
00897 #ifdef PCAP_FDDIPAD
00898     pad = p->fddipad;
00899 #endif
00900     while (bp < ep) {
00901         register int caplen, hdrlen;
00902 
00903         /*
00904          * Has "pcap_breakloop()" been called?
00905          * If so, return immediately - if we haven't read any
00906          * packets, clear the flag and return PCAP_ERROR_BREAK
00907          * to indicate that we were told to break out of the loop,
00908          * otherwise leave the flag set, so that the *next* call
00909          * will break out of the loop without having read any
00910          * packets, and return the number of packets we've
00911          * processed so far.
00912          */
00913         if (p->break_loop) {
00914             if (n == 0) {
00915                 p->break_loop = 0;
00916                 return (PCAP_ERROR_BREAK);
00917             } else {
00918                 p->bp = bp;
00919                 p->cc = ep - bp;
00920                 return (n);
00921             }
00922         }
00923 
00924         caplen = bhp->bh_caplen;
00925         hdrlen = bhp->bh_hdrlen;
00926         datap = bp + hdrlen;
00927         /*
00928          * Short-circuit evaluation: if using BPF filter
00929          * in kernel, no need to do it now - we already know
00930          * the packet passed the filter.
00931          *
00932 #ifdef PCAP_FDDIPAD
00933          * Note: the filter code was generated assuming
00934          * that p->fddipad was the amount of padding
00935          * before the header, as that's what's required
00936          * in the kernel, so we run the filter before
00937          * skipping that padding.
00938 #endif
00939          */
00940         if (p->md.use_bpf ||
00941             bpf_filter(p->fcode.bf_insns, datap, bhp->bh_datalen, caplen)) {
00942             struct pcap_pkthdr pkthdr;
00943 
00944             pkthdr.ts.tv_sec = bhp->bh_tstamp.tv_sec;
00945 #ifdef _AIX
00946             /*
00947              * AIX's BPF returns seconds/nanoseconds time
00948              * stamps, not seconds/microseconds time stamps.
00949              */
00950             pkthdr.ts.tv_usec = bhp->bh_tstamp.tv_usec/1000;
00951 #else
00952             pkthdr.ts.tv_usec = bhp->bh_tstamp.tv_usec;
00953 #endif
00954 #ifdef PCAP_FDDIPAD
00955             if (caplen > pad)
00956                 pkthdr.caplen = caplen - pad;
00957             else
00958                 pkthdr.caplen = 0;
00959             if (bhp->bh_datalen > pad)
00960                 pkthdr.len = bhp->bh_datalen - pad;
00961             else
00962                 pkthdr.len = 0;
00963             datap += pad;
00964 #else
00965             pkthdr.caplen = caplen;
00966             pkthdr.len = bhp->bh_datalen;
00967 #endif
00968             (*callback)(user, &pkthdr, datap);
00969             bp += BPF_WORDALIGN(caplen + hdrlen);
00970             if (++n >= cnt && cnt > 0) {
00971                 p->bp = bp;
00972                 p->cc = ep - bp;
00973                 return (n);
00974             }
00975         } else {
00976             /*
00977              * Skip this packet.
00978              */
00979             bp += BPF_WORDALIGN(caplen + hdrlen);
00980         }
00981     }
00982 #undef bhp
00983     p->cc = 0;
00984     return (n);
00985 }

02494 pcap_setfilter_bpf(pcap_t *p, struct bpf_program *fp)
02495 {
02496     /*
02497      * Free any user-mode filter we might happen to have installed.
02498      */
02499     pcap_freecode(&p->fcode);
02500 
02501     /*
02502      * Try to install the kernel filter.
02503      */
02504     if (ioctl(p->fd, BIOCSETF, (caddr_t)fp) == 0) {
02505         /*
02506          * It worked.
02507          */
02508         p->md.use_bpf = 1;  /* filtering in the kernel */
02509 
02510         /*
02511          * Discard any previously-received packets, as they might
02512          * have passed whatever filter was formerly in effect, but
02513          * might not pass this filter (BIOCSETF discards packets
02514          * buffered in the kernel, so you can lose packets in any
02515          * case).
02516          */
02517         p->cc = 0;
02518         return (0);
02519     }
02520 
02521     /*
02522      * We failed.
02523      *
02524      * If it failed with EINVAL, that's probably because the program
02525      * is invalid or too big.  Validate it ourselves; if we like it
02526      * (we currently allow backward branches, to support protochain),
02527      * run it in userland.  (There's no notion of "too big" for
02528      * userland.)
02529      *
02530      * Otherwise, just give up.
02531      * XXX - if the copy of the program into the kernel failed,
02532      * we will get EINVAL rather than, say, EFAULT on at least
02533      * some kernels.
02534      */
02535     if (errno != EINVAL) {
02536         snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "BIOCSETF: %s",
02537             pcap_strerror(errno));
02538         return (-1);
02539     }
02540 
02541     /*
02542      * install_bpf_program() validates the program.
02543      *
02544      * XXX - what if we already have a filter in the kernel?
02545      */
02546     if (install_bpf_program(p, fp) < 0)
02547         return (-1);
02548     p->md.use_bpf = 0;  /* filtering in userland */
02549     return (0);
02550 }
