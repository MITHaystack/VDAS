00044 /*
00045  * Internal representation of dev_t, because different NFS servers
00046  * that we might be spying upon use different external representations.
00047  */
00048 typedef struct {
00049     u_int32_t Minor;    /* upper case to avoid clashing with macro names */
00050     u_int32_t Major;
00051 } my_devt;
00052 
00053 #define dev_eq(a,b) ((a.Minor == b.Minor) && (a.Major == b.Major))
00054 
00055 /*
00056  * Many file servers now use a large file system ID.  This is
00057  * our internal representation of that.
00058  */
00059 typedef struct {
00060     my_devt Fsid_dev;       /* XXX avoid name conflict with AIX */
00061     char Opaque_Handle[2 * 32 + 1];
00062     u_int32_t fsid_code;
00063 } my_fsid;
00064 
00065 #define fsid_eq(a,b)    ((a.fsid_code == b.fsid_code) &&\
00066              dev_eq(a.Fsid_dev, b.Fsid_dev))
