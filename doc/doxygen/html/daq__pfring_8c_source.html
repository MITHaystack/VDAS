00240         }
00241             }
00242     }
00243 
00244       else

00246       snprintf(errbuf, len,

00286     return DAQ_ERROR;

00291     if (context->filter_string)

00331 static int pfring_daq_acquire(void *handle, int cnt, DAQ_Analysis_Func_t callback, void *user)
00332 {
00333   Pfring_Context_t *context = (Pfring_Context_t *) handle;
00334   int ret;
00335   pfring *current_ring;
00336 
00337   context->analysis_func = callback;
00338   context->breakloop = 0;
00339 
00340   while ((!context->breakloop) && ((cnt == -1) || (cnt > 0)))
00341     {
00342       struct pfring_pkthdr phdr;
00343       DAQ_PktHdr_t hdr;
00344       DAQ_Verdict verdict;
00345 
00346       ret = pfring_recv((current_ring = context->ring_handle), &context->pkt_buffer, context->snaplen, &phdr, 0 /* Dont't wait */);
00347 
00348       if((ret == -1) && (context->twin_ring_handle != NULL))
00349     ret = pfring_recv((current_ring = context->twin_ring_handle), &context->pkt_buffer, context->snaplen, &phdr, 0 /* Dont't wait */);      
00350 
00351       if(ret == -1) {
00352     /* No packet to read: let's poll */
00353     struct pollfd pfd[2];
00354     int num = 1, rc;
00355 
00356     pfd[0].fd = context->ring_handle->fd, pfd[0].events = POLLIN, pfd[0].revents = 0;
00357 
00358     if(context->twin_ring_handle)
00359       pfd[1].fd = context->twin_ring_handle->fd, pfd[1].events = POLLIN, pfd[1].revents = 0, num = 2;
00360 
00361     rc = poll(pfd, num, context->timeout);
00362 
00363     if(rc < 0) {
00364       if(errno == EINTR)
00365         break;
00366 
00367       DPE(context->errbuf, "%s: Poll failed: %s (%d)", __FUNCTION__, strerror(errno), errno);
00368       return DAQ_ERROR;
00369     }
00370       } else {
00371     hdr.caplen = phdr.caplen;
00372     hdr.pktlen = phdr.len;
00373     hdr.ts = phdr.ts;
00374     hdr.device_index = phdr.extended_hdr.if_index;
00375     hdr.flags = 0;
00376 
00377     context->stats.packets_received++;
00378     verdict = context->analysis_func(user, &hdr, (u_char*)context->pkt_buffer);
00379     if (verdict >= MAX_DAQ_VERDICT)
00380       verdict = DAQ_VERDICT_PASS;
00381 
00382     if(verdict == DAQ_VERDICT_BLACKLIST) {
00383       hash_filtering_rule hash_rule;
00384       int rc;
00385 
00386       /* Block the packet and block all future packets in the same flow systemwide. */
00387 
00388       memset(&hash_rule, 0, sizeof(hash_rule));
00389 
00390       hash_rule.vlan_id     = phdr.extended_hdr.parsed_pkt.vlan_id;
00391       hash_rule.proto       = phdr.extended_hdr.parsed_pkt.l3_proto;
00392       memcpy(&hash_rule.host_peer_a, &phdr.extended_hdr.parsed_pkt.ipv4_src, sizeof(ip_addr));
00393       memcpy(&hash_rule.host_peer_b, &phdr.extended_hdr.parsed_pkt.ipv4_dst, sizeof(ip_addr));
00394       hash_rule.port_peer_a = phdr.extended_hdr.parsed_pkt.l4_src_port;
00395       hash_rule.port_peer_b = phdr.extended_hdr.parsed_pkt.l4_dst_port;
00396       hash_rule.rule_action = dont_forward_packet_and_stop_rule_evaluation;
00397       hash_rule.plugin_action.plugin_id = NO_PLUGIN_ID;
00398 
00399       rc = pfring_handle_hash_filtering_rule(context->ring_handle, &hash_rule, 1 /* add_rule */);
00400 
00401       /* printf("Verdict=%d [pfring_handle_hash_filtering_rule=%d]\n", verdict, rc); */
00402     } else if((verdict == DAQ_VERDICT_PASS) && (context->twin_ring_handle /* DAQ_MODE_INLINE */)) {
00403       /* Userland PF_RING bridge */
00404       if (pfring_send((current_ring == context->ring_handle) ? context->twin_ring_handle : context->ring_handle,
00405               (char*)context->pkt_buffer, hdr.caplen) < 0)
00406         {
00407           DPE(context->errbuf, "%s", "pfring_send() error");
00408           return DAQ_ERROR;
00409         }
00410 
00411       context->stats.packets_injected++;
00412     }
00413 
00414     context->stats.verdicts[verdict]++;
00415     if(cnt > 0) cnt--;
00416       }
00417     }
00418 
00419   return 0;
00420 }
