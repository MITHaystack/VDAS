00062 ripng_print(const u_char *dat, unsigned int length)
00063 {
00064     register const struct rip6 *rp = (struct rip6 *)dat;
00065     register const struct netinfo6 *ni;
00066     register u_int amt;
00067     register u_int i;
00068     int j;
00069     int trunc;
00070 
00071     if (snapend < dat)
00072         return;
00073     amt = snapend - dat;
00074     i = min(length, amt);
00075     if (i < (sizeof(struct rip6) - sizeof(struct netinfo6)))
00076         return;
00077     i -= (sizeof(struct rip6) - sizeof(struct netinfo6));
00078 
00079     switch (rp->rip6_cmd) {
00080 
00081     case RIP6_REQUEST:
00082         j = length / sizeof(*ni);
00083         if (j == 1
00084             &&  rp->rip6_nets->rip6_metric == HOPCNT_INFINITY6
00085             &&  IN6_IS_ADDR_UNSPECIFIED(&rp->rip6_nets->rip6_dest)) {
00086             printf(" ripng-req dump");
00087             break;
00088         }
00089         if (j * sizeof(*ni) != length - 4)
00090             printf(" ripng-req %d[%u]:", j, length);
00091         else
00092             printf(" ripng-req %d:", j);
00093         trunc = ((i / sizeof(*ni)) * sizeof(*ni) != i);
00094         for (ni = rp->rip6_nets; i >= sizeof(*ni);
00095             i -= sizeof(*ni), ++ni) {
00096             if (vflag > 1)
00097                 printf("\n\t");
00098             else
00099                 printf(" ");
00100             rip6_entry_print(ni, 0);
00101         }
00102         break;
00103     case RIP6_RESPONSE:
00104         j = length / sizeof(*ni);
00105         if (j * sizeof(*ni) != length - 4)
00106             printf(" ripng-resp %d[%u]:", j, length);
00107         else
00108             printf(" ripng-resp %d:", j);
00109         trunc = ((i / sizeof(*ni)) * sizeof(*ni) != i);
00110         for (ni = rp->rip6_nets; i >= sizeof(*ni);
00111             i -= sizeof(*ni), ++ni) {
00112             if (vflag > 1)
00113                 printf("\n\t");
00114             else
00115                 printf(" ");
00116             rip6_entry_print(ni, ni->rip6_metric);
00117         }
00118         if (trunc)
00119             printf("[|ripng]");
00120         break;
00121     default:
00122         printf(" ripng-%d ?? %u", rp->rip6_cmd, length);
00123         break;
00124     }
00125     if (rp->rip6_vers != RIP6_VERSION)
