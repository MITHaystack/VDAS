00350 s32 ixgbe_read_pba_string_generic(struct ixgbe_hw *hw, u8 *pba_num,
00351                                   u32 pba_num_size)
00352 {
00353     s32 ret_val;
00354     u16 data;
00355     u16 pba_ptr;
00356     u16 offset;
00357     u16 length;
00358 
00359     if (pba_num == NULL) {
00360         hw_dbg(hw, "PBA string buffer was null\n");
00361         return IXGBE_ERR_INVALID_ARGUMENT;
00362     }
00363 
00364     ret_val = hw->eeprom.ops.read(hw, IXGBE_PBANUM0_PTR, &data);
00365     if (ret_val) {
00366         hw_dbg(hw, "NVM Read Error\n");
00367         return ret_val;
00368     }
00369 
00370     ret_val = hw->eeprom.ops.read(hw, IXGBE_PBANUM1_PTR, &pba_ptr);
00371     if (ret_val) {
00372         hw_dbg(hw, "NVM Read Error\n");
00373         return ret_val;
00374     }
00375 
00376     /*
00377      * if data is not ptr guard the PBA must be in legacy format which
00378      * means pba_ptr is actually our second data word for the PBA number
00379      * and we can decode it into an ascii string
00380      */
00381     if (data != IXGBE_PBANUM_PTR_GUARD) {
00382         hw_dbg(hw, "NVM PBA number is not stored as string\n");
00383 
00384         /* we will need 11 characters to store the PBA */
00385         if (pba_num_size < 11) {
00386             hw_dbg(hw, "PBA string buffer too small\n");
00387             return IXGBE_ERR_NO_SPACE;
00388         }
00389 
00390         /* extract hex string from data and pba_ptr */
00391         pba_num[0] = (data >> 12) & 0xF;
00392         pba_num[1] = (data >> 8) & 0xF;
00393         pba_num[2] = (data >> 4) & 0xF;
00394         pba_num[3] = data & 0xF;
00395         pba_num[4] = (pba_ptr >> 12) & 0xF;
00396         pba_num[5] = (pba_ptr >> 8) & 0xF;
00397         pba_num[6] = '-';
00398         pba_num[7] = 0;
00399         pba_num[8] = (pba_ptr >> 4) & 0xF;
00400         pba_num[9] = pba_ptr & 0xF;
00401 
00402         /* put a null character on the end of our string */
00403         pba_num[10] = '\0';
00404 
00405         /* switch all the data but the '-' to hex char */
00406         for (offset = 0; offset < 10; offset++) {
00407             if (pba_num[offset] < 0xA)
00408                 pba_num[offset] += '0';
00409             else if (pba_num[offset] < 0x10)
00410                 pba_num[offset] += 'A' - 0xA;
00411         }
00412 
00413         return 0;
00414     }
00415 
00416     ret_val = hw->eeprom.ops.read(hw, pba_ptr, &length);
00417     if (ret_val) {
00418         hw_dbg(hw, "NVM Read Error\n");
00419         return ret_val;
00420     }
00421 
00422     if (length == 0xFFFF || length == 0) {
00423         hw_dbg(hw, "NVM PBA number section invalid length\n");
00424         return IXGBE_ERR_PBA_SECTION;
00425     }
00426 
00427     /* check if pba_num buffer is big enough */
00428     if (pba_num_size  < (((u32)length * 2) - 1)) {
00429         hw_dbg(hw, "PBA string buffer too small\n");
00430         return IXGBE_ERR_NO_SPACE;
00431     }
00432 
00433     /* trim pba length from start of string */
00434     pba_ptr++;
00435     length--;
00436 
00437     for (offset = 0; offset < length; offset++) {
00438         ret_val = hw->eeprom.ops.read(hw, pba_ptr + offset, &data);
00439         if (ret_val) {
00440             hw_dbg(hw, "NVM Read Error\n");
00441             return ret_val;
00442         }
00443         pba_num[offset * 2] = (u8)(data >> 8);
00444         pba_num[(offset * 2) + 1] = (u8)(data & 0xFF);
00445     }
00446     pba_num[offset * 2] = '\0';
00447 
00448     return 0;
00449 }

00698 s32 ixgbe_write_eeprom_generic(struct ixgbe_hw *hw, u16 offset, u16 data)
00699 {
00700     s32 status;
00701     u8 write_opcode = IXGBE_EEPROM_WRITE_OPCODE_SPI;
00702 
00703     hw->eeprom.ops.init_params(hw);
00704 
00705     if (offset >= hw->eeprom.word_size) {
00706         status = IXGBE_ERR_EEPROM;
00707         goto out;
00708     }
00709 
00710     /* Prepare the EEPROM for writing  */
00711     status = ixgbe_acquire_eeprom(hw);
00712 
00713     if (status == 0) {
00714         if (ixgbe_ready_eeprom(hw) != 0) {
00715             ixgbe_release_eeprom(hw);
00716             status = IXGBE_ERR_EEPROM;
00717         }
00718     }
00719

00721         ixgbe_standby_eeprom(hw);
00722

00725                                     IXGBE_EEPROM_OPCODE_BITS);
00726

00730          * Some SPI eeproms use the 8th address bit embedded in the

00738                                     IXGBE_EEPROM_OPCODE_BITS);

00740                                     hw->eeprom.address_bits);

00748         ixgbe_release_eeprom(hw);

00928 static s32 ixgbe_acquire_eeprom(struct ixgbe_hw *hw)
00929 {
00930     s32 status = 0;
00931     u32 eec;
00932     u32 i;
00933 
00934     if (hw->mac.ops.acquire_swfw_sync(hw, IXGBE_GSSR_EEP_SM)
00935         != 0)
00936         status = IXGBE_ERR_SWFW_SYNC;
00937 
00938     if (status == 0) {
00939         eec = IXGBE_READ_REG(hw, IXGBE_EEC);
00940 
00941         /* Request EEPROM Access */
00942         eec |= IXGBE_EEC_REQ;
00943         IXGBE_WRITE_REG(hw, IXGBE_EEC, eec);
00944 
00945         for (i = 0; i < IXGBE_EEPROM_GRANT_ATTEMPTS; i++) {
00946             eec = IXGBE_READ_REG(hw, IXGBE_EEC);
00947             if (eec & IXGBE_EEC_GNT)
00948                 break;
00949             udelay(5);
00950         }
00951 
00952         /* Release if grant not acquired */
00953         if (!(eec & IXGBE_EEC_GNT)) {
00954             eec &= ~IXGBE_EEC_REQ;
00955             IXGBE_WRITE_REG(hw, IXGBE_EEC, eec);
00956             hw_dbg(hw, "Could not acquire EEPROM grant\n");
00957 
00958             hw->mac.ops.release_swfw_sync(hw, IXGBE_GSSR_EEP_SM);
00959             status = IXGBE_ERR_EEPROM;
00960         }
00961 
00962         /* Setup EEPROM for Read/Write */
00963         if (status == 0) {
00964             /* Clear CS and SK */
00965             eec &= ~(IXGBE_EEC_CS | IXGBE_EEC_SK);
00966             IXGBE_WRITE_REG(hw, IXGBE_EEC, eec);
00967             IXGBE_WRITE_FLUSH(hw);
00968             udelay(1);
00969         }
00970     }
00971     return status;
00972 }

00980 static s32 ixgbe_get_eeprom_semaphore(struct ixgbe_hw *hw)
00981 {
00982     s32 status = IXGBE_ERR_EEPROM;
00983     u32 timeout = 2000;
00984     u32 i;
00985     u32 swsm;
00986 
00987     /* Get SMBI software semaphore between device drivers first */
00988     for (i = 0; i < timeout; i++) {
00989         /*
00990          * If the SMBI bit is 0 when we read it, then the bit will be
00991          * set and we have the semaphore
00992          */
00993         swsm = IXGBE_READ_REG(hw, IXGBE_SWSM);
00994         if (!(swsm & IXGBE_SWSM_SMBI)) {
00995             status = 0;
00996             break;
00997         }
00998         udelay(50);
00999     }
01000 
01001     if (i == timeout) {
01002         hw_dbg(hw, "Driver can't access the Eeprom - SMBI Semaphore "
01003                  "not granted.\n");
01004         /*
01005          * this release is particularly important because our attempts
01006          * above to get the semaphore may have succeeded, and if there
01007          * was a timeout, we should unconditionally clear the semaphore
01008          * bits to free the driver to make progress
01009          */
01010         ixgbe_release_eeprom_semaphore(hw);
01011 
01012         udelay(50);
01013         /*
01014          * one last try
01015          * If the SMBI bit is 0 when we read it, then the bit will be
01016          * set and we have the semaphore
01017          */
01018         swsm = IXGBE_READ_REG(hw, IXGBE_SWSM);
01019         if (!(swsm & IXGBE_SWSM_SMBI))
01020             status = 0;
01021     }
01022 
01023     /* Now get the semaphore between SW/FW through the SWESMBI bit */
01024     if (status == 0) {
01025         for (i = 0; i < timeout; i++) {
01026             swsm = IXGBE_READ_REG(hw, IXGBE_SWSM);
01027 
01028             /* Set the SW EEPROM semaphore bit to request access */
01029             swsm |= IXGBE_SWSM_SWESMBI;
01030             IXGBE_WRITE_REG(hw, IXGBE_SWSM, swsm);
01031 
01032             /*
01033              * If we set the bit successfully then we got the
01034              * semaphore.
01035              */
01036             swsm = IXGBE_READ_REG(hw, IXGBE_SWSM);
01037             if (swsm & IXGBE_SWSM_SWESMBI)
01038                 break;
01039 
01040             udelay(50);
01041         }
01042 
01043         /*
01044          * Release semaphores and return error if SW EEPROM semaphore
01045          * was not granted because we don't have access to the EEPROM
01046          */
01047         if (i >= timeout) {
01048             hw_dbg(hw, "SWESMBI Software EEPROM semaphore "
01049                      "not granted.\n");
01050             ixgbe_release_eeprom_semaphore(hw);
01051             status = IXGBE_ERR_EEPROM;
01052         }
01053     } else {
01054         hw_dbg(hw, "Software semaphore SMBI between device drivers "
01055                  "not granted.\n");
01056     }
01057 
01058     return status;
01059 }

01083 static s32 ixgbe_ready_eeprom(struct ixgbe_hw *hw)
01084 {
01085     s32 status = 0;
01086     u16 i;
01087     u8 spi_stat_reg;
01088 
01089     /*
01090      * Read "Status Register" repeatedly until the LSB is cleared.  The
01091      * EEPROM will signal that the command has been completed by clearing
01092      * bit 0 of the internal status register.  If it's not cleared within
01093      * 5 milliseconds, then error out.
01094      */
01095     for (i = 0; i < IXGBE_EEPROM_MAX_RETRY_SPI; i += 5) {
01096         ixgbe_shift_out_eeprom_bits(hw, IXGBE_EEPROM_RDSR_OPCODE_SPI,
01097                                     IXGBE_EEPROM_OPCODE_BITS);
01098         spi_stat_reg = (u8)ixgbe_shift_in_eeprom_bits(hw, 8);
01099         if (!(spi_stat_reg & IXGBE_EEPROM_STATUS_RDY_SPI))
01100             break;
01101 
01102         udelay(5);
01103         ixgbe_standby_eeprom(hw);
01104     };
01105 
01106     /*
01107      * On some parts, SPI write time could vary from 0-20mSec on 3.3V
01108      * devices (and only 0-5mSec on 5V devices)
01109      */
01110     if (i >= IXGBE_EEPROM_MAX_RETRY_SPI) {
01111         hw_dbg(hw, "SPI EEPROM Status error\n");
01112         status = IXGBE_ERR_EEPROM;
01113     }
01114 
01115     return status;
01116 }

01145 static void ixgbe_shift_out_eeprom_bits(struct ixgbe_hw *hw, u16 data,
01146                                         u16 count)
01147 {
01148     u32 eec;
01149     u32 mask;
01150     u32 i;
01151 
01152     eec = IXGBE_READ_REG(hw, IXGBE_EEC);
01153 
01154     /*
01155      * Mask is used to shift "count" bits of "data" out to the EEPROM
01156      * one bit at a time.  Determine the starting bit based on count
01157      */
01158     mask = 0x01 << (count - 1);
01159 
01160     for (i = 0; i < count; i++) {
01161         /*
01162          * A "1" is shifted out to the EEPROM by setting bit "DI" to a
01163          * "1", and then raising and then lowering the clock (the SK
01164          * bit controls the clock input to the EEPROM).  A "0" is
01165          * shifted out to the EEPROM by setting "DI" to "0" and then
01166          * raising and then lowering the clock.
01167          */
01168         if (data & mask)
01169             eec |= IXGBE_EEC_DI;
01170         else
01171             eec &= ~IXGBE_EEC_DI;
01172 
01173         IXGBE_WRITE_REG(hw, IXGBE_EEC, eec);
01174         IXGBE_WRITE_FLUSH(hw);
01175 
01176         udelay(1);
01177 
01178         ixgbe_raise_eeprom_clk(hw, &eec);
01179         ixgbe_lower_eeprom_clk(hw, &eec);
01180 
01181         /*
01182          * Shift mask to signify next bit of data to shift in to the
01183          * EEPROM
01184          */
01185         mask = mask >> 1;
01186     };
01187 
01188     /* We leave the "DI" bit set to "0" when we leave this routine. */
01189     eec &= ~IXGBE_EEC_DI;
01190     IXGBE_WRITE_REG(hw, IXGBE_EEC, eec);
01191     IXGBE_WRITE_FLUSH(hw);
01192 }

01198 static u16 ixgbe_shift_in_eeprom_bits(struct ixgbe_hw *hw, u16 count)
01199 {
01200     u32 eec;
01201     u32 i;
01202     u16 data = 0;
01203 
01204     /*
01205      * In order to read a register from the EEPROM, we need to shift
01206      * 'count' bits in from the EEPROM. Bits are "shifted in" by raising
01207      * the clock input to the EEPROM (setting the SK bit), and then reading
01208      * the value of the "DO" bit.  During this "shifting in" process the
01209      * "DI" bit should always be clear.
01210      */
01211     eec = IXGBE_READ_REG(hw, IXGBE_EEC);
01212 
01213     eec &= ~(IXGBE_EEC_DO | IXGBE_EEC_DI);
01214 
01215     for (i = 0; i < count; i++) {
01216         data = data << 1;
01217         ixgbe_raise_eeprom_clk(hw, &eec);
01218 
01219         eec = IXGBE_READ_REG(hw, IXGBE_EEC);
01220 
01221         eec &= ~(IXGBE_EEC_DI);
01222         if (eec & IXGBE_EEC_DO)
01223             data |= 1;
01224 
01225         ixgbe_lower_eeprom_clk(hw, &eec);
01226     }
01227 
01228     return data;
01229 }

01258      */

01269 static void ixgbe_release_eeprom(struct ixgbe_hw *hw)
01270 {
01271     u32 eec;
01272 
01273     eec = IXGBE_READ_REG(hw, IXGBE_EEC);
01274 
01275     eec |= IXGBE_EEC_CS;  /* Pull CS high */
01276     eec &= ~IXGBE_EEC_SK; /* Lower SCK */
01277 
01278     IXGBE_WRITE_REG(hw, IXGBE_EEC, eec);
01279     IXGBE_WRITE_FLUSH(hw);
01280 
01281     udelay(1);
01282 
01283     /* Stop requesting EEPROM access */

01297 u16 ixgbe_calc_eeprom_checksum_generic(struct ixgbe_hw *hw)
01298 {
01299     u16 i;
01300     u16 j;
01301     u16 checksum = 0;
01302     u16 length = 0;
01303     u16 pointer = 0;
01304     u16 word = 0;
01305 
01306     /* Include 0x0-0x3F in the checksum */
01307     for (i = 0; i < IXGBE_EEPROM_CHECKSUM; i++) {
01308         if (hw->eeprom.ops.read(hw, i, &word) != 0) {
01309             hw_dbg(hw, "EEPROM read failed\n");
01310             break;
01311         }
01312         checksum += word;
01313     }
01314 
01315     /* Include all data from pointers except for the fw pointer */
01316     for (i = IXGBE_PCIE_ANALOG_PTR; i < IXGBE_FW_PTR; i++) {
01317         hw->eeprom.ops.read(hw, i, &pointer);
01318 
01319         /* Make sure the pointer seems valid */
01320         if (pointer != 0xFFFF && pointer != 0) {
01321             hw->eeprom.ops.read(hw, pointer, &length);
01322 
01323             if (length != 0xFFFF && length != 0) {
01324                 for (j = pointer+1; j <= pointer+length; j++) {
01325                     hw->eeprom.ops.read(hw, j, &word);
01326                     checksum += word;
01327                 }
01328             }
01329         }
01330     }
01331 
01332     checksum = (u16)IXGBE_EEPROM_SUM - checksum;
01333 
01334     return checksum;
01335 }

01345 s32 ixgbe_validate_eeprom_checksum_generic(struct ixgbe_hw *hw,

01350     u16 read_checksum = 0;
01351 
01352     /*
01353      * Read the first word from the EEPROM. If this times out or fails, do
01354      * not continue or we could be in for a very long wait while every

01450 
01451     /* Make sure we are using a valid rar index range */
01452     if (index >= rar_entries) {

01698 static s32 ixgbe_mta_vector(struct ixgbe_hw *hw, u8 *mc_addr)
01699 {
01700     u32 vector = 0;
01701 
01702     switch (hw->mac.mc_filter_type) {
01703     case 0:   /* use bits [47:36] of the address */
01704         vector = ((mc_addr[4] >> 4) | (((u16)mc_addr[5]) << 4));
01705         break;
01706     case 1:   /* use bits [46:35] of the address */
01707         vector = ((mc_addr[4] >> 3) | (((u16)mc_addr[5]) << 5));
01708         break;
01709     case 2:   /* use bits [45:34] of the address */
01710         vector = ((mc_addr[4] >> 2) | (((u16)mc_addr[5]) << 6));
01711         break;
01712     case 3:   /* use bits [43:32] of the address */
01713         vector = ((mc_addr[4]) | (((u16)mc_addr[5]) << 8));
01714         break;
01715     default:  /* Invalid mc_filter_type */
01716         hw_dbg(hw, "MC filter type param set incorrectly\n");
01717         break;
01718     }
01719 
01720     /* vector can only be 12-bits or boundary will be exceeded */
01721     vector &= 0xFFF;
01722     return vector;
01723 }

01732 void ixgbe_set_mta(struct ixgbe_hw *hw, u8 *mc_addr)
01733 {
01734     u32 vector;
01735     u32 vector_bit;
01736     u32 vector_reg;
01737 
01738     hw->addr_ctrl.mta_in_use++;
01739 
01740     vector = ixgbe_mta_vector(hw, mc_addr);
01741     hw_dbg(hw, " bit-vector = 0x%03X\n", vector);
01742 
01743     /*
01744      * The MTA is a register array of 128 32-bit registers. It is treated
01745      * like an array of 4096 bits.  We want to set bit
01746      * BitArray[vector_value]. So we figure out what register the bit is
01747      * in, read it, OR in the new bit, then write back the new value.  The
01748      * register is determined by the upper 7 bits of the vector value and
01749      * the bit within that register are determined by the lower 5 bits of
01750      * the value.
01751      */
01752     vector_reg = (vector >> 5) & 0x7F;
01753     vector_bit = vector & 0x1F;
01754     hw->mac.mta_shadow[vector_reg] |= (1 << vector_bit);
01755 }

01845 s32 ixgbe_fc_enable_generic(struct ixgbe_hw *hw, s32 packetbuf_num)
01846 {
01847     s32 ret_val = 0;
01848     u32 mflcn_reg, fccfg_reg;
01849     u32 reg;
01850     u32 rx_pba_size;
01851     u32 fcrtl, fcrth;
01852 
01853 #ifdef CONFIG_DCB
01854     if (hw->fc.requested_mode == ixgbe_fc_pfc)
01855         goto out;
01856 
01857 #endif /* CONFIG_DCB */
01858     /* Negotiate the fc mode to use */
01859     ret_val = ixgbe_fc_autoneg(hw);
01860     if (ret_val == IXGBE_ERR_FLOW_CONTROL)
01861         goto out;
01862 
01863     /* Disable any previous flow control settings */
01864     mflcn_reg = IXGBE_READ_REG(hw, IXGBE_MFLCN);
01865     mflcn_reg &= ~(IXGBE_MFLCN_RFCE | IXGBE_MFLCN_RPFCE);
01866 
01867     fccfg_reg = IXGBE_READ_REG(hw, IXGBE_FCCFG);
01868     fccfg_reg &= ~(IXGBE_FCCFG_TFCE_802_3X | IXGBE_FCCFG_TFCE_PRIORITY);
01869 
01870     /*
01871      * The possible values of fc.current_mode are:
01872      * 0: Flow control is completely disabled
01873      * 1: Rx flow control is enabled (we can receive pause frames,
01874      *    but not send pause frames).
01875      * 2: Tx flow control is enabled (we can send pause frames but
01876      *    we do not support receiving pause frames).
01877      * 3: Both Rx and Tx flow control (symmetric) are enabled.
01878 #ifdef CONFIG_DCB
01879      * 4: Priority Flow Control is enabled.
01880 #endif
01881      * other: Invalid.
01882      */
01883     switch (hw->fc.current_mode) {
01884     case ixgbe_fc_none:
01885         /*
01886          * Flow control is disabled by software override or autoneg.
01887          * The code below will actually disable it in the HW.
01888          */
01889         break;
01890     case ixgbe_fc_rx_pause:
01891         /*
01892          * Rx Flow control is enabled and Tx Flow control is
01893          * disabled by software override. Since there really
01894          * isn't a way to advertise that we are capable of RX
01895          * Pause ONLY, we will advertise that we support both
01896          * symmetric and asymmetric Rx PAUSE.  Later, we will
01897          * disable the adapter's ability to send PAUSE frames.
01898          */
01899         mflcn_reg |= IXGBE_MFLCN_RFCE;
01900         break;
01901     case ixgbe_fc_tx_pause:
01902         /*
01903          * Tx Flow control is enabled, and Rx Flow control is
01904          * disabled by software override.
01905          */
01906         fccfg_reg |= IXGBE_FCCFG_TFCE_802_3X;
01907         break;
01908     case ixgbe_fc_full:
01909         /* Flow control (both Rx and Tx) is enabled by SW override. */
01910         mflcn_reg |= IXGBE_MFLCN_RFCE;
01911         fccfg_reg |= IXGBE_FCCFG_TFCE_802_3X;
01912         break;
01913 #ifdef CONFIG_DCB
01914     case ixgbe_fc_pfc:
01915         goto out;
01916         break;
01917 #endif /* CONFIG_DCB */
01918     default:
01919         hw_dbg(hw, "Flow control param set incorrectly\n");
01920         ret_val = IXGBE_ERR_CONFIG;
01921         goto out;
01922         break;
01923     }
01924 
01925     /* Set 802.3x based flow control settings. */
01926     mflcn_reg |= IXGBE_MFLCN_DPF;
01927     IXGBE_WRITE_REG(hw, IXGBE_MFLCN, mflcn_reg);
01928     IXGBE_WRITE_REG(hw, IXGBE_FCCFG, fccfg_reg);
01929 
01930     rx_pba_size = IXGBE_READ_REG(hw, IXGBE_RXPBSIZE(packetbuf_num));
01931     rx_pba_size >>= IXGBE_RXPBSIZE_SHIFT;
01932 
01933     fcrth = (rx_pba_size - hw->fc.high_water) << 10;
01934     fcrtl = (rx_pba_size - hw->fc.low_water) << 10;
01935 
01936     if (hw->fc.current_mode & ixgbe_fc_tx_pause) {
01937         fcrth |= IXGBE_FCRTH_FCEN;
01938         if (hw->fc.send_xon)
01939             fcrtl |= IXGBE_FCRTL_XONE;
01940     }
01941 
01942     IXGBE_WRITE_REG(hw, IXGBE_FCRTH_82599(packetbuf_num), fcrth);
01943     IXGBE_WRITE_REG(hw, IXGBE_FCRTL_82599(packetbuf_num), fcrtl);
01944 
01945     /* Configure pause time (2 TCs per register) */
01946     reg = IXGBE_READ_REG(hw, IXGBE_FCTTV(packetbuf_num / 2));
01947     if ((packetbuf_num & 1) == 0)
01948         reg = (reg & 0xFFFF0000) | hw->fc.pause_time;
01949     else
01950         reg = (reg & 0x0000FFFF) | (hw->fc.pause_time << 16);

01966 s32 ixgbe_fc_autoneg(struct ixgbe_hw *hw)
01967 {
01968     s32 ret_val = IXGBE_ERR_FC_NOT_NEGOTIATED;
01969     ixgbe_link_speed speed;
01970     bool link_up;
01971 
01972     if (hw->fc.disable_fc_autoneg)
01973         goto out;
01974 
01975     /*
01976      * AN should have completed when the cable was plugged in.
01977      * Look for reasons to bail out.  Bail out if:
01978      * - FC autoneg is disabled, or if
01979      * - link is not up.
01980      *
01981      * Since we're being called from an LSC, link is already known to be up.
01982      * So use link_up_wait_to_complete=false.
01983      */
01984     hw->mac.ops.check_link(hw, &speed, &link_up, false);
01985     if (!link_up) {
01986         ret_val = IXGBE_ERR_FLOW_CONTROL;
01987         goto out;
01988     }
01989 
01990     switch (hw->phy.media_type) {
01991     /* Autoneg flow control on fiber adapters */
01992     case ixgbe_media_type_fiber:
01993         if (speed == IXGBE_LINK_SPEED_1GB_FULL)
01994             ret_val = ixgbe_fc_autoneg_fiber(hw);
01995         break;
01996 
01997     /* Autoneg flow control on backplane adapters */
01998     case ixgbe_media_type_backplane:
01999         ret_val = ixgbe_fc_autoneg_backplane(hw);
02000         break;
02001 
02002     /* Autoneg flow control on copper adapters */
02003     case ixgbe_media_type_copper:
02004         if (ixgbe_device_supports_autoneg_fc(hw) == 0)
02005             ret_val = ixgbe_fc_autoneg_copper(hw);
02006         break;
02007 
02008     default:
02009         break;
02010     }
02011 
02012 out:
02013     if (ret_val == 0) {
02014         hw->fc.fc_was_autonegged = true;
02015     } else {
02016         hw->fc.fc_was_autonegged = false;
02017         hw->fc.current_mode = hw->fc.requested_mode;
02018     }
02019     return ret_val;

02144 static s32 ixgbe_negotiate_fc(struct ixgbe_hw *hw, u32 adv_reg, u32 lp_reg,
02145                   u32 adv_sym, u32 adv_asm, u32 lp_sym, u32 lp_asm)
02146 {
02147     if ((!(adv_reg)) ||  (!(lp_reg)))
02148         return IXGBE_ERR_FC_NOT_NEGOTIATED;
02149 
02150     if ((adv_reg & adv_sym) && (lp_reg & lp_sym)) {
02151         /*
02152          * Now we need to check if the user selected Rx ONLY
02153          * of pause frames.  In this case, we had to advertise
02154          * FULL flow control because we could not advertise RX
02155          * ONLY. Hence, we must now check to see if we need to
02156          * turn OFF the TRANSMISSION of PAUSE frames.
02157          */
02158         if (hw->fc.requested_mode == ixgbe_fc_full) {
02159             hw->fc.current_mode = ixgbe_fc_full;
02160             hw_dbg(hw, "Flow Control = FULL.\n");
02161         } else {
02162             hw->fc.current_mode = ixgbe_fc_rx_pause;
02163             hw_dbg(hw, "Flow Control=RX PAUSE frames only\n");
02164         }
02165     } else if (!(adv_reg & adv_sym) && (adv_reg & adv_asm) &&
02166            (lp_reg & lp_sym) && (lp_reg & lp_asm)) {
02167         hw->fc.current_mode = ixgbe_fc_tx_pause;
02168         hw_dbg(hw, "Flow Control = TX PAUSE frames only.\n");
02169     } else if ((adv_reg & adv_sym) && (adv_reg & adv_asm) &&
02170            !(lp_reg & lp_sym) && (lp_reg & lp_asm)) {
02171         hw->fc.current_mode = ixgbe_fc_rx_pause;
02172         hw_dbg(hw, "Flow Control = RX PAUSE frames only.\n");
02173     } else {
02174         hw->fc.current_mode = ixgbe_fc_none;
02175         hw_dbg(hw, "Flow Control = NONE.\n");
02176     }
02177     return 0;
02178 }

02186 static s32 ixgbe_setup_fc(struct ixgbe_hw *hw, s32 packetbuf_num)
02187 {
02188     s32 ret_val = 0;
02189     u32 reg = 0, reg_bp = 0;
02190     u16 reg_cu = 0;
02191 
02192 #ifdef CONFIG_DCB
02193     if (hw->fc.requested_mode == ixgbe_fc_pfc) {
02194         hw->fc.current_mode = hw->fc.requested_mode;
02195         goto out;
02196     }
02197 
02198 #endif /* CONFIG_DCB */
02199     /* Validate the packetbuf configuration */
02200     if (packetbuf_num < 0 || packetbuf_num > 7) {
02201         hw_dbg(hw, "Invalid packet buffer number [%d], expected range "
02202                   "is 0-7\n", packetbuf_num);
02203         ret_val = IXGBE_ERR_INVALID_LINK_SETTINGS;
02204         goto out;
02205     }
02206 
02207     /*
02208      * Validate the water mark configuration.  Zero water marks are invalid
02209      * because it causes the controller to just blast out fc packets.
02210      */
02211     if (!hw->fc.low_water || !hw->fc.high_water || !hw->fc.pause_time) {
02212         hw_dbg(hw, "Invalid water mark configuration\n");
02213         ret_val = IXGBE_ERR_INVALID_LINK_SETTINGS;
02214         goto out;
02215     }
02216 
02217     /*
02218      * Validate the requested mode.  Strict IEEE mode does not allow
02219      * ixgbe_fc_rx_pause because it will cause us to fail at UNH.
02220      */
02221     if (hw->fc.strict_ieee && hw->fc.requested_mode == ixgbe_fc_rx_pause) {
02222         hw_dbg(hw, "ixgbe_fc_rx_pause not valid in strict IEEE mode\n");
02223         ret_val = IXGBE_ERR_INVALID_LINK_SETTINGS;
02224         goto out;
02225     }
02226 
02227     /*
02228      * 10gig parts do not have a word in the EEPROM to determine the
02229      * default flow control setting, so we explicitly set it to full.
02230      */
02231     if (hw->fc.requested_mode == ixgbe_fc_default)
02232         hw->fc.requested_mode = ixgbe_fc_full;
02233 
02234     /*
02235      * Set up the 1G and 10G flow control advertisement registers so the
02236      * HW will be able to do fc autoneg once the cable is plugged in.  If
02237      * we link at 10G, the 1G advertisement is harmless and vice versa.
02238      */
02239 
02240     switch (hw->phy.media_type) {
02241     case ixgbe_media_type_fiber:
02242     case ixgbe_media_type_backplane:
02243         reg = IXGBE_READ_REG(hw, IXGBE_PCS1GANA);
02244         reg_bp = IXGBE_READ_REG(hw, IXGBE_AUTOC);
02245         break;
02246 
02247     case ixgbe_media_type_copper:
02248         hw->phy.ops.read_reg(hw, IXGBE_MDIO_AUTO_NEG_ADVT,
02249                     IXGBE_MDIO_AUTO_NEG_DEV_TYPE, &reg_cu);
02250         break;
02251 
02252     default:
02253         ;
02254     }
02255 
02256     /*
02257      * The possible values of fc.requested_mode are:
02258      * 0: Flow control is completely disabled
02259      * 1: Rx flow control is enabled (we can receive pause frames,
02260      *    but not send pause frames).
02261      * 2: Tx flow control is enabled (we can send pause frames but
02262      *    we do not support receiving pause frames).
02263      * 3: Both Rx and Tx flow control (symmetric) are enabled.
02264 #ifdef CONFIG_DCB
02265      * 4: Priority Flow Control is enabled.
02266 #endif
02267      * other: Invalid.
02268      */
02269     switch (hw->fc.requested_mode) {
02270     case ixgbe_fc_none:
02271         /* Flow control completely disabled by software override. */
02272         reg &= ~(IXGBE_PCS1GANA_SYM_PAUSE | IXGBE_PCS1GANA_ASM_PAUSE);
02273         if (hw->phy.media_type == ixgbe_media_type_backplane)
02274             reg_bp &= ~(IXGBE_AUTOC_SYM_PAUSE |
02275                     IXGBE_AUTOC_ASM_PAUSE);
02276         else if (hw->phy.media_type == ixgbe_media_type_copper)
02277             reg_cu &= ~(IXGBE_TAF_SYM_PAUSE | IXGBE_TAF_ASM_PAUSE);
02278         break;
02279     case ixgbe_fc_rx_pause:
02280         /*
02281          * Rx Flow control is enabled and Tx Flow control is
02282          * disabled by software override. Since there really
02283          * isn't a way to advertise that we are capable of RX
02284          * Pause ONLY, we will advertise that we support both
02285          * symmetric and asymmetric Rx PAUSE.  Later, we will
02286          * disable the adapter's ability to send PAUSE frames.
02287          */
02288         reg |= (IXGBE_PCS1GANA_SYM_PAUSE | IXGBE_PCS1GANA_ASM_PAUSE);
02289         if (hw->phy.media_type == ixgbe_media_type_backplane)
02290             reg_bp |= (IXGBE_AUTOC_SYM_PAUSE |
02291                    IXGBE_AUTOC_ASM_PAUSE);
02292         else if (hw->phy.media_type == ixgbe_media_type_copper)
02293             reg_cu |= (IXGBE_TAF_SYM_PAUSE | IXGBE_TAF_ASM_PAUSE);
02294         break;
02295     case ixgbe_fc_tx_pause:
02296         /*
02297          * Tx Flow control is enabled, and Rx Flow control is
02298          * disabled by software override.
02299          */
02300         reg |= (IXGBE_PCS1GANA_ASM_PAUSE);
02301         reg &= ~(IXGBE_PCS1GANA_SYM_PAUSE);
02302         if (hw->phy.media_type == ixgbe_media_type_backplane) {
02303             reg_bp |= (IXGBE_AUTOC_ASM_PAUSE);
02304             reg_bp &= ~(IXGBE_AUTOC_SYM_PAUSE);
02305         } else if (hw->phy.media_type == ixgbe_media_type_copper) {
02306             reg_cu |= (IXGBE_TAF_ASM_PAUSE);
02307             reg_cu &= ~(IXGBE_TAF_SYM_PAUSE);
02308         }
02309         break;
02310     case ixgbe_fc_full:
02311         /* Flow control (both Rx and Tx) is enabled by SW override. */
02312         reg |= (IXGBE_PCS1GANA_SYM_PAUSE | IXGBE_PCS1GANA_ASM_PAUSE);
02313         if (hw->phy.media_type == ixgbe_media_type_backplane)
02314             reg_bp |= (IXGBE_AUTOC_SYM_PAUSE |
02315                    IXGBE_AUTOC_ASM_PAUSE);
02316         else if (hw->phy.media_type == ixgbe_media_type_copper)
02317             reg_cu |= (IXGBE_TAF_SYM_PAUSE | IXGBE_TAF_ASM_PAUSE);
02318         break;
02319 #ifdef CONFIG_DCB
02320     case ixgbe_fc_pfc:
02321         goto out;
02322         break;
02323 #endif /* CONFIG_DCB */
02324     default:
02325         hw_dbg(hw, "Flow control param set incorrectly\n");
02326         ret_val = IXGBE_ERR_CONFIG;
02327         goto out;
02328         break;
02329     }
02330 
02331     if (hw->mac.type != ixgbe_mac_X540) {
02332         /*
02333          * Enable auto-negotiation between the MAC & PHY;
02334          * the MAC will advertise clause 37 flow control.
02335          */
02336         IXGBE_WRITE_REG(hw, IXGBE_PCS1GANA, reg);
02337         reg = IXGBE_READ_REG(hw, IXGBE_PCS1GLCTL);
02338 
02339         /* Disable AN timeout */
02340         if (hw->fc.strict_ieee)
02341             reg &= ~IXGBE_PCS1GLCTL_AN_1G_TIMEOUT_EN;
02342 
02343         IXGBE_WRITE_REG(hw, IXGBE_PCS1GLCTL, reg);
02344         hw_dbg(hw, "Set up FC; PCS1GLCTL = 0x%08X\n", reg);
02345     }
02346 
02347     /*
02348      * AUTOC restart handles negotiation of 1G and 10G on backplane
02349      * and copper. There is no need to set the PCS1GCTL register.
02350      *
02351      */
02352     if (hw->phy.media_type == ixgbe_media_type_backplane) {
02353         reg_bp |= IXGBE_AUTOC_AN_RESTART;
02354         IXGBE_WRITE_REG(hw, IXGBE_AUTOC, reg_bp);
02355     } else if ((hw->phy.media_type == ixgbe_media_type_copper) &&
02356             (ixgbe_device_supports_autoneg_fc(hw) == 0)) {
02357         hw->phy.ops.write_reg(hw, IXGBE_MDIO_AUTO_NEG_ADVT,
02358                       IXGBE_MDIO_AUTO_NEG_DEV_TYPE, reg_cu);
02359     }
02360 
02361     hw_dbg(hw, "Set up FC; IXGBE_AUTOC = 0x%08X\n", reg);
02362 out:
02363     return ret_val;
02364 }

02375 s32 ixgbe_disable_pcie_master(struct ixgbe_hw *hw)
02376 {
02377     u32 i;
02378     u32 reg_val;
02379     u32 number_of_queues;
02380     s32 status = 0;
02381 
02382     /* Just jump out if bus mastering is already disabled */
02383     if (!(IXGBE_READ_REG(hw, IXGBE_STATUS) & IXGBE_STATUS_GIO))
02384         goto out;
02385 
02386     /* Disable the receive unit by stopping each queue */
02387     number_of_queues = hw->mac.max_rx_queues;
02388     for (i = 0; i < number_of_queues; i++) {
02389         reg_val = IXGBE_READ_REG(hw, IXGBE_RXDCTL(i));
02390         if (reg_val & IXGBE_RXDCTL_ENABLE) {
02391             reg_val &= ~IXGBE_RXDCTL_ENABLE;
02392             IXGBE_WRITE_REG(hw, IXGBE_RXDCTL(i), reg_val);
02393         }
02394     }
02395 
02396     reg_val = IXGBE_READ_REG(hw, IXGBE_CTRL);
02397     reg_val |= IXGBE_CTRL_GIO_DIS;
02398     IXGBE_WRITE_REG(hw, IXGBE_CTRL, reg_val);
02399 
02400     for (i = 0; i < IXGBE_PCI_MASTER_DISABLE_TIMEOUT; i++) {
02401         if (!(IXGBE_READ_REG(hw, IXGBE_STATUS) & IXGBE_STATUS_GIO))
02402             goto check_device_status;
02403         udelay(100);
02404     }
02405 
02406     hw_dbg(hw, "GIO Master Disable bit didn't clear - requesting resets\n");
02407     status = IXGBE_ERR_MASTER_REQUESTS_PENDING;
02408 
02409     /*
02410      * Before proceeding, make sure that the PCIe block does not have
02411      * transactions pending.
02412      */
02413 check_device_status:
02414     for (i = 0; i < IXGBE_PCI_MASTER_DISABLE_TIMEOUT; i++) {
02415         if (!(IXGBE_READ_PCIE_WORD(hw, IXGBE_PCI_DEVICE_STATUS) &
02416             IXGBE_PCI_DEVICE_STATUS_TRANSACTION_PENDING))
02417             break;
02418         udelay(100);
02419     }
02420 
02421     if (i == IXGBE_PCI_MASTER_DISABLE_TIMEOUT)
02422         hw_dbg(hw, "PCIe transaction pending bit also did not clear.\n");
02423     else
02424         goto out;
02425 
02426     /*
02427      * Two consecutive resets are required via CTRL.RST per datasheet
02428      * 5.2.5.3.2 Master Disable.  We set a flag to inform the reset routine
02429      * of this need.  The first reset prevents new master requests from
02430      * being issued by our device.  We then must wait 1usec for any
02431      * remaining completions from the PCIe bus to trickle in, and then reset
02432      * again to clear out any effects they may have had on our device.
02433      */
02434      hw->mac.flags |= IXGBE_FLAGS_DOUBLE_RESET_REQUIRED;
02435 
02436 out:
02437     return status;
02438 }

02449 s32 ixgbe_acquire_swfw_sync(struct ixgbe_hw *hw, u16 mask)
02450 {
02451     u32 gssr;
02452     u32 swmask = mask;
02453     u32 fwmask = mask << 5;
02454     s32 timeout = 200;
02455 
02456     while (timeout) {
02457         /*
02458          * SW EEPROM semaphore bit is used for access to all
02459          * SW_FW_SYNC/GSSR bits (not just EEPROM)
02460          */
02461         if (ixgbe_get_eeprom_semaphore(hw))
02462             return IXGBE_ERR_SWFW_SYNC;
02463 
02464         gssr = IXGBE_READ_REG(hw, IXGBE_GSSR);
02465         if (!(gssr & (fwmask | swmask)))
02466             break;
02467 
02468         /*
02469          * Firmware currently using resource (fwmask) or other software
02470          * thread currently using resource (swmask)
02471          */
02472         ixgbe_release_eeprom_semaphore(hw);
02473         msleep(5);
02474         timeout--;
02475     }
02476 
02477     if (!timeout) {
02478         hw_dbg(hw, "Driver can't access resource, SW_FW_SYNC timeout.\n");
02479         return IXGBE_ERR_SWFW_SYNC;
02480     }
02481 
02482     gssr |= swmask;
02483     IXGBE_WRITE_REG(hw, IXGBE_GSSR, gssr);
02484 
02485     ixgbe_release_eeprom_semaphore(hw);
02486     return 0;
02487 }

02613 {
02614     u16 san_mac_data, san_mac_offset;
02615     u8 i;
02616 
02617     /*
02618      * First read the EEPROM pointer to see if the MAC addresses are
02619      * available.  If they're not, no point in calling set_lan_id() here.
02620      */

02721 s32 ixgbe_insert_mac_addr_generic(struct ixgbe_hw *hw, u8 *addr, u32 vmdq)
02722 {
02723     static const u32 NO_EMPTY_RAR_FOUND = 0xFFFFFFFF;
02724     u32 first_empty_rar = NO_EMPTY_RAR_FOUND;
02725     u32 rar;
02726     u32 rar_low, rar_high;
02727     u32 addr_low, addr_high;
02728 
02729     /* swap bytes for HW little endian */
02730     addr_low  = addr[0] | (addr[1] << 8)
02731                 | (addr[2] << 16)
02732                 | (addr[3] << 24);
02733     addr_high = addr[4] | (addr[5] << 8);
02734 
02735     /*
02736      * Either find the mac_id in rar or find the first empty space.
02737      * rar_highwater points to just after the highest currently used
02738      * rar in order to shorten the search.  It grows when we add a new
02739      * rar to the top.
02740      */
02741     for (rar = 0; rar < hw->mac.rar_highwater; rar++) {
02742         rar_high = IXGBE_READ_REG(hw, IXGBE_RAH(rar));
02743 
02744         if (((IXGBE_RAH_AV & rar_high) == 0)
02745             && first_empty_rar == NO_EMPTY_RAR_FOUND) {
02746             first_empty_rar = rar;
02747         } else if ((rar_high & 0xFFFF) == addr_high) {
02748             rar_low = IXGBE_READ_REG(hw, IXGBE_RAL(rar));
02749             if (rar_low == addr_low)
02750                 break;    /* found it already in the rars */
02751         }
02752     }
02753 
02754     if (rar < hw->mac.rar_highwater) {
02755         /* already there so just add to the pool bits */
02756         ixgbe_set_vmdq(hw, rar, vmdq);
02757     } else if (first_empty_rar != NO_EMPTY_RAR_FOUND) {
02758         /* stick it into first empty RAR slot we found */
02759         rar = first_empty_rar;
02760         ixgbe_set_rar(hw, rar, addr, vmdq, IXGBE_RAH_AV);
02761     } else if (rar == hw->mac.rar_highwater) {
02762         /* add it to the top of the list and inc the highwater mark */
02763         ixgbe_set_rar(hw, rar, addr, vmdq, IXGBE_RAH_AV);
02764         hw->mac.rar_highwater++;
02765     } else if (rar >= hw->mac.num_rar_entries) {
02766         return IXGBE_ERR_INVALID_MAC_ADDR;
02767     }
02768 
02769     /*
02770      * If we found rar[0], make sure the default pool bit (we use pool 0)
02771      * remains cleared to be sure default pool packets will get delivered
02772      */
02773     if (rar == 0)
02774         ixgbe_clear_vmdq(hw, rar, 0);
02775 
02776     return rar;
02777 }

02785 s32 ixgbe_clear_vmdq_generic(struct ixgbe_hw *hw, u32 rar, u32 vmdq)
02786 {
02787     u32 mpsar_lo, mpsar_hi;
02788     u32 rar_entries = hw->mac.num_rar_entries;
02789 
02790     /* Make sure we are using a valid rar index range */
02791     if (rar >= rar_entries) {
02792         hw_dbg(hw, "RAR index %d is out of range.\n", rar);
02793         return IXGBE_ERR_INVALID_ARGUMENT;
02794     }
02795 
02796     mpsar_lo = IXGBE_READ_REG(hw, IXGBE_MPSAR_LO(rar));
02797     mpsar_hi = IXGBE_READ_REG(hw, IXGBE_MPSAR_HI(rar));
02798 
02799     if (!mpsar_lo && !mpsar_hi)
02800         goto done;
02801 
02802     if (vmdq == IXGBE_CLEAR_VMDQ_ALL) {
02803         if (mpsar_lo) {
02804             IXGBE_WRITE_REG(hw, IXGBE_MPSAR_LO(rar), 0);
02805             mpsar_lo = 0;
02806         }
02807         if (mpsar_hi) {
02808             IXGBE_WRITE_REG(hw, IXGBE_MPSAR_HI(rar), 0);
02809             mpsar_hi = 0;
02810         }
02811     } else if (vmdq < 32) {
02812         mpsar_lo &= ~(1 << vmdq);
02813         IXGBE_WRITE_REG(hw, IXGBE_MPSAR_LO(rar), mpsar_lo);
02814     } else {
02815         mpsar_hi &= ~(1 << (vmdq - 32));
02816         IXGBE_WRITE_REG(hw, IXGBE_MPSAR_HI(rar), mpsar_hi);
02817     }
02818 
02819     /* was that the last pool using this rar? */
02820     if (mpsar_lo == 0 && mpsar_hi == 0 && rar != 0)
02821         hw->mac.ops.clear_rar(hw, rar);
02822 done:
02823     return 0;
02824 }

02879 s32 ixgbe_find_vlvf_slot(struct ixgbe_hw *hw, u32 vlan)
02880 {
02881     u32 bits = 0;
02882     u32 first_empty_slot = 0;
02883     s32 regindex;
02884 
02885     /* short cut the special case */
02886     if (vlan == 0)
02887         return 0;
02888 
02889     /*
02890       * Search for the vlan id in the VLVF entries. Save off the first empty
02891       * slot found along the way
02892       */
02893     for (regindex = 1; regindex < IXGBE_VLVF_ENTRIES; regindex++) {
02894         bits = IXGBE_READ_REG(hw, IXGBE_VLVF(regindex));
02895         if (!bits && !(first_empty_slot))
02896             first_empty_slot = regindex;
02897         else if ((bits & 0x0FFF) == vlan)
02898             break;
02899     }
02900 
02901     /*
02902       * If regindex is less than IXGBE_VLVF_ENTRIES, then we found the vlan
02903       * in the VLVF. Else use the first empty VLVF register for this
02904       * vlan id.
02905       */
02906     if (regindex >= IXGBE_VLVF_ENTRIES) {
02907         if (first_empty_slot)
02908             regindex = first_empty_slot;
02909         else {
02910             hw_dbg(hw, "No space in VLVF.\n");
02911             regindex = IXGBE_ERR_NO_SPACE;
02912         }
02913     }
02914 
02915     return regindex;

02927 s32 ixgbe_set_vfta_generic(struct ixgbe_hw *hw, u32 vlan, u32 vind,
02928                            bool vlan_on)
02929 {
02930     s32 regindex;
02931     u32 bitindex;
02932     u32 vfta;
02933     u32 bits;
02934     u32 vt;
02935     u32 targetbit;
02936     bool vfta_changed = false;
02937 
02938     if (vlan > 4095)
02939         return IXGBE_ERR_PARAM;
02940 
02941     /*
02942      * this is a 2 part operation - first the VFTA, then the
02943      * VLVF and VLVFB if VT Mode is set
02944      * We don't write the VFTA until we know the VLVF part succeeded.
02945      */
02946 
02947     /* Part 1
02948      * The VFTA is a bitstring made up of 128 32-bit registers
02949      * that enable the particular VLAN id, much like the MTA:
02950      *    bits[11-5]: which register
02951      *    bits[4-0]:  which bit in the register
02952      */
02953     regindex = (vlan >> 5) & 0x7F;
02954     bitindex = vlan & 0x1F;
02955     targetbit = (1 << bitindex);
02956     vfta = IXGBE_READ_REG(hw, IXGBE_VFTA(regindex));
02957 
02958     if (vlan_on) {
02959         if (!(vfta & targetbit)) {
02960             vfta |= targetbit;
02961             vfta_changed = true;
02962         }
02963     } else {
02964         if ((vfta & targetbit)) {
02965             vfta &= ~targetbit;
02966             vfta_changed = true;
02967         }
02968     }
02969 
02970     /* Part 2
02971      * If VT Mode is set
02972      *   Either vlan_on
02973      *     make sure the vlan is in VLVF
02974      *     set the vind bit in the matching VLVFB
02975      *   Or !vlan_on
02976      *     clear the pool bit and possibly the vind
02977      */
02978     vt = IXGBE_READ_REG(hw, IXGBE_VT_CTL);
02979     if (vt & IXGBE_VT_CTL_VT_ENABLE) {
02980         s32 vlvf_index;
02981 
02982         vlvf_index = ixgbe_find_vlvf_slot(hw, vlan);
02983         if (vlvf_index < 0)
02984             return vlvf_index;
02985 
02986         if (vlan_on) {
02987             /* set the pool bit */
02988             if (vind < 32) {
02989                 bits = IXGBE_READ_REG(hw,
02990                         IXGBE_VLVFB(vlvf_index*2));
02991                 bits |= (1 << vind);
02992                 IXGBE_WRITE_REG(hw,
02993                         IXGBE_VLVFB(vlvf_index*2),
02994                         bits);
02995             } else {
02996                 bits = IXGBE_READ_REG(hw,
02997                         IXGBE_VLVFB((vlvf_index*2)+1));
02998                 bits |= (1 << (vind-32));
02999                 IXGBE_WRITE_REG(hw,
03000                         IXGBE_VLVFB((vlvf_index*2)+1),
03001                         bits);
03002             }
03003         } else {
03004             /* clear the pool bit */
03005             if (vind < 32) {
03006                 bits = IXGBE_READ_REG(hw,
03007                         IXGBE_VLVFB(vlvf_index*2));
03008                 bits &= ~(1 << vind);
03009                 IXGBE_WRITE_REG(hw,
03010                         IXGBE_VLVFB(vlvf_index*2),
03011                         bits);
03012                 bits |= IXGBE_READ_REG(hw,
03013                         IXGBE_VLVFB((vlvf_index*2)+1));
03014             } else {
03015                 bits = IXGBE_READ_REG(hw,
03016                         IXGBE_VLVFB((vlvf_index*2)+1));
03017                 bits &= ~(1 << (vind-32));
03018                 IXGBE_WRITE_REG(hw,
03019                         IXGBE_VLVFB((vlvf_index*2)+1),
03020                         bits);
03021                 bits |= IXGBE_READ_REG(hw,
03022                         IXGBE_VLVFB(vlvf_index*2));
03023             }
03024         }
03025 
03026         /*
03027          * If there are still bits set in the VLVFB registers
03028          * for the VLAN ID indicated we need to see if the
03029          * caller is requesting that we clear the VFTA entry bit.
03030          * If the caller has requested that we clear the VFTA
03031          * entry bit but there are still pools/VFs using this VLAN
03032          * ID entry then ignore the request.  We're not worried
03033          * about the case where we're turning the VFTA VLAN ID
03034          * entry bit on, only when requested to turn it off as
03035          * there may be multiple pools and/or VFs using the
03036          * VLAN ID entry.  In that case we cannot clear the
03037          * VFTA bit until all pools/VFs using that VLAN ID have also
03038          * been cleared.  This will be indicated by "bits" being
03039          * zero.
03040          */
03041         if (bits) {
03042             IXGBE_WRITE_REG(hw, IXGBE_VLVF(vlvf_index),
03043                     (IXGBE_VLVF_VIEN | vlan));
03044             if (!vlan_on) {
03045                 /* someone wants to clear the vfta entry
03046                  * but some pools/VFs are still using it.
03047                  * Ignore it. */
03048                 vfta_changed = false;
03049             }
03050         }
03051         else
03052             IXGBE_WRITE_REG(hw, IXGBE_VLVF(vlvf_index), 0);
03053     }
03054 
03055     if (vfta_changed)
03056         IXGBE_WRITE_REG(hw, IXGBE_VFTA(regindex), vfta);
03057 
03058     return 0;
03059 }

03396     hicr = IXGBE_READ_REG(hw, IXGBE_HICR);

03429         goto out;
03430     }

03433     dword_len = hdr_size >> 2;
03434

03436     for (i = 0; i < dword_len; i++)
03437         *((u32 *)buffer + i) =
03438             IXGBE_READ_REG_ARRAY(hw, IXGBE_FLEX_MNG, i);

03448         goto out;

03455     for (; i < buf_len; i++)

03457             IXGBE_READ_REG_ARRAY(hw, IXGBE_FLEX_MNG, i);
