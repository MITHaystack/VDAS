00102 static const u_char *
00103 c_print(register const u_char *s, register const u_char *ep)
00104 {
00105     register u_char c;
00106     register int flag;
00107 
00108     flag = 1;
00109     while (s < ep) {
00110         c = *s++;
00111         if (c == '\0') {
00112             flag = 0;
00113             break;
00114         }
00115         if (!isascii(c)) {
00116             c = toascii(c);
00117             putchar('M');
00118             putchar('-');
00119         }
00120         if (!isprint(c)) {
00121             c ^= 0x40;  /* DEL to ?, others to alpha */
00122             putchar('^');
00123         }
00124         putchar(c);
00125     }
00126     if (flag)
00127         return NULL;
00128     return (s);
00129 }

00152 static void
00153 krb4_print(const u_char *cp)
00154 {
00155     register const struct krb *kp;
00156     u_char type;
00157     u_short len;
00158 
00159 #define PRINT       if ((cp = c_print(cp, snapend)) == NULL) goto trunc
00160 /*  True if struct krb is little endian */
00161 #define IS_LENDIAN(kp)  (((kp)->type & 0x01) != 0)
00162 #define KTOHSP(kp, cp)  (IS_LENDIAN(kp) ? EXTRACT_LE_16BITS(cp) : EXTRACT_16BITS(cp))
00163 
00164     kp = (struct krb *)cp;
00165 
00166     if ((&kp->type) >= snapend) {
00167         fputs(tstr, stdout);
00168         return;
00169     }
00170 
00171     type = kp->type & (0xFF << 1);
00172 
00173     printf(" %s %s: ",
00174         IS_LENDIAN(kp) ? "le" : "be", tok2str(type2str, NULL, type));
00175 
00176     switch (type) {
00177 
00178     case AUTH_MSG_KDC_REQUEST:
00179         if ((cp = krb4_print_hdr(cp)) == NULL)
00180             return;
00181         cp += 4;    /* ctime */
00182         TCHECK(*cp);
00183         printf(" %dmin ", *cp++ * 5);
00184         PRINT;
00185         putchar('.');
00186         PRINT;
00187         break;
00188 
00189     case AUTH_MSG_APPL_REQUEST:
00190         cp += 2;
00191         TCHECK(*cp);
00192         printf("v%d ", *cp++);
00193         PRINT;
00194         TCHECK(*cp);
00195         printf(" (%d)", *cp++);
00196         TCHECK(*cp);
00197         printf(" (%d)", *cp);
00198         break;
00199 
00200     case AUTH_MSG_KDC_REPLY:
00201         if ((cp = krb4_print_hdr(cp)) == NULL)
00202             return;
00203         cp += 10;   /* timestamp + n + exp + kvno */
00204         TCHECK2(*cp, sizeof(short));
00205         len = KTOHSP(kp, cp);
00206         printf(" (%d)", len);
00207         break;
00208 
00209     case AUTH_MSG_ERR_REPLY:
00210         if ((cp = krb4_print_hdr(cp)) == NULL)
00211             return;
00212         cp += 4;      /* timestamp */
00213         TCHECK2(*cp, sizeof(short));
00214         printf(" %s ", tok2str(kerr2str, NULL, KTOHSP(kp, cp)));
00215         cp += 4;
00216         PRINT;
00217         break;
00218 
00219     default:
00220         fputs("(unknown)", stdout);
00221         break;
00222     }
00223 
00224     return;
00225 trunc:
00226     fputs(tstr, stdout);
00227 }
