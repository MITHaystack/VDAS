00257 int acn_parse_hosts_file(char *errbuf) {                /* returns: -1 = error, 0 = OK */
00258     FILE    *fp;
00259     char    buf[MAX_LINE_SIZE];
00260     char    *ptr, *ptr2;
00261     int     pos;
00262     int     chassis, geoslot;
00263     unit_t  *u;
00264 
00265     empty_unit_table();
00266     if ((fp = fopen("/etc/hosts", "r")) == NULL) {                                      /* try to open the hosts file and if it fails */
00267         snprintf(errbuf, PCAP_ERRBUF_SIZE, "Cannot open '/etc/hosts' for reading.");    /* return the nohostsfile error response */
00268         return -1;
00269     }
00270     while (fgets(buf, MAX_LINE_SIZE-1, fp)) {           /* while looping over the file */
00271 
00272         pos = strcspn(buf, "#\n\r");                    /* find the first comment character or EOL */
00273         *(buf + pos) = '\0';                            /* and clobber it and anything that follows it */
00274 
00275         pos = strspn(buf, " \t");                       /* then find the first non-white space */
00276         if (pos == strlen(buf))                         /* if there is nothing but white space on the line */
00277             continue;                                   /* ignore that empty line */
00278         ptr = buf + pos;                                /* and skip over any of that leading whitespace */
00279 
00280         if ((ptr2 = strstr(ptr, "_I_")) == NULL)        /* skip any lines that don't have names that look like they belong to IOPs */
00281             continue;
00282         if (*(ptr2 + 4) != '_')                         /* and skip other lines that have names that don't look like ACN components */
00283             continue;
00284         *(ptr + strcspn(ptr, " \t")) = '\0';            /* null terminate the IP address so its a standalone string */
00285 
00286         chassis = *(ptr2 + 3) - '0';                    /* extract the chassis number */
00287         geoslot = *(ptr2 + 5) - '0';                    /* and geo-slot number */
00288         if (chassis < 1 || chassis > MAX_CHASSIS ||
00289             geoslot < 1 || geoslot > MAX_GEOSLOT) {     /* if the chassis and/or slot numbers appear to be bad... */
00290             snprintf(errbuf, PCAP_ERRBUF_SIZE, "Invalid ACN name in '/etc/hosts'.");    /* warn the user */
00291             continue;                                                                   /* and ignore the entry */
00292         }
00293         if ((ptr2 = (char *)malloc(strlen(ptr) + 1)) == NULL) {
00294             snprintf(errbuf, PCAP_ERRBUF_SIZE, "malloc: %s", pcap_strerror(errno));
00295             continue;
00296         }
00297         strcpy(ptr2, ptr);                              /* copy the IP address into our malloc'ed memory */
00298         u = &units[chassis][geoslot];
00299         u->ip = ptr2;                                   /* and remember the whole shebang */
00300         u->chassis = chassis;
00301         u->geoslot = geoslot;
00302     }
00303     fclose(fp);

00330     else                u->find_fd = sockfd;

00465 static int if_sort(char *s1, char *s2) {
00466     char    *s1_p2, *s2_p2;
00467     char    str1[MAX_LINE_SIZE], str2[MAX_LINE_SIZE];
00468     int     s1_p1_len, s2_p1_len;
00469     int     retval;
00470 
00471     if ((s1_p2 = strchr(s1, '_'))) {    /* if an underscore is found... */
00472         s1_p1_len = s1_p2 - s1;         /* the prefix length is the difference in pointers */
00473         s1_p2++;                        /* the suffix actually starts _after_ the underscore */
00474     } else {                            /* otherwise... */
00475         s1_p1_len = strlen(s1);         /* the prefix length is the length of the string itself */
00476         s1_p2 = 0;                      /* and there is no suffix */
00477     }
00478     if ((s2_p2 = strchr(s2, '_'))) {    /* now do the same for the second string */
00479         s2_p1_len = s2_p2 - s2;
00480         s2_p2++;
00481     } else {
00482         s2_p1_len = strlen(s2);
00483         s2_p2 = 0;
00484     }
00485     strncpy(str1, s1, (s1_p1_len > sizeof(str1)) ? s1_p1_len : sizeof(str1));   *(str1 + s1_p1_len) = 0;
00486     strncpy(str2, s2, (s2_p1_len > sizeof(str2)) ? s2_p1_len : sizeof(str2));   *(str2 + s2_p1_len) = 0;
00487     retval = strcmp(str1, str2);
00488     if (retval != 0) return retval;     /* if they are not identical, then we can quit now and return the indication */
00489     return strcmp(s1_p2, s2_p2);        /* otherwise we return the result of comparing the 2nd half of the string */
00490 }
