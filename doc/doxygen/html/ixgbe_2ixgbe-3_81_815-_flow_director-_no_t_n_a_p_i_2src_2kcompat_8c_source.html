00038     int i=0;

00053 static char * number(char * buf, char * end, long long num, int base, int size, int precision, int type)
00054 {
00055     char c,sign,tmp[66];
00056     const char *digits;
00057     const char small_digits[] = "0123456789abcdefghijklmnopqrstuvwxyz";
00058     const char large_digits[] = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
00059     int i;
00060 
00061     digits = (type & _kc_LARGE) ? large_digits : small_digits;
00062     if (type & _kc_LEFT)
00063         type &= ~_kc_ZEROPAD;
00064     if (base < 2 || base > 36)
00065         return 0;
00066     c = (type & _kc_ZEROPAD) ? '0' : ' ';
00067     sign = 0;
00068     if (type & _kc_SIGN) {
00069         if (num < 0) {
00070             sign = '-';
00071             num = -num;
00072             size--;
00073         } else if (type & _kc_PLUS) {
00074             sign = '+';
00075             size--;
00076         } else if (type & _kc_SPACE) {
00077             sign = ' ';
00078             size--;
00079         }
00080     }
00081     if (type & _kc_SPECIAL) {
00082         if (base == 16)
00083             size -= 2;
00084         else if (base == 8)
00085             size--;
00086     }
00087     i = 0;
00088     if (num == 0)
00089         tmp[i++]='0';
00090     else while (num != 0)
00091         tmp[i++] = digits[do_div(num,base)];
00092     if (i > precision)
00093         precision = i;
00094     size -= precision;
00095     if (!(type&(_kc_ZEROPAD+_kc_LEFT))) {
00096         while(size-->0) {
00097             if (buf <= end)
00098                 *buf = ' ';
00099             ++buf;
00100         }
00101     }
00102     if (sign) {
00103         if (buf <= end)
00104             *buf = sign;
00105         ++buf;
00106     }
00107     if (type & _kc_SPECIAL) {
00108         if (base==8) {
00109             if (buf <= end)
00110                 *buf = '0';
00111             ++buf;
00112         } else if (base==16) {
00113             if (buf <= end)
00114                 *buf = '0';
00115             ++buf;
00116             if (buf <= end)
00117                 *buf = digits[33];
00118             ++buf;
00119         }
00120     }
00121     if (!(type & _kc_LEFT)) {
00122         while (size-- > 0) {
00123             if (buf <= end)
00124                 *buf = c;
00125             ++buf;
00126         }
00127     }
00128     while (i < precision--) {
00129         if (buf <= end)
00130             *buf = '0';
00131         ++buf;
00132     }
00133     while (i-- > 0) {
00134         if (buf <= end)
00135             *buf = tmp[i];
00136         ++buf;
00137     }
00138     while (size-- > 0) {
00139         if (buf <= end)
00140             *buf = ' ';
00141         ++buf;
00142     }
00143     return buf;
00144 }

00146 int _kc_vsnprintf(char *buf, size_t size, const char *fmt, va_list args)
00147 {
00148     int len;
00149     unsigned long long num;
00150     int i, base;
00151     char *str, *end, c;
00152     const char *s;
00153 
00154     int flags;      /* flags to number() */
00155 
00156     int field_width;    /* width of output field */
00157     int precision;      /* min. # of digits for integers; max
00158                    number of chars for from string */
00159     int qualifier;      /* 'h', 'l', or 'L' for integer fields */
00160                 /* 'z' support added 23/7/1999 S.H.    */
00161                 /* 'z' changed to 'Z' --davidm 1/25/99 */
00162 
00163     str = buf;
00164     end = buf + size - 1;
00165 
00166     if (end < buf - 1) {
00167         end = ((void *) -1);
00168         size = end - buf + 1;
00169     }
00170 
00171     for (; *fmt ; ++fmt) {
00172         if (*fmt != '%') {
00173             if (str <= end)
00174                 *str = *fmt;
00175             ++str;
00176             continue;
00177         }
00178 
00179         /* process flags */
00180         flags = 0;
00181         repeat:
00182             ++fmt;      /* this also skips first '%' */
00183             switch (*fmt) {
00184                 case '-': flags |= _kc_LEFT; goto repeat;
00185                 case '+': flags |= _kc_PLUS; goto repeat;
00186                 case ' ': flags |= _kc_SPACE; goto repeat;
00187                 case '#': flags |= _kc_SPECIAL; goto repeat;
00188                 case '0': flags |= _kc_ZEROPAD; goto repeat;
00189             }
00190 
00191         /* get field width */
00192         field_width = -1;
00193         if (isdigit(*fmt))
00194             field_width = skip_atoi(&fmt);
00195         else if (*fmt == '*') {
00196             ++fmt;
00197             /* it's the next argument */
00198             field_width = va_arg(args, int);
00199             if (field_width < 0) {
00200                 field_width = -field_width;
00201                 flags |= _kc_LEFT;
00202             }
00203         }
00204 
00205         /* get the precision */
00206         precision = -1;
00207         if (*fmt == '.') {
00208             ++fmt;  
00209             if (isdigit(*fmt))
00210                 precision = skip_atoi(&fmt);
00211             else if (*fmt == '*') {
00212                 ++fmt;
00213                 /* it's the next argument */
00214                 precision = va_arg(args, int);
00215             }
00216             if (precision < 0)
00217                 precision = 0;
00218         }
00219 
00220         /* get the conversion qualifier */
00221         qualifier = -1;
00222         if (*fmt == 'h' || *fmt == 'l' || *fmt == 'L' || *fmt =='Z') {
00223             qualifier = *fmt;
00224             ++fmt;
00225         }
00226 
00227         /* default base */
00228         base = 10;
00229 
00230         switch (*fmt) {
00231             case 'c':
00232                 if (!(flags & _kc_LEFT)) {
00233                     while (--field_width > 0) {
00234                         if (str <= end)
00235                             *str = ' ';
00236                         ++str;
00237                     }
00238                 }
00239                 c = (unsigned char) va_arg(args, int);
00240                 if (str <= end)
00241                     *str = c;
00242                 ++str;
00243                 while (--field_width > 0) {
00244                     if (str <= end)
00245                         *str = ' ';
00246                     ++str;
00247                 }
00248                 continue;
00249 
00250             case 's':
00251                 s = va_arg(args, char *);
00252                 if (!s)
00253                     s = "<NULL>";
00254 
00255                 len = strnlen(s, precision);
00256 
00257                 if (!(flags & _kc_LEFT)) {
00258                     while (len < field_width--) {
00259                         if (str <= end)
00260                             *str = ' ';
00261                         ++str;
00262                     }
00263                 }
00264                 for (i = 0; i < len; ++i) {
00265                     if (str <= end)
00266                         *str = *s;
00267                     ++str; ++s;
00268                 }
00269                 while (len < field_width--) {
00270                     if (str <= end)
00271                         *str = ' ';
00272                     ++str;
00273                 }
00274                 continue;
00275 
00276             case 'p':
00277                 if (field_width == -1) {
00278                     field_width = 2*sizeof(void *);
00279                     flags |= _kc_ZEROPAD;
00280                 }
00281                 str = number(str, end,
00282                         (unsigned long) va_arg(args, void *),
00283                         16, field_width, precision, flags);
00284                 continue;
00285 
00286 
00287             case 'n':
00288                 /* FIXME:
00289                 * What does C99 say about the overflow case here? */
00290                 if (qualifier == 'l') {
00291                     long * ip = va_arg(args, long *);
00292                     *ip = (str - buf);
00293                 } else if (qualifier == 'Z') {
00294                     size_t * ip = va_arg(args, size_t *);
00295                     *ip = (str - buf);
00296                 } else {
00297                     int * ip = va_arg(args, int *);
00298                     *ip = (str - buf);
00299                 }
00300                 continue;
00301 
00302             case '%':
00303                 if (str <= end)
00304                     *str = '%';
00305                 ++str;
00306                 continue;
00307 
00308                 /* integer number formats - set up the flags and "break" */
00309             case 'o':
00310                 base = 8;
00311                 break;
00312 
00313             case 'X':
00314                 flags |= _kc_LARGE;
00315             case 'x':
00316                 base = 16;
00317                 break;
00318 
00319             case 'd':
00320             case 'i':
00321                 flags |= _kc_SIGN;
00322             case 'u':
00323                 break;
00324 
00325             default:
00326                 if (str <= end)
00327                     *str = '%';
00328                 ++str;
00329                 if (*fmt) {
00330                     if (str <= end)
00331                         *str = *fmt;
00332                     ++str;
00333                 } else {
00334                     --fmt;
00335                 }
00336                 continue;
00337         }
00338         if (qualifier == 'L')
00339             num = va_arg(args, long long);
00340         else if (qualifier == 'l') {
00341             num = va_arg(args, unsigned long);
00342             if (flags & _kc_SIGN)
00343                 num = (signed long) num;
00344         } else if (qualifier == 'Z') {
00345             num = va_arg(args, size_t);
00346         } else if (qualifier == 'h') {
00347             num = (unsigned short) va_arg(args, int);
00348             if (flags & _kc_SIGN)
00349                 num = (signed short) num;
00350         } else {
00351             num = va_arg(args, unsigned int);
00352             if (flags & _kc_SIGN)
00353                 num = (signed int) num;
00354         }
00355         str = number(str, end, num, base,
00356                 field_width, precision, flags);
00357     }
00358     if (str <= end)
00359         *str = '\0';
00360     else if (size > 0)
00361         /* don't write out a null byte if the buf size is zero */
00362         *end = '\0';
00363     /* the trailing null byte doesn't count towards the total
00364     * ++str;
00365     */
00366     return str-buf;
00367 }

00576 unsigned long find_next_bit(const unsigned long *addr, unsigned long size,
00577                             unsigned long offset)
00578 {
00579     const unsigned long *p = addr + BITOP_WORD(offset);
00580     unsigned long result = offset & ~(BITS_PER_LONG-1);
00581     unsigned long tmp;
00582 
00583     if (offset >= size)
00584         return size;
00585     size -= result;
00586     offset %= BITS_PER_LONG;
00587     if (offset) {
00588         tmp = *(p++);
00589         tmp &= (~0UL << offset);
00590         if (size < BITS_PER_LONG)
00591             goto found_first;
00592         if (tmp)
00593             goto found_middle;
00594         size -= BITS_PER_LONG;
00595         result += BITS_PER_LONG;
00596     }
00597     while (size & ~(BITS_PER_LONG-1)) {
00598         if ((tmp = *(p++)))
00599             goto found_middle;
00600         result += BITS_PER_LONG;
00601         size -= BITS_PER_LONG;
00602     }
00603     if (!size)
00604         return result;
00605     tmp = *p;
00606 
00607 found_first:
00608     tmp &= (~0UL >> (BITS_PER_LONG - size));
00609     if (tmp == 0UL)     /* Are any bits set? */
00610         return result + size;   /* Nope. */
00611 found_middle:
00612     return result + ffs(tmp);
00613 }
