00270 /*
00271  * Inline versions of get/put short/long.  Pointer is advanced.
00272  * We also assume that a "u_int16_t" holds 2 "chars"
00273  * and that a "u_int32_t" holds 4 "chars".
00274  *
00275  * These macros demonstrate the property of C whereby it can be
00276  * portable or it can be elegant but never both.
00277  */
00278 #define GETSHORT(s, cp) { \
00279     register u_char *t_cp = (u_char *)(cp); \
00280     (s) = ((u_int16_t)t_cp[0] << 8) | (u_int16_t)t_cp[1]; \
00281     (cp) += 2; \
00282 }
00283 
00284 #define GETLONG(l, cp) { \
00285     register u_char *t_cp = (u_char *)(cp); \
00286     (l) = (((u_int32_t)t_cp[0]) << 24) \
00287         | (((u_int32_t)t_cp[1]) << 16) \
00288         | (((u_int32_t)t_cp[2]) << 8) \
00289         | (((u_int32_t)t_cp[3])); \
00290     (cp) += 4; \
00291 }
00292 
00293 #define PUTSHORT(s, cp) { \
00294     register u_int16_t t_s = (u_int16_t)(s); \
00295     register u_char *t_cp = (u_char *)(cp); \
00296     *t_cp++ = t_s >> 8; \
00297     *t_cp   = t_s; \
00298     (cp) += 2; \
00299 }
00300 
00301 /*
00302  * Warning: PUTLONG --no-longer-- destroys its first argument.  if you
00303  * were depending on this "feature", you will lose.
00304  */
00305 #define PUTLONG(l, cp) { \
00306     register u_int32_t t_l = (u_int32_t)(l); \
00307     register u_char *t_cp = (u_char *)(cp); \
00308     *t_cp++ = t_l >> 24; \
00309     *t_cp++ = t_l >> 16; \
00310     *t_cp++ = t_l >> 8; \
00311     *t_cp   = t_l; \
00312     (cp) += 4; \
00313 }
