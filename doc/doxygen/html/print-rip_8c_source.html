00098 static void
00099 rip_entry_print_v1(register const struct rip_netinfo *ni)
00100 {
00101     register u_short family;
00102 
00103     /* RFC 1058 */
00104     family = EXTRACT_16BITS(&ni->rip_family);
00105     if (family != BSD_AFNUM_INET) {
00106         printf("\n\t AFI %s, ", tok2str(bsd_af_values, "Unknown (%u)", family));
00107                 print_unknown_data((u_int8_t *)&ni->rip_family,"\n\t  ",RIP_ROUTELEN);
00108         return;
00109     }
00110     if (EXTRACT_16BITS(&ni->rip_tag) ||
00111         EXTRACT_32BITS(&ni->rip_dest_mask) ||
00112         EXTRACT_32BITS(&ni->rip_router)) {
00113         /* MBZ fields not zero */
00114                 print_unknown_data((u_int8_t *)&ni->rip_family,"\n\t  ",RIP_ROUTELEN);
00115         return;
00116     } /* BSD_AFNUM_INET */
00117     printf("\n\t  %s, metric: %u",
00118                ipaddr_string(&ni->rip_dest),
00119            EXTRACT_32BITS(&ni->rip_metric));
00120 }

00122 static void
00123 rip_entry_print_v2(register const struct rip_netinfo *ni)
00124 {
00125     register u_char *p;
00126     register u_short family;
00127     u_char buf[RIP_AUTHLEN];
00128 
00129     family = EXTRACT_16BITS(&ni->rip_family);
00130     if (family == 0xFFFF) { /* 16 bytes authentication ? */
00131                 if (EXTRACT_16BITS(&ni->rip_tag) == 2) { /* simple text authentication ? */
00132             memcpy(buf, &ni->rip_dest, sizeof(buf));
00133             buf[sizeof(buf)-1] = '\0';
00134             for (p = buf; *p; p++) {
00135                 if (!isprint(*p))
00136                     break;
00137             }
00138                         printf("\n\t  Simple Text Authentication data: %s", buf);
00139                 } else {
00140             printf("\n\t  Unknown (%u) Authentication data:",
00141                    EXTRACT_16BITS(&ni->rip_tag));
00142                         print_unknown_data((u_int8_t *)&ni->rip_dest,"\n\t  ",RIP_AUTHLEN);
00143         }
00144     } else if (family != BSD_AFNUM_INET) {
00145         printf("\n\t  AFI %s", tok2str(bsd_af_values, "Unknown (%u)", family));
00146                 print_unknown_data((u_int8_t *)&ni->rip_tag,"\n\t  ",RIP_ROUTELEN-2);
00147         return;
00148     } else { /* BSD_AFNUM_INET */
00149         printf("\n\t  AFI %s, %15s/%-2d, tag 0x%04x, metric: %u, next-hop: ",
00150                        tok2str(bsd_af_values, "Unknown (%u)", family),
00151                        ipaddr_string(&ni->rip_dest),
00152                mask2plen(EXTRACT_32BITS(&ni->rip_dest_mask)),
00153                        EXTRACT_16BITS(&ni->rip_tag),
00154                        EXTRACT_32BITS(&ni->rip_metric));
00155         if (EXTRACT_32BITS(&ni->rip_router))
00156             printf("%s", ipaddr_string(&ni->rip_router));
00157                 else
00158                     printf("self");
00159     }
00160 }

00184 
00185         printf("%sRIPv%u",

00188 
00189     switch (rp->rip_vers) {
00190     case 0:
00191         /*
00192          * RFC 1058.
00193          *
00194          * XXX - RFC 1058 says
00195          *
00196          * 0  Datagrams whose version number is zero are to be ignored.
00197          *    These are from a previous version of the protocol, whose
00198          *    packet format was machine-specific.
00199          *
00200          * so perhaps we should just dump the packet, in hex.
00201          */
00202                 print_unknown_data((u_int8_t *)&rp->rip_cmd,"\n\t",length);
00203         break;
00204     default:
00205                 /* dump version and lets see if we know the commands name*/
00206                 printf(", %s, length: %u",
00207                        tok2str(rip_cmd_values,
00208                                "unknown command (%u)",
00209                                rp->rip_cmd),
00210                        length);
00211 
00212                 if (vflag < 1)
00213                     return;
00214 
00215         switch (rp->rip_cmd) {
00216         case RIPCMD_RESPONSE:
00217             j = length / sizeof(*ni);
00218                         printf(", routes: %u",j);
00219             trunc = (i / sizeof(*ni)) != j;
00220             ni = (struct rip_netinfo *)(rp + 1);
00221             for (; i >= sizeof(*ni); ++ni) {
00222                 if (rp->rip_vers == 1)
00223                     rip_entry_print_v1(ni);
00224                 else if (rp->rip_vers == 2)
00225                     rip_entry_print_v2(ni);
00226                                 else
00227                                     break;
00228                 i -= sizeof(*ni);
00229             }
00230             if (trunc)
00231                 printf("[|rip]");
00232             break;
00233 
00234         case RIPCMD_REQUEST:
00235         case RIPCMD_TRACEOFF:
00236         case RIPCMD_POLL:
00237         case RIPCMD_POLLENTRY:
00238             break;
00239 
00240         case RIPCMD_TRACEON:
00241                     /* fall through */
00242             default:
00243                     if (vflag <= 1) {
00244                         if(!print_unknown_data((u_int8_t *)rp,"\n\t",length))
00245                             return;
00246                     }
00247                     break;
00248                 }
00249                 /* do we want to see an additionally hexdump ? */
