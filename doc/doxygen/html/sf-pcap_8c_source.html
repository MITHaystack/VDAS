00126  * Check whether this is a pcap savefile and, if it is, extract the
00127  * relevant information from the header.

00131 {
00132     struct pcap_file_header hdr;
00133     size_t amt_read;
00134 
00135     /*
00136      * Check whether the first 4 bytes of the file are the magic
00137      * number for a pcap savefile, or for a byte-swapped pcap
00138      * savefile.
00139      */
00140     if (magic != TCPDUMP_MAGIC && magic != KUZNETZOV_TCPDUMP_MAGIC) {
00141         magic = SWAPLONG(magic);
00142         if (magic != TCPDUMP_MAGIC && magic != KUZNETZOV_TCPDUMP_MAGIC)
00143             return (0); /* nope */
00144         p->sf.swapped = 1;
00145     }
00146 
00147     /*
00148      * They are.  Put the magic number in the header, and read
00149      * the rest of the header.
00150      */
00151     hdr.magic = magic;
00152     amt_read = fread(((char *)&hdr) + sizeof hdr.magic, 1,
00153         sizeof(hdr) - sizeof(hdr.magic), fp);
00154     if (amt_read != sizeof(hdr) - sizeof(hdr.magic)) {
00155         if (ferror(fp)) {
00156             snprintf(errbuf, PCAP_ERRBUF_SIZE,
00157                 "error reading dump file: %s",
00158                 pcap_strerror(errno));
00159         } else {
00160             snprintf(errbuf, PCAP_ERRBUF_SIZE,
00161                 "truncated dump file; tried to read %lu file header bytes, only got %lu",
00162                 (unsigned long)sizeof(hdr),
00163                 (unsigned long)amt_read);
00164         }
00165         return (-1);
00166     }
00167 
00168     /*
00169      * If it's a byte-swapped capture file, byte-swap the header.
00170      */
00171     if (p->sf.swapped) {
00172         hdr.version_major = SWAPSHORT(hdr.version_major);
00173         hdr.version_minor = SWAPSHORT(hdr.version_minor);
00174         hdr.thiszone = SWAPLONG(hdr.thiszone);
00175         hdr.sigfigs = SWAPLONG(hdr.sigfigs);
00176         hdr.snaplen = SWAPLONG(hdr.snaplen);
00177         hdr.linktype = SWAPLONG(hdr.linktype);
00178     }
00179 
00180     if (hdr.version_major < PCAP_VERSION_MAJOR) {
00181         snprintf(errbuf, PCAP_ERRBUF_SIZE,
00182             "archaic pcap savefile format");
00183         return (-1);
00184     }
00185     p->sf.version_major = hdr.version_major;
00186     p->sf.version_minor = hdr.version_minor;
00187     p->tzoff = hdr.thiszone;
00188     p->snapshot = hdr.snaplen;
00189     p->linktype = linktype_to_dlt(LT_LINKTYPE(hdr.linktype));
00190     p->linktype_ext = LT_LINKTYPE_EXT(hdr.linktype);
00191 
00192     p->sf.next_packet_op = pcap_next_packet;
00193 
00194     /*
00195      * We interchanged the caplen and len fields at version 2.3,
00196      * in order to match the bpf header layout.  But unfortunately
00197      * some files were written with version 2.3 in their headers
00198      * but without the interchanged fields.
00199      *
00200      * In addition, DG/UX tcpdump writes out files with a version
00201      * number of 543.0, and with the caplen and len fields in the
00202      * pre-2.3 order.
00203      */
00204     switch (hdr.version_major) {
00205 
00206     case 2:
00207         if (hdr.version_minor < 3)
00208             p->sf.lengths_swapped = SWAPPED;
00209         else if (hdr.version_minor == 3)
00210             p->sf.lengths_swapped = MAYBE_SWAPPED;
00211         else
00212             p->sf.lengths_swapped = NOT_SWAPPED;
00213         break;
00214 
00215     case 543:
00216         p->sf.lengths_swapped = SWAPPED;
00217         break;
00218 
00219     default:
00220         p->sf.lengths_swapped = NOT_SWAPPED;
00221         break;
00222     }
00223 
00224     if (magic == KUZNETZOV_TCPDUMP_MAGIC) {
00225         /*
00226          * XXX - the patch that's in some versions of libpcap
00227          * changes the packet header but not the magic number,
00228          * and some other versions with this magic number have
00229          * some extra debugging information in the packet header;
00230          * we'd have to use some hacks^H^H^H^H^Hheuristics to
00231          * detect those variants.
00232          *
00233          * Ethereal does that, but it does so by trying to read
00234          * the first two packets of the file with each of the
00235          * record header formats.  That currently means it seeks
00236          * backwards and retries the reads, which doesn't work
00237          * on pipes.  We want to be able to read from a pipe, so
00238          * that strategy won't work; we'd have to buffer some
00239          * data ourselves and read from that buffer in order to
00240          * make that work.
00241          */
00242         p->sf.hdrsize = sizeof(struct pcap_sf_patched_pkthdr);
00243 
00244         if (p->linktype == DLT_EN10MB) {
00245             /*
00246              * This capture might have been done in raw mode
00247              * or cooked mode.
00248              *
00249              * If it was done in cooked mode, p->snapshot was
00250              * passed to recvfrom() as the buffer size, meaning
00251              * that the most packet data that would be copied
00252              * would be p->snapshot.  However, a faked Ethernet
00253              * header would then have been added to it, so the
00254              * most data that would be in a packet in the file
00255              * would be p->snapshot + 14.
00256              *
00257              * We can't easily tell whether the capture was done
00258              * in raw mode or cooked mode, so we'll assume it was
00259              * cooked mode, and add 14 to the snapshot length.
00260              * That means that, for a raw capture, the snapshot
00261              * length will be misleading if you use it to figure
00262              * out why a capture doesn't have all the packet data,
00263              * but there's not much we can do to avoid that.
00264              */
00265             p->snapshot += 14;
00266         }
00267     } else
00268         p->sf.hdrsize = sizeof(struct pcap_sf_pkthdr);
00269 
00270     /*
00271      * Allocate a buffer for the packet data.
00272      */
00273     p->bufsize = p->snapshot;
00274     if (p->bufsize <= 0)
00275         p->bufsize = BPF_MAXBUFSIZE;
00276     p->buffer = malloc(p->bufsize);
00277     if (p->buffer == NULL) {
00278         snprintf(errbuf, PCAP_ERRBUF_SIZE, "out of memory");
00279         return (-1);
00280     }
00281 
00282     return (1);
00283 }

00286  * Read and return the next packet from the savefile.  Return the header
00287  * in hdr and a pointer to the contents in data.  Return 0 on success, 1
00288  * if there were no more packets, and -1 on an error.
00289  */
00290 static int
00291 pcap_next_packet(pcap_t *p, struct pcap_pkthdr *hdr, u_char **data)
00292 {
00293     struct pcap_sf_patched_pkthdr sf_hdr;
00294     FILE *fp = p->sf.rfile;
00295     size_t amt_read;
00296     bpf_u_int32 t;
00297 
00298     /*
00299      * Read the packet header; the structure we use as a buffer
00300      * is the longer structure for files generated by the patched
00301      * libpcap, but if the file has the magic number for an
00302      * unpatched libpcap we only read as many bytes as the regular
00303      * header has.
00304      */
00305     amt_read = fread(&sf_hdr, 1, p->sf.hdrsize, fp);
00306     if (amt_read != p->sf.hdrsize) {
00307         if (ferror(fp)) {
00308             snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00309                 "error reading dump file: %s",
00310                 pcap_strerror(errno));
00311             return (-1);
00312         } else {
00313             if (amt_read != 0) {
00314                 snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00315                     "truncated dump file; tried to read %lu header bytes, only got %lu",
00316                     (unsigned long)p->sf.hdrsize,
00317                     (unsigned long)amt_read);
00318                 return (-1);
00319             }
00320             /* EOF */
00321             return (1);
00322         }
00323     }
00324 
00325     if (p->sf.swapped) {
00326         /* these were written in opposite byte order */
00327         hdr->caplen = SWAPLONG(sf_hdr.caplen);
00328         hdr->len = SWAPLONG(sf_hdr.len);
00329         hdr->ts.tv_sec = SWAPLONG(sf_hdr.ts.tv_sec);
00330         hdr->ts.tv_usec = SWAPLONG(sf_hdr.ts.tv_usec);
00331     } else {
00332         hdr->caplen = sf_hdr.caplen;
00333         hdr->len = sf_hdr.len;
00334         hdr->ts.tv_sec = sf_hdr.ts.tv_sec;
00335         hdr->ts.tv_usec = sf_hdr.ts.tv_usec;
00336     }
00337     /* Swap the caplen and len fields, if necessary. */
00338     switch (p->sf.lengths_swapped) {
00339 
00340     case NOT_SWAPPED:
00341         break;
00342 
00343     case MAYBE_SWAPPED:
00344         if (hdr->caplen <= hdr->len) {
00345             /*
00346              * The captured length is <= the actual length,
00347              * so presumably they weren't swapped.
00348              */
00349             break;
00350         }
00351         /* FALLTHROUGH */
00352 
00353     case SWAPPED:
00354         t = hdr->caplen;
00355         hdr->caplen = hdr->len;
00356         hdr->len = t;
00357         break;
00358     }
00359 
00360     if (hdr->caplen > p->bufsize) {
00361         /*
00362          * This can happen due to Solaris 2.3 systems tripping
00363          * over the BUFMOD problem and not setting the snapshot
00364          * correctly in the savefile header.  If the caplen isn't
00365          * grossly wrong, try to salvage.
00366          */
00367         static u_char *tp = NULL;
00368         static size_t tsize = 0;
00369 
00370         if (hdr->caplen > 65535) {
00371             snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00372                 "bogus savefile header");
00373             return (-1);
00374         }
00375 
00376         if (tsize < hdr->caplen) {
00377             tsize = ((hdr->caplen + 1023) / 1024) * 1024;
00378             if (tp != NULL)
00379                 free((u_char *)tp);
00380             tp = (u_char *)malloc(tsize);
00381             if (tp == NULL) {
00382                 tsize = 0;
00383                 snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00384                     "BUFMOD hack malloc");
00385                 return (-1);
00386             }
00387         }
00388         amt_read = fread((char *)tp, 1, hdr->caplen, fp);
00389         if (amt_read != hdr->caplen) {
00390             if (ferror(fp)) {
00391                 snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00392                     "error reading dump file: %s",
00393                     pcap_strerror(errno));
00394             } else {
00395                 snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00396                     "truncated dump file; tried to read %u captured bytes, only got %lu",
00397                     hdr->caplen, (unsigned long)amt_read);
00398             }
00399             return (-1);
00400         }
00401         /*
00402          * We can only keep up to p->bufsize bytes.  Since
00403          * caplen > p->bufsize is exactly how we got here,
00404          * we know we can only keep the first p->bufsize bytes
00405          * and must drop the remainder.  Adjust caplen accordingly,
00406          * so we don't get confused later as to how many bytes we
00407          * have to play with.
00408          */
00409         hdr->caplen = p->bufsize;
00410         memcpy(p->buffer, (char *)tp, p->bufsize);
00411     } else {
00412         /* read the packet itself */
00413         amt_read = fread(p->buffer, 1, hdr->caplen, fp);
00414         if (amt_read != hdr->caplen) {
00415             if (ferror(fp)) {
00416                 snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00417                     "error reading dump file: %s",
00418                     pcap_strerror(errno));
00419             } else {
00420                 snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00421                     "truncated dump file; tried to read %u captured bytes, only got %lu",
00422                     hdr->caplen, (unsigned long)amt_read);
00423             }
00424             return (-1);
00425         }
00426     }
00427     *data = p->buffer;
00428 
00429     if (p->sf.swapped) {
00430         /*
00431          * Convert pseudo-headers from the byte order of
00432          * the host on which the file was saved to our
00433          * byte order, as necessary.
00434          */
00435         switch (p->linktype) {
00436 
00437         case DLT_USB_LINUX:
00438             swap_linux_usb_header(hdr, *data, 0);
00439             break;
00440 
00441         case DLT_USB_LINUX_MMAPPED:
00442             swap_linux_usb_header(hdr, *data, 1);
00443             break;
00444         }
00445     }
00446

00448 }
