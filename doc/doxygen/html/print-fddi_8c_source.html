00053 /*
00054  * FDDI support for tcpdump, by Jeffrey Mogul [DECWRL], June 1992
00055  *
00056  * Based in part on code by Van Jacobson, which bears this note:
00057  *
00058  * NOTE:  This is a very preliminary hack for FDDI support.
00059  * There are all sorts of wired in constants & nothing (yet)
00060  * to print SMT packets as anything other than hex dumps.
00061  * Most of the necessary changes are waiting on my redoing
00062  * the "header" that a kernel fddi driver supplies to bpf:  I
00063  * want it to look like one byte of 'direction' (0 or 1
00064  * depending on whether the packet was inbound or outbound),
00065  * two bytes of system/driver dependent data (anything an
00066  * implementor thinks would be useful to filter on and/or
00067  * save per-packet, then the real 21-byte FDDI header.
00068  * Steve McCanne & I have also talked about adding the
00069  * 'direction' byte to all bpf headers (e.g., in the two
00070  * bytes of padding on an ethernet header).  It's not clear
00071  * we could do this in a backwards compatible way & we hate
00072  * the idea of an incompatible bpf change.  Discussions are
00073  * proceeding.
00074  *
00075  * Also, to really support FDDI (and better support 802.2
00076  * over ethernet) we really need to re-think the rather simple
00077  * minded assumptions about fixed length & fixed format link
00078  * level headers made in gencode.c.  One day...
00079  *
00080  *  - vj
00081  */

00122 print_fddi_fc(u_char fc)
00123 {
00124     switch (fc) {
00125 
00126     case FDDIFC_VOID:                         /* Void frame */
00127         printf("void ");
00128         break;
00129 
00130     case FDDIFC_NRT:                          /* Nonrestricted token */
00131         printf("nrt ");
00132         break;
00133 
00134     case FDDIFC_RT:                           /* Restricted token */
00135         printf("rt ");
00136         break;
00137 
00138     case FDDIFC_SMT_INFO:                     /* SMT Info */
00139         printf("info ");
00140         break;
00141 
00142     case FDDIFC_SMT_NSA:                      /* SMT Next station adrs */
00143         printf("nsa ");
00144         break;
00145 
00146     case FDDIFC_MAC_BEACON:                   /* MAC Beacon frame */
00147         printf("beacon ");
00148         break;
00149 
00150     case FDDIFC_MAC_CLAIM:                    /* MAC Claim frame */
00151         printf("claim ");
00152         break;
00153 
00154     default:
00155         switch (fc & FDDIFC_CLFF) {
00156 
00157         case FDDIFC_MAC:
00158             printf("mac%1x ", fc & FDDIFC_ZZZZ);
00159             break;
00160 
00161         case FDDIFC_SMT:
00162             printf("smt%1x ", fc & FDDIFC_ZZZZ);
00163             break;
00164 
00165         case FDDIFC_LLC_ASYNC:
00166             printf("async%1x ", fc & FDDIFC_ZZZZ);
00167             break;
00168 
00169         case FDDIFC_LLC_SYNC:
00170             printf("sync%1x ", fc & FDDIFC_ZZZZ);
00171             break;
00172 
00173         case FDDIFC_IMP_ASYNC:
00174             printf("imp_async%1x ", fc & FDDIFC_ZZZZ);
00175             break;
00176 
00177         case FDDIFC_IMP_SYNC:
00178             printf("imp_sync%1x ", fc & FDDIFC_ZZZZ);

00188 /* Extract src, dst addresses */
00189 static inline void
00190 extract_fddi_addrs(const struct fddi_header *fddip, char *fsrc, char *fdst)
00191 {
00192     register int i;
00193 
00194     if (fddi_bitswap) {
00195         /*
00196          * bit-swap the fddi addresses (isn't the IEEE standards
00197          * process wonderful!) then convert them to names.
00198          */
00199         for (i = 0; i < 6; ++i)
00200             fdst[i] = fddi_bit_swap[fddip->fddi_dhost[i]];
00201         for (i = 0; i < 6; ++i)
00202             fsrc[i] = fddi_bit_swap[fddip->fddi_shost[i]];
00203     }
00204     else {
00205         memcpy(fdst, (const char *)fddip->fddi_dhost, 6);
00206         memcpy(fsrc, (const char *)fddip->fddi_shost, 6);
00207     }
00208 }

00241 void
00242 fddi_print(const u_char *p, u_int length, u_int caplen)
00243 {
00244     const struct fddi_header *fddip = (const struct fddi_header *)p;
00245     struct ether_header ehdr;
00246     u_short extracted_ethertype;
00247 
00248     if (caplen < FDDI_HDRLEN) {
00249         printf("[|fddi]");
00250         return;
00251     }
00252 
00253     /*
00254      * Get the FDDI addresses into a canonical form
00255      */
00256     extract_fddi_addrs(fddip, (char *)ESRC(&ehdr), (char *)EDST(&ehdr));
00257 
00258     if (eflag)
00259         fddi_hdr_print(fddip, length, ESRC(&ehdr), EDST(&ehdr));
00260 
00261     /* Skip over FDDI MAC header */
00262     length -= FDDI_HDRLEN;
00263     p += FDDI_HDRLEN;
00264     caplen -= FDDI_HDRLEN;
00265 
00266     /* Frame Control field determines interpretation of packet */
00267     if ((fddip->fddi_fc & FDDIFC_CLFF) == FDDIFC_LLC_ASYNC) {
00268         /* Try to print the LLC-layer header & higher layers */
00269         if (llc_print(p, length, caplen, ESRC(&ehdr), EDST(&ehdr),
00270             &extracted_ethertype) == 0) {
00271             /*
00272              * Some kinds of LLC packet we cannot
00273              * handle intelligently
00274              */
00275             if (!eflag)
00276                 fddi_hdr_print(fddip, length + FDDI_HDRLEN,
00277                     ESRC(&ehdr), EDST(&ehdr));
00278             if (extracted_ethertype) {
00279                 printf("(LLC %s) ",
00280             etherproto_string(htons(extracted_ethertype)));
00281             }
00282             if (!suppress_default_print)
00283                 default_print(p, caplen);
00284         }
00285     } else if ((fddip->fddi_fc & FDDIFC_CLFF) == FDDIFC_SMT)
00286         fddi_smt_print(p, caplen);
00287     else {
00288         /* Some kinds of FDDI packet we cannot handle intelligently */
00289         if (!eflag)
00290             fddi_hdr_print(fddip, length + FDDI_HDRLEN, ESRC(&ehdr),
00291                 EDST(&ehdr));
00292         if (!suppress_default_print)
00293             default_print(p, caplen);
00294     }
00295 }
