00079 void ixgbe_set_pcie_completion_timeout(struct ixgbe_hw *hw)
00080 {
00081     u32 gcr = IXGBE_READ_REG(hw, IXGBE_GCR);
00082     u16 pcie_devctl2;
00083 
00084     /* only take action if timeout value is defaulted to 0 */
00085     if (gcr & IXGBE_GCR_CMPL_TMOUT_MASK)
00086         goto out;
00087 
00088     /*
00089      * if capababilities version is type 1 we can write the
00090      * timeout of 10ms to 250ms through the GCR register
00091      */
00092     if (!(gcr & IXGBE_GCR_CAP_VER2)) {
00093         gcr |= IXGBE_GCR_CMPL_TMOUT_10ms;
00094         goto out;
00095     }
00096 
00097     /*
00098      * for version 2 capabilities we need to write the config space
00099      * directly in order to set the completion timeout value for
00100      * 16ms to 55ms
00101      */
00102     pcie_devctl2 = IXGBE_READ_PCIE_WORD(hw, IXGBE_PCI_DEVICE_CONTROL2);
00103     pcie_devctl2 |= IXGBE_PCI_DEVICE_CONTROL2_16ms;
00104     IXGBE_WRITE_PCIE_WORD(hw, IXGBE_PCI_DEVICE_CONTROL2, pcie_devctl2);
00105 out:
00106     /* disable completion timeout resend */
00107     gcr &= ~IXGBE_GCR_CMPL_TMOUT_RESEND;
00108     IXGBE_WRITE_REG(hw, IXGBE_GCR, gcr);
00109 }

00267 s32 ixgbe_start_hw_82598(struct ixgbe_hw *hw)
00268 {
00269     u32 regval;
00270     u32 i;
00271     s32 ret_val = 0;
00272 
00273     ret_val = ixgbe_start_hw_generic(hw);
00274 
00275     /* Disable relaxed ordering */
00276     for (i = 0; ((i < hw->mac.max_tx_queues) &&
00277          (i < IXGBE_DCA_MAX_QUEUES_82598)); i++) {
00278         regval = IXGBE_READ_REG(hw, IXGBE_DCA_TXCTRL(i));
00279         regval &= ~IXGBE_DCA_TXCTRL_TX_WB_RO_EN;
00280         IXGBE_WRITE_REG(hw, IXGBE_DCA_TXCTRL(i), regval);
00281     }
00282 
00283     for (i = 0; ((i < hw->mac.max_rx_queues) &&
00284          (i < IXGBE_DCA_MAX_QUEUES_82598)); i++) {
00285         regval = IXGBE_READ_REG(hw, IXGBE_DCA_RXCTRL(i));
00286         regval &= ~(IXGBE_DCA_RXCTRL_DESC_WRO_EN |
00287                     IXGBE_DCA_RXCTRL_DESC_HSRO_EN);
00288         IXGBE_WRITE_REG(hw, IXGBE_DCA_RXCTRL(i), regval);
00289     }
00290 
00291     /* set the completion timeout for interface */
00292     if (ret_val == 0)
00293         ixgbe_set_pcie_completion_timeout(hw);
00294 
00295     return ret_val;
00296 }

00306 static s32 ixgbe_get_link_capabilities_82598(struct ixgbe_hw *hw,
00307                                              ixgbe_link_speed *speed,
00308                                              bool *autoneg)
00309 {
00310     s32 status = 0;
00311     u32 autoc = 0;
00312 
00313     /*
00314      * Determine link capabilities based on the stored value of AUTOC,
00315      * which represents EEPROM defaults.  If AUTOC value has not been
00316      * stored, use the current register value.
00317      */
00318     if (hw->mac.orig_link_settings_stored)
00319         autoc = hw->mac.orig_autoc;
00320     else
00321         autoc = IXGBE_READ_REG(hw, IXGBE_AUTOC);
00322 
00323     switch (autoc & IXGBE_AUTOC_LMS_MASK) {
00324     case IXGBE_AUTOC_LMS_1G_LINK_NO_AN:
00325         *speed = IXGBE_LINK_SPEED_1GB_FULL;
00326         *autoneg = false;
00327         break;
00328 
00329     case IXGBE_AUTOC_LMS_10G_LINK_NO_AN:
00330         *speed = IXGBE_LINK_SPEED_10GB_FULL;
00331         *autoneg = false;
00332         break;
00333 
00334     case IXGBE_AUTOC_LMS_1G_AN:
00335         *speed = IXGBE_LINK_SPEED_1GB_FULL;
00336         *autoneg = true;
00337         break;
00338 
00339     case IXGBE_AUTOC_LMS_KX4_AN:
00340     case IXGBE_AUTOC_LMS_KX4_AN_1G_AN:
00341         *speed = IXGBE_LINK_SPEED_UNKNOWN;
00342         if (autoc & IXGBE_AUTOC_KX4_SUPP)
00343             *speed |= IXGBE_LINK_SPEED_10GB_FULL;
00344         if (autoc & IXGBE_AUTOC_KX_SUPP)
00345             *speed |= IXGBE_LINK_SPEED_1GB_FULL;
00346         *autoneg = true;
00347         break;
00348 
00349     default:
00350         status = IXGBE_ERR_LINK_SETUP;
00351         break;
00352     }
00353 
00354     return status;
00355 }

00390     case IXGBE_DEV_ID_82598EB_SFP_LOM:
00391         media_type = ixgbe_media_type_fiber;
00392         break;
00393     case IXGBE_DEV_ID_82598EB_CX4:
00394     case IXGBE_DEV_ID_82598_CX4_DUAL_PORT:
00395         media_type = ixgbe_media_type_cx4;
00396         break;
00397     case IXGBE_DEV_ID_82598AT:
00398     case IXGBE_DEV_ID_82598AT2:

00416 s32 ixgbe_fc_enable_82598(struct ixgbe_hw *hw, s32 packetbuf_num)
00417 {
00418     s32 ret_val = 0;
00419     u32 fctrl_reg;
00420     u32 rmcs_reg;
00421     u32 reg;
00422     u32 rx_pba_size;
00423     u32 link_speed = 0;
00424     bool link_up;
00425 
00426 #ifdef CONFIG_DCB
00427     if (hw->fc.requested_mode == ixgbe_fc_pfc)
00428         goto out;
00429 
00430 #endif /* CONFIG_DCB */
00431     /*
00432      * On 82598 having Rx FC on causes resets while doing 1G
00433      * so if it's on turn it off once we know link_speed. For
00434      * more details see 82598 Specification update.
00435      */
00436     hw->mac.ops.check_link(hw, &link_speed, &link_up, false);
00437     if (link_up && link_speed == IXGBE_LINK_SPEED_1GB_FULL) {
00438         switch (hw->fc.requested_mode) {
00439         case ixgbe_fc_full:
00440             hw->fc.requested_mode = ixgbe_fc_tx_pause;
00441             break;
00442         case ixgbe_fc_rx_pause:
00443             hw->fc.requested_mode = ixgbe_fc_none;
00444             break;
00445         default:
00446             /* no change */
00447             break;
00448         }
00449     }
00450 
00451     /* Negotiate the fc mode to use */
00452     ret_val = ixgbe_fc_autoneg(hw);
00453     if (ret_val == IXGBE_ERR_FLOW_CONTROL)
00454         goto out;
00455 
00456     /* Disable any previous flow control settings */
00457     fctrl_reg = IXGBE_READ_REG(hw, IXGBE_FCTRL);
00458     fctrl_reg &= ~(IXGBE_FCTRL_RFCE | IXGBE_FCTRL_RPFCE);
00459 
00460     rmcs_reg = IXGBE_READ_REG(hw, IXGBE_RMCS);
00461     rmcs_reg &= ~(IXGBE_RMCS_TFCE_PRIORITY | IXGBE_RMCS_TFCE_802_3X);
00462 
00463     /*
00464      * The possible values of fc.current_mode are:
00465      * 0: Flow control is completely disabled
00466      * 1: Rx flow control is enabled (we can receive pause frames,
00467      *    but not send pause frames).
00468      * 2: Tx flow control is enabled (we can send pause frames but
00469      *     we do not support receiving pause frames).
00470      * 3: Both Rx and Tx flow control (symmetric) are enabled.
00471 #ifdef CONFIG_DCB
00472      * 4: Priority Flow Control is enabled.
00473 #endif
00474      * other: Invalid.
00475      */
00476     switch (hw->fc.current_mode) {
00477     case ixgbe_fc_none:
00478         /*
00479          * Flow control is disabled by software override or autoneg.
00480          * The code below will actually disable it in the HW.
00481          */
00482         break;
00483     case ixgbe_fc_rx_pause:
00484         /*
00485          * Rx Flow control is enabled and Tx Flow control is
00486          * disabled by software override. Since there really
00487          * isn't a way to advertise that we are capable of RX
00488          * Pause ONLY, we will advertise that we support both
00489          * symmetric and asymmetric Rx PAUSE.  Later, we will
00490          * disable the adapter's ability to send PAUSE frames.
00491          */
00492         fctrl_reg |= IXGBE_FCTRL_RFCE;
00493         break;
00494     case ixgbe_fc_tx_pause:
00495         /*
00496          * Tx Flow control is enabled, and Rx Flow control is
00497          * disabled by software override.
00498          */
00499         rmcs_reg |= IXGBE_RMCS_TFCE_802_3X;
00500         break;
00501     case ixgbe_fc_full:
00502         /* Flow control (both Rx and Tx) is enabled by SW override. */
00503         fctrl_reg |= IXGBE_FCTRL_RFCE;
00504         rmcs_reg |= IXGBE_RMCS_TFCE_802_3X;
00505         break;
00506 #ifdef CONFIG_DCB
00507     case ixgbe_fc_pfc:
00508         goto out;
00509         break;
00510 #endif /* CONFIG_DCB */
00511     default:
00512         hw_dbg(hw, "Flow control param set incorrectly\n");
00513         ret_val = IXGBE_ERR_CONFIG;
00514         goto out;
00515         break;
00516     }
00517 
00518     /* Set 802.3x based flow control settings. */
00519     fctrl_reg |= IXGBE_FCTRL_DPF;
00520     IXGBE_WRITE_REG(hw, IXGBE_FCTRL, fctrl_reg);
00521     IXGBE_WRITE_REG(hw, IXGBE_RMCS, rmcs_reg);
00522 
00523     /* Set up and enable Rx high/low water mark thresholds, enable XON. */
00524     if (hw->fc.current_mode & ixgbe_fc_tx_pause) {
00525         rx_pba_size = IXGBE_READ_REG(hw, IXGBE_RXPBSIZE(packetbuf_num));
00526         rx_pba_size >>= IXGBE_RXPBSIZE_SHIFT;
00527 
00528         reg = (rx_pba_size - hw->fc.low_water) << 6;
00529         if (hw->fc.send_xon)
00530             reg |= IXGBE_FCRTL_XONE;
00531 
00532         IXGBE_WRITE_REG(hw, IXGBE_FCRTL(packetbuf_num), reg);
00533 
00534         reg = (rx_pba_size - hw->fc.high_water) << 6;
00535         reg |= IXGBE_FCRTH_FCEN;
00536 
00537         IXGBE_WRITE_REG(hw, IXGBE_FCRTH(packetbuf_num), reg);
00538     }
00539 
00540     /* Configure pause time (2 TCs per register) */
00541     reg = IXGBE_READ_REG(hw, IXGBE_FCTTV(packetbuf_num / 2));
00542     if ((packetbuf_num & 1) == 0)
00543         reg = (reg & 0xFFFF0000) | hw->fc.pause_time;
00544     else
00545         reg = (reg & 0x0000FFFF) | (hw->fc.pause_time << 16);
00546     IXGBE_WRITE_REG(hw, IXGBE_FCTTV(packetbuf_num / 2), reg);
00547 
00548     IXGBE_WRITE_REG(hw, IXGBE_FCRTV, (hw->fc.pause_time >> 1));
00549 
00550 out:
00551     return ret_val;
00552 }

00561 static s32 ixgbe_start_mac_link_82598(struct ixgbe_hw *hw,
00562                                       bool autoneg_wait_to_complete)
00563 {
00564     u32 autoc_reg;
00565     u32 links_reg;
00566     u32 i;
00567     s32 status = 0;
00568 
00569     /* Restart link */
00570     autoc_reg = IXGBE_READ_REG(hw, IXGBE_AUTOC);
00571     autoc_reg |= IXGBE_AUTOC_AN_RESTART;
00572     IXGBE_WRITE_REG(hw, IXGBE_AUTOC, autoc_reg);
00573 
00574     /* Only poll for autoneg to complete if specified to do so */
00575     if (autoneg_wait_to_complete) {
00576         if ((autoc_reg & IXGBE_AUTOC_LMS_MASK) ==
00577              IXGBE_AUTOC_LMS_KX4_AN ||
00578             (autoc_reg & IXGBE_AUTOC_LMS_MASK) ==
00579              IXGBE_AUTOC_LMS_KX4_AN_1G_AN) {
00580             links_reg = 0; /* Just in case Autoneg time = 0 */
00581             for (i = 0; i < IXGBE_AUTO_NEG_TIME; i++) {
00582                 links_reg = IXGBE_READ_REG(hw, IXGBE_LINKS);
00583                 if (links_reg & IXGBE_LINKS_KX_AN_COMP)
00584                     break;
00585                 msleep(100);
00586             }
00587             if (!(links_reg & IXGBE_LINKS_KX_AN_COMP)) {

00607 static s32 ixgbe_validate_link_ready(struct ixgbe_hw *hw)
00608 {
00609     u32 timeout;
00610     u16 an_reg;
00611 
00612     if (hw->device_id != IXGBE_DEV_ID_82598AT2)
00613         return 0;
00614 
00615     for (timeout = 0;
00616          timeout < IXGBE_VALIDATE_LINK_READY_TIMEOUT; timeout++) {
00617         hw->phy.ops.read_reg(hw, IXGBE_MDIO_AUTO_NEG_STATUS,
00618                              IXGBE_MDIO_AUTO_NEG_DEV_TYPE, &an_reg);

00621             (an_reg & IXGBE_MII_AUTONEG_LINK_UP))
00622             break;

00644 static s32 ixgbe_check_mac_link_82598(struct ixgbe_hw *hw,
00645                                       ixgbe_link_speed *speed, bool *link_up,
00646                                       bool link_up_wait_to_complete)
00647 {
00648     u32 links_reg;
00649     u32 i;
00650     u16 link_reg, adapt_comp_reg;
00651 
00652     /*
00653      * SERDES PHY requires us to read link status from undocumented
00654      * register 0xC79F.  Bit 0 set indicates link is up/ready; clear
00655      * indicates link down.  OxC00C is read to check that the XAUI lanes
00656      * are active.  Bit 0 clear indicates active; set indicates inactive.
00657      */
00658     if (hw->phy.type == ixgbe_phy_nl) {
00659         hw->phy.ops.read_reg(hw, 0xC79F, IXGBE_TWINAX_DEV, &link_reg);
00660         hw->phy.ops.read_reg(hw, 0xC79F, IXGBE_TWINAX_DEV, &link_reg);
00661         hw->phy.ops.read_reg(hw, 0xC00C, IXGBE_TWINAX_DEV,
00662                              &adapt_comp_reg);
00663         if (link_up_wait_to_complete) {
00664             for (i = 0; i < IXGBE_LINK_UP_TIME; i++) {
00665                 if ((link_reg & 1) &&
00666                     ((adapt_comp_reg & 1) == 0)) {
00667                     *link_up = true;
00668                     break;
00669                 } else {
00670                     *link_up = false;
00671                 }
00672                 msleep(100);
00673                 hw->phy.ops.read_reg(hw, 0xC79F,
00674                                      IXGBE_TWINAX_DEV,
00675                                      &link_reg);
00676                 hw->phy.ops.read_reg(hw, 0xC00C,
00677                                      IXGBE_TWINAX_DEV,
00678                                      &adapt_comp_reg);
00679             }
00680         } else {
00681             if ((link_reg & 1) && ((adapt_comp_reg & 1) == 0))
00682                 *link_up = true;
00683             else
00684                 *link_up = false;
00685         }
00686 
00687         if (*link_up == false)
00688             goto out;
00689     }
00690 
00691     links_reg = IXGBE_READ_REG(hw, IXGBE_LINKS);
00692     if (link_up_wait_to_complete) {
00693         for (i = 0; i < IXGBE_LINK_UP_TIME; i++) {
00694             if (links_reg & IXGBE_LINKS_UP) {
00695                 *link_up = true;
00696                 break;
00697             } else {
00698                 *link_up = false;
00699             }
00700             msleep(100);
00701             links_reg = IXGBE_READ_REG(hw, IXGBE_LINKS);
00702         }
00703     } else {
00704         if (links_reg & IXGBE_LINKS_UP)
00705             *link_up = true;
00706         else
00707             *link_up = false;
00708     }
00709 
00710     if (links_reg & IXGBE_LINKS_SPEED)
00711         *speed = IXGBE_LINK_SPEED_10GB_FULL;
00712     else
00713         *speed = IXGBE_LINK_SPEED_1GB_FULL;
00714 
00715     if ((hw->device_id == IXGBE_DEV_ID_82598AT2) && (*link_up == true) &&
00716         (ixgbe_validate_link_ready(hw) != 0))
00717         *link_up = false;
00718 
00719     /* if link is down, zero out the current_mode */
00720     if (*link_up == false) {
00721         hw->fc.current_mode = ixgbe_fc_none;
00722         hw->fc.fc_was_autonegged = false;
00723     }
00724 out:
00725     return 0;
00726 }

00813 static s32 ixgbe_reset_hw_82598(struct ixgbe_hw *hw)
00814 {
00815     s32 status = 0;
00816     s32 phy_status = 0;
00817     u32 ctrl;
00818     u32 gheccr;
00819     u32 i;
00820     u32 autoc;
00821     u8  analog_val;
00822 
00823     /* Call adapter stop to disable tx/rx and clear interrupts */
00824     hw->mac.ops.stop_adapter(hw);
00825 
00826     /*
00827      * Power up the Atlas Tx lanes if they are currently powered down.
00828      * Atlas Tx lanes are powered down for MAC loopback tests, but
00829      * they are not automatically restored on reset.
00830      */
00831     hw->mac.ops.read_analog_reg8(hw, IXGBE_ATLAS_PDN_LPBK, &analog_val);
00832     if (analog_val & IXGBE_ATLAS_PDN_TX_REG_EN) {
00833         /* Enable Tx Atlas so packets can be transmitted again */
00834         hw->mac.ops.read_analog_reg8(hw, IXGBE_ATLAS_PDN_LPBK,
00835                                      &analog_val);
00836         analog_val &= ~IXGBE_ATLAS_PDN_TX_REG_EN;
00837         hw->mac.ops.write_analog_reg8(hw, IXGBE_ATLAS_PDN_LPBK,
00838                                       analog_val);
00839 
00840         hw->mac.ops.read_analog_reg8(hw, IXGBE_ATLAS_PDN_10G,
00841                                      &analog_val);
00842         analog_val &= ~IXGBE_ATLAS_PDN_TX_10G_QL_ALL;
00843         hw->mac.ops.write_analog_reg8(hw, IXGBE_ATLAS_PDN_10G,
00844                                       analog_val);
00845 
00846         hw->mac.ops.read_analog_reg8(hw, IXGBE_ATLAS_PDN_1G,
00847                                      &analog_val);
00848         analog_val &= ~IXGBE_ATLAS_PDN_TX_1G_QL_ALL;
00849         hw->mac.ops.write_analog_reg8(hw, IXGBE_ATLAS_PDN_1G,
00850                                       analog_val);
00851 
00852         hw->mac.ops.read_analog_reg8(hw, IXGBE_ATLAS_PDN_AN,
00853                                      &analog_val);
00854         analog_val &= ~IXGBE_ATLAS_PDN_TX_AN_QL_ALL;
00855         hw->mac.ops.write_analog_reg8(hw, IXGBE_ATLAS_PDN_AN,
00856                                       analog_val);
00857     }
00858 
00859     /* Reset PHY */
00860     if (hw->phy.reset_disable == false) {
00861         /* PHY ops must be identified and initialized prior to reset */
00862 
00863         /* Init PHY and function pointers, perform SFP setup */
00864         phy_status = hw->phy.ops.init(hw);
00865         if (phy_status == IXGBE_ERR_SFP_NOT_SUPPORTED)
00866             goto reset_hw_out;
00867         else if (phy_status == IXGBE_ERR_SFP_NOT_PRESENT)
00868             goto no_phy_reset;
00869 
00870         hw->phy.ops.reset(hw);
00871     }
00872 
00873 no_phy_reset:
00874     /*
00875      * Prevent the PCI-E bus from from hanging by disabling PCI-E master
00876      * access and verify no pending requests before reset
00877      */
00878     ixgbe_disable_pcie_master(hw);
00879 
00880 mac_reset_top:
00881     /*
00882      * Issue global reset to the MAC.  This needs to be a SW reset.
00883      * If link reset is used, it might reset the MAC when mng is using it
00884      */
00885     ctrl = IXGBE_READ_REG(hw, IXGBE_CTRL);
00886     IXGBE_WRITE_REG(hw, IXGBE_CTRL, (ctrl | IXGBE_CTRL_RST));
00887     IXGBE_WRITE_FLUSH(hw);
00888 
00889     /* Poll for reset bit to self-clear indicating reset is complete */
00890     for (i = 0; i < 10; i++) {
00891         udelay(1);
00892         ctrl = IXGBE_READ_REG(hw, IXGBE_CTRL);
00893         if (!(ctrl & IXGBE_CTRL_RST))
00894             break;
00895     }
00896     if (ctrl & IXGBE_CTRL_RST) {
00897         status = IXGBE_ERR_RESET_FAILED;
00898         hw_dbg(hw, "Reset polling failed to complete.\n");
00899     }
00900 
00901     /*
00902      * Double resets are required for recovery from certain error
00903      * conditions.  Between resets, it is necessary to stall to allow time
00904      * for any pending HW events to complete.  We use 1usec since that is
00905      * what is needed for ixgbe_disable_pcie_master().  The second reset
00906      * then clears out any effects of those events.
00907      */
00908     if (hw->mac.flags & IXGBE_FLAGS_DOUBLE_RESET_REQUIRED) {
00909         hw->mac.flags &= ~IXGBE_FLAGS_DOUBLE_RESET_REQUIRED;
00910         udelay(1);
00911         goto mac_reset_top;
00912     }
00913 
00914     msleep(50);
00915 
00916     gheccr = IXGBE_READ_REG(hw, IXGBE_GHECCR);
00917     gheccr &= ~((1 << 21) | (1 << 18) | (1 << 9) | (1 << 6));
00918     IXGBE_WRITE_REG(hw, IXGBE_GHECCR, gheccr);
00919 
00920     /*
00921      * Store the original AUTOC value if it has not been
00922      * stored off yet.  Otherwise restore the stored original
00923      * AUTOC value since the reset operation sets back to deaults.
00924      */
00925     autoc = IXGBE_READ_REG(hw, IXGBE_AUTOC);
00926     if (hw->mac.orig_link_settings_stored == false) {
00927         hw->mac.orig_autoc = autoc;
00928         hw->mac.orig_link_settings_stored = true;
00929     } else if (autoc != hw->mac.orig_autoc) {
00930         IXGBE_WRITE_REG(hw, IXGBE_AUTOC, hw->mac.orig_autoc);
00931     }
00932 
00933     /* Store the permanent mac address */
00934     hw->mac.ops.get_mac_addr(hw, hw->mac.perm_addr);
00935 
00936     /*
00937      * Store MAC address from RAR0, clear receive address registers, and
00938      * clear the multicast table
00939      */
00940     hw->mac.ops.init_rx_addrs(hw);
00941 
00942 reset_hw_out:
00943     if (phy_status != 0)
00944         status = phy_status;
00945 
00946     return status;
00947 }

01009 s32 ixgbe_set_vfta_82598(struct ixgbe_hw *hw, u32 vlan, u32 vind,
01010                                                   bool vlan_on)
01011 {
01012     u32 regindex;
01013     u32 bitindex;
01014     u32 bits;
01015     u32 vftabyte;
01016 
01017     if (vlan > 4095)
01018         return IXGBE_ERR_PARAM;
01019 
01020     /* Determine 32-bit word position in array */
01021     regindex = (vlan >> 5) & 0x7F;   /* upper seven bits */
01022 
01023     /* Determine the location of the (VMD) queue index */
01024     vftabyte =  ((vlan >> 3) & 0x03); /* bits (4:3) indicating byte array */
01025     bitindex = (vlan & 0x7) << 2;    /* lower 3 bits indicate nibble */
01026

01028     bits = IXGBE_READ_REG(hw, IXGBE_VFTAVIND(vftabyte, regindex));
01029     bits &= (~(0x0F << bitindex));
01030     bits |= (vind << bitindex);
01031     IXGBE_WRITE_REG(hw, IXGBE_VFTAVIND(vftabyte, regindex), bits);
01032 
01033     /* Determine the location of the bit for this VLAN id */
01034     bitindex = vlan & 0x1F;   /* lower five bits */
01035 
01036     bits = IXGBE_READ_REG(hw, IXGBE_VFTA(regindex));
01037     if (vlan_on)
01038         /* Turn on this VLAN id */
01039         bits |= (1 << bitindex);
01040     else
01041         /* Turn off this VLAN id */
01042         bits &= ~(1 << bitindex);
01043     IXGBE_WRITE_REG(hw, IXGBE_VFTA(regindex), bits);
01044 
01045     return 0;
01046 }

01120 s32 ixgbe_read_i2c_eeprom_82598(struct ixgbe_hw *hw, u8 byte_offset,
01121                                 u8 *eeprom_data)
01122 {
01123     s32 status = 0;
01124     u16 sfp_addr = 0;
01125     u16 sfp_data = 0;
01126     u16 sfp_stat = 0;
01127     u32 i;
01128 
01129     if (hw->phy.type == ixgbe_phy_nl) {
01130         /*
01131          * NetLogic phy SDA/SCL registers are at addresses 0xC30A to
01132          * 0xC30D. These registers are used to talk to the SFP+
01133          * module's EEPROM through the SDA/SCL (I2C) interface.
01134          */
01135         sfp_addr = (IXGBE_I2C_EEPROM_DEV_ADDR << 8) + byte_offset;
01136         sfp_addr = (sfp_addr | IXGBE_I2C_EEPROM_READ_MASK);

01139                               IXGBE_MDIO_PMA_PMD_DEV_TYPE,
01140                               sfp_addr);
01141 
01142         /* Poll status */
01143         for (i = 0; i < 100; i++) {
01144             hw->phy.ops.read_reg(hw,
01145                                  IXGBE_MDIO_PMA_PMD_SDA_SCL_STAT,
01146                                  IXGBE_MDIO_PMA_PMD_DEV_TYPE,
01147                                  &sfp_stat);
01148             sfp_stat = sfp_stat & IXGBE_I2C_EEPROM_STATUS_MASK;
01149             if (sfp_stat != IXGBE_I2C_EEPROM_STATUS_IN_PROGRESS)
01150                 break;
01151             msleep(10);
01152         }
01153 
01154         if (sfp_stat != IXGBE_I2C_EEPROM_STATUS_PASS) {
01155             hw_dbg(hw, "EEPROM read did not pass.\n");
01156             status = IXGBE_ERR_SFP_NOT_PRESENT;
01157             goto out;
01158         }
01159 
01160         /* Read data */
01161         hw->phy.ops.read_reg(hw, IXGBE_MDIO_PMA_PMD_SDA_SCL_DATA,
01162                              IXGBE_MDIO_PMA_PMD_DEV_TYPE, &sfp_data);
01163 
01164         *eeprom_data = (u8)(sfp_data >> 8);
01165     } else {
01166         status = IXGBE_ERR_PHY;
01167         goto out;
01168     }
01169 
01170 out:
01171     return status;
01172 }

01180 u32 ixgbe_get_supported_physical_layer_82598(struct ixgbe_hw *hw)
01181 {
01182     u32 physical_layer = IXGBE_PHYSICAL_LAYER_UNKNOWN;
01183     u32 autoc = IXGBE_READ_REG(hw, IXGBE_AUTOC);
01184     u32 pma_pmd_10g = autoc & IXGBE_AUTOC_10G_PMA_PMD_MASK;
01185     u32 pma_pmd_1g = autoc & IXGBE_AUTOC_1G_PMA_PMD_MASK;
01186     u16 ext_ability = 0;
01187 
01188     hw->phy.ops.identify(hw);
01189 
01190     /* Copper PHY must be checked before AUTOC LMS to determine correct
01191      * physical layer because 10GBase-T PHYs use LMS = KX4/KX */
01192     switch (hw->phy.type) {
01193     case ixgbe_phy_tn:
01194     case ixgbe_phy_aq:
01195     case ixgbe_phy_cu_unknown:
01196         hw->phy.ops.read_reg(hw, IXGBE_MDIO_PHY_EXT_ABILITY,
01197         IXGBE_MDIO_PMA_PMD_DEV_TYPE, &ext_ability);
01198         if (ext_ability & IXGBE_MDIO_PHY_10GBASET_ABILITY)
01199             physical_layer |= IXGBE_PHYSICAL_LAYER_10GBASE_T;
01200         if (ext_ability & IXGBE_MDIO_PHY_1000BASET_ABILITY)
01201             physical_layer |= IXGBE_PHYSICAL_LAYER_1000BASE_T;
01202         if (ext_ability & IXGBE_MDIO_PHY_100BASETX_ABILITY)
01203             physical_layer |= IXGBE_PHYSICAL_LAYER_100BASE_TX;
01204         goto out;
01205     default:
01206         break;
01207     }
01208 
01209     switch (autoc & IXGBE_AUTOC_LMS_MASK) {
01210     case IXGBE_AUTOC_LMS_1G_AN:
01211     case IXGBE_AUTOC_LMS_1G_LINK_NO_AN:
01212         if (pma_pmd_1g == IXGBE_AUTOC_1G_KX)
01213             physical_layer = IXGBE_PHYSICAL_LAYER_1000BASE_KX;
01214         else
01215             physical_layer = IXGBE_PHYSICAL_LAYER_1000BASE_BX;
01216         break;
01217     case IXGBE_AUTOC_LMS_10G_LINK_NO_AN:
01218         if (pma_pmd_10g == IXGBE_AUTOC_10G_CX4)
01219             physical_layer = IXGBE_PHYSICAL_LAYER_10GBASE_CX4;
01220         else if (pma_pmd_10g == IXGBE_AUTOC_10G_KX4)
01221             physical_layer = IXGBE_PHYSICAL_LAYER_10GBASE_KX4;
01222         else /* XAUI */
01223             physical_layer = IXGBE_PHYSICAL_LAYER_UNKNOWN;
01224         break;
