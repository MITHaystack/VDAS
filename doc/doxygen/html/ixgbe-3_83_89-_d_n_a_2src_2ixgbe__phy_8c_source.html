00207 {
00208     u32 i;
00209     u16 ctrl = 0;
00210     s32 status = 0;
00211 
00212     if (hw->phy.type == ixgbe_phy_unknown)
00213         status = ixgbe_identify_phy_generic(hw);
00214 
00215     if (status != 0 || hw->phy.type == ixgbe_phy_none)
00216         goto out;
00217 
00218     /* Don't reset PHY if it's shut down due to overtemp. */
00219     if (!hw->phy.reset_if_overtemp &&
00220         (IXGBE_ERR_OVERTEMP == hw->phy.ops.check_overtemp(hw)))
00221         goto out;
00222 
00223     /*
00224      * Perform soft PHY reset to the PHY_XS.
00225      * This will cause a soft reset to the PHY
00226      */
00227     hw->phy.ops.write_reg(hw, IXGBE_MDIO_PHY_XS_CONTROL,
00228                           IXGBE_MDIO_PHY_XS_DEV_TYPE,
00229                           IXGBE_MDIO_PHY_XS_RESET);
00230 
00231     /*
00232      * Poll for reset bit to self-clear indicating reset is complete.
00233      * Some PHYs could take up to 3 seconds to complete and need about
00234      * 1.7 usec delay after the reset is complete.
00235      */
00236     for (i = 0; i < 30; i++) {
00237         msleep(100);
00238         hw->phy.ops.read_reg(hw, IXGBE_MDIO_PHY_XS_CONTROL,
00239                              IXGBE_MDIO_PHY_XS_DEV_TYPE, &ctrl);
00240         if (!(ctrl & IXGBE_MDIO_PHY_XS_RESET)) {
00241             udelay(2);
00242             break;
00243         }
00244     }
00245 
00246     if (ctrl & IXGBE_MDIO_PHY_XS_RESET) {
00247         status = IXGBE_ERR_RESET_FAILED;
00248         hw_dbg(hw, "PHY reset polling failed to complete.\n");
00249     }
00250 
00251 out:
00252     return status;
00253 }

00312                        (device_type << IXGBE_MSCA_DEV_TYPE_SHIFT) |

00730                                       IXGBE_MDIO_AUTO_NEG_DEV_TYPE,

00740         status = IXGBE_ERR_LINK_SETUP;

00742     }
00743 
00744     return status;

00785 s32 ixgbe_reset_phy_nl(struct ixgbe_hw *hw)
00786 {

00788     bool end_data = false;
00789     u16 list_offset, data_offset;
00790     u16 phy_data = 0;
00791     s32 ret_val = 0;
00792     u32 i;
00793 
00794     hw->phy.ops.read_reg(hw, IXGBE_MDIO_PHY_XS_CONTROL,
00795                          IXGBE_MDIO_PHY_XS_DEV_TYPE, &phy_data);
00796 
00797     /* reset the PHY and poll for completion */
00798     hw->phy.ops.write_reg(hw, IXGBE_MDIO_PHY_XS_CONTROL,
00799                           IXGBE_MDIO_PHY_XS_DEV_TYPE,
00800                           (phy_data | IXGBE_MDIO_PHY_XS_RESET));
00801 
00802     for (i = 0; i < 100; i++) {
00803         hw->phy.ops.read_reg(hw, IXGBE_MDIO_PHY_XS_CONTROL,
00804                              IXGBE_MDIO_PHY_XS_DEV_TYPE, &phy_data);
00805         if ((phy_data & IXGBE_MDIO_PHY_XS_RESET) == 0)
00806             break;
00807         msleep(10);
00808     }
00809 
00810     if ((phy_data & IXGBE_MDIO_PHY_XS_RESET) != 0) {
00811         hw_dbg(hw, "PHY reset did not complete.\n");
00812         ret_val = IXGBE_ERR_PHY;
00813         goto out;
00814     }
00815 
00816     /* Get init offsets */
00817     ret_val = ixgbe_get_sfp_init_sequence_offsets(hw, &list_offset,
00818                                                   &data_offset);
00819     if (ret_val != 0)
00820         goto out;
00821 
00822     ret_val = hw->eeprom.ops.read(hw, data_offset, &block_crc);
00823     data_offset++;
00824     while (!end_data) {
00825         /*
00826          * Read control word from PHY init contents offset
00827          */
00828         ret_val = hw->eeprom.ops.read(hw, data_offset, &eword);
00829         control = (eword & IXGBE_CONTROL_MASK_NL) >>
00830                    IXGBE_CONTROL_SHIFT_NL;
00831         edata = eword & IXGBE_DATA_MASK_NL;
00832         switch (control) {
00833         case IXGBE_DELAY_NL:
00834             data_offset++;
00835             hw_dbg(hw, "DELAY: %d MS\n", edata);
00836             msleep(edata);
00837             break;
00838         case IXGBE_DATA_NL:
00839             hw_dbg(hw, "DATA:  \n");
00840             data_offset++;
00841             hw->eeprom.ops.read(hw, data_offset++,
00842                                 &phy_offset);
00843             for (i = 0; i < edata; i++) {
00844                 hw->eeprom.ops.read(hw, data_offset, &eword);
00845                 hw->phy.ops.write_reg(hw, phy_offset,
00846                                       IXGBE_TWINAX_DEV, eword);
00847                 hw_dbg(hw, "Wrote %4.4x to %4.4x\n", eword,
00848                           phy_offset);
00849                 data_offset++;
00850                 phy_offset++;
00851             }
00852             break;
00853         case IXGBE_CONTROL_NL:
00854             data_offset++;
00855             hw_dbg(hw, "CONTROL: \n");
00856             if (edata == IXGBE_CONTROL_EOL_NL) {
00857                 hw_dbg(hw, "EOL\n");
00858                 end_data = true;
00859             } else if (edata == IXGBE_CONTROL_SOL_NL) {
00860                 hw_dbg(hw, "SOL\n");
00861             } else {
00862                 hw_dbg(hw, "Bad control value\n");
00863                 ret_val = IXGBE_ERR_PHY;
00864                 goto out;
00865             }
00866             break;
00867         default:
00868             hw_dbg(hw, "Bad control type\n");
00869             ret_val = IXGBE_ERR_PHY;
00870             goto out;
00871         }
00872     }
00873 
00874 out:
00875     return ret_val;
00876 }
