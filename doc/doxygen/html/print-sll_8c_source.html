00108 /*
00109  * This is the top level routine of the printer.  'p' points to the
00110  * Linux "cooked capture" header of the packet, 'h->ts' is the timestamp,
00111  * 'h->len' is the length of the packet off the wire, and 'h->caplen'
00112  * is the number of bytes actually captured.
00113  */
00114 u_int
00115 sll_if_print(const struct pcap_pkthdr *h, const u_char *p)
00116 {
00117     u_int caplen = h->caplen;
00118     u_int length = h->len;
00119     register const struct sll_header *sllp;
00120     u_short ether_type;
00121     u_short extracted_ethertype;
00122 
00123     if (caplen < SLL_HDR_LEN) {
00124         /*
00125          * XXX - this "can't happen" because "pcap-linux.c" always
00126          * adds this many bytes of header to every packet in a
00127          * cooked socket capture.
00128          */
00129         printf("[|sll]");
00130         return (caplen);
00131     }
00132 
00133     sllp = (const struct sll_header *)p;
00134 
00135     if (eflag)
00136         sll_print(sllp, length);
00137 
00138     /*
00139      * Go past the cooked-mode header.
00140      */
00141     length -= SLL_HDR_LEN;
00142     caplen -= SLL_HDR_LEN;
00143     p += SLL_HDR_LEN;
00144 
00145     ether_type = EXTRACT_16BITS(&sllp->sll_protocol);
00146 
00147 recurse:
00148     /*
00149      * Is it (gag) an 802.3 encapsulation, or some non-Ethernet
00150      * packet type?
00151      */
00152     if (ether_type <= ETHERMTU) {
00153         /*
00154          * Yes - what type is it?
00155          */
00156         switch (ether_type) {
00157 
00158         case LINUX_SLL_P_802_3:
00159             /*
00160              * Ethernet_802.3 IPX frame.
00161              */
00162             ipx_print(p, length);
00163             break;
00164 
00165         case LINUX_SLL_P_802_2:
00166             /*
00167              * 802.2.
00168              * Try to print the LLC-layer header & higher layers.
00169              */
00170             if (llc_print(p, length, caplen, NULL, NULL,
00171                 &extracted_ethertype) == 0)
00172                 goto unknown;   /* unknown LLC type */
00173             break;
00174 
00175         default:
00176             extracted_ethertype = 0;
00177             /*FALLTHROUGH*/
00178 
00179         unknown:
00180             /* ether_type not known, print raw packet */
00181             if (!eflag)
00182                 sll_print(sllp, length + SLL_HDR_LEN);
00183             if (extracted_ethertype) {
00184                 printf("(LLC %s) ",
00185                    etherproto_string(htons(extracted_ethertype)));
00186             }
00187             if (!suppress_default_print)
00188                 default_print(p, caplen);
00189             break;
00190         }
00191     } else if (ether_type == ETHERTYPE_8021Q) {
00192         /*
00193          * Print VLAN information, and then go back and process
00194          * the enclosed type field.
00195          */
00196         if (caplen < 4 || length < 4) {
00197             printf("[|vlan]");
00198             return (SLL_HDR_LEN);
00199         }
00200             if (eflag) {
00201                 u_int16_t tag = EXTRACT_16BITS(p);
00202 
00203             printf("vlan %u, p %u%s, ",
00204                 tag & 0xfff,
00205                 tag >> 13,
00206                 (tag & 0x1000) ? ", CFI" : "");
00207         }
00208 
00209         ether_type = EXTRACT_16BITS(p + 2);
00210         if (ether_type <= ETHERMTU)
00211             ether_type = LINUX_SLL_P_802_2;
00212         if (!qflag) {
00213             (void)printf("ethertype %s, ",
00214                 tok2str(ethertype_values, "Unknown", ether_type));
00215         }
00216         p += 4;
00217         length -= 4;
00218         caplen -= 4;
00219         goto recurse;
00220     } else {
00221         if (ethertype_print(ether_type, p, length, caplen) == 0) {
00222             /* ether_type not known, print raw packet */
00223             if (!eflag)
00224                 sll_print(sllp, length + SLL_HDR_LEN);
00225             if (!suppress_default_print)
00226                 default_print(p, caplen);
00227         }
00228     }
00229 
00230     return (SLL_HDR_LEN);
