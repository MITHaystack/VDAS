00047 /*
00048  * Compute a HMAC MD5 sum.
00049  * Taken from rfc2104, Appendix.
00050  */
00051 static void
00052 signature_compute_hmac_md5(const u_int8_t *text, int text_len, unsigned char *key,
00053                            unsigned int key_len, u_int8_t *digest)
00054 {
00055     MD5_CTX context;
00056     unsigned char k_ipad[65];    /* inner padding - key XORd with ipad */
00057     unsigned char k_opad[65];    /* outer padding - key XORd with opad */
00058     unsigned char tk[16];
00059     int i;
00060 
00061     /* if key is longer than 64 bytes reset it to key=MD5(key) */
00062     if (key_len > 64) {
00063 
00064         MD5_CTX tctx;
00065 
00066         MD5_Init(&tctx);
00067         MD5_Update(&tctx, key, key_len);
00068         MD5_Final(tk, &tctx);
00069 
00070         key = tk;
00071         key_len = 16;
00072     }
00073 
00074     /*
00075      * the HMAC_MD5 transform looks like:
00076      *
00077      * MD5(K XOR opad, MD5(K XOR ipad, text))
00078      *
00079      * where K is an n byte key
00080      * ipad is the byte 0x36 repeated 64 times
00081      * opad is the byte 0x5c repeated 64 times
00082      * and text is the data being protected
00083      */
00084 
00085     /* start out by storing key in pads */
00086     memset(k_ipad, 0, sizeof k_ipad);
00087     memset(k_opad, 0, sizeof k_opad);
00088     memcpy(k_ipad, key, key_len);
00089     memcpy(k_opad, key, key_len);
00090 
00091     /* XOR key with ipad and opad values */
00092     for (i=0; i<64; i++) {
00093         k_ipad[i] ^= 0x36;
00094         k_opad[i] ^= 0x5c;
00095     }
00096 
00097     /*
00098      * perform inner MD5
00099      */
00100     MD5_Init(&context);                   /* init context for 1st pass */
00101     MD5_Update(&context, k_ipad, 64);     /* start with inner pad */
00102     MD5_Update(&context, text, text_len); /* then text of datagram */
00103     MD5_Final(digest, &context);          /* finish up 1st pass */
00104 
00105     /*
00106      * perform outer MD5
00107      */
00108     MD5_Init(&context);                   /* init context for 2nd pass */
00109     MD5_Update(&context, k_opad, 64);     /* start with outer pad */
00110     MD5_Update(&context, digest, 16);     /* then results of 1st hash */
00111     MD5_Final(digest, &context);          /* finish up 2nd pass */
00112 }
