00091             snprintf(errBuf, errMax,

00093                 return DAQ_ERROR;

00096         {
00097             char* end = entry->value;

00099 
00100             if ( *end || impl->port <= 0 || impl->port > 65535 )
00101             {
00102                 snprintf(errBuf, errMax, "%s: bad port (%s)\n",
00103                     __FUNCTION__, entry->value);
00104                 return DAQ_ERROR;
00105             }
00106         }
00107         else
00108         {
00109             snprintf(errBuf, errMax,
00110                 "%s: unsupported variable (%s=%s)\n",
00111                     __FUNCTION__, entry->key, entry->value);
00112                 return DAQ_ERROR;
00113         }
00114     }
00115 
00116     impl->snaplen = cfg->snaplen ? cfg->snaplen : IP_MAXPACKET;
00117     impl->timeout = cfg->timeout;
00118     impl->passive = ( cfg->mode == DAQ_MODE_PASSIVE );
00119 
00120     impl->sin.sin_family = impl->proto;
00121     impl->sin.sin_addr.s_addr = INADDR_ANY;

00286 static int ipfw_daq_acquire (
00287     void* handle, int cnt, DAQ_Analysis_Func_t callback, void* user)
00288 {
00289     IpfwImpl* impl = (IpfwImpl*)handle;
00290 
00291     fd_set fdset;
00292     int n = 0;
00293     struct timeval tv;
00294 
00295     tv.tv_usec = 0;
00296     // If cnt is <= 0, don't limit the packets acquired.  However,
00297     // impl->count = 0 has a special meaning, so interpret accordingly.
00298     impl->count = (cnt == 0) ? -1 : cnt;
00299 
00300     while ( impl->count < 0 || n < impl->count )
00301     {
00302         FD_ZERO(&fdset);
00303         FD_SET(impl->sock, &fdset);
00304 
00305         // set this per call
00306         tv.tv_sec = impl->timeout;
00307 
00308         if ( select(impl->sock+1, &fdset, NULL, NULL, &tv) < 0 )
00309         {
00310             DPE(impl->error, "%s: can't select divert socket (%s)\n",
00311                 __FUNCTION__, strerror(errno));
00312             return DAQ_ERROR;
00313         }
00314 
00315         if (FD_ISSET(impl->sock, &fdset))
00316         {
00317             socklen_t sinlen = sizeof(impl->sin);
00318             ssize_t pktlen;
00319 
00320             DAQ_PktHdr_t hdr;
00321             DAQ_Verdict verdict;
00322 
00323             if ((pktlen = recvfrom(impl->sock, impl->buf, impl->snaplen, 0,
00324                 (struct sockaddr *)&impl->sin, &sinlen)) == -1)
00325             {
00326                 if (errno != EINTR)
00327                 {
00328                     DPE(impl->error, "%s: can't readfrom divert socket (%s)\n",
00329                         __FUNCTION__, strerror(errno));
00330                     return DAQ_ERROR;
00331                 }
00332             }
00333 
00334             SetPktHdr(&hdr, pktlen);
00335             impl->stats.hw_packets_received++;
00336 
00337             if (
00338                 impl->fcode.bf_insns &&
00339                 sfbpf_filter(impl->fcode.bf_insns, impl->buf,
00340                     hdr.caplen, hdr.caplen) == 0
00341             ) {
00342                 verdict = DAQ_VERDICT_PASS;
00343                 impl->stats.packets_filtered++;
00344             }
00345             else
00346             {
00347                 verdict = callback(NULL, &hdr, impl->buf);
00348                 impl->stats.verdicts[verdict]++;
00349                 impl->stats.packets_received++;
00350             }
00351             if ( impl->passive || s_fwd[verdict] )
00352                 ipfw_daq_forward(impl, &hdr, impl->buf, hdr.pktlen, 0);
00353 
00354             n++;
00355         }
00356     }
00357     return 0;
00358 }
