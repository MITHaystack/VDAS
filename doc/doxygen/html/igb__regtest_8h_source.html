00028 /* ethtool register test data */

00030     u16 reg;

00038 /* In the hardware, registers are laid out either singly, in arrays
00039  * spaced 0x100 bytes apart, or in contiguous tables.  We assume
00040  * most tests take place on arrays or single registers (handled
00041  * as a single-element array) and special-case the tables.
00042  * Table tests are always pattern tests.
00043  *
00044  * We also make provision for some required setup steps by specifying
00045  * registers to be written without any read-back testing.
00046  */
00047 
00048 #define PATTERN_TEST    1
00049 #define SET_READ_TEST   2
00050 #define WRITE_NO_TEST   3
00051 #define TABLE32_TEST    4
00052 #define TABLE64_TEST_LO 5
00053 #define TABLE64_TEST_HI 6
