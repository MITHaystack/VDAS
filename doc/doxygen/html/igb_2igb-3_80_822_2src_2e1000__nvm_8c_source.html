00088 static void e1000_shift_out_eec_bits(struct e1000_hw *hw, u16 data, u16 count)
00089 {
00090     struct e1000_nvm_info *nvm = &hw->nvm;
00091     u32 eecd = E1000_READ_REG(hw, E1000_EECD);
00092     u32 mask;
00093 
00094     DEBUGFUNC("e1000_shift_out_eec_bits");
00095 
00096     mask = 0x01 << (count - 1);
00097     if (nvm->type == e1000_nvm_eeprom_spi)
00098         eecd |= E1000_EECD_DO;
00099 
00100     do {
00101         eecd &= ~E1000_EECD_DI;
00102 
00103         if (data & mask)
00104             eecd |= E1000_EECD_DI;
00105 
00106         E1000_WRITE_REG(hw, E1000_EECD, eecd);
00107         E1000_WRITE_FLUSH(hw);
00108 
00109         usec_delay(nvm->delay_usec);
00110 
00111         e1000_raise_eec_clk(hw, &eecd);
00112         e1000_lower_eec_clk(hw, &eecd);
00113 
00114         mask >>= 1;
00115     } while (mask);
00116 
00117     eecd &= ~E1000_EECD_DI;
00118     E1000_WRITE_REG(hw, E1000_EECD, eecd);

00302 static s32 e1000_ready_nvm_eeprom(struct e1000_hw *hw)
00303 {
00304     struct e1000_nvm_info *nvm = &hw->nvm;
00305     u32 eecd = E1000_READ_REG(hw, E1000_EECD);
00306     s32 ret_val = E1000_SUCCESS;
00307     u8 spi_stat_reg;
00308 
00309     DEBUGFUNC("e1000_ready_nvm_eeprom");
00310 
00311     if (nvm->type == e1000_nvm_eeprom_spi) {
00312         u16 timeout = NVM_MAX_RETRY_SPI;
00313 
00314         /* Clear SK and CS */
00315         eecd &= ~(E1000_EECD_CS | E1000_EECD_SK);
00316         E1000_WRITE_REG(hw, E1000_EECD, eecd);
00317         usec_delay(1);
00318 
00319         /*
00320          * Read "Status Register" repeatedly until the LSB is cleared.
00321          * The EEPROM will signal that the command has been completed
00322          * by clearing bit 0 of the internal status register.  If it's
00323          * not cleared within 'timeout', then error out.
00324          */
00325         while (timeout) {
00326             e1000_shift_out_eec_bits(hw, NVM_RDSR_OPCODE_SPI,
00327                                      hw->nvm.opcode_bits);
00328             spi_stat_reg = (u8)e1000_shift_in_eec_bits(hw, 8);
00329             if (!(spi_stat_reg & NVM_STATUS_RDY_SPI))
00330                 break;
00331 
00332             usec_delay(5);
00333             e1000_standby_nvm(hw);
00334             timeout--;
00335         }
00336 
00337         if (!timeout) {
00338             DEBUGOUT("SPI NVM Status error\n");
00339             ret_val = -E1000_ERR_NVM;
00340             goto out;
00341         }
00342     }
00343 
00344 out:
00345     return ret_val;
00346 }

00357 s32 e1000_read_nvm_spi(struct e1000_hw *hw, u16 offset, u16 words, u16 *data)
00358 {
00359     struct e1000_nvm_info *nvm = &hw->nvm;
00360     u32 i = 0;
00361     s32 ret_val;
00362     u16 word_in;
00363     u8 read_opcode = NVM_READ_OPCODE_SPI;
00364 
00365     DEBUGFUNC("e1000_read_nvm_spi");
00366 
00367     /*
00368      * A check for invalid values:  offset too large, too many words,
00369      * and not enough words.
00370      */
00371     if ((offset >= nvm->word_size) || (words > (nvm->word_size - offset)) ||
00372         (words == 0)) {
00373         DEBUGOUT("nvm parameter(s) out of bounds\n");
00374         ret_val = -E1000_ERR_NVM;
00375         goto out;
00376     }
00377 
00378     ret_val = nvm->ops.acquire(hw);
00379     if (ret_val)
00380         goto out;
00381 
00382     ret_val = e1000_ready_nvm_eeprom(hw);
00383     if (ret_val)
00384         goto release;
00385 
00386     e1000_standby_nvm(hw);
00387 
00388     if ((nvm->address_bits == 8) && (offset >= 128))
00389         read_opcode |= NVM_A8_OPCODE_SPI;
00390 
00391     /* Send the READ command (opcode + addr) */
00392     e1000_shift_out_eec_bits(hw, read_opcode, nvm->opcode_bits);
00393     e1000_shift_out_eec_bits(hw, (u16)(offset*2), nvm->address_bits);
00394 
00395     /*
00396      * Read the data.  SPI NVMs increment the address with each byte
00397      * read and will roll over if reading beyond the end.  This allows
00398      * us to read the whole NVM from any offset
00399      */
00400     for (i = 0; i < words; i++) {
00401         word_in = e1000_shift_in_eec_bits(hw, 16);
00402         data[i] = (word_in >> 8) | (word_in << 8);
00403     }
00404 
00405 release:
00406     nvm->ops.release(hw);
00407 
00408 out:
00409     return ret_val;
00410 }

00469 s32 e1000_write_nvm_spi(struct e1000_hw *hw, u16 offset, u16 words, u16 *data)
00470 {
00471     struct e1000_nvm_info *nvm = &hw->nvm;
00472     s32 ret_val;
00473     u16 widx = 0;
00474 
00475     DEBUGFUNC("e1000_write_nvm_spi");
00476 
00477     /*
00478      * A check for invalid values:  offset too large, too many words,
00479      * and not enough words.
00480      */
00481     if ((offset >= nvm->word_size) || (words > (nvm->word_size - offset)) ||
00482         (words == 0)) {
00483         DEBUGOUT("nvm parameter(s) out of bounds\n");
00484         ret_val = -E1000_ERR_NVM;
00485         goto out;
00486     }
00487 
00488     ret_val = nvm->ops.acquire(hw);
00489     if (ret_val)
00490         goto out;
00491 
00492     while (widx < words) {
00493         u8 write_opcode = NVM_WRITE_OPCODE_SPI;
00494 
00495         ret_val = e1000_ready_nvm_eeprom(hw);
00496         if (ret_val)
00497             goto release;
00498 
00499         e1000_standby_nvm(hw);
00500 
00501         /* Send the WRITE ENABLE command (8 bit opcode) */
00502         e1000_shift_out_eec_bits(hw, NVM_WREN_OPCODE_SPI,
00503                                  nvm->opcode_bits);
00504 
00505         e1000_standby_nvm(hw);
00506 
00507         /*
00508          * Some SPI eeproms use the 8th address bit embedded in the
00509          * opcode
00510          */
00511         if ((nvm->address_bits == 8) && (offset >= 128))
00512             write_opcode |= NVM_A8_OPCODE_SPI;
00513 
00514         /* Send the Write command (8-bit opcode + addr) */
00515         e1000_shift_out_eec_bits(hw, write_opcode, nvm->opcode_bits);
00516         e1000_shift_out_eec_bits(hw, (u16)((offset + widx) * 2),
00517                                  nvm->address_bits);
00518 
00519         /* Loop to allow for up to whole page write of eeprom */
00520         while (widx < words) {
00521             u16 word_out = data[widx];
00522             word_out = (word_out >> 8) | (word_out << 8);
00523             e1000_shift_out_eec_bits(hw, word_out, 16);
00524             widx++;
00525 
00526             if ((((offset + widx) * 2) % nvm->page_size) == 0) {
00527                 e1000_standby_nvm(hw);
00528                 break;
00529             }
00530         }
00531     }
00532 
00533     msec_delay(10);
00534 release:
00535     nvm->ops.release(hw);
00536 
00537 out:
00538     return ret_val;
00539 }

00550 s32 e1000_read_pba_string_generic(struct e1000_hw *hw, u8 *pba_num,
00551                                   u32 pba_num_size)
00552 {
00553     s32 ret_val;
00554     u16 nvm_data;
00555     u16 pba_ptr;
00556     u16 offset;
00557     u16 length;
00558 
00559     DEBUGFUNC("e1000_read_pba_string_generic");
00560 
00561     if (pba_num == NULL) {
00562         DEBUGOUT("PBA string buffer was null\n");
00563         ret_val = E1000_ERR_INVALID_ARGUMENT;
00564         goto out;
00565     }
00566 
00567     ret_val = hw->nvm.ops.read(hw, NVM_PBA_OFFSET_0, 1, &nvm_data);
00568     if (ret_val) {
00569         DEBUGOUT("NVM Read Error\n");
00570         goto out;
00571     }
00572 
00573     ret_val = hw->nvm.ops.read(hw, NVM_PBA_OFFSET_1, 1, &pba_ptr);
00574     if (ret_val) {
00575         DEBUGOUT("NVM Read Error\n");
00576         goto out;
00577     }
00578 
00579     /*
00580      * if nvm_data is not ptr guard the PBA must be in legacy format which
00581      * means pba_ptr is actually our second data word for the PBA number
00582      * and we can decode it into an ascii string
00583      */
00584     if (nvm_data != NVM_PBA_PTR_GUARD) {
00585         DEBUGOUT("NVM PBA number is not stored as string\n");
00586 
00587         /* we will need 11 characters to store the PBA */
00588         if (pba_num_size < 11) {
00589             DEBUGOUT("PBA string buffer too small\n");
00590             return E1000_ERR_NO_SPACE;
00591         }
00592 
00593         /* extract hex string from data and pba_ptr */
00594         pba_num[0] = (nvm_data >> 12) & 0xF;
00595         pba_num[1] = (nvm_data >> 8) & 0xF;
00596         pba_num[2] = (nvm_data >> 4) & 0xF;
00597         pba_num[3] = nvm_data & 0xF;
00598         pba_num[4] = (pba_ptr >> 12) & 0xF;
00599         pba_num[5] = (pba_ptr >> 8) & 0xF;
00600         pba_num[6] = '-';
00601         pba_num[7] = 0;
00602         pba_num[8] = (pba_ptr >> 4) & 0xF;
00603         pba_num[9] = pba_ptr & 0xF;
00604 
00605         /* put a null character on the end of our string */
00606         pba_num[10] = '\0';
00607 
00608         /* switch all the data but the '-' to hex char */
00609         for (offset = 0; offset < 10; offset++) {
00610             if (pba_num[offset] < 0xA)
00611                 pba_num[offset] += '0';
00612             else if (pba_num[offset] < 0x10)
00613                 pba_num[offset] += 'A' - 0xA;
00614         }
00615 
00616         goto out;
00617     }
00618 
00619     ret_val = hw->nvm.ops.read(hw, pba_ptr, 1, &length);
00620     if (ret_val) {
00621         DEBUGOUT("NVM Read Error\n");
00622         goto out;
00623     }
00624 
00625     if (length == 0xFFFF || length == 0) {
00626         DEBUGOUT("NVM PBA number section invalid length\n");
00627         ret_val = E1000_ERR_NVM_PBA_SECTION;
00628         goto out;
00629     }
00630     /* check if pba_num buffer is big enough */
00631     if (pba_num_size < (((u32)length * 2) - 1)) {
00632         DEBUGOUT("PBA string buffer too small\n");
00633         ret_val = E1000_ERR_NO_SPACE;
00634         goto out;
00635     }
00636 
00637     /* trim pba length from start of string */
00638     pba_ptr++;
00639     length--;
00640 
00641     for (offset = 0; offset < length; offset++) {
00642         ret_val = hw->nvm.ops.read(hw, pba_ptr + offset, 1, &nvm_data);
00643         if (ret_val) {
00644             DEBUGOUT("NVM Read Error\n");
00645             goto out;
00646         }
00647         pba_num[offset * 2] = (u8)(nvm_data >> 8);
00648         pba_num[(offset * 2) + 1] = (u8)(nvm_data & 0xFF);
00649     }
00650     pba_num[offset * 2] = '\0';
00651 
00652 out:
00653     return ret_val;
00654 }
