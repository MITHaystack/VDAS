00048 s32 ixgbe_dcb_check_config(struct ixgbe_dcb_config *dcb_config)
00049 {
00050     struct tc_bw_alloc *p;
00051     s32 ret_val = 0;
00052     u8 i, j, bw = 0, bw_id;
00053     u8 bw_sum[2][MAX_BW_GROUP];
00054     bool link_strict[2][MAX_BW_GROUP];
00055 
00056     memset(bw_sum, 0, sizeof(bw_sum));
00057     memset(link_strict, 0, sizeof(link_strict));
00058 
00059     /* First Tx, then Rx */
00060     for (i = 0; i < 2; i++) {
00061         /* Check each traffic class for rule violation */
00062         for (j = 0; j < dcb_config->num_tcs.pg_tcs; j++) {
00063             p = &dcb_config->tc_config[j].path[i];
00064 
00065             bw = p->bwg_percent;
00066             bw_id = p->bwg_id;
00067 
00068             if (bw_id >= MAX_BW_GROUP) {
00069                 ret_val = DCB_ERR_CONFIG;
00070                 goto err_config;
00071             }
00072             if (p->prio_type == prio_link) {
00073                 link_strict[i][bw_id] = true;
00074                 /* Link strict should have zero bandwidth */
00075                 if (bw) {
00076                     ret_val = DCB_ERR_LS_BW_NONZERO;
00077                     goto err_config;
00078                 }
00079             } else if (!bw) {
00080                 /*
00081                  * Traffic classes without link strict
00082                  * should have non-zero bandwidth.
00083                  */
00084                 ret_val = DCB_ERR_TC_BW_ZERO;
00085                 goto err_config;
00086             }
00087             bw_sum[i][bw_id] += bw;
00088         }
00089 
00090         bw = 0;
00091 
00092         /* Check each bandwidth group for rule violation */
00093         for (j = 0; j < MAX_BW_GROUP; j++) {
00094             bw += dcb_config->bw_percentage[i][j];
00095             /*
00096              * Sum of bandwidth percentages of all traffic classes
00097              * within a Bandwidth Group must total 100 except for
00098              * link strict group (zero bandwidth).
00099              */
00100             if (link_strict[i][j]) {
00101                 if (bw_sum[i][j]) {
00102                     /*
00103                      * Link strict group should have zero
00104                      * bandwidth.
00105                      */
00106                     ret_val = DCB_ERR_LS_BWG_NONZERO;
00107                     goto err_config;
00108                 }
00109             } else if (bw_sum[i][j] != BW_PERCENT &&
00110                        bw_sum[i][j] != 0) {
00111                 ret_val = DCB_ERR_TC_BW;
00112                 goto err_config;
00113             }
00114         }
00115 
00116         if (bw != BW_PERCENT) {
00117             ret_val = DCB_ERR_BW_GROUP;
00118             goto err_config;
00119         }
00120     }
00121 
00122     return DCB_SUCCESS;
00123 
00124 err_config:
00125     hw_dbg(hw, "DCB error code %d while checking %s settings.\n",
00126               ret_val, (j == DCB_TX_CONFIG) ? "Tx" : "Rx");
00127 
00128     return ret_val;
00129 }

00140 s32 ixgbe_dcb_calculate_tc_credits(struct ixgbe_hw *hw,
00141                                    struct ixgbe_dcb_config *dcb_config,
00142                                    u32 max_frame, u8 direction)
00143 {
00144     struct tc_bw_alloc *p;
00145     s32 ret_val = 0;
00146     /* Initialization values default for Tx settings */
00147     u32 credit_refill       = 0;
00148     u32 credit_max          = 0;
00149     u32 min_credit          = 0;
00150     u32 min_percent         = 100;
00151     u16 min_multiplier      = 0;
00152     u16 link_percentage     = 0;
00153     u8  bw_percent          = 0;
00154     u8  i;
00155 
00156     if (dcb_config == NULL) {
00157         ret_val = DCB_ERR_CONFIG;
00158         goto out;
00159     }
00160 
00161     min_credit = ((max_frame / 2) + DCB_CREDIT_QUANTUM - 1) /
00162                  DCB_CREDIT_QUANTUM;
00163 
00164     /* Find smallest link percentage */
00165     for (i = 0; i < dcb_config->num_tcs.pg_tcs; i++) {
00166         p = &dcb_config->tc_config[i].path[direction];
00167         bw_percent = dcb_config->bw_percentage[direction][p->bwg_id];
00168         link_percentage = p->bwg_percent;
00169 
00170         link_percentage = (link_percentage * bw_percent) / 100;
00171 
00172         if (link_percentage && link_percentage < min_percent)
00173             min_percent = link_percentage;
00174     }
00175 
00176     /*
00177      * The ratio between traffic classes will control the bandwidth
00178      * percentages seen on the wire. To calculate this ratio we use
00179      * a multiplier. It is required that the refill credits must be
00180      * larger than the max frame size so here we find the smallest
00181      * multiplier that will allow all bandwidth percentages to be
00182      * greater than the max frame size.
00183      */
00184     min_multiplier = (min_credit / min_percent) + 1;
00185 
00186     /* Find out the link percentage for each TC first */
00187     for (i = 0; i < dcb_config->num_tcs.pg_tcs; i++) {
00188         p = &dcb_config->tc_config[i].path[direction];
00189         bw_percent = dcb_config->bw_percentage[direction][p->bwg_id];
00190 
00191         link_percentage = p->bwg_percent;
00192         /* Must be careful of integer division for very small nums */
00193         link_percentage = (link_percentage * bw_percent) / 100;
00194         if (p->bwg_percent > 0 && link_percentage == 0)
00195             link_percentage = 1;
00196 
00197         /* Save link_percentage for reference */
00198         p->link_percent = (u8)link_percentage;
00199 
00200         /* Calculate credit refill and save it */
00201         credit_refill = min(link_percentage * min_multiplier,
00202                             MAX_CREDIT_REFILL);
00203         p->data_credits_refill = (u16)credit_refill;
00204 
00205         /* Calculate maximum credit for the TC */
00206         credit_max = (link_percentage * MAX_CREDIT) / 100;
00207 
00208         /*
00209          * Adjustment based on rule checking, if the percentage
00210          * of a TC is too small, the maximum credit may not be
00211          * enough to send out a jumbo frame in data plane arbitration.
00212          */
00213         if (credit_max && credit_max < min_credit)
00214             credit_max = min_credit;
00215 
00216         if (direction == DCB_TX_CONFIG) {
00217             /*
00218              * Adjustment based on rule checking, if the
00219              * percentage of a TC is too small, the maximum
00220              * credit may not be enough to send out a TSO
00221              * packet in descriptor plane arbitration.
00222              */
00223             if (credit_max && (credit_max < MINIMUM_CREDIT_FOR_TSO)
00224                 && (hw->mac.type == ixgbe_mac_82598EB))
00225                 credit_max = MINIMUM_CREDIT_FOR_TSO;
00226 
00227             dcb_config->tc_config[i].desc_credits_max =
00228                (u16)credit_max;
00229         }
00230 
00231         p->data_credits_max = (u16)credit_max;
00232     }
00233 
00234 out:
00235     return ret_val;
00236 }
