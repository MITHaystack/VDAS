00091     int type, len, i, j;
00092         const u_char *tptr;
00093 
00094     if (caplen < CDP_HEADER_LEN) {
00095         (void)printf("[|cdp]");
00096         return;
00097     }
00098 
00099         tptr = pptr; /* temporary pointer */
00100 
00101         if (!TTEST2(*tptr, CDP_HEADER_LEN))
00102                 goto trunc;
00103     printf("CDPv%u, ttl: %us", *tptr, *(tptr+1));
00104         if (vflag)
00105                 printf(", checksum: %u (unverified), length %u", EXTRACT_16BITS(tptr), length);
00106     tptr += CDP_HEADER_LEN;
00107 
00108     while (tptr < (pptr+length)) {
00109 
00110                 if (!TTEST2(*tptr, 4)) /* read out Type and Length */
00111                     goto trunc;
00112         type = EXTRACT_16BITS(tptr);
00113         len  = EXTRACT_16BITS(tptr+2); /* object length includes the 4 bytes header length */
00114                 tptr += 4;
00115                 len -= 4;
00116 
00117         if (!TTEST2(*tptr, len))
00118             goto trunc;
00119 
00120                 if (vflag || type == 1) { /* in non-verbose mode just print Device-ID */
00121 
00122                     if (vflag)
00123                         printf("\n\t%s (0x%02x), length: %u byte%s: ",
00124                                tok2str(cdp_tlv_values,"unknown field type", type),
00125                                type,
00126                                len,
00127                                len>1 ? "s" : ""); /* plural */
00128 
00129                     switch (type) {
00130 
00131                     case 0x01: /* Device-ID */
00132                         if (!vflag)
00133                             printf(", Device-ID ");
00134                         printf("'");
00135                         fn_printn(tptr, len, NULL);
00136                         printf("'");
00137             break;

00140                             goto trunc;
00141             break;

00145                         printf("'");

00150                                bittok2str(cdp_capability_values, "none",EXTRACT_32BITS(tptr)));

00152                     case 0x05: /* Version */

00155                             j = *(tptr+i);
00156                             putchar(j);
00157                             if (j == 0x0a) /* lets rework the version string to get a nice identation */
00158                                 printf("\t  ");
00159                         }

00161                     case 0x06: /* Platform */
00162                         printf("'");
00163                         fn_printn(tptr, len, NULL);

00165             break;
00166                     case 0x07: /* Prefixes */
00167             if (cdp_print_prefixes(tptr, len) < 0)
00168                             goto trunc;

00170                     case 0x08: /* Protocol Hello Option - not documented */
00171             break;

00238 /*
00239  * Protocol type values.
00240  *
00241  * PT_NLPID means that the protocol type field contains an OSI NLPID.
00242  *
00243  * PT_IEEE_802_2 means that the protocol type field contains an IEEE 802.2
00244  * LLC header that specifies that the payload is for that protocol.
00245  */
00246 #define PT_NLPID        1   /* OSI NLPID */
00247 #define PT_IEEE_802_2       2   /* IEEE 802.2 LLC header */

00249 static int
00250 cdp_print_addr(const u_char * p, int l)
00251 {
00252     int pt, pl, al, num;
00253     const u_char *endp = p + l;
00254 #ifdef INET6
00255     static u_char prot_ipv6[] = {
00256         0xaa, 0xaa, 0x03, 0x00, 0x00, 0x00, 0x86, 0xdd
00257     };
00258 #endif
00259 
00260     TCHECK2(*p, 2);
00261     num = EXTRACT_32BITS(p);
00262     p += 4;
00263 
00264     while (p < endp && num >= 0) {
00265         TCHECK2(*p, 2);
00266         if (p + 2 > endp)
00267             goto trunc;
00268         pt = p[0];      /* type of "protocol" field */
00269         pl = p[1];      /* length of "protocol" field */
00270         p += 2;
00271 
00272         TCHECK2(p[pl], 2);
00273         if (p + pl + 2 > endp)
00274             goto trunc;
00275         al = EXTRACT_16BITS(&p[pl]);    /* address length */
00276 
00277         if (pt == PT_NLPID && pl == 1 && *p == NLPID_IP && al == 4) {
00278             /*
00279              * IPv4: protocol type = NLPID, protocol length = 1
00280              * (1-byte NLPID), protocol = 0xcc (NLPID for IPv4),
00281              * address length = 4
00282              */
00283             p += 3;
00284 
00285             TCHECK2(*p, 4);
00286             if (p + 4 > endp)
00287                 goto trunc;
00288             printf("IPv4 (%u) %s",
00289                                num,
00290                                ipaddr_string(p));
00291             p += 4;
00292         }
00293 #ifdef INET6
00294         else if (pt == PT_IEEE_802_2 && pl == 8 &&
00295             memcmp(p, prot_ipv6, 8) == 0 && al == 16) {
00296             /*
00297              * IPv6: protocol type = IEEE 802.2 header,
00298              * protocol length = 8 (size of LLC+SNAP header),
00299              * protocol = LLC+SNAP header with the IPv6
00300              * Ethertype, address length = 16
00301              */
00302             p += 10;
00303             TCHECK2(*p, al);
00304             if (p + al > endp)
00305                 goto trunc;
00306 
00307             printf("IPv6 (%u) %s",
00308                                num,
00309                                ip6addr_string(p));
00310             p += al;
00311         }
00312 #endif
00313         else {
00314             /*
00315              * Generic case: just print raw data
00316              */
00317             TCHECK2(*p, pl);
00318             if (p + pl > endp)
00319                 goto trunc;
00320             printf("pt=0x%02x, pl=%d, pb=", *(p - 2), pl);
00321             while (pl-- > 0)
00322                 printf(" %02x", *p++);
00323             TCHECK2(*p, 2);
00324             if (p + 2 > endp)
00325                 goto trunc;
00326             al = (*p << 8) + *(p + 1);
00327             printf(", al=%d, a=", al);
00328             p += 2;
00329             TCHECK2(*p, al);
00330             if (p + al > endp)
00331                 goto trunc;
00332             while (al-- > 0)
00333                 printf(" %02x", *p++);
00334         }
00335         num--;
00336         if (num)
00337             printf(" ");
00338     }
00339 
00340     return 0;
00341 
00342 trunc:
00343     return -1;
00344 }

00367 /* read in a <n>-byte number, MSB first

00370 static unsigned long cdp_get_number(const u_char * p, int l)
00371 {
00372     unsigned long res=0;
00373     while( l>0 )
