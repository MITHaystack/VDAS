00163 /*
00164  * Convert a port name to its port and protocol numbers.
00165  * We assume only TCP or UDP.
00166  * Return 0 upon failure.
00167  */
00168 int
00169 pcap_nametoport(const char *name, int *port, int *proto)
00170 {
00171     struct servent *sp;
00172     int tcp_port = -1;
00173     int udp_port = -1;
00174 
00175     /*
00176      * We need to check /etc/services for ambiguous entries.
00177      * If we find the ambiguous entry, and it has the
00178      * same port number, change the proto to PROTO_UNDEF
00179      * so both TCP and UDP will be checked.
00180      */
00181     sp = getservbyname(name, "tcp");
00182     if (sp != NULL) tcp_port = ntohs(sp->s_port);
00183     sp = getservbyname(name, "udp");
00184     if (sp != NULL) udp_port = ntohs(sp->s_port);
00185     if (tcp_port >= 0) {
00186         *port = tcp_port;
00187         *proto = IPPROTO_TCP;
00188         if (udp_port >= 0) {
00189             if (udp_port == tcp_port)
00190                 *proto = PROTO_UNDEF;
00191 #ifdef notdef
00192             else
00193                 /* Can't handle ambiguous names that refer
00194                    to different port numbers. */
00195                 warning("ambiguous port %s in /etc/services",
00196                     name);
00197 #endif
00198         }
00199         return 1;
00200     }
00201     if (udp_port >= 0) {
00202         *port = udp_port;
00203         *proto = IPPROTO_UDP;
00204         return 1;
00205     }
00206 #if defined(ultrix) || defined(__osf__)
00207     /* Special hack in case NFS isn't in /etc/services */
00208     if (strcmp(name, "nfs") == 0) {
00209         *port = 2049;
00210         *proto = PROTO_UNDEF;
00211         return 1;
00212     }
00213 #endif
00214     return 0;
00215 }

00398 /*
00399  * Convert 's', which can have the one of the forms:
00400  *
00401  *  "xx:xx:xx:xx:xx:xx"
00402  *  "xx.xx.xx.xx.xx.xx"
00403  *  "xx-xx-xx-xx-xx-xx"
00404  *  "xxxx.xxxx.xxxx"
00405  *  "xxxxxxxxxxxx"
00406  *
00407  * (or various mixes of ':', '.', and '-') into a new
00408  * ethernet address.  Assumes 's' is well formed.
00409  */
00410 u_char *
00411 pcap_ether_aton(const char *s)
00412 {
00413     register u_char *ep, *e;
00414     register u_int d;
00415 
00416     e = ep = (u_char *)malloc(6);
00417 
00418     while (*s) {
00419         if (*s == ':' || *s == '.' || *s == '-')
00420             s += 1;
00421         d = xdtoi(*s++);
00422         if (isxdigit((unsigned char)*s)) {
00423             d <<= 4;
00424             d |= xdtoi(*s++);
00425         }
00426         *ep++ = d;
00427     }
00428 
00429     return (e);
00430 }
