00089 /*
00090  * Struct return by BIOCVERSION.  This represents the version number of 
00091  * the filter language described by the instruction encodings below.
00092  * bpf understands a program iff kernel_major == filter_major &&
00093  * kernel_minor >= filter_minor, that is, if the value returned by the
00094  * running kernel has the same major number and a minor number equal
00095  * equal to or less than the filter being downloaded.  Otherwise, the
00096  * results are undefined, meaning an error may be returned or packets
00097  * may be accepted haphazardly.
00098  * It has nothing to do with the source code version.
00099  */
00100 struct sfbpf_version {
00101     u_short bv_major;
00102     u_short bv_minor;
00103 };
00104 /* Current version number of filter architecture. */
00105 #define SFBPF_MAJOR_VERSION 1
00106 #define SFBPF_MINOR_VERSION 1
