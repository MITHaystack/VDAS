00189 static int
00190 pcap_read_dlpi(pcap_t *p, int cnt, pcap_handler callback, u_char *user)
00191 {
00192     int cc;
00193     u_char *bp;
00194     int flags;
00195     struct strbuf data;
00196 
00197     flags = 0;
00198     cc = p->cc;
00199     if (cc == 0) {
00200         data.buf = (char *)p->buffer + p->offset;
00201         data.maxlen = p->bufsize;
00202         data.len = 0;
00203         do {
00204             /*
00205              * Has "pcap_breakloop()" been called?
00206              */
00207             if (p->break_loop) {
00208                 /*
00209                  * Yes - clear the flag that indicates
00210                  * that it has, and return -2 to
00211                  * indicate that we were told to
00212                  * break out of the loop.
00213                  */
00214                 p->break_loop = 0;
00215                 return (-2);
00216             }
00217             /*
00218              * XXX - check for the DLPI primitive, which
00219              * would be DL_HP_RAWDATA_IND on HP-UX
00220              * if we're in raw mode?
00221              */
00222             if (getmsg(p->fd, &ctl, &data, &flags) < 0) {
00223                 /* Don't choke when we get ptraced */
00224                 switch (errno) {
00225 
00226                 case EINTR:
00227                     cc = 0;
00228                     continue;
00229 
00230                 case EAGAIN:
00231                     return (0);
00232                 }
00233                 strlcpy(p->errbuf, pcap_strerror(errno),
00234                     sizeof(p->errbuf));
00235                 return (-1);
00236             }
00237             cc = data.len;
00238         } while (cc == 0);
00239         bp = p->buffer + p->offset;
00240     } else
00241         bp = p->bp;
00242 
00243     return (pcap_process_pkts(p, callback, user, cnt, bp, cc));
00244 }

00246 static int
00247 pcap_inject_dlpi(pcap_t *p, const void *buf, size_t size)
00248 {
00249     int ret;
00250 
00251 #if defined(DLIOCRAW)
00252     ret = write(p->fd, buf, size);
00253     if (ret == -1) {
00254         snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "send: %s",
00255             pcap_strerror(errno));
00256         return (-1);
00257     }
00258 #elif defined(DL_HP_RAWDLS)
00259     if (p->send_fd < 0) {
00260         snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00261             "send: Output FD couldn't be opened");
00262         return (-1);
00263     }
00264     ret = dlrawdatareq(p->send_fd, buf, size);
00265     if (ret == -1) {
00266         snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "send: %s",
00267             pcap_strerror(errno));
00268         return (-1);
00269     }
00270     /*
00271      * putmsg() returns either 0 or -1; it doesn't indicate how
00272      * many bytes were written (presumably they were all written
00273      * or none of them were written).  OpenBSD's pcap_inject()
00274      * returns the number of bytes written, so, for API compatibility,
00275      * we return the number of bytes we were told to write.
00276      */
00277     ret = size;
00278 #else /* no raw mode */
00279     /*
00280      * XXX - this is a pain, because you might have to extract
00281      * the address from the packet and use it in a DL_UNITDATA_REQ
00282      * request.  That would be dependent on the link-layer type.
00283      *
00284      * I also don't know what SAP you'd have to bind the descriptor
00285      * to, or whether you'd need separate "receive" and "send" FDs,
00286      * nor do I know whether you'd need different bindings for
00287      * D/I/X Ethernet and 802.3, or for {FDDI,Token Ring} plus
00288      * 802.2 and {FDDI,Token Ring} plus 802.2 plus SNAP.
00289      *
00290      * So, for now, we just return a "you can't send" indication,
00291      * and leave it up to somebody with a DLPI-based system lacking
00292      * both DLIOCRAW and DL_HP_RAWDLS to supply code to implement
00293      * packet transmission on that system.  If they do, they should
00294      * send it to us - but should not send us code that assumes
00295      * Ethernet; if the code doesn't work on non-Ethernet interfaces,
00296      * it should check "p->linktype" and reject the send request if
00297      * it's anything other than DLT_EN10MB.
00298      */
00299     strlcpy(p->errbuf, "send: Not supported on this version of this OS",
00300         PCAP_ERRBUF_SIZE);
00301     ret = -1;
00302 #endif /* raw mode */
00303     return (ret);
00304 }

00332 static int
00333 pcap_activate_dlpi(pcap_t *p)
00334 {
00335     register char *cp;
00336     int ppa;
00337 #ifdef HAVE_SOLARIS
00338     int isatm = 0;
00339 #endif
00340     register dl_info_ack_t *infop;
00341 #ifdef HAVE_SYS_BUFMOD_H
00342     bpf_u_int32 ss;
00343 #ifdef HAVE_SOLARIS
00344     register char *release;
00345     bpf_u_int32 osmajor, osminor, osmicro;
00346 #endif
00347 #endif
00348     bpf_u_int32 buf[MAXDLBUF];
00349     char dname[100];
00350 #ifndef HAVE_DEV_DLPI
00351     char dname2[100];
00352 #endif
00353     int status = PCAP_ERROR;
00354 
00355 #ifdef HAVE_DEV_DLPI
00356     /*
00357     ** Remove any "/dev/" on the front of the device.
00358     */
00359     cp = strrchr(p->opt.source, '/');
00360     if (cp == NULL)
00361         strlcpy(dname, p->opt.source, sizeof(dname));
00362     else
00363         strlcpy(dname, cp + 1, sizeof(dname));
00364 
00365     /*
00366      * Split the device name into a device type name and a unit number;
00367      * chop off the unit number, so "dname" is just a device type name.
00368      */
00369     cp = split_dname(dname, &ppa, p->errbuf);
00370     if (cp == NULL) {
00371         status = PCAP_ERROR_NO_SUCH_DEVICE;
00372         goto bad;
00373     }
00374     *cp = '\0';
00375 
00376     /*
00377      * Use "/dev/dlpi" as the device.
00378      *
00379      * XXX - HP's DLPI Programmer's Guide for HP-UX 11.00 says that
00380      * the "dl_mjr_num" field is for the "major number of interface
00381      * driver"; that's the major of "/dev/dlpi" on the system on
00382      * which I tried this, but there may be DLPI devices that
00383      * use a different driver, in which case we may need to
00384      * search "/dev" for the appropriate device with that major
00385      * device number, rather than hardwiring "/dev/dlpi".
00386      */
00387     cp = "/dev/dlpi";
00388     if ((p->fd = open(cp, O_RDWR)) < 0) {
00389         if (errno == EPERM || errno == EACCES)
00390             status = PCAP_ERROR_PERM_DENIED;
00391         snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00392             "%s: %s", cp, pcap_strerror(errno));
00393         goto bad;
00394     }
00395 
00396 #ifdef DL_HP_RAWDLS
00397     /*
00398      * XXX - HP-UX 10.20 and 11.xx don't appear to support sending and
00399      * receiving packets on the same descriptor - you need separate
00400      * descriptors for sending and receiving, bound to different SAPs.
00401      *
00402      * If the open fails, we just leave -1 in "p->send_fd" and reject
00403      * attempts to send packets, just as if, in pcap-bpf.c, we fail
00404      * to open the BPF device for reading and writing, we just try
00405      * to open it for reading only and, if that succeeds, just let
00406      * the send attempts fail.
00407      */
00408     p->send_fd = open(cp, O_RDWR);
00409 #endif
00410 
00411     /*
00412      * Get a table of all PPAs for that device, and search that
00413      * table for the specified device type name and unit number.
00414      */
00415     ppa = get_dlpi_ppa(p->fd, dname, ppa, p->errbuf);
00416     if (ppa < 0) {
00417         status = ppa;
00418         goto bad;
00419     }
00420 #else
00421     /*
00422      * If the device name begins with "/", assume it begins with
00423      * the pathname of the directory containing the device to open;
00424      * otherwise, concatenate the device directory name and the
00425      * device name.
00426      */
00427     if (*p->opt.source == '/')
00428         strlcpy(dname, p->opt.source, sizeof(dname));
00429     else
00430         snprintf(dname, sizeof(dname), "%s/%s", PCAP_DEV_PREFIX,
00431             p->opt.source);
00432 
00433     /*
00434      * Get the unit number, and a pointer to the end of the device
00435      * type name.
00436      */
00437     cp = split_dname(dname, &ppa, p->errbuf);
00438     if (cp == NULL) {
00439         status = PCAP_ERROR_NO_SUCH_DEVICE;
00440         goto bad;
00441     }
00442 
00443     /*
00444      * Make a copy of the device pathname, and then remove the unit
00445      * number from the device pathname.
00446      */
00447     strlcpy(dname2, dname, sizeof(dname));
00448     *cp = '\0';
00449 
00450     /* Try device without unit number */
00451     if ((p->fd = open(dname, O_RDWR)) < 0) {
00452         if (errno != ENOENT) {
00453             if (errno == EACCES)
00454                 status = PCAP_ERROR_PERM_DENIED;
00455             snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "%s: %s", dname,
00456                 pcap_strerror(errno));
00457             goto bad;
00458         }
00459 
00460         /* Try again with unit number */
00461         if ((p->fd = open(dname2, O_RDWR)) < 0) {
00462             if (errno == ENOENT) {
00463                 status = PCAP_ERROR_NO_SUCH_DEVICE;
00464 
00465                 /*
00466                  * We provide an error message even
00467                  * for this error, for diagnostic
00468                  * purposes (so that, for example,
00469                  * the app can show the message if the
00470                  * user requests it).
00471                  *
00472                  * In it, we just report "No DLPI device
00473                  * found" with the device name, so people
00474                  * don't get confused and think, for example,
00475                  * that if they can't capture on "lo0"
00476                  * on Solaris the fix is to change libpcap
00477                  * (or the application that uses it) to
00478                  * look for something other than "/dev/lo0",
00479                  * as the fix is to look for an operating
00480                  * system other than Solaris - you just
00481                  * *can't* capture on a loopback interface
00482                  * on Solaris, the lack of a DLPI device
00483                  * for the loopback interface is just a
00484                  * symptom of that inability.
00485                  */
00486                 snprintf(p->errbuf, PCAP_ERRBUF_SIZE,
00487                     "%s: No DLPI device found", p->opt.source);
00488             } else {
00489                 if (errno == EACCES)
00490                     status = PCAP_ERROR_PERM_DENIED;
00491                 snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "%s: %s",
00492                     dname2, pcap_strerror(errno));
00493             }
00494             goto bad;
00495         }
00496         /* XXX Assume unit zero */
00497         ppa = 0;
00498     }
00499 #endif
00500 
00501     /*
00502     ** Attach if "style 2" provider
00503     */
00504     if (dlinforeq(p->fd, p->errbuf) < 0 ||
00505         dlinfoack(p->fd, (char *)buf, p->errbuf) < 0)
00506         goto bad;
00507     infop = &(MAKE_DL_PRIMITIVES(buf))->info_ack;
00508 #ifdef HAVE_SOLARIS
00509     if (infop->dl_mac_type == DL_IPATM)
00510         isatm = 1;
00511 #endif
00512     if (infop->dl_provider_style == DL_STYLE2) {
00513         status = dl_doattach(p->fd, ppa, p->errbuf);
00514         if (status < 0)
00515             goto bad;
00516 #ifdef DL_HP_RAWDLS
00517         if (p->send_fd >= 0) {
00518             if (dl_doattach(p->send_fd, ppa, p->errbuf) < 0)
00519                 goto bad;
00520         }
00521 #endif
00522     }
00523 
00524     if (p->opt.rfmon) {
00525         /*
00526          * This device exists, but we don't support monitor mode
00527          * any platforms that support DLPI.
00528          */
00529         status = PCAP_ERROR_RFMON_NOTSUP;
00530         goto bad;
00531     }
00532 
00533 #ifdef HAVE_DLPI_PASSIVE
00534     /*
00535      * Enable Passive mode to be able to capture on aggregated link.
00536      * Not supported in all Solaris versions.
00537      */
00538     dlpassive(p->fd, p->errbuf);
00539 #endif
00540     /*
00541     ** Bind (defer if using HP-UX 9 or HP-UX 10.20 or later, totally
00542     ** skip if using SINIX)
00543     */
00544 #if !defined(HAVE_HPUX9) && !defined(HAVE_HPUX10_20_OR_LATER) && !defined(sinix)
00545 #ifdef _AIX
00546     /*
00547     ** AIX.
00548     ** According to IBM's AIX Support Line, the dl_sap value
00549     ** should not be less than 0x600 (1536) for standard Ethernet.
00550     ** However, we seem to get DL_BADADDR - "DLSAP addr in improper
00551     ** format or invalid" - errors if we use 1537 on the "tr0"
00552     ** device, which, given that its name starts with "tr" and that
00553     ** it's IBM, probably means a Token Ring device.  (Perhaps we
00554     ** need to use 1537 on "/dev/dlpi/en" because that device is for
00555     ** D/I/X Ethernet, the "SAP" is actually an Ethernet type, and
00556     ** it rejects invalid Ethernet types.)
00557     **
00558     ** So if 1537 fails, we try 2, as Hyung Sik Yoon of IBM Korea
00559     ** says that works on Token Ring (he says that 0 does *not*
00560     ** work; perhaps that's considered an invalid LLC SAP value - I
00561     ** assume the SAP value in a DLPI bind is an LLC SAP for network
00562     ** types that use 802.2 LLC).
00563     */
00564     if ((dlbindreq(p->fd, 1537, p->errbuf) < 0 &&
00565          dlbindreq(p->fd, 2, p->errbuf) < 0) ||
00566          dlbindack(p->fd, (char *)buf, p->errbuf, NULL) < 0)
00567         goto bad;
00568 #elif defined(DL_HP_RAWDLS)
00569     /*
00570     ** HP-UX 10.0x and 10.1x.
00571     */

00573         goto bad;

00575         /*
00576         ** XXX - if this fails, just close send_fd and
00577         ** set it to -1, so that you can't send but can
00578         ** still receive?
00579         */
00580         if (dl_dohpuxbind(p->send_fd, p->errbuf) < 0)
00581             goto bad;
00582     }
00583 #else /* neither AIX nor HP-UX */
00584     /*
00585     ** Not Sinix, and neither AIX nor HP-UX - Solaris, and any other
00586     ** OS using DLPI.
00587     **/
00588     if (dlbindreq(p->fd, 0, p->errbuf) < 0 ||
00589         dlbindack(p->fd, (char *)buf, p->errbuf, NULL) < 0)
00590         goto bad;
00591 #endif /* AIX vs. HP-UX vs. other */
00592 #endif /* !HP-UX 9 and !HP-UX 10.20 or later and !SINIX */
00593

00596         /*
00597         ** Have to turn on some special ATM promiscuous mode
00598         ** for SunATM.

00766 /*
00767  * Split a device name into a device type name and a unit number;
00768  * return the a pointer to the beginning of the unit number, which
00769  * is the end of the device type name, and set "*unitp" to the unit
00770  * number.
00771  *
00772  * Returns NULL on error, and fills "ebuf" with an error message.
00773  */
00774 static char *
00775 split_dname(char *device, int *unitp, char *ebuf)
00776 {
00777     char *cp;
00778     char *eos;
00779     long unit;
00780 
00781     /*
00782      * Look for a number at the end of the device name string.
00783      */
00784     cp = device + strlen(device) - 1;
00785     if (*cp < '0' || *cp > '9') {
00786         snprintf(ebuf, PCAP_ERRBUF_SIZE, "%s missing unit number",
00787             device);
00788         return (NULL);
00789     }
00790 
00791     /* Digits at end of string are unit number */
00792     while (cp-1 >= device && *(cp-1) >= '0' && *(cp-1) <= '9')
00793         cp--;
00794 
00795     errno = 0;
00796     unit = strtol(cp, &eos, 10);
00797     if (*eos != '\0') {
00798         snprintf(ebuf, PCAP_ERRBUF_SIZE, "%s bad unit number", device);
00799         return (NULL);
00800     }
00801     if (errno == ERANGE || unit > INT_MAX) {
00802         snprintf(ebuf, PCAP_ERRBUF_SIZE, "%s unit number too large",
00803             device);
00804         return (NULL);
00805     }
00806     if (unit < 0) {
00807         snprintf(ebuf, PCAP_ERRBUF_SIZE, "%s unit number is negative",
00808             device);
00809         return (NULL);
00810     }
00811     *unitp = (int)unit;
00812     return (cp);
00813 }
