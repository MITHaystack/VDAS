00138 egpnrprint(register const struct egp_packet *egp)
00139 {
00140     register const u_int8_t *cp;
00141     u_int32_t addr;
00142     register u_int32_t net;
00143     register u_int netlen;
00144     int gateways, distances, networks;
00145     int t_gateways;
00146     const char *comma;
00147 
00148     addr = egp->egp_sourcenet;
00149     if (IN_CLASSA(addr)) {
00150         net = addr & IN_CLASSA_NET;
00151         netlen = 1;
00152     } else if (IN_CLASSB(addr)) {
00153         net = addr & IN_CLASSB_NET;
00154         netlen = 2;
00155     } else if (IN_CLASSC(addr)) {
00156         net = addr & IN_CLASSC_NET;
00157         netlen = 3;
00158     } else {
00159         net = 0;
00160         netlen = 0;
00161     }
00162     cp = (u_int8_t *)(egp + 1);
00163 
00164     t_gateways = egp->egp_intgw + egp->egp_extgw;
00165     for (gateways = 0; gateways < t_gateways; ++gateways) {
00166         /* Pickup host part of gateway address */
00167         addr = 0;
00168         TCHECK2(cp[0], 4 - netlen);
00169         switch (netlen) {
00170 
00171         case 1:
00172             addr = *cp++;
00173             /* fall through */
00174         case 2:
00175             addr = (addr << 8) | *cp++;
00176             /* fall through */
00177         case 3:
00178             addr = (addr << 8) | *cp++;
00179         }
00180         addr |= net;
00181         TCHECK2(cp[0], 1);
00182         distances = *cp++;
00183         printf(" %s %s ",
00184                gateways < (int)egp->egp_intgw ? "int" : "ext",
00185                ipaddr_string(&addr));
00186 
00187         comma = "";
00188         putchar('(');
00189         while (--distances >= 0) {
00190             TCHECK2(cp[0], 2);
00191             printf("%sd%d:", comma, (int)*cp++);
00192             comma = ", ";
00193             networks = *cp++;
00194             while (--networks >= 0) {
00195                 /* Pickup network number */
00196                 TCHECK2(cp[0], 1);
00197                 addr = (u_int32_t)*cp++ << 24;
00198                 if (IN_CLASSB(addr)) {
00199                     TCHECK2(cp[0], 1);
00200                     addr |= (u_int32_t)*cp++ << 16;
00201                 } else if (!IN_CLASSA(addr)) {
00202                     TCHECK2(cp[0], 2);
00203                     addr |= (u_int32_t)*cp++ << 16;
00204                     addr |= (u_int32_t)*cp++ << 8;
00205                 }
00206                 printf(" %s", ipaddr_string(&addr));
00207             }
00208         }
00209         putchar(')');
00210     }
00211     return;
00212 trunc:
00213     fputs("[|]", stdout);
00214 }
