00066 void
00067 decnet_print(register const u_char *ap, register u_int length,
00068          register u_int caplen)
00069 {
00070     register const union routehdr *rhp;
00071     register int mflags;
00072     int dst, src, hops;
00073     u_int nsplen, pktlen;
00074     const u_char *nspp;
00075 
00076     if (length < sizeof(struct shorthdr)) {
00077         (void)printf("[|decnet]");
00078         return;
00079     }
00080 
00081     TCHECK2(*ap, sizeof(short));
00082     pktlen = EXTRACT_LE_16BITS(ap);
00083     if (pktlen < sizeof(struct shorthdr)) {
00084         (void)printf("[|decnet]");
00085         return;
00086     }
00087     if (pktlen > length) {
00088         (void)printf("[|decnet]");
00089         return;
00090     }
00091     length = pktlen;
00092 
00093     rhp = (const union routehdr *)&(ap[sizeof(short)]);
00094     TCHECK(rhp->rh_short.sh_flags);
00095     mflags = EXTRACT_LE_8BITS(rhp->rh_short.sh_flags);
00096 
00097     if (mflags & RMF_PAD) {
00098         /* pad bytes of some sort in front of message */
00099         u_int padlen = mflags & RMF_PADMASK;
00100         if (vflag)
00101         (void) printf("[pad:%d] ", padlen);
00102         if (length < padlen + 2) {
00103         (void)printf("[|decnet]");
00104         return;
00105         }
00106         TCHECK2(ap[sizeof(short)], padlen);
00107         ap += padlen;
00108         length -= padlen;
00109         caplen -= padlen;
00110         rhp = (const union routehdr *)&(ap[sizeof(short)]);
00111         mflags = EXTRACT_LE_8BITS(rhp->rh_short.sh_flags);
00112     }
00113 
00114     if (mflags & RMF_FVER) {
00115         (void) printf("future-version-decnet");
00116         default_print(ap, min(length, caplen));
00117         return;
00118     }
00119 
00120     /* is it a control message? */
00121     if (mflags & RMF_CTLMSG) {
00122         if (!print_decnet_ctlmsg(rhp, length, caplen))
00123             goto trunc;
00124         return;
00125     }
00126 
00127     switch (mflags & RMF_MASK) {
00128     case RMF_LONG:
00129         if (length < sizeof(struct longhdr)) {
00130         (void)printf("[|decnet]");
00131         return;
00132         }
00133         TCHECK(rhp->rh_long);
00134         dst =
00135         EXTRACT_LE_16BITS(rhp->rh_long.lg_dst.dne_remote.dne_nodeaddr);
00136         src =
00137         EXTRACT_LE_16BITS(rhp->rh_long.lg_src.dne_remote.dne_nodeaddr);
00138         hops = EXTRACT_LE_8BITS(rhp->rh_long.lg_visits);
00139         nspp = &(ap[sizeof(short) + sizeof(struct longhdr)]);
00140         nsplen = length - sizeof(struct longhdr);
00141         break;
00142     case RMF_SHORT:
00143         TCHECK(rhp->rh_short);
00144         dst = EXTRACT_LE_16BITS(rhp->rh_short.sh_dst);
00145         src = EXTRACT_LE_16BITS(rhp->rh_short.sh_src);
00146         hops = (EXTRACT_LE_8BITS(rhp->rh_short.sh_visits) & VIS_MASK)+1;
00147         nspp = &(ap[sizeof(short) + sizeof(struct shorthdr)]);
00148         nsplen = length - sizeof(struct shorthdr);
00149         break;
00150     default:
00151         (void) printf("unknown message flags under mask");
00152         default_print((u_char *)ap, min(length, caplen));
00153         return;
00154     }
00155 
00156     (void)printf("%s > %s %d ",
00157             dnaddr_string(src), dnaddr_string(dst), pktlen);
00158     if (vflag) {
00159         if (mflags & RMF_RQR)
00160         (void)printf("RQR ");
00161         if (mflags & RMF_RTS)
00162         (void)printf("RTS ");
00163         if (mflags & RMF_IE)
00164         (void)printf("IE ");
00165         (void)printf("%d hops ", hops);
00166     }
00167 
00168     if (!print_nsp(nspp, nsplen))
00169         goto trunc;
00170     return;
00171 
00172 trunc:
00173     (void)printf("[|decnet]");
00174     return;
00175 }

00226         (void)printf("src %s data %o", dnaddr_string(src), other);

00247                 length - sizeof(struct l2rout));
00248         break;
00249     case RMF_RHELLO:
00250         (void)printf("router-hello ");
00251         if (length < sizeof(struct rhellomsg))
00252         goto trunc;
00253         TCHECK(cmp->cm_rhello);
00254         vers = EXTRACT_LE_8BITS(cmp->cm_rhello.rh_vers);
00255         eco = EXTRACT_LE_8BITS(cmp->cm_rhello.rh_eco);
00256         ueco = EXTRACT_LE_8BITS(cmp->cm_rhello.rh_ueco);
00257         memcpy((char *)&srcea, (char *)&(cmp->cm_rhello.rh_src),
00258         sizeof(srcea));

00267             vers, eco, ueco, dnaddr_string(src),

00272     case RMF_EHELLO:

00279         ueco = EXTRACT_LE_8BITS(cmp->cm_ehello.eh_ueco);
00280         memcpy((char *)&srcea, (char *)&(cmp->cm_ehello.eh_src),
00281         sizeof(srcea));
00282         src = EXTRACT_LE_16BITS(srcea.dne_remote.dne_nodeaddr);
00283         info = EXTRACT_LE_8BITS(cmp->cm_ehello.eh_info);
00284         blksize = EXTRACT_LE_16BITS(cmp->cm_ehello.eh_blksize);
00285         /*seed*/
00286         memcpy((char *)&rtea, (char *)&(cmp->cm_ehello.eh_router),
00287         sizeof(rtea));
00288         dst = EXTRACT_LE_16BITS(rtea.dne_remote.dne_nodeaddr);
00289         hello = EXTRACT_LE_16BITS(cmp->cm_ehello.eh_hello);
00290         other = EXTRACT_LE_8BITS(cmp->cm_ehello.eh_data);
00291         print_i_info(info);
00292         (void)printf(
00293     "vers %d eco %d ueco %d src %s blksize %d rtr %s hello %d data %o",
00294             vers, eco, ueco, dnaddr_string(src),
00295             blksize, dnaddr_string(dst), hello, other);
00296         ret = 1;
00297         break;
00298 
00299     default:
00300         (void)printf("unknown control message");
00301         default_print((u_char *)rhp, min(length, caplen));
00302         ret = 1;
00303         break;
00304     }
00305     return (ret);
00306 
00307 trunc:
00308     return (0);

00327 static int
00328 print_l1_routes(const char *rp, u_int len)
00329 {
00330     int count;
00331     int id;
00332     int info;
00333 
00334     /* The last short is a checksum */
00335     while (len > (3 * sizeof(short))) {
00336         TCHECK2(*rp, 3 * sizeof(short));
00337         count = EXTRACT_LE_16BITS(rp);
00338         if (count > 1024)
00339         return (1); /* seems to be bogus from here on */
00340         rp += sizeof(short);
00341         len -= sizeof(short);
00342         id = EXTRACT_LE_16BITS(rp);
00343         rp += sizeof(short);
00344         len -= sizeof(short);
00345         info = EXTRACT_LE_16BITS(rp);
00346         rp += sizeof(short);
00347         len -= sizeof(short);
00348         (void)printf("{ids %d-%d cost %d hops %d} ", id, id + count,
00349                 RI_COST(info), RI_HOPS(info));
00350     }
00351     return (1);
00352 
00353 trunc:
00354     return (0);
00355 }

00413 print_nsp(const u_char *nspp, u_int nsplen)
00414 {
00415     const struct nsphdr *nsphp = (struct nsphdr *)nspp;
00416     int dst, src, flags;
00417 
00418     if (nsplen < sizeof(struct nsphdr))
00419         goto trunc;
00420     TCHECK(*nsphp);
00421     flags = EXTRACT_LE_8BITS(nsphp->nh_flags);
00422     dst = EXTRACT_LE_16BITS(nsphp->nh_dst);
00423     src = EXTRACT_LE_16BITS(nsphp->nh_src);
00424 
00425     switch (flags & NSP_TYPEMASK) {
00426     case MFT_DATA:
00427         switch (flags & NSP_SUBMASK) {
00428         case MFS_BOM:
00429         case MFS_MOM:
00430         case MFS_EOM:
00431         case MFS_BOM+MFS_EOM:
00432         printf("data %d>%d ", src, dst);
00433         {
00434             struct seghdr *shp = (struct seghdr *)nspp;
00435             int ack;
00436 #ifdef  PRINT_NSPDATA
00437             u_char *dp;
00438 #endif
00439             u_int data_off = sizeof(struct minseghdr);
00440 
00441             if (nsplen < data_off)
00442             goto trunc;
00443             TCHECK(shp->sh_seq[0]);
00444             ack = EXTRACT_LE_16BITS(shp->sh_seq[0]);
00445             if (ack & SGQ_ACK) {    /* acknum field */
00446             if ((ack & SGQ_NAK) == SGQ_NAK)
00447                 (void)printf("nak %d ", ack & SGQ_MASK);
00448             else
00449                 (void)printf("ack %d ", ack & SGQ_MASK);
00450             data_off += sizeof(short);
00451             if (nsplen < data_off)
00452                 goto trunc;
00453             TCHECK(shp->sh_seq[1]);
00454                 ack = EXTRACT_LE_16BITS(shp->sh_seq[1]);
00455             if (ack & SGQ_OACK) {   /* ackoth field */
00456                 if ((ack & SGQ_ONAK) == SGQ_ONAK)
00457                 (void)printf("onak %d ", ack & SGQ_MASK);
00458                 else
00459                 (void)printf("oack %d ", ack & SGQ_MASK);
00460                 data_off += sizeof(short);
00461                 if (nsplen < data_off)
00462                 goto trunc;
00463                 TCHECK(shp->sh_seq[2]);
00464                 ack = EXTRACT_LE_16BITS(shp->sh_seq[2]);
00465             }
00466             }
00467             (void)printf("seg %d ", ack & SGQ_MASK);
00468 #ifdef  PRINT_NSPDATA
00469             if (nsplen > data_off) {
00470             dp = &(nspp[data_off]);
00471             TCHECK2(*dp, nsplen - data_off);
00472             pdata(dp, nsplen - data_off);
00473             }
00474 #endif
00475         }
00476         break;
00477         case MFS_ILS+MFS_INT:
00478         printf("intr ");
00479         {
00480             struct seghdr *shp = (struct seghdr *)nspp;
00481             int ack;
00482 #ifdef  PRINT_NSPDATA
00483             u_char *dp;
00484 #endif
00485             u_int data_off = sizeof(struct minseghdr);
00486 
00487             if (nsplen < data_off)
00488             goto trunc;
00489             TCHECK(shp->sh_seq[0]);
00490             ack = EXTRACT_LE_16BITS(shp->sh_seq[0]);
00491             if (ack & SGQ_ACK) {    /* acknum field */
00492             if ((ack & SGQ_NAK) == SGQ_NAK)
00493                 (void)printf("nak %d ", ack & SGQ_MASK);
00494             else
00495                 (void)printf("ack %d ", ack & SGQ_MASK);
00496             data_off += sizeof(short);
00497             if (nsplen < data_off)
00498                 goto trunc;
00499             TCHECK(shp->sh_seq[1]);
00500                 ack = EXTRACT_LE_16BITS(shp->sh_seq[1]);
00501             if (ack & SGQ_OACK) {   /* ackdat field */
00502                 if ((ack & SGQ_ONAK) == SGQ_ONAK)
00503                 (void)printf("nakdat %d ", ack & SGQ_MASK);
00504                 else
00505                 (void)printf("ackdat %d ", ack & SGQ_MASK);
00506                 data_off += sizeof(short);
00507                 if (nsplen < data_off)
00508                 goto trunc;
00509                 TCHECK(shp->sh_seq[2]);
00510                 ack = EXTRACT_LE_16BITS(shp->sh_seq[2]);
00511             }
00512             }
00513             (void)printf("seg %d ", ack & SGQ_MASK);
00514 #ifdef  PRINT_NSPDATA
00515             if (nsplen > data_off) {
00516             dp = &(nspp[data_off]);
00517             TCHECK2(*dp, nsplen - data_off);
00518             pdata(dp, nsplen - data_off);
00519             }
00520 #endif
00521         }
00522         break;
00523         case MFS_ILS:
00524         (void)printf("link-service %d>%d ", src, dst);
00525         {
00526             struct seghdr *shp = (struct seghdr *)nspp;
00527             struct lsmsg *lsmp =
00528             (struct lsmsg *)&(nspp[sizeof(struct seghdr)]);
00529             int ack;
00530             int lsflags, fcval;
00531 
00532             if (nsplen < sizeof(struct seghdr) + sizeof(struct lsmsg))
00533             goto trunc;
00534             TCHECK(shp->sh_seq[0]);
00535             ack = EXTRACT_LE_16BITS(shp->sh_seq[0]);
00536             if (ack & SGQ_ACK) {    /* acknum field */
00537             if ((ack & SGQ_NAK) == SGQ_NAK)
00538                 (void)printf("nak %d ", ack & SGQ_MASK);
00539             else
00540                 (void)printf("ack %d ", ack & SGQ_MASK);
00541             TCHECK(shp->sh_seq[1]);
00542                 ack = EXTRACT_LE_16BITS(shp->sh_seq[1]);
00543             if (ack & SGQ_OACK) {   /* ackdat field */
00544                 if ((ack & SGQ_ONAK) == SGQ_ONAK)
00545                 (void)printf("nakdat %d ", ack & SGQ_MASK);
00546                 else
00547                 (void)printf("ackdat %d ", ack & SGQ_MASK);
00548                 TCHECK(shp->sh_seq[2]);
00549                 ack = EXTRACT_LE_16BITS(shp->sh_seq[2]);
00550             }
00551             }
00552             (void)printf("seg %d ", ack & SGQ_MASK);
00553             TCHECK(*lsmp);
00554             lsflags = EXTRACT_LE_8BITS(lsmp->ls_lsflags);
00555             fcval = EXTRACT_LE_8BITS(lsmp->ls_fcval);
00556             switch (lsflags & LSI_MASK) {
