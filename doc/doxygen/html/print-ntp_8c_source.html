00085 ntp_print(register const u_char *cp, u_int length)
00086 {
00087     register const struct ntpdata *bp;
00088     int mode, version, leapind;
00089 
00090     bp = (struct ntpdata *)cp;
00091 
00092     TCHECK(bp->status);
00093 
00094     version = (int)(bp->status & VERSIONMASK) >> 3;
00095     printf("NTPv%d", version);
00096 
00097     mode = bp->status & MODEMASK;
00098         if (!vflag) {
00099             printf (", %s, length %u",
00100                     tok2str(ntp_mode_values, "Unknown mode", mode),
00101                     length);
00102             return;
00103         }
00104         
00105         printf (", length %u\n\t%s",
00106                 length,
00107                 tok2str(ntp_mode_values, "Unknown mode", mode));        
00108 
00109     leapind = bp->status & LEAPMASK;
00110         printf (", Leap indicator: %s (%u)",
00111                 tok2str(ntp_leapind_values, "Unknown", leapind),
00112                 leapind);
00113 
00114     TCHECK(bp->stratum);
00115     printf(", Stratum %u (%s)",     
00116         bp->stratum,
00117         tok2str(ntp_stratum_values, (bp->stratum >=2 && bp->stratum<=15) ? "secondary reference" : "reserved", bp->stratum));
00118 
00119     TCHECK(bp->ppoll);
00120     printf(", poll %us", bp->ppoll);
00121 
00122     /* Can't TCHECK bp->precision bitfield so bp->distance + 0 instead */
00123     TCHECK2(bp->root_delay, 0);
00124     printf(", precision %d", bp->precision);
00125 
00126     TCHECK(bp->root_delay);
00127     fputs("\n\tRoot Delay: ", stdout);
00128     p_sfix(&bp->root_delay);
00129 
00130     TCHECK(bp->root_dispersion);
00131     fputs(", Root dispersion: ", stdout);
00132     p_sfix(&bp->root_dispersion);
00133 
00134     TCHECK(bp->refid);
00135     fputs(", Reference-ID: ", stdout);
00136     /* Interpretation depends on stratum */
00137     switch (bp->stratum) {
00138 
00139     case UNSPECIFIED:
00140         printf("(unspec)");
00141         break;
00142 
00143     case PRIM_REF:
00144         if (fn_printn((u_char *)&(bp->refid), 4, snapend))
00145             goto trunc;
00146         break;
00147 
00148     case INFO_QUERY:
00149         printf("%s INFO_QUERY", ipaddr_string(&(bp->refid)));
00150         /* this doesn't have more content */
00151         return;
00152 
00153     case INFO_REPLY:
00154         printf("%s INFO_REPLY", ipaddr_string(&(bp->refid)));
00155         /* this is too complex to be worth printing */
00156         return;
00157 
00158     default:
00159         printf("%s", ipaddr_string(&(bp->refid)));
00160         break;
00161     }
00162 
00163     TCHECK(bp->ref_timestamp);
00164     fputs("\n\t  Reference Timestamp:  ", stdout);
00165     p_ntp_time(&(bp->ref_timestamp));
00166 
00167     TCHECK(bp->org_timestamp);
00168     fputs("\n\t  Originator Timestamp: ", stdout);
00169     p_ntp_time(&(bp->org_timestamp));
00170 
00171     TCHECK(bp->rec_timestamp);
00172     fputs("\n\t  Receive Timestamp:    ", stdout);
00173     p_ntp_time(&(bp->rec_timestamp));
00174 
00175     TCHECK(bp->xmt_timestamp);
00176     fputs("\n\t  Transmit Timestamp:   ", stdout);
00177     p_ntp_time(&(bp->xmt_timestamp));
00178 
00179     fputs("\n\t    Originator - Receive Timestamp:  ", stdout);
00180     p_ntp_delta(&(bp->org_timestamp), &(bp->rec_timestamp));
00181 
00182     fputs("\n\t    Originator - Transmit Timestamp: ", stdout);
00183     p_ntp_delta(&(bp->org_timestamp), &(bp->xmt_timestamp));
00184 
00185     if ( (sizeof(struct ntpdata) - length) == 16) {     /* Optional: key-id */
00186         TCHECK(bp->key_id);
00187         printf("\n\tKey id: %u", bp->key_id);
00188     } else if ( (sizeof(struct ntpdata) - length) == 0) {   /* Optional: key-id + authentication */
00189         TCHECK(bp->key_id);
00190         printf("\n\tKey id: %u", bp->key_id);
00191         TCHECK2(bp->message_digest, sizeof (bp->message_digest));
00192                 printf("\n\tAuthentication: %08x%08x%08x%08x",
00193                        EXTRACT_32BITS(bp->message_digest),
00194                        EXTRACT_32BITS(bp->message_digest + 4),
00195                        EXTRACT_32BITS(bp->message_digest + 8),
00196                        EXTRACT_32BITS(bp->message_digest + 12));
00197         }
00198     return;
00199 
00200 trunc:

00202 }

00205 p_sfix(register const struct s_fixedpt *sfp)
00206 {
00207     register int i;
00208     register int f;
00209     register float ff;
00210 
00211     i = EXTRACT_16BITS(&sfp->int_part);
00212     f = EXTRACT_16BITS(&sfp->fraction);
00213     ff = f / 65536.0;   /* shift radix point by 16 bits */
00214     f = ff * 1000000.0; /* Treat fraction as parts per million */
00215     printf("%d.%06d", i, f);
00216 }

00221 p_ntp_time(register const struct l_fixedpt *lfp)
00222 {
00223     register int32_t i;
00224     register u_int32_t uf;
00225     register u_int32_t f;
00226     register float ff;
00227 
00228     i = EXTRACT_32BITS(&lfp->int_part);
00229     uf = EXTRACT_32BITS(&lfp->fraction);
00230     ff = uf;
00231     if (ff < 0.0)       /* some compilers are buggy */
00232         ff += FMAXINT;
00233     ff = ff / FMAXINT;  /* shift radix point by 32 bits */
00234     f = ff * 1000000000.0;  /* treat fraction as parts per billion */
00235     printf("%u.%09d", i, f);

00239      * print the time in human-readable format.

00253 /* Prints time difference between *lfp and *olfp */
00254 static void
00255 p_ntp_delta(register const struct l_fixedpt *olfp,
00256         register const struct l_fixedpt *lfp)
00257 {
00258     register int32_t i;
00259     register u_int32_t u, uf;
00260     register u_int32_t ou, ouf;
00261     register u_int32_t f;
00262     register float ff;
00263     int signbit;
00264 
00265     u = EXTRACT_32BITS(&lfp->int_part);
00266     ou = EXTRACT_32BITS(&olfp->int_part);
00267     uf = EXTRACT_32BITS(&lfp->fraction);
00268     ouf = EXTRACT_32BITS(&olfp->fraction);
00269     if (ou == 0 && ouf == 0) {
00270         p_ntp_time(lfp);
00271         return;
00272     }
00273 
00274     i = u - ou;
00275 
00276     if (i > 0) {        /* new is definitely greater than old */
00277         signbit = 0;
00278         f = uf - ouf;
00279         if (ouf > uf)   /* must borrow from high-order bits */
00280             i -= 1;
00281     } else if (i < 0) { /* new is definitely less than old */
00282         signbit = 1;
00283         f = ouf - uf;
00284         if (uf > ouf)   /* must carry into the high-order bits */
00285             i += 1;
00286         i = -i;
00287     } else {        /* int_part is zero */
00288         if (uf > ouf) {
00289             signbit = 0;
00290             f = uf - ouf;
00291         } else {
00292             signbit = 1;
00293             f = ouf - uf;
00294         }
00295     }
00296 
00297     ff = f;
00298     if (ff < 0.0)       /* some compilers are buggy */
00299         ff += FMAXINT;
00300     ff = ff / FMAXINT;  /* shift radix point by 32 bits */
00301     f = ff * 1000000000.0;  /* treat fraction as parts per billion */
00302     if (signbit)
00303         putchar('-');
00304     else
00305         putchar('+');
00306     printf("%d.%09d", i, f);
00307 }
