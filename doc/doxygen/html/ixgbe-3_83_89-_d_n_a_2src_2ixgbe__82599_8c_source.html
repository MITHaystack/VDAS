00157 s32 ixgbe_setup_sfp_modules_82599(struct ixgbe_hw *hw)
00158 {
00159     s32 ret_val = 0;
00160     u32 reg_anlp1 = 0;
00161     u32 i = 0;
00162     u16 list_offset, data_offset, data_value;
00163 
00164     if (hw->phy.sfp_type != ixgbe_sfp_type_unknown) {
00165         ixgbe_init_mac_link_ops_82599(hw);
00166 
00167         hw->phy.ops.reset = NULL;
00168 
00169         ret_val = ixgbe_get_sfp_init_sequence_offsets(hw, &list_offset,
00170                                                       &data_offset);
00171         if (ret_val != 0)
00172             goto setup_sfp_out;
00173 
00174         /* PHY config will finish before releasing the semaphore */
00175         ret_val = hw->mac.ops.acquire_swfw_sync(hw, 
00176                                                 IXGBE_GSSR_MAC_CSR_SM);
00177         if (ret_val != 0) {
00178             ret_val = IXGBE_ERR_SWFW_SYNC;
00179             goto setup_sfp_out;
00180         }
00181 
00182         hw->eeprom.ops.read(hw, ++data_offset, &data_value);
00183         while (data_value != 0xffff) {
00184             IXGBE_WRITE_REG(hw, IXGBE_CORECTL, data_value);
00185             IXGBE_WRITE_FLUSH(hw);
00186             hw->eeprom.ops.read(hw, ++data_offset, &data_value);
00187         }
00188 
00189         /* Release the semaphore */
00190         hw->mac.ops.release_swfw_sync(hw, IXGBE_GSSR_MAC_CSR_SM);
00191         /* Delay obtaining semaphore again to allow FW access */
00192         msleep(hw->eeprom.semaphore_delay);
00193 
00194         /* Now restart DSP by setting Restart_AN and clearing LMS */
00195         IXGBE_WRITE_REG(hw, IXGBE_AUTOC, ((IXGBE_READ_REG(hw,
00196                         IXGBE_AUTOC) & ~IXGBE_AUTOC_LMS_MASK) |
00197                         IXGBE_AUTOC_AN_RESTART));
00198 
00199         /* Wait for AN to leave state 0 */
00200         for (i = 0; i < 10; i++) {
00201             msleep(4);
00202             reg_anlp1 = IXGBE_READ_REG(hw, IXGBE_ANLP1);
00203             if (reg_anlp1 & IXGBE_ANLP1_AN_STATE_MASK)
00204                 break;
00205         }
00206         if (!(reg_anlp1 & IXGBE_ANLP1_AN_STATE_MASK)) {
00207             hw_dbg(hw, "sfp module setup not complete\n");
00208             ret_val = IXGBE_ERR_SFP_SETUP_NOT_COMPLETE;
00209             goto setup_sfp_out;
00210         }
00211 
00212         /* Restart DSP by setting Restart_AN and return to SFI mode */
00213         IXGBE_WRITE_REG(hw, IXGBE_AUTOC, (IXGBE_READ_REG(hw,
00214                         IXGBE_AUTOC) | IXGBE_AUTOC_LMS_10G_SERIAL |
00215                         IXGBE_AUTOC_AN_RESTART));
00216     }
00217 
00218 setup_sfp_out:
00219     return ret_val;
00220 }

00307     /* Check if 1G SFP module. */
00308     if (hw->phy.sfp_type == ixgbe_sfp_type_1g_cu_core0 ||
00309         hw->phy.sfp_type == ixgbe_sfp_type_1g_cu_core1) {
00310         *speed = IXGBE_LINK_SPEED_1GB_FULL;
00311         *negotiation = true;

00313     }
00314 
00315     /*
00316      * Determine link capabilities based on the stored value of AUTOC,

00321         autoc = hw->mac.orig_autoc;

00323         autoc = IXGBE_READ_REG(hw, IXGBE_AUTOC);

00326     case IXGBE_AUTOC_LMS_1G_LINK_NO_AN:
00327         *speed = IXGBE_LINK_SPEED_1GB_FULL;
00328         *negotiation = false;
00329         break;

00332         *speed = IXGBE_LINK_SPEED_10GB_FULL;
00333         *negotiation = false;
00334         break;

00337         *speed = IXGBE_LINK_SPEED_1GB_FULL;
00338         *negotiation = true;
00339         break;
00340

00342         *speed = IXGBE_LINK_SPEED_10GB_FULL;
00343         *negotiation = false;

00566     u32 i = 0;
00567     bool link_up = false;
00568     bool negotiation;
00569 
00570     /* Mask off requested but non-supported speeds */

00572     if (status != 0)
00573         return status;
00574 
00575     speed &= link_speed;
00576 
00577     /*
00578      * Try each speed one by one, highest priority first.  We do this in
00579      * software because 10gb fiber doesn't support speed autonegotiation.
00580      */

00582         speedcnt++;
00583         highest_link_speed = IXGBE_LINK_SPEED_10GB_FULL;
00584 
00585         /* If we already have link at this speed, just jump out */
00586         status = ixgbe_check_link(hw, &link_speed, &link_up, false);

00588             return status;
00589

00591             goto out;

00594         esdp_reg |= (IXGBE_ESDP_SDP5_DIR | IXGBE_ESDP_SDP5);
00595         IXGBE_WRITE_REG(hw, IXGBE_ESDP, esdp_reg);
00596         IXGBE_WRITE_FLUSH(hw);
00597 
00598         /* Allow module to change analog characteristics (1G->10G) */
00599         msleep(40);
00600

00602                         IXGBE_LINK_SPEED_10GB_FULL,
00603                         autoneg,
00604                         autoneg_wait_to_complete);
00605         if (status != 0)
00606             return status;
00607 
00608         /* Flap the tx laser if it has not already been done */
00609         ixgbe_flap_tx_laser(hw);
00610

00612          * Wait for the controller to acquire link.  Per IEEE 802.3ap,
00613          * Section 73.10.2, we may have to wait up to 500ms if KR is
00614          * attempted.  82599 uses the same timing for 10g SFI.
00615          */

00618             msleep(100);

00625 
00626             if (link_up)

00628         }
00629     }
00630

00635 
00636         /* If we already have link at this speed, just jump out */
00637         status = ixgbe_check_link(hw, &link_speed, &link_up, false);
00638         if (status != 0)
00639             return status;
00640 
00641         if ((link_speed == IXGBE_LINK_SPEED_1GB_FULL) && link_up)
00642             goto out;
00643 
00644         /* Set the module link speed */
00645         esdp_reg &= ~IXGBE_ESDP_SDP5;
00646         esdp_reg |= IXGBE_ESDP_SDP5_DIR;
00647         IXGBE_WRITE_REG(hw, IXGBE_ESDP, esdp_reg);
00648         IXGBE_WRITE_FLUSH(hw);
00649 
00650         /* Allow module to change analog characteristics (10G->1G) */
00651         msleep(40);
00652 
00653         status = ixgbe_setup_mac_link_82599(hw,
00654                             IXGBE_LINK_SPEED_1GB_FULL,
00655                             autoneg,
00656                             autoneg_wait_to_complete);
00657         if (status != 0)
00658             return status;
00659 
00660         /* Flap the tx laser if it has not already been done */
00661         ixgbe_flap_tx_laser(hw);
00662 
00663         /* Wait for the link partner to also set speed */
00664         msleep(100);
00665 
00666         /* If we have link, just jump out */
00667         status = ixgbe_check_link(hw, &link_speed, &link_up, false);
00668         if (status != 0)
00669             return status;
00670 
00671         if (link_up)
00672             goto out;
00673     }

00676      * We didn't get link.  Configure back to the highest speed we tried,
00677      * (if there was more than one).  We call ourselves back with just the

00679      */

00681         status = ixgbe_setup_mac_link_multispeed_fiber(hw,
00682                 highest_link_speed, autoneg, autoneg_wait_to_complete);
00683

00686     hw->phy.autoneg_advertised = 0;

00689         hw->phy.autoneg_advertised |= IXGBE_LINK_SPEED_10GB_FULL;
00690 
00691     if (speed & IXGBE_LINK_SPEED_1GB_FULL)
00692         hw->phy.autoneg_advertised |= IXGBE_LINK_SPEED_1GB_FULL;
00693 
00694     return status;
00695 }

00706 s32 ixgbe_setup_mac_link_smartspeed(struct ixgbe_hw *hw,
00707                      ixgbe_link_speed speed, bool autoneg,
00708                      bool autoneg_wait_to_complete)
00709 {
00710     s32 status = 0;
00711     ixgbe_link_speed link_speed = IXGBE_LINK_SPEED_UNKNOWN;
00712     s32 i, j;
00713     bool link_up = false;
00714     u32 autoc_reg = IXGBE_READ_REG(hw, IXGBE_AUTOC);
00715 
00716      /* Set autoneg_advertised value based on input link speed */
00717     hw->phy.autoneg_advertised = 0;
00718 
00719     if (speed & IXGBE_LINK_SPEED_10GB_FULL)
00720         hw->phy.autoneg_advertised |= IXGBE_LINK_SPEED_10GB_FULL;
00721 
00722     if (speed & IXGBE_LINK_SPEED_1GB_FULL)
00723         hw->phy.autoneg_advertised |= IXGBE_LINK_SPEED_1GB_FULL;
00724 
00725     if (speed & IXGBE_LINK_SPEED_100_FULL)
00726         hw->phy.autoneg_advertised |= IXGBE_LINK_SPEED_100_FULL;
00727 
00728     /*
00729      * Implement Intel SmartSpeed algorithm.  SmartSpeed will reduce the
00730      * autoneg advertisement if link is unable to be established at the
00731      * highest negotiated rate.  This can sometimes happen due to integrity
00732      * issues with the physical media connection.
00733      */
00734 
00735     /* First, try to get link with full advertisement */
00736     hw->phy.smart_speed_active = false;
00737     for (j = 0; j < IXGBE_SMARTSPEED_MAX_RETRIES; j++) {
00738         status = ixgbe_setup_mac_link_82599(hw, speed, autoneg,
00739                             autoneg_wait_to_complete);
00740         if (status != 0)
00741             goto out;
00742 
00743         /*
00744          * Wait for the controller to acquire link.  Per IEEE 802.3ap,
00745          * Section 73.10.2, we may have to wait up to 500ms if KR is
00746          * attempted, or 200ms if KX/KX4/BX/BX4 is attempted, per
00747          * Table 9 in the AN MAS.
00748          */
00749         for (i = 0; i < 5; i++) {
00750             msleep(100);
00751 
00752             /* If we have link, just jump out */
00753             status = ixgbe_check_link(hw, &link_speed, &link_up,
00754                           false);
00755             if (status != 0)
00756                 goto out;
00757 
00758             if (link_up)
00759                 goto out;
00760         }
00761     }
00762 
00763     /*
00764      * We didn't get link.  If we advertised KR plus one of KX4/KX
00765      * (or BX4/BX), then disable KR and try again.
00766      */
00767     if (((autoc_reg & IXGBE_AUTOC_KR_SUPP) == 0) ||
00768         ((autoc_reg & IXGBE_AUTOC_KX4_KX_SUPP_MASK) == 0))
00769         goto out;
00770 
00771     /* Turn SmartSpeed on to disable KR support */
00772     hw->phy.smart_speed_active = true;
00773     status = ixgbe_setup_mac_link_82599(hw, speed, autoneg,
00774                         autoneg_wait_to_complete);
00775     if (status != 0)
00776         goto out;
00777 
00778     /*
00779      * Wait for the controller to acquire link.  600ms will allow for
00780      * the AN link_fail_inhibit_timer as well for multiple cycles of
00781      * parallel detect, both 10g and 1g. This allows for the maximum
00782      * connect attempts as defined in the AN MAS table 73-7.
00783      */
00784     for (i = 0; i < 6; i++) {
00785         msleep(100);
00786 
00787         /* If we have link, just jump out */
00788         status = ixgbe_check_link(hw, &link_speed, &link_up, false);
00789         if (status != 0)
00790             goto out;
00791 
00792         if (link_up)
00793             goto out;
00794     }
00795 
00796     /* We didn't get link.  Turn SmartSpeed back off. */
00797     hw->phy.smart_speed_active = false;
00798     status = ixgbe_setup_mac_link_82599(hw, speed, autoneg,
00799                         autoneg_wait_to_complete);
00800 
00801 out:
00802     if (link_up && (link_speed == IXGBE_LINK_SPEED_1GB_FULL))
00803         hw_dbg(hw, "Smartspeed has downgraded the link speed "
00804         "from the maximum advertised\n");
00805     return status;
00806 }

00841         status = IXGBE_ERR_LINK_SETUP;

00894                 link_mode == IXGBE_AUTOC_LMS_KX4_KX_KR_1G_AN ||
00895                 link_mode == IXGBE_AUTOC_LMS_KX4_KX_KR_SGMII) {

00899                            IXGBE_READ_REG(hw, IXGBE_LINKS);

00901                         break;

00953 s32 ixgbe_reset_hw_82599(struct ixgbe_hw *hw)
00954 {
00955     s32 status = 0;
00956     u32 ctrl;
00957     u32 i;
00958     u32 autoc;
00959     u32 autoc2;
00960 
00961     /* Call adapter stop to disable tx/rx and clear interrupts */
00962     hw->mac.ops.stop_adapter(hw);
00963 
00964     /* PHY ops must be identified and initialized prior to reset */
00965 
00966     /* Identify PHY and related function pointers */
00967     status = hw->phy.ops.init(hw);
00968 
00969     if (status == IXGBE_ERR_SFP_NOT_SUPPORTED)
00970         goto reset_hw_out;
00971 
00972     /* Setup SFP module if there is one present. */
00973     if (hw->phy.sfp_setup_needed) {
00974         status = hw->mac.ops.setup_sfp(hw);
00975         hw->phy.sfp_setup_needed = false;
00976     }
00977 
00978     if (status == IXGBE_ERR_SFP_NOT_SUPPORTED)
00979         goto reset_hw_out;
00980 
00981     /* Reset PHY */
00982     if (hw->phy.reset_disable == false && hw->phy.ops.reset != NULL)
00983         hw->phy.ops.reset(hw);
00984 
00985     /*
00986      * Prevent the PCI-E bus from from hanging by disabling PCI-E master
00987      * access and verify no pending requests before reset
00988      */
00989     ixgbe_disable_pcie_master(hw);
00990 
00991 mac_reset_top:
00992     /*
00993      * Issue global reset to the MAC.  This needs to be a SW reset.
00994      * If link reset is used, it might reset the MAC when mng is using it
00995      */
00996     ctrl = IXGBE_READ_REG(hw, IXGBE_CTRL);
00997     IXGBE_WRITE_REG(hw, IXGBE_CTRL, (ctrl | IXGBE_CTRL_RST));
00998     IXGBE_WRITE_FLUSH(hw);
00999 
01000     /* Poll for reset bit to self-clear indicating reset is complete */
01001     for (i = 0; i < 10; i++) {
01002         udelay(1);
01003         ctrl = IXGBE_READ_REG(hw, IXGBE_CTRL);
01004         if (!(ctrl & IXGBE_CTRL_RST))
01005             break;
01006     }
01007     if (ctrl & IXGBE_CTRL_RST) {
01008         status = IXGBE_ERR_RESET_FAILED;
01009         hw_dbg(hw, "Reset polling failed to complete.\n");
01010     }
01011 
01012     /*
01013      * Double resets are required for recovery from certain error
01014      * conditions.  Between resets, it is necessary to stall to allow time
01015      * for any pending HW events to complete.  We use 1usec since that is
01016      * what is needed for ixgbe_disable_pcie_master().  The second reset
01017      * then clears out any effects of those events.
01018      */
01019     if (hw->mac.flags & IXGBE_FLAGS_DOUBLE_RESET_REQUIRED) {
01020         hw->mac.flags &= ~IXGBE_FLAGS_DOUBLE_RESET_REQUIRED;
01021         udelay(1);
01022         goto mac_reset_top;
01023     }
01024 
01025     msleep(50);
01026 
01027     /*
01028      * Store the original AUTOC/AUTOC2 values if they have not been
01029      * stored off yet.  Otherwise restore the stored original
01030      * values since the reset operation sets back to defaults.
01031      */
01032     autoc = IXGBE_READ_REG(hw, IXGBE_AUTOC);
01033     autoc2 = IXGBE_READ_REG(hw, IXGBE_AUTOC2);
01034     if (hw->mac.orig_link_settings_stored == false) {
01035         hw->mac.orig_autoc = autoc;
01036         hw->mac.orig_autoc2 = autoc2;
01037         hw->mac.orig_link_settings_stored = true;
01038     } else {
01039         if (autoc != hw->mac.orig_autoc)
01040             IXGBE_WRITE_REG(hw, IXGBE_AUTOC, (hw->mac.orig_autoc |
01041                     IXGBE_AUTOC_AN_RESTART));
01042 
01043         if ((autoc2 & IXGBE_AUTOC2_UPPER_MASK) !=
01044             (hw->mac.orig_autoc2 & IXGBE_AUTOC2_UPPER_MASK)) {
01045             autoc2 &= ~IXGBE_AUTOC2_UPPER_MASK;
01046             autoc2 |= (hw->mac.orig_autoc2 &
01047                        IXGBE_AUTOC2_UPPER_MASK);
01048             IXGBE_WRITE_REG(hw, IXGBE_AUTOC2, autoc2);
01049         }
01050     }
01051 
01052     /* Store the permanent mac address */
01053     hw->mac.ops.get_mac_addr(hw, hw->mac.perm_addr);
01054 
01055     /*
01056      * Store MAC address from RAR0, clear receive address registers, and
01057      * clear the multicast table.  Also reset num_rar_entries to 128,
01058      * since we modify this value when programming the SAN MAC address.
01059      */
01060     hw->mac.num_rar_entries = 128;
01061     hw->mac.ops.init_rx_addrs(hw);
01062 
01063     /* Store the permanent SAN mac address */
01064     hw->mac.ops.get_san_mac_addr(hw, hw->mac.san_addr);
01065 
01066     /* Add the SAN MAC address to the RAR only if it's a valid address */
01067     if (ixgbe_validate_mac_addr(hw->mac.san_addr) == 0) {
01068         hw->mac.ops.set_rar(hw, hw->mac.num_rar_entries - 1,
01069                             hw->mac.san_addr, 0, IXGBE_RAH_AV);
01070 
01071         /* Reserve the last RAR for the SAN MAC address */
01072         hw->mac.num_rar_entries--;
01073     }
01074 
01075     /* Store the alternative WWNN/WWPN prefix */
01076     hw->mac.ops.get_wwn_prefix(hw, &hw->mac.wwnn_prefix,
01077                                    &hw->mac.wwpn_prefix);
01078 
01079 reset_hw_out:
01080     return status;
01081 }

01087 s32 ixgbe_reinit_fdir_tables_82599(struct ixgbe_hw *hw)
01088 {
01089     int i;
01090     u32 fdirctrl = IXGBE_READ_REG(hw, IXGBE_FDIRCTRL);
01091     fdirctrl &= ~IXGBE_FDIRCTRL_INIT_DONE;
01092 
01093     /*
01094      * Before starting reinitialization process,
01095      * FDIRCMD.CMD must be zero.
01096      */
01097     for (i = 0; i < IXGBE_FDIRCMD_CMD_POLL; i++) {
01098         if (!(IXGBE_READ_REG(hw, IXGBE_FDIRCMD) &
01099               IXGBE_FDIRCMD_CMD_MASK))
01100             break;
01101         udelay(10);
01102     }
01103     if (i >= IXGBE_FDIRCMD_CMD_POLL) {
01104         hw_dbg(hw, "Flow Director previous command isn't complete, "
01105                  "aborting table re-initialization. \n");
01106         return IXGBE_ERR_FDIR_REINIT_FAILED;
01107     }
01108 
01109     IXGBE_WRITE_REG(hw, IXGBE_FDIRFREE, 0);
01110     IXGBE_WRITE_FLUSH(hw);
01111     /*
01112      * 82599 adapters flow director init flow cannot be restarted,
01113      * Workaround 82599 silicon errata by performing the following steps
01114      * before re-writing the FDIRCTRL control register with the same value.
01115      * - write 1 to bit 8 of FDIRCMD register &
01116      * - write 0 to bit 8 of FDIRCMD register
01117      */
01118     IXGBE_WRITE_REG(hw, IXGBE_FDIRCMD,
01119                     (IXGBE_READ_REG(hw, IXGBE_FDIRCMD) |
01120                      IXGBE_FDIRCMD_CLEARHT));
01121     IXGBE_WRITE_FLUSH(hw);
01122     IXGBE_WRITE_REG(hw, IXGBE_FDIRCMD,
01123                     (IXGBE_READ_REG(hw, IXGBE_FDIRCMD) &
01124                      ~IXGBE_FDIRCMD_CLEARHT));
01125     IXGBE_WRITE_FLUSH(hw);
01126     /*
01127      * Clear FDIR Hash register to clear any leftover hashes
01128      * waiting to be programmed.
01129      */
01130     IXGBE_WRITE_REG(hw, IXGBE_FDIRHASH, 0x00);
01131     IXGBE_WRITE_FLUSH(hw);
01132 
01133     IXGBE_WRITE_REG(hw, IXGBE_FDIRCTRL, fdirctrl);
01134     IXGBE_WRITE_FLUSH(hw);
01135 
01136     /* Poll init-done after we write FDIRCTRL register */
01137     for (i = 0; i < IXGBE_FDIR_INIT_DONE_POLL; i++) {
01138         if (IXGBE_READ_REG(hw, IXGBE_FDIRCTRL) &
01139                            IXGBE_FDIRCTRL_INIT_DONE)
01140             break;
01141         udelay(10);
01142     }
01143     if (i >= IXGBE_FDIR_INIT_DONE_POLL) {
01144         hw_dbg(hw, "Flow Director Signature poll time exceeded!\n");
01145         return IXGBE_ERR_FDIR_REINIT_FAILED;

01163 static s32 ixgbe_set_fdir_rxpba_82599(struct ixgbe_hw *hw, const u32 pballoc)
01164 {
01165     u32 fdir_pbsize = hw->mac.rx_pb_size << IXGBE_RXPBSIZE_SHIFT;
01166     u32 current_rxpbsize = 0;
01167     int i;
01168 
01169     /* reserve space for Flow Director filters */
01170     switch (pballoc) {
01171     case IXGBE_FDIR_PBALLOC_256K:
01172         fdir_pbsize -= 256 << IXGBE_RXPBSIZE_SHIFT;
01173         break;
01174     case IXGBE_FDIR_PBALLOC_128K:
01175         fdir_pbsize -= 128 << IXGBE_RXPBSIZE_SHIFT;
01176         break;
01177     case IXGBE_FDIR_PBALLOC_64K:
01178         fdir_pbsize -= 64 << IXGBE_RXPBSIZE_SHIFT;
01179         break;
01180     case IXGBE_FDIR_PBALLOC_NONE:
01181     default:
01182         return IXGBE_ERR_PARAM;
01183     }
01184 
01185     /* determine current RX packet buffer size */
01186     for (i = 0; i < 8; i++)
01187         current_rxpbsize += IXGBE_READ_REG(hw, IXGBE_RXPBSIZE(i));
01188 
01189     /* if there is already room for the filters do nothing */
01190     if (current_rxpbsize <= fdir_pbsize)
01191         return 0;
01192 
01193     if (current_rxpbsize > hw->mac.rx_pb_size) {
01194         /*
01195          * if rxpbsize is greater than max then HW max the Rx buffer
01196          * sizes are unconfigured or misconfigured since HW default is
01197          * to give the full buffer to each traffic class resulting in
01198          * the total size being buffer size 8x actual size
01199          *
01200          * This assumes no DCB since the RXPBSIZE registers appear to
01201          * be unconfigured.
01202          */
01203         IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(0), fdir_pbsize);
01204         for (i = 1; i < 8; i++)
01205             IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(i), 0);
01206     } else {
01207         /*
01208          * Since the Rx packet buffer appears to have already been
01209          * configured we need to shrink each packet buffer by enough
01210          * to make room for the filters.  As such we take each rxpbsize
01211          * value and multiply it by a fraction representing the size
01212          * needed over the size we currently have.
01213          *
01214          * We need to reduce fdir_pbsize and current_rxpbsize to
01215          * 1/1024 of their original values in order to avoid
01216          * overflowing the u32 being used to store rxpbsize.
01217          */
01218         fdir_pbsize >>= IXGBE_RXPBSIZE_SHIFT;
01219         current_rxpbsize >>= IXGBE_RXPBSIZE_SHIFT;
01220         for (i = 0; i < 8; i++) {
01221             u32 rxpbsize = IXGBE_READ_REG(hw, IXGBE_RXPBSIZE(i));
01222             rxpbsize *= fdir_pbsize;
01223             rxpbsize /= current_rxpbsize;
01224             IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(i), rxpbsize);
01225         }
01226     }
01227 
01228     return 0;
01229 }

01372 u32 ixgbe_atr_compute_sig_hash_82599(union ixgbe_atr_hash_dword input,
01373                                      union ixgbe_atr_hash_dword common)
01374 {
01375     u32 hi_hash_dword, lo_hash_dword, flow_vm_vlan;
01376     u32 sig_hash = 0, bucket_hash = 0, common_hash = 0;
01377 
01378     /* record the flow_vm_vlan bits as they are a key part to the hash */
01379     flow_vm_vlan = IXGBE_NTOHL(input.dword);
01380 
01381     /* generate common hash dword */
01382     hi_hash_dword = IXGBE_NTOHL(common.dword);
01383 
01384     /* low dword is word swapped version of common */
01385     lo_hash_dword = (hi_hash_dword >> 16) | (hi_hash_dword << 16);
01386 
01387     /* apply flow ID/VM pool/VLAN ID bits to hash words */
01388     hi_hash_dword ^= flow_vm_vlan ^ (flow_vm_vlan >> 16);
01389 
01390     /* Process bits 0 and 16 */
01391     IXGBE_COMPUTE_SIG_HASH_ITERATION(0);
01392 
01393     /*
01394      * apply flow ID/VM pool/VLAN ID bits to lo hash dword, we had to
01395      * delay this because bit 0 of the stream should not be processed
01396      * so we do not add the vlan until after bit 0 was processed
01397      */

01399 
01400     /* Process remaining 30 bit of the key */
01401     IXGBE_COMPUTE_SIG_HASH_ITERATION(1);
01402     IXGBE_COMPUTE_SIG_HASH_ITERATION(2);
01403     IXGBE_COMPUTE_SIG_HASH_ITERATION(3);
01404     IXGBE_COMPUTE_SIG_HASH_ITERATION(4);
01405     IXGBE_COMPUTE_SIG_HASH_ITERATION(5);
01406     IXGBE_COMPUTE_SIG_HASH_ITERATION(6);

01415     IXGBE_COMPUTE_SIG_HASH_ITERATION(15);
