00058  * RFC3032: MPLS label stack encoding
00059  */
00060 void
00061 mpls_print(const u_char *bp, u_int length)
00062 {
00063     const u_char *p;
00064     u_int32_t label_entry;
00065         u_int16_t label_stack_depth = 0;
00066 
00067     p = bp;
00068     printf("MPLS");
00069     do {
00070         TCHECK2(*p, sizeof(label_entry));
00071         label_entry = EXTRACT_32BITS(p);
00072         printf("%s(label %u",
00073                        label_stack_depth ? "\n\t" : " ",
00074                        MPLS_LABEL(label_entry));
00075                 label_stack_depth++;
00076         if (vflag &&
00077             MPLS_LABEL(label_entry) < sizeof(mpls_labelname) / sizeof(mpls_labelname[0]))
00078             printf(" (%s)", mpls_labelname[MPLS_LABEL(label_entry)]);
00079         printf(", exp %u", MPLS_EXP(label_entry));
00080         if (MPLS_STACK(label_entry))
00081             printf(", [S]");
00082         printf(", ttl %u)", MPLS_TTL(label_entry));
00083 
00084         p += sizeof(label_entry);
00085     } while (!MPLS_STACK(label_entry));
00086 
00087     switch (MPLS_LABEL(label_entry)) {
00088     case 0: /* IPv4 explicit NULL label */
00089         case 3: /* IPv4 implicit NULL label */
00090                 if (vflag>0) {
00091                         printf("\n\t");
00092                         ip_print(gndo, p, length - (p - bp));
00093                 }
00094                 else printf(", IP, length: %u",length);
00095         break;
00096 #ifdef INET6
00097     case 2: /* IPv6 explicit NULL label */
00098                 if (vflag>0) {
00099                         printf("\n\t");
00100                         ip6_print(p, length - (p - bp));
00101                 }
00102                 else printf(", IPv6, length: %u",length);
00103         break;
00104 #endif
00105     default:
00106         /*
00107          * Generally there's no indication of protocol in MPLS label
00108          * encoding, however draft-hsmit-isis-aal5mux-00.txt describes
00109                  * a technique that looks at the first payload byte if the BOS (bottom of stack)
00110                  * bit is set and tries to determine the network layer protocol
00111                  * 0x45-0x4f is IPv4
00112                  * 0x60-0x6f is IPv6
00113                  * 0x81-0x83 is OSI (CLNP,ES-IS,IS-IS)
00114                  * this technique is sometimes known as NULL encapsulation
00115                  * and decoding is particularly useful for control-plane traffic [BGP]
00116                  * which cisco by default sends MPLS encapsulated
00117          */
00118 
00119                 if (MPLS_STACK(label_entry)) { /* only do this if the stack bit is set */
00120                     switch(*p) {
00121                     case 0x45:
00122                     case 0x46:
00123                     case 0x47:
00124                     case 0x48:
00125                     case 0x49:
00126                     case 0x4a:
00127                     case 0x4b:
00128                     case 0x4c:
00129                     case 0x4d:
00130                     case 0x4e:
00131                     case 0x4f:
00132                 if (vflag>0) {
00133                             printf("\n\t");
00134                             ip_print(gndo, p, length - (p - bp));
00135                 }
00136                         else printf(", IP, length: %u",length);
00137                         break;
00138 #ifdef INET6
00139                     case 0x60:
00140                     case 0x61:
00141                     case 0x62:
00142                     case 0x63:
00143                     case 0x64:
00144                     case 0x65:
00145                     case 0x66:
00146                     case 0x67:
00147                     case 0x68:
00148                     case 0x69:
00149                     case 0x6a:
00150                     case 0x6b:
00151                     case 0x6c:
00152                     case 0x6d:
00153                     case 0x6e:
00154                     case 0x6f:
00155                 if (vflag>0) {
00156                             printf("\n\t");
00157                             ip6_print(p, length - (p - bp));
00158                             }
00159             else printf(", IPv6, length: %u",length);
00160                         break;
00161 #endif
00162                     case 0x81:
00163                     case 0x82:
00164                     case 0x83:
00165                 if (vflag>0) {
00166                             printf("\n\t");
00167                             isoclns_print(p, length - (p - bp), length - (p - bp));
00168                 }
00169             else printf(", OSI, length: %u",length);
00170                         break;
00171                     default:
00172                         /* ok bail out - we did not figure out what it is*/
00173                         break;
00174                     }
00175                 }
00176                 return;
00177     }
00178 
00179 trunc:
00180     printf("[|MPLS]");
00181 }
