00041 s32 ixgbe_dcb_get_tc_stats_82599(struct ixgbe_hw *hw,
00042                                  struct ixgbe_hw_stats *stats,
00043                                  u8 tc_count)
00044 {
00045     int tc;
00046 
00047     if (tc_count > MAX_TRAFFIC_CLASS)
00048         return DCB_ERR_PARAM;
00049     /* Statistics pertaining to each traffic class */
00050     for (tc = 0; tc < tc_count; tc++) {
00051         /* Transmitted Packets */
00052         stats->qptc[tc] += IXGBE_READ_REG(hw, IXGBE_QPTC(tc));
00053         /* Transmitted Bytes (read low first to prevent missed carry) */
00054         stats->qbtc[tc] += IXGBE_READ_REG(hw, IXGBE_QBTC_L(tc));
00055         stats->qbtc[tc] +=
00056             (((u64)(IXGBE_READ_REG(hw, IXGBE_QBTC_H(tc)))) << 32);
00057         /* Received Packets */
00058         stats->qprc[tc] += IXGBE_READ_REG(hw, IXGBE_QPRC(tc));
00059         /* Received Bytes (read low first to prevent missed carry) */
00060         stats->qbrc[tc] += IXGBE_READ_REG(hw, IXGBE_QBRC_L(tc));
00061         stats->qbrc[tc] +=
00062             (((u64)(IXGBE_READ_REG(hw, IXGBE_QBRC_H(tc)))) << 32);
00063 
00064         /* Received Dropped Packet */
00065         stats->qprdc[tc] += IXGBE_READ_REG(hw, IXGBE_QPRDC(tc));
00066     }
00067 
00068     return 0;

00108     u32 rx_pb_size = hw->mac.rx_pb_size << IXGBE_RXPBSIZE_SHIFT;
00109     u32 rxpktsize;

00113 
00114     /* reserve space for Flow Director filters */
00115     switch (dcb_config->fdir_pballoc) {
00116     case IXGBE_FDIR_PBALLOC_256K:
00117         rx_pb_size -= 256 << IXGBE_RXPBSIZE_SHIFT;
00118         break;
00119     case IXGBE_FDIR_PBALLOC_128K:
00120         rx_pb_size -= 128 << IXGBE_RXPBSIZE_SHIFT;
00121         break;
00122     case IXGBE_FDIR_PBALLOC_64K:
00123         rx_pb_size -= 64 << IXGBE_RXPBSIZE_SHIFT;
00124         break;
00125     case IXGBE_FDIR_PBALLOC_NONE:
00126     default:
00127         /* do nothing */
00128         break;
00129     }
00130 
00131     /*
00132      * This really means configure the first half of the TCs
00133      * (Traffic Classes) to use 5/8 of the Rx packet buffer
00134      * space.  To determine the size of the buffer for each TC,
00135      * we are multiplying the average size by 5/4 and applying
00136      * it to half of the traffic classes.
00137      */
00138     if (dcb_config->rx_pba_cfg == pba_80_48) {
00139         rxpktsize = (rx_pb_size * 5) / (num_tcs * 4);
00140         rx_pb_size -= rxpktsize * (num_tcs / 2);
00141         for (; i < (num_tcs / 2); i++)
00142             IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(i), rxpktsize);
00143     }
00144 
00145     /* Divide the remaining Rx packet buffer evenly among the TCs */
00146     rxpktsize = rx_pb_size / (num_tcs - i);
00147     for (; i < num_tcs; i++)
00148         IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(i), rxpktsize);
00149 
00150     /*
00151      * Setup Tx packet buffer and threshold equally for all TCs
00152      * TXPBTHRESH register is set in K so divide by 1024 and subtract
00153      * 10 since the largest packet we support is just over 9K.
00154      */
00155     txpktsize = IXGBE_TXPBSIZE_MAX / num_tcs;
00156     txpbthresh = (txpktsize / 1024) - IXGBE_TXPKT_SIZE_MAX;
00157     for (i = 0; i < num_tcs; i++) {
00158         IXGBE_WRITE_REG(hw, IXGBE_TXPBSIZE(i), txpktsize);
00159         IXGBE_WRITE_REG(hw, IXGBE_TXPBTHRESH(i), txpbthresh);
00160     }
00161 
00162     /* Clear unused TCs, if any, to zero buffer size*/
00163     for (; i < MAX_TRAFFIC_CLASS; i++) {
00164         IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(i), 0);
00165         IXGBE_WRITE_REG(hw, IXGBE_TXPBSIZE(i), 0);
00166         IXGBE_WRITE_REG(hw, IXGBE_TXPBTHRESH(i), 0);
00167     }
00168 
00169     return 0;
00170 }

00184     u32    credit_refill = 0;
00185     u32    credit_max    = 0;
00186     u8     i             = 0;
00187     u8     j;
00188 
00189     /*
00190      * Disable the arbiter before changing parameters
00191      * (always enable recycle mode; WSP)
00192      */

00195

00200      * up_to_tc_bitmap value for that TC will be 11000000 in binary.
00201      */

00203     for (i = 0; i < MAX_TRAFFIC_CLASS; i++) {
00204         p = &dcb_config->tc_config[i].path[DCB_RX_CONFIG];
00205         for (j = 0; j < MAX_USER_PRIORITY; j++) {
00206             if (p->up_to_tc_bitmap & (1 << j))
00207                 reg |= (i << (j * IXGBE_RTRUP2TC_UP_SHIFT));
00208         }
00209     }
00210     IXGBE_WRITE_REG(hw, IXGBE_RTRUP2TC, reg);
00211 
00212     /* Configure traffic class credits and priority */
00213     for (i = 0; i < dcb_config->num_tcs.pg_tcs; i++) {
00214         p = &dcb_config->tc_config[i].path[DCB_RX_CONFIG];
00215 
00216         credit_refill = p->data_credits_refill;
00217         credit_max    = p->data_credits_max;
00218         reg = credit_refill | (credit_max << IXGBE_RTRPT4C_MCL_SHIFT);
00219 
00220         reg |= (u32)(p->bwg_id) << IXGBE_RTRPT4C_BWG_SHIFT;
00221 
00222         if (p->prio_type == prio_link)
00223             reg |= IXGBE_RTRPT4C_LSP;
00224 
00225         IXGBE_WRITE_REG(hw, IXGBE_RTRPT4C(i), reg);
00226     }
00227 
00228     /*
00229      * Configure Rx packet plane (recycle mode; WSP) and
00230      * enable arbiter
00231      */
00232     reg = IXGBE_RTRPCS_RRM | IXGBE_RTRPCS_RAC;
00233     IXGBE_WRITE_REG(hw, IXGBE_RTRPCS, reg);
00234
