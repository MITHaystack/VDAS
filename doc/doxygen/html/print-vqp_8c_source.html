00109 vqp_print(register const u_char *pptr, register u_int len) 
00110 {
00111     const struct vqp_common_header_t *vqp_common_header;
00112     const struct vqp_obj_tlv_t *vqp_obj_tlv;
00113 
00114     const u_char *tptr;
00115     u_int16_t vqp_obj_len;
00116     u_int32_t vqp_obj_type;
00117     int tlen;
00118     u_int8_t nitems;
00119 
00120     tptr=pptr;
00121     tlen = len;
00122     vqp_common_header = (const struct vqp_common_header_t *)pptr;
00123     TCHECK(*vqp_common_header);
00124 
00125     /*
00126      * Sanity checking of the header.
00127      */
00128     if (VQP_EXTRACT_VERSION(vqp_common_header->version) != VQP_VERSION) {
00129     printf("VQP version %u packet not supported",
00130                VQP_EXTRACT_VERSION(vqp_common_header->version));
00131     return;
00132     }
00133 
00134     /* in non-verbose mode just lets print the basic Message Type */
00135     if (vflag < 1) {
00136         printf("VQPv%u %s Message, error-code %s (%u), length %u",
00137                VQP_EXTRACT_VERSION(vqp_common_header->version),
00138                tok2str(vqp_msg_type_values, "unknown (%u)",vqp_common_header->msg_type),
00139                tok2str(vqp_error_code_values, "unknown (%u)",vqp_common_header->error_code),
00140            vqp_common_header->error_code,
00141                len);
00142         return;
00143     }
00144     
00145     /* ok they seem to want to know everything - lets fully decode it */
00146     nitems = vqp_common_header->nitems;
00147     printf("\n\tVQPv%u, %s Message, error-code %s (%u), seq 0x%08x, items %u, length %u",
00148            VQP_EXTRACT_VERSION(vqp_common_header->version),
00149        tok2str(vqp_msg_type_values, "unknown (%u)",vqp_common_header->msg_type),
00150        tok2str(vqp_error_code_values, "unknown (%u)",vqp_common_header->error_code),
00151        vqp_common_header->error_code,
00152            EXTRACT_32BITS(&vqp_common_header->sequence),
00153            nitems,
00154            len);
00155 
00156     /* skip VQP Common header */
00157     tptr+=sizeof(const struct vqp_common_header_t);
00158     tlen-=sizeof(const struct vqp_common_header_t);
00159 
00160     while (nitems > 0 && tlen > 0) {
00161 
00162         vqp_obj_tlv = (const struct vqp_obj_tlv_t *)tptr;
00163         vqp_obj_type = EXTRACT_32BITS(vqp_obj_tlv->obj_type);
00164         vqp_obj_len = EXTRACT_16BITS(vqp_obj_tlv->obj_length);
00165         tptr+=sizeof(struct vqp_obj_tlv_t);
00166         tlen-=sizeof(struct vqp_obj_tlv_t);
00167 
00168         printf("\n\t  %s Object (0x%08x), length %u, value: ",
00169                tok2str(vqp_obj_values, "Unknown", vqp_obj_type),
00170                vqp_obj_type, vqp_obj_len);
00171 
00172         /* basic sanity check */
00173         if (vqp_obj_type == 0 || vqp_obj_len ==0) {
00174             return;
00175         }
00176 
00177         /* did we capture enough for fully decoding the object ? */
00178         if (!TTEST2(*tptr, vqp_obj_len)) 
00179             goto trunc;
00180 
00181         switch(vqp_obj_type) {
00182     case VQP_OBJ_IP_ADDRESS:
00183             printf("%s (0x%08x)", ipaddr_string(tptr), EXTRACT_32BITS(tptr));
00184             break;
00185             /* those objects have similar semantics - fall through */
00186         case VQP_OBJ_PORT_NAME:
00187     case VQP_OBJ_VLAN_NAME:
00188     case VQP_OBJ_VTP_DOMAIN:
00189     case VQP_OBJ_ETHERNET_PKT:
00190             safeputs((const char *)tptr, vqp_obj_len);
00191             break;
00192             /* those objects have similar semantics - fall through */
00193     case VQP_OBJ_MAC_ADDRESS:
00194     case VQP_OBJ_MAC_NULL:
00195           printf("%s", etheraddr_string(tptr));
00196               break;
00197         default:
00198             if (vflag <= 1)
00199                 print_unknown_data(tptr, "\n\t    ", vqp_obj_len);
00200             break;
00201         }
00202     tptr += vqp_obj_len;
00203     tlen -= vqp_obj_len;
00204     nitems--;
00205     }
00206     return;
00207 trunc:
00208     printf("\n\t[|VQP]");
00209 }
