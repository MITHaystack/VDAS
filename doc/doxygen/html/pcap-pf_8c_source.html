00103         cc = read(pc->fd, (char *)pc->buffer + pc->offset, pc->bufsize);
00104         if (cc < 0) {

00106                 return (0);
00107             if (errno == EINVAL &&
00108                 lseek(pc->fd, 0L, SEEK_CUR) + pc->bufsize < 0) {
00109                 /*
00110                  * Due to a kernel bug, after 2^31 bytes,
00111                  * the kernel file offset overflows and
00112                  * read fails with EINVAL. The lseek()
00113                  * to 0 will fix things.
00114                  */

00119                 pcap_strerror(errno));

00123     } else

00125     /*
00126      * Loop through each packet.
00127      */
00128     n = 0;
00129 #ifdef PCAP_FDDIPAD
00130     pad = pc->fddipad;
00131 #endif
00132     while (cc > 0) {
00133         /*
00134          * Has "pcap_breakloop()" been called?
00135          * If so, return immediately - if we haven't read any
00136          * packets, clear the flag and return -2 to indicate
00137          * that we were told to break out of the loop, otherwise
00138          * leave the flag set, so that the *next* call will break
00139          * out of the loop without having read any packets, and
00140          * return the number of packets we've processed so far.
00141          */
00142         if (pc->break_loop) {
00143             if (n == 0) {
00144                 pc->break_loop = 0;
00145                 return (-2);
00146             } else {
00147                 pc->cc = cc;
00148                 pc->bp = bp;
00149                 return (n);
00150             }
00151         }
00152         if (cc < sizeof(*sp)) {
00153             snprintf(pc->errbuf, sizeof(pc->errbuf),
00154                 "pf short read (%d)", cc);
00155             return (-1);
00156         }
00157 #ifdef LBL_ALIGN
00158         if ((long)bp & 3) {
00159             sp = &stamp;
00160             memcpy((char *)sp, (char *)bp, sizeof(*sp));
00161         } else
00162 #endif
00163             sp = (struct enstamp *)bp;
00164         if (sp->ens_stamplen != sizeof(*sp)) {
00165             snprintf(pc->errbuf, sizeof(pc->errbuf),
00166                 "pf short stamplen (%d)",
00167                 sp->ens_stamplen);
00168             return (-1);
00169         }
00170 
00171         p = bp + sp->ens_stamplen;
00172         buflen = sp->ens_count;
00173         if (buflen > pc->snapshot)
00174             buflen = pc->snapshot;
00175

00179         bp += inc;

00184             pc->md.OrigMissed = pc->md.TotMissed;

00186         /*

00188          * in kernel, no need to do it now - we already know
00189          * the packet passed the filter.
00190          *
00191 #ifdef PCAP_FDDIPAD

00193          * that pc->fddipad was the amount of padding

00196          * skipping that padding.
00197 #endif
00198          */

00200             bpf_filter(pc->fcode.bf_insns, p, sp->ens_count, buflen)) {

00202             pc->md.TotAccepted++;

00204 #ifdef PCAP_FDDIPAD
00205             h.len = sp->ens_count - pad;
00206 #else

00208 #endif
00209 #ifdef PCAP_FDDIPAD
00210             p += pad;
00211             buflen -= pad;
00212 #endif
00213             h.caplen = buflen;
00214             (*callback)(user, &h, p);
00215             if (++n >= cnt && cnt > 0) {
00216                 pc->cc = cc;
00217                 pc->bp = bp;
00218                 return (n);
00219             }
00220         }
00221     }
00222     pc->cc = 0;

00263      *
00264      * If packet filtering is not being done in the kernel:
00265      *
00266      *  "ps_recv" counts only packets that passed the filter.
00267      *
00268      *  "ps_drop" counts packets that were dropped because the
00269      *  input queue was full, regardless of whether they passed
00270      *  the userland filter.
00271      *
00272      *  "ps_ifdrop" counts packets dropped by the network
00273      *  inteface (regardless of whether they would have passed
00274      *  the input filter, of course).
00275      *
00276      * These statistics don't include packets not yet read from
00277      * the kernel by libpcap, but they may include packets not
00278      * yet read from libpcap by the application.
00279      */
00280     ps->ps_recv = p->md.TotAccepted;
00281     ps->ps_drop = p->md.TotDrops;

00520 static int
00521 pcap_setfilter_pf(pcap_t *p, struct bpf_program *fp)
00522 {
00523     struct bpf_version bv;
00524 
00525     /*
00526      * See if BIOCVERSION works.  If not, we assume the kernel doesn't
00527      * support BPF-style filters (it's not documented in the bpf(7)
00528      * or packetfiler(7) man pages, but the code used to fail if
00529      * BIOCSETF worked but BIOCVERSION didn't, and I've seen it do
00530      * kernel filtering in DU 4.0, so presumably BIOCVERSION works
00531      * there, at least).
00532      */
00533     if (ioctl(p->fd, BIOCVERSION, (caddr_t)&bv) >= 0) {
00534         /*
00535          * OK, we have the version of the BPF interpreter;
00536          * is it the same major version as us, and the same
00537          * or better minor version?
00538          */
00539         if (bv.bv_major == BPF_MAJOR_VERSION &&
00540             bv.bv_minor >= BPF_MINOR_VERSION) {
00541             /*
00542              * Yes.  Try to install the filter.
00543              */
00544             if (ioctl(p->fd, BIOCSETF, (caddr_t)fp) < 0) {
00545                 snprintf(p->errbuf, sizeof(p->errbuf),
00546                     "BIOCSETF: %s", pcap_strerror(errno));
00547                 return (-1);
00548             }
00549 
00550             /*
00551              * OK, that succeeded.  We're doing filtering in
00552              * the kernel.  (We assume we don't have a
00553              * userland filter installed - that'd require
00554              * a previous version check to have failed but
00555              * this one to succeed.)
00556              *
00557              * XXX - this message should be supplied to the
00558              * application as a warning of some sort,
00559              * except that if it's a GUI application, it's
00560              * not clear that it should be displayed in
00561              * a window to annoy the user.
00562              */
00563             fprintf(stderr, "tcpdump: Using kernel BPF filter\n");
00564             p->md.use_bpf = 1;
00565 
00566             /*
00567              * Discard any previously-received packets,
00568              * as they might have passed whatever filter
00569              * was formerly in effect, but might not pass
00570              * this filter (BIOCSETF discards packets buffered
00571              * in the kernel, so you can lose packets in any
00572              * case).
00573              */
00574             p->cc = 0;
00575             return (0);
00576         }
00577 
00578         /*
00579          * We can't use the kernel's BPF interpreter; don't give
00580          * up, just log a message and be inefficient.
00581          *
00582          * XXX - this should really be supplied to the application
00583          * as a warning of some sort.
00584          */
00585         fprintf(stderr,
00586         "tcpdump: Requires BPF language %d.%d or higher; kernel is %d.%d\n",
00587             BPF_MAJOR_VERSION, BPF_MINOR_VERSION,
00588             bv.bv_major, bv.bv_minor);
00589     }
00590 
00591     /*
00592      * We couldn't do filtering in the kernel; do it in userland.
00593      */
00594     if (install_bpf_program(p, fp) < 0)
00595         return (-1);
00596 
00597     /*
00598      * XXX - this message should be supplied by the application as
00599      * a warning of some sort.
00600      */
00601     fprintf(stderr, "tcpdump: Filtering in user process\n");
00602     p->md.use_bpf = 0;
00603     return (0);
00604 }
