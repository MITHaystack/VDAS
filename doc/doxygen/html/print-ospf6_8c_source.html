00162 static int
00163 ospf6_print_lsaprefix(register const struct lsa6_prefix *lsapp)
00164 {
00165     u_int wordlen;
00166     struct in6_addr prefix;
00167 
00168     TCHECK(*lsapp);
00169     wordlen = (lsapp->lsa_p_len + 31) / 32;
00170     if (wordlen * 4 > sizeof(struct in6_addr)) {
00171         printf(" bogus prefixlen /%d", lsapp->lsa_p_len);
00172         goto trunc;
00173     }
00174     memset(&prefix, 0, sizeof(prefix));
00175     memcpy(&prefix, lsapp->lsa_p_prefix, wordlen * 4);
00176     printf("\n\t\t%s/%d", ip6addr_string(&prefix),
00177         lsapp->lsa_p_len);
00178         if (lsapp->lsa_p_opt) {
00179             printf(", Options [%s]",
00180                    bittok2str(ospf6_lsa_prefix_option_values,
00181                               "none", lsapp->lsa_p_opt));
00182         }
00183         printf(", metric %u", EXTRACT_16BITS(&lsapp->lsa_p_metric));
00184     return sizeof(*lsapp) - 4 + wordlen * 4;
00185 
00186 trunc:
00187     return -1;
00188 }

00194 static int
00195 ospf6_print_lsa(register const struct lsa6 *lsap)
00196 {
00197     register const u_char *ls_end, *ls_opt;
00198     register const struct rlalink6 *rlp;
00199 #if 0
00200     register const struct tos_metric *tosp;
00201 #endif
00202     register const rtrid_t *ap;
00203 #if 0
00204     register const struct aslametric *almp;
00205     register const struct mcla *mcp;
00206 #endif
00207     register const struct llsa *llsap;
00208     register const struct lsa6_prefix *lsapp;
00209 #if 0
00210     register const u_int32_t *lp;
00211 #endif
00212     register u_int prefixes;
00213     register int bytelen, length, lsa_length;
00214     u_int32_t flags32;
00215         u_int8_t *tptr;
00216 
00217     if (ospf6_print_lshdr(&lsap->ls_hdr))
00218         return (1);
00219     TCHECK(lsap->ls_hdr.ls_length);
00220         length = EXTRACT_16BITS(&lsap->ls_hdr.ls_length);
00221         lsa_length = length - sizeof(struct lsa6_hdr);
00222     ls_end = (u_char *)lsap + length;
00223         tptr = (u_int8_t *)lsap+sizeof(struct lsa6_hdr);
00224 
00225     switch (EXTRACT_16BITS(&lsap->ls_hdr.ls_type)) {
00226     case LS_TYPE_ROUTER | LS_SCOPE_AREA:
00227         TCHECK(lsap->lsa_un.un_rla.rla_options);
00228                 printf("\n\t      Options [%s]",
00229                        bittok2str(ospf6_option_values, "none",
00230                                   EXTRACT_32BITS(&lsap->lsa_un.un_rla.rla_options)));
00231 
00232         TCHECK(lsap->lsa_un.un_rla.rla_flags);
00233                 printf(", RLA-Flags [%s]",
00234                        bittok2str(ospf6_rla_flag_values, "none",
00235                                   lsap->lsa_un.un_rla.rla_flags));
00236 
00237 
00238         TCHECK(lsap->lsa_un.un_rla.rla_link);
00239         rlp = lsap->lsa_un.un_rla.rla_link;
00240         while (rlp + 1 <= (struct rlalink6 *)ls_end) {
00241             TCHECK(*rlp);
00242             switch (rlp->link_type) {
00243 
00244             case RLA_TYPE_VIRTUAL:
00245                 printf("\n\t      Virtual Link: Neighbor Router-ID %s"
00246                                        "\n\t      Neighbor Interface-ID %s, Interface %s",
00247                                        ipaddr_string(&rlp->link_nrtid),
00248                                        ipaddr_string(&rlp->link_nifid),
00249                                        ipaddr_string(&rlp->link_ifid)); 
00250                                 break;
00251 
00252             case RLA_TYPE_ROUTER:
00253                 printf("\n\t      Neighbor Router-ID %s"
00254                                        "\n\t      Neighbor Interface-ID %s, Interface %s",
00255                                        ipaddr_string(&rlp->link_nrtid),
00256                                        ipaddr_string(&rlp->link_nifid),
00257                                        ipaddr_string(&rlp->link_ifid)); 
00258                 break;
00259 
00260             case RLA_TYPE_TRANSIT:
00261                 printf("\n\t      Neighbor Network-ID %s"
00262                                        "\n\t      Neighbor Interface-ID %s, Interface %s",
00263                     ipaddr_string(&rlp->link_nrtid),
00264                     ipaddr_string(&rlp->link_nifid),
00265                     ipaddr_string(&rlp->link_ifid));
00266                 break;
00267 
00268             default:
00269                 printf("\n\t      Unknown Router Links Type 0x%02x",
00270                     rlp->link_type);
00271                 return (0);
00272             }
00273             printf(", metric %d", EXTRACT_16BITS(&rlp->link_metric));
00274             rlp++;
00275         }
00276         break;
00277 
00278     case LS_TYPE_NETWORK | LS_SCOPE_AREA:
00279         TCHECK(lsap->lsa_un.un_nla.nla_options);
00280                 printf("\n\t      Options [%s]",
00281                        bittok2str(ospf6_option_values, "none",
00282                                   EXTRACT_32BITS(&lsap->lsa_un.un_nla.nla_options)));
00283         printf("\n\t      Connected Routers:");
00284         ap = lsap->lsa_un.un_nla.nla_router;
00285         while ((u_char *)ap < ls_end) {
00286             TCHECK(*ap);
00287             printf("\n\t\t%s", ipaddr_string(ap));
00288             ++ap;
00289         }
00290         break;
00291 
00292     case LS_TYPE_INTER_AP | LS_SCOPE_AREA:
00293         TCHECK(lsap->lsa_un.un_inter_ap.inter_ap_metric);
00294         printf(", metric %u",
00295             EXTRACT_32BITS(&lsap->lsa_un.un_inter_ap.inter_ap_metric) & SLA_MASK_METRIC);
00296         lsapp = lsap->lsa_un.un_inter_ap.inter_ap_prefix;
00297         while (lsapp + sizeof(lsapp) <= (struct lsa6_prefix *)ls_end) {
00298             bytelen = ospf6_print_lsaprefix(lsapp);
00299             if (bytelen)
00300                 goto trunc;
00301             lsapp = (struct lsa6_prefix *)(((u_char *)lsapp) + bytelen);
00302         }
00303         break;
00304     case LS_SCOPE_AS | LS_TYPE_ASE:
00305         TCHECK(lsap->lsa_un.un_asla.asla_metric);
00306         flags32 = EXTRACT_32BITS(&lsap->lsa_un.un_asla.asla_metric);
00307                 printf("\n\t     Flags [%s]",
00308                        bittok2str(ospf6_asla_flag_values, "none", flags32));
00309         printf(" metric %u",
00310                EXTRACT_32BITS(&lsap->lsa_un.un_asla.asla_metric) &
00311                ASLA_MASK_METRIC);
00312         lsapp = lsap->lsa_un.un_asla.asla_prefix;
00313         bytelen = ospf6_print_lsaprefix(lsapp);
00314         if (bytelen < 0)
00315             goto trunc;
00316         if ((ls_opt = (u_char *)(((u_char *)lsapp) + bytelen)) < ls_end) {
00317             struct in6_addr *fwdaddr6;
00318 
00319             if ((flags32 & ASLA_FLAG_FWDADDR) != 0) {
00320                 fwdaddr6 = (struct in6_addr *)ls_opt;
00321                 TCHECK(*fwdaddr6);
00322                 printf(" forward %s",
00323                        ip6addr_string(fwdaddr6));
00324 
00325                 ls_opt += sizeof(struct in6_addr);
00326             }
00327 
00328             if ((flags32 & ASLA_FLAG_ROUTETAG) != 0) {
00329                 TCHECK(*(u_int32_t *)ls_opt);
00330                 printf(" tag %s",
00331                        ipaddr_string((u_int32_t *)ls_opt));
00332 
00333                 ls_opt += sizeof(u_int32_t);
00334             }
00335 
00336             if (lsapp->lsa_p_metric) {
00337                 TCHECK(*(u_int32_t *)ls_opt);
00338                 printf(" RefLSID: %s",
00339                        ipaddr_string((u_int32_t *)ls_opt));
00340 
00341                 ls_opt += sizeof(u_int32_t);
00342             }
00343         }
00344         break;
00345 
00346     case LS_TYPE_LINK:
00347         /* Link LSA */
00348         llsap = &lsap->lsa_un.un_llsa;
00349         TCHECK(llsap->llsa_options);
00350                 printf("\n\t      Options [%s]",
00351                        bittok2str(ospf6_option_values, "none",
00352                                   EXTRACT_32BITS(&llsap->llsa_options)));
00353         TCHECK(llsap->llsa_nprefix);
00354                 prefixes = EXTRACT_32BITS(&llsap->llsa_nprefix);
00355         printf("\n\t      Priority %d, Link-local address %s, Prefixes %d:",
00356                        llsap->llsa_priority,
00357                        ip6addr_string(&llsap->llsa_lladdr),
00358                        prefixes);
00359 
00360         tptr = (u_int8_t *)llsap->llsa_prefix;
00361                 while (prefixes > 0) {
00362                     lsapp = (struct lsa6_prefix *)tptr;
00363                     if ((bytelen = ospf6_print_lsaprefix(lsapp)) == -1) {
00364                         goto trunc;
00365                     }
00366                     prefixes--;
00367                     tptr += bytelen;
00368                 }
00369         break;
00370 
00371     case LS_TYPE_INTRA_AP | LS_SCOPE_AREA:
00372         /* Intra-Area-Prefix LSA */
00373         TCHECK(lsap->lsa_un.un_intra_ap.intra_ap_rtid);
00374         ospf6_print_ls_type(
00375             EXTRACT_16BITS(&lsap->lsa_un.un_intra_ap.intra_ap_lstype),
00376             &lsap->lsa_un.un_intra_ap.intra_ap_lsid);
00377         TCHECK(lsap->lsa_un.un_intra_ap.intra_ap_nprefix);
00378                 prefixes = EXTRACT_16BITS(&lsap->lsa_un.un_intra_ap.intra_ap_nprefix);
00379         printf("\n\t      Prefixes %d:", prefixes);
00380 
00381                 tptr = (u_int8_t *)lsap->lsa_un.un_intra_ap.intra_ap_prefix;
00382 
00383                 while (prefixes > 0) {
00384                     lsapp = (struct lsa6_prefix *)tptr;
00385                     if ((bytelen = ospf6_print_lsaprefix(lsapp)) == -1) {
00386                         goto trunc;
00387                     }
00388                     prefixes--;
00389                     tptr += bytelen;
00390                 }
00391         break;
00392 
00393         case LS_TYPE_GRACE | LS_SCOPE_LINKLOCAL:
00394                 if (ospf_print_grace_lsa(tptr, lsa_length) == -1) {
00395                     return 1;
00396                 }
00397             
00398             break;
00399 
00400         case LS_TYPE_INTRA_ATE | LS_SCOPE_LINKLOCAL:
00401             if (ospf_print_te_lsa(tptr, lsa_length) == -1) {
00402                 return 1;
00403             }
00404             break;
00405 
00406     default:
00407             if(!print_unknown_data(tptr,
00408                                    "\n\t      ",
00409                                    lsa_length)) {
00410                 return (1);
00411             }
00412     }
00413 
00414     return (0);
00415 trunc:
00416     return (1);
00417 }

00419 static int
00420 ospf6_decode_v3(register const struct ospf6hdr *op,
00421     register const u_char *dataend)
00422 {
00423     register const rtrid_t *ap;
00424     register const struct lsr6 *lsrp;
00425     register const struct lsa6_hdr *lshp;
00426     register const struct lsa6 *lsap;
00427     register int i;
00428 
00429     switch (op->ospf6_type) {
00430 
00431     case OSPF_TYPE_HELLO:
00432                 printf("\n\tOptions [%s]",
00433                        bittok2str(ospf6_option_values, "none",
00434                                   EXTRACT_32BITS(&op->ospf6_hello.hello_options)));
00435 
00436                 TCHECK(op->ospf6_hello.hello_deadint);
00437                 printf("\n\t  Hello Timer %us, Dead Timer %us, Interface-ID %s, Priority %u",
00438                        EXTRACT_16BITS(&op->ospf6_hello.hello_helloint),
00439                        EXTRACT_16BITS(&op->ospf6_hello.hello_deadint),
00440                        ipaddr_string(&op->ospf6_hello.hello_ifid),
00441                        op->ospf6_hello.hello_priority);
00442 
00443         TCHECK(op->ospf6_hello.hello_dr);
00444         if (op->ospf6_hello.hello_dr != 0)
00445             printf("\n\t  Designated Router %s",
00446                 ipaddr_string(&op->ospf6_hello.hello_dr));
00447         TCHECK(op->ospf6_hello.hello_bdr);
00448         if (op->ospf6_hello.hello_bdr != 0)
00449             printf(", Backup Designated Router %s",
00450                 ipaddr_string(&op->ospf6_hello.hello_bdr));
00451         if (vflag) {
00452             printf("\n\t  Neighbor List:");
00453             ap = op->ospf6_hello.hello_neighbor;
00454             while ((u_char *)ap < dataend) {
00455                 TCHECK(*ap);
00456                 printf("\n\t    %s", ipaddr_string(ap));
00457                 ++ap;
00458             }
00459         }
00460         break;  /* HELLO */
00461 
00462     case OSPF_TYPE_DD:
00463         TCHECK(op->ospf6_db.db_options);
00464                 printf("\n\tOptions [%s]",
00465                        bittok2str(ospf6_option_values, "none",
00466                                   EXTRACT_32BITS(&op->ospf6_db.db_options)));
00467         TCHECK(op->ospf6_db.db_flags);
00468                 printf(", DD Flags [%s]",
00469                        bittok2str(ospf6_dd_flag_values,"none",op->ospf6_db.db_flags));
00470 
00471         TCHECK(op->ospf6_db.db_seq);
00472         printf(", MTU %u, DD-Sequence 0x%08x",
00473                        EXTRACT_16BITS(&op->ospf6_db.db_mtu),
00474                        EXTRACT_32BITS(&op->ospf6_db.db_seq));
00475 
00476                 /* Print all the LS adv's */
00477                 lshp = op->ospf6_db.db_lshdr;
00478                 while (!ospf6_print_lshdr(lshp)) {
00479                     ++lshp;
00480                 }
00481         break;
00482 
00483     case OSPF_TYPE_LS_REQ:
00484         if (vflag) {
00485             lsrp = op->ospf6_lsr;
00486             while ((u_char *)lsrp < dataend) {
00487                 TCHECK(*lsrp);
00488                                 printf("\n\t  Advertising Router %s",
00489                                        ipaddr_string(&lsrp->ls_router));
00490                 ospf6_print_ls_type(EXTRACT_16BITS(&lsrp->ls_type),
00491                                                     &lsrp->ls_stateid);
00492                 ++lsrp;
00493             }
00494         }
00495         break;
00496 
00497     case OSPF_TYPE_LS_UPDATE:
00498         if (vflag) {
00499             lsap = op->ospf6_lsu.lsu_lsa;
00500             TCHECK(op->ospf6_lsu.lsu_count);
00501             i = EXTRACT_32BITS(&op->ospf6_lsu.lsu_count);
00502             while (i--) {
00503                 if (ospf6_print_lsa(lsap))
00504                     goto trunc;
00505                 lsap = (struct lsa6 *)((u_char *)lsap +
00506                     EXTRACT_16BITS(&lsap->ls_hdr.ls_length));
00507             }
00508         }
00509         break;
00510 
00511 
00512     case OSPF_TYPE_LS_ACK:
00513         if (vflag) {
00514             lshp = op->ospf6_lsa.lsa_lshdr;
00515 
00516             while (!ospf6_print_lshdr(lshp)) {
00517                 ++lshp;
00518             }
00519         }
00520         break;
00521 
00522     default:
00523         break;
00524     }
00525     return (0);
00526 trunc:
00527     return (1);
00528 }

00530 void
00531 ospf6_print(register const u_char *bp, register u_int length)
00532 {
00533     register const struct ospf6hdr *op;
00534     register const u_char *dataend;
00535     register const char *cp;
00536 
00537     op = (struct ospf6hdr *)bp;
00538 
00539     /* If the type is valid translate it, or just print the type */
00540     /* value.  If it's not valid, say so and return */
