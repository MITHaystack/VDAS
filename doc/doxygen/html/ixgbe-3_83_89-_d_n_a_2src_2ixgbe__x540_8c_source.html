00198 s32 ixgbe_reset_hw_X540(struct ixgbe_hw *hw)
00199 {
00200     ixgbe_link_speed link_speed;
00201     s32 status = 0;
00202     u32 ctrl, ctrl_ext, reset_bit;
00203     u32 i;
00204     bool link_up = false;
00205 
00206     /* Call adapter stop to disable tx/rx and clear interrupts */
00207     hw->mac.ops.stop_adapter(hw);
00208 
00209     /*
00210      * Prevent the PCI-E bus from from hanging by disabling PCI-E master
00211      * access and verify no pending requests before reset
00212      */
00213     ixgbe_disable_pcie_master(hw);
00214 
00215 mac_reset_top:
00216     /*
00217      * Issue global reset to the MAC. Needs to be SW reset if link is up.
00218      * If link reset is used when link is up, it might reset the PHY when
00219      * mng is using it.  If link is down or the flag to force full link
00220      * reset is set, then perform link reset.
00221      */
00222     if (hw->force_full_reset) {
00223         reset_bit = IXGBE_CTRL_LNK_RST;
00224     } else {
00225         hw->mac.ops.check_link(hw, &link_speed, &link_up, false);
00226         if (!link_up)
00227             reset_bit = IXGBE_CTRL_LNK_RST;
00228         else
00229             reset_bit = IXGBE_CTRL_RST;
00230     }
00231 
00232     ctrl = IXGBE_READ_REG(hw, IXGBE_CTRL);
00233     IXGBE_WRITE_REG(hw, IXGBE_CTRL, (ctrl | reset_bit));
00234     IXGBE_WRITE_FLUSH(hw);
00235 
00236     /* Poll for reset bit to self-clear indicating reset is complete */
00237     for (i = 0; i < 10; i++) {
00238         udelay(1);
00239         ctrl = IXGBE_READ_REG(hw, IXGBE_CTRL);
00240         if (!(ctrl & reset_bit))
00241             break;
00242     }
00243 
00244 
00245     if (ctrl & reset_bit) {
00246         status = IXGBE_ERR_RESET_FAILED;
00247         hw_dbg(hw, "Reset polling failed to complete.\n");
00248     }
00249 
00250     /*
00251      * Double resets are required for recovery from certain error
00252      * conditions.  Between resets, it is necessary to stall to allow time
00253      * for any pending HW events to complete.  We use 1usec since that is
00254      * what is needed for ixgbe_disable_pcie_master().  The second reset
00255      * then clears out any effects of those events.
00256      */
00257     if (hw->mac.flags & IXGBE_FLAGS_DOUBLE_RESET_REQUIRED) {
00258         hw->mac.flags &= ~IXGBE_FLAGS_DOUBLE_RESET_REQUIRED;
00259         udelay(1);
00260         goto mac_reset_top;
00261     }
00262 
00263     /* Clear PF Reset Done bit so PF/VF Mail Ops can work */
00264     ctrl_ext = IXGBE_READ_REG(hw, IXGBE_CTRL_EXT);
00265     ctrl_ext |= IXGBE_CTRL_EXT_PFRSTD;
00266     IXGBE_WRITE_REG(hw, IXGBE_CTRL_EXT, ctrl_ext);
00267 
00268     msleep(50);
00269 
00270     /* Set the Rx packet buffer size. */
00271     IXGBE_WRITE_REG(hw, IXGBE_RXPBSIZE(0), 384 << IXGBE_RXPBSIZE_SHIFT);
00272 
00273     /* Store the permanent mac address */
00274     hw->mac.ops.get_mac_addr(hw, hw->mac.perm_addr);
00275 
00276     /*
00277      * Store MAC address from RAR0, clear receive address registers, and
00278      * clear the multicast table.  Also reset num_rar_entries to 128,
00279      * since we modify this value when programming the SAN MAC address.
00280      */
00281     hw->mac.num_rar_entries = 128;
00282     hw->mac.ops.init_rx_addrs(hw);
00283 
00284     /* Store the permanent SAN mac address */
00285     hw->mac.ops.get_san_mac_addr(hw, hw->mac.san_addr);
00286 
00287     /* Add the SAN MAC address to the RAR only if it's a valid address */
00288     if (ixgbe_validate_mac_addr(hw->mac.san_addr) == 0) {
00289         hw->mac.ops.set_rar(hw, hw->mac.num_rar_entries - 1,
00290                             hw->mac.san_addr, 0, IXGBE_RAH_AV);
00291 
00292         /* Reserve the last RAR for the SAN MAC address */
00293         hw->mac.num_rar_entries--;
00294     }
00295 
00296     return status;
00297 }

00426 u16 ixgbe_calc_eeprom_checksum_X540(struct ixgbe_hw *hw)
00427 {
00428     u16 i;
00429     u16 j;
00430     u16 checksum = 0;
00431     u16 length = 0;
00432     u16 pointer = 0;
00433     u16 word = 0;
00434 
00435     /*
00436      * Do not use hw->eeprom.ops.read because we do not want to take
00437      * the synchronization semaphores here. Instead use
00438      * ixgbe_read_eerd_generic
00439      */
00440 
00441     /* Include 0x0-0x3F in the checksum */
00442     for (i = 0; i < IXGBE_EEPROM_CHECKSUM; i++) {
00443         if (ixgbe_read_eerd_generic(hw, i, &word) != 0) {
00444             hw_dbg(hw, "EEPROM read failed\n");
00445             break;
00446         }
00447         checksum += word;
00448     }
00449 
00450     /*
00451      * Include all data from pointers 0x3, 0x6-0xE.  This excludes the
00452      * FW, PHY module, and PCIe Expansion/Option ROM pointers.
00453      */
00454     for (i = IXGBE_PCIE_ANALOG_PTR; i < IXGBE_FW_PTR; i++) {
00455         if (i == IXGBE_PHY_PTR || i == IXGBE_OPTION_ROM_PTR)
00456             continue;
00457 
00458         if (ixgbe_read_eerd_generic(hw, i, &pointer) != 0) {
00459             hw_dbg(hw, "EEPROM read failed\n");
00460             break;
00461         }
00462 
00463         /* Skip pointer section if the pointer is invalid. */
00464         if (pointer == 0xFFFF || pointer == 0 ||
00465             pointer >= hw->eeprom.word_size)
00466             continue;
00467 
00468         if (ixgbe_read_eerd_generic(hw, pointer, &length)!=
00469             0) {
00470             hw_dbg(hw, "EEPROM read failed\n");
00471             break;
00472         }
00473 
00474         /* Skip pointer section if length is invalid. */
00475         if (length == 0xFFFF || length == 0 ||
00476             (pointer + length) >= hw->eeprom.word_size)
00477             continue;
00478 
00479         for (j = pointer+1; j <= pointer+length; j++) {
00480             if (ixgbe_read_eerd_generic(hw, j, &word) !=
00481                 0) {
00482                 hw_dbg(hw, "EEPROM read failed\n");
00483                 break;
00484             }
00485             checksum += word;
00486         }
00487     }
00488 
00489     checksum = (u16)IXGBE_EEPROM_SUM - checksum;
00490 
00491     return checksum;
00492 }

00502 s32 ixgbe_validate_eeprom_checksum_X540(struct ixgbe_hw *hw,
00503                                         u16 *checksum_val)
00504 {
00505     s32 status;
00506     u16 checksum;
00507     u16 read_checksum = 0;
00508 
00509     /*
00510      * Read the first word from the EEPROM. If this times out or fails, do
00511      * not continue or we could be in for a very long wait while every
00512      * EEPROM read fails
00513      */
00514     status = hw->eeprom.ops.read(hw, 0, &checksum);
00515 
00516     if (status != 0) {
00517         hw_dbg(hw, "EEPROM read failed\n");
00518         goto out;
00519     }
00520 
00521     if (hw->mac.ops.acquire_swfw_sync(hw, IXGBE_GSSR_EEP_SM) ==
00522         0) {
00523         checksum = hw->eeprom.ops.calc_checksum(hw);
00524 
00525         /*
00526          * Do not use hw->eeprom.ops.read because we do not want to take
00527          * the synchronization semaphores twice here.
00528         */
00529         ixgbe_read_eerd_generic(hw, IXGBE_EEPROM_CHECKSUM,
00530                                 &read_checksum);
00531 
00532         /*
00533          * Verify read checksum from EEPROM is the same as
00534          * calculated checksum
00535          */
00536         if (read_checksum != checksum)
00537             status = IXGBE_ERR_EEPROM_CHECKSUM;
00538 
00539         /* If the user cares, return the calculated checksum */
00540         if (checksum_val)
00541             *checksum_val = checksum;
00542     } else {
00543         status = IXGBE_ERR_SWFW_SYNC;
00544     }
00545 
00546     hw->mac.ops.release_swfw_sync(hw, IXGBE_GSSR_EEP_SM);
00547 out:
00548     return status;
00549 }

00775 static s32 ixgbe_get_swfw_sync_semaphore(struct ixgbe_hw *hw)
00776 {
00777     s32 status = IXGBE_ERR_EEPROM;
00778     u32 timeout = 2000;
00779     u32 i;
00780     u32 swsm;
00781 
00782     /* Get SMBI software semaphore between device drivers first */
00783     for (i = 0; i < timeout; i++) {
00784         /*
00785          * If the SMBI bit is 0 when we read it, then the bit will be
00786          * set and we have the semaphore
00787          */
00788         swsm = IXGBE_READ_REG(hw, IXGBE_SWSM);
00789         if (!(swsm & IXGBE_SWSM_SMBI)) {
00790             status = 0;
00791             break;
00792         }
00793         udelay(50);
00794     }
00795 
00796     /* Now get the semaphore between SW/FW through the REGSMP bit */
00797     if (status == 0) {
00798         for (i = 0; i < timeout; i++) {
00799             swsm = IXGBE_READ_REG(hw, IXGBE_SWFW_SYNC);
00800             if (!(swsm & IXGBE_SWFW_REGSMP))
00801                 break;
00802 
00803             udelay(50);
00804         }
00805 
00806         /*
00807          * Release semaphores and return error if SW NVM semaphore
00808          * was not granted because we don't have access to the EEPROM
00809          */
00810         if (i >= timeout) {
00811             hw_dbg(hw, "REGSMP Software NVM semaphore not granted.\n");
00812             ixgbe_release_swfw_sync_semaphore(hw);
00813             status = IXGBE_ERR_EEPROM;
00814         }
00815     } else {
00816         hw_dbg(hw, "Software semaphore SMBI between device drivers "
00817                  "not granted.\n");
00818     }
00819 
00820     return status;
00821 }
