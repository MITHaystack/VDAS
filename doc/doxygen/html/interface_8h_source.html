00104 /*
00105  * True if  "l" bytes of "var" were captured.
00106  *
00107  * The "snapend - (l) <= snapend" checks to make sure "l" isn't so large
00108  * that "snapend - (l)" underflows.
00109  *
00110  * The check is for <= rather than < because "l" might be 0.
00111  */
00112 #define TTEST2(var, l) (snapend - (l) <= snapend && \
00113             (const u_char *)&(var) <= snapend - (l))
00114 
00115 /* True if "var" was captured */
00116 #define TTEST(var) TTEST2(var, sizeof(var))
00117 
00118 /* Bail if "l" bytes of "var" were not captured */
00119 #define TCHECK2(var, l) if (!TTEST2(var, l)) goto trunc
00120 
00121 /* Bail if "var" was not captured */
00122 #define TCHECK(var) TCHECK2(var, sizeof(var))
