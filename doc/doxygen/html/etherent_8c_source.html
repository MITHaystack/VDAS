00101 struct pcap_etherent *
00102 pcap_next_etherent(FILE *fp)
00103 {
00104     register int c, d, i;
00105     char *bp;
00106     static struct pcap_etherent e;
00107 
00108     memset((char *)&e, 0, sizeof(e));
00109     do {
00110         /* Find addr */
00111         c = skip_space(fp);
00112         if (c == '\n')
00113             continue;
00114 
00115         /* If this is a comment, or first thing on line
00116            cannot be etehrnet address, skip the line. */
00117         if (!isxdigit(c)) {
00118             c = skip_line(fp);
00119             continue;
00120         }
00121 
00122         /* must be the start of an address */
00123         for (i = 0; i < 6; i += 1) {
00124             d = xdtoi(c);
00125             c = getc(fp);
00126             if (isxdigit(c)) {
00127                 d <<= 4;
00128                 d |= xdtoi(c);
00129                 c = getc(fp);
00130             }
00131             e.addr[i] = d;
00132             if (c != ':')
00133                 break;
00134             c = getc(fp);
00135         }
00136         if (c == EOF)
00137             break;
00138 
00139         /* Must be whitespace */
00140         if (!isspace(c)) {
00141             c = skip_line(fp);
00142             continue;
00143         }
00144         c = skip_space(fp);
00145 
00146         /* hit end of line... */
00147         if (c == '\n')
00148             continue;
00149 
00150         if (c == '#') {
00151             c = skip_line(fp);
00152             continue;
00153         }
00154 
00155         /* pick up name */
00156         bp = e.name;
00157         /* Use 'd' to prevent buffer overflow. */
00158         d = sizeof(e.name) - 1;
00159         do {
00160             *bp++ = c;
00161             c = getc(fp);
00162         } while (!isspace(c) && c != EOF && --d > 0);
00163         *bp = '\0';
00164 
00165         /* Eat trailing junk */
00166         if (c != '\n')
00167             (void)skip_line(fp);
00168 
00169         return &e;
00170 
00171     } while (c != EOF);
00172 
00173     return (NULL);
00174 }
