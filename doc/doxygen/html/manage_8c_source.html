00122     struct e1000_host_mng_dhcp_cookie *hdr = &hw->mng_cookie;

00129 
00130     /* No manageability, no filtering */
00131     if (!hw->mac.ops.check_mng_mode(hw)) {
00132         hw->mac.tx_pkt_filtering = false;
00133         goto out;
00134     }
00135 
00136     /*
00137      * If we can't read from the host interface for whatever

00139      */
00140     ret_val = hw->mac.ops.mng_enable_host_if(hw);
00141     if (ret_val) {
00142         hw->mac.tx_pkt_filtering = false;
00143         goto out;
00144     }
00145 
00146     /* Read in the header.  Length and offset are in dwords. */
00147     len    = E1000_MNG_DHCP_COOKIE_LENGTH >> 2;
00148     offset = E1000_MNG_DHCP_COOKIE_OFFSET >> 2;
00149     for (i = 0; i < len; i++)
00150         *(buffer + i) = E1000_READ_REG_ARRAY(hw, E1000_HOST_IF,
00151                                                    offset + i);
00152     hdr_csum = hdr->checksum;
00153     hdr->checksum = 0;
00154     csum = e1000_calculate_checksum((u8 *)hdr,
00155                                     E1000_MNG_DHCP_COOKIE_LENGTH);
00156     /*
00157      * If either the checksums or signature don't match, then
00158      * the cookie area isn't considered valid, in which case we
00159      * take the safe route of assuming Tx filtering is enabled.
00160      */
00161     if ((hdr_csum != csum) || (hdr->signature != E1000_IAMT_SIGNATURE)) {
00162         hw->mac.tx_pkt_filtering = true;
00163         goto out;
00164     }
00165 
00166     /* Cookie area is valid, make the final check for filtering. */
00167     if (!(hdr->status & E1000_MNG_DHCP_COOKIE_STATUS_PARSING)) {
00168         hw->mac.tx_pkt_filtering = false;
00169         goto out;
00170     }
