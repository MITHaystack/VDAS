00088 static int ipq_daq_get_setup (
00089     IpqImpl* impl, const DAQ_Config_t* cfg, char* errBuf, size_t errMax)
00090 {
00091     DAQ_Dict* entry;
00092 
00093     impl->proto = PF_INET;
00094 
00095     for ( entry = cfg->values; entry; entry = entry->next)
00096     {
00097         if ( !entry->value || !*entry->value )
00098         {
00099             snprintf(errBuf, errMax,
00100                 "%s: variable needs value (%s)\n", __FUNCTION__, entry->key);
00101                 return DAQ_ERROR;
00102         }
00103         else if ( !strcmp(entry->key, "proto") )
00104         {
00105             impl->proto = ipq_daq_get_proto(entry->value);
00106             if ( !impl->proto )
00107             {
00108                 snprintf(errBuf, errMax, "%s: bad proto (%s)\n",
00109                     __FUNCTION__, entry->value);
00110                 return DAQ_ERROR;
00111             }
00112         }
00113         else if ( !strcmp(entry->key, "device") )
00114         {
00115             impl->device = strdup(entry->value);
00116             if ( !impl->device )
00117             {
00118                 snprintf(errBuf, errMax,
00119                     "%s: can't allocate memory for device (%s)\n",
00120                     __FUNCTION__, entry->value);
00121                 return DAQ_ERROR;
00122             }
00123         }
00124         else
00125         {
00126             snprintf(errBuf, errMax,
00127                 "%s: unsupported variable (%s=%s)\n",
00128                     __FUNCTION__, entry->key, entry->value);
00129                 return DAQ_ERROR;
00130         }
00131     }
00132 
00133     impl->snaplen = cfg->snaplen ? cfg->snaplen : IP_MAXPACKET;
00134     impl->timeout = cfg->timeout * 1000;  // cfg is ms, ipq is us
00135     impl->passive = ( cfg->mode == DAQ_MODE_PASSIVE );
00136 
00137     return DAQ_SUCCESS;
00138 }

00281 static int ipq_daq_acquire (
00282     void* handle, int cnt, DAQ_Analysis_Func_t callback, void* user)
00283 {
00284     IpqImpl* impl = (IpqImpl*)handle;
00285 
00286     int n = 0;
00287     DAQ_PktHdr_t hdr;
00288 
00289     // If cnt is <= 0, don't limit the packets acquired.  However,
00290     // impl->count = 0 has a special meaning, so interpret accordingly.
00291     impl->count = (cnt == 0) ? -1 : cnt;
00292     hdr.device_index = 0;
00293     hdr.flags = 0;
00294 
00295     while ( impl->count < 0 || n < impl->count )
00296     {
00297         int ipqt, status = ipq_read(
00298             impl->ipqh, impl->buf, MSG_BUF_SIZE, impl->timeout);
00299 
00300         if ( status <= 0 )
00301         {
00302             if ( status < 0 )
00303             {
00304                 DPE(impl->error, "%s: ipq_read=%d error %s",
00305                     __FUNCTION__, status, ipq_errstr());
00306                 return DAQ_ERROR;
00307             }
00308             return 0;
00309         }
00310 
00311         ipqt = ipq_message_type(impl->buf);
00312 
00313         if ( ipqt == IPQM_PACKET )
00314         {
00315             DAQ_Verdict verdict;
00316             ipq_packet_msg_t* ipqm = ipq_get_packet(impl->buf);
00317             SetPktHdr(impl, ipqm, &hdr);
00318             impl->stats.hw_packets_received++;
00319 
00320             if (
00321                 impl->fcode.bf_insns &&
00322                 sfbpf_filter(impl->fcode.bf_insns, ipqm->payload,
00323                     hdr.caplen, hdr.caplen) == 0
00324             ) {
00325                 verdict = DAQ_VERDICT_PASS;
00326                 impl->stats.packets_filtered++;
00327             }
00328             else
00329             {
00330                 verdict = callback(user, &hdr, (uint8_t*)ipqm->payload);
00331                 impl->stats.verdicts[verdict]++;
00332                 impl->stats.packets_received++;
00333             }
00334             if ( impl->passive ) verdict = DAQ_VERDICT_PASS;
00335 
00336             switch ( verdict ) {
00337             case DAQ_VERDICT_BLOCK:
00338             case DAQ_VERDICT_BLACKLIST:
00339                 status = ipq_set_verdict(
00340                     impl->ipqh, ipqm->packet_id, NF_DROP, 0, NULL);
00341                 break;
00342 
00343             case DAQ_VERDICT_REPLACE:
00344                 status = ipq_set_verdict(
00345                     impl->ipqh, ipqm->packet_id, NF_ACCEPT,
00346                     hdr.pktlen, ipqm->payload);
00347                 break;
00348 
00349             case DAQ_VERDICT_PASS:
00350             case DAQ_VERDICT_WHITELIST:
00351             case DAQ_VERDICT_IGNORE:
00352             default:
00353                 status = ipq_set_verdict(
00354                     impl->ipqh, ipqm->packet_id, NF_ACCEPT, 0, NULL);
00355                 break;
00356             }
00357             if ( status < 0 )
00358             {
00359                 DPE(impl->error, "%s: ipq_set_verdict=%d error %s",
00360                     __FUNCTION__, status, ipq_errstr());
00361                 return DAQ_ERROR;
00362             }
00363             n++;
00364         }
00365         else
00366         {
00367             // NLMSG_ERROR is supposed to be the only other valid type
00368             status = ipq_get_msgerr(impl->buf);
00369             DPE(impl->error, "%s: ipq_message_type=%d error=%d %s",
00370                 __FUNCTION__, ipqt, status, ipq_errstr());
00371             // ipq_message_type=2 error=1 Timeout
00372             // keep looping upon timeout or other errors
00373         }
00374     }
00375     return 0;
00376 }
