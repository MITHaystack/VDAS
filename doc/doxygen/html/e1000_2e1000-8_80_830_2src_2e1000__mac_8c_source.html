00451 void e1000_rar_set_generic(struct e1000_hw *hw, u8 *addr, u32 index)
00452 {
00453     u32 rar_low, rar_high;
00454 
00455     DEBUGFUNC("e1000_rar_set_generic");
00456 
00457     /*
00458      * HW expects these in little endian so we reverse the byte order
00459      * from network order (big endian) to little endian
00460      */
00461     rar_low = ((u32) addr[0] |
00462                ((u32) addr[1] << 8) |
00463                ((u32) addr[2] << 16) | ((u32) addr[3] << 24));
00464 
00465     rar_high = ((u32) addr[4] | ((u32) addr[5] << 8));
00466 
00467     /* If MAC address zero, no need to set the AV bit */
00468     if (rar_low || rar_high)
00469         rar_high |= E1000_RAH_AV;
00470 
00471     /*
00472      * Some bridges will combine consecutive 32-bit writes into
00473      * a single burst write, which will malfunction on some parts.
00474      * The flushes avoid this.
00475      */
00476     E1000_WRITE_REG(hw, E1000_RAL(index), rar_low);
00477     E1000_WRITE_FLUSH(hw);
00478     E1000_WRITE_REG(hw, E1000_RAH(index), rar_high);
00479     E1000_WRITE_FLUSH(hw);

00527 u32 e1000_hash_mc_addr_generic(struct e1000_hw *hw, u8 *mc_addr)
00528 {
00529     u32 hash_value, hash_mask;
00530     u8 bit_shift = 0;
00531 
00532     DEBUGFUNC("e1000_hash_mc_addr_generic");
00533 
00534     /* Register count multiplied by bits per register */
00535     hash_mask = (hw->mac.mta_reg_count * 32) - 1;
00536 
00537     /*
00538      * For a mc_filter_type of 0, bit_shift is the number of left-shifts
00539      * where 0xFF would still fall within the hash mask.
00540      */
00541     while (hash_mask >> bit_shift != 0xFF)
00542         bit_shift++;
00543 
00544     /*
00545      * The portion of the address that is used for the hash table
00546      * is determined by the mc_filter_type setting.
00547      * The algorithm is such that there is a total of 8 bits of shifting.
00548      * The bit_shift for a mc_filter_type of 0 represents the number of
00549      * left-shifts where the MSB of mc_addr[5] would still fall within
00550      * the hash_mask.  Case 0 does this exactly.  Since there are a total
00551      * of 8 bits of shifting, then mc_addr[4] will shift right the
00552      * remaining number of bits. Thus 8 - bit_shift.  The rest of the
00553      * cases are a variation of this algorithm...essentially raising the
00554      * number of bits to shift mc_addr[5] left, while still keeping the
00555      * 8-bit shifting total.
00556      *
00557      * For example, given the following Destination MAC Address and an
00558      * mta register count of 128 (thus a 4096-bit vector and 0xFFF mask),
00559      * we can see that the bit_shift for case 0 is 4.  These are the hash
00560      * values resulting from each mc_filter_type...
00561      * [0] [1] [2] [3] [4] [5]
00562      * 01  AA  00  12  34  56
00563      * LSB                 MSB
00564      *
00565      * case 0: hash_value = ((0x34 >> 4) | (0x56 << 4)) & 0xFFF = 0x563
00566      * case 1: hash_value = ((0x34 >> 3) | (0x56 << 5)) & 0xFFF = 0xAC6
00567      * case 2: hash_value = ((0x34 >> 2) | (0x56 << 6)) & 0xFFF = 0x163
00568      * case 3: hash_value = ((0x34 >> 0) | (0x56 << 8)) & 0xFFF = 0x634
00569      */
00570     switch (hw->mac.mc_filter_type) {
00571     default:
00572     case 0:
00573         break;
00574     case 1:
00575         bit_shift += 1;
00576         break;
00577     case 2:
00578         bit_shift += 2;
00579         break;
00580     case 3:
00581         bit_shift += 4;
00582         break;
00583     }
00584 
00585     hash_value = hash_mask & (((mc_addr[4] >> (8 - bit_shift)) |
00586                               (((u16) mc_addr[5]) << bit_shift)));
00587 
00588     return hash_value;
00589 }

01138 s32 e1000_commit_fc_settings_generic(struct e1000_hw *hw)
01139 {
01140     struct e1000_mac_info *mac = &hw->mac;
01141     u32 txcw;
01142     s32 ret_val = E1000_SUCCESS;
01143 
01144     DEBUGFUNC("e1000_commit_fc_settings_generic");
01145 
01146     /*
01147      * Check for a software override of the flow control settings, and
01148      * setup the device accordingly.  If auto-negotiation is enabled, then
01149      * software will have to set the "PAUSE" bits to the correct value in
01150      * the Transmit Config Word Register (TXCW) and re-start auto-
01151      * negotiation.  However, if auto-negotiation is disabled, then
01152      * software will have to manually configure the two flow control enable
01153      * bits in the CTRL register.
01154      *
01155      * The possible values of the "fc" parameter are:
01156      *      0:  Flow control is completely disabled
01157      *      1:  Rx flow control is enabled (we can receive pause frames,
01158      *          but not send pause frames).
01159      *      2:  Tx flow control is enabled (we can send pause frames but we
01160      *          do not support receiving pause frames).
01161      *      3:  Both Rx and Tx flow control (symmetric) are enabled.
01162      */
01163     switch (hw->fc.current_mode) {
01164     case e1000_fc_none:
01165         /* Flow control completely disabled by a software over-ride. */
01166         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD);
01167         break;
01168     case e1000_fc_rx_pause:
01169         /*
01170          * Rx Flow control is enabled and Tx Flow control is disabled

01175          * PAUSE frames.
01176          */
01177         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_PAUSE_MASK);
01178         break;
01179     case e1000_fc_tx_pause:
01180         /*
01181          * Tx Flow control is enabled, and Rx Flow control is disabled,
01182          * by a software over-ride.
01183          */
01184         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_ASM_DIR);
01185         break;
01186     case e1000_fc_full:
01187         /*
01188          * Flow control (both Rx and Tx) is enabled by a software
01189          * over-ride.
01190          */
01191         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_PAUSE_MASK);

01195         ret_val = -E1000_ERR_CONFIG;

01198     }

01376      * configuration of the MAC to match the "fc" parameter.
01377      */
01378     if (mac->autoneg_failed) {
01379         if (hw->phy.media_type == e1000_media_type_fiber ||
01380             hw->phy.media_type == e1000_media_type_internal_serdes)
01381             ret_val = e1000_force_mac_fc_generic(hw);
01382     } else {
01383         if (hw->phy.media_type == e1000_media_type_copper)
01384             ret_val = e1000_force_mac_fc_generic(hw);
01385     }
01386 
01387     if (ret_val) {
01388         DEBUGOUT("Error forcing flow control settings\n");
01389         goto out;
01390     }
01391 
01392     /*
01393      * Check for the case where we have copper media and auto-neg is
01394      * enabled.  In this case, we need to check and see if Auto-Neg
01395      * has completed, and if so, how the PHY and link partner has
01396      * flow control configured.
01397      */
01398     if ((hw->phy.media_type == e1000_media_type_copper) && mac->autoneg) {
01399         /*
01400          * Read the MII Status Register and check to see if AutoNeg
01401          * has completed.  We read this twice because this reg has
01402          * some "sticky" (latched) bits.
01403          */
01404         ret_val = hw->phy.ops.read_reg(hw, PHY_STATUS, &mii_status_reg);
01405         if (ret_val)
01406             goto out;
01407         ret_val = hw->phy.ops.read_reg(hw, PHY_STATUS, &mii_status_reg);
01408         if (ret_val)
01409             goto out;
01410 
01411         if (!(mii_status_reg & MII_SR_AUTONEG_COMPLETE)) {
01412             DEBUGOUT("Copper PHY and Auto Neg "
01413                      "has not completed.\n");
01414             goto out;
01415         }
01416 
01417         /*
01418          * The AutoNeg process has completed, so we now need to
01419          * read both the Auto Negotiation Advertisement
01420          * Register (Address 4) and the Auto_Negotiation Base
01421          * Page Ability Register (Address 5) to determine how
01422          * flow control was negotiated.
01423          */
01424         ret_val = hw->phy.ops.read_reg(hw, PHY_AUTONEG_ADV,
01425                                      &mii_nway_adv_reg);
01426         if (ret_val)
01427             goto out;
01428         ret_val = hw->phy.ops.read_reg(hw, PHY_LP_ABILITY,
01429                                      &mii_nway_lp_ability_reg);
01430         if (ret_val)
01431             goto out;
01432 
01433         /*
01434          * Two bits in the Auto Negotiation Advertisement Register
01435          * (Address 4) and two bits in the Auto Negotiation Base
01436          * Page Ability Register (Address 5) determine flow control
01437          * for both the PHY and the link partner.  The following
01438          * table, taken out of the IEEE 802.3ab/D6.0 dated March 25,
01439          * 1999, describes these PAUSE resolution bits and how flow
01440          * control is determined based upon these settings.
01441          * NOTE:  DC = Don't Care
01442          *
01443          *   LOCAL DEVICE  |   LINK PARTNER
01444          * PAUSE | ASM_DIR | PAUSE | ASM_DIR | NIC Resolution
01445          *-------|---------|-------|---------|--------------------
01446          *   0   |    0    |  DC   |   DC    | e1000_fc_none
01447          *   0   |    1    |   0   |   DC    | e1000_fc_none
01448          *   0   |    1    |   1   |    0    | e1000_fc_none
01449          *   0   |    1    |   1   |    1    | e1000_fc_tx_pause
01450          *   1   |    0    |   0   |   DC    | e1000_fc_none
01451          *   1   |   DC    |   1   |   DC    | e1000_fc_full
01452          *   1   |    1    |   0   |    0    | e1000_fc_none
01453          *   1   |    1    |   0   |    1    | e1000_fc_rx_pause
01454          *
01455          * Are both PAUSE bits set to 1?  If so, this implies
01456          * Symmetric Flow Control is enabled at both ends.  The
01457          * ASM_DIR bits are irrelevant per the spec.
01458          *
01459          * For Symmetric Flow Control:
01460          *
01461          *   LOCAL DEVICE  |   LINK PARTNER
01462          * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
01463          *-------|---------|-------|---------|--------------------
01464          *   1   |   DC    |   1   |   DC    | E1000_fc_full
01465          *
01466          */
01467         if ((mii_nway_adv_reg & NWAY_AR_PAUSE) &&
01468             (mii_nway_lp_ability_reg & NWAY_LPAR_PAUSE)) {
01469             /*
01470              * Now we need to check if the user selected Rx ONLY
01471              * of pause frames.  In this case, we had to advertise
01472              * FULL flow control because we could not advertise Rx
01473              * ONLY. Hence, we must now check to see if we need to
01474              * turn OFF  the TRANSMISSION of PAUSE frames.
01475              */
01476             if (hw->fc.requested_mode == e1000_fc_full) {
01477                 hw->fc.current_mode = e1000_fc_full;
01478                 DEBUGOUT("Flow Control = FULL.\r\n");
01479             } else {
01480                 hw->fc.current_mode = e1000_fc_rx_pause;
01481                 DEBUGOUT("Flow Control = "
01482                          "Rx PAUSE frames only.\r\n");
01483             }
01484         }
01485         /*
01486          * For receiving PAUSE frames ONLY.
01487          *
01488          *   LOCAL DEVICE  |   LINK PARTNER
01489          * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
01490          *-------|---------|-------|---------|--------------------
01491          *   0   |    1    |   1   |    1    | e1000_fc_tx_pause
01492          */
01493         else if (!(mii_nway_adv_reg & NWAY_AR_PAUSE) &&
01494                   (mii_nway_adv_reg & NWAY_AR_ASM_DIR) &&
01495                   (mii_nway_lp_ability_reg & NWAY_LPAR_PAUSE) &&
01496                   (mii_nway_lp_ability_reg & NWAY_LPAR_ASM_DIR)) {
01497             hw->fc.current_mode = e1000_fc_tx_pause;
01498             DEBUGOUT("Flow Control = Tx PAUSE frames only.\r\n");
01499         }
01500         /*
01501          * For transmitting PAUSE frames ONLY.
01502          *
01503          *   LOCAL DEVICE  |   LINK PARTNER
01504          * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
01505          *-------|---------|-------|---------|--------------------
01506          *   1   |    1    |   0   |    1    | e1000_fc_rx_pause
01507          */
01508         else if ((mii_nway_adv_reg & NWAY_AR_PAUSE) &&
01509                  (mii_nway_adv_reg & NWAY_AR_ASM_DIR) &&
01510                  !(mii_nway_lp_ability_reg & NWAY_LPAR_PAUSE) &&
01511                  (mii_nway_lp_ability_reg & NWAY_LPAR_ASM_DIR)) {
01512             hw->fc.current_mode = e1000_fc_rx_pause;
01513             DEBUGOUT("Flow Control = Rx PAUSE frames only.\r\n");
01514         } else {
01515             /*
01516              * Per the IEEE spec, at this point flow control
01517              * should be disabled.
01518              */
01519             hw->fc.current_mode = e1000_fc_none;
01520             DEBUGOUT("Flow Control = NONE.\r\n");
01521         }
01522 
01523         /*
01524          * Now we need to do one last check...  If we auto-
01525          * negotiated to HALF DUPLEX, flow control should not be
01526          * enabled per IEEE 802.3 spec.
01527          */
01528         ret_val = mac->ops.get_link_up_info(hw, &speed, &duplex);
01529         if (ret_val) {
01530             DEBUGOUT("Error getting link speed and duplex\n");
01531             goto out;
01532         }
01533 
01534         if (duplex == HALF_DUPLEX)
01535             hw->fc.current_mode = e1000_fc_none;
01536 
01537         /*
01538          * Now we call a subroutine to actually force the MAC
01539          * controller to use the correct flow control settings.
01540          */
01541         ret_val = e1000_force_mac_fc_generic(hw);
01542         if (ret_val) {
01543             DEBUGOUT("Error forcing flow control settings\n");
01544             goto out;
01545         }
01546     }
01547 
01548 out:
01549     return ret_val;
01550 }

01617 s32 e1000_get_hw_semaphore_generic(struct e1000_hw *hw)
01618 {
01619     u32 swsm;
01620     s32 ret_val = E1000_SUCCESS;
01621     s32 timeout = hw->nvm.word_size + 1;
01622     s32 i = 0;
01623 
01624     DEBUGFUNC("e1000_get_hw_semaphore_generic");
01625 
01626     /* Get the SW semaphore */
01627     while (i < timeout) {
01628         swsm = E1000_READ_REG(hw, E1000_SWSM);
01629         if (!(swsm & E1000_SWSM_SMBI))
01630             break;
01631 
01632         usec_delay(50);
01633         i++;
01634     }
01635 
01636     if (i == timeout) {
01637         DEBUGOUT("Driver can't access device - SMBI bit is set.\n");
01638         ret_val = -E1000_ERR_NVM;
01639         goto out;
01640     }
01641 
01642     /* Get the FW semaphore. */
01643     for (i = 0; i < timeout; i++) {
01644         swsm = E1000_READ_REG(hw, E1000_SWSM);
01645         E1000_WRITE_REG(hw, E1000_SWSM, swsm | E1000_SWSM_SWESMBI);
01646 
01647         /* Semaphore acquired if bit latched */
01648         if (E1000_READ_REG(hw, E1000_SWSM) & E1000_SWSM_SWESMBI)
01649             break;
01650 
01651         usec_delay(50);
01652     }
01653 
01654     if (i == timeout) {
01655         /* Release semaphores */
01656         e1000_put_hw_semaphore_generic(hw);
01657         DEBUGOUT("Driver can't access the NVM\n");
01658         ret_val = -E1000_ERR_NVM;
01659         goto out;
01660     }
01661 
01662 out:
01663     return ret_val;
01664 }
