00047 /* Entries EAI_ADDRFAMILY (1) and EAI_NODATA (7) are obsoleted, but left */

00050     "Success",                  /* 0 */
00051     "Address family for hostname not supported",    /* 1 */
00052     "Temporary failure in name resolution",     /* EAI_AGAIN */
00053     "Invalid value for ai_flags",           /* EAI_BADFLAGS */
00054     "Non-recoverable failure in name resolution",   /* EAI_FAIL */
00055     "ai_family not supported",          /* EAI_FAMILY */
00056     "Memory allocation failure",            /* EAI_MEMORY */
00057     "No address associated with hostname",      /* 7 */
00058     "hostname nor servname provided, or not known", /* EAI_NONAME */
00059     "servname not supported for ai_socktype",   /* EAI_SERVICE */
00060     "ai_socktype not supported",            /* EAI_SOCKTYPE */
00061     "System error returned in errno",       /* EAI_SYSTEM */
00062     "Invalid value for hints",          /* EAI_BADHINTS */
00063     "Resolved protocol is unknown"          /* EAI_PROTOCOL */
00064 };
