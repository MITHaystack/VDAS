00511 static s32 e1000_get_hw_semaphore_82571(struct e1000_hw *hw)
00512 {
00513     u32 swsm;
00514     s32 ret_val = E1000_SUCCESS;
00515     s32 sw_timeout = hw->nvm.word_size + 1;
00516     s32 fw_timeout = hw->nvm.word_size + 1;
00517     s32 i = 0;
00518 
00519     /*
00520      * If we have timedout 3 times on trying to acquire
00521      * the inter-port SMBI semaphore, there is old code
00522      * operating on the other port, and it is not
00523      * releasing SMBI. Modify the number of times that
00524      * we try for the semaphore to interwork with this
00525      * older code.
00526      */
00527     if (hw->dev_spec.e82571.smb_counter > 2)
00528         sw_timeout = 1;
00529 
00530     /* Get the SW semaphore */
00531     while (i < sw_timeout) {
00532         swsm = er32(SWSM);
00533         if (!(swsm & E1000_SWSM_SMBI))
00534             break;
00535 
00536         udelay(50);
00537         i++;
00538     }

00810 static s32 e1000_update_nvm_checksum_82571(struct e1000_hw *hw)
00811 {

00814     u16 i;
00815 
00816     ret_val = e1000e_update_nvm_checksum_generic(hw);
00817     if (ret_val)
00818         goto out;
00819 
00820     /*
00821      * If our nvm is an EEPROM, then we're done
00822      * otherwise, commit the checksum to the flash NVM.
00823      */
00824     if (hw->nvm.type != e1000_nvm_flash_hw)
00825         goto out;

00827     /* Check for pending operations. */
00828     for (i = 0; i < E1000_FLASH_UPDATES; i++) {

00846         e1e_flush();

00849 
00850     /* Commit the write to flash */
00851     eecd = er32(EECD) | E1000_EECD_FLUPD;

00853 
00854     for (i = 0; i < E1000_FLASH_UPDATES; i++) {
00855         msleep(1);

00857             break;

01050 {
01051     u32 ctrl, ctrl_ext;
01052     s32 ret_val;
01053 
01054     /*
01055      * Prevent the PCI-E bus from sticking if there is no TLP connection
01056      * on the last TLP read/write transaction when MAC is reset.
01057      */
01058     ret_val = e1000e_disable_pcie_master(hw);
01059     if (ret_val)
01060         e_dbg("PCI-E Master disable polling has failed.\n");
01061

01064 
01065     ew32(RCTL, 0);
01066     ew32(TCTL, E1000_TCTL_PSP);

01068 
01069     msleep(10);
01070

01073      * Ownership defaults to firmware after a reset.

01076     case e1000_82573:
01077         ret_val = e1000_get_hw_semaphore_82573(hw);
01078         break;
01079     case e1000_82574:
01080     case e1000_82583:
01081         ret_val = e1000_get_hw_semaphore_82574(hw);
01082         break;
01083     default:

01087         e_dbg("Cannot acquire MDIO ownership\n");
01088

01095     switch (hw->mac.type) {
01096     case e1000_82574:
01097     case e1000_82583:
01098         e1000_put_hw_semaphore_82574(hw);
01099         break;
01100     default:
01101         break;
01102     }
01103

01105         udelay(10);

01125     case e1000_82574:
01126     case e1000_82583:
