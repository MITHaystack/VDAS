00121 static s32 e1000_init_nvm_params_82541(struct e1000_hw *hw)
00122 {
00123     struct   e1000_nvm_info *nvm = &hw->nvm;
00124     s32  ret_val = E1000_SUCCESS;
00125     u32 eecd = E1000_READ_REG(hw, E1000_EECD);
00126     u16 size;
00127 
00128     DEBUGFUNC("e1000_init_nvm_params_82541");
00129 
00130     switch (nvm->override) {
00131     case e1000_nvm_override_spi_large:
00132         nvm->type = e1000_nvm_eeprom_spi;
00133         eecd |= E1000_EECD_ADDR_BITS;
00134         break;
00135     case e1000_nvm_override_spi_small:
00136         nvm->type = e1000_nvm_eeprom_spi;
00137         eecd &= ~E1000_EECD_ADDR_BITS;
00138         break;
00139     case e1000_nvm_override_microwire_large:
00140         nvm->type = e1000_nvm_eeprom_microwire;
00141         eecd |= E1000_EECD_SIZE;
00142         break;
00143     case e1000_nvm_override_microwire_small:
00144         nvm->type = e1000_nvm_eeprom_microwire;
00145         eecd &= ~E1000_EECD_SIZE;
00146         break;
00147     default:
00148         nvm->type = eecd & E1000_EECD_TYPE
00149                     ? e1000_nvm_eeprom_spi
00150                     : e1000_nvm_eeprom_microwire;
00151         break;
00152     }
00153 
00154     if (nvm->type == e1000_nvm_eeprom_spi) {
00155         nvm->address_bits       = (eecd & E1000_EECD_ADDR_BITS)
00156                                   ? 16 : 8;
00157         nvm->delay_usec         = 1;
00158         nvm->opcode_bits        = 8;
00159         nvm->page_size          = (eecd & E1000_EECD_ADDR_BITS)
00160                                   ? 32 : 8;
00161 
00162         /* Function Pointers */
00163         nvm->ops.acquire        = e1000_acquire_nvm_generic;
00164         nvm->ops.read           = e1000_read_nvm_spi;
00165         nvm->ops.release        = e1000_release_nvm_generic;
00166         nvm->ops.update         = e1000_update_nvm_checksum_generic;
00167         nvm->ops.valid_led_default = e1000_valid_led_default_generic;
00168         nvm->ops.validate       = e1000_validate_nvm_checksum_generic;
00169         nvm->ops.write          = e1000_write_nvm_spi;
00170 
00171         /*
00172          * nvm->word_size must be discovered after the pointers
00173          * are set so we can verify the size from the nvm image
00174          * itself.  Temporarily set it to a dummy value so the
00175          * read will work.
00176          */
00177         nvm->word_size = 64;
00178         ret_val = nvm->ops.read(hw, NVM_CFG, 1, &size);
00179         if (ret_val)
00180             goto out;
00181         size = (size & NVM_SIZE_MASK) >> NVM_SIZE_SHIFT;
00182         /*
00183          * if size != 0, it can be added to a constant and become
00184          * the left-shift value to set the word_size.  Otherwise,
00185          * word_size stays at 64.
00186          */
00187         if (size) {
00188             size += NVM_WORD_SIZE_BASE_SHIFT_82541;
00189             nvm->word_size = 1 << size;
00190         }
00191     } else {
00192         nvm->address_bits       = (eecd & E1000_EECD_ADDR_BITS)
00193                                   ? 8 : 6;
00194         nvm->delay_usec         = 50;
00195         nvm->opcode_bits        = 3;
00196         nvm->word_size          = (eecd & E1000_EECD_ADDR_BITS)
00197                                   ? 256 : 64;
00198 
00199         /* Function Pointers */
00200         nvm->ops.acquire        = e1000_acquire_nvm_generic;
00201         nvm->ops.read           = e1000_read_nvm_microwire;
00202         nvm->ops.release        = e1000_release_nvm_generic;
00203         nvm->ops.update         = e1000_update_nvm_checksum_generic;
00204         nvm->ops.valid_led_default = e1000_valid_led_default_generic;
00205         nvm->ops.validate       = e1000_validate_nvm_checksum_generic;
00206         nvm->ops.write          = e1000_write_nvm_microwire;

00292 static s32 e1000_reset_hw_82541(struct e1000_hw *hw)
00293 {
00294     u32 ledctl, ctrl, manc;
00295 
00296     DEBUGFUNC("e1000_reset_hw_82541");
00297 
00298     DEBUGOUT("Masking off all interrupts\n");
00299     E1000_WRITE_REG(hw, E1000_IMC, 0xFFFFFFFF);
00300 
00301     E1000_WRITE_REG(hw, E1000_RCTL, 0);
00302     E1000_WRITE_REG(hw, E1000_TCTL, E1000_TCTL_PSP);
00303     E1000_WRITE_FLUSH(hw);
00304 
00305     /*
00306      * Delay to allow any outstanding PCI transactions to complete
00307      * before resetting the device.
00308      */
00309     msec_delay(10);
00310 
00311     ctrl = E1000_READ_REG(hw, E1000_CTRL);
00312 
00313     /* Must reset the Phy before resetting the MAC */
00314     if ((hw->mac.type == e1000_82541) || (hw->mac.type == e1000_82547)) {
00315         E1000_WRITE_REG(hw, E1000_CTRL, (ctrl | E1000_CTRL_PHY_RST));
00316         msec_delay(5);
00317     }
00318 
00319     DEBUGOUT("Issuing a global reset to 82541/82547 MAC\n");
00320     switch (hw->mac.type) {
00321     case e1000_82541:
00322     case e1000_82541_rev_2:
00323         /*
00324          * These controllers can't ack the 64-bit write when
00325          * issuing the reset, so we use IO-mapping as a
00326          * workaround to issue the reset.
00327          */
00328         E1000_WRITE_REG_IO(hw, E1000_CTRL, ctrl | E1000_CTRL_RST);
00329         break;
00330     default:
00331         E1000_WRITE_REG(hw, E1000_CTRL, ctrl | E1000_CTRL_RST);
00332         break;
00333     }
00334 
00335     /* Wait for NVM reload */
00336     msec_delay(20);
00337 
00338     /* Disable HW ARPs on ASF enabled adapters */
00339     manc = E1000_READ_REG(hw, E1000_MANC);
00340     manc &= ~E1000_MANC_ARP_EN;
00341     E1000_WRITE_REG(hw, E1000_MANC, manc);
00342 
00343     if ((hw->mac.type == e1000_82541) || (hw->mac.type == e1000_82547)) {
00344         e1000_phy_init_script_82541(hw);
00345 
00346         /* Configure activity LED after Phy reset */
00347         ledctl = E1000_READ_REG(hw, E1000_LEDCTL);
00348         ledctl &= IGP_ACTIVITY_LED_MASK;
00349         ledctl |= (IGP_ACTIVITY_LED_ENABLE | IGP_LED3_MODE);
00350         E1000_WRITE_REG(hw, E1000_LEDCTL, ledctl);
00351     }
00352 
00353     /* Once again, mask the interrupts */
00354     DEBUGOUT("Masking off all interrupts\n");
00355     E1000_WRITE_REG(hw, E1000_IMC, 0xFFFFFFFF);
00356 
00357     /* Clear any pending interrupt events. */
00358     E1000_READ_REG(hw, E1000_ICR);
00359 
00360     return E1000_SUCCESS;
00361 }

00369 static s32 e1000_init_hw_82541(struct e1000_hw *hw)
00370 {
00371     struct e1000_mac_info *mac = &hw->mac;
00372     struct e1000_dev_spec_82541 *dev_spec = &hw->dev_spec._82541;
00373     u32 i, txdctl;
00374     s32 ret_val;
00375 
00376     DEBUGFUNC("e1000_init_hw_82541");
00377 
00378     /* Initialize identification LED */
00379     ret_val = mac->ops.id_led_init(hw);
00380     if (ret_val) {
00381         DEBUGOUT("Error initializing identification LED\n");
00382         /* This is not fatal and we should not stop init due to this */
00383     }
00384         
00385     /* Storing the Speed Power Down  value for later use */
00386     ret_val = hw->phy.ops.read_reg(hw,
00387                                    IGP01E1000_GMII_FIFO,
00388                                    &dev_spec->spd_default);
00389     if (ret_val)
00390         goto out;
00391 
00392     /* Disabling VLAN filtering */
00393     DEBUGOUT("Initializing the IEEE VLAN\n");
00394     mac->ops.clear_vfta(hw);
00395 
00396     /* Setup the receive address. */
00397     e1000_init_rx_addrs_generic(hw, mac->rar_entry_count);
00398 
00399     /* Zero out the Multicast HASH table */
00400     DEBUGOUT("Zeroing the MTA\n");
00401     for (i = 0; i < mac->mta_reg_count; i++) {
00402         E1000_WRITE_REG_ARRAY(hw, E1000_MTA, i, 0);
00403         /*
00404          * Avoid back to back register writes by adding the register
00405          * read (flush).  This is to protect against some strange
00406          * bridge configurations that may issue Memory Write Block
00407          * (MWB) to our register space.
00408          */
00409         E1000_WRITE_FLUSH(hw);
00410     }
00411 
00412     /* Setup link and flow control */
00413     ret_val = mac->ops.setup_link(hw);
00414 
00415     txdctl = E1000_READ_REG(hw, E1000_TXDCTL(0));
00416     txdctl = (txdctl & ~E1000_TXDCTL_WTHRESH) |
00417              E1000_TXDCTL_FULL_TX_DESC_WB;
00418     E1000_WRITE_REG(hw, E1000_TXDCTL(0), txdctl);
00419 
00420     /*
00421      * Clear all of the statistics registers (clear on read).  It is
00422      * important that we do this after we have tried to establish link
00423      * because the symbol error count will increment wildly if there
00424      * is no link.
00425      */
00426     e1000_clear_hw_cntrs_82541(hw);
00427 
00428 out:
00429     return ret_val;
00430 }

00441                                         u16 *duplex)
00442 {

00444     s32 ret_val;

00446 
00447     DEBUGFUNC("e1000_get_link_up_info_82541");

00450     if (ret_val)

00454         goto out;

00470         if (ret_val)
00471             goto out;
00472 
00473         if (*speed == SPEED_100) {
00474             if (!(data & NWAY_LPAR_100TX_FD_CAPS))
00475                 *duplex = HALF_DUPLEX;
00476         } else if (*speed == SPEED_10) {
00477             if (!(data & NWAY_LPAR_10T_FD_CAPS))
00478                 *duplex = HALF_DUPLEX;

00480     }

00482 out:

00589     /*
00590      * We only want to go out to the PHY registers to see if Auto-Neg
00591      * has completed and/or if our link status has changed.  The
00592      * get_link_status flag is set upon receiving a Link Status
00593      * Change or Rx Sequence Error interrupt.
00594      */
00595     if (!mac->get_link_status) {
00596         ret_val = E1000_SUCCESS;
00597         goto out;
00598     }
00599 
00600     /*
00601      * First we want to see if the MII Status Register reports
00602      * link.  If so, then we want to get the current speed/duplex
00603      * of the PHY.
00604      */
00605     ret_val = e1000_phy_has_link_generic(hw, 1, 0, &link);
00606     if (ret_val)
00607         goto out;
00608 
00609     if (!link) {
00610         ret_val = e1000_config_dsp_after_link_change_82541(hw, false);
00611         goto out; /* No link detected */
00612     }
00613 
00614     mac->get_link_status = false;
00615 
00616     /*
00617      * Check if there was DownShift, must be checked
00618      * immediately after link-up
00619      */
00620     e1000_check_downshift_generic(hw);
00621 
00622     /*
00623      * If we are forcing speed/duplex, then we simply return since
00624      * we have already determined whether we have link or not.
00625      */
00626     if (!mac->autoneg) {
00627         ret_val = -E1000_ERR_CONFIG;
00628         goto out;
00629     }
00630 
00631     ret_val = e1000_config_dsp_after_link_change_82541(hw, true);
00632 
00633     /*
00634      * Auto-Neg is enabled.  Auto Speed Detection takes care
00635      * of MAC speed/duplex configuration.  So we only need to
00636      * configure Collision Distance in the MAC.
00637      */
00638     e1000_config_collision_dist_generic(hw);
00639 
00640     /*
00641      * Configure Flow Control now that Auto-Neg has completed.
00642      * First, we need to restore the desired flow control
00643      * settings because we may have had to re-autoneg with a
00644      * different link partner.
00645      */
00646     ret_val = e1000_config_fc_after_link_up_generic(hw);
00647     if (ret_val) {
00648         DEBUGOUT("Error configuring flow control\n");
00649     }
00650 
00651 out:
00652     return ret_val;
00653 }

00666 static s32 e1000_config_dsp_after_link_change_82541(struct e1000_hw *hw,
00667                                                     bool link_up)
00668 {
00669     struct e1000_phy_info *phy = &hw->phy;
00670     struct e1000_dev_spec_82541 *dev_spec = &hw->dev_spec._82541;
00671     s32 ret_val;
00672     u32 idle_errs = 0;
00673     u16 phy_data, phy_saved_data, speed, duplex, i;
00674     u16 ffe_idle_err_timeout = FFE_IDLE_ERR_COUNT_TIMEOUT_20;
00675     u16 dsp_reg_array[IGP01E1000_PHY_CHANNEL_NUM] =
00676                                                {IGP01E1000_PHY_AGC_PARAM_A,
00677                                                 IGP01E1000_PHY_AGC_PARAM_B,
00678                                                 IGP01E1000_PHY_AGC_PARAM_C,
00679                                                 IGP01E1000_PHY_AGC_PARAM_D};
00680 
00681     DEBUGFUNC("e1000_config_dsp_after_link_change_82541");
00682 
00683     if (link_up) {
00684         ret_val = hw->mac.ops.get_link_up_info(hw, &speed, &duplex);
00685         if (ret_val) {
00686             DEBUGOUT("Error getting link speed and duplex\n");
00687             goto out;
00688         }
00689 
00690         if (speed != SPEED_1000) {
00691             ret_val = E1000_SUCCESS;
00692             goto out;
00693         }
00694 
00695         ret_val = phy->ops.get_cable_length(hw);
00696         if (ret_val)
00697             goto out;
00698 
00699         if ((dev_spec->dsp_config == e1000_dsp_config_enabled) &&
00700             phy->min_cable_length >= 50) {
00701 
00702             for (i = 0; i < IGP01E1000_PHY_CHANNEL_NUM; i++) {
00703                 ret_val = phy->ops.read_reg(hw,
00704                                             dsp_reg_array[i],
00705                                             &phy_data);
00706                 if (ret_val)
00707                     goto out;
00708 
00709                 phy_data &= ~IGP01E1000_PHY_EDAC_MU_INDEX;
00710 
00711                 ret_val = phy->ops.write_reg(hw,
00712                                              dsp_reg_array[i],
00713                                              phy_data);
00714                 if (ret_val)
00715                     goto out;
00716             }
00717             dev_spec->dsp_config = e1000_dsp_config_activated;
00718         }
00719 
00720         if ((dev_spec->ffe_config != e1000_ffe_config_enabled) ||
00721             (phy->min_cable_length >= 50)) {
00722             ret_val = E1000_SUCCESS;
00723             goto out;
00724         }
00725 
00726         /* clear previous idle error counts */
00727         ret_val = phy->ops.read_reg(hw, PHY_1000T_STATUS, &phy_data);
00728         if (ret_val)
00729             goto out;
00730 
00731         for (i = 0; i < ffe_idle_err_timeout; i++) {
00732             usec_delay(1000);
00733             ret_val = phy->ops.read_reg(hw,
00734                                         PHY_1000T_STATUS,
00735                                         &phy_data);
00736             if (ret_val)
00737                 goto out;
00738 
00739             idle_errs += (phy_data & SR_1000T_IDLE_ERROR_CNT);
00740             if (idle_errs > SR_1000T_PHY_EXCESSIVE_IDLE_ERR_COUNT) {
00741                 dev_spec->ffe_config = e1000_ffe_config_active;
00742 
00743                 ret_val = phy->ops.write_reg(hw,
00744                                   IGP01E1000_PHY_DSP_FFE,
00745                                   IGP01E1000_PHY_DSP_FFE_CM_CP);
00746                 if (ret_val)
00747                     goto out;
00748                 break;
00749             }
00750 
00751             if (idle_errs)
00752                 ffe_idle_err_timeout =
00753                                  FFE_IDLE_ERR_COUNT_TIMEOUT_100;
00754         }
00755     } else {
00756         if (dev_spec->dsp_config == e1000_dsp_config_activated) {
00757             /*
00758              * Save off the current value of register 0x2F5B
00759              * to be restored at the end of the routines.
00760              */
00761             ret_val = phy->ops.read_reg(hw,
00762                                         0x2F5B,
00763                                         &phy_saved_data);
00764             if (ret_val)
00765                 goto out;
00766 
00767             /* Disable the PHY transmitter */
00768             ret_val = phy->ops.write_reg(hw, 0x2F5B, 0x0003);
00769             if (ret_val)
00770                 goto out;
00771 
00772             msec_delay_irq(20);
00773 
00774             ret_val = phy->ops.write_reg(hw,
00775                                          0x0000,
00776                                          IGP01E1000_IEEE_FORCE_GIG);
00777             if (ret_val)
00778                 goto out;
00779             for (i = 0; i < IGP01E1000_PHY_CHANNEL_NUM; i++) {
00780                 ret_val = phy->ops.read_reg(hw,
00781                                             dsp_reg_array[i],
00782                                             &phy_data);
00783                 if (ret_val)
00784                     goto out;
00785 
00786                 phy_data &= ~IGP01E1000_PHY_EDAC_MU_INDEX;
00787                 phy_data |= IGP01E1000_PHY_EDAC_SIGN_EXT_9_BITS;
00788 
00789                 ret_val = phy->ops.write_reg(hw,
00790                                              dsp_reg_array[i],
00791                                              phy_data);
00792                 if (ret_val)
00793                     goto out;
00794             }
00795 
00796             ret_val = phy->ops.write_reg(hw,
00797                                    0x0000,
00798                                    IGP01E1000_IEEE_RESTART_AUTONEG);
00799             if (ret_val)
00800                 goto out;
00801 
00802             msec_delay_irq(20);
00803 
00804             /* Now enable the transmitter */
00805             ret_val = phy->ops.write_reg(hw,
00806                                          0x2F5B,
00807                                          phy_saved_data);
00808             if (ret_val)
00809                 goto out;
00810 
00811             dev_spec->dsp_config = e1000_dsp_config_enabled;
00812         }
00813 
00814         if (dev_spec->ffe_config != e1000_ffe_config_active) {
00815             ret_val = E1000_SUCCESS;
00816             goto out;
00817         }
00818 
00819         /*
00820          * Save off the current value of register 0x2F5B
00821          * to be restored at the end of the routines.
00822          */
00823         ret_val = phy->ops.read_reg(hw, 0x2F5B, &phy_saved_data);
00824         if (ret_val)
00825             goto out;
00826 
00827         /* Disable the PHY transmitter */
00828         ret_val = phy->ops.write_reg(hw, 0x2F5B, 0x0003);
00829         if (ret_val)
00830             goto out;
00831 
00832         msec_delay_irq(20);
00833 
00834         ret_val = phy->ops.write_reg(hw,
00835                                      0x0000,
00836                                      IGP01E1000_IEEE_FORCE_GIG);
00837         if (ret_val)
00838             goto out;
00839 
00840         ret_val = phy->ops.write_reg(hw,
00841                                      IGP01E1000_PHY_DSP_FFE,
00842                                      IGP01E1000_PHY_DSP_FFE_DEFAULT);
00843         if (ret_val)
00844             goto out;
00845 
00846         ret_val = phy->ops.write_reg(hw,
00847                                      0x0000,
00848                                      IGP01E1000_IEEE_RESTART_AUTONEG);
00849         if (ret_val)
00850             goto out;
00851 
00852         msec_delay_irq(20);
00853 
00854         /* Now enable the transmitter */
00855         ret_val = phy->ops.write_reg(hw, 0x2F5B, phy_saved_data);
00856 
00857         if (ret_val)
00858             goto out;
00859 
00860         dev_spec->ffe_config = e1000_ffe_config_enabled;
00861     }
00862 
00863 out:
00864     return ret_val;
00865 }

00878 static s32 e1000_get_cable_length_igp_82541(struct e1000_hw *hw)
00879 {
00880     struct e1000_phy_info *phy = &hw->phy;
00881     s32 ret_val = E1000_SUCCESS;
00882     u16 i, data;
00883     u16 cur_agc_value, agc_value = 0;
00884     u16 min_agc_value = IGP01E1000_AGC_LENGTH_TABLE_SIZE;
00885     u16 agc_reg_array[IGP01E1000_PHY_CHANNEL_NUM] =
00886                                                      {IGP01E1000_PHY_AGC_A,
00887                                                       IGP01E1000_PHY_AGC_B,
00888                                                       IGP01E1000_PHY_AGC_C,
00889                                                       IGP01E1000_PHY_AGC_D};
00890 
00891     DEBUGFUNC("e1000_get_cable_length_igp_82541");
00892 
00893     /* Read the AGC registers for all channels */
00894     for (i = 0; i < IGP01E1000_PHY_CHANNEL_NUM; i++) {
00895         ret_val = phy->ops.read_reg(hw, agc_reg_array[i], &data);
00896         if (ret_val)
00897             goto out;
00898 
00899         cur_agc_value = data >> IGP01E1000_AGC_LENGTH_SHIFT;
00900 
00901         /* Bounds checking */
00902         if ((cur_agc_value >= IGP01E1000_AGC_LENGTH_TABLE_SIZE - 1) ||
00903             (cur_agc_value == 0)) {
00904             ret_val = -E1000_ERR_PHY;
00905             goto out;
00906         }
00907 
00908         agc_value += cur_agc_value;
00909 
00910         if (min_agc_value > cur_agc_value)
00911             min_agc_value = cur_agc_value;
00912     }
00913 
00914     /* Remove the minimal AGC result for length < 50m */
00915     if (agc_value < IGP01E1000_PHY_CHANNEL_NUM * 50) {
00916         agc_value -= min_agc_value;
00917         /* Average the three remaining channels for the length. */
00918         agc_value /= (IGP01E1000_PHY_CHANNEL_NUM - 1);
00919     } else {
00920         /* Average the channels for the length. */
00921         agc_value /= IGP01E1000_PHY_CHANNEL_NUM;
00922     }
00923 
00924     phy->min_cable_length = (e1000_igp_cable_length_table[agc_value] >
00925                              IGP01E1000_AGC_RANGE)
00926                             ? (e1000_igp_cable_length_table[agc_value] -
00927                                IGP01E1000_AGC_RANGE)
00928                             : 0;
00929     phy->max_cable_length = e1000_igp_cable_length_table[agc_value] +
00930                             IGP01E1000_AGC_RANGE;
00931 
00932     phy->cable_length = (phy->min_cable_length + phy->max_cable_length) / 2;
00933 
00934 out:
00935     return ret_val;
00936 }

00952 static s32 e1000_set_d3_lplu_state_82541(struct e1000_hw *hw, bool active)
00953 {
00954     struct e1000_phy_info *phy = &hw->phy;
00955     s32 ret_val;
00956     u16 data;
00957 
00958     DEBUGFUNC("e1000_set_d3_lplu_state_82541");
00959 
00960     switch (hw->mac.type) {
00961     case e1000_82541_rev_2:
00962     case e1000_82547_rev_2:
00963         break;
00964     default:
00965         ret_val = e1000_set_d3_lplu_state_generic(hw, active);
00966         goto out;
00967         break;
00968     }
00969 
00970     ret_val = phy->ops.read_reg(hw, IGP01E1000_GMII_FIFO, &data);
00971     if (ret_val)
00972         goto out;
00973 
00974     if (!active) {
00975         data &= ~IGP01E1000_GMII_FLEX_SPD;
00976         ret_val = phy->ops.write_reg(hw, IGP01E1000_GMII_FIFO, data);
00977         if (ret_val)
00978             goto out;
00979 
00980         /*
00981          * LPLU and SmartSpeed are mutually exclusive.  LPLU is used
00982          * during Dx states where the power conservation is most
00983          * important.  During driver activity we should enable
00984          * SmartSpeed, so performance is maintained.
00985          */
00986         if (phy->smart_speed == e1000_smart_speed_on) {
00987             ret_val = phy->ops.read_reg(hw,
00988                                         IGP01E1000_PHY_PORT_CONFIG,
00989                                         &data);
00990             if (ret_val)
00991                 goto out;
00992 
00993             data |= IGP01E1000_PSCFR_SMART_SPEED;
00994             ret_val = phy->ops.write_reg(hw,
00995                                          IGP01E1000_PHY_PORT_CONFIG,
00996                                          data);
00997             if (ret_val)
00998                 goto out;
00999         } else if (phy->smart_speed == e1000_smart_speed_off) {
01000             ret_val = phy->ops.read_reg(hw,
01001                                         IGP01E1000_PHY_PORT_CONFIG,
01002                                         &data);
01003             if (ret_val)
01004                 goto out;
01005 
01006             data &= ~IGP01E1000_PSCFR_SMART_SPEED;
01007             ret_val = phy->ops.write_reg(hw,
01008                                          IGP01E1000_PHY_PORT_CONFIG,
01009                                          data);
01010             if (ret_val)
01011                 goto out;
01012         }
01013     } else if ((phy->autoneg_advertised == E1000_ALL_SPEED_DUPLEX) ||
01014                (phy->autoneg_advertised == E1000_ALL_NOT_GIG) ||
01015                (phy->autoneg_advertised == E1000_ALL_10_SPEED)) {
01016         data |= IGP01E1000_GMII_FLEX_SPD;
01017         ret_val = phy->ops.write_reg(hw, IGP01E1000_GMII_FIFO, data);
01018         if (ret_val)
01019             goto out;
01020 
01021         /* When LPLU is enabled, we should disable SmartSpeed */
01022         ret_val = phy->ops.read_reg(hw,
01023                                     IGP01E1000_PHY_PORT_CONFIG,
01024                                     &data);
01025         if (ret_val)
01026             goto out;
01027 
01028         data &= ~IGP01E1000_PSCFR_SMART_SPEED;
01029         ret_val = phy->ops.write_reg(hw,
01030                                      IGP01E1000_PHY_PORT_CONFIG,
01031                                      data);
01032     }
01033 
01034 out:
01035     return ret_val;
