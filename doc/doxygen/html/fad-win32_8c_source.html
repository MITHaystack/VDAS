00054 add_addr_to_list(pcap_if_t *curdev, struct sockaddr *addr,

00056     struct sockaddr *dstaddr, char *errbuf)

00091         curaddr->netmask = NULL;

00095         if (curaddr->broadaddr == NULL) {

00097                 "malloc: %s", pcap_strerror(errno));
00098             free(curaddr);
00099             return (-1);
00100         }
00101     } else
00102         curaddr->broadaddr = NULL;

00105         curaddr->dstaddr = (struct sockaddr*)dup_sockaddr(dstaddr, sizeof(struct sockaddr_storage));
00106         if (curaddr->dstaddr == NULL) {

00109             free(curaddr);

00112     } else
00113         curaddr->dstaddr = NULL;
00114         
00115     /*
00116      * Find the end of the list of addresses.
00117      */
00118     for (prevaddr = curdev->addresses; prevaddr != NULL; prevaddr = nextaddr) {
00119         nextaddr = prevaddr->next;
00120         if (nextaddr == NULL) {
00121             /*
00122              * This is the end of the list.

00124             break;

00129         /*

00132         curdev->addresses = curaddr;
00133     } else {
00134         /*
00135          * "prevaddr" is the last member of the list; append
00136          * this member to it.
00137          */
00138         prevaddr->next = curaddr;

00146 pcap_add_if_win32(pcap_if_t **devlist, char *name, const char *desc,
00147     char *errbuf)
00148 {
00149     pcap_if_t *curdev;
00150     npf_if_addr if_addrs[MAX_NETWORK_ADDRESSES];
00151     LONG if_addr_size;
00152     int res = 0;
00153 
00154     if_addr_size = MAX_NETWORK_ADDRESSES;
00155 
00156     /*
00157      * Add an entry for this interface, with no addresses.
00158      */
00159     if (add_or_find_if(&curdev, devlist, name, 0, desc, errbuf) == -1) {
00160         /*
00161          * Failure.
00162          */
00163         return (-1);
00164     }
00165 
00166     /*
00167      * Get the list of addresses for the interface.
00168      */
00169     if (!PacketGetNetInfoEx((void *)name, if_addrs, &if_addr_size)) {
00170         /*
00171          * Failure.
00172          *
00173          * We don't return an error, because this can happen with
00174          * NdisWan interfaces, and we want to supply them even
00175          * if we can't supply their addresses.
00176          *
00177          * We return an entry with an empty address list.
00178          */
00179         return (0);
00180     }
00181 
00182     /*
00183      * Now add the addresses.
00184      */
00185     while (if_addr_size-- > 0) {
00186         /*
00187          * "curdev" is an entry for this interface; add an entry for
00188          * this address to its list of addresses.
00189          */
00190         if(curdev == NULL)
00191             break;
00192         res = add_addr_to_list(curdev,
00193             (struct sockaddr *)&if_addrs[if_addr_size].IPAddress,
00194             (struct sockaddr *)&if_addrs[if_addr_size].SubnetMask,
00195             (struct sockaddr *)&if_addrs[if_addr_size].Broadcast,
00196             NULL,
00197             errbuf);
00198         if (res == -1) {
00199             /*
00200              * Failure.
00201              */
00202             break;
00203         }
00204     }
00205 
00206     return (res);

00210 /*
00211  * Get a list of all interfaces that are up and that we can open.
00212  * Returns -1 on error, 0 otherwise.
00213  * The list, as returned through "alldevsp", may be null if no interfaces
00214  * were up and could be opened.
00215  *
00216  * Win32 implementation, based on WinPcap
00217  */
00218 int
00219 pcap_findalldevs(pcap_if_t **alldevsp, char *errbuf)
00220 {
00221     pcap_if_t *devlist = NULL;
00222     int ret = 0;
00223     const char *desc;
00224     char *AdaptersName;
00225     ULONG NameLength;
00226     char *name;
00227     
00228     if (!PacketGetAdapterNames(NULL, &NameLength))
00229     {
00230         DWORD last_error = GetLastError();
00231 
00232         if (last_error != ERROR_INSUFFICIENT_BUFFER)
00233         {
00234             snprintf(errbuf, PCAP_ERRBUF_SIZE,
00235                 "PacketGetAdapterNames: %s",
00236                 pcap_win32strerror());
00237             return (-1);
00238         }
00239     }
00240 
00241     if (NameLength > 0)
00242         AdaptersName = (char*) malloc(NameLength);
00243     else
00244     {
00245         *alldevsp = NULL;
00246         return 0;
00247     }
00248     if (AdaptersName == NULL)
00249     {
00250         snprintf(errbuf, PCAP_ERRBUF_SIZE, "Cannot allocate enough memory to list the adapters.");
00251         return (-1);
00252     }           
00253 
00254     if (!PacketGetAdapterNames(AdaptersName, &NameLength)) {
00255         snprintf(errbuf, PCAP_ERRBUF_SIZE,
00256             "PacketGetAdapterNames: %s",
00257             pcap_win32strerror());
00258         free(AdaptersName);
00259         return (-1);
00260     }
00261     
00262     /*
00263      * "PacketGetAdapterNames()" returned a list of
00264      * null-terminated ASCII interface name strings,
00265      * terminated by a null string, followed by a list
00266      * of null-terminated ASCII interface description
00267      * strings, terminated by a null string.
00268      * This means there are two ASCII nulls at the end
00269      * of the first list.
00270      *
00271      * Find the end of the first list; that's the
00272      * beginning of the second list.
00273      */
00274     desc = &AdaptersName[0];
00275     while (*desc != '\0' || *(desc + 1) != '\0')
00276         desc++;
00277     
00278     /*
00279      * Found it - "desc" points to the first of the two
00280      * nulls at the end of the list of names, so the
00281      * first byte of the list of descriptions is two bytes
00282      * after it.
00283      */
00284     desc += 2;
00285     
00286     /*
00287      * Loop over the elements in the first list.
00288      */
00289     name = &AdaptersName[0];
00290     while (*name != '\0') {
00291         /*
00292          * Add an entry for this interface.
00293          */
00294         if (pcap_add_if_win32(&devlist, name, desc, errbuf) == -1) {
00295             /*
00296              * Failure.
00297              */
00298             ret = -1;
00299             break;
00300         }
00301         name += strlen(name) + 1;
00302         desc += strlen(desc) + 1;
00303     }
00304 
00305     if (ret != -1) {
00306         /*
00307          * We haven't had any errors yet; do any platform-specific
00308          * operations to add devices.
00309          */
00310         if (pcap_platform_finddevs(&devlist, errbuf) < 0)
00311             ret = -1;
00312     }
00313     
00314     if (ret == -1) {
00315         /*
00316          * We had an error; free the list we've been constructing.
00317          */
00318         if (devlist != NULL) {
00319             pcap_freealldevs(devlist);
00320             devlist = NULL;
00321         }
00322     }
00323     
00324     *alldevsp = devlist;
00325     free(AdaptersName);
00326     return (ret);
00327 }
