00739  * This function is used to work around a quirk of the linux kernel which
00740  * would otherwise cause our driver to leak 16MB of ram per interface
00741  * when PAT write-combining is used.
00742  *
00743  * Early in the boot process, Linux maps linearly all physical space
00744  * at: [PAGE_OFFSET, PAGE_OFFSET + <end-of-usable-physical-space]
00745  * ioremap() gives a new mapping for the same physical space in a
00746  * different virtual region (the "vmalloc" region). Linux tries to
00747  * keep the same attributes for the two virtual mapping of the same
00748  * physical space through the clumsy change_page_attr(). After
00749  * ioremap() has more or less established the new mapping,
00750  * change_page_attr() is called on the corresponding interval of the
00751  * "linear mapping" to fix it, and this part is confused by the
00752  * presence of the PAT (_PAGE_PSE) bit in the pte entries and leaks
00753  * memory.  This function is designed to be called prior to iounmap()
00754  * to clear the _PAGE_PSE bits in the linear mapping, and eliminate
00755  * this leak of vmalloc space.

00775     if (mgp->wc_enabled != 2 || myri10ge_pat_idx < 4)
00776         return;

00779     if (addr == 0)

00786         return;
00787 #endif

00899 static int
00900 myri10ge_send_cmd(struct myri10ge_priv *mgp, u32 cmd,
00901           struct myri10ge_cmd *data, int atomic)
00902 {
00903     struct mcp_cmd *buf;
00904     char buf_bytes[sizeof(*buf) + 8];
00905     struct mcp_cmd_response *response = mgp->cmd;
00906     char __iomem *cmd_addr = mgp->sram + MXGEFW_ETH_CMD;
00907     u32 dma_low, dma_high, result, value;
00908     int sleep_total = 0;
00909 
00910     /* ensure buf is aligned to 8 bytes */
00911     buf = (struct mcp_cmd *) ALIGN((unsigned long) buf_bytes, 8);
00912 
00913     buf->data0 = htonl(data->data0);
00914     buf->data1 = htonl(data->data1);
00915     buf->data2 = htonl(data->data2);
00916     buf->cmd = htonl(cmd);
00917     dma_low = MYRI10GE_LOWPART_TO_U32(mgp->cmd_bus);
00918     dma_high = MYRI10GE_HIGHPART_TO_U32(mgp->cmd_bus);
00919 
00920     buf->response_addr.low = htonl(dma_low);
00921     buf->response_addr.high = htonl(dma_high);
00922     response->result = htonl(MYRI10GE_NO_RESPONSE_RESULT);
00923     mb();
00924     myri10ge_pio_copy(cmd_addr, buf, sizeof (*buf));
00925 
00926     /* wait up to 15ms. Longest command is the DMA benchmark,
00927      * which is capped at 5ms, but runs from a timeout handler
00928      * that runs every 7.8ms. So a 15ms timeout leaves us with
00929      * a 2.2ms margin
00930      */
00931     if (atomic) {
00932         /* if atomic is set, do not sleep,
00933          * and try to get the completion quickly
00934          * (1ms will be enough for those commands) */
00935         for (sleep_total = 0;
00936              sleep_total < 1000
00937              && response->result == htonl(MYRI10GE_NO_RESPONSE_RESULT);
00938              sleep_total += 10) {
00939             udelay(10);
00940             mb();
00941         }
00942     } else {
00943         /* use msleep for most command */
00944         for (sleep_total = 0;
00945              sleep_total < MYRI10GE_NONATOMIC_SLEEP_MS
00946              && response->result == htonl(MYRI10GE_NO_RESPONSE_RESULT);
00947              sleep_total++)
00948             myri10ge_msleep(1);
00949     }
00950 
00951     result = ntohl(response->result);
00952     value = ntohl(response->data);
00953     if (result != MYRI10GE_NO_RESPONSE_RESULT) {
00954         if (result == 0) {
00955             data->data0 = value;
00956             return 0;
00957         } else if (result == MXGEFW_CMD_UNKNOWN) {
00958             return -ENOSYS;
00959         } else if (result == MXGEFW_CMD_ERROR_UNALIGNED) {
00960             return -E2BIG;
00961         } else if (result == MXGEFW_CMD_ERROR_RANGE &&
00962                cmd == MXGEFW_CMD_ENABLE_RSS_QUEUES &&
00963                (data->data1 & MXGEFW_SLICE_ENABLE_MULTIPLE_TX_QUEUES) != 0) {
00964             return -ERANGE;
00965         } else {
00966             dev_err(&mgp->pdev->dev,
00967                 "command %d failed, result = %d\n",
00968                 cmd, result);
00969             return -ENXIO;
00970         }
00971     }
00972 
00973     dev_err(&mgp->pdev->dev, "command %d timed out, result = %d\n",
00974            cmd, result);
00975     return -EAGAIN;
00976 }

01386 static int
01387 myri10ge_load_firmware(struct myri10ge_priv *mgp, int adopt)
01388 {
01389     char __iomem *submit;
01390     __be32 buf[16] __attribute__((__aligned__(8)));
01391     u32 dma_low, dma_high, size;
01392     int status, i;
01393 
01394     size = 0;
01395 #if MYRI10GE_BUILTIN_FW
01396     status = myri10ge_load_builtin_firmware(mgp, &size);
01397 #else /* MYRI10GE_BUILTIN_FW */
01398     status = myri10ge_load_hotplug_firmware(mgp, &size);
01399 #endif /* MYRI10GE_BUILTIN_FW */
01400     if (status) {
01401         if (!adopt)
01402             return status;
01403 #if MYRI10GE_BUILTIN_FW
01404         dev_warn(&mgp->pdev->dev, "firmware loading failed\n");
01405 #else
01406         dev_warn(&mgp->pdev->dev, "hotplug firmware loading failed\n");
01407 #endif /* MYRI10GE_BUILTIN_FW */
01408 
01409         /* Do not attempt to adopt firmware if there
01410            was a bad crc */
01411         if (status == -EIO)
01412             return status;
01413 
01414         status = myri10ge_adopt_running_firmware(mgp);
01415         if (status != 0) {
01416             dev_err(&mgp->pdev->dev,
01417                 "failed to adopt running firmware\n");
01418             return status;
01419         }
01420         dev_info(&mgp->pdev->dev,
01421              "Successfully adopted running firmware\n");
01422         if (mgp->tx_boundary == 4096) {
01423             dev_warn(&mgp->pdev->dev,
01424                 "Using firmware currently running on NIC"
01425                  ".  For optimal\n");
01426             dev_warn(&mgp->pdev->dev,
01427                  "performance consider loading optimized "
01428                  "firmware\n");
01429             dev_warn(&mgp->pdev->dev, "via hotplug\n");
01430         }
01431 
01432         mgp->fw_name = "adopted";
01433         mgp->tx_boundary = 2048;
01434         myri10ge_dummy_rdma(mgp, 1);
01435         status = myri10ge_get_firmware_capabilities(mgp);
01436         return status;
01437     }
01438 
01439     /* clear confirmation addr */
01440     mgp->cmd->data = 0;
01441     mb();
01442 
01443     /* send a reload command to the bootstrap MCP, and wait for the
01444      *  response in the confirmation address.  The firmware should
01445      * write a -1 there to indicate it is alive and well
01446      */
01447     dma_low = MYRI10GE_LOWPART_TO_U32(mgp->cmd_bus);
01448     dma_high = MYRI10GE_HIGHPART_TO_U32(mgp->cmd_bus);
01449 
01450     buf[0] = htonl(dma_high);   /* confirm addr MSW */
01451     buf[1] = htonl(dma_low);    /* confirm addr LSW */
01452     buf[2] = MYRI10GE_NO_CONFIRM_DATA;  /* confirm data */
01453 
01454     /* FIX: All newest firmware should un-protect the bottom of
01455      * the sram before handoff. However, the very first interfaces
01456      * do not. Therefore the handoff copy must skip the first 8 bytes
01457      */
01458     buf[3] = htonl(MYRI10GE_FW_OFFSET + 8); /* where the code starts */
01459     buf[4] = htonl(size - 8);       /* length of code */
01460     buf[5] = htonl(8);          /* where to copy to */
01461     buf[6] = htonl(0);          /* where to jump to */
01462 
01463     submit = mgp->sram + MXGEFW_BOOT_HANDOFF;
01464 
01465     myri10ge_pio_copy(submit, &buf, sizeof (buf));
01466     mb();
01467     myri10ge_msleep(1);
01468     mb();
01469     i = 0;
01470     while (mgp->cmd->data != MYRI10GE_NO_CONFIRM_DATA && i < 9) {
01471         myri10ge_msleep(1 << i);
01472         i++;
01473     }
01474     if (mgp->cmd->data != MYRI10GE_NO_CONFIRM_DATA) {
01475         dev_err(&mgp->pdev->dev, "handoff failed\n");
01476         return -ENXIO;
01477     }
01478     myri10ge_dummy_rdma(mgp, 1);
01479     status = myri10ge_get_firmware_capabilities(mgp);
01480 
01481     return status;
01482 }

01639                 &cmd, 0);   
01640 
01641     /* Now exchange information about interrupts  */
01642 
01643     bytes = mgp->max_intr_slots * sizeof (*mgp->ss[0].rx_done.entry);
01644     cmd.data0 = (u32) bytes;
01645     status = myri10ge_send_cmd(mgp, MXGEFW_CMD_SET_INTRQ_SIZE, &cmd, 0);
01646 
01647     /* 
01648      * Even though we already know how many slices are supported
01649      * via myri10ge_probe_slices() MXGEFW_CMD_GET_MAX_RSS_QUEUES
01650      * has magic side effects, and must be called after a reset.
01651      * It must be called prior to calling any RSS related cmds,
01652      * including assigning an interrupt queue for anything but
01653      * slice 0.  It must also be called *after*
01654      * MXGEFW_CMD_SET_INTRQ_SIZE, since the intrq size is used by
01655      * the firmware to compute offsets.
01656      */
01657      
01658     if (mgp->num_slices > 1) {
01659 
01660         /* ask the maximum number of slices it supports */
01661         status = myri10ge_send_cmd(mgp, MXGEFW_CMD_GET_MAX_RSS_QUEUES,
01662                        &cmd, 0);
01663         if (status != 0) {
01664             dev_err(&mgp->pdev->dev,
01665                 "failed to get number of slices\n");
01666         }
01667 
01668         /* 
01669          * MXGEFW_CMD_ENABLE_RSS_QUEUES must be called prior
01670          * to setting up the interrupt queue DMA
01671          */
01672          
01673         cmd.data0 = mgp->num_slices;
01674         cmd.data1 = MXGEFW_SLICE_INTR_MODE_ONE_PER_SLICE;
01675         if (MYRI10GE_GET_NUM_TXQ(mgp->dev) > 1)

01678                        &cmd, 0);

01681            RX queues, so if we get an error, first retry using a
01682            single TX queue before giving up */
01683         if (status != 0 && MYRI10GE_GET_NUM_TXQ(mgp->dev) > 1) {

01686             cmd.data1 = MXGEFW_SLICE_INTR_MODE_ONE_PER_SLICE;
01687             status = myri10ge_send_cmd(mgp,
01688                            MXGEFW_CMD_ENABLE_RSS_QUEUES,
01689                            &cmd, 0);
01690         }           
01691 
01692         if (status != 0) {
01693             dev_err(&mgp->pdev->dev,
01694                 "failed to set number of slices\n");
01695                 
01696             return status;
01697         }
01698     }
01699     for (i = 0; i < mgp->num_slices; i++) {
01700         ss = &mgp->ss[i];
01701         cmd.data0 = MYRI10GE_LOWPART_TO_U32(ss->rx_done.bus);
01702         cmd.data1 = MYRI10GE_HIGHPART_TO_U32(ss->rx_done.bus);
01703         cmd.data2 = i;
01704         status |= myri10ge_send_cmd(mgp, MXGEFW_CMD_SET_INTRQ_DMA,
01705                                     &cmd, 0);
01706     };
01707

01712     }

01723     }

01729     for (i = 0; i < mgp->num_slices; i++) {
01730         ss = &mgp->ss[i];
01731         if (status == 0) {
01732             ss->dca_tag = (__iomem __be32 *)
01733                 (mgp->sram + dca_tag_off + 4 * i);
01734         } else {
01735             ss->dca_tag = NULL;
01736         }
01737     }
01738 #endif /* MYRI10GE_HAVE_DCA */
01739 #ifdef MYRI10GE_RELAX_RX_ALIGN
01740     status = myri10ge_send_cmd(mgp, MXGEFW_CMD_RELAX_RXBUFFER_ALIGNMENT, &cmd, 0);
01741     if (status != 0) {
01742         dev_err(&mgp->pdev->dev, "failed set relaxex rx alignment\n");
01743         return status;
01744     }
01745 #endif /* MYRI10GE_RELAX_RX_ALIGN */
01746 
01747     /* reset mcp/driver shared state back to 0 */
01748 
01749     mgp->link_changes = 0;
01750     for (i = 0; i < mgp->num_slices; i++) {
01751         ss = &mgp->ss[i];
01752 
01753         memset(ss->rx_done.entry, 0, bytes);
01754         ss->tx.req = 0;
01755         ss->tx.done = 0;
01756         ss->tx.pkt_start = 0;
01757         ss->tx.pkt_done = 0;
01758         ss->rx_big.cnt = 0;
01759         ss->rx_small.cnt = 0;
01760         ss->rx_done.idx = 0;
01761         ss->rx_done.cnt = 0;
01762         ss->tx.wake_queue = 0;
01763         ss->tx.stop_queue = 0;
01764     }
01765 
01766 #ifndef LINUX_KERNEL_SPECIFIC   
01767     mgp->adapt_coal.usecs = -1;
01768     mgp->adapt_coal.old_rx_bytes = 0;
01769     mgp->adapt_coal.old_tx_bytes = 0;
01770 #endif
01771     status = myri10ge_update_mac_address(mgp, mgp->dev->dev_addr);
01772     myri10ge_change_pause(mgp, mgp->pause);
01773     myri10ge_set_multicast_list(mgp->dev);
01774 #if MYRI10GE_THROTTLE
01775     if (mgp->throttle) {
01776         cmd.data0 = myri10ge_throttle;
01777         if (myri10ge_send_cmd(mgp, MXGEFW_CMD_SET_THROTTLE_FACTOR,
01778                       &cmd, 0) != 0) {
01779             dev_err(&mgp->pdev->dev, "failed to set throttle\n");
01780         }
01781     }
01782 #endif
01783     return status;
01784 }

01898 static inline void
01899 myri10ge_rx_skb_build(struct sk_buff *skb, u8 *va, struct skb_frag_struct *rx_frags, 
01900               int len, int hlen)
01901 {
01902     struct skb_frag_struct *skb_frags;
01903 
01904     skb->len = skb->data_len = len;
01905     myri10ge_set_truesize(skb, len + sizeof (struct sk_buff));
01906     /* attach the page(s) */
01907 
01908     skb_frags = skb_shinfo(skb)->frags;
01909     while (len > 0) {
01910         memcpy(skb_frags, rx_frags, sizeof (*skb_frags));
01911         len -= rx_frags->size;
01912         skb_frags++;
01913         rx_frags++;
01914         skb_shinfo(skb)->nr_frags++;
01915     }
01916 
01917     /* pskb_may_pull is not available in irq context, but
01918        skb_pull() (for ether_pad and eth_type_trans()) requires
01919        the beginning of the packet in skb_headlen(), move it
01920        manually */
01921     myri10ge_skb_copy_to_linear_data(skb, va, hlen);
01922     skb_shinfo(skb)->frags[0].page_offset += hlen;
01923     skb_shinfo(skb)->frags[0].size -= hlen;
01924     skb->data_len -= hlen;
01925     skb->tail += hlen;
01926     skb_pull(skb, MXGEFW_PAD);
01927 }

01932 static void
01933 myri10ge_alloc_rx_pages(struct myri10ge_priv *mgp, struct myri10ge_rx_buf *rx,
01934             int bytes, int watchdog)
01935 {
01936     struct page *page;
01937     int idx;
01938 
01939 
01940     if (unlikely(rx->watchdog_needed && !watchdog))
01941         return;
01942 
01943     /* try to refill entire ring */
01944     while (rx->fill_cnt != (rx->cnt + rx->mask + 1)) {
01945         idx = rx->fill_cnt & rx->mask;
01946         if (rx->page_offset + bytes <= MYRI10GE_ALLOC_SIZE) {
01947             /* we can use part of previous page */
01948             get_page(rx->page);
01949         } else {
01950             /* we need a new page */
01951             page =
01952                 myri10ge_alloc_pages(GFP_ATOMIC | __GFP_COMP | __GFP_NOWARN,
01953                          MYRI10GE_ALLOC_ORDER);
01954             if (unlikely(page == NULL)) {
01955                 if (rx->fill_cnt - rx->cnt < 16)
01956                     rx->watchdog_needed = 1;
01957                 return;
01958             }
01959             rx->page = page;
01960             rx->page_offset = 0;
01961             rx->bus = pci_map_page(mgp->pdev, page, 0,
01962                            MYRI10GE_ALLOC_SIZE, PCI_DMA_FROMDEVICE);
01963         }
01964         rx->info[idx].rx__page = rx->page;
01965         rx->info[idx].rx__page_offset = rx->page_offset;
01966         /* note that this is the address of the start of the
01967          * page */
01968         pci_unmap_addr_set(&rx->info[idx], bus, rx->bus);
01969         rx->shadow[idx].addr_low = 
01970             htonl(MYRI10GE_LOWPART_TO_U32(rx->bus) + rx->page_offset);
01971         rx->shadow[idx].addr_high = 
01972             htonl(MYRI10GE_HIGHPART_TO_U32(rx->bus));
01973 
01974         /* start next packet on a cacheline boundary */
01975         rx->page_offset += SKB_DATA_ALIGN(bytes);
01976 
01977 #if MYRI10GE_ALLOC_SIZE > 4096
01978         /* don't cross a 4KB boundary */
01979         if ((rx->page_offset >> 12) !=
01980             ((rx->page_offset + bytes - 1) >> 12))
01981             rx->page_offset = (rx->page_offset + 4096) & ~4095;
01982 #endif
01983         rx->fill_cnt++;
01984 
01985         /* copy 8 descriptors to the firmware at a time */
01986         if ((idx & 7) == 7) {
01987             myri10ge_submit_8rx(&rx->lanai[idx - 7],
01988                         &rx->shadow[idx - 7]);
01989         }
01990     }
01991 }

02113       }
02114     }
02115 
02116 #endif

02118         myri10ge_set_last_rx(dev, jiffies);
02119         return 1;
02120     }
02121 #endif /* MYRI10GE_HAVE_GRO_FRAGS */
02122 
02123 #if MYRI10GE_LRO
02124     if (dev->features & NETIF_F_LRO) {

02183           /* PF_RING has already freed the memory */
02184           return 0;
02185         }
02186           }
02187         } else {
02188           if(debug) printk(KERN_INFO "[PF_RING] not present on %s\n", 
02189                    skb->dev->name);
02190         }
02191       }
02192     }
02193 
02194 #endif

02197 #if LINUX_VERSION_CODE <= KERNEL_VERSION(2,6,21)
02198     /* eth_type_trans sets skb->dev in 2.6.22 */
02199     skb->dev = dev;
02200 #endif
02201     if (unlikely(myri10ge_linearize_non_ip(skb))) {
02202         dev_kfree_skb_any(skb);
02203         ss->stats.rx_dropped++;
02204         return 0;
02205     }
02206 #endif /* LINUX_KERNEL_SPECIFIC */
02207 
02208         if (mgp->csum_flag) {
02209         if ((skb->protocol == htons(ETH_P_IP)) ||
02210             (skb->protocol == htons(ETH_P_IPV6))) {

02212             skb->ip_summed = CHECKSUM_COMPLETE;
02213         } else
02214             myri10ge_vlan_ip_csum(skb, csum);
02215         }
02216     myri10ge_report_queue(skb, ss - mgp->ss);
02217 #ifdef MYRI10GE_NAPI
02218     netif_receive_skb(skb);
02219 #else
02220     netif_rx(skb);
02221 #endif
02222     myri10ge_set_last_rx(dev, jiffies);
02223     return 1;

02226 static inline void
02227 myri10ge_tx_done(struct myri10ge_slice_state *ss, int mcp_index)
02228 {
02229     struct pci_dev *pdev = ss->mgp->pdev;
02230     struct myri10ge_tx_buf *tx = &ss->tx;
02231     struct netdev_queue *dev_queue;
02232     struct sk_buff *skb;
02233     int idx, len;
02234 
02235     while (tx->pkt_done != mcp_index) {
02236         idx = tx->done & tx->mask;
02237         skb = tx->info[idx].skb;
02238 
02239         /* Mark as free */
02240         tx->info[idx].skb = NULL;
02241         if (tx->info[idx].last) {
02242             tx->pkt_done++;
02243             tx->info[idx].last = 0;
02244         }
02245         tx->done++;
02246         len = pci_unmap_len(&tx->info[idx], len);
02247         pci_unmap_len_set(&tx->info[idx], len, 0);
02248         if (skb) {
02249             ss->stats.tx_bytes += skb->len;
02250             ss->stats.tx_packets++;
02251             dev_kfree_skb_irq(skb);
02252             if (len)
02253                 pci_unmap_single(pdev,
02254                          pci_unmap_addr(&tx->info[idx], bus),
02255                          len, PCI_DMA_TODEVICE);
02256         } else {
02257             if (len)
02258                 pci_unmap_page(pdev,
02259                            pci_unmap_addr(&tx->info[idx], bus),
02260                            len, PCI_DMA_TODEVICE);
02261         }
02262     }
02263 
02264     dev_queue = netdev_get_tx_queue(ss->dev, ss - ss->mgp->ss);
02265 #ifdef MYRI10GE_HAVE_MULTI_TX
02266     /* 
02267      * Make a minimal effort to prevent the NIC from polling an
02268      * idle tx queue.  If we can't get the lock we leave the queue
02269      * active. In this case, either a thread was about to start
02270      * using the queue anyway, or we lost a race and the NIC will
02271      * waste some of its resources polling an inactive queue for a
02272      * while.
02273      */
02274      
02275         if ((MYRI10GE_GET_NUM_TXQ(ss->mgp->dev) > 1) &&
02276         __netif_tx_trylock(dev_queue)) {
02277         if (tx->req == tx->done) {
02278             tx->queue_active = 0;
02279             put_be32(htonl(1), tx->send_stop);
02280             mb();
02281             myri10ge_mmiowb();
02282         }
02283         __netif_tx_unlock(dev_queue);
02284     }
02285 #endif /* MYRI10GE_HAVE_MULTI_TX */
02286 
02287     /* start the queue if we've stopped it */
02288     if (netif_tx_queue_stopped(dev_queue)
02289         && tx->req - tx->done < (tx->mask >> 1)) {
02290         tx->wake_queue++;
02291         netif_tx_wake_queue(dev_queue);
02292     }
02293 }

02296 static inline int
02297 myri10ge_clean_rx_done(struct myri10ge_slice_state *ss, int budget)
02298 #else
02299 static inline void
02300 myri10ge_clean_rx_done(struct myri10ge_slice_state *ss, int *limit)
02301 #endif
02302 {
02303     struct myri10ge_rx_done *rx_done = &ss->rx_done;
02304     struct myri10ge_priv *mgp = ss->mgp;
02305 #if MYRI10GE_LRO
02306     struct net_device *netdev = mgp->dev;
02307 #endif
02308     unsigned long rx_bytes = 0;
02309     unsigned long rx_packets = 0;
02310     unsigned long rx_ok;
02311 
02312     int idx = rx_done->idx;
02313     int cnt = rx_done->cnt;
02314 #ifdef MYRI10GE_HAVE_NEW_NAPI
02315     int work_done = 0;
02316 #endif
02317     u16 length;
02318     __wsum checksum;
02319 
02320     while (rx_done->entry[idx].length != 0 &&
02321 #ifdef MYRI10GE_HAVE_NEW_NAPI
02322         work_done < budget
02323 #else
02324         *limit != 0
02325 #endif
02326         ) {
02327         length = ntohs(rx_done->entry[idx].length);
02328         rx_done->entry[idx].length = 0;
02329         checksum = csum_unfold(rx_done->entry[idx].checksum);
02330         if (length <= mgp->small_bytes)
02331             rx_ok = myri10ge_rx_done(ss, &ss->rx_small,
02332                          mgp->small_bytes,
02333                          length, checksum);
02334         else
02335             rx_ok = myri10ge_rx_done(ss, &ss->rx_big,
02336                          mgp->big_bytes,
02337                          length, checksum);
02338         rx_packets += rx_ok;
02339         rx_bytes += rx_ok * (unsigned long)length;
02340         cnt++;
02341         idx = cnt & (mgp->max_intr_slots - 1);
02342 #ifdef MYRI10GE_HAVE_NEW_NAPI
02343         work_done++;
02344 #else
02345         /* limit potential for livelock by only handling a
02346          * limited number of frames. */
02347         (*limit)--;
02348 #endif
02349     }
02350     rx_done->idx = idx;
02351     rx_done->cnt = cnt;
02352     ss->stats.rx_packets += rx_packets;
02353     ss->stats.rx_bytes += rx_bytes;
02354 
02355 #if MYRI10GE_LRO
02356     if (netdev->features & NETIF_F_LRO)
02357         lro_flush_all(&rx_done->lro_mgr);
02358 #endif
02359 
02360 #if MYRI10GE_RX_SKBS
02361     /* Only call restock functions if not using skbs */
02362     if (!myri10ge_rx_skbs) {
02363 #endif /* MYRI10GE_RX_SKBS */
02364 
02365       /* restock receive rings if needed */
02366       if (ss->rx_small.fill_cnt - ss->rx_small.cnt < myri10ge_fill_thresh)
02367           myri10ge_alloc_rx_pages(mgp, &ss->rx_small,
02368                       mgp->small_bytes + MXGEFW_PAD, 0);
02369       if (ss->rx_big.fill_cnt - ss->rx_big.cnt < myri10ge_fill_thresh)
02370           myri10ge_alloc_rx_pages(mgp, &ss->rx_big, mgp->big_bytes, 0);
02371 
02372 #if MYRI10GE_RX_SKBS
02373     }
02374 #endif /* MYRI10GE_RX_SKBS */
02375 #ifdef MYRI10GE_HAVE_NEW_NAPI
02376     return work_done;
02377 #endif
02378 }

02514 static irqreturn_t
02515 #ifdef MYRI10GE_HAVE_IRQ_HANDLER_REGS
02516 myri10ge_intr(int irq __unused, void *arg, struct pt_regs *regs __unused)
02517 #else /* ~MYRI10GE_HAVE_IRQ_HANDLER_REGS */
02518 myri10ge_intr(int irq __unused, void *arg)
02519 #endif /* ~MYRI10GE_HAVE_IRQ_HANDLER_REGS */
02520 {
02521     struct myri10ge_slice_state *ss = arg;
02522     struct myri10ge_priv *mgp = ss->mgp;
02523     struct mcp_irq_data *stats = ss->fw_stats;
02524     struct myri10ge_tx_buf *tx = &ss->tx;
02525     u32 send_done_count;
02526     int i;
02527 
02528     /* an interrupt on a non-zero receive-only slice is implicitly
02529        valid  since MSI-X irqs are not shared */
02530     if ((MYRI10GE_GET_NUM_TXQ(mgp->dev) == 1) && (ss != mgp->ss)) {
02531         myri10ge_netif_rx_schedule(ss->dev, &ss->napi);
02532         myri10ge_inc_intrcnt(mgp->dev);
02533         return IRQ_HANDLED;
02534     }
02535         
02536     /* make sure it is our IRQ, and that the DMA has finished */
02537     if (unlikely(!stats->valid))
02538         return IRQ_NONE;
02539 
02540     myri10ge_inc_intrcnt(mgp->dev);
02541 
02542     /* low bit indicates receives are present, so schedule
02543        napi poll handler */
02544     if (stats->valid & 1)
02545         myri10ge_netif_rx_schedule(ss->dev, &ss->napi);
02546 
02547     if (!mgp->msi_enabled && !mgp->msix_enabled) {
02548         put_be32(0, mgp->irq_deassert);
02549         if (!myri10ge_deassert_wait)
02550             stats->valid = 0;
02551         mb();
02552     } else
02553         stats->valid = 0;
02554 
02555 
02556     /* Wait for IRQ line to go low, if using INTx */
02557     i = 0;
02558     while (1) {
02559         i++;
02560         /* check for transmit completes and receives */
02561         send_done_count = ntohl(stats->send_done_count);
02562         if (send_done_count != tx->pkt_done)
02563             myri10ge_tx_done(ss, (int)send_done_count);
02564         if (unlikely(i > myri10ge_max_irq_loops)) {
02565             printk(KERN_WARNING "myri10ge: %s: irq stuck?\n",
02566                    mgp->dev->name);
02567             stats->valid = 0;
02568             schedule_work(&mgp->watchdog_work);
02569         }
02570         if (likely(stats->valid == 0))
02571             break;
02572         cpu_relax();
02573         barrier();
02574     }
02575 
02576         /* Only slice 0 updates stats */
02577         if (ss == mgp->ss)
02578                 myri10ge_check_statblock(mgp);
02579 
02580 #if MYRI10GE_VPUMP
02581     if (mgp->vpump != (vpump_dev_t *)NULL) {
02582       wake_up(&mgp->vpump->wait_evt);
02583     }
02584 #endif
02585     put_be32(htonl(3), ss->irq_claim + 1);
02586     return IRQ_HANDLED;
02587 }
02588 #else
02589 static irqreturn_t
02590 #ifdef MYRI10GE_HAVE_IRQ_HANDLER_REGS
02591 myri10ge_intr(int irq __unused, void *arg, struct pt_regs *regs __unused)
02592 #else /* ~MYRI10GE_HAVE_IRQ_HANDLER_REGS */
02593 myri10ge_intr(int irq __unused, void *arg)
02594 #endif /* ~MYRI10GE_HAVE_IRQ_HANDLER_REGS */
02595 {
02596     struct myri10ge_slice_state *ss = arg;
02597     struct myri10ge_priv *mgp = ss->mgp;
02598     struct mcp_irq_data *stats = ss->fw_stats;
02599     struct myri10ge_tx_buf *tx = &ss->tx;
02600     struct myri10ge_rx_done *rx_done = &ss->rx_done;
02601     u32 send_done_count;
02602     int i, limit;
02603     u8 valid;
02604 
02605     /* an interrupt on a non-zero receive only slice is implicitly
02606        valid since MSI-X irqs are not shared */
02607     if ((MYRI10GE_GET_NUM_TXQ(mgp->dev) == 1) && (ss != mgp->ss)) {
02608 #ifdef MYRI10GE_HAVE_DCA
02609         if (ss->mgp->dca_enabled)
02610             myri10ge_update_dca(ss);
02611 #endif
02612 
02613         limit = mgp->max_intr_slots;
02614 #ifdef MYRI10GE_HAVE_NEW_NAPI
02615         myri10ge_clean_rx_done(ss, limit);
02616 #else
02617         myri10ge_clean_rx_done(ss, &limit);
02618 #endif
02619         put_be32(htonl(3), ss->irq_claim);
02620         myri10ge_inc_intrcnt(mgp->dev);
02621         return IRQ_HANDLED;
02622     }
02623         
02624     /* make sure it is our IRQ, and that the DMA has finished */
02625     if (unlikely(!stats->valid))
02626         return IRQ_NONE;
02627 
02628 #ifdef MYRI10GE_HAVE_DCA
02629     if (ss->mgp->dca_enabled)
02630         myri10ge_update_dca(ss);
02631 #endif
02632 
02633     myri10ge_inc_intrcnt(mgp->dev);
02634     valid = stats->valid;
02635     if (!mgp->msi_enabled && !mgp->msix_enabled) {
02636         put_be32(0, mgp->irq_deassert);
02637         if (!myri10ge_deassert_wait)
02638             stats->valid = 0;
02639         mb();
02640     } else
02641         stats->valid = 0;
02642 
02643 
02644     /* Wait for IRQ line to go low, if using INTx */
02645     i = 0;
02646     while (1) {
02647         i++;
02648         /* check for transmit completes and receives */
02649         send_done_count = ntohl(stats->send_done_count);
02650         while ((send_done_count != tx->pkt_done) ||
02651                (rx_done->entry[rx_done->idx].length != 0)) {
02652             myri10ge_tx_done(ss, (int)send_done_count);
02653             limit = 32;
02654 #ifdef MYRI10GE_HAVE_NEW_NAPI
02655             myri10ge_clean_rx_done(ss, limit);
02656 #else
02657             myri10ge_clean_rx_done(ss, &limit);
02658 #endif
02659         }
02660         if (unlikely(i > myri10ge_max_irq_loops)) {
02661             printk(KERN_WARNING "myri10ge: %s: irq stuck?\n",
02662                    mgp->dev->name);
02663             stats->valid = 0;
02664             schedule_work(&mgp->watchdog_work);
02665         }
02666         if (likely(stats->valid == 0))
02667             break;
02668         cpu_relax();
02669         barrier();
02670     }
02671 
02672     myri10ge_check_statblock(mgp);
02673 
02674     /* pass back rx token if we own it */
02675     if (valid & 0x1) {
02676         put_be32(htonl(3), ss->irq_claim);
02677     }
02678 
02679 #if MYRI10GE_VPUMP
02680     if (mgp->vpump != (vpump_dev_t *)NULL) {
02681       wake_up(&mgp->vpump->wait_evt);
02682     }
02683 #endif
02684 
02685     put_be32(htonl(3), ss->irq_claim + 1);
02686 
02687     return IRQ_HANDLED;
02688 }

03123 static int
03124 myri10ge_allocate_rings(struct myri10ge_slice_state *ss)
03125 {
03126     struct myri10ge_priv *mgp = ss->mgp;
03127     struct myri10ge_cmd cmd;
03128     struct net_device *dev = mgp->dev;
03129     int tx_ring_size, rx_ring_size;
03130     int tx_ring_entries, rx_ring_entries;
03131     int i, slice, status;
03132     size_t bytes;
03133 
03134     /* get ring sizes */
03135     slice = ss - mgp->ss;
03136     cmd.data0 = slice;
03137     status = myri10ge_send_cmd(mgp, MXGEFW_CMD_GET_SEND_RING_SIZE, &cmd, 0);
03138     tx_ring_size = cmd.data0;
03139     cmd.data0 = slice;
03140     status |= myri10ge_send_cmd(mgp, MXGEFW_CMD_GET_RX_RING_SIZE, &cmd, 0);
03141     if (status != 0)
03142         return status;
03143     rx_ring_size = cmd.data0;
03144 
03145     tx_ring_entries = tx_ring_size / sizeof (struct mcp_kreq_ether_send);
03146     rx_ring_entries = rx_ring_size / sizeof (struct mcp_dma_addr);
03147     ss->tx.mask = tx_ring_entries - 1;
03148     ss->rx_small.mask = ss->rx_big.mask = rx_ring_entries - 1;
03149 
03150     status = -ENOMEM;
03151 
03152     /* allocate the host shadow rings */
03153 
03154     bytes = 8 + (MYRI10GE_MAX_SEND_DESC_TSO + 4)
03155         * sizeof (*ss->tx.req_list);
03156     ss->tx.req_bytes = myri10ge_kzalloc(bytes, GFP_KERNEL);
03157     if (ss->tx.req_bytes == NULL)
03158         goto abort_with_nothing;
03159 
03160     /* ensure req_list entries are aligned to 8 bytes */
03161     ss->tx.req_list = (struct mcp_kreq_ether_send *)
03162         ALIGN((unsigned long) ss->tx.req_bytes, 8);
03163         ss->tx.queue_active = 0;
03164         
03165     bytes = rx_ring_entries * sizeof (*ss->rx_small.shadow);
03166     ss->rx_small.shadow = myri10ge_kzalloc(bytes, GFP_KERNEL);
03167     if (ss->rx_small.shadow == NULL)
03168         goto abort_with_tx_req_bytes;
03169 
03170     bytes = rx_ring_entries * sizeof (*ss->rx_big.shadow);
03171     ss->rx_big.shadow = myri10ge_kzalloc(bytes, GFP_KERNEL);
03172     if (ss->rx_big.shadow == NULL)
03173         goto abort_with_rx_small_shadow;
03174 
03175     /* allocate the host info rings */
03176 
03177     bytes = tx_ring_entries * sizeof (*ss->tx.info);
03178     ss->tx.info = myri10ge_kzalloc(bytes, GFP_KERNEL);
03179     if (ss->tx.info == NULL)
03180         goto abort_with_rx_big_shadow;
03181 
03182     bytes = rx_ring_entries * sizeof (*ss->rx_small.info);
03183     ss->rx_small.info = myri10ge_kzalloc(bytes, GFP_KERNEL);
03184     if (ss->rx_small.info == NULL)
03185         goto abort_with_tx_info;
03186 
03187     bytes = rx_ring_entries * sizeof (*ss->rx_big.info);
03188     ss->rx_big.info = myri10ge_kzalloc(bytes, GFP_KERNEL);
03189     if (ss->rx_big.info == NULL)
03190         goto abort_with_rx_small_info;
03191 
03192     /* Fill the receive rings */
03193     ss->rx_big.cnt = 0;
03194     ss->rx_small.cnt = 0;
03195     ss->rx_big.fill_cnt = 0;
03196     ss->rx_small.fill_cnt = 0;
03197     ss->rx_small.page_offset = MYRI10GE_ALLOC_SIZE;
03198     ss->rx_big.page_offset = MYRI10GE_ALLOC_SIZE;
03199     ss->rx_small.watchdog_needed = 0;
03200     ss->rx_big.watchdog_needed = 0;
03201 #if MYRI10GE_RX_SKBS
03202     if (myri10ge_rx_skbs) {
03203         unsigned int max;
03204 
03205         ss->rx_big.fill_offset = 0;
03206         ss->rx_small.fill_offset = 0;
03207 
03208         if (mgp->skb_alloc_limit != 0)
03209             max = mgp->skb_alloc_limit - 1;
03210         else
03211             max = ss->rx_small.mask;
03212 
03213         for (i = 0; i <= max; i++) {
03214             status = myri10ge_getbuf(&ss->rx_small, mgp,
03215                          mgp->small_bytes, i);
03216             if (status) {
03217                 printk(KERN_ERR "myri10ge: %s:slice-%d: alloced only %d/%d small bufs\n",
03218                        dev->name, slice, i, max);
03219                 goto abort_with_rx_small_ring;
03220             }
03221         }
03222         if (max != ss->rx_small.mask)
03223             ss->rx_small.fill_offset = max + 1;
03224 
03225         if (mgp->skb_alloc_limit != 0)
03226             max = mgp->skb_alloc_limit - 1;
03227         else
03228             max = ss->rx_big.mask;
03229         for (i = 0; i <= max; i++) {
03230             status = myri10ge_getbuf(&ss->rx_big, mgp, mgp->big_bytes, i);
03231             if (status) {
03232                 printk(KERN_ERR "myri10ge: %s:slice-%d: alloced only %d/%d big bufs\n",
03233                        dev->name, slice, i, max);
03234                 goto abort_with_rx_big_ring;
03235             }
03236         }
03237         if (max != ss->rx_big.mask)
03238             ss->rx_big.fill_offset = max + 1;
03239     } else {
03240 #endif /* MYRI10GE_RX_SKBS */
03241         myri10ge_alloc_rx_pages(mgp, &ss->rx_small, 
03242                     mgp->small_bytes + MXGEFW_PAD, 0);
03243 
03244         if (ss->rx_small.fill_cnt < ss->rx_small.mask + 1) {
03245             printk(KERN_ERR
03246                    "myri10ge: %s:slice-%d: alloced only %d small bufs\n",
03247                    dev->name, slice, ss->rx_small.fill_cnt);
03248             goto abort_with_rx_small_ring;
03249         }
03250 
03251         myri10ge_alloc_rx_pages(mgp, &ss->rx_big, mgp->big_bytes, 0);
03252         if (ss->rx_big.fill_cnt < ss->rx_big.mask + 1) {
03253             printk(KERN_ERR
03254                    "myri10ge: %s:slice-%d: alloced only %d big bufs\n",
03255                    dev->name, slice, ss->rx_big.fill_cnt);
03256             goto abort_with_rx_big_ring;
03257         }
03258 #if MYRI10GE_RX_SKBS
03259     }
03260 #endif
03261 
03262     return 0;
03263 
03264 abort_with_rx_big_ring:
03265 #if MYRI10GE_RX_SKBS
03266     if (myri10ge_rx_skbs) {
03267         for (i = 0; i <= ss->rx_big.mask; i++) {
03268             if (ss->rx_big.info[i].rx__skb != NULL)
03269                 myri10ge_dev_kfree_skb_any(ss->rx_big.info[i].rx__skb);
03270             if (pci_unmap_len(&ss->rx_big.info[i], len))
03271                 pci_unmap_single(mgp->pdev,
03272                          pci_unmap_addr(&ss->rx_big.info[i], bus),
03273                          pci_unmap_len(&ss->rx_big.info[i], len),
03274                          PCI_DMA_FROMDEVICE);
03275         }
03276     } else {
03277 #endif /* MYRI10GE_RX_SKBS */
03278         for (i = ss->rx_big.cnt; i< ss->rx_big.fill_cnt; i++) {
03279             int idx = i & ss->rx_big.mask;
03280             myri10ge_unmap_rx_page(mgp->pdev, &ss->rx_big.info[idx],
03281                            mgp->big_bytes);
03282             put_page(ss->rx_big.info[idx].rx__page);
03283         }
03284 #if MYRI10GE_RX_SKBS
03285     }
03286 #endif

03614 static int
03615 myri10ge_init_toeplitz(struct myri10ge_priv *mgp)
03616 {
03617     struct myri10ge_cmd cmd;
03618     int i, b, s, t, j;
03619     int status;
03620     u32 k[8];
03621     u32 tmp;
03622     u8 *key;
03623 
03624     status = myri10ge_send_cmd(mgp, MXGEFW_CMD_GET_RSS_KEY_OFFSET,
03625                    &cmd, 0);
03626     if (status != 0) {
03627         printk(KERN_ERR

03629                mgp->dev->name);
03630         return -EIO;
03631     }
03632     memcpy_fromio(mgp->rss_key, mgp->sram + cmd.data0,
03633               sizeof(mgp->rss_key));
03634

03636                        GFP_KERNEL);
03637     if (mgp->toeplitz_hash_table == NULL)
03638         return -ENOMEM;
03639     key = (u8 *)mgp->rss_key;
03640     t = 0;
03641     for (b = 0; b < 12; b++) {
03642         for (s = 0; s < 8; s++) {
03643             /* Bits: b*8+s, ..., b*8+s+31 */
03644             k[s] = 0;
03645             for (j = 0; j < 32; j++) {
03646                 int bit = b*8+s+j;
03647                 bit = 0x1 & (key[bit / 8] >> (7 -(bit & 0x7)));
03648                 k[s] |= bit << (31 - j);
03649             }
03650         }
03651 
03652         for (i = 0; i <= 0xff; i++) {
03653             tmp = 0;
03654             if (i & (1 << 7)) { tmp ^= k[0]; }
03655             if (i & (1 << 6)) { tmp ^= k[1]; }
03656             if (i & (1 << 5)) { tmp ^= k[2]; }
03657             if (i & (1 << 4)) { tmp ^= k[3]; }
03658             if (i & (1 << 3)) { tmp ^= k[4]; }
03659             if (i & (1 << 2)) { tmp ^= k[5]; }
03660             if (i & (1 << 1)) { tmp ^= k[6]; }
03661             if (i & (1 << 0)) { tmp ^= k[7]; }
03662             mgp->toeplitz_hash_table[t++] = tmp;
03663         }
03664     }
03665     return 0;
03666 }

03668 static inline u16
03669 myri10ge_toeplitz_select_queue(struct net_device *dev, struct iphdr *ip)
03670 {
03671     struct myri10ge_priv *mgp = netdev_priv(dev);
03672     struct tcphdr *hdr;
03673     u32 saddr, daddr;
03674     u32 hash;
03675     u32 *table = mgp->toeplitz_hash_table;
03676     u16 src, dst;
03677 
03678     /*
03679      * Note hashing order is reversed from how it is done
03680      * in the NIC, so as to generate the same hash value
03681      * for the connection to try to keep connections CPU local
03682      */
03683 
03684     /* hash on IPv4 src/dst address */
03685     saddr = ntohl(ip->saddr);
03686     daddr = ntohl(ip->daddr);
03687     hash = table[(256 * 0) + ((daddr >> 24) & 0xff)];
03688     hash ^= table[(256 * 1) + ((daddr >> 16) & 0xff)];
03689     hash ^= table[(256 * 2) + ((daddr >> 8) & 0xff)];
03690     hash ^= table[(256 * 3) + ((daddr) & 0xff)];
03691     hash ^= table[(256 * 4) + ((saddr >> 24) & 0xff)];
03692     hash ^= table[(256 * 5) + ((saddr >> 16) & 0xff)];
03693     hash ^= table[(256 * 6) + ((saddr >> 8) & 0xff)];
03694     hash ^= table[(256 * 7) + ((saddr) & 0xff)];
03695     /* hash on TCP port, if required */
03696     if ((myri10ge_rss_hash & MXGEFW_RSS_HASH_TYPE_TCP_IPV4) &&
03697         ip->protocol == IPPROTO_TCP) {
03698         hdr = (struct tcphdr *)(((u8 *)ip) +  (ip->ihl << 2));
03699         src = ntohs(hdr->source);
03700         dst = ntohs(hdr->dest);
03701 
03702         hash ^= table[(256 * 8) + ((dst >> 8) & 0xff)];
03703         hash ^= table[(256 * 9) + ((dst) & 0xff)];
03704         hash ^= table[(256 * 10) + ((src >> 8) & 0xff)];
03705         hash ^= table[(256 * 11) + ((src) & 0xff)];
03706     }
03707     return (u16)(hash & (dev->real_num_tx_queues - 1));
03708 }

03712 {

03731 static u16
03732 myri10ge_select_queue(struct net_device *dev, struct sk_buff *skb)
03733 {
03734     struct iphdr *ip;
03735     struct vlan_hdr *vh;
03736 
03737     if (myri10ge_tx_hash == MYRI10GE_TX_HASH_SKB)
03738         return (u16)(skb_get_queue_mapping(skb) &
03739                  (dev->real_num_tx_queues - 1));
03740 
03741     if (skb->protocol == __constant_htons(ETH_P_IP)) {
03742         ip = ip_hdr(skb);
03743     } else if (skb->protocol == __constant_htons(ETH_P_8021Q)) {
03744         vh = (struct vlan_hdr *) skb->data;
03745         if ((vh->h_vlan_encapsulated_proto != 
03746              __constant_htons(ETH_P_IP)))
03747             return 0;
03748         ip = (struct iphdr *)skb->data + sizeof (*vh);
03749     } else {
03750         return 0;
03751     }
03752 
03753     switch (myri10ge_rss_hash) {
03754     case MXGEFW_RSS_HASH_TYPE_IPV4:
03755         /* fallthru */
03756     case MXGEFW_RSS_HASH_TYPE_TCP_IPV4:
03757         /* fallthru */
03758     case (MXGEFW_RSS_HASH_TYPE_IPV4|MXGEFW_RSS_HASH_TYPE_TCP_IPV4):
03759         return (myri10ge_toeplitz_select_queue(dev, ip));
03760         break;
03761     case MXGEFW_RSS_HASH_TYPE_SRC_PORT:
03762         /* fallthru */
03763     case MXGEFW_RSS_HASH_TYPE_SRC_DST_PORT:
03764         return (myri10ge_simple_select_queue(dev, ip));
03765     default:
03766         return (0);
03767     }
03768 }

03885             goto abort_with_nothing;
03886         }

03890         for (i = 0; i < mgp->num_slices; i++)
03891             __raw_writeb(i, &itable[i]);
03892

03907 
03908 #endif /* MYRI10GE_HAVE_TOEPLITZ_MULTI_TX */

03910         cmd.data1 = myri10ge_rss_hash;

03912                        &cmd, 0);

03915                    "myri10ge: %s: failed to enable slices\n",

03925 #endif
03926     }

03929     if (status != 0)
03930         goto abort_with_nothing;

03932     /* decide what small buffer size to use.  For good TCP rx
03933      * performance, it is important to not receive 1514 byte
03934      * frames into jumbo buffers, as it confuses the socket buffer
03935      * accounting code, leading to drops and erratic performance.
03936      */

03956 #if MYRI10GE_RX_SKBS

03959             big_pow2++;
03960         mgp->big_bytes = dev->mtu + ETH_HLEN + VLAN_HLEN + MXGEFW_PAD;
03961     } else {
03962 #endif

03970         }
03971 #if MYRI10GE_RX_SKBS
03972     }

03976 #endif

04118         myri10ge_napi_disable(&mgp->ss[slice]);
04119     }
04120 #endif /* MYRI10GE_NAPI */
04121     for (i = 0; i < mgp->num_slices; i++)

04123 
04124     myri10ge_free_irq(mgp);
04125

04198 /* copy an array of struct mcp_kreq_ether_send's to the mcp.  Copy
04199  * backwards one at a time and handle ring wraps */
04200 
04201 static inline void
04202 myri10ge_submit_req_backwards(struct myri10ge_tx_buf *tx,
04203                   struct mcp_kreq_ether_send *src, int cnt)
04204 {
04205     int idx, starting_slot;
04206     starting_slot = tx->req;
04207     while (cnt > 1) {
04208         cnt--;
04209         idx = (starting_slot + cnt) & tx->mask;
04210         myri10ge_pio_copy(&tx->lanai[idx],
04211                   &src[cnt], sizeof(*src));
04212         mb();
04213     }
04214 }
04215 
04216 /*
04217  * copy an array of struct mcp_kreq_ether_send's to the mcp.  Copy
04218  * at most 32 bytes at a time, so as to avoid involving the software
04219  * pio handler in the nic.   We re-write the first segment's flags
04220  * to mark them valid only after writing the entire chain.
04221  */
04222 
04223 static inline void
04224 myri10ge_submit_req(struct myri10ge_tx_buf *tx, struct mcp_kreq_ether_send *src,
04225             int cnt)
04226 {
04227     int idx, i;
04228     struct mcp_kreq_ether_send __iomem *dstp, *dst;
04229     struct mcp_kreq_ether_send *srcp;
04230     u8 last_flags;
04231 
04232     idx = tx->req & tx->mask;
04233 
04234     last_flags = src->flags;
04235     src->flags = 0;
04236     mb();
04237     dst = dstp = &tx->lanai[idx];
04238     srcp = src;
04239 
04240     if ((idx + cnt) < tx->mask) {
04241         for (i = 0; i < (cnt - 1); i += 2) {
04242             myri10ge_pio_copy(dstp, srcp, 2 * sizeof(*src));
04243             mb();   /* force write every 32 bytes */
04244             srcp += 2;
04245             dstp += 2;
04246         }
04247     } else {
04248         /* submit all but the first request, and ensure
04249            that it is submitted below */
04250         myri10ge_submit_req_backwards(tx, src, cnt);
04251         i = 0;
04252     }
04253     if (i < cnt) {
04254         /* submit the first request */
04255         myri10ge_pio_copy(dstp, srcp, sizeof(*src));
04256         mb(); /* barrier before setting valid flag */
04257     }
04258 
04259     /* re-write the last 32-bits with the valid flags */
04260     src->flags = last_flags;
04261     put_be32(*((__be32 *) src + 3), (__be32 __iomem *) dst + 3);
04262     tx->req += cnt;
04263     mb();
04264 }

04352 /*
04353  * Transmit a packet.  We need to split the packet so that a single
04354  * segment does not cross myri10ge->tx_boundary, so this makes segment
04355  * counting tricky.  So rather than try to count segments up front, we
04356  * just give up if there are too few segments to hold a reasonably
04357  * fragmented packet currently available.  If we run
04358  * out of segments while preparing a packet for DMA, we just linearize
04359  * it and try again.
04360  */
04361 
04362 static netdev_tx_t
04363 myri10ge_xmit(struct sk_buff *skb, struct net_device *dev)
04364 {
04365     struct myri10ge_priv *mgp = NETDEV_TO_MGP(dev);
04366     struct myri10ge_slice_state *ss;
04367     struct mcp_kreq_ether_send *req;
04368     struct myri10ge_tx_buf *tx;
04369     struct skb_frag_struct *frag;
04370     struct netdev_queue *netdev_queue;
04371     dma_addr_t bus;
04372     u32 low;
04373     __be32 high_swapped;
04374     unsigned int len;
04375     int idx, last_idx, avail, frag_cnt, frag_idx, count, mss, max_segments;
04376     u16 pseudo_hdr_offset, cksum_offset, queue;
04377     int cum_len, seglen, boundary, rdma_count;
04378     u8 flags, odd_flag;
04379 
04380     queue = skb_get_queue_mapping(skb);
04381 #ifdef __VMKERNEL_MODULE__
04382     queue = (mgp->num_slices - 1) & queue;
04383 #endif
04384     ss = &mgp->ss[queue];        
04385     netdev_queue = netdev_get_tx_queue(mgp->dev, queue);
04386     tx = &ss->tx;
04387 
04388 #ifdef MYRI10GE_HAVE_VLAN_OFFLOAD
04389     if (mgp->vlan_group != NULL && vlan_tx_tag_present(skb)) {
04390         skb = myri10ge_tx_vlan(skb);
04391         if (unlikely(skb == NULL)) {
04392             ss->stats.tx_dropped += 1;
04393             return 0;
04394         }
04395     }
04396 #endif
04397 again:
04398     req = tx->req_list;
04399     avail = tx->mask - 1 - (tx->req - tx->done);
04400 
04401     mss = 0;
04402     max_segments = MXGEFW_MAX_SEND_DESC;
04403 
04404     if (myri10ge_skb_is_gso(skb)) {
04405 #if MYRI10GE_HAVE_TSO
04406         mss = skb_shinfo(skb)->MYRI10GE_GSO_SIZE;
04407 #endif
04408         max_segments = MYRI10GE_MAX_SEND_DESC_TSO;
04409     }
04410 
04411     if ((unlikely(avail < max_segments))) {
04412         /* we are out of transmit resources */
04413         tx->stop_queue++;
04414         netif_tx_stop_queue(netdev_queue);
04415         return NETDEV_TX_BUSY;
04416     }
04417 
04418     /* Setup checksum offloading, if needed */
04419     cksum_offset = 0;
04420     pseudo_hdr_offset = 0;
04421     odd_flag = 0;
04422     flags = (MXGEFW_FLAGS_NO_TSO |
04423          MXGEFW_FLAGS_FIRST);
04424     if (likely(skb->ip_summed == CHECKSUM_PARTIAL)) {
04425         cksum_offset = myri10ge_skb_transport_offset(skb);
04426         pseudo_hdr_offset = cksum_offset + skb->MYRI10GE_SKB_CSUM_OFFSET;
04427         /* If the headers are excessively large, then we must
04428          * fall back to a software checksum */
04429         if (unlikely(!mss && (cksum_offset > 255 ||
04430                       pseudo_hdr_offset > 127))) {
04431 #ifdef LINUX_KERNEL_SPECIFIC
04432             if (skb_checksum_help(skb))
04433                 goto drop;
04434 #else  /* !LINUX_KERNEL_SPECIFIC */
04435             myri10ge_csum_fixup(skb, cksum_offset, pseudo_hdr_offset);
04436 #endif /* LINUX_KERNEL_SPECIFIC */
04437             cksum_offset = 0;
04438             pseudo_hdr_offset = 0;
04439         } else {
04440             odd_flag = MXGEFW_FLAGS_ALIGN_ODD;
04441             flags |= MXGEFW_FLAGS_CKSUM;
04442         }
04443     }
04444 
04445     cum_len = 0;
04446 
04447     if (mss) { /* TSO */
04448 #ifndef LINUX_KERNEL_SPECIFIC
04449         /* 
04450          * The firmware expects to do checksum offloading on
04451          * TSO segments.  If we see a frame with the checksum
04452          * already calculated, we need to replace the complete
04453          * checksum with the partial checksum that the
04454          * firmware expects.  Further, ESX Windows guests pass down
04455          * a th->check which sets the total length of the frame
04456          * to zero, so we must recalculate it for ESX.
04457          */
04458 #ifndef __VMKERNEL_MODULE__
04459         if (unlikely(skb->ip_summed == CHECKSUM_NONE))
04460 #endif
04461             myri10ge_tso_csum_fixup(skb, &pseudo_hdr_offset,
04462                         &cksum_offset);
04463 #endif
04464         /* this removes any CKSUM flag from before */
04465         flags = (MXGEFW_FLAGS_TSO_HDR |
04466              MXGEFW_FLAGS_FIRST);
04467 
04468         /* negative cum_len signifies to the
04469          * send loop that we are still in the
04470          * header portion of the TSO packet.
04471          * TSO header can be at most 1KB long */
04472         cum_len = -(myri10ge_skb_transport_offset(skb) + myri10ge_tcp_hdrlen(skb));
04473 
04474         /* for IPv6 TSO, the checksum offset stores the
04475          * TCP header length, to save the firmware from
04476          * the need to parse the headers */
04477 #ifdef NETIF_F_TSO6
04478         if (myri10ge_skb_is_gso_v6(skb)) {
04479             cksum_offset = myri10ge_tcp_hdrlen(skb);
04480             /* Can only handle headers <= max_tso6 long */
04481             if (unlikely (-cum_len > mgp->max_tso6))
04482                 return myri10ge_sw_tso(skb, dev);
04483         }
04484 #endif
04485         /* for TSO, pseudo_hdr_offset holds mss.
04486          * The firmware figures out where to put
04487          * the checksum by parsing the header. */
04488         pseudo_hdr_offset = mss;
04489     } else
04490     /* Mark small packets, and pad out tiny packets */
04491     if (skb->len <= MXGEFW_SEND_SMALL_SIZE) {
04492         flags |= MXGEFW_FLAGS_SMALL;
04493 
04494         /* pad frames to at least ETH_ZLEN bytes */
04495         if (unlikely(skb->len < ETH_ZLEN)) {
04496             if (myri10ge_skb_padto(skb, ETH_ZLEN)) {
04497                 /* The packet is gone, so we must
04498                    return 0 */
04499                 ss->stats.tx_dropped += 1;
04500                 return NETDEV_TX_OK;
04501             }
04502             /* adjust the len to account for the zero pad
04503                so that the nic can know how long it is */
04504             skb->len = ETH_ZLEN;
04505         }
04506     }
04507 
04508     /* map the skb for DMA */
04509     len = skb->len - skb->data_len;
04510     idx = tx->req & tx->mask;
04511     tx->info[idx].skb = skb;
04512     bus = myri10ge_pci_map_skb_data(mgp->pdev, skb, len, PCI_DMA_TODEVICE);
04513     pci_unmap_addr_set(&tx->info[idx], bus, bus);
04514     pci_unmap_len_set(&tx->info[idx], len, len);
04515 
04516     frag_cnt = skb_shinfo(skb)->nr_frags;
04517     frag_idx = 0;
04518     count = 0;
04519     rdma_count = 0;
04520 
04521     /* "rdma_count" is the number of RDMAs belonging to the
04522      * current packet BEFORE the current send request. For
04523      * non-TSO packets, this is equal to "count".
04524      * For TSO packets, rdma_count needs to be reset
04525      * to 0 after a segment cut.
04526      *
04527      * The rdma_count field of the send request is
04528      * the number of RDMAs of the packet starting at
04529      * that request. For TSO send requests with one ore more cuts
04530      * in the middle, this is the number of RDMAs starting
04531      * after the last cut in the request. All previous
04532      * segments before the last cut implicitly have 1 RDMA.
04533      *
04534      * Since the number of RDMAs is not known beforehand,
04535      * it must be filled-in retroactively - after each
04536      * segmentation cut or at the end of the entire packet.
04537      */
04538 
04539     while (1) {
04540         /* Break the SKB or Fragment up into pieces which
04541            do not cross mgp->tx_boundary */
04542         low = MYRI10GE_LOWPART_TO_U32(bus);
04543         high_swapped = htonl(MYRI10GE_HIGHPART_TO_U32(bus));
04544         while (len) {
04545             u8 flags_next;
04546             int cum_len_next;
04547 
04548             if (unlikely(count == max_segments))
04549                 goto abort_linearize;
04550 
04551             boundary = (low + mgp->tx_boundary) & ~(mgp->tx_boundary - 1);
04552             seglen = boundary - low;
04553             if (seglen > len)
04554                 seglen = len;
04555             flags_next = flags & ~MXGEFW_FLAGS_FIRST;
04556             cum_len_next = cum_len + seglen;
04557             if (mss) { /* TSO */
04558                 (req-rdma_count)->rdma_count = rdma_count + 1;
04559 
04560                 if (likely(cum_len >= 0)) { /* payload */
04561                     int next_is_first, chop;
04562 
04563                     chop = (cum_len_next>mss);
04564                     cum_len_next = cum_len_next % mss;
04565                     next_is_first = (cum_len_next == 0);
04566                     flags |= chop *
04567                         MXGEFW_FLAGS_TSO_CHOP;
04568                     flags_next |= next_is_first *
04569                         MXGEFW_FLAGS_FIRST;
04570                     rdma_count |= -(chop | next_is_first);
04571                     rdma_count += chop & !next_is_first;
04572                 } else if (likely(cum_len_next >= 0)) { /* header ends */
04573                     int small;
04574 
04575                     rdma_count = -1;
04576                     cum_len_next = 0;
04577                     seglen = -cum_len;
04578                     small = (mss <= MXGEFW_SEND_SMALL_SIZE);
04579                     flags_next = MXGEFW_FLAGS_TSO_PLD |
04580                         MXGEFW_FLAGS_FIRST |
04581                         (small * MXGEFW_FLAGS_SMALL);
04582                 }
04583             }
04584             req->addr_high = high_swapped;
04585             req->addr_low = htonl(low);
04586             req->pseudo_hdr_offset = htons(pseudo_hdr_offset);
04587             req->pad = 0;   /* complete solid 16-byte block; does this matter? */
04588             req->rdma_count = 1;
04589             req->length = htons(seglen);
04590             req->cksum_offset = cksum_offset;
04591             req->flags = flags | ((cum_len & 1) * odd_flag);
04592 
04593             low += seglen;
04594             len -= seglen;
04595             cum_len = cum_len_next;
04596             flags = flags_next;
04597             req++;
04598             count++;
04599             rdma_count++;
04600             if (cksum_offset != 0 &&
04601                 !(mss && myri10ge_skb_is_gso_v6(skb))) {
04602                 if (unlikely(cksum_offset > seglen))
04603                     cksum_offset -= seglen;
04604                 else
04605                     cksum_offset = 0;
04606             }
04607         }
04608         if (frag_idx == frag_cnt)
04609             break;
04610 
04611         /* map next fragment for DMA */
04612         idx = (count + tx->req) & tx->mask;
04613         frag = &skb_shinfo(skb)->frags[frag_idx];
04614         frag_idx++;
04615         len = frag->size;
04616         bus = pci_map_page(mgp->pdev, frag->page, frag->page_offset,
04617                    len, PCI_DMA_TODEVICE);
04618         pci_unmap_addr_set(&tx->info[idx], bus, bus);
04619         pci_unmap_len_set(&tx->info[idx], len, len);
04620     }
04621 
04622     (req-rdma_count)->rdma_count = rdma_count;
04623     if (mss)
04624         do {
04625             req--;
04626             req->flags |= MXGEFW_FLAGS_TSO_LAST;
04627         } while (!(req->flags & (MXGEFW_FLAGS_TSO_CHOP |
04628                      MXGEFW_FLAGS_FIRST)));
04629     idx = ((count - 1) + tx->req) & tx->mask;
04630     tx->info[idx].last = 1;
04631     myri10ge_submit_req(tx, tx->req_list, count);
04632     /* if using multiple tx queues, make sure NIC polls the
04633      * current slice */
04634     if ((MYRI10GE_GET_NUM_TXQ(mgp->dev) > 1) && tx->queue_active == 0) {
04635         tx->queue_active = 1;
04636         put_be32(htonl(1), tx->send_go);
04637         mb();
04638         myri10ge_mmiowb();
04639     }
04640     tx->pkt_start++;
04641     if ((avail - count) < MXGEFW_MAX_SEND_DESC) {
04642         tx->stop_queue++;
04643         netif_tx_stop_queue(netdev_queue);
04644     }
04645     myri10ge_set_trans_start(dev, jiffies);
04646     return NETDEV_TX_OK;
04647 
04648 
04649 abort_linearize:
04650     /* Free any DMA resources we've alloced and clear out the skb
04651      * slot so as to not trip up assertions, and to avoid a
04652      * double-free if linearizing fails */
04653 
04654     last_idx = (idx + 1) & tx->mask;
04655     idx = tx->req & tx->mask;
04656     tx->info[idx].skb = NULL;
04657     do {
04658         len = pci_unmap_len(&tx->info[idx], len);
04659         if (len) {
04660             if (tx->info[idx].skb != NULL)
04661                 pci_unmap_single(mgp->pdev,
04662                          pci_unmap_addr(&tx->info[idx], bus),
04663                          len, PCI_DMA_TODEVICE);
04664             else
04665                 pci_unmap_page(mgp->pdev,
04666                            pci_unmap_addr(&tx->info[idx], bus),
04667                            len, PCI_DMA_TODEVICE);
04668             pci_unmap_len_set(&tx->info[idx], len, 0);
04669             tx->info[idx].skb = NULL;
04670         }
04671         idx = (idx + 1) & tx->mask;
04672     } while (idx != last_idx);
04673     if (myri10ge_skb_is_gso(skb)) {
04674         printk(KERN_ERR "myri10ge: %s: TSO but wanted to linearize?!?!?\n",
04675                mgp->dev->name);
04676         goto drop;
04677     }
04678 
04679     if (myri10ge_skb_linearize(skb))
04680         goto drop;
04681 
04682     tx->linearized++;
04683     goto again;
04684 
04685 drop:
04686     dev_kfree_skb_any(skb);
04687     ss->stats.tx_dropped += 1;
04688     return NETDEV_TX_OK;
04689 
04690 
04691 }

04803 myri10ge_set_multicast_list(struct net_device *dev)
04804 {
04805     struct myri10ge_priv *mgp = NETDEV_TO_MGP(dev);
04806     struct myri10ge_cmd cmd;
04807     struct dev_mc_list *mc_list;
04808     __be32 data[2] = {0, 0};
04809     int err;
04810     MYRI10GE_DECLARE_MAC_BUF(mac);
04811 
04812     /* can be called from atomic contexts,
04813      * pass 1 to force atomicity in myri10ge_send_cmd() */
04814     myri10ge_change_promisc(mgp, dev->flags & IFF_PROMISC, 1);
04815 
04816     /* This firmware is known to not support multicast */
04817     if (!mgp->fw_multicast_support)
04818         return;
04819 
04820     /* Disable multicast filtering */
04821 
04822     err = myri10ge_send_cmd(mgp, MXGEFW_ENABLE_ALLMULTI,
04823                 &cmd, 1);
04824     if (err != 0) {
04825         printk(KERN_ERR "myri10ge: %s: Failed MXGEFW_ENABLE_ALLMULTI,"
04826                " error status: %d\n", dev->name, err);
04827         goto abort;
04828     }
04829 
04830     if ((dev->flags & IFF_ALLMULTI) || mgp->adopted_rx_filter_bug) {
04831         /* request to disable multicast filtering, so quit here */
04832         return;
04833     }
04834 
04835     /* Flush the filters */
04836 
04837     err = myri10ge_send_cmd(mgp, MXGEFW_LEAVE_ALL_MULTICAST_GROUPS,
04838                 &cmd, 1);
04839     if (err != 0) {
04840         printk(KERN_ERR
04841                "myri10ge: %s: Failed MXGEFW_LEAVE_ALL_MULTICAST_GROUPS"
04842                ", error status: %d\n", dev->name, err);
04843         goto abort;
04844     }
04845 
04846     /* Walk the multicast list, and add each address */
04847     for (mc_list = dev->mc_list; mc_list != NULL;
04848          mc_list = mc_list->next) {
04849         memcpy(data, &mc_list->dmi_addr, 6);
04850         cmd.data0 = ntohl(data[0]);
04851         cmd.data1 = ntohl(data[1]);
04852         err = myri10ge_send_cmd(mgp, MXGEFW_JOIN_MULTICAST_GROUP,
04853                     &cmd, 1);
04854 
04855         if (err != 0) {
04856             printk(KERN_ERR "myri10ge: %s: Failed "
04857                    "MXGEFW_JOIN_MULTICAST_GROUP, error status:"
04858                    "%d\t", dev->name, err);
04859             printk(KERN_ERR "MAC " MYRI10GE_MAC_FMT "\n",
04860                    myri10ge_print_mac(mac, mc_list->dmi_addr));
04861             goto abort;
04862         }
04863     }
04864     /* Enable multicast filtering */
04865     err = myri10ge_send_cmd(mgp, MXGEFW_DISABLE_ALLMULTI,
04866                 &cmd, 1);
04867     if (err != 0) {
04868         printk(KERN_ERR "myri10ge: %s: Failed MXGEFW_DISABLE_ALLMULTI,"
04869                "error status: %d\n", dev->name, err);
04870         goto abort;
04871     }
04872 
04873     return;
04874 
04875   abort:
04876     return;
04877 }

04938 /*
04939  * Enable ECRC to align PCI-E Completion packets on an 8-byte boundary.
04940  * Only do it if the bridge is a root port since we don't want to disturb
04941  * any other device, except if forced with myri10ge_ecrc_enable > 1.
04942  */
04943 
04944 static void
04945 myri10ge_enable_ecrc(struct myri10ge_priv *mgp)
04946 {
04947     struct pci_dev *bridge = mgp->pdev->bus->self;
04948     DECLARE_INIT_DEV(dev,&mgp->pdev->dev);
04949     unsigned cap;
04950     unsigned err_cap;
04951     u16 val;
04952     u8 ext_type;
04953     int ret;
04954 
04955     if (!myri10ge_ecrc_enable || !bridge)
04956         return;
04957 
04958     /* check that the bridge is a root port */
04959     cap = pci_find_capability(bridge, PCI_CAP_ID_EXP);
04960     pci_read_config_word(bridge, cap + PCI_CAP_FLAGS, &val);
04961     ext_type = (val & PCI_EXP_FLAGS_TYPE) >> 4;
04962     if (ext_type != PCI_EXP_TYPE_ROOT_PORT) {
04963         if (myri10ge_ecrc_enable > 1) {
04964             struct pci_dev *prev_bridge, *old_bridge = bridge;
04965             
04966             /* Walk the hierarchy up to the root port
04967              * where ECRC has to be enabled */
04968             do {
04969                 prev_bridge = bridge;
04970                 bridge = bridge->bus->self;
04971                 if (!bridge || prev_bridge == bridge) {
04972                     dev_err(dev,
04973                         "Failed to find root port"
04974                         " to force ECRC\n");
04975                     return;
04976                 }
04977                 cap = pci_find_capability(bridge, PCI_CAP_ID_EXP);
04978                 pci_read_config_word(bridge, cap + PCI_CAP_FLAGS, &val);
04979                 ext_type = (val & PCI_EXP_FLAGS_TYPE) >> 4;
04980             } while (ext_type != PCI_EXP_TYPE_ROOT_PORT);
04981 
04982             dev_info(dev,
04983                  "Forcing ECRC on non-root port %s"
04984                  " (enabling on root port %s)\n",
04985                  pci_name(old_bridge), pci_name(bridge));
04986         } else {
04987 #ifndef LINUX_KERNEL_SPECIFIC

04993             return;
04994         }
04995     }
04996 
04997     cap = pci_find_ext_capability(bridge, PCI_EXT_CAP_ID_ERR);
04998 #ifndef LINUX_KERNEL_SPECIFIC
04999     /* nvidia ext cap is not always linked in ext cap chain,
05000      * fixed with a quirk in 2.6.18 */
05001     if (!cap
05002         && bridge->vendor == PCI_VENDOR_ID_NVIDIA
05003         && (bridge->device == PCI_DEVICE_ID_NVIDIA_NFORCE_CK804_PCIE
05004         || (bridge->device >= PCI_DEVICE_ID_NVIDIA_NFORCE_MCP55_PCIE_374 
05005             && bridge->device <= PCI_DEVICE_ID_NVIDIA_NFORCE_MCP55_PCIE_378)))
05006         cap = 0x160;
05007 #endif
05008     if (!cap)
05009         return;

05011 #ifdef LINUX_KERNEL_SPECIFIC
05012     ret = pci_read_config_dword(bridge, cap + PCI_ERR_CAP, &err_cap);
05013 #else /* LINUX_KERNEL_SPECIFIC */

05015 #endif /* LINUX_KERNEL_SPECIFIC */

05024         return;

05031 #endif /* LINUX_KERNEL_SPECIFIC */

05033          "Enabled ECRC on upstream bridge %s\n",

05037 /*
05038  * The Lanai Z8E PCI-E interface achieves higher Read-DMA throughput
05039  * when the PCI-E Completion packets are aligned on an 8-byte
05040  * boundary.  Some PCI-E chip sets always align Completion packets; on
05041  * the ones that do not, the alignment can be enforced by enabling
05042  * ECRC generation (if supported).
05043  *
05044  * When PCI-E Completion packets are not aligned, it is actually more
05045  * efficient to limit Read-DMA transactions to 2KB, rather than 4KB.
05046  *
05047  * If the driver can neither enable ECRC nor verify that it has
05048  * already been enabled, then it must use a firmware image which works
05049  * around unaligned completion packets (myri10ge_rss_ethp_z8e.dat), and it
05050  * should also ensure that it never gives the device a Read-DMA which is
05051  * larger than 2KB by setting the tx_boundary to 2KB.  If ECRC is
05052  * enabled, then the driver should use the aligned (myri10ge_rss_eth_z8e.dat)
05053  * firmware image, and set tx_boundary to 4KB.
05054  */
05055 
05056 static void
05057 myri10ge_firmware_probe(struct myri10ge_priv *mgp)
05058 {
05059     struct pci_dev *pdev = mgp->pdev;
05060     DECLARE_INIT_DEV(dev,&pdev->dev);
05061     int status;
05062 
05063     mgp->tx_boundary = 4096;
05064     /*
05065      * Verify the max read request size was set to 4KB
05066      * before trying the test with 4KB.
05067      */
05068     status = myri10ge_pcie_get_readrq(pdev);
05069     if (status < 0) {
05070         dev_err(dev, "Couldn't read max read req size: %d\n", status);
05071         goto abort;
05072     }
05073     if (status != 4096) {
05074         dev_warn(dev, "Max Read Request size != 4096 (%d)\n", status);
05075         mgp->tx_boundary = 2048;
05076     }
05077     /* 
05078      * load the optimized firmware (which assumes aligned PCIe
05079      * completions) in order to see if it works on this host.
05080      */
05081     mgp->fw_name = myri10ge_fw_aligned;
05082     status = myri10ge_load_firmware(mgp, 1);
05083     if (status != 0) {
05084         goto abort;
05085     }
05086 
05087     /* 
05088      * Enable ECRC if possible
05089      */
05090     myri10ge_enable_ecrc(mgp);
05091 
05092     /* 
05093      * Run a DMA test which watches for unaligned completions and
05094      * aborts on the first one seen.
05095      */
05096 
05097     status = myri10ge_dma_test(mgp, MXGEFW_CMD_UNALIGNED_TEST);
05098     if (status == 0)
05099         return; /* keep the aligned firmware */
05100 
05101     if (status != -E2BIG)
05102         dev_warn(dev, "DMA test failed: %d\n", status);
05103     if (status == -ENOSYS)
05104         dev_warn(dev, "Falling back to ethp! "
05105              "Please install up to date fw\n");
05106 abort:
05107     /* fall back to using the unaligned firmware */
05108     mgp->tx_boundary = 2048;
05109     mgp->fw_name = myri10ge_fw_unaligned;
05110     
05111     
05112 }

05173  * Older kernels init the MSI-X table once, and then use the MSI-X
05174  * table stored in PCI config space as the definative data store.
05175  * When then NIC is reset or looses power, the table is lost, and

05180 static u8 *
05181 myri10ge_find_msix_table(struct myri10ge_priv *mgp)
05182 {
05183     struct mcp_gen_header *hdr;
05184     size_t hdr_offset;
05185     DECLARE_INIT_DEV(dev,&mgp->pdev->dev);
05186 
05187     /* find running firmware header */
05188     hdr_offset = ntohl(__raw_readl(mgp->sram + MCP_HEADER_PTR_OFFSET));
05189 
05190     if ((hdr_offset & 3) || hdr_offset + sizeof(*hdr) > mgp->sram_size) {
05191         dev_err(dev, "Running firmware has bad header offset (%d)\n",
05192             (int)hdr_offset);
05193         return NULL;
05194     }
05195     hdr = (struct mcp_gen_header *) (mgp->sram + hdr_offset);
05196 
05197     if (ntohl(hdr->header_length) <
05198         offsetof(struct mcp_gen_header, msix_table_addr)) {
05199         /* header does not include valide MSI-X table address */
05200         return NULL;
05201     }
05202     if (ntohl(hdr->msix_table_addr))
05203         return (mgp->sram + ntohl(hdr->msix_table_addr));
05204     return NULL;
05205 }

05418 /*
05419  * This watchdog is used to check whether the board has suffered
05420  * from a parity error and needs to be recovered.
05421  */
05422 static void
05423 myri10ge_watchdog(MYRI10GE_WATCHDOG_ARG_TYPE work)
05424 {
05425     struct myri10ge_priv *mgp = MYRI10GE_WATCHDOG_ARG_CONTAINER_OF_MGP(work, struct myri10ge_priv, watchdog_work);
05426     struct myri10ge_tx_buf *tx;
05427     u32 reboot;
05428     int status, rebooted;
05429     int i;
05430     u16 cmd, vendor;
05431 
05432     mgp->watchdog_resets++;
05433     pci_read_config_word(mgp->pdev, PCI_COMMAND, &cmd);
05434     rebooted = 0;
05435     if ((cmd & PCI_COMMAND_MASTER) == 0) {
05436         /* Bus master DMA disabled?  Check to see
05437          * if the card rebooted due to a parity error
05438          * For now, just report it */
05439         reboot = myri10ge_read_reboot(mgp);
05440         printk(KERN_ERR "myri10ge: %s: NIC rebooted (0x%x),%s resetting\n",
05441                mgp->dev->name, reboot, 
05442                myri10ge_reset_recover ? " " : " not");
05443         if (myri10ge_reset_recover == 0)
05444             return;
05445         rtnl_lock();
05446         mgp->rebooted = 1;
05447         rebooted = 1;
05448         myri10ge_close(mgp->dev);
05449         myri10ge_reset_recover--;
05450         mgp->rebooted = 0;      
05451         /*
05452          * A rebooted nic will come back with config space as
05453          * it was after power was applied to PCIe bus.
05454          * Attempt to restore config space which was saved
05455          * when the driver was loaded, or the last time the
05456          * nic was resumed from power saving mode.
05457          */
05458         myri10ge_restore_state(mgp->pdev);
05459 
05460         /* save state again for accounting reasons */
05461         myri10ge_save_state(mgp->pdev);
05462 
05463     } else {
05464         /* if we get back -1's from our slot, perhaps somebody
05465            powered off our card.  Don't try to reset it in
05466            this case */
05467         if (cmd == 0xffff) {
05468             pci_read_config_word(mgp->pdev, PCI_VENDOR_ID, &vendor);
05469             if (vendor == 0xffff) {
05470                 printk(KERN_ERR "myri10ge: %s: device disappeared!\n",
05471                        mgp->dev->name);
05472                 return;
05473             }
05474         }
05475         /* Perhaps it is a software error.  Try to reset */
05476 
05477         printk(KERN_ERR "myri10ge: %s: device timeout, resetting\n",
05478                mgp->dev->name);
05479 #ifndef __VMKERNEL_MODULE__
05480         for (i = 0; i < mgp->num_slices; i++) {
05481             tx = &mgp->ss[i].tx;
05482             printk(KERN_INFO
05483                    "myri10ge: %s: (%d): %d %d %d %d %d %d\n",
05484                    mgp->dev->name, i, tx->queue_active, tx->req, tx->done,
05485                    tx->pkt_start, tx->pkt_done,
05486                    (int)ntohl(mgp->ss[i].fw_stats->send_done_count));
05487             myri10ge_msleep(2000);
05488             printk(KERN_INFO
05489                    "myri10ge: %s: (%d): %d %d %d %d %d %d\n",
05490                    mgp->dev->name, i, tx->queue_active, tx->req, tx->done,
05491                    tx->pkt_start, tx->pkt_done,
05492                    (int)ntohl(mgp->ss[i].fw_stats->send_done_count));
05493         }
05494 #endif
05495     }
05496 
05497     if (!rebooted) {
05498         rtnl_lock();
05499         myri10ge_close(mgp->dev);
05500     }
05501     status = myri10ge_load_firmware(mgp, 1);
05502     if (status != 0)
05503         printk(KERN_ERR "myri10ge: %s: failed to load firmware\n",
05504                mgp->dev->name);
05505     else
05506 #ifdef __VMKERNEL_MODULE__
05507     {
05508         atomic_set(&mgp->reset_pending, 0);
05509         myri10ge_open(mgp->dev);
05510     }
05511 #else
05512         myri10ge_open(mgp->dev);
05513 #endif
05514     rtnl_unlock();
05515 }

05517 /*
05518  * We use our own timer routine rather than relying upon
05519  * netdev->tx_timeout because we have a very large hardware transmit
05520  * queue.  Due to the large queue, the netdev->tx_timeout function
05521  * cannot detect a NIC with a parity error in a timely fashion if the
05522  * NIC is lightly loaded.
05523  */
05524 static void
05525 myri10ge_watchdog_timer(unsigned long arg)
05526 {
05527     struct myri10ge_priv *mgp;
05528     struct myri10ge_slice_state *ss;
05529     int i, reset_needed, busy_slice_cnt;
05530     u32 rx_pause_cnt;
05531     u16 cmd;
05532 
05533     mgp = (struct myri10ge_priv *) arg;
05534 
05535     rx_pause_cnt = ntohl(mgp->ss[0].fw_stats->dropped_pause);
05536     busy_slice_cnt = 0;
05537     for (i = 0, reset_needed = 0;
05538          i < mgp->num_slices && reset_needed == 0;
05539          ++i) {
05540 
05541         ss = &mgp->ss[i];
05542         if (ss->rx_small.watchdog_needed) {
05543             myri10ge_alloc_rx_pages(mgp, &ss->rx_small,
05544                         mgp->small_bytes + MXGEFW_PAD, 1);
05545             if (ss->rx_small.fill_cnt -  ss->rx_small.cnt >= myri10ge_fill_thresh)
05546                 ss->rx_small.watchdog_needed = 0;
05547         }
05548         if (ss->rx_big.watchdog_needed) {
05549             myri10ge_alloc_rx_pages(mgp, &ss->rx_big,
05550                         mgp->big_bytes, 1);
05551             if (ss->rx_big.fill_cnt -  ss->rx_big.cnt >= myri10ge_fill_thresh)
05552                 ss->rx_big.watchdog_needed = 0;
05553         }
05554 
05555         if (ss->tx.req != ss->tx.done &&
05556             ss->tx.done == ss->watchdog_tx_done &&
05557             ss->watchdog_tx_req != ss->watchdog_tx_done) {
05558             /* nic seems like it might be stuck.. */
05559             if (rx_pause_cnt != mgp->watchdog_pause) {
05560                 if (net_ratelimit())
05561                     printk(KERN_WARNING "myri10ge %s slice %d:"
05562                            "TX paused, check link partner\n",
05563                            mgp->dev->name, i);
05564             } else {
05565                 printk(KERN_WARNING "myri10ge %s slice %d stuck:",
05566                        mgp->dev->name, i);
05567                 reset_needed = 1;
05568             }
05569         }
05570         if (ss->watchdog_tx_done != ss->tx.done ||
05571             ss->watchdog_rx_done != ss->rx_done.cnt) {
05572             busy_slice_cnt++;
05573         }
05574         ss->watchdog_tx_done = ss->tx.done;
05575         ss->watchdog_tx_req = ss->tx.req;
05576         ss->watchdog_rx_done = ss->rx_done.cnt;
05577     }
05578     /* if we've sent or received no traffic, poll the NIC to
05579        ensure it is still there.  Otherwise, we risk not noticing
05580        an error in a timely fashion */
05581     if (busy_slice_cnt == 0) {
05582         pci_read_config_word(mgp->pdev, PCI_COMMAND, &cmd);
05583         if ((cmd & PCI_COMMAND_MASTER) == 0) {
05584             reset_needed = 1;
05585         }
05586     }
05587     mgp->watchdog_pause = rx_pause_cnt;
05588 
05589     if (reset_needed) {
05590 #ifdef __VMKERNEL_MODULE__
05591         if (atomic_cmpxchg(&mgp->reset_pending, 0, 1) == 0)
05592 #endif
05593             schedule_work(&mgp->watchdog_work);
05594     } else {
05595         /* rearm timer */
05596         mod_timer(&mgp->watchdog_timer,
05597               jiffies + myri10ge_watchdog_timeout * HZ);
05598     }
05599 }

05699 /*
05700   * This function determines the number of slices supported.
05701   * The number slices is the minumum of the number of CPUS,
05702   * the number of MSI-X irqs supported, the number of slices
05703   * supported by the firmware
05704   */
05705 static void
05706 myri10ge_probe_slices(struct myri10ge_priv *mgp)
05707 {
05708     struct myri10ge_cmd cmd;
05709     struct pci_dev *pdev = mgp->pdev;
05710     char *old_fw;
05711     int i, status, ncpus, msix_cap;
05712 
05713     mgp->num_slices = 1;
05714 #if MYRI10GE_VPUMP
05715     /* no rss vpump fw; stick with what we have */
05716     return;
05717 #endif
05718     msix_cap = pci_find_capability(pdev, PCI_CAP_ID_MSIX);
05719     ncpus = num_online_cpus();
05720 
05721     if (myri10ge_max_slices == 1 || msix_cap == 0 ||
05722         (myri10ge_max_slices == -1 && ncpus < 2))
05723         return;
05724 
05725     /* try to load the slice aware rss firmware */
05726     old_fw = mgp->fw_name;
05727     if (myri10ge_fw_name != NULL) {
05728         dev_info(&mgp->pdev->dev, "overriding rss firmware to %s\n",
05729              myri10ge_fw_name);
05730         mgp->fw_name = myri10ge_fw_name;
05731     } else if (old_fw == myri10ge_fw_aligned)
05732         mgp->fw_name = myri10ge_fw_rss_aligned;
05733     else
05734         mgp->fw_name = myri10ge_fw_rss_unaligned;
05735     status = myri10ge_load_firmware(mgp, 0);
05736     if (status != 0) {
05737         dev_info(&pdev->dev, "Rss firmware not found\n");
05738         return;
05739     }
05740 
05741     /* hit the board with a reset to ensure it is alive */
05742     memset(&cmd, 0, sizeof (cmd));
05743     status = myri10ge_send_cmd(mgp, MXGEFW_CMD_RESET, &cmd, 0);
05744     if (status != 0) {
05745         dev_err(&mgp->pdev->dev, "failed reset\n");
05746         goto abort_with_fw;
05747         return;
05748     }
05749 
05750     mgp->max_intr_slots = cmd.data0 / sizeof (struct mcp_slot);
05751 
05752     /* tell it the size of the interrupt queues */
05753     cmd.data0 = mgp->max_intr_slots * sizeof (struct mcp_slot);
05754     status = myri10ge_send_cmd(mgp, MXGEFW_CMD_SET_INTRQ_SIZE, &cmd, 0);
05755     if (status != 0) {
05756         dev_err(&mgp->pdev->dev, "failed MXGEFW_CMD_SET_INTRQ_SIZE\n");
05757         goto abort_with_fw;
05758     }
05759 
05760     /* ask the maximum number of slices it supports */
05761     status = myri10ge_send_cmd(mgp, MXGEFW_CMD_GET_MAX_RSS_QUEUES,
05762                    &cmd, 0);
05763     if (status != 0)
05764         goto abort_with_fw;
05765     else
05766         mgp->num_slices = cmd.data0;
05767 
05768     /* Only allow multiple slices if MSI-X is usable */
05769     if (!myri10ge_try_msi(pdev)) {
05770         goto abort_with_fw;
05771     }
05772 
05773     /* if the admin did not specify a limit to how many
05774        slices we should use, cap it automatically to the
05775        number of CPUs currently online */
05776     if (myri10ge_max_slices == -1)
05777         myri10ge_max_slices = ncpus;
05778 
05779     if (mgp->num_slices > myri10ge_max_slices)
05780         mgp->num_slices = myri10ge_max_slices;
05781 
05782 #ifdef MYRI10GE_HAVE_MSI
05783     /* Now try to allocate as many MSI-X vectors as we have
05784        slices. We give up on MSI-X if we can only get a single
05785        vector. */
05786 
05787     mgp->msix_vectors = myri10ge_kzalloc(mgp->num_slices *
05788                          sizeof (*mgp->msix_vectors),
05789                          GFP_KERNEL);
05790     if (mgp->msix_vectors == NULL)
05791         goto disable_msix;
05792     for (i = 0; i < mgp->num_slices; i++) {
05793         mgp->msix_vectors[i].entry = i;
05794     }
05795 
05796     while (mgp->num_slices > 1) {
05797         /* make sure it is a power of two */
05798         while (!myri10ge_is_power_of_2(mgp->num_slices))
05799             mgp->num_slices--;
05800         if (mgp->num_slices == 1)
05801             goto disable_msix;
05802         status = pci_enable_msix(pdev, mgp->msix_vectors,
05803                      mgp->num_slices);
05804         if (status == 0) {
05805             pci_disable_msix(pdev);
05806             return;
05807         }
05808         if (status > 0)
05809             mgp->num_slices = status;
05810         else
05811 #ifdef __VMKERNEL_MODULE__
05812         {
05813             if (status == -EINVAL) {
05814                 mgp->num_slices = mgp->num_slices / 2;
05815             } else {
05816                 goto disable_msix;
05817             }
05818         }
05819 #else
05820             goto disable_msix;
05821 #endif /* __VMKERNEL_MODULE__ */
05822     }
05823 
05824  disable_msix:
05825     if (mgp->msix_vectors != NULL) {
05826         kfree(mgp->msix_vectors);
05827         mgp->msix_vectors = NULL;
05828     }
05829 #else
05830     i = 0; /* defeat gcc -Wunused */
05831 #endif /*MYRI10GE_HAVE_MSI */
05832 
05833 abort_with_fw:
05834     mgp->num_slices = 1;
05835     mgp->fw_name = old_fw;
05836     myri10ge_load_firmware(mgp, 0);
05837 }
