00196 static int e1000_set_settings(struct net_device *netdev,
00197                               struct ethtool_cmd *ecmd)
00198 {
00199     struct e1000_adapter *adapter = netdev_priv(netdev);
00200     struct e1000_hw *hw = &adapter->hw;
00201 
00202     /* If speed is 1000, autoneg cannot be turned off */
00203     if ((ecmd->autoneg == 0) && (ecmd->speed == SPEED_1000) &&
00204         (ecmd->duplex = DUPLEX_FULL))
00205         return -EINVAL;
00206 
00207     /* When SoL/IDER sessions are active, autoneg/speed/duplex
00208      * cannot be changed */
00209     if (e1000_check_reset_block(hw)) {
00210         DPRINTK(DRV, ERR, "Cannot change link characteristics "
00211                 "when SoL/IDER is active.\n");
00212         return -EINVAL;
00213     }
00214 
00215     while (test_and_set_bit(__E1000_RESETTING, &adapter->state))
00216         msleep(1);
00217 
00218     if (ecmd->autoneg == AUTONEG_ENABLE) {
00219         hw->mac.autoneg = 1;
00220         if (hw->phy.media_type == e1000_media_type_fiber)
00221             hw->phy.autoneg_advertised = ADVERTISED_1000baseT_Full |
00222                                          ADVERTISED_FIBRE |
00223                                          ADVERTISED_Autoneg;
00224         else
00225             hw->phy.autoneg_advertised = ecmd->advertising |
00226                                          ADVERTISED_TP |
00227                                          ADVERTISED_Autoneg;
00228         ecmd->advertising = hw->phy.autoneg_advertised;
00229         if (adapter->fc_autoneg)
00230             hw->fc.requested_mode = e1000_fc_default;
00231     } else {
00232         if (e1000_set_spd_dplx(adapter, ecmd->speed + ecmd->duplex)) {
00233             clear_bit(__E1000_RESETTING, &adapter->state);
00234             return -EINVAL;
00235         }
00236     }
00237 
00238     /* reset the link */
00239 
00240     if (netif_running(adapter->netdev)) {
00241         e1000_down(adapter);
00242         e1000_up(adapter);
00243     } else {
00244         e1000_reset(adapter);
00245     }
00246 
00247     clear_bit(__E1000_RESETTING, &adapter->state);
00248     return 0;
00249 }

00568 static int e1000_set_eeprom(struct net_device *netdev,
00569                             struct ethtool_eeprom *eeprom, u8 *bytes)
00570 {

00572     struct e1000_hw *hw = &adapter->hw;
00573     u16 *eeprom_buff;
00574     void *ptr;
00575     int max_len, first_word, last_word, ret_val = 0;
00576     u16 i;
00577 
00578     if (eeprom->len == 0)
00579         return -EOPNOTSUPP;
00580 
00581     if (eeprom->magic != (hw->vendor_id | (hw->device_id << 16)))
00582         return -EFAULT;
00583 
00584     max_len = hw->nvm.word_size * 2;
00585 
00586     first_word = eeprom->offset >> 1;
00587     last_word = (eeprom->offset + eeprom->len - 1) >> 1;

00590         return -ENOMEM;
00591 
00592     ptr = (void *)eeprom_buff;
00593 
00594     if (eeprom->offset & 1) {
00595         /* need read/modify/write of first changed EEPROM word */
00596         /* only the second byte of the word is being modified */
00597         ret_val = e1000_read_nvm(hw, first_word, 1,
00598                         &eeprom_buff[0]);
00599         ptr++;

00602         /* need read/modify/write of last changed EEPROM word */
00603         /* only the first byte of the word is being modified */
00604         ret_val = e1000_read_nvm(hw, last_word, 1,
00605                           &eeprom_buff[last_word - first_word]);
00606     }
00607 
00608     /* Device's eeprom is always little-endian, word addressable */
00609     for (i = 0; i < last_word - first_word + 1; i++)
00610         le16_to_cpus(&eeprom_buff[i]);
00611 
00612     memcpy(ptr, bytes, eeprom->len);
00613 
00614     for (i = 0; i < last_word - first_word + 1; i++)
00615         eeprom_buff[i] = cpu_to_le16(eeprom_buff[i]);
00616 
00617     ret_val = e1000_write_nvm(hw, first_word,
00618                               last_word - first_word + 1, eeprom_buff);
00619 
00620     /* Update the checksum over the first part of the EEPROM if needed */
00621     if ((ret_val == 0) && (first_word <= NVM_CHECKSUM_REG))
00622         e1000_update_nvm_checksum(hw);

00624     kfree(eeprom_buff);
00625     return ret_val;

01215 static int e1000_nonintegrated_phy_loopback(struct e1000_adapter *adapter)
01216 {
01217     u32 ctrl_reg;
01218     u16 phy_reg;
01219 
01220     /* Setup the Device Control Register for PHY loopback test. */
01221 
01222     ctrl_reg = E1000_READ_REG(&adapter->hw, E1000_CTRL);
01223     ctrl_reg |= (E1000_CTRL_ILOS |      /* Invert Loss-Of-Signal */
01224              E1000_CTRL_FRCSPD |    /* Set the Force Speed Bit */
01225              E1000_CTRL_FRCDPX |    /* Set the Force Duplex Bit */
01226              E1000_CTRL_SPD_1000 |  /* Force Speed to 1000 */
01227              E1000_CTRL_FD);        /* Force Duplex to FULL */
01228 
01229     E1000_WRITE_REG(&adapter->hw, E1000_CTRL, ctrl_reg);
01230 
01231     /* Read the PHY Specific Control Register (0x10) */
01232     e1000_read_phy_reg(&adapter->hw, M88E1000_PHY_SPEC_CTRL, &phy_reg);
01233 
01234     /* Clear Auto-Crossover bits in PHY Specific Control Register
01235      * (bits 6:5).
01236      */
01237     phy_reg &= ~M88E1000_PSCR_AUTO_X_MODE;
01238     e1000_write_phy_reg(&adapter->hw, M88E1000_PHY_SPEC_CTRL, phy_reg);
01239 
01240     /* Perform software reset on the PHY */
01241     e1000_phy_commit(&adapter->hw);
01242 
01243     /* Have to setup TX_CLK and TX_CRS after software reset */
01244     e1000_phy_reset_clk_and_crs(adapter);
01245 
01246     e1000_write_phy_reg(&adapter->hw, PHY_CONTROL, 0x8100);
01247 
01248     /* Wait for reset to complete. */
01249     udelay(500);
01250 
01251     /* Have to setup TX_CLK and TX_CRS after software reset */
01252     e1000_phy_reset_clk_and_crs(adapter);
01253 
01254     /* Write out to PHY registers 29 and 30 to disable the Receiver. */
01255     e1000_phy_disable_receiver(adapter);
01256 
01257     /* Set the loopback bit in the PHY control register. */
01258     e1000_read_phy_reg(&adapter->hw, PHY_CONTROL, &phy_reg);
01259     phy_reg |= MII_CR_LOOPBACK;
01260     e1000_write_phy_reg(&adapter->hw, PHY_CONTROL, phy_reg);
01261 
01262     /* Setup TX_CLK and TX_CRS one more time. */
01263     e1000_phy_reset_clk_and_crs(adapter);
01264 
01265     /* Check Phy Configuration */
01266     e1000_read_phy_reg(&adapter->hw, PHY_CONTROL, &phy_reg);
01267     if (phy_reg != 0x4100)
01268          return 9;
01269 
01270     e1000_read_phy_reg(&adapter->hw, M88E1000_EXT_PHY_SPEC_CTRL, &phy_reg);
01271     if (phy_reg != 0x0070)
01272         return 10;
01273 
01274     e1000_read_phy_reg(&adapter->hw, 29, &phy_reg);
01275     if (phy_reg != 0x001A)
01276         return 11;
01277 
01278     return 0;
01279 }

01281 static int e1000_integrated_phy_loopback(struct e1000_adapter *adapter)
01282 {
01283     u32 ctrl_reg = 0;
01284     u32 stat_reg = 0;
01285 
01286     adapter->hw.mac.autoneg = false;
01287 
01288     if (adapter->hw.phy.type == e1000_phy_m88) {
01289         /* Auto-MDI/MDIX Off */
01290         e1000_write_phy_reg(&adapter->hw,
01291                     M88E1000_PHY_SPEC_CTRL, 0x0808);
01292         /* reset to update Auto-MDI/MDIX */
01293         e1000_write_phy_reg(&adapter->hw, PHY_CONTROL, 0x9140);
01294         /* autoneg off */
01295         e1000_write_phy_reg(&adapter->hw, PHY_CONTROL, 0x8140);
01296     } else if (adapter->hw.phy.type == e1000_phy_gg82563)
01297         e1000_write_phy_reg(&adapter->hw,
01298                             GG82563_PHY_KMRN_MODE_CTRL,
01299                             0x1CC);
01300 
01301     ctrl_reg = E1000_READ_REG(&adapter->hw, E1000_CTRL);
01302 
01303     if (adapter->hw.phy.type == e1000_phy_ife) {
01304         /* force 100, set loopback */
01305         e1000_write_phy_reg(&adapter->hw, PHY_CONTROL, 0x6100);
01306 
01307         /* Now set up the MAC to the same speed/duplex as the PHY. */
01308         ctrl_reg &= ~E1000_CTRL_SPD_SEL; /* Clear the speed sel bits */
01309         ctrl_reg |= (E1000_CTRL_FRCSPD | /* Set the Force Speed Bit */
01310                  E1000_CTRL_FRCDPX | /* Set the Force Duplex Bit */
01311                  E1000_CTRL_SPD_100 |/* Force Speed to 100 */
01312                  E1000_CTRL_FD);     /* Force Duplex to FULL */
01313     } else {
01314         /* force 1000, set loopback */
01315         e1000_write_phy_reg(&adapter->hw, PHY_CONTROL, 0x4140);
01316 
01317         /* Now set up the MAC to the same speed/duplex as the PHY. */
01318         ctrl_reg = E1000_READ_REG(&adapter->hw, E1000_CTRL);
01319         ctrl_reg &= ~E1000_CTRL_SPD_SEL; /* Clear the speed sel bits */
01320         ctrl_reg |= (E1000_CTRL_FRCSPD | /* Set the Force Speed Bit */
01321                  E1000_CTRL_FRCDPX | /* Set the Force Duplex Bit */
01322                  E1000_CTRL_SPD_1000 |/* Force Speed to 1000 */
01323                  E1000_CTRL_FD);     /* Force Duplex to FULL */
01324     }
01325 
01326     if (adapter->hw.phy.media_type == e1000_media_type_copper &&
01327        adapter->hw.phy.type == e1000_phy_m88) {
01328         ctrl_reg |= E1000_CTRL_ILOS; /* Invert Loss of Signal */
01329     } else {
01330         /* Set the ILOS bit on the fiber Nic if half duplex link is
01331          * detected. */
01332         stat_reg = E1000_READ_REG(&adapter->hw, E1000_STATUS);
01333         if ((stat_reg & E1000_STATUS_FD) == 0)
01334             ctrl_reg |= (E1000_CTRL_ILOS | E1000_CTRL_SLU);
01335     }
01336 
01337     E1000_WRITE_REG(&adapter->hw, E1000_CTRL, ctrl_reg);
01338 
01339     /* Disable the receiver on the PHY so when a cable is plugged in, the
01340      * PHY does not begin to autoneg when a cable is reconnected to the NIC.

01342     if (adapter->hw.phy.type == e1000_phy_m88)
01343         e1000_phy_disable_receiver(adapter);
01344 
01345     udelay(500);
01346 
01347     return 0;
01348 }

01465 static int e1000_run_loopback_test(struct e1000_adapter *adapter)
01466 {
01467     struct e1000_tx_ring *tx_ring = &adapter->test_tx_ring;
01468     struct e1000_rx_ring *rx_ring = &adapter->test_rx_ring;
01469     struct pci_dev *pdev = adapter->pdev;
01470     int i, j, k, l, lc, good_cnt, ret_val=0;
01471     unsigned long time;
01472 
01473     E1000_WRITE_REG(&adapter->hw, E1000_RDT(0), rx_ring->count - 1);
01474 
01475     /* Calculate the loop count based on the largest descriptor ring
01476      * The idea is to wrap the largest ring a number of times using 64
01477      * send/receive pairs during each loop
01478      */
01479 
01480     if (rx_ring->count <= tx_ring->count)
01481         lc = ((tx_ring->count / 64) * 2) + 1;
01482     else
01483         lc = ((rx_ring->count / 64) * 2) + 1;
01484 
01485     k = l = 0;
01486     for (j = 0; j <= lc; j++) { /* loop count loop */
01487         for (i = 0; i < 64; i++) { /* send the packets */
01488             e1000_create_lbtest_frame(tx_ring->buffer_info[k].skb,
01489                     1024);
01490             dma_sync_single_for_device(pci_dev_to_dev(pdev),
01491                     tx_ring->buffer_info[k].dma,
01492                         tx_ring->buffer_info[k].length,
01493                         DMA_TO_DEVICE);
01494             if (unlikely(++k == tx_ring->count)) k = 0;
01495         }
01496         E1000_WRITE_REG(&adapter->hw, E1000_TDT(0), k);
01497         msleep(200);
01498         time = jiffies; /* set the start time for the receive */
01499         good_cnt = 0;
01500         do { /* receive the sent packets */
01501             dma_sync_single_for_cpu(pci_dev_to_dev(pdev),
01502                             rx_ring->buffer_info[l].dma,
01503                             E1000_RXBUFFER_2048,
01504                             DMA_FROM_DEVICE);
01505 
01506             ret_val = e1000_check_lbtest_frame(
01507                     rx_ring->buffer_info[l].skb,
01508                     1024);
01509             if (!ret_val)
01510                 good_cnt++;
01511             if (unlikely(++l == rx_ring->count)) l = 0;
01512             /* time + 20 msecs (200 msecs on 2.4) is more than
01513              * enough time to complete the receives, if it's
01514              * exceeded, break and error off
01515              */
01516         } while (good_cnt < 64 && jiffies < (time + 20));
01517         if (good_cnt != 64) {
01518             ret_val = 13; /* ret_val is the same as mis-compare */
01519             break;
01520         }
01521         if (jiffies >= (time + 20)) {
01522             ret_val = 14; /* error code for time out error */
01523             break;
01524         }
01525     } /* end loop count loop */
01526     return ret_val;
01527 }
