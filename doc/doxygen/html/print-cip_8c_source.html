00063 /*
00064  * This is the top level routine of the printer.  'p' points
00065  * to the LLC/SNAP or raw header of the packet, 'h->ts' is the timestamp,
00066  * 'h->len' is the length of the packet off the wire, and 'h->caplen'
00067  * is the number of bytes actually captured.
00068  */
00069 u_int
00070 cip_if_print(const struct pcap_pkthdr *h, const u_char *p)
00071 {
00072     u_int caplen = h->caplen;
00073     u_int length = h->len;
00074     u_short extracted_ethertype;
00075 
00076     if (memcmp(rfcllc, p, sizeof(rfcllc))==0 && caplen < RFC1483LLC_LEN) {
00077         printf("[|cip]");
00078         return (0);
00079     }
00080 
00081     if (eflag)
00082         cip_print(length);
00083 
00084     if (memcmp(rfcllc, p, sizeof(rfcllc)) == 0) {
00085         /*
00086          * LLC header is present.  Try to print it & higher layers.
00087          */
00088         if (llc_print(p, length, caplen, NULL, NULL,
00089             &extracted_ethertype) == 0) {
00090             /* ether_type not known, print raw packet */
00091             if (!eflag)
00092                 cip_print(length);
00093             if (extracted_ethertype) {
00094                 printf("(LLC %s) ",
00095                    etherproto_string(htons(extracted_ethertype)));
00096             }
00097             if (!suppress_default_print)
00098                 default_print(p, caplen);
00099         }
00100     } else {
00101         /*
00102          * LLC header is absent; treat it as just IP.
00103          */
00104         ip_print(gndo, p, length);
00105     }
00106 
00107     return (0);
00108 }
