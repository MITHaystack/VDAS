00299     u32 reg, rx_pba_size;

00302     if (!dcb_config->pfc_mode_enable)

00305     /* Enable Transmit Priority Flow Control */
00306     reg = IXGBE_READ_REG(hw, IXGBE_RMCS);
00307     reg &= ~IXGBE_RMCS_TFCE_802_3X;

00315     reg |= IXGBE_FCTRL_RPFCE;
00316     IXGBE_WRITE_REG(hw, IXGBE_FCTRL, reg);
00317 
00318     /*
00319      * Configure flow control thresholds and enable priority flow control
00320      * for each traffic class.
00321      */
00322     for (i = 0; i < MAX_TRAFFIC_CLASS; i++) {
00323         rx_pba_size = IXGBE_READ_REG(hw, IXGBE_RXPBSIZE(i));
00324         rx_pba_size >>= IXGBE_RXPBSIZE_SHIFT;
00325         reg = (rx_pba_size - hw->fc.low_water) << 10;
00326 
00327         if (dcb_config->tc_config[i].dcb_pfc == pfc_enabled_tx ||
00328             dcb_config->tc_config[i].dcb_pfc == pfc_enabled_full)
00329             reg |= IXGBE_FCRTL_XONE;
00330 
00331         IXGBE_WRITE_REG(hw, IXGBE_FCRTL(i), reg);
00332 
00333         reg = (rx_pba_size - hw->fc.high_water) << 10;
00334         if (dcb_config->tc_config[i].dcb_pfc == pfc_enabled_tx ||
00335             dcb_config->tc_config[i].dcb_pfc == pfc_enabled_full)
00336             reg |= IXGBE_FCRTH_FCEN;
00337 
00338         IXGBE_WRITE_REG(hw, IXGBE_FCRTH(i), reg);
00339     }
00340 
00341     /* Configure pause time */
00342     for (i = 0; i < (MAX_TRAFFIC_CLASS >> 1); i++)
00343         IXGBE_WRITE_REG(hw, IXGBE_FCTTV(i), 0x68006800);
