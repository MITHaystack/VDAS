00071 int ixgbe_fcoe_ddp_put(struct net_device *netdev, u16 xid)
00072 {
00073     int len = 0;
00074     struct ixgbe_fcoe *fcoe;
00075     struct ixgbe_adapter *adapter;
00076     struct ixgbe_fcoe_ddp *ddp;
00077     u32 fcbuff;
00078 
00079     if (!netdev)
00080         goto out_ddp_put;
00081 
00082     if (xid >= IXGBE_FCOE_DDP_MAX)
00083         goto out_ddp_put;
00084 
00085     adapter = netdev_priv(netdev);
00086     fcoe = &adapter->fcoe;
00087     ddp = &fcoe->ddp[xid];
00088     if (!ddp->udl)

00091     len = ddp->len;
00092     /* if there an error, force to invalidate ddp context */
00093     if (ddp->err) {
00094         spin_lock_bh(&fcoe->lock);

00096         IXGBE_WRITE_REG(&adapter->hw, IXGBE_FCFLTRW,
00097                 (xid | IXGBE_FCFLTRW_WE));
00098         IXGBE_WRITE_REG(&adapter->hw, IXGBE_FCBUFF, 0);
00099         IXGBE_WRITE_REG(&adapter->hw, IXGBE_FCDMARW,
00100                 (xid | IXGBE_FCDMARW_WE));
00101 
00102         /* guaranteed to be invalidated after 100us */
00103         IXGBE_WRITE_REG(&adapter->hw, IXGBE_FCDMARW,
00104                 (xid | IXGBE_FCDMARW_RE));
00105         fcbuff = IXGBE_READ_REG(&adapter->hw, IXGBE_FCBUFF);
00106         spin_unlock_bh(&fcoe->lock);
00107         if (fcbuff & IXGBE_FCBUFF_VALID)
00108             udelay(100);
00109     }

00111         pci_unmap_sg(adapter->pdev, ddp->sgl, ddp->sgc,
00112                  DMA_FROM_DEVICE);
00113     pci_pool_free(fcoe->pool, ddp->udl, ddp->udp);
00114     ixgbe_fcoe_clear_ddp(ddp);
00115 
00116 out_ddp_put:

00129 static int ixgbe_fcoe_ddp_setup(struct net_device *netdev, u16 xid,
00130                 struct scatterlist *sgl, unsigned int sgc,
00131                 int target_mode)
00132 {
00133     struct ixgbe_adapter *adapter;
00134     struct ixgbe_hw *hw;
00135     struct ixgbe_fcoe *fcoe;
00136     struct ixgbe_fcoe_ddp *ddp;
00137     struct scatterlist *sg;
00138     unsigned int i, j, dmacount;
00139     unsigned int len;
00140     static const unsigned int bufflen = IXGBE_FCBUFF_MIN;
00141     unsigned int firstoff = 0;
00142     unsigned int lastsize;
00143     unsigned int thisoff = 0;
00144     unsigned int thislen = 0;
00145     u32 fcbuff, fcdmarw, fcfltrw, fcrxctl;
00146     dma_addr_t addr = 0;
00147 
00148     if (!netdev || !sgl || !sgc)
00149         return 0;
00150 
00151     adapter = netdev_priv(netdev);
00152     if (xid >= IXGBE_FCOE_DDP_MAX) {
00153         e_warn(drv, "xid=0x%x out-of-range\n", xid);
00154         return 0;
00155     }
00156 
00157     /* no DDP if we are already down or resetting */
00158     if (test_bit(__IXGBE_DOWN, &adapter->state) ||
00159         test_bit(__IXGBE_RESETTING, &adapter->state))
00160         return 0;
00161 
00162     fcoe = &adapter->fcoe;
00163     if (!fcoe->pool) {
00164         e_warn(drv, "xid=0x%x no ddp pool for fcoe\n", xid);
00165         return 0;
00166     }
00167 
00168     ddp = &fcoe->ddp[xid];
00169     if (ddp->sgl) {
00170         e_err(drv, "xid 0x%x w/ non-null sgl=%p nents=%d\n",
00171             xid, ddp->sgl, ddp->sgc);
00172         return 0;
00173     }
00174     ixgbe_fcoe_clear_ddp(ddp);
00175 
00176     /* setup dma from scsi command sgl */
00177     dmacount = pci_map_sg(adapter->pdev, sgl, sgc, DMA_FROM_DEVICE);
00178     if (dmacount == 0) {
00179         e_err(drv, "xid 0x%x DMA map error\n", xid);
00180         return 0;
00181     }
00182 
00183     /* alloc the udl from our ddp pool */
00184     ddp->udl = pci_pool_alloc(fcoe->pool, GFP_ATOMIC, &ddp->udp);
00185     if (!ddp->udl) {
00186         e_err(drv, "failed allocated ddp context\n");
00187         goto out_noddp_unmap;
00188     }
00189     ddp->sgl = sgl;
00190     ddp->sgc = sgc;
00191 
00192     j = 0;
00193     for_each_sg(sgl, sg, dmacount, i) {
00194         addr = sg_dma_address(sg);
00195         len = sg_dma_len(sg);
00196         while (len) {
00197             /* max number of buffers allowed in one DDP context */
00198             if (j >= IXGBE_BUFFCNT_MAX) {
00199                 e_err(drv, "xid=%x:%d,%d,%d:addr=%llx "
00200                       "not enough descriptors\n",
00201                     xid, i, j, dmacount, (u64)addr);
00202                 goto out_noddp_free;
00203             }
00204 
00205             /* get the offset of length of current buffer */
00206             thisoff = addr & ((dma_addr_t)bufflen - 1);
00207             thislen = min((bufflen - thisoff), len);
00208             /*
00209              * all but the 1st buffer (j == 0)
00210              * must be aligned on bufflen
00211              */
00212             if ((j != 0) && (thisoff))
00213                 goto out_noddp_free;
00214             /*
00215              * all but the last buffer
00216              * ((i == (dmacount - 1)) && (thislen == len))
00217              * must end at bufflen
00218              */
00219             if (((i != (dmacount - 1)) || (thislen != len))
00220                 && ((thislen + thisoff) != bufflen))
00221                 goto out_noddp_free;
00222 
00223             ddp->udl[j] = (u64)(addr - thisoff);
00224             /* only the first buffer may have none-zero offset */
00225             if (j == 0)
00226                 firstoff = thisoff;
00227             len -= thislen;
00228             addr += thislen;
00229             j++;
00230         }
00231     }
00232     /* only the last buffer may have non-full bufflen */
00233     lastsize = thisoff + thislen;
00234 
00235     /*
00236      * lastsize can not be bufflen.
00237      * If it is then adding another buffer with lastsize = 1.
00238      * Since lastsize is 1 there will be no HW access to this buffer.
00239      */
00240     if (lastsize == bufflen) {
00241         if (j >= IXGBE_BUFFCNT_MAX) {
00242             e_err(drv, "xid=%x:%d,%d,%d:addr=%llx "
00243                   "not enough user buffers. We need an extra "
00244                   "buffer because lastsize is bufflen.\n",
00245                   xid, i, j, dmacount, (u64)addr);
00246             goto out_noddp_free;
00247         }
00248 
00249         ddp->udl[j] = (u64)(fcoe->extra_ddp_buffer_dma);
00250         j++;
00251         lastsize = 1;
00252     }
00253 
00254     fcbuff = (IXGBE_FCBUFF_4KB << IXGBE_FCBUFF_BUFFSIZE_SHIFT);
00255     fcbuff |= ((j & 0xff) << IXGBE_FCBUFF_BUFFCNT_SHIFT);
00256     fcbuff |= (firstoff << IXGBE_FCBUFF_OFFSET_SHIFT);
00257     /* Set WRCONTX bit to allow DDP for target */
00258     if (target_mode)
00259         fcbuff |= (IXGBE_FCBUFF_WRCONTX);
00260     fcbuff |= (IXGBE_FCBUFF_VALID);
00261 
00262     fcdmarw = xid;
00263     fcdmarw |= IXGBE_FCDMARW_WE;
00264     fcdmarw |= (lastsize << IXGBE_FCDMARW_LASTSIZE_SHIFT);
00265 
00266     fcfltrw = xid;
00267     fcfltrw |= IXGBE_FCFLTRW_WE;
00268 
00269     /* program DMA context */
00270     hw = &adapter->hw;
00271     spin_lock_bh(&fcoe->lock);
00272 
00273     /* turn on last frame indication for target mode as FCP_RSPtarget is
00274      * supposed to send FCP_RSP when it is done. */
00275     if (target_mode && !test_bit(__IXGBE_FCOE_TARGET, &fcoe->mode)) {
00276         set_bit(__IXGBE_FCOE_TARGET, &fcoe->mode);
00277         fcrxctl = IXGBE_READ_REG(hw, IXGBE_FCRXCTRL);
00278         fcrxctl |= IXGBE_FCRXCTRL_LASTSEQH;
00279         IXGBE_WRITE_REG(hw, IXGBE_FCRXCTRL, fcrxctl);
00280     }
00281 
00282     IXGBE_WRITE_REG(hw, IXGBE_FCPTRL, ddp->udp & DMA_BIT_MASK(32));
00283     IXGBE_WRITE_REG(hw, IXGBE_FCPTRH, (u64)ddp->udp >> 32);
00284     IXGBE_WRITE_REG(hw, IXGBE_FCBUFF, fcbuff);
00285     IXGBE_WRITE_REG(hw, IXGBE_FCDMARW, fcdmarw);
00286     /* program filter context */
00287     IXGBE_WRITE_REG(hw, IXGBE_FCPARAM, 0);
00288     IXGBE_WRITE_REG(hw, IXGBE_FCFLT, IXGBE_FCFLT_VALID);
00289     IXGBE_WRITE_REG(hw, IXGBE_FCFLTRW, fcfltrw);
00290 
00291     spin_unlock_bh(&fcoe->lock);
00292 
00293     return 1;
00294 
00295 out_noddp_free:
00296     pci_pool_free(fcoe->pool, ddp->udl, ddp->udp);
00297     ixgbe_fcoe_clear_ddp(ddp);
00298 
00299 out_noddp_unmap:
00300     pci_unmap_sg(adapter->pdev, sgl, sgc, DMA_FROM_DEVICE);
00301     return 0;
00302 }

00359 int ixgbe_fcoe_ddp(struct ixgbe_adapter *adapter,
00360            union ixgbe_adv_rx_desc *rx_desc,
00361            struct sk_buff *skb,
00362            u32 staterr)
00363 {
00364     struct ixgbe_fcoe *fcoe = &adapter->fcoe;
00365     struct ixgbe_fcoe_ddp *ddp;
00366     struct fc_frame_header *fh;
00367     struct fcoe_crc_eof *crc;
00368     int rc = -EINVAL;
00369     u32 fcerr = staterr & IXGBE_RXDADV_ERR_FCERR;
00370     u32 ddp_err, fctl;
00371     u16 xid;
00372 
00373     skb->ip_summed = (fcerr == IXGBE_FCERR_BADCRC) ? CHECKSUM_NONE :
00374                              CHECKSUM_UNNECESSARY;
00375 
00376     fh = (struct fc_frame_header *)(skb->data + sizeof(struct fcoe_hdr));
00377 
00378     if (skb->protocol == htons(ETH_P_8021Q))
00379         fh = (struct fc_frame_header *)((char *)fh + VLAN_HLEN);
00380 
00381     fctl = ntoh24(fh->fh_f_ctl);
00382     if (fctl & FC_FC_EX_CTX)
00383         xid =  ntohs(fh->fh_ox_id);
00384     else
00385         xid =  ntohs(fh->fh_rx_id);
00386 
00387     if (xid >= IXGBE_FCOE_DDP_MAX)
00388         goto ddp_out;
00389 
00390     ddp = &fcoe->ddp[xid];
00391     if (!ddp->udl)
00392         goto ddp_out;
00393 
00394     ddp_err = staterr & (IXGBE_RXDADV_ERR_FCEOFE | IXGBE_RXDADV_ERR_FCERR);
00395     ddp->err = ddp_err;
00396     if (ddp_err)
00397         goto ddp_out;
00398 
00399     switch (staterr & IXGBE_RXDADV_STAT_FCSTAT) {
00400     /* return 0 to bypass going to ULD for DDPed data */
00401     case IXGBE_RXDADV_STAT_FCSTAT_DDP:
00402         /* update length of DDPed data */
00403         ddp->len = le32_to_cpu(rx_desc->wb.lower.hi_dword.rss);
00404         rc = 0;
00405         break;
00406     /* unmap the sg list when FCPRSP is received */
00407     case IXGBE_RXDADV_STAT_FCSTAT_FCPRSP:
00408         pci_unmap_sg(adapter->pdev, ddp->sgl,
00409                  ddp->sgc, DMA_FROM_DEVICE);
00410         ddp->sgl = NULL;
00411         ddp->sgc = 0;
00412         /* fall through */
00413     /* if DDP length is present pass it through to ULD */
00414     case IXGBE_RXDADV_STAT_FCSTAT_NODDP:
00415         /* update length of DDPed data */
00416         ddp->len = le32_to_cpu(rx_desc->wb.lower.hi_dword.rss);
00417         if (rx_desc->wb.lower.hi_dword.rss)
00418             rc = ddp->len;
00419         break;
00420     /* no match will return as an error */
00421     case IXGBE_RXDADV_STAT_FCSTAT_NOMTCH:
00422     default:
00423         break;
00424     }
00425 
00426     /* In target mode, check the last data frame of the sequence.
00427      * For DDP in target mode, data is already DDPed but the header
00428      * indication of the last data frame ould allow is to tell if we
00429      * got all the data and the ULP can send FCP_RSP back, as this is
00430      * not a full fcoe frame, we fill the trailer here so it won't be
00431      * dropped by the ULP stack.
00432      */
00433     if ((fh->fh_r_ctl == FC_RCTL_DD_SOL_DATA) &&
00434         (fctl & FC_FC_END_SEQ)) {
00435         crc = (struct fcoe_crc_eof *)skb_put(skb, sizeof(*crc));
00436         crc->fcoe_eof = FC_EOF_T;
00437     }
00438 ddp_out:
00439     return rc;
00440 }

00453 int ixgbe_fso(struct ixgbe_ring *tx_ring, struct sk_buff *skb,
00454               u32 tx_flags, u8 *hdr_len)
00455 {
00456     struct fc_frame_header *fh;
00457     u32 vlan_macip_lens;
00458     u32 fcoe_sof_eof = 0;
00459     u32 mss_l4len_idx;
00460     u8 sof, eof;
00461 
00462 #ifdef NETIF_F_FSO
00463     if (skb_is_gso(skb) && skb_shinfo(skb)->gso_type != SKB_GSO_FCOE) {
00464         dev_err(tx_ring->dev, "Wrong gso type %d:expecting SKB_GSO_FCOE\n",
00465             skb_shinfo(skb)->gso_type);
00466         return -EINVAL;
00467     }
00468 
00469 #endif
00470     /* resets the header to point fcoe/fc */
00471     skb_set_network_header(skb, skb->mac_len);
00472     skb_set_transport_header(skb, skb->mac_len +
00473                  sizeof(struct fcoe_hdr));
00474 
00475     /* sets up SOF and ORIS */
00476     sof = ((struct fcoe_hdr *)skb_network_header(skb))->fcoe_sof;
00477     switch (sof) {
00478     case FC_SOF_I2:
00479         fcoe_sof_eof = IXGBE_ADVTXD_FCOEF_ORIS;
00480         break;
00481     case FC_SOF_I3:
00482         fcoe_sof_eof = IXGBE_ADVTXD_FCOEF_SOF |
00483                    IXGBE_ADVTXD_FCOEF_ORIS;
00484         break;
00485     case FC_SOF_N2:
00486         break;
00487     case FC_SOF_N3:
00488         fcoe_sof_eof = IXGBE_ADVTXD_FCOEF_SOF;
00489         break;
00490     default:
00491         dev_warn(tx_ring->dev, "unknown sof = 0x%x\n", sof);
00492         return -EINVAL;
00493     }
00494 
00495     /* the first byte of the last dword is EOF */
00496     skb_copy_bits(skb, skb->len - 4, &eof, 1);
00497     /* sets up EOF and ORIE */
00498     switch (eof) {
00499     case FC_EOF_N:
00500         fcoe_sof_eof |= IXGBE_ADVTXD_FCOEF_EOF_N;
00501         break;
00502     case FC_EOF_T:
00503         /* lso needs ORIE */
00504         if (skb_is_gso(skb))
00505             fcoe_sof_eof |= IXGBE_ADVTXD_FCOEF_EOF_N |
00506                     IXGBE_ADVTXD_FCOEF_ORIE;
00507         else
00508             fcoe_sof_eof |= IXGBE_ADVTXD_FCOEF_EOF_T;
00509         break;
00510     case FC_EOF_NI:
00511         fcoe_sof_eof |= IXGBE_ADVTXD_FCOEF_EOF_NI;
00512         break;
00513     case FC_EOF_A:
00514         fcoe_sof_eof |= IXGBE_ADVTXD_FCOEF_EOF_A;
00515         break;
00516     default:
00517         dev_warn(tx_ring->dev, "unknown eof = 0x%x\n", eof);
00518         return -EINVAL;
00519     }
00520 
00521     /* sets up PARINC indicating data offset */
00522     fh = (struct fc_frame_header *)skb_transport_header(skb);
00523     if (fh->fh_f_ctl[2] & FC_FC_REL_OFF)
00524         fcoe_sof_eof |= IXGBE_ADVTXD_FCOEF_PARINC;
00525 
00526     /* include trailer in headlen as it is replicated per frame */
00527     *hdr_len = sizeof(struct fcoe_crc_eof);
00528 
00529     /* hdr_len includes fc_hdr if FCoE LSO is enabled */
00530     if (skb_is_gso(skb))
00531         *hdr_len += skb_transport_offset(skb) +
00532                 sizeof(struct fc_frame_header);
00533 
00534     /* mss_l4len_id: use 1 for FSO as TSO, no need for L4LEN */
00535     mss_l4len_idx = skb_shinfo(skb)->gso_size << IXGBE_ADVTXD_MSS_SHIFT;
00536     mss_l4len_idx |= 1 << IXGBE_ADVTXD_IDX_SHIFT;
00537 
00538     /* vlan_macip_lens: HEADLEN, MACLEN, VLAN tag */
00539     vlan_macip_lens = skb_transport_offset(skb) +
00540               sizeof(struct fc_frame_header);
00541     vlan_macip_lens |= (skb_transport_offset(skb) - 4)
00542                << IXGBE_ADVTXD_MACLEN_SHIFT;
00543     vlan_macip_lens |= tx_flags & IXGBE_TX_FLAGS_VLAN_MASK;
00544 
00545     /* write context desc */
00546     ixgbe_tx_ctxtdesc(tx_ring, vlan_macip_lens, fcoe_sof_eof,
00547               IXGBE_ADVTXT_TUCMD_FCOE, mss_l4len_idx);
00548 
00549     return skb_is_gso(skb);
00550 }
