00207  * Poll the receiver queue and call the pcap callback-handler
00208  * with the packet.
00209  */
00210 static int
00211 pcap_read_one (pcap_t *p, pcap_handler callback, u_char *data)
00212 {
00213   struct pcap_pkthdr pcap;
00214   struct timeval     now, expiry;
00215   BYTE  *rx_buf;
00216   int    rx_len = 0;
00217 
00218   if (p->md.timeout > 0)
00219   {
00220     gettimeofday2 (&now, NULL);
00221     expiry.tv_usec = now.tv_usec + 1000UL * p->md.timeout;
00222     expiry.tv_sec  = now.tv_sec;
00223     while (expiry.tv_usec >= 1000000L)
00224     {
00225       expiry.tv_usec -= 1000000L;
00226       expiry.tv_sec++;
00227     }
00228   }
00229 
00230   while (!exc_occured)
00231   {
00232     volatile struct device *dev; /* might be reset by sig_handler */
00233 
00234     dev = get_device (p->fd);
00235     if (!dev)
00236        break;
00237 
00238     PCAP_ASSERT (dev->copy_rx_buf || dev->peek_rx_buf);
00239     FLUSHK();
00240 
00241     /* If driver has a zero-copy receive facility, peek at the queue,
00242      * filter it, do the callback and release the buffer.
00243      */
00244     if (dev->peek_rx_buf)
00245     {
00246       PCAP_ASSERT (dev->release_rx_buf);
00247       rx_len = (*dev->peek_rx_buf) (&rx_buf);
00248     }
00249     else
00250     {
00251       BYTE buf [ETH_MAX+100]; /* add some margin */
00252       rx_len = (*dev->copy_rx_buf) (buf, p->snapshot);
00253       rx_buf = buf;
00254     }
00255 
00256     if (rx_len > 0)  /* got a packet */
00257     {
00258       mac_count++;
00259 
00260       FLUSHK();
00261 
00262       pcap.caplen = min (rx_len, p->snapshot);
00263       pcap.len    = rx_len;
00264 
00265       if (callback &&
00266           (!p->fcode.bf_insns || bpf_filter(p->fcode.bf_insns, rx_buf, pcap.len, pcap.caplen)))
00267       {
00268         filter_count++;
00269 
00270         /* Fix-me!! Should be time of arrival. Not time of
00271          * capture.
00272          */
00273         gettimeofday2 (&pcap.ts, NULL);
00274         (*callback) (data, &pcap, rx_buf);
00275       }
00276 
00277       if (dev->release_rx_buf)
00278         (*dev->release_rx_buf) (rx_buf);
00279 
00280       if (pcap_pkt_debug > 0)
00281       {
00282         if (callback == watt32_recv_hook)
00283              dbug_write ("pcap_recv_hook\n");
00284         else dbug_write ("pcap_read_op\n");
00285       }
00286       FLUSHK();
00287       return (1);
00288     }
00289 
00290     /* If not to wait for a packet or pcap_close() called from
00291      * e.g. SIGINT handler, exit loop now.
00292      */
00293     if (p->md.timeout <= 0 || (volatile int)p->fd <= 0)
00294        break;
00295 
00296     gettimeofday2 (&now, NULL);
00297 
00298     if (timercmp(&now, &expiry, >))
00299        break;
00300 
00301 #ifndef DJGPP
00302     kbhit();    /* a real CPU hog */
00303 #endif
00304 
00305     if (p->wait_proc)
00306       (*p->wait_proc)();     /* call yield func */
00307   }
00308 
00309   if (rx_len < 0)            /* receive error */
00310   {
00311     p->md.stat.ps_drop++;
00312 #ifdef USE_32BIT_DRIVERS
00313     if (pcap_pkt_debug > 1)
00314        printk ("pkt-err %s\n", pktInfo.error);
00315 #endif
00316     return (-1);
00317   }
00318   return (0);
00319 }

00321 static int
00322 pcap_read_dos (pcap_t *p, int cnt, pcap_handler callback, u_char *data)
00323 {
00324   int rc, num = 0;
00325 
00326   while (num <= cnt || (cnt < 0))
00327   {
00328     if (p->fd <= 0)
00329        return (-1);
00330     rc = pcap_read_one (p, callback, data);
00331     if (rc > 0)
00332        num++;
00333     if (rc < 0)
00334        break;
00335     _w32_os_yield();  /* allow SIGINT generation, yield to Win95/NT */
00336   }
00337   return (num);

00510 /*
00511  * Get a list of all interfaces that are present and that we probe okay.
00512  * Returns -1 on error, 0 otherwise.
00513  * The list, as returned through "alldevsp", may be null if no interfaces
00514  * were up and could be opened.
00515  */
00516 int pcap_findalldevs (pcap_if_t **alldevsp, char *errbuf)
00517 {
00518   struct device     *dev;
00519   struct sockaddr_ll sa_ll_1, sa_ll_2;
00520   struct sockaddr   *addr, *netmask, *broadaddr, *dstaddr;
00521   pcap_if_t *devlist = NULL;
00522   int       ret = 0;
00523   size_t    addr_size = sizeof(struct sockaddr_ll);
00524 
00525   for (dev = (struct device*)dev_base; dev; dev = dev->next)
00526   {
00527     PCAP_ASSERT (dev->probe);
00528 
00529     if (!(*dev->probe)(dev))
00530        continue;
00531 
00532     PCAP_ASSERT (dev->close);  /* set by probe routine */
00533     FLUSHK();
00534     (*dev->close) (dev);
00535 
00536     memset (&sa_ll_1, 0, sizeof(sa_ll_1));
00537     memset (&sa_ll_2, 0, sizeof(sa_ll_2));
00538     sa_ll_1.sll_family = AF_PACKET;
00539     sa_ll_2.sll_family = AF_PACKET;
00540 
00541     addr      = (struct sockaddr*) &sa_ll_1;
00542     netmask   = (struct sockaddr*) &sa_ll_1;
00543     dstaddr   = (struct sockaddr*) &sa_ll_1;
00544     broadaddr = (struct sockaddr*) &sa_ll_2;
00545     memset (&sa_ll_2.sll_addr, 0xFF, sizeof(sa_ll_2.sll_addr));
00546 
00547     if (pcap_add_if(&devlist, dev->name, dev->flags,
00548                     dev->long_name, errbuf) < 0)
00549     {
00550       ret = -1;
00551       break;
00552     }
00553     if (add_addr_to_iflist(&devlist,dev->name, dev->flags, addr, addr_size,
00554                            netmask, addr_size, broadaddr, addr_size,
00555                            dstaddr, addr_size, errbuf) < 0)
00556     {
00557       ret = -1;
00558       break;
00559     }
00560   }
00561 
00562   if (devlist && ret < 0)
00563   {
00564     pcap_freealldevs (devlist);
00565     devlist = NULL;
00566   }
00567   else
00568   if (!devlist)
00569      strcpy (errbuf, "No drivers found");
00570 
00571   *alldevsp = devlist;
00572   return (ret);
00573 }

00600 /*
00601  * Initialise a named network device.
00602  */
00603 static struct device *
00604 open_driver (const char *dev_name, char *ebuf, int promisc)
00605 {
00606   struct device *dev;
00607 
00608   for (dev = (struct device*)dev_base; dev; dev = dev->next)
00609   {
00610     PCAP_ASSERT (dev->name);
00611 
00612     if (strcmp (dev_name,dev->name))
00613        continue;
00614 
00615     if (!probed_dev)   /* user didn't call pcap_lookupdev() first */
00616     {
00617       PCAP_ASSERT (dev->probe);
00618 
00619       if (!(*dev->probe)(dev))    /* call the xx_probe() function */
00620       {
00621         sprintf (ebuf, "failed to detect device `%s'", dev_name);
00622         return (NULL);
00623       }
00624       probed_dev = dev;  /* device is probed okay and may be used */
00625     }
00626     else if (dev != probed_dev)
00627     {
00628       goto not_probed;
00629     }
00630 
00631     FLUSHK();
00632 
00633     /* Select what traffic to receive
00634      */
00635     if (promisc)
00636          dev->flags |=  (IFF_ALLMULTI | IFF_PROMISC);
00637     else dev->flags &= ~(IFF_ALLMULTI | IFF_PROMISC);
00638 
00639     PCAP_ASSERT (dev->open);
00640 
00641     if (!(*dev->open)(dev))
00642     {
00643       sprintf (ebuf, "failed to activate device `%s'", dev_name);
00644       if (pktInfo.error && !strncmp(dev->name,"pkt",3))
00645       {
00646         strcat (ebuf, ": ");
00647         strcat (ebuf, pktInfo.error);
00648       }
00649       return (NULL);
00650     }
00651 
00652     /* Some devices need this to operate in promiscous mode
00653      */
00654     if (promisc && dev->set_multicast_list)
00655        (*dev->set_multicast_list) (dev);
00656 
00657     active_dev = dev;   /* remember our active device */
00658     break;
00659   }
00660 
00661   /* 'dev_name' not matched in 'dev_base' list.
00662    */
00663   if (!dev)
00664   {
00665     sprintf (ebuf, "device `%s' not supported", dev_name);
00666     return (NULL);
00667   }
00668 
00669 not_probed:
00670   if (!probed_dev)
00671   {
00672     sprintf (ebuf, "device `%s' not probed", dev_name);
00673     return (NULL);
00674   }
00675   return (dev);
00676 }

00901 /*
00902  * To use features of Watt-32 (netdb functions and socket etc.)
00903  * we must call sock_init(). But we set various hooks to prevent
00904  * using normal PKTDRVR functions in pcpkt.c. This should hopefully
00905  * make Watt-32 and pcap co-operate.
00906  */
00907 static int init_watt32 (struct pcap *pcap, const char *dev_name, char *err_buf)
00908 {
00909   char *env;
00910   int   rc, MTU, has_ip_addr;
00911   int   using_pktdrv = 1;
00912 
00913   /* If user called sock_init() first, we need to reinit in
00914    * order to open debug/trace-file properly
00915    */
00916   if (_watt_is_init)
00917      sock_exit();
00918 
00919   env = getenv ("PCAP_DEBUG");
00920   if (env && atoi(env) > 0 &&
00921       pcap_pkt_debug < 0)   /* if not already set */
00922   {
00923     dbug_init();
00924     pcap_pkt_debug = atoi (env);
00925   }
00926 
00927   _watt_do_exit      = 0;    /* prevent sock_init() calling exit() */
00928   prev_post_hook     = _w32_usr_post_init;
00929   _w32_usr_post_init = pcap_init_hook;
00930   _w32_print_hook    = null_print;
00931 
00932   if (dev_name && strncmp(dev_name,"pkt",3))
00933      using_pktdrv = FALSE;
00934 
00935   rc = sock_init();
00936   has_ip_addr = (rc != 8);  /* IP-address assignment failed */
00937 
00938   /* if pcap is using a 32-bit driver w/o a pktdrvr loaded, we
00939    * just pretend Watt-32 is initialised okay.
00940    *
00941    * !! fix-me: The Watt-32 config isn't done if no pktdrvr
00942    *            was found. In that case my_ip_addr + sin_mask
00943    *            have default values. Should be taken from another
00944    *            ini-file/environment in any case (ref. tcpdump.ini)
00945    */
00946   _watt_is_init = 1;  
00947 
00948   if (!using_pktdrv || !has_ip_addr)  /* for now .... */
00949   {
00950     static const char myip[] = "192.168.0.1";
00951     static const char mask[] = "255.255.255.0";
00952 
00953     printf ("Just guessing, using IP %s and netmask %s\n", myip, mask);
00954     my_ip_addr    = aton (myip);
00955     _w32_sin_mask = aton (mask);
00956   }
00957   else if (rc && using_pktdrv)
00958   {
00959     sprintf (err_buf, "sock_init() failed, code %d", rc);
00960     return (0);
00961   }
00962 
00963   /* Set recv-hook for peeking in _eth_arrived().
00964    */
00965 #if (WATTCP_VER >= 0x0224)
00966   _eth_recv_hook = pcap_recv_hook;
00967   _eth_xmit_hook = pcap_xmit_hook;
00968 #endif
00969 
00970   /* Free the pkt-drvr handle allocated in pkt_init().
00971    * The above hooks should thus use the handle reopened in open_driver()
00972    */
00973   if (using_pktdrv)
00974   {
00975     _eth_release();
00976 /*  _eth_is_init = 1; */  /* hack to get Rx/Tx-hooks in Watt-32 working */
00977   }
00978 
00979   memcpy (&pcap_save, pcap, sizeof(pcap_save));
00980   MTU = pkt_get_mtu();
00981   pcap_save.fcode.bf_insns = NULL;
00982   pcap_save.linktype       = _eth_get_hwtype (NULL, NULL);
00983   pcap_save.snapshot       = MTU > 0 ? MTU : ETH_MAX; /* assume 1514 */
00984 
00985 #if 1
00986   /* prevent use of resolve() and resolve_ip()
00987    */
00988   last_nameserver = 0;
00989 #endif
00990   return (1);
00991 }

01278 /*
01279  * Dequeue routine is called by polling.
01280  * NOTE: the queue-element is not copied, only a pointer is
01281  * returned at '*buf'
01282  */
01283 int peek_rxbuf (BYTE **buf)
01284 {
01285   struct rx_elem *tail, *head;
01286 
01287   PCAP_ASSERT (pktq_check (&active_dev->queue));
01288 
01289   DISABLE();
01290   tail = pktq_out_elem (&active_dev->queue);
01291   head = pktq_in_elem (&active_dev->queue);
01292   ENABLE();
01293 
01294   if (head != tail)
01295   {
01296     PCAP_ASSERT (tail->size < active_dev->queue.elem_size-4-2);
01297 
01298     *buf = &tail->data[0];
01299     return (tail->size);
01300   }
01301   *buf = NULL;
01302   return (0);
01303 }
01304 
01305 /*
01306  * Release buffer we peeked at above.
01307  */
01308 int release_rxbuf (BYTE *buf)
01309 {
01310 #ifndef NDEBUG
01311   struct rx_elem *tail = pktq_out_elem (&active_dev->queue);
01312 
01313   PCAP_ASSERT (&tail->data[0] == buf);
01314 #else
01315   ARGSUSED (buf);
01316 #endif
01317   pktq_inc_out (&active_dev->queue);
01318   return (1);
01319 }
