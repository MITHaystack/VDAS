00051 /*
00052  * Print IPX datagram packets.
00053  */
00054 void
00055 ipx_print(const u_char *p, u_int length)
00056 {
00057     const struct ipxHdr *ipx = (const struct ipxHdr *)p;
00058 
00059     if (!eflag)
00060         printf("IPX ");
00061 
00062     TCHECK(ipx->srcSkt);
00063     (void)printf("%s.%04x > ",
00064              ipxaddr_string(EXTRACT_32BITS(ipx->srcNet), ipx->srcNode),
00065              EXTRACT_16BITS(&ipx->srcSkt));
00066 
00067     (void)printf("%s.%04x: ",
00068              ipxaddr_string(EXTRACT_32BITS(ipx->dstNet), ipx->dstNode),
00069              EXTRACT_16BITS(&ipx->dstSkt));
00070 
00071     /* take length from ipx header */
00072     TCHECK(ipx->length);
00073     length = EXTRACT_16BITS(&ipx->length);
00074 
00075     ipx_decode(ipx, (u_char *)ipx + ipxSize, length - ipxSize);
00076     return;
00077 trunc:
00078     printf("[|ipx %d]", length);
00079 }

00092 void
00093 ipx_decode(const struct ipxHdr *ipx, const u_char *datap, u_int length)
00094 {
00095     register u_short dstSkt;
00096 
00097     dstSkt = EXTRACT_16BITS(&ipx->dstSkt);
00098     switch (dstSkt) {
00099       case IPX_SKT_NCP:
00100     (void)printf("ipx-ncp %d", length);
00101     break;
00102       case IPX_SKT_SAP:
00103     ipx_sap_print((u_short *)datap, length);
00104     break;
00105       case IPX_SKT_RIP:
00106     ipx_rip_print((u_short *)datap, length);
00107     break;
00108       case IPX_SKT_NETBIOS:
00109     (void)printf("ipx-netbios %d", length);
00110 #ifdef TCPDUMP_DO_SMB
00111     ipx_netbios_print(datap, length);
00112 #endif
00113     break;
00114       case IPX_SKT_DIAGNOSTICS:
00115     (void)printf("ipx-diags %d", length);
00116     break;
00117       case IPX_SKT_NWLINK_DGM:
00118     (void)printf("ipx-nwlink-dgm %d", length);
00119 #ifdef TCPDUMP_DO_SMB
00120     ipx_netbios_print(datap, length);
00121 #endif
00122     break;
00123       case IPX_SKT_EIGRP:
00124     eigrp_print(datap, length);
00125     break;
00126       default:
00127     (void)printf("ipx-#%x %d", dstSkt, length);
00128     break;
00129     }
00130 }
