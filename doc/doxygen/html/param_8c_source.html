00208 static int __devinit e1000_validate_option(unsigned int *value,
00209                        const struct e1000_option *opt,
00210                        struct e1000_adapter *adapter)
00211 {
00212     if (*value == OPTION_UNSET) {
00213         *value = opt->def;
00214         return 0;
00215     }
00216 
00217     switch (opt->type) {
00218     case enable_option:
00219         switch (*value) {
00220         case OPTION_ENABLED:
00221             e_info("%s Enabled\n", opt->name);
00222             return 0;
00223         case OPTION_DISABLED:
00224             e_info("%s Disabled\n", opt->name);
00225             return 0;
00226         }
00227         break;
00228     case range_option:
00229         if (*value >= opt->arg.r.min && *value <= opt->arg.r.max) {
00230             e_info("%s set to %i\n", opt->name, *value);
00231             return 0;
00232         }
00233         break;
00234     case list_option: {
00235         int i;
00236         struct e1000_opt_list *ent;
00237 
00238         for (i = 0; i < opt->arg.l.nr; i++) {
00239             ent = &opt->arg.l.p[i];
00240             if (*value == ent->i) {
00241                 if (ent->str[0] != '\0')
00242                     e_info("%s\n", ent->str);
00243                 return 0;
00244             }
00245         }
00246     }
00247         break;
00248     default:
00249         BUG();
00250     }
00251 
00252     e_info("Invalid %s value specified (%i) %s\n", opt->name, *value,
00253            opt->err);
00254     *value = opt->def;
00255     return -1;
00256 }
