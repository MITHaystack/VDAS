00161         (void)printf("[|llc]");

00169     /*
00170      * OK, what type of LLC frame is this?  The length
00171      * of the control field depends on that - I frames
00172      * have a two-byte control field, and U frames have
00173      * a one-byte control field.
00174      */
00175     control = *(p + 2);
00176     if ((control & LLC_U_FMT) == LLC_U_FMT) {
00177         /*
00178          * U frame.
00179          */
00180         is_u = 1;
00181     } else {
00182         /*
00183          * The control field in I and S frames is
00184          * 2 bytes...
00185          */
00186         if (caplen < 4) {
00187             (void)printf("[|llc]");
00188             default_print((u_char *)p, caplen);
00189             return(0);
00190         }
00191 
00192         /*
00193          * ...and is little-endian.
00194          */
00195         control = EXTRACT_LE_16BITS(p + 2);
00196         is_u = 0;
00197     }
00198 
00199     if (ssap_field == LLCSAP_GLOBAL && dsap_field == LLCSAP_GLOBAL) {
00200         /*
00201          * This is an Ethernet_802.3 IPX frame; it has an
00202          * 802.3 header (i.e., an Ethernet header where the
00203          * type/length field is <= ETHERMTU, i.e. it's a length
00204          * field, not a type field), but has no 802.2 header -
00205          * the IPX packet starts right after the Ethernet header,
00206          * with a signature of two bytes of 0xFF (which is
00207          * LLCSAP_GLOBAL).
00208          *
00209          * (It might also have been an Ethernet_802.3 IPX at
00210          * one time, but got bridged onto another network,
00211          * such as an 802.11 network; this has appeared in at
00212          * least one capture file.)
00213          */
00214 
00215             if (eflag)
00216         printf("IPX 802.3: ");
00217 
00218             ipx_print(p, length);
00219             return (1);
00220     }
00221 
00222     dsap = dsap_field & ~LLC_IG;
00223     ssap = ssap_field & ~LLC_GSAP;
00224 
00225     if (eflag) {
00226                 printf("LLC, dsap %s (0x%02x) %s, ssap %s (0x%02x) %s",
00227                        tok2str(llc_values, "Unknown", dsap),
00228                        dsap,
00229                        tok2str(llc_ig_flag_values, "Unknown", dsap_field & LLC_IG),
00230                        tok2str(llc_values, "Unknown", ssap),
00231                        ssap,
00232                        tok2str(llc_flag_values, "Unknown", ssap_field & LLC_GSAP));
00233 
00234         if (is_u) {
00235             printf(", ctrl 0x%02x: ", control);
00236         } else {
00237             printf(", ctrl 0x%04x: ", control);
00238         }
00239     }
00240 
00241     if (ssap == LLCSAP_8021D && dsap == LLCSAP_8021D &&
00242         control == LLC_UI) {
00243         stp_print(p+3, length-3);
00244         return (1);
00245     }
00246 
00247     if (ssap == LLCSAP_IP && dsap == LLCSAP_IP &&
00248         control == LLC_UI) {
00249         ip_print(gndo, p+4, length-4);
00250         return (1);
00251     }
00252 
00253     if (ssap == LLCSAP_IPX && dsap == LLCSAP_IPX &&
00254         control == LLC_UI) {
00255         /*
00256          * This is an Ethernet_802.2 IPX frame, with an 802.3
00257          * header and an 802.2 LLC header with the source and
00258          * destination SAPs being the IPX SAP.
00259          *
00260          * Skip DSAP, LSAP, and control field.
00261          */
00262                 if (eflag)
00263                         printf("IPX 802.2: ");
00264 
00265         ipx_print(p+3, length-3);
00266         return (1);
00267     }
00268 
00269 #ifdef TCPDUMP_DO_SMB
00270     if (ssap == LLCSAP_NETBEUI && dsap == LLCSAP_NETBEUI
00271         && (!(control & LLC_S_FMT) || control == LLC_U_FMT)) {
00272         /*
00273          * we don't actually have a full netbeui parser yet, but the
00274          * smb parser can handle many smb-in-netbeui packets, which
00275          * is very useful, so we call that
00276          *
00277          * We don't call it for S frames, however, just I frames
00278          * (which are frames that don't have the low-order bit,
00279          * LLC_S_FMT, set in the first byte of the control field)
00280          * and UI frames (whose control field is just 3, LLC_U_FMT).
00281          */
00282 
00283         /*
00284          * Skip the LLC header.
00285          */
00286         if (is_u) {
00287             p += 3;
00288             length -= 3;
00289             caplen -= 3;
00290         } else {
00291             p += 4;
00292             length -= 4;
00293             caplen -= 4;
00294         }
00295         netbeui_print(control, p, length);
00296         return (1);
00297     }
00298 #endif
00299     if (ssap == LLCSAP_ISONS && dsap == LLCSAP_ISONS
00300         && control == LLC_UI) {
00301         isoclns_print(p + 3, length - 3, caplen - 3);
00302         return (1);
00303     }
00304 
00305     if (ssap == LLCSAP_SNAP && dsap == LLCSAP_SNAP
00306         && control == LLC_UI) {
00307         /*
00308          * XXX - what *is* the right bridge pad value here?
00309          * Does anybody ever bridge one form of LAN traffic
00310          * over a networking type that uses 802.2 LLC?
00311          */
00312         ret = snap_print(p+3, length-3, caplen-3, 2);
00313         if (ret)
00314             return (ret);
00315     }
00316 
00317     if (!eflag) {
00318         if (ssap == dsap) {
00319             if (esrc == NULL || edst == NULL)
00320                 (void)printf("%s ", tok2str(llc_values, "Unknown DSAP 0x%02x", dsap));
00321             else
00322                 (void)printf("%s > %s %s ",
00323                         etheraddr_string(esrc),
00324                         etheraddr_string(edst),
00325                         tok2str(llc_values, "Unknown DSAP 0x%02x", dsap));
00326         } else {
00327             if (esrc == NULL || edst == NULL)
00328                 (void)printf("%s > %s ",
00329                                         tok2str(llc_values, "Unknown SSAP 0x%02x", ssap),
00330                     tok2str(llc_values, "Unknown DSAP 0x%02x", dsap));
00331             else
00332                 (void)printf("%s %s > %s %s ",
00333                     etheraddr_string(esrc),
00334                                         tok2str(llc_values, "Unknown SSAP 0x%02x", ssap),
00335                     etheraddr_string(edst),
00336                     tok2str(llc_values, "Unknown DSAP 0x%02x", dsap));
00337         }
00338     }
00339 
00340     if (is_u) {
00341         printf("Unnumbered, %s, Flags [%s], length %u",
00342                        tok2str(llc_cmd_values, "%02x", LLC_U_CMD(control)),
00343                        tok2str(llc_flag_values,"?",(ssap_field & LLC_GSAP) | (control & LLC_U_POLL)),
00344                        length);
00345 
00346         p += 3;
00347         length -= 3;
00348         caplen -= 3;
00349 
00350         if ((control & ~LLC_U_POLL) == LLC_XID) {
00351             if (*p == LLC_XID_FI) {
00352                 printf(": %02x %02x", p[1], p[2]);
00353                 p += 3;
00354                 length -= 3;
00355                 caplen -= 3;
00356             }
00357         }
00358     } else {
00359         if ((control & LLC_S_FMT) == LLC_S_FMT) {
00360             (void)printf("Supervisory, %s, rcv seq %u, Flags [%s], length %u",
00361                 tok2str(llc_supervisory_values,"?",LLC_S_CMD(control)),
00362                 LLC_IS_NR(control),
00363                 tok2str(llc_flag_values,"?",(ssap_field & LLC_GSAP) | (control & LLC_IS_POLL)),
00364                                 length);
00365         } else {
00366             (void)printf("Information, send seq %u, rcv seq %u, Flags [%s], length %u",
00367                 LLC_I_NS(control),
00368                 LLC_IS_NR(control),
00369                 tok2str(llc_flag_values,"?",(ssap_field & LLC_GSAP) | (control & LLC_IS_POLL)),
00370                                 length);
00371         }
00372         p += 4;
00373         length -= 4;
00374         caplen -= 4;
00375     }

00383     register u_short et;
00384     register int ret;
00385 
00386     TCHECK2(*p, 5);
00387     orgcode = EXTRACT_24BITS(p);
00388     et = EXTRACT_16BITS(p + 3);
00389 
00390     if (eflag) {
00391         const struct tok *tok = null_values;
00392         const struct oui_tok *otp;
00393 
00394         for (otp = &oui_to_tok[0]; otp->tok != NULL; otp++) {
00395             if (otp->oui == orgcode) {
00396                 tok = otp->tok;
00397                 break;
00398             }
00399         }
00400         (void)printf("oui %s (0x%06x), %s %s (0x%04x): ",
00401              tok2str(oui_values, "Unknown", orgcode),
00402              orgcode,
00403              (orgcode == 0x000000 ? "ethertype" : "pid"),
00404              tok2str(tok, "Unknown", et),
00405              et);
00406     }
00407     p += 5;
00408     length -= 5;
00409     caplen -= 5;
00410 
00411     switch (orgcode) {
00412     case OUI_ENCAP_ETHER:
00413     case OUI_CISCO_90:
00414         /*
00415          * This is an encapsulated Ethernet packet,
00416          * or a packet bridged by some piece of
00417          * Cisco hardware; the protocol ID is
00418          * an Ethernet protocol type.
00419          */
00420         ret = ethertype_print(et, p, length, caplen);
00421         if (ret)
00422             return (ret);
00423         break;
00424 
00425     case OUI_APPLETALK:
00426         if (et == ETHERTYPE_ATALK) {
00427             /*
00428              * No, I have no idea why Apple used one
00429              * of their own OUIs, rather than
00430              * 0x000000, and an Ethernet packet
00431              * type, for Appletalk data packets,
00432              * but used 0x000000 and an Ethernet
00433              * packet type for AARP packets.
00434              */
00435             ret = ethertype_print(et, p, length, caplen);
00436             if (ret)
00437                 return (ret);
00438         }
00439         break;
00440 
00441     case OUI_CISCO:
00442                 switch (et) {
00443                 case PID_CISCO_CDP:
00444                         cdp_print(p, length, caplen);
00445                         return (1);
00446                 case PID_CISCO_DTP:
00447                         dtp_print(p, length); 
00448                         return (1);
00449                 case PID_CISCO_UDLD:
00450                         udld_print(p, length);
00451                         return (1);
00452                 case PID_CISCO_VTP:
00453                         vtp_print(p, length);
00454                         return (1);
00455                 case PID_CISCO_PVST:
00456                         stp_print(p, length);
00457                         return (1);
00458                 default:
00459                         break;
00460                 }
00461 
00462     case OUI_RFC2684:
00463         switch (et) {
00464 
00465         case PID_RFC2684_ETH_FCS:
00466         case PID_RFC2684_ETH_NOFCS:
00467             /*
00468              * XXX - remove the last two bytes for
00469              * PID_RFC2684_ETH_FCS?
00470              */
00471             /*
00472              * Skip the padding.
00473              */
00474             TCHECK2(*p, bridge_pad);
00475             caplen -= bridge_pad;
00476             length -= bridge_pad;
00477             p += bridge_pad;

00479             /*

00485         case PID_RFC2684_802_5_FCS:
00486         case PID_RFC2684_802_5_NOFCS:

00488              * XXX - remove the last two bytes for
00489              * PID_RFC2684_ETH_FCS?

00506 
00507         case PID_RFC2684_FDDI_FCS:
00508         case PID_RFC2684_FDDI_NOFCS:
