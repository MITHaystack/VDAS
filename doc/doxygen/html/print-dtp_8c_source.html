00048 void
00049 dtp_print (const u_char *pptr, u_int length)
00050 {
00051     int type, len;
00052     const u_char *tptr;
00053 
00054     if (length < DTP_HEADER_LEN)
00055         goto trunc;
00056 
00057     tptr = pptr; 
00058 
00059     if (!TTEST2(*tptr, DTP_HEADER_LEN)) 
00060     goto trunc;
00061 
00062     printf("DTPv%u, length %u", 
00063            (*tptr),
00064            length);
00065 
00066     /*
00067      * In non-verbose mode, just print version.
00068      */
00069     if (vflag < 1) {
00070     return;
00071     }
00072 
00073     tptr += DTP_HEADER_LEN;
00074 
00075     while (tptr < (pptr+length)) {
00076 
00077         if (!TTEST2(*tptr, 4)) 
00078             goto trunc;
00079 
00080     type = EXTRACT_16BITS(tptr);
00081         len  = EXTRACT_16BITS(tptr+2); 
00082 
00083         /* infinite loop check */
00084         if (type == 0 || len == 0) {
00085             return;
00086         }
00087 
00088         printf("\n\t%s (0x%04x) TLV, length %u",
00089                tok2str(dtp_tlv_values, "Unknown", type),
00090                type, len);
00091 
00092         switch (type) {
00093     case DTP_DOMAIN_TLV:
00094         printf(", %s", tptr+4);
00095         break;
00096 
00097     case DTP_STATUS_TLV:            
00098     case DTP_DTP_TYPE_TLV:
00099                 printf(", 0x%x", *(tptr+4));
00100                 break;
00101 
00102     case DTP_NEIGHBOR_TLV:
00103                 printf(", %s", etheraddr_string(tptr+4));
00104                 break;
00105 
00106         default:
00107             break;
00108         }   
00109         tptr += len;
00110     }
00111 
00112     return;
00113 
00114  trunc:
00115     printf("[|dtp]");
00116 }
