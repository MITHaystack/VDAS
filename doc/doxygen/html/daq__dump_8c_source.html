00059 static int daq_dump_get_vars (
00060     DumpImpl* impl, DAQ_Config_t* cfg, char* errBuf, size_t errMax
00061 ) {
00062     const char* s = NULL;
00063     DAQ_Dict* entry;
00064 
00065     for ( entry = cfg->values; entry; entry = entry->next)
00066     {
00067         if ( !strcmp(entry->key, "load-mode") )
00068         {
00069             s = entry->value;
00070         }
00071         else if ( !strcmp(entry->key, "file") )
00072         {
00073             impl->name = strdup(entry->value);
00074         }
00075     }
00076     if ( !s )
00077         return 1;
00078 
00079     if ( !strcasecmp(s, "read-file") )
00080     {
00081         cfg->mode = DAQ_MODE_READ_FILE;
00082         return 1;
00083     }
00084     else if ( !strcasecmp(s, "passive") )
00085     {
00086         cfg->mode = DAQ_MODE_PASSIVE;
00087         return 1;
00088     }
00089     else if ( !strcasecmp(s, "inline") )
00090     {
00091         cfg->mode = DAQ_MODE_INLINE;
00092         return 1;
00093     }
00094     snprintf(errBuf, errMax, "invalid load-mode (%s)", s);
00095     return 0;
00096 }

00155 static DAQ_Verdict daq_dump_capture (
00156     void* user, const DAQ_PktHdr_t* hdr, const uint8_t* pkt)
00157 {
00158     DumpImpl* impl = (DumpImpl*)user;
00159     DAQ_Verdict verdict = impl->callback(impl->user, hdr, pkt);
00160 
00161     if ( verdict > MAX_DAQ_VERDICT )
00162         verdict = DAQ_VERDICT_BLOCK;
00163 
00164     impl->stats.verdicts[verdict]++;
00165 
00166     if ( s_fwd[verdict] )
00167         pcap_dump((u_char*)impl->dump, (struct pcap_pkthdr*)hdr, pkt);
