00139 /* RX descriptor control thresholds.
00140  * PTHRESH - MAC will consider prefetch if it has fewer than this number of
00141  *           descriptors available in its onboard memory.
00142  *           Setting this to 0 disables RX descriptor prefetch.
00143  * HTHRESH - MAC will only prefetch if there are at least this many descriptors
00144  *           available in host memory.
00145  *           If PTHRESH is 0, this should also be 0.
00146  * WTHRESH - RX descriptor writeback threshold - MAC will delay writing back
00147  *           descriptors until either it has this many to write back, or the
00148  *           ITR timer expires.
00149  */
00150 #define IGB_RX_PTHRESH                     8
00151 #define IGB_RX_HTHRESH                     8
00152 #define IGB_TX_PTHRESH                     8
00153 #define IGB_TX_HTHRESH                     1
00154 #define IGB_RX_WTHRESH                     ((hw->mac.type == e1000_82576 && \
00155                                              adapter->msix_entries) ? 1 : 4)
00156 #define IGB_TX_WTHRESH                     ((hw->mac.type == e1000_82576 && \
00157                                              adapter->msix_entries) ? 1 : 16)

00390 {

00392         return ring->next_to_clean - ring->next_to_use - 1;
