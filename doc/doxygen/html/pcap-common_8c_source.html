00059 /*
00060  * We don't write DLT_* values to capture files, because they're not the
00061  * same on all platforms.
00062  *
00063  * Unfortunately, the various flavors of BSD have not always used the same
00064  * numerical values for the same data types, and various patches to
00065  * libpcap for non-BSD OSes have added their own DLT_* codes for link
00066  * layer encapsulation types seen on those OSes, and those codes have had,
00067  * in some cases, values that were also used, on other platforms, for other
00068  * link layer encapsulation types.
00069  *
00070  * This means that capture files of a type whose numerical DLT_* code
00071  * means different things on different BSDs, or with different versions
00072  * of libpcap, can't always be read on systems other than those like
00073  * the one running on the machine on which the capture was made.
00074  *
00075  * Instead, we define here a set of LINKTYPE_* codes, and map DLT_* codes
00076  * to LINKTYPE_* codes when writing a savefile header, and map LINKTYPE_*
00077  * codes to DLT_* codes when reading a savefile header.
00078  *
00079  * For those DLT_* codes that have, as far as we know, the same values on
00080  * all platforms (DLT_NULL through DLT_FDDI), we define LINKTYPE_xxx as
00081  * DLT_xxx; that way, captures of those types can still be read by
00082  * versions of libpcap that map LINKTYPE_* values to DLT_* values, and
00083  * captures of those types written by versions of libpcap that map DLT_
00084  * values to LINKTYPE_ values can still be read by older versions
00085  * of libpcap.
00086  *
00087  * The other LINKTYPE_* codes are given values starting at 100, in the
00088  * hopes that no DLT_* code will be given one of those values.
00089  *
00090  * In order to ensure that a given LINKTYPE_* code's value will refer to
00091  * the same encapsulation type on all platforms, you should not allocate
00092  * a new LINKTYPE_* value without consulting
00093  * "tcpdump-workers@lists.tcpdump.org".  The tcpdump developers will
00094  * allocate a value for you, and will not subsequently allocate it to
00095  * anybody else; that value will be added to the "pcap.h" in the
00096  * tcpdump.org Git repository, so that a future libpcap release will
00097  * include it.
00098  *
00099  * You should, if possible, also contribute patches to libpcap and tcpdump
00100  * to handle the new encapsulation type, so that they can also be checked
00101  * into the tcpdump.org Git repository and so that they will appear in
00102  * future libpcap and tcpdump releases.
00103  *
00104  * Do *NOT* assume that any values after the largest value in this file
00105  * are available; you might not have the most up-to-date version of this
00106  * file, and new values after that one might have been assigned.  Also,
00107  * do *NOT* use any values below 100 - those might already have been
00108  * taken by one (or more!) organizations.
00109  */

00671  * the pseudo-header is:
00672  *
00673  * struct dl_ipnetinfo {
00674  *     u_int8_t   dli_version;
00675  *     u_int8_t   dli_family;
00676  *     u_int16_t  dli_htype;

01080 int
01081 dlt_to_linktype(int dlt)
01082 {
01083     int i;
01084 
01085     for (i = 0; map[i].dlt != -1; i++) {
01086         if (map[i].dlt == dlt)
01087             return (map[i].linktype);
01088     }
01089 
01090     /*
01091      * If we don't have a mapping for this DLT_ code, return an
01092      * error; that means that the table above needs to have an
01093      * entry added.
01094      */
01095     return (-1);
01096 }
