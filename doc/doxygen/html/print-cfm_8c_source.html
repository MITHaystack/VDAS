00242     u_int mgmt_addr_type;

00245     /*

00256      * Resolve the passed in Address.
00257      */

00264     case AFNUM_INET6:
00265         printf(", %s", ip6addr_string(tptr + 1));
00266         break;
00267 #endif
00268

00270         hexdump = TRUE;
00271         break;
00272     }
00273

00292 void
00293 cfm_print(register const u_char *pptr, register u_int length) {
00294 
00295     const struct cfm_common_header_t *cfm_common_header;
00296     const struct cfm_tlv_header_t *cfm_tlv_header;
00297     const u_int8_t *tptr, *tlv_ptr, *ma_name, *ma_nameformat, *ma_namelength;
00298     u_int hexdump, tlen, cfm_tlv_len, cfm_tlv_type, ccm_interval;
00299 
00300 
00301     union {
00302         const struct cfm_ccm_t *cfm_ccm;
00303         const struct cfm_lbm_t *cfm_lbm;
00304         const struct cfm_ltm_t *cfm_ltm;
00305         const struct cfm_ltr_t *cfm_ltr;
00306     } msg_ptr;
00307 
00308     tptr=pptr;
00309     cfm_common_header = (const struct cfm_common_header_t *)pptr;
00310     TCHECK(*cfm_common_header);
00311 
00312     /*
00313      * Sanity checking of the header.
00314      */
00315     if (CFM_EXTRACT_VERSION(cfm_common_header->mdlevel_version) != CFM_VERSION) {
00316     printf("CFMv%u not supported, length %u",
00317                CFM_EXTRACT_VERSION(cfm_common_header->mdlevel_version), length);
00318     return;
00319     }
00320 
00321     printf("CFMv%u %s, MD Level %u, length %u",
00322            CFM_EXTRACT_VERSION(cfm_common_header->mdlevel_version),
00323            tok2str(cfm_opcode_values, "unknown (%u)", cfm_common_header->opcode),
00324            CFM_EXTRACT_MD_LEVEL(cfm_common_header->mdlevel_version),
00325            length);
00326 
00327     /*
00328      * In non-verbose mode just print the opcode and md-level.
00329      */
00330     if (vflag < 1) {
00331         return;
00332     }
00333 
00334     printf("\n\tFirst TLV offset %u", cfm_common_header->first_tlv_offset);
00335 
00336     tptr += sizeof(const struct cfm_common_header_t);
00337     tlen = length - sizeof(struct cfm_common_header_t);
00338 
00339     switch (cfm_common_header->opcode) {
00340     case CFM_OPCODE_CCM:
00341         msg_ptr.cfm_ccm = (const struct cfm_ccm_t *)tptr;
00342 
00343         ccm_interval = CFM_EXTRACT_CCM_INTERVAL(cfm_common_header->flags);
00344         printf(", Flags [CCM Interval %u%s]",
00345                ccm_interval,
00346                cfm_common_header->flags & CFM_CCM_RDI_FLAG ?
00347                ", RDI" : "");
00348 
00349         /*
00350          * Resolve the CCM interval field.
00351          */
00352         if (ccm_interval) {
00353             printf("\n\t  CCM Interval %.3fs"
00354                    ", min CCM Lifetime %.3fs, max CCM Lifetime %.3fs",
00355                    ccm_interval_base[ccm_interval],
00356                    ccm_interval_base[ccm_interval] * CCM_INTERVAL_MIN_MULTIPLIER,
00357                    ccm_interval_base[ccm_interval] * CCM_INTERVAL_MAX_MULTIPLIER);
00358         }
00359 
00360         printf("\n\t  Sequence Number 0x%08x, MA-End-Point-ID 0x%04x",
00361                EXTRACT_32BITS(msg_ptr.cfm_ccm->sequence),
00362                EXTRACT_16BITS(msg_ptr.cfm_ccm->ma_epi));
00363 
00364 
00365         /*
00366          * Resolve the MD fields.
00367          */
00368         printf("\n\t  MD Name Format %s (%u), MD Name length %u",
00369                tok2str(cfm_md_nameformat_values, "Unknown",
00370                        msg_ptr.cfm_ccm->md_nameformat),
00371                msg_ptr.cfm_ccm->md_nameformat,
00372                msg_ptr.cfm_ccm->md_namelength);
00373 
00374         if (msg_ptr.cfm_ccm->md_nameformat != CFM_CCM_MD_FORMAT_NONE) {
00375             printf("\n\t  MD Name: ");
00376             switch (msg_ptr.cfm_ccm->md_nameformat) {
00377             case CFM_CCM_MD_FORMAT_DNS:
00378             case CFM_CCM_MD_FORMAT_CHAR:
00379                 safeputs((const char *)msg_ptr.cfm_ccm->md_name, msg_ptr.cfm_ccm->md_namelength);
00380                 break;
00381 
00382             case CFM_CCM_MD_FORMAT_MAC:
00383                 printf("\n\t  MAC %s", etheraddr_string(
00384                            msg_ptr.cfm_ccm->md_name));
00385                 break;
00386 
00387                 /* FIXME add printers for those MD formats - hexdump for now */
00388             case CFM_CCM_MA_FORMAT_8021:
00389             default:
00390                 print_unknown_data(msg_ptr.cfm_ccm->md_name, "\n\t    ",
00391                                    msg_ptr.cfm_ccm->md_namelength);
00392             }
00393         }
00394 
00395 
00396         /*
00397          * Resolve the MA fields.
00398          */
00399         ma_nameformat = msg_ptr.cfm_ccm->md_name + msg_ptr.cfm_ccm->md_namelength;
00400         ma_namelength = msg_ptr.cfm_ccm->md_name + msg_ptr.cfm_ccm->md_namelength + 1;
00401         ma_name = msg_ptr.cfm_ccm->md_name + msg_ptr.cfm_ccm->md_namelength + 2;
00402 
00403         printf("\n\t  MA Name-Format %s (%u), MA name length %u",
00404                tok2str(cfm_ma_nameformat_values, "Unknown",
00405                        *ma_nameformat),
00406                *ma_nameformat,
00407                *ma_namelength);        
00408 
00409         printf("\n\t  MA Name: ");
00410         switch (*ma_nameformat) {
00411         case CFM_CCM_MA_FORMAT_CHAR:
00412             safeputs((const char *)ma_name, *ma_namelength);
00413             break;
00414 
00415             /* FIXME add printers for those MA formats - hexdump for now */
00416         case CFM_CCM_MA_FORMAT_8021:
00417         case CFM_CCM_MA_FORMAT_VID:
00418         case CFM_CCM_MA_FORMAT_INT:
00419         case CFM_CCM_MA_FORMAT_VPN:
00420         default:
00421             print_unknown_data(ma_name, "\n\t    ", *ma_namelength);
00422         }
00423         break;
00424 
00425     case CFM_OPCODE_LTM:
00426         msg_ptr.cfm_ltm = (const struct cfm_ltm_t *)tptr;
00427 
00428         printf(", Flags [%s]",
00429                bittok2str(cfm_ltm_flag_values, "none",  cfm_common_header->flags));
00430 
00431         printf("\n\t  Transaction-ID 0x%08x, Egress-ID %s, ttl %u",
00432                EXTRACT_32BITS(msg_ptr.cfm_ltm->transaction_id),
00433                cfm_egress_id_string(msg_ptr.cfm_ltm->egress_id),
00434                msg_ptr.cfm_ltm->ttl);
00435 
00436         printf("\n\t  Original-MAC %s, Target-MAC %s",
00437                etheraddr_string(msg_ptr.cfm_ltm->original_mac),
00438                etheraddr_string(msg_ptr.cfm_ltm->target_mac));
00439         break;
00440 
00441     case CFM_OPCODE_LTR:
00442         msg_ptr.cfm_ltr = (const struct cfm_ltr_t *)tptr;
00443 
00444         printf(", Flags [%s]",
00445                bittok2str(cfm_ltr_flag_values, "none",  cfm_common_header->flags));
00446 
00447         printf("\n\t  Transaction-ID 0x%08x, Last-Egress-ID %s",
00448                EXTRACT_32BITS(msg_ptr.cfm_ltr->transaction_id),
00449                cfm_egress_id_string(msg_ptr.cfm_ltr->last_egress_id));
00450 
00451         printf("\n\t  Next-Egress-ID %s, ttl %u",
00452                cfm_egress_id_string(msg_ptr.cfm_ltr->next_egress_id),
00453                msg_ptr.cfm_ltr->ttl);
00454 
00455         printf("\n\t  Replay-Action %s (%u)",
00456                tok2str(cfm_ltr_replay_action_values,
00457                        "Unknown",
00458                        msg_ptr.cfm_ltr->replay_action),
00459                msg_ptr.cfm_ltr->replay_action);
00460         break;
00461 
00462         /*
00463          * No message decoder yet.
00464          * Hexdump everything up until the start of the TLVs
00465          */
00466     case CFM_OPCODE_LBR:
00467     case CFM_OPCODE_LBM:
00468     default:
00469         if (tlen > cfm_common_header->first_tlv_offset) {
00470             print_unknown_data(tptr, "\n\t  ",
00471                                tlen -  cfm_common_header->first_tlv_offset);
00472         }
00473         break;
00474     }
00475 
00476     /*
00477      * Sanity check for not walking off.
00478      */
00479     if (tlen <= cfm_common_header->first_tlv_offset) {
00480         return;
00481     }
00482 
00483     tptr += cfm_common_header->first_tlv_offset;
00484     tlen -= cfm_common_header->first_tlv_offset;
00485     
00486     while (tlen > 0) {
00487         cfm_tlv_header = (const struct cfm_tlv_header_t *)tptr;
00488 
00489         /* Enough to read the tlv type ? */
00490         TCHECK2(*tptr, 1);
00491         cfm_tlv_type=cfm_tlv_header->type;
00492 
00493         if (cfm_tlv_type != CFM_TLV_END) {
00494             /* did we capture enough for fully decoding the object header ? */
00495             TCHECK2(*tptr, sizeof(struct cfm_tlv_header_t));            
00496             cfm_tlv_len=EXTRACT_16BITS(&cfm_tlv_header->length);
00497         } else {
00498             cfm_tlv_len = 0;
00499         }
00500 
00501         printf("\n\t%s TLV (0x%02x), length %u",
00502                tok2str(cfm_tlv_values, "Unknown", cfm_tlv_type),
00503                cfm_tlv_type,
00504                cfm_tlv_len);
00505 
00506         /* sanity check for not walking off and infinite loop check. */
00507         if ((cfm_tlv_type != CFM_TLV_END) &&
00508             ((cfm_tlv_len + sizeof(struct cfm_tlv_header_t) > tlen) ||
00509              (!cfm_tlv_len))) {
00510             print_unknown_data(tptr,"\n\t  ",tlen);
00511             return;
00512         }
00513 
00514         tptr += sizeof(struct cfm_tlv_header_t);
00515         tlen -= sizeof(struct cfm_tlv_header_t);
00516         tlv_ptr = tptr;
00517 
00518         /* did we capture enough for fully decoding the object ? */
00519         if (cfm_tlv_type != CFM_TLV_END) {
00520             TCHECK2(*tptr, cfm_tlv_len);
00521         }
00522         hexdump = FALSE;
00523 
00524         switch(cfm_tlv_type) {
00525         case CFM_TLV_END:
00526             /* we are done - bail out */
00527             return;
00528 
00529         case CFM_TLV_PORT_STATUS:
00530             printf(", Status: %s (%u)",
00531                    tok2str(cfm_tlv_port_status_values, "Unknown", *tptr),
00532                    *tptr);
00533             break;
00534 
00535         case CFM_TLV_INTERFACE_STATUS:
00536             printf(", Status: %s (%u)",
00537                    tok2str(cfm_tlv_interface_status_values, "Unknown", *tptr),
00538                    *tptr);
00539             break;
00540 
00541         case CFM_TLV_PRIVATE:
00542             printf(", Vendor: %s (%u), Sub-Type %u",
00543                    tok2str(oui_values,"Unknown", EXTRACT_24BITS(tptr)),
00544                    EXTRACT_24BITS(tptr),
00545                    *(tptr+3));
00546             hexdump = TRUE;
00547             break;
00548 
00549         case CFM_TLV_SENDER_ID:
00550         {
00551             u_int chassis_id_type, chassis_id_length;
00552             u_int mgmt_addr_length;
00553 
00554             /*
00555              * Check if there is a Chassis-ID.
00556              */
00557             chassis_id_length = *tptr;
00558             if (chassis_id_length > tlen) {
00559                 hexdump = TRUE;
00560                 break;
00561             }
00562 
00563             tptr++;
00564             tlen--;
00565 
00566             if (chassis_id_length) {
