00210 static int __devinit igb_validate_option(unsigned int *value,
00211                                          struct igb_option *opt,
00212                                          struct igb_adapter *adapter)
00213 {
00214     if (*value == OPTION_UNSET) {
00215         *value = opt->def;
00216         return 0;
00217     }
00218 
00219     switch (opt->type) {
00220     case enable_option:
00221         switch (*value) {
00222         case OPTION_ENABLED:
00223             DPRINTK(PROBE, INFO, "%s Enabled\n", opt->name);
00224             return 0;
00225         case OPTION_DISABLED:
00226             DPRINTK(PROBE, INFO, "%s Disabled\n", opt->name);
00227             return 0;
00228         }
00229         break;
00230     case range_option:
00231         if (*value >= opt->arg.r.min && *value <= opt->arg.r.max) {
00232             DPRINTK(PROBE, INFO,
00233                     "%s set to %d\n", opt->name, *value);
00234             return 0;
00235         }
00236         break;
00237     case list_option: {
00238         int i;
00239         struct igb_opt_list *ent;
00240 
00241         for (i = 0; i < opt->arg.l.nr; i++) {
00242             ent = &opt->arg.l.p[i];
00243             if (*value == ent->i) {
00244                 if (ent->str[0] != '\0')
00245                     DPRINTK(PROBE, INFO, "%s\n", ent->str);
00246                 return 0;
00247             }
00248         }
00249     }
00250         break;
00251     default:
00252         BUG();
00253     }
00254 
00255     DPRINTK(PROBE, INFO, "Invalid %s value specified (%d) %s\n",
00256            opt->name, *value, opt->err);
00257     *value = opt->def;
00258     return -1;
00259 }
