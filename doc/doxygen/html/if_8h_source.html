00150 /*
00151  * Interface request structure used for socket
00152  * ioctl's.  All interface ioctl's must have parameter
00153  * definitions which begin with ifr_name.  The
00154  * remainder may be interface specific.
00155  */
00156 struct  ifreq {
00157 #define IFNAMSIZ    16
00158     char    ifr_name[IFNAMSIZ];     /* if name, e.g. "en0" */
00159     union {
00160         struct  sockaddr ifru_addr;
00161         struct  sockaddr ifru_dstaddr;
00162         struct  sockaddr ifru_broadaddr;
00163         short   ifru_flags;
00164         int ifru_metric;
00165         int ifru_mtu;
00166         int ifru_phys;
00167         int ifru_media;
00168         caddr_t ifru_data;
00169     } ifr_ifru;
00170 #define ifr_addr    ifr_ifru.ifru_addr  /* address */
00171 #define ifr_dstaddr ifr_ifru.ifru_dstaddr   /* other end of p-to-p link */
00172 #define ifr_broadaddr   ifr_ifru.ifru_broadaddr /* broadcast address */
00173 #define ifr_flags   ifr_ifru.ifru_flags /* flags */
00174 #define ifr_metric  ifr_ifru.ifru_metric    /* metric */
00175 #define ifr_mtu     ifr_ifru.ifru_mtu   /* mtu */
00176 #define ifr_phys    ifr_ifru.ifru_phys  /* physical wire */
00177 #define ifr_media   ifr_ifru.ifru_media /* physical media */
00178 #define ifr_data    ifr_ifru.ifru_data  /* for use by interface */
00179 };
