00103             struct pfring_pkthdr *hdr,
00104             u_int8_t wait_for_incoming_packet) {
00105   u_char *pkt = NULL;

00107 
00108   if(ring->is_shutting_down) return(-1);
00109 
00110   ring->break_recv_loop = 0;
00111   if(ring->reentrant) pthread_spin_lock(&ring->spinlock);
00112 
00113   redo_pfring_recv:
00114     if(ring->is_shutting_down || ring->break_recv_loop) {
00115       if(ring->reentrant) pthread_spin_unlock(&ring->spinlock);
00116       return(-1);
00117     }
00118 
00119     pkt = ring->dna_next_packet(ring, buffer, buffer_len, hdr);
00120 
00121     if(pkt && (hdr->len > 0)) {
00122       /* Set the (1) below to (0) for enabling packet parsing for DNA devices */
00123       if(0)
00124     hdr->extended_hdr.parsed_header_len = 0;
00125       else if(buffer_len > 0)
00126     parse_pkt(*buffer, hdr);
00127 
00128       if(ring->reentrant) pthread_spin_unlock(&ring->spinlock);
00129       return(1);
00130     }
00131 
00132     if(wait_for_incoming_packet) {
00133       status = ring->dna_check_packet_to_read(ring, wait_for_incoming_packet);
00134     }
00135 
00136     if(status > 0)
00137       goto redo_pfring_recv;
00138 
00139     if(ring->reentrant) pthread_spin_unlock(&ring->spinlock);
00140     return(0);
00141  }
