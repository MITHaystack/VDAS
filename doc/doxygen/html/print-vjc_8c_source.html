00042 /*
00043  * XXX - for BSD/OS PPP, what packets get supplied with a PPP header type
00044  * of PPP_VJC and what packets get supplied with a PPP header type of
00045  * PPP_VJNC?  PPP_VJNC is for "UNCOMPRESSED_TCP" packets, and PPP_VJC
00046  * is for COMPRESSED_TCP packets (PPP_IP is used for TYPE_IP packets).
00047  *
00048  * RFC 1144 implies that, on the wire, the packet type is *not* needed
00049  * for PPP, as different PPP protocol types can be used; it only needs
00050  * to be put on the wire for SLIP.
00051  *
00052  * It also indicates that, for compressed SLIP:
00053  *
00054  *  If the COMPRESSED_TCP bit is set in the first byte, it's
00055  *  a COMPRESSED_TCP packet; that byte is the change byte, and
00056  *  the COMPRESSED_TCP bit, 0x80, isn't used in the change byte.
00057  *
00058  *  If the upper 4 bits of the first byte are 7, it's an
00059  *  UNCOMPRESSED_TCP packet; that byte is the first byte of
00060  *  the UNCOMPRESSED_TCP modified IP header, with a connection
00061  *  number in the protocol field, and with the version field
00062  *  being 7, not 4.
00063  *
00064  *  Otherwise, the packet is an IPv4 packet (where the upper 4 bits
00065  *  of the packet are 4).
00066  *
00067  * So this routine looks as if it's sort-of intended to handle
00068  * compressed SLIP, although it doesn't handle UNCOMPRESSED_TCP
00069  * correctly for that (it doesn't fix the version number and doesn't
00070  * do anything to the protocol field), and doesn't check for COMPRESSED_TCP
00071  * packets correctly for that (you only check the first bit - see
00072  * B.1 in RFC 1144).
00073  *
00074  * But it's called for BSD/OS PPP, not SLIP - perhaps BSD/OS does weird
00075  * things with the headers?
00076  *
00077  * Without a BSD/OS VJC-compressed PPP trace, or knowledge of what the
00078  * BSD/OS VJC code does, we can't say what's the case.
00079  *
00080  * We therefore leave "proto" - which is the PPP protocol type - in place,
00081  * *not* marked as unused, for now, so that GCC warnings about the
00082  * unused argument remind us that we should fix this some day.
00083  */
00084 int
00085 vjc_print(register const char *bp, u_short proto _U_)
00086 {
00087     int i;
00088 
00089     switch (bp[0] & 0xf0) {
00090     case TYPE_IP:
00091         if (eflag)
00092             printf("(vjc type=IP) ");
00093         return PPP_IP;
00094     case TYPE_UNCOMPRESSED_TCP:
00095         if (eflag)
00096             printf("(vjc type=raw TCP) ");
00097         return PPP_IP;
00098     case TYPE_COMPRESSED_TCP:
00099         if (eflag)
00100             printf("(vjc type=compressed TCP) ");
00101         for (i = 0; i < 8; i++) {
00102             if (bp[1] & (0x80 >> i))
00103                 printf("%c", "?CI?SAWU"[i]);
00104         }
00105         if (bp[1])
00106             printf(" ");
00107         printf("C=0x%02x ", bp[2]);
00108         printf("sum=0x%04x ", *(u_short *)&bp[3]);
00109         return -1;
00110     case TYPE_ERROR:
00111         if (eflag)
00112             printf("(vjc type=error) ");
00113         return -1;
00114     default:
00115         if (eflag)
00116             printf("(vjc type=0x%02x) ", bp[0] & 0xf0);
00117         return -1;
00118     }
00119 }
