00043 static void
00044 igrp_entry_print(register struct igrprte *igr, register int is_interior,
00045     register int is_exterior)
00046 {
00047     register u_int delay, bandwidth;
00048     u_int metric, mtu;
00049 
00050     if (is_interior)
00051         printf(" *.%d.%d.%d", igr->igr_net[0],
00052             igr->igr_net[1], igr->igr_net[2]);
00053     else if (is_exterior)
00054         printf(" X%d.%d.%d.0", igr->igr_net[0],
00055             igr->igr_net[1], igr->igr_net[2]);
00056     else
00057         printf(" %d.%d.%d.0", igr->igr_net[0],
00058             igr->igr_net[1], igr->igr_net[2]);
00059 
00060     delay = EXTRACT_24BITS(igr->igr_dly);
00061     bandwidth = EXTRACT_24BITS(igr->igr_bw);
00062     metric = bandwidth + delay;
00063     if (metric > 0xffffff)
00064         metric = 0xffffff;
00065     mtu = EXTRACT_16BITS(igr->igr_mtu);
00066 
00067     printf(" d=%d b=%d r=%d l=%d M=%d mtu=%d in %d hops",
00068         10 * delay, bandwidth == 0 ? 0 : 10000000 / bandwidth,
00069         igr->igr_rel, igr->igr_ld, metric,
00070         mtu, igr->igr_hct);
00071 }

00079 void
00080 igrp_print(register const u_char *bp, u_int length, const u_char *bp2 _U_)
00081 {
00082     register struct igrphdr *hdr;
00083     register u_char *cp;
00084     u_int nint, nsys, next;
00085 
00086     hdr = (struct igrphdr *)bp;
00087     cp = (u_char *)(hdr + 1);
00088         (void)printf("igrp:");
00089 
00090     /* Header */
00091     TCHECK(*hdr);
00092     nint = EXTRACT_16BITS(&hdr->ig_ni);
00093     nsys = EXTRACT_16BITS(&hdr->ig_ns);
00094     next = EXTRACT_16BITS(&hdr->ig_nx);
00095 
00096     (void)printf(" %s V%d edit=%d AS=%d (%d/%d/%d)",
00097         tok2str(op2str, "op-#%d", IGRP_OP(hdr->ig_vop)),
00098         IGRP_V(hdr->ig_vop),
00099         hdr->ig_ed,
00100         EXTRACT_16BITS(&hdr->ig_as),
00101         nint,
00102         nsys,
00103         next);
00104 
00105     length -= sizeof(*hdr);
00106     while (length >= IGRP_RTE_SIZE) {
00107         if (nint > 0) {
00108             TCHECK2(*cp, IGRP_RTE_SIZE);
00109             igrp_entry_print((struct igrprte *)cp, 1, 0);
00110             --nint;
00111         } else if (nsys > 0) {
00112             TCHECK2(*cp, IGRP_RTE_SIZE);
00113             igrp_entry_print((struct igrprte *)cp, 0, 0);
00114             --nsys;
00115         } else if (next > 0) {
00116             TCHECK2(*cp, IGRP_RTE_SIZE);
00117             igrp_entry_print((struct igrprte *)cp, 0, 1);
00118             --next;
00119         } else {
00120             (void)printf(" [extra bytes %d]", length);
00121             break;
00122         }
00123         cp += IGRP_RTE_SIZE;
00124         length -= IGRP_RTE_SIZE;
00125     }
00126     if (nint == 0 && nsys == 0 && next == 0)
00127         return;
00128 trunc:
00129     fputs(" [|igrp]", stdout);
00130 }
