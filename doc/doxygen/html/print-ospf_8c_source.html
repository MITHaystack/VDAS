00205     while (ls_length > 0) {
00206         TCHECK2(*tptr, 4);
00207         if (ls_length < 4) {
00208             printf("\n\t    Remaining LS length %u < 4", ls_length);

00224             return -1;

00226 
00227         /* Infinite loop protection. */
00228         if (tlv_type == 0 || tlv_length ==0) {

00487         switch (lshp->ls_type) {

00491         case LS_TYPE_OPAQUE_DW:

00498                *(&lshp->un_lsa_id.opaque_field.opaque_type)),

00501

00516

00834 static int
00835 ospf_decode_lls(register const struct ospfhdr *op,
00836         register u_int length)
00837 {
00838     register const u_char *dptr;
00839     register const u_char *dataend;
00840     register u_int length2;
00841     register u_int16_t lls_type, lls_len;
00842     register u_int32_t lls_flags;
00843 
00844     switch (op->ospf_type) {
00845 
00846     case OSPF_TYPE_HELLO:
00847         if (!(op->ospf_hello.hello_options & OSPF_OPTION_L))
00848             return (0);
00849         break;
00850 
00851     case OSPF_TYPE_DD:
00852         if (!(op->ospf_db.db_options & OSPF_OPTION_L))
00853             return (0);
00854         break;
00855 
00856     default:
00857         return (0);
00858     }
00859 
00860     /* dig deeper if LLS data is available; see RFC4813 */
00861     length2 = EXTRACT_16BITS(&op->ospf_len);
00862     dptr = (u_char *)op + length2;
00863     dataend = (u_char *)op + length;
00864 
00865     if (EXTRACT_16BITS(&op->ospf_authtype) == OSPF_AUTH_MD5) {
00866         dptr = dptr + op->ospf_authdata[3];
00867         length2 += op->ospf_authdata[3];
00868     }
00869     if (length2 >= length) {
00870         printf("\n\t[LLS truncated]");
00871         return (1);
00872     }
00873     TCHECK2(*dptr, 2);
00874     printf("\n\t  LLS: checksum: 0x%04x", (u_int)EXTRACT_16BITS(dptr));
00875 
00876     dptr += 2;
00877     TCHECK2(*dptr, 2);
00878     length2 = EXTRACT_16BITS(dptr);
00879     printf(", length: %u", length2);
00880 
00881     dptr += 2;
00882     TCHECK(*dptr);
00883     while (dptr < dataend) {
00884         TCHECK2(*dptr, 2);
00885         lls_type = EXTRACT_16BITS(dptr);
00886         printf("\n\t    %s (%u)",
00887                tok2str(ospf_lls_tlv_values,"Unknown TLV",lls_type),
00888                lls_type);
00889         dptr += 2;
00890         TCHECK2(*dptr, 2);
00891         lls_len = EXTRACT_16BITS(dptr);
00892         printf(", length: %u", lls_len);
00893         dptr += 2;
00894         switch (lls_type) {
00895 
00896         case OSPF_LLS_EO:
00897             if (lls_len != 4) {
00898                 printf(" [should be 4]");
00899                 lls_len = 4;
00900             }
00901             TCHECK2(*dptr, 4);
00902             lls_flags = EXTRACT_32BITS(dptr);
00903             printf("\n\t      Options: 0x%08x [%s]", lls_flags,
00904                    bittok2str(ospf_lls_eo_options,"?",lls_flags));
00905 
00906             break;
00907 
00908         case OSPF_LLS_MD5:
00909             if (lls_len != 20) {
00910                 printf(" [should be 20]");
00911                 lls_len = 20;
00912             }
00913             TCHECK2(*dptr, 4);
00914             printf("\n\t      Sequence number: 0x%08x", EXTRACT_32BITS(dptr));
00915             break;
00916         }
00917 
00918         dptr += lls_len;
00919     }
00920 
00921     return (0);
00922 trunc:
00923     return (1);
00924 }

01032                 for (lsa_count=1;lsa_count <= lsa_count_max;lsa_count++) {

01047     default:

01049     }
01050     return (0);
01051 trunc:

01055 void
01056 ospf_print(register const u_char *bp, register u_int length,
01057     const u_char *bp2 _U_)
01058 {
01059     register const struct ospfhdr *op;
01060     register const u_char *dataend;
01061     register const char *cp;
01062 
01063     op = (struct ospfhdr *)bp;
01064 
01065         /* XXX Before we do anything else, strip off the MD5 trailer */
01066         TCHECK(op->ospf_authtype);
01067         if (EXTRACT_16BITS(&op->ospf_authtype) == OSPF_AUTH_MD5) {
01068                 length -= OSPF_AUTH_MD5_LEN;
01069                 snapend -= OSPF_AUTH_MD5_LEN;
01070         }
01071 
01072     /* If the type is valid translate it, or just print the type */
01073     /* value.  If it's not valid, say so and return */
01074     TCHECK(op->ospf_type);
01075     cp = tok2str(type2str, "unknown LS-type", op->ospf_type);
01076     printf("OSPFv%u, %s, length %u",
01077            op->ospf_version,
01078            cp,
01079            length);
01080     if (*cp == 'u')
01081         return;
01082 
01083         if(!vflag) { /* non verbose - so lets bail out here */
01084                 return;
01085         }
01086 
01087     TCHECK(op->ospf_len);
01088     if (length != EXTRACT_16BITS(&op->ospf_len)) {
01089         printf(" [len %d]", EXTRACT_16BITS(&op->ospf_len));
01090     }
01091 
01092     if (length > EXTRACT_16BITS(&op->ospf_len)) {
01093         dataend = bp + EXTRACT_16BITS(&op->ospf_len);
01094     } else {
01095         dataend = bp + length;
01096     }
