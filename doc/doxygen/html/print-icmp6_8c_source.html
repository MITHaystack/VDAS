00197 static int icmp6_cksum(const struct ip6_hdr *ip6, const struct icmp6_hdr *icp,
00198     u_int len)
00199 {
00200     size_t i;
00201     register const u_int16_t *sp;
00202     u_int32_t sum;
00203     union {
00204         struct {
00205             struct in6_addr ph_src;
00206             struct in6_addr ph_dst;
00207             u_int32_t   ph_len;
00208             u_int8_t    ph_zero[3];
00209             u_int8_t    ph_nxt;
00210         } ph;
00211         u_int16_t pa[20];
00212     } phu;
00213 
00214     /* pseudo-header */
00215     memset(&phu, 0, sizeof(phu));
00216     phu.ph.ph_src = ip6->ip6_src;
00217     phu.ph.ph_dst = ip6->ip6_dst;
00218     phu.ph.ph_len = htonl(len);
00219     phu.ph.ph_nxt = IPPROTO_ICMPV6;
00220 
00221     sum = 0;
00222     for (i = 0; i < sizeof(phu.pa) / sizeof(phu.pa[0]); i++)
00223         sum += phu.pa[i];
00224 
00225     sp = (const u_int16_t *)icp;
00226 
00227     for (i = 0; i < (len & ~1); i += 2)
00228         sum += *sp++;
00229 
00230     if (len & 1)
00231         sum += htons((*(const u_int8_t *)sp) << 8);
00232 
00233     while (sum > 0xffff)
00234         sum = (sum & 0xffff) + (sum >> 16);
00235     sum = ~sum & 0xffff;
00236 
00237     return (sum);
00238 }

00329 
00330     dp = (struct icmp6_hdr *)bp;
00331     ip = (struct ip6_hdr *)bp2;

00333     /* 'ep' points to the end of available data. */
00334     ep = snapend;
00335 
00336     TCHECK(dp->icmp6_cksum);
00337 
00338     if (vflag && !fragmented) {
00339         int sum = dp->icmp6_cksum;
00340 
00341         if (TTEST2(bp[0], length)) {
00342             sum = icmp6_cksum(ip, dp, length);
00343             if (sum != 0)
00344                 (void)printf("[bad icmp6 cksum %x!] ", sum);
00345             else
00346                 (void)printf("[icmp6 sum ok] ");
00347         }
00348     }
00349 
00350         printf("ICMP6, %s", tok2str(icmp6_type_values,"unknown icmp6 type (%u)",dp->icmp6_type));
00351 
00352         /* display cosmetics: print the packet length for printer that use the vflag now */
00353         if (vflag && (dp->icmp6_type ==
00354                       ND_ROUTER_SOLICIT ||
00355                       ND_ROUTER_ADVERT ||
00356                       ND_NEIGHBOR_ADVERT ||
00357                       ND_NEIGHBOR_SOLICIT ||
00358                       ND_REDIRECT ||
00359                       ICMP6_HADISCOV_REPLY ||
00360                       ICMP6_MOBILEPREFIX_ADVERT ))
00361             printf(", length %u", length);
00362                       
00363     switch (dp->icmp6_type) {
00364     case ICMP6_DST_UNREACH:
00365         TCHECK(oip->ip6_dst);
00366                 printf(", %s", tok2str(icmp6_dst_unreach_code_values,"unknown unreach code (%u)",dp->icmp6_code));
00367         switch (dp->icmp6_code) {
00368 
00369         case ICMP6_DST_UNREACH_NOROUTE: /* fall through */
00370         case ICMP6_DST_UNREACH_ADMIN:
00371         case ICMP6_DST_UNREACH_ADDR:
00372                         printf(" %s",ip6addr_string(&oip->ip6_dst));
00373                         break;

00375             printf(" %s, source address %s",
00376                    ip6addr_string(&oip->ip6_dst),
00377                    ip6addr_string(&oip->ip6_src));
00378             break;
00379         case ICMP6_DST_UNREACH_NOPORT:
00380             if ((ouh = get_upperlayer((u_char *)oip, &prot))
00381                 == NULL)
00382                 goto trunc;

00608 static struct udphdr *
00609 get_upperlayer(u_char *bp, u_int *prot)
00610 {
00611     const u_char *ep;
00612     struct ip6_hdr *ip6 = (struct ip6_hdr *)bp;
00613     struct udphdr *uh;
00614     struct ip6_hbh *hbh;
00615     struct ip6_frag *fragh;
00616     struct ah *ah;
00617     u_int nh;
00618     int hlen;
00619 
00620     /* 'ep' points to the end of available data. */
00621     ep = snapend;
00622 
00623     if (!TTEST(ip6->ip6_nxt))
00624         return NULL;
00625 
00626     nh = ip6->ip6_nxt;
00627     hlen = sizeof(struct ip6_hdr);
00628 
00629     while (bp < ep) {
00630         bp += hlen;
00631 
00632         switch(nh) {
00633         case IPPROTO_UDP:
00634         case IPPROTO_TCP:
00635             uh = (struct udphdr *)bp;
00636             if (TTEST(uh->uh_dport)) {
00637                 *prot = nh;
00638                 return(uh);
00639             }
00640             else
00641                 return(NULL);
00642             /* NOTREACHED */
00643 
00644         case IPPROTO_HOPOPTS:
00645         case IPPROTO_DSTOPTS:
00646         case IPPROTO_ROUTING:
00647             hbh = (struct ip6_hbh *)bp;
00648             if (!TTEST(hbh->ip6h_len))
00649                 return(NULL);
00650             nh = hbh->ip6h_nxt;
00651             hlen = (hbh->ip6h_len + 1) << 3;
00652             break;
00653 
00654         case IPPROTO_FRAGMENT: /* this should be odd, but try anyway */
00655             fragh = (struct ip6_frag *)bp;
00656             if (!TTEST(fragh->ip6f_offlg))
00657                 return(NULL);
00658             /* fragments with non-zero offset are meaningless */
00659             if ((EXTRACT_16BITS(&fragh->ip6f_offlg) & IP6F_OFF_MASK) != 0)
00660                 return(NULL);
00661             nh = fragh->ip6f_nxt;
00662             hlen = sizeof(struct ip6_frag);
00663             break;
00664 
00665         case IPPROTO_AH:
00666             ah = (struct ah *)bp;
00667             if (!TTEST(ah->ah_len))
00668                 return(NULL);
00669             nh = ah->ah_nxt;
00670             hlen = (ah->ah_len + 2) << 2;
00671             break;
00672 
00673         default:    /* unknown or undecodable header */
00674             *prot = nh; /* meaningless, but set here anyway */
00675             return(NULL);
00676         }
00677     }
00678 
00679     return(NULL);       /* should be notreached, though */
00680 }
