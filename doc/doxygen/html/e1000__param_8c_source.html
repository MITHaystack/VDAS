00252 static int __devinit e1000_validate_option(unsigned int *value,
00253                                            const struct e1000_option *opt,
00254                                            struct e1000_adapter *adapter)
00255 {
00256     if (*value == OPTION_UNSET) {
00257         *value = opt->def;
00258         return 0;
00259     }
00260 
00261     switch (opt->type) {
00262     case enable_option:
00263         switch (*value) {
00264         case OPTION_ENABLED:
00265             DPRINTK(PROBE, INFO, "%s Enabled\n", opt->name);
00266             return 0;
00267         case OPTION_DISABLED:
00268             DPRINTK(PROBE, INFO, "%s Disabled\n", opt->name);
00269             return 0;
00270         }
00271         break;
00272     case range_option:
00273         if (*value >= opt->arg.r.min && *value <= opt->arg.r.max) {
00274             DPRINTK(PROBE, INFO,
00275                     "%s set to %i\n", opt->name, *value);
00276             return 0;
00277         }
00278         break;
00279     case list_option: {
00280         int i;
00281         struct e1000_opt_list *ent;
00282 
00283         for (i = 0; i < opt->arg.l.nr; i++) {
00284             ent = &opt->arg.l.p[i];
00285             if (*value == ent->i) {
00286                 if (ent->str[0] != '\0')
00287                     DPRINTK(PROBE, INFO, "%s\n", ent->str);
00288                 return 0;
00289             }
00290         }
00291     }
00292         break;
00293     default:
00294         BUG();
00295     }
00296 
00297     DPRINTK(PROBE, INFO, "Invalid %s value specified (%i) %s\n",
00298            opt->name, *value, opt->err);
00299     *value = opt->def;
00300     return -1;
00301 }
