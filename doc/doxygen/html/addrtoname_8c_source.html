00173 intoa(u_int32_t addr)
00174 {
00175     register char *cp;
00176     register u_int byte;
00177     register int n;
00178     static char buf[sizeof(".xxx.xxx.xxx.xxx")];
00179 
00180     NTOHL(addr);
00181     cp = buf + sizeof(buf);
00182     *--cp = '\0';
00183

00185     do {
00186         byte = addr & 0xff;
00187         *--cp = byte % 10 + '0';
00188         byte /= 10;
00189         if (byte > 0) {
00190             *--cp = byte % 10 + '0';
00191             byte /= 10;
00192             if (byte > 0)
00193                 *--cp = byte + '0';
00194         }
00195         *--cp = '.';
00196         addr >>= 8;
00197     } while (--n > 0);
00198 
00199     return cp + 1;

00205 /*
00206  * Return a name for the IP address pointed to by ap.  This address
00207  * is assumed to be in network byte order.
00208  *
00209  * NOTE: ap is *NOT* necessarily part of the packet data (not even if
00210  * this is being called with the "ipaddr_string()" macro), so you
00211  * *CANNOT* use the TCHECK{2}/TTEST{2} macros on it.  Furthermore,
00212  * even in cases where it *is* part of the packet data, the caller
00213  * would still have to check for a null return value, even if it's
00214  * just printing the return value with "%s" - not all versions of
00215  * printf print "(null)" with "%s" and a null pointer, some of them
00216  * don't check for a null pointer and crash in that case.
00217  *
00218  * The callers of this routine should, before handing this routine
00219  * a pointer to packet data, be sure that the data is present in
00220  * the packet buffer.  They should probably do those checks anyway,
00221  * as other data at that layer might not be IP addresses, and it
00222  * also needs to check whether they're present in the packet buffer.
00223  */
00224 const char *
00225 getname(const u_char *ap)
00226 {
00227     register struct hostent *hp;
00228     u_int32_t addr;
00229     static struct hnamemem *p;      /* static for longjmp() */
00230 
00231     memcpy(&addr, ap, sizeof(addr));
00232     p = &hnametable[addr & (HASHNAMESIZE-1)];
00233     for (; p->nxt; p = p->nxt) {
00234         if (p->addr == addr)
00235             return (p->name);
00236     }
00237     p->addr = addr;
00238     p->nxt = newhnamemem();
00239 
00240     /*
00241      * Print names unless:
00242      *  (1) -n was given.
00243      *      (2) Address is foreign and -f was given. (If -f was not
00244      *      given, f_netmask and f_localnet are 0 and the test
00245      *      evaluates to true)
00246      */
00247     if (!nflag &&
00248         (addr & f_netmask) == f_localnet) {
00249         hp = gethostbyaddr((char *)&addr, 4, AF_INET);
00250         if (hp) {
00251             char *dotp;
00252 
00253             p->name = strdup(hp->h_name);
00254             if (Nflag) {
00255                 /* Remove domain qualifications */
00256                 dotp = strchr(p->name, '.');
00257                 if (dotp)
00258                     *dotp = '\0';
00259             }
00260             return (p->name);
00261         }
00262     }
00263     p->name = strdup(intoa(addr));
00264     return (p->name);
00265 }

00319 static inline struct enamemem *
00320 lookup_emem(const u_char *ep)
00321 {
00322     register u_int i, j, k;
00323     struct enamemem *tp;
00324 
00325     k = (ep[0] << 8) | ep[1];
00326     j = (ep[2] << 8) | ep[3];
00327     i = (ep[4] << 8) | ep[5];
00328 
00329     tp = &enametable[(i ^ j) & (HASHNAMESIZE-1)];
00330     while (tp->e_nxt)
00331         if (tp->e_addr0 == i &&
00332             tp->e_addr1 == j &&
00333             tp->e_addr2 == k)
00334             return tp;
00335         else
00336             tp = tp->e_nxt;
00337     tp->e_addr0 = i;
00338     tp->e_addr1 = j;
00339     tp->e_addr2 = k;
00340     tp->e_nxt = (struct enamemem *)calloc(1, sizeof(*tp));
00341     if (tp->e_nxt == NULL)
00342         error("lookup_emem: calloc");
00343 
00344     return tp;
00345 }

00353 lookup_bytestring(register const u_char *bs, const unsigned int nlen)
00354 {
00355     struct enamemem *tp;
00356     register u_int i, j, k;
00357 
00358     if (nlen >= 6) {

00360         j = (bs[2] << 8) | bs[3];
00361         i = (bs[4] << 8) | bs[5];
00362     } else if (nlen >= 4) {
00363         k = (bs[0] << 8) | bs[1];
00364         j = (bs[2] << 8) | bs[3];

00366     } else

00368

00370     while (tp->e_nxt)
00371         if (tp->e_addr0 == i &&
00372             tp->e_addr1 == j &&
00373             tp->e_addr2 == k &&
00374             memcmp((const char *)bs, (const char *)(tp->e_bs), nlen) == 0)
00375             return tp;
00376         else
00377             tp = tp->e_nxt;
00378 
00379     tp->e_addr0 = i;
00380     tp->e_addr1 = j;
00381     tp->e_addr2 = k;
00382 
00383     tp->e_bs = (u_char *) calloc(1, nlen + 1);
00384     memcpy(tp->e_bs, bs, nlen);

00804 /*
00805  * Initialize the ethers hash table.  We take two different approaches
00806  * depending on whether or not the system provides the ethers name
00807  * service.  If it does, we just wire in a few names at startup,
00808  * and etheraddr_string() fills in the table on demand.  If it doesn't,
00809  * then we suck in the entire /etc/ethers file at startup.  The idea
00810  * is that parsing the local file will be fast, but spinning through
00811  * all the ethers entries via NIS & next_etherent might be very slow.
00812  *
00813  * XXX pcap_next_etherent doesn't belong in the pcap interface, but
00814  * since the pcap module already does name-to-address translation,
00815  * it's already does most of the work for the ethernet address-to-name
00816  * translation, so we just pcap_next_etherent as a convenience.
00817  */
00818 static void
00819 init_etherarray(void)
00820 {
00821     register const struct etherlist *el;
00822     register struct enamemem *tp;
00823 #ifdef USE_ETHER_NTOHOST
00824     char name[256];
00825 #else
00826     register struct pcap_etherent *ep;
00827     register FILE *fp;
00828 
00829     /* Suck in entire ethers file */
00830     fp = fopen(PCAP_ETHERS_FILE, "r");
00831     if (fp != NULL) {
00832         while ((ep = pcap_next_etherent(fp)) != NULL) {
00833             tp = lookup_emem(ep->addr);
00834             tp->e_name = strdup(ep->name);
00835         }
00836         (void)fclose(fp);
00837     }
00838 #endif
00839 
00840     /* Hardwire some ethernet names */
00841     for (el = etherlist; el->name != NULL; ++el) {
00842         tp = lookup_emem(el->addr);
00843         /* Don't override existing name */
00844         if (tp->e_name != NULL)
00845             continue;
00846 
00847 #ifdef USE_ETHER_NTOHOST
00848         /*
00849          * Use YP/NIS version of name if available.
00850          *
00851          * We don't cast it to "const struct ether_addr *"
00852          * because some systems don't modify the Ethernet
00853          * address but fail to declare the second argument
00854          * as a "const" pointer.
00855          */
00856         if (ether_ntohost(name, (struct ether_addr *)el->addr) == 0) {
00857             tp->e_name = strdup(name);
00858             continue;
00859         }
00860 #endif
00861         tp->e_name = el->name;
00862     }
00863 }
