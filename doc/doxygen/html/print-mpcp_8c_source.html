00177                grant_numbers,

00179                           "?",

00181         tptr++;
00182 
00183         for (grant = 1; grant <= grant_numbers; grant++) {
00184             if (!TTEST2(*tptr, sizeof(const struct mpcp_grant_t)))
00185                 goto trunc;
00186             mpcp.grant = (const struct mpcp_grant_t *)tptr;        
00187             printf("\n\tGrant #%u, Start-Time %u ticks, duration %u ticks",

00190                    EXTRACT_16BITS(mpcp.grant->duration));

00197         break;
00198 
00199 
00200     case MPCP_OPCODE_REPORT:
00201     if (!TTEST2(*tptr, MPCP_REPORT_QUEUESETS_LEN))
00202         goto trunc;

00204         tptr+=MPCP_REPORT_QUEUESETS_LEN;
00205         printf("\n\tTotal Queue-Sets %u", queue_sets);
00206 
00207         for (queue_set = 1; queue_set < queue_sets; queue_set++) {
00208             if (!TTEST2(*tptr, MPCP_REPORT_REPORTBITMAP_LEN))
00209                 goto trunc;
00210             report_bitmap = *(tptr);
00211             printf("\n\t  Queue-Set #%u, Report-Bitmap [ %s ]",
00212                    queue_sets,
00213                    bittok2str(mpcp_report_bitmap_values, "Unknown", report_bitmap));
00214             tptr++;
00215 
00216             report=1;
00217             while (report_bitmap != 0) { 
00218                 if (report_bitmap & 1) {
00219                     if (!TTEST2(*tptr, MPCP_TIMESTAMP_DURATION_LEN))
00220                         goto trunc;
00221                     printf("\n\t    Q%u Report, Duration %u ticks",

00228             }
00229         }
00230         break;

00234             goto trunc;

00245         printf("\n\tAssigned-Port %u, Flags [ %s ]" \

00247                EXTRACT_16BITS(mpcp.reg->assigned_port),
00248                bittok2str(mpcp_reg_flag_values, "Reserved", mpcp.reg->flags),
00249                EXTRACT_16BITS(mpcp.reg->sync_time),
00250                mpcp.reg->echoed_pending_grants);
00251         break;
00252 
00253     case MPCP_OPCODE_REG_ACK:

00259                EXTRACT_16BITS(mpcp.reg_ack->echoed_assigned_port),
