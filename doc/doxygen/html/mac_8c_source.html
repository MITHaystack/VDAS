00280 void e1000e_rar_set(struct e1000_hw *hw, u8 *addr, u32 index)
00281 {
00282     u32 rar_low, rar_high;
00283 
00284     /*
00285      * HW expects these in little endian so we reverse the byte order
00286      * from network order (big endian) to little endian
00287      */
00288     rar_low = ((u32) addr[0] |
00289                ((u32) addr[1] << 8) |
00290                ((u32) addr[2] << 16) | ((u32) addr[3] << 24));
00291 
00292     rar_high = ((u32) addr[4] | ((u32) addr[5] << 8));
00293 
00294     /* If MAC address zero, no need to set the AV bit */
00295     if (rar_low || rar_high)
00296         rar_high |= E1000_RAH_AV;
00297 
00298     /*
00299      * Some bridges will combine consecutive 32-bit writes into
00300      * a single burst write, which will malfunction on some parts.
00301      * The flushes avoid this.
00302      */
00303     ew32(RAL(index), rar_low);
00304     e1e_flush();
00305     ew32(RAH(index), rar_high);
00306     e1e_flush();
00307 }

00352 static u32 e1000_hash_mc_addr(struct e1000_hw *hw, u8 *mc_addr)
00353 {
00354     u32 hash_value, hash_mask;
00355     u8 bit_shift = 0;
00356 
00357     /* Register count multiplied by bits per register */
00358     hash_mask = (hw->mac.mta_reg_count * 32) - 1;
00359 
00360     /*
00361      * For a mc_filter_type of 0, bit_shift is the number of left-shifts
00362      * where 0xFF would still fall within the hash mask.
00363      */
00364     while (hash_mask >> bit_shift != 0xFF)
00365         bit_shift++;
00366 
00367     /*
00368      * The portion of the address that is used for the hash table
00369      * is determined by the mc_filter_type setting.
00370      * The algorithm is such that there is a total of 8 bits of shifting.
00371      * The bit_shift for a mc_filter_type of 0 represents the number of
00372      * left-shifts where the MSB of mc_addr[5] would still fall within
00373      * the hash_mask.  Case 0 does this exactly.  Since there are a total
00374      * of 8 bits of shifting, then mc_addr[4] will shift right the
00375      * remaining number of bits. Thus 8 - bit_shift.  The rest of the
00376      * cases are a variation of this algorithm...essentially raising the
00377      * number of bits to shift mc_addr[5] left, while still keeping the
00378      * 8-bit shifting total.
00379      *
00380      * For example, given the following Destination MAC Address and an
00381      * mta register count of 128 (thus a 4096-bit vector and 0xFFF mask),
00382      * we can see that the bit_shift for case 0 is 4.  These are the hash
00383      * values resulting from each mc_filter_type...
00384      * [0] [1] [2] [3] [4] [5]
00385      * 01  AA  00  12  34  56
00386      * LSB                 MSB
00387      *
00388      * case 0: hash_value = ((0x34 >> 4) | (0x56 << 4)) & 0xFFF = 0x563
00389      * case 1: hash_value = ((0x34 >> 3) | (0x56 << 5)) & 0xFFF = 0xAC6
00390      * case 2: hash_value = ((0x34 >> 2) | (0x56 << 6)) & 0xFFF = 0x163
00391      * case 3: hash_value = ((0x34 >> 0) | (0x56 << 8)) & 0xFFF = 0x634
00392      */
00393     switch (hw->mac.mc_filter_type) {
00394     default:
00395     case 0:
00396         break;
00397     case 1:
00398         bit_shift += 1;
00399         break;
00400     case 2:
00401         bit_shift += 2;
00402         break;
00403     case 3:
00404         bit_shift += 4;
00405         break;
00406     }
00407 
00408     hash_value = hash_mask & (((mc_addr[4] >> (8 - bit_shift)) |
00409                               (((u16) mc_addr[5]) << bit_shift)));
00410 
00411     return hash_value;
00412 }

00542 s32 e1000e_check_for_fiber_link(struct e1000_hw *hw)
00543 {
00544     struct e1000_mac_info *mac = &hw->mac;
00545     u32 rxcw;
00546     u32 ctrl;
00547     u32 status;
00548     s32 ret_val = E1000_SUCCESS;
00549 
00550     ctrl = er32(CTRL);
00551     status = er32(STATUS);
00552     rxcw = er32(RXCW);
00553 
00554     /*
00555      * If we don't have link (auto-negotiation failed or link partner
00556      * cannot auto-negotiate), the cable is plugged in (we have signal),
00557      * and our link partner is not trying to auto-negotiate with us (we
00558      * are receiving idles or data), we need to force link up. We also
00559      * need to give auto-negotiation time to complete, in case the cable
00560      * was just plugged in. The autoneg_failed flag does this.
00561      */
00562     /* (ctrl & E1000_CTRL_SWDPIN1) == 1 == have signal */
00563     if ((ctrl & E1000_CTRL_SWDPIN1) && (!(status & E1000_STATUS_LU)) &&
00564         (!(rxcw & E1000_RXCW_C))) {
00565         if (mac->autoneg_failed == 0) {
00566             mac->autoneg_failed = 1;
00567             goto out;
00568         }
00569         e_dbg("NOT Rx'ing /C/, disable AutoNeg and force link.\n");
00570 
00571         /* Disable auto-negotiation in the TXCW register */
00572         ew32(TXCW, (mac->txcw & ~E1000_TXCW_ANE));
00573 
00574         /* Force link-up and also force full-duplex. */
00575         ctrl = er32(CTRL);
00576         ctrl |= (E1000_CTRL_SLU | E1000_CTRL_FD);
00577         ew32(CTRL, ctrl);
00578 
00579         /* Configure Flow Control after forcing link up. */
00580         ret_val = e1000e_config_fc_after_link_up(hw);
00581         if (ret_val) {
00582             e_dbg("Error configuring flow control\n");
00583             goto out;
00584         }
00585     } else if ((ctrl & E1000_CTRL_SLU) && (rxcw & E1000_RXCW_C)) {
00586         /*
00587          * If we are forcing link and we are receiving /C/ ordered
00588          * sets, re-enable auto-negotiation in the TXCW register
00589          * and disable forced link in the Device Control register
00590          * in an attempt to auto-negotiate with our link partner.
00591          */
00592         e_dbg("Rx'ing /C/, enable AutoNeg and stop forcing link.\n");
00593         ew32(TXCW, mac->txcw);
00594         ew32(CTRL, (ctrl & ~E1000_CTRL_SLU));
00595 
00596         mac->serdes_has_link = true;
00597     }
00598 
00599 out:
00600     return ret_val;

00907 static s32 e1000_commit_fc_settings_generic(struct e1000_hw *hw)
00908 {
00909     struct e1000_mac_info *mac = &hw->mac;
00910     u32 txcw;
00911     s32 ret_val = E1000_SUCCESS;
00912 
00913     /*
00914      * Check for a software override of the flow control settings, and
00915      * setup the device accordingly.  If auto-negotiation is enabled, then
00916      * software will have to set the "PAUSE" bits to the correct value in
00917      * the Transmit Config Word Register (TXCW) and re-start auto-
00918      * negotiation.  However, if auto-negotiation is disabled, then
00919      * software will have to manually configure the two flow control enable
00920      * bits in the CTRL register.
00921      *
00922      * The possible values of the "fc" parameter are:
00923      *      0:  Flow control is completely disabled
00924      *      1:  Rx flow control is enabled (we can receive pause frames,
00925      *          but not send pause frames).
00926      *      2:  Tx flow control is enabled (we can send pause frames but we
00927      *          do not support receiving pause frames).
00928      *      3:  Both Rx and Tx flow control (symmetric) are enabled.
00929      */
00930     switch (hw->fc.current_mode) {
00931     case e1000_fc_none:
00932         /* Flow control completely disabled by a software over-ride. */
00933         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD);
00934         break;
00935     case e1000_fc_rx_pause:
00936         /*
00937          * Rx Flow control is enabled and Tx Flow control is disabled
00938          * by a software over-ride. Since there really isn't a way to
00939          * advertise that we are capable of Rx Pause ONLY, we will
00940          * advertise that we support both symmetric and asymmetric Rx
00941          * PAUSE.  Later, we will disable the adapter's ability to send
00942          * PAUSE frames.
00943          */
00944         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_PAUSE_MASK);
00945         break;
00946     case e1000_fc_tx_pause:
00947         /*
00948          * Tx Flow control is enabled, and Rx Flow control is disabled,
00949          * by a software over-ride.
00950          */
00951         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_ASM_DIR);
00952         break;
00953     case e1000_fc_full:
00954         /*
00955          * Flow control (both Rx and Tx) is enabled by a software
00956          * over-ride.
00957          */
00958         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_PAUSE_MASK);
00959         break;
00960     default:
00961         e_dbg("Flow control param set incorrectly\n");
00962         ret_val = -E1000_ERR_CONFIG;
00963         goto out;
00964         break;
00965     }
00966 
00967     ew32(TXCW, txcw);
00968     mac->txcw = txcw;
00969 
00970 out:
00971     return ret_val;
00972 }

01061 s32 e1000e_force_mac_fc(struct e1000_hw *hw)
01062 {
01063     u32 ctrl;
01064     s32 ret_val = E1000_SUCCESS;
01065 
01066     ctrl = er32(CTRL);
01067 
01068     /*
01069      * Because we didn't get link via the internal auto-negotiation
01070      * mechanism (we either forced link or we got link via PHY
01071      * auto-neg), we have to manually enable/disable transmit an
01072      * receive flow control.
01073      *
01074      * The "Case" statement below enables/disable flow control
01075      * according to the "hw->fc.current_mode" parameter.
01076      *
01077      * The possible values of the "fc" parameter are:
01078      *      0:  Flow control is completely disabled
01079      *      1:  Rx flow control is enabled (we can receive pause
01080      *          frames but not send pause frames).
01081      *      2:  Tx flow control is enabled (we can send pause frames
01082      *          frames but we do not receive pause frames).
01083      *      3:  Both Rx and Tx flow control (symmetric) is enabled.
01084      *  other:  No other values should be possible at this point.
01085      */
01086     e_dbg("hw->fc.current_mode = %u\n", hw->fc.current_mode);
01087 
01088     switch (hw->fc.current_mode) {
01089     case e1000_fc_none:
01090         ctrl &= (~(E1000_CTRL_TFCE | E1000_CTRL_RFCE));
01091         break;
01092     case e1000_fc_rx_pause:
01093         ctrl &= (~E1000_CTRL_TFCE);
01094         ctrl |= E1000_CTRL_RFCE;
01095         break;
01096     case e1000_fc_tx_pause:
01097         ctrl &= (~E1000_CTRL_RFCE);
01098         ctrl |= E1000_CTRL_TFCE;
01099         break;
01100     case e1000_fc_full:
01101         ctrl |= (E1000_CTRL_TFCE | E1000_CTRL_RFCE);
01102         break;
01103     default:
01104         e_dbg("Flow control param set incorrectly\n");
01105         ret_val = -E1000_ERR_CONFIG;
01106         goto out;
01107     }
01108 
01109     ew32(CTRL, ctrl);
01110 
01111 out:
01112     return ret_val;
01113 }

01125 s32 e1000e_config_fc_after_link_up(struct e1000_hw *hw)
01126 {
01127     struct e1000_mac_info *mac = &hw->mac;
01128     s32 ret_val = E1000_SUCCESS;
01129     u16 mii_status_reg, mii_nway_adv_reg, mii_nway_lp_ability_reg;
01130     u16 speed, duplex;
01131 
01132     /*
01133      * Check for the case where we have fiber media and auto-neg failed
01134      * so we had to force link.  In this case, we need to force the
01135      * configuration of the MAC to match the "fc" parameter.
01136      */
01137     if (mac->autoneg_failed) {
01138         if (hw->phy.media_type == e1000_media_type_fiber ||
01139             hw->phy.media_type == e1000_media_type_internal_serdes)
01140             ret_val = e1000e_force_mac_fc(hw);
01141     } else {
01142         if (hw->phy.media_type == e1000_media_type_copper)
01143             ret_val = e1000e_force_mac_fc(hw);
01144     }
01145 
01146     if (ret_val) {
01147         e_dbg("Error forcing flow control settings\n");
01148         goto out;
01149     }
01150 
01151     /*
01152      * Check for the case where we have copper media and auto-neg is
01153      * enabled.  In this case, we need to check and see if Auto-Neg
01154      * has completed, and if so, how the PHY and link partner has
01155      * flow control configured.
01156      */
01157     if ((hw->phy.media_type == e1000_media_type_copper) && mac->autoneg) {
01158         /*
01159          * Read the MII Status Register and check to see if AutoNeg
01160          * has completed.  We read this twice because this reg has
01161          * some "sticky" (latched) bits.
01162          */
01163         ret_val = e1e_rphy(hw, PHY_STATUS, &mii_status_reg);
01164         if (ret_val)
01165             goto out;
01166         ret_val = e1e_rphy(hw, PHY_STATUS, &mii_status_reg);
01167         if (ret_val)
01168             goto out;
01169 
01170         if (!(mii_status_reg & MII_SR_AUTONEG_COMPLETE)) {
01171             e_dbg("Copper PHY and Auto Neg "
01172                      "has not completed.\n");
01173             goto out;
01174         }
01175 
01176         /*
01177          * The AutoNeg process has completed, so we now need to
01178          * read both the Auto Negotiation Advertisement
01179          * Register (Address 4) and the Auto_Negotiation Base
01180          * Page Ability Register (Address 5) to determine how
01181          * flow control was negotiated.
01182          */
01183         ret_val = e1e_rphy(hw, PHY_AUTONEG_ADV,
01184                                      &mii_nway_adv_reg);
01185         if (ret_val)
01186             goto out;
01187         ret_val = e1e_rphy(hw, PHY_LP_ABILITY,
01188                                      &mii_nway_lp_ability_reg);
01189         if (ret_val)
01190             goto out;
01191 
01192         /*
01193          * Two bits in the Auto Negotiation Advertisement Register
01194          * (Address 4) and two bits in the Auto Negotiation Base
01195          * Page Ability Register (Address 5) determine flow control
01196          * for both the PHY and the link partner.  The following
01197          * table, taken out of the IEEE 802.3ab/D6.0 dated March 25,
01198          * 1999, describes these PAUSE resolution bits and how flow
01199          * control is determined based upon these settings.
01200          * NOTE:  DC = Don't Care
01201          *
01202          *   LOCAL DEVICE  |   LINK PARTNER
01203          * PAUSE | ASM_DIR | PAUSE | ASM_DIR | NIC Resolution
01204          *-------|---------|-------|---------|--------------------
01205          *   0   |    0    |  DC   |   DC    | e1000_fc_none
01206          *   0   |    1    |   0   |   DC    | e1000_fc_none
01207          *   0   |    1    |   1   |    0    | e1000_fc_none
01208          *   0   |    1    |   1   |    1    | e1000_fc_tx_pause
01209          *   1   |    0    |   0   |   DC    | e1000_fc_none
01210          *   1   |   DC    |   1   |   DC    | e1000_fc_full
01211          *   1   |    1    |   0   |    0    | e1000_fc_none
01212          *   1   |    1    |   0   |    1    | e1000_fc_rx_pause
01213          *
01214          * Are both PAUSE bits set to 1?  If so, this implies
01215          * Symmetric Flow Control is enabled at both ends.  The
01216          * ASM_DIR bits are irrelevant per the spec.
01217          *
01218          * For Symmetric Flow Control:
01219          *
01220          *   LOCAL DEVICE  |   LINK PARTNER
01221          * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
01222          *-------|---------|-------|---------|--------------------
01223          *   1   |   DC    |   1   |   DC    | E1000_fc_full
01224          *
01225          */
01226         if ((mii_nway_adv_reg & NWAY_AR_PAUSE) &&
01227             (mii_nway_lp_ability_reg & NWAY_LPAR_PAUSE)) {
01228             /*
01229              * Now we need to check if the user selected Rx ONLY
01230              * of pause frames.  In this case, we had to advertise
01231              * FULL flow control because we could not advertise Rx
01232              * ONLY. Hence, we must now check to see if we need to
01233              * turn OFF  the TRANSMISSION of PAUSE frames.
01234              */
01235             if (hw->fc.requested_mode == e1000_fc_full) {
01236                 hw->fc.current_mode = e1000_fc_full;
01237                 e_dbg("Flow Control = FULL.\r\n");
01238             } else {
01239                 hw->fc.current_mode = e1000_fc_rx_pause;
01240                 e_dbg("Flow Control = "
01241                          "Rx PAUSE frames only.\r\n");
01242             }
01243         }
01244         /*
01245          * For receiving PAUSE frames ONLY.
01246          *
01247          *   LOCAL DEVICE  |   LINK PARTNER
01248          * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
01249          *-------|---------|-------|---------|--------------------
01250          *   0   |    1    |   1   |    1    | e1000_fc_tx_pause
01251          */
01252         else if (!(mii_nway_adv_reg & NWAY_AR_PAUSE) &&
01253                   (mii_nway_adv_reg & NWAY_AR_ASM_DIR) &&
01254                   (mii_nway_lp_ability_reg & NWAY_LPAR_PAUSE) &&
01255                   (mii_nway_lp_ability_reg & NWAY_LPAR_ASM_DIR)) {
01256             hw->fc.current_mode = e1000_fc_tx_pause;
01257             e_dbg("Flow Control = Tx PAUSE frames only.\r\n");
01258         }
01259         /*
01260          * For transmitting PAUSE frames ONLY.
01261          *
01262          *   LOCAL DEVICE  |   LINK PARTNER
01263          * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
01264          *-------|---------|-------|---------|--------------------
01265          *   1   |    1    |   0   |    1    | e1000_fc_rx_pause
01266          */
01267         else if ((mii_nway_adv_reg & NWAY_AR_PAUSE) &&
01268                  (mii_nway_adv_reg & NWAY_AR_ASM_DIR) &&
01269                  !(mii_nway_lp_ability_reg & NWAY_LPAR_PAUSE) &&
01270                  (mii_nway_lp_ability_reg & NWAY_LPAR_ASM_DIR)) {
01271             hw->fc.current_mode = e1000_fc_rx_pause;
01272             e_dbg("Flow Control = Rx PAUSE frames only.\r\n");
01273         } else {
01274             /*
01275              * Per the IEEE spec, at this point flow control
01276              * should be disabled.
01277              */
01278             hw->fc.current_mode = e1000_fc_none;
01279             e_dbg("Flow Control = NONE.\r\n");
01280         }
01281 
01282         /*
01283          * Now we need to do one last check...  If we auto-
01284          * negotiated to HALF DUPLEX, flow control should not be
01285          * enabled per IEEE 802.3 spec.
01286          */
01287         ret_val = mac->ops.get_link_up_info(hw, &speed, &duplex);
01288         if (ret_val) {
01289             e_dbg("Error getting link speed and duplex\n");
01290             goto out;
01291         }
01292 
01293         if (duplex == HALF_DUPLEX)
01294             hw->fc.current_mode = e1000_fc_none;
01295 
01296         /*
01297          * Now we call a subroutine to actually force the MAC
01298          * controller to use the correct flow control settings.
01299          */
01300         ret_val = e1000e_force_mac_fc(hw);
01301         if (ret_val) {
01302             e_dbg("Error forcing flow control settings\n");
01303             goto out;
01304         }
01305     }
01306 
01307 out:
01308     return ret_val;
01309 }

01372 s32 e1000e_get_hw_semaphore(struct e1000_hw *hw)
01373 {
01374     u32 swsm;
01375     s32 ret_val = E1000_SUCCESS;
01376     s32 timeout = hw->nvm.word_size + 1;
01377     s32 i = 0;
01378 
01379     /* Get the SW semaphore */
01380     while (i < timeout) {
01381         swsm = er32(SWSM);
01382         if (!(swsm & E1000_SWSM_SMBI))
01383             break;
01384

01386         i++;
01387     }
01388 
01389     if (i == timeout) {
