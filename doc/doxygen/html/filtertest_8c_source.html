00071 static char *
00072 read_infile(char *fname)
00073 {
00074     register int i, fd, cc;
00075     register char *cp;
00076     struct stat buf;
00077 
00078     fd = open(fname, O_RDONLY|O_BINARY);
00079     if (fd < 0)
00080         error("can't open %s: %s", fname, pcap_strerror(errno));
00081 
00082     if (fstat(fd, &buf) < 0)
00083         error("can't stat %s: %s", fname, pcap_strerror(errno));
00084 
00085     cp = malloc((u_int)buf.st_size + 1);
00086     if (cp == NULL)
00087         error("malloc(%d) for %s: %s", (u_int)buf.st_size + 1,
00088             fname, pcap_strerror(errno));
00089     cc = read(fd, cp, (u_int)buf.st_size);
00090     if (cc < 0)
00091         error("read %s: %s", fname, pcap_strerror(errno));
00092     if (cc != buf.st_size)
00093         error("short read %s (%d != %d)", fname, cc, (int)buf.st_size);
00094 
00095     close(fd);
00096     /* replace "# comment" with spaces */
00097     for (i = 0; i < cc; i++) {
00098         if (cp[i] == '#')
00099             while (i < cc && cp[i] != '\n')
00100                 cp[i++] = ' ';
00101     }
00102     cp[cc] = '\0';
00103     return (cp);
00104 }

00130 {
00131     register char **p;
00132     register u_int len = 0;
00133     char *buf;
00134     char *src, *dst;

00136     p = argv;

00145         error("copy_argv: malloc");

00147     p = argv;

00150         while ((*dst++ = *src++) != '\0')
00151             ;
