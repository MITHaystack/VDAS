00135 int 
00136 pcap_next_ex(pcap_t *p, struct pcap_pkthdr **pkt_header,
00137     const u_char **pkt_data)
00138 {
00139     struct oneshot_userdata s;
00140 
00141     s.hdr = &p->pcap_header;
00142     s.pkt = pkt_data;
00143     s.pd = p;
00144 
00145     /* Saves a pointer to the packet headers */
00146     *pkt_header= &p->pcap_header;
00147 
00148     if (p->sf.rfile != NULL) {
00149         int status;
00150 
00151         /* We are on an offline capture */
00152         status = pcap_offline_read(p, 1, pcap_oneshot, (u_char *)&s);
00153 
00154         /*
00155          * Return codes for pcap_offline_read() are:
00156          *   -  0: EOF
00157          *   - -1: error
00158          *   - >1: OK
00159          * The first one ('0') conflicts with the return code of
00160          * 0 from pcap_read() meaning "no packets arrived before
00161          * the timeout expired", so we map it to -2 so you can
00162          * distinguish between an EOF from a savefile and a
00163          * "no packets arrived before the timeout expired, try
00164          * again" from a live capture.
00165          */
00166         if (status == 0)
00167             return (-2);
00168         else
00169             return (status);
00170     }
00171 
00172     /*
00173      * Return codes for pcap_read() are:
00174      *   -  0: timeout
00175      *   - -1: error
00176      *   - -2: loop was broken out of with pcap_breakloop()
00177      *   - >1: OK
00178      * The first one ('0') conflicts with the return code of 0 from
00179      * pcap_offline_read() meaning "end of file".
00180     */
00181     return (p->read_op(p, 1, pcap_oneshot, (u_char *)&s));
00182 }

00320 int
00321 pcap_activate(pcap_t *p)
00322 {
00323     int status;
00324 
00325     status = p->activate_op(p);
00326     if (status >= 0)
00327         p->activated = 1;
00328     else {
00329         if (p->errbuf[0] == '\0') {
00330             /*
00331              * No error message supplied by the activate routine;
00332              * for the benefit of programs that don't specially
00333              * handle errors other than PCAP_ERROR, return the
00334              * error message corresponding to the status.
00335              */
00336             snprintf(p->errbuf, PCAP_ERRBUF_SIZE, "%s",
00337                 pcap_statustostr(status));
00338         }
00339 
00340         /*
00341          * Undo any operation pointer setting, etc. done by
00342          * the activate operation.
00343          */
00344         initialize_ops(p);
00345     }
00346     return (status);
00347 }

00349 pcap_t *
00350 pcap_open_live(const char *source, int snaplen, int promisc, int to_ms, char *errbuf)
00351 {
00352     pcap_t *p;
00353     int status;
00354 
00355     p = pcap_create(source, errbuf);
00356     if (p == NULL)
00357         return (NULL);
00358     status = pcap_set_snaplen(p, snaplen);
00359     if (status < 0)
00360         goto fail;
00361     status = pcap_set_promisc(p, promisc);
00362     if (status < 0)
00363         goto fail;
00364     status = pcap_set_timeout(p, to_ms);
00365     if (status < 0)
00366         goto fail;
00367     /*
00368      * Mark this as opened with pcap_open_live(), so that, for
00369      * example, we show the full list of DLT_ values, rather
00370      * than just the ones that are compatible with capturing
00371      * when not in monitor mode.  That allows existing applications
00372      * to work the way they used to work, but allows new applications
00373      * that know about the new open API to, for example, find out the
00374      * DLT_ values that they can select without changing whether
00375      * the adapter is in monitor mode or not.
00376      */
00377     p->oldstyle = 1;
00378     status = pcap_activate(p);
00379     if (status < 0)
00380         goto fail;
00381     return (p);
00382 fail:
00383     if (status == PCAP_ERROR)
00384         snprintf(errbuf, PCAP_ERRBUF_SIZE, "%s: %s", source,
00385             p->errbuf);
00386     else if (status == PCAP_ERROR_NO_SUCH_DEVICE ||
00387         status == PCAP_ERROR_PERM_DENIED)
00388         snprintf(errbuf, PCAP_ERRBUF_SIZE, "%s: %s (%s)", source,
00389             pcap_statustostr(status), p->errbuf);
00390     else
00391         snprintf(errbuf, PCAP_ERRBUF_SIZE, "%s: %s", source,
00392             pcap_statustostr(status));
00393     pcap_close(p);
00394     return (NULL);
00395 }

00413 int
00414 pcap_loop(pcap_t *p, int cnt, pcap_handler callback, u_char *user)
00415 {
00416     register int n;
00417 
00418     for (;;) {
00419         if (p->sf.rfile != NULL) {
00420             /*
00421              * 0 means EOF, so don't loop if we get 0.
00422              */
00423             n = pcap_offline_read(p, cnt, callback, user);
00424         } else {
00425             /*
00426              * XXX keep reading until we get something
00427              * (or an error occurs)
00428              */
00429             do {
00430                 n = p->read_op(p, cnt, callback, user);
00431             } while (n == 0);
00432         }
00433         if (n <= 0)
00434             return (n);
00435         if (cnt > 0) {
00436             cnt -= n;
00437             if (cnt <= 0)
00438                 return (0);
00439         }
00440     }
00441 }

00469 pcap_list_datalinks(pcap_t *p, int **dlt_buffer)
00470 {
00471     if (p->dlt_count == 0) {
00472         /*
00473          * We couldn't fetch the list of DLTs, which means

00494             sizeof(**dlt_buffer) * p->dlt_count);

00516 int
00517 pcap_set_datalink(pcap_t *p, int dlt)
00518 {
00519     int i;
00520     const char *dlt_name;
00521 
00522     if (p->dlt_count == 0 || p->set_datalink_op == NULL) {
00523         /*
00524          * We couldn't fetch the list of DLTs, or we don't
00525          * have a "set datalink" operation, which means
00526          * this platform doesn't support changing the
00527          * DLT for an interface.  Check whether the new
00528          * DLT is the one this interface supports.
00529          */
00530         if (p->linktype != dlt)
00531             goto unsupported;
00532 
00533         /*
00534          * It is, so there's nothing we need to do here.
00535          */
00536         return (0);
00537     }
00538     for (i = 0; i < p->dlt_count; i++)
00539         if (p->dlt_list[i] == dlt)
00540             break;
00541     if (i >= p->dlt_count)
00542         goto unsupported;
00543     if (p->dlt_count == 2 && p->dlt_list[0] == DLT_EN10MB &&
00544         dlt == DLT_DOCSIS) {
00545         /*
00546          * This is presumably an Ethernet device, as the first
00547          * link-layer type it offers is DLT_EN10MB, and the only
00548          * other type it offers is DLT_DOCSIS.  That means that
00549          * we can't tell the driver to supply DOCSIS link-layer
00550          * headers - we're just pretending that's what we're
00551          * getting, as, presumably, we're capturing on a dedicated
00552          * link to a Cisco Cable Modem Termination System, and
00553          * it's putting raw DOCSIS frames on the wire inside low-level
00554          * Ethernet framing.
00555          */
00556         p->linktype = dlt;
00557         return (0);
00558     }
00559     if (p->set_datalink_op(p, dlt) == -1)
00560         return (-1);
00561     p->linktype = dlt;
00562     return (0);
00563 
00564 unsupported:
00565     dlt_name = pcap_datalink_val_to_name(dlt);
00566     if (dlt_name != NULL) {
00567         (void) snprintf(p->errbuf, sizeof(p->errbuf),
00568             "%s is not one of the DLTs supported by this device",
00569             dlt_name);
00570     } else {
00571         (void) snprintf(p->errbuf, sizeof(p->errbuf),
00572             "DLT %d is not one of the DLTs supported by this device",
00573             dlt);
00574     }
00575     return (-1);
00576 }

00759                 *us1 = (const u_char *)s1,
00760                 *us2 = (const u_char *)s2;

00764             return(0);
00765     return (cm[*us1] - cm[*--us2]);
