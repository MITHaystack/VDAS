00067 /*
00068  * The time difference in millisecond
00069  */
00070 double delta_time (struct timeval * now,
00071            struct timeval * before) {
00072   time_t delta_seconds;
00073   time_t delta_microseconds;
00074 
00075   /*
00076    * compute delta in second, 1/10's and 1/1000's second units
00077    */
00078   delta_seconds      = now -> tv_sec  - before -> tv_sec;
00079   delta_microseconds = now -> tv_usec - before -> tv_usec;
00080 
00081   if(delta_microseconds < 0) {
00082     /* manually carry a one from the seconds field */
00083     delta_microseconds += 1000000;  /* 1e6 */
00084     -- delta_seconds;
00085   }
00086   return((double)(delta_seconds * 1000) + (double)delta_microseconds/1000);
00087 }

00206 char* _intoa(unsigned int addr, char* buf, u_short bufLen) {
00207   char *cp, *retStr;
00208   u_int byte;
00209   int n;
00210 
00211   cp = &buf[bufLen];
00212   *--cp = '\0';
00213 
00214   n = 4;
00215   do {
00216     byte = addr & 0xff;
00217     *--cp = byte % 10 + '0';
00218     byte /= 10;
00219     if (byte > 0) {
00220       *--cp = byte % 10 + '0';
00221       byte /= 10;
00222       if (byte > 0)
00223     *--cp = byte + '0';
00224     }
00225     *--cp = '.';
00226     addr >>= 8;
00227   } while (--n > 0);
00228 
00229   /* Convert the string to lowercase */
00230   retStr = (char*)(cp+1);
00231 
00232   return(retStr);

00318 int32_t gmt2local(time_t t) {
00319   int dt, dir;
00320   struct tm *gmt, *loc;
00321   struct tm sgmt;
00322 
00323   if (t == 0)
00324     t = time(NULL);
00325   gmt = &sgmt;
00326   *gmt = *gmtime(&t);
00327   loc = localtime(&t);
00328   dt = (loc->tm_hour - gmt->tm_hour) * 60 * 60 +
00329     (loc->tm_min - gmt->tm_min) * 60;
00330 
00331   /*
00332    * If the year or julian day is different, we span 00:00 GMT
00333    * and must add or subtract a day. Check the year first to
00334    * avoid problems when the julian day wraps.
00335    */
00336   dir = loc->tm_year - gmt->tm_year;
00337   if (dir == 0)
00338     dir = loc->tm_yday - gmt->tm_yday;
00339   dt += dir * 24 * 60 * 60;
00340 
00341   return (dt);
00342 }
