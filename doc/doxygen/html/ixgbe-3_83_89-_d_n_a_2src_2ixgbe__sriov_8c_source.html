00044 int ixgbe_set_vf_multicasts(struct ixgbe_adapter *adapter,
00045                 int entries, u16 *hash_list, u32 vf)
00046 {
00047     struct vf_data_storage *vfinfo = &adapter->vfinfo[vf];
00048     struct ixgbe_hw *hw = &adapter->hw;
00049     int i;
00050     u32 vector_bit;
00051     u32 vector_reg;
00052     u32 mta_reg;
00053     u32 vmolr = IXGBE_READ_REG(hw, IXGBE_VMOLR(vf));
00054 
00055     /* only so many hash values supported */
00056     entries = min(entries, IXGBE_MAX_VF_MC_ENTRIES);
00057 
00058     /* salt away the number of multi cast addresses assigned
00059      * to this VF for later use to restore when the PF multi cast
00060      * list changes
00061      */
00062     vfinfo->num_vf_mc_hashes = entries;
00063 
00064     /* VFs are limited to using the MTA hash table for their multicast
00065      * addresses */
00066     for (i = 0; i < entries; i++) {
00067         vfinfo->vf_mc_hashes[i] = hash_list[i];;
00068     }
00069 
00070     for (i = 0; i < vfinfo->num_vf_mc_hashes; i++) {
00071         vector_reg = (vfinfo->vf_mc_hashes[i] >> 5) & 0x7F;
00072         vector_bit = vfinfo->vf_mc_hashes[i] & 0x1F;
00073         mta_reg = IXGBE_READ_REG(hw, IXGBE_MTA(vector_reg));
00074         mta_reg |= (1 << vector_bit);
00075         IXGBE_WRITE_REG(hw, IXGBE_MTA(vector_reg), mta_reg);
00076     }
00077     vmolr |= IXGBE_VMOLR_ROMPE;
00078     IXGBE_WRITE_REG(hw, IXGBE_VMOLR(vf), vmolr);
00079 
00080     return 0;
00081 }

00083 void ixgbe_restore_vf_multicasts(struct ixgbe_adapter *adapter)
00084 {
00085     struct ixgbe_hw *hw = &adapter->hw;
00086     struct vf_data_storage *vfinfo;
00087     int i, j;
00088     u32 vector_bit;
00089     u32 vector_reg;
00090     u32 mta_reg;
00091 
00092     for (i = 0; i < adapter->num_vfs; i++) {
00093         u32 vmolr = IXGBE_READ_REG(hw, IXGBE_VMOLR(i));
00094         vfinfo = &adapter->vfinfo[i];
00095         for (j = 0; j < vfinfo->num_vf_mc_hashes; j++) {
00096             hw->addr_ctrl.mta_in_use++;
00097             vector_reg = (vfinfo->vf_mc_hashes[j] >> 5) & 0x7F;
00098             vector_bit = vfinfo->vf_mc_hashes[j] & 0x1F;
00099             mta_reg = IXGBE_READ_REG(hw, IXGBE_MTA(vector_reg));
00100             mta_reg |= (1 << vector_bit);
00101             IXGBE_WRITE_REG(hw, IXGBE_MTA(vector_reg), mta_reg);
00102         }
00103         if (vfinfo->num_vf_mc_hashes)
00104             vmolr |= IXGBE_VMOLR_ROMPE;
00105         else
00106             vmolr &= ~IXGBE_VMOLR_ROMPE;
00107         IXGBE_WRITE_REG(hw, IXGBE_VMOLR(i), vmolr);
00108     }

00116 void ixgbe_set_vf_lpe(struct ixgbe_adapter *adapter, u32 *msgbuf)
00117 {
00118     struct ixgbe_hw *hw = &adapter->hw;
00119     int new_mtu = msgbuf[1];
00120     u32 max_frs;
00121     int max_frame = new_mtu + ETH_HLEN + ETH_FCS_LEN;
00122 
00123     /* Only X540 supports jumbo frames in IOV mode */
00124     if (adapter->hw.mac.type != ixgbe_mac_X540)
00125         return;
00126 
00127     /* MTU < 68 is an error and causes problems on some kernels */
00128     if ((new_mtu < 68) || (max_frame > IXGBE_MAX_JUMBO_FRAME_SIZE)) {
00129         e_err(drv, "VF mtu %d out of range\n", new_mtu);
00130         return;
00131     }
00132 
00133     max_frs = (IXGBE_READ_REG(hw, IXGBE_MAXFRS) &
00134            IXGBE_MHADD_MFS_MASK) >> IXGBE_MHADD_MFS_SHIFT;
00135     if (max_frs < new_mtu) {
00136         max_frs = new_mtu << IXGBE_MHADD_MFS_SHIFT;
00137         IXGBE_WRITE_REG(hw, IXGBE_MAXFRS, max_frs);
00138     }
00139 
00140     e_info(drv, "VF requests change max MTU to %d\n", new_mtu);
00141 }

00165 inline void ixgbe_vf_reset_event(struct ixgbe_adapter *adapter, u32 vf)
00166 {
00167     struct ixgbe_hw *hw = &adapter->hw;
00168     int rar_entry = hw->mac.num_rar_entries - (vf + 1);
00169 
00170     /* reset offloads to defaults */
00171     if (adapter->vfinfo[vf].pf_vlan) {
00172         ixgbe_set_vf_vlan(adapter, true,
00173                   adapter->vfinfo[vf].pf_vlan, vf);
00174         ixgbe_set_vmvir(adapter,
00175                 (adapter->vfinfo[vf].pf_vlan |
00176                  (adapter->vfinfo[vf].pf_qos << 
00177                   VLAN_PRIO_SHIFT)), vf);
00178         ixgbe_set_vmolr(hw, vf, false);
00179     } else {
00180         ixgbe_set_vmvir(adapter, 0, vf);
00181         ixgbe_set_vmolr(hw, vf, true);
00182     }
00183 
00184 
00185     /* reset multicast table array for vf */
00186     adapter->vfinfo[vf].num_vf_mc_hashes = 0;
00187 
00188     /* Flush and reset the mta with the new values */
00189     ixgbe_set_rx_mode(adapter->netdev);
00190 
00191     hw->mac.ops.clear_rar(hw, rar_entry);
00192 }

00230 inline void ixgbe_vf_reset_msg(struct ixgbe_adapter *adapter, u32 vf)
00231 {
00232     struct ixgbe_hw *hw = &adapter->hw;
00233     u32 reg;
00234     u32 reg_offset, vf_shift;
00235 
00236     vf_shift = vf % 32;
00237     reg_offset = vf / 32;
00238 
00239     /* enable transmit and receive for vf */
00240     reg = IXGBE_READ_REG(hw, IXGBE_VFTE(reg_offset));
00241     reg |= (reg | (1 << vf_shift));
00242     IXGBE_WRITE_REG(hw, IXGBE_VFTE(reg_offset), reg);
00243 
00244     reg = IXGBE_READ_REG(hw, IXGBE_VFRE(reg_offset));
00245     reg |= (reg | (1 << vf_shift));
00246     IXGBE_WRITE_REG(hw, IXGBE_VFRE(reg_offset), reg);
00247 
00248     reg = IXGBE_READ_REG(hw, IXGBE_VMECM(reg_offset));
00249     reg |= (1 << vf_shift);
00250     IXGBE_WRITE_REG(hw, IXGBE_VMECM(reg_offset), reg);
00251 
00252     ixgbe_vf_reset_event(adapter, vf);
00253 }

00255 static int ixgbe_rcv_msg_from_vf(struct ixgbe_adapter *adapter, u32 vf)
00256 {
00257     u32 mbx_size = IXGBE_VFMAILBOX_SIZE;
00258     u32 msgbuf[mbx_size];
00259     struct ixgbe_hw *hw = &adapter->hw;
00260     s32 retval;
00261     int entries;
00262     u16 *hash_list;
00263     int add, vid;
00264     u8 *new_mac;
00265 
00266     retval = ixgbe_read_mbx(hw, msgbuf, mbx_size, vf);
00267 
00268     if (retval)
00269         printk(KERN_ERR "Error receiving message from VF\n");
00270 
00271     /* this is a message we already processed, do nothing */
00272     if (msgbuf[0] & (IXGBE_VT_MSGTYPE_ACK | IXGBE_VT_MSGTYPE_NACK))
00273         return retval;
00274 
00275     /*
00276      * until the vf completes a virtual function reset it should not be
00277      * allowed to start any configuration.
00278      */
00279 
00280     if (msgbuf[0] == IXGBE_VF_RESET) {
00281         unsigned char *vf_mac = adapter->vfinfo[vf].vf_mac_addresses;
00282         new_mac = (u8 *)(&msgbuf[1]);
00283         adapter->vfinfo[vf].clear_to_send = false;
00284         ixgbe_vf_reset_msg(adapter, vf);
00285         adapter->vfinfo[vf].clear_to_send = true;
00286 
00287         if (is_valid_ether_addr(new_mac) &&
00288             !adapter->vfinfo[vf].pf_set_mac)
00289             ixgbe_set_vf_mac(adapter, vf, vf_mac);
00290         else
00291             ixgbe_set_vf_mac(adapter,
00292                  vf, adapter->vfinfo[vf].vf_mac_addresses);
00293 
00294         /* reply to reset with ack and vf mac address */
00295         msgbuf[0] = IXGBE_VF_RESET | IXGBE_VT_MSGTYPE_ACK;
00296         memcpy(new_mac, vf_mac, IXGBE_ETH_LENGTH_OF_ADDRESS);
00297         /* Piggyback the multicast filter type so VF can compute the
00298          * correct vectors */
00299         msgbuf[3] = hw->mac.mc_filter_type;
00300         ixgbe_write_mbx(hw, msgbuf, IXGBE_VF_PERMADDR_MSG_LEN, vf);
00301 
00302         return retval;
00303     }
00304 
00305     if (!adapter->vfinfo[vf].clear_to_send) {
00306         msgbuf[0] |= IXGBE_VT_MSGTYPE_NACK;
00307         ixgbe_write_mbx(hw, msgbuf, 1, vf);
00308         return retval;
00309     }
00310 
00311     switch ((msgbuf[0] & 0xFFFF)) {
00312     case IXGBE_VF_SET_MAC_ADDR:
00313         new_mac = ((u8 *)(&msgbuf[1]));
00314         if (is_valid_ether_addr(new_mac) &&
00315             !adapter->vfinfo[vf].pf_set_mac) {
00316             ixgbe_set_vf_mac(adapter, vf, new_mac);
00317             e_info(probe, "Set MAC msg received from VF %d\n", vf);
00318         } else if (memcmp(adapter->vfinfo[vf].vf_mac_addresses,
00319                   new_mac, ETH_ALEN)) {
00320             e_warn(drv, "VF %d attempted to override "
00321                    "administratively set MAC address\nReload "
00322                    "the VF driver to resume operations\n", vf);
00323             retval = -1;
00324         }
00325         break;
00326     case IXGBE_VF_SET_MULTICAST:
00327         entries = (msgbuf[0] & IXGBE_VT_MSGINFO_MASK)
00328                     >> IXGBE_VT_MSGINFO_SHIFT;
00329         hash_list = (u16 *)&msgbuf[1];
00330         retval = ixgbe_set_vf_multicasts(adapter, entries,
00331                          hash_list, vf);
00332         break;
00333     case IXGBE_VF_SET_LPE:
00334         e_info(probe, "Set LPE msg received from vf %d\n", vf);
00335         ixgbe_set_vf_lpe(adapter, msgbuf);
00336         break;
00337     case IXGBE_VF_SET_VLAN:
00338         add = (msgbuf[0] & IXGBE_VT_MSGINFO_MASK)
00339                     >> IXGBE_VT_MSGINFO_SHIFT;
00340         vid = (msgbuf[1] & IXGBE_VLVF_VLANID_MASK);
00341         if (adapter->vfinfo[vf].pf_vlan) {
00342             e_warn(drv, "VF %d attempted to override "
00343                    "administratively set VLAN configuration\n"
00344                    "Reload the VF driver to resume operations\n",
00345                    vf);
00346             retval = -1;
00347         } else {
00348             retval = ixgbe_set_vf_vlan(adapter, add, vid, vf);
00349         }
00350         break;
00351     default:
00352         e_err(drv, "Unhandled Msg %8.8x\n", msgbuf[0]);
00353         retval = IXGBE_ERR_MBX;
00354         break;
00355     }
00356 
00357     /* notify the VF of the results of what it sent us */
00358     if (retval)
00359         msgbuf[0] |= IXGBE_VT_MSGTYPE_NACK;
00360     else
00361         msgbuf[0] |= IXGBE_VT_MSGTYPE_ACK;
00362 
00363     msgbuf[0] |= IXGBE_VT_MSGTYPE_CTS;
00364 
00365     ixgbe_write_mbx(hw, msgbuf, 1, vf);
00366 
00367     return retval;
00368 }

00380 void ixgbe_msg_task(struct ixgbe_adapter *adapter)
00381 {
00382     struct ixgbe_hw *hw = &adapter->hw;
00383     u32 vf;
00384 
00385     for (vf = 0; vf < adapter->num_vfs; vf++) {
00386         /* process any reset requests */
00387         if (!ixgbe_check_for_rst(hw, vf))
00388             ixgbe_vf_reset_event(adapter, vf);
00389 
00390         /* process any messages pending */
00391         if (!ixgbe_check_for_msg(hw, vf))
00392             ixgbe_rcv_msg_from_vf(adapter, vf);
00393 
00394         /* process any acks */
00395         if (!ixgbe_check_for_ack(hw, vf))
00396             ixgbe_rcv_ack_from_vf(adapter, vf);
00397     }
00398 }

00500 static void ixgbe_set_vf_rate_limit(struct ixgbe_hw *hw, int vf, int tx_rate,
00501                     int link_speed)
00502 {
00503     int rf_dec, rf_int;
00504     u32 bcnrc_val;
00505 
00506     if (tx_rate != 0) {
00507         /* Calculate the rate factor values to set */
00508         rf_int = link_speed / tx_rate;
00509         rf_dec = (link_speed - (rf_int * tx_rate));
00510         rf_dec = (rf_dec * (1<<IXGBE_RTTBCNRC_RF_INT_SHIFT)) / tx_rate;
00511 
00512         bcnrc_val = IXGBE_RTTBCNRC_RS_ENA;
00513         bcnrc_val |= ((rf_int<<IXGBE_RTTBCNRC_RF_INT_SHIFT) &
00514                        IXGBE_RTTBCNRC_RF_INT_MASK);
00515         bcnrc_val |= (rf_dec & IXGBE_RTTBCNRC_RF_DEC_MASK);
00516     } else {
00517         bcnrc_val = 0;
00518     }
00519 
00520     IXGBE_WRITE_REG(hw, IXGBE_RTTDQSEL, 2*vf); /* vf Y uses queue 2*Y */
00521     IXGBE_WRITE_REG(hw, IXGBE_RTTBCNRC, bcnrc_val);
00522 }
