00300     struct addrinfo sentinel;

00304     struct addrinfo ai0;

00308 
00309 #ifdef FAITH
00310     static int firsttime = 1;
00311 
00312     if (firsttime) {
00313         /* translator hack */
00314         char *q = getenv("GAI");
00315         if (q && inet_pton(AF_INET6, q, &faith_prefix) == 1)
00316             translate = YES;
00317         firsttime = 0;

00322     cur = &sentinel;
00323     pai = &ai;
00324     pai->ai_flags = 0;
00325     pai->ai_family = PF_UNSPEC;
00326     pai->ai_socktype = ANY;
00327     pai->ai_protocol = ANY;
00328     pai->ai_addrlen = 0;
00329     pai->ai_canonname = NULL;
00330     pai->ai_addr = NULL;
00331     pai->ai_next = NULL;
00332     
00333     if (hostname == NULL && servname == NULL)
00334         return EAI_NONAME;
00335     if (hints) {
00336         /* error check for hints */
00337         if (hints->ai_addrlen || hints->ai_canonname ||
00338             hints->ai_addr || hints->ai_next)
00339             ERR(EAI_BADHINTS); /* xxx */
00340         if (hints->ai_flags & ~AI_MASK)
00341             ERR(EAI_BADFLAGS);
00342         switch (hints->ai_family) {
00343         case PF_UNSPEC:
00344         case PF_INET:
00345 #ifdef INET6
00346         case PF_INET6:
00347 #endif
00348             break;
00349         default:
00350             ERR(EAI_FAMILY);
00351         }
00352         memcpy(pai, hints, sizeof(*pai));
00353 
00354         /*
00355          * if both socktype/protocol are specified, check if they
00356          * are meaningful combination.
00357          */
00358         if (pai->ai_socktype != ANY && pai->ai_protocol != ANY) {
00359             for (ex = explore; ex->e_af >= 0; ex++) {
00360                 if (pai->ai_family != ex->e_af)
00361                     continue;
00362                 if (ex->e_socktype == ANY)
00363                     continue;
00364                 if (ex->e_protocol == ANY)
00365                     continue;
00366                 if (pai->ai_socktype == ex->e_socktype
00367                  && pai->ai_protocol != ex->e_protocol) {
00368                     ERR(EAI_BADHINTS);
00369                 }
00370             }
00371         }
00372     }
00373 
00374     /*
00375      * check for special cases.  (1) numeric servname is disallowed if
00376      * socktype/protocol are left unspecified. (2) servname is disallowed
00377      * for raw and other inet{,6} sockets.
00378      */
00379     if (MATCH_FAMILY(pai->ai_family, PF_INET, 1)
00380 #ifdef PF_INET6
00381      || MATCH_FAMILY(pai->ai_family, PF_INET6, 1)
00382 #endif
00383         ) {
00384         ai0 = *pai;
00385 
00386         if (pai->ai_family == PF_UNSPEC) {
00387 #ifdef PF_INET6
00388             pai->ai_family = PF_INET6;
00389 #else
00390             pai->ai_family = PF_INET;
00391 #endif
00392         }
00393         error = get_portmatch(pai, servname);
00394         if (error)
00395             ERR(error);
00396 
00397         *pai = ai0;
00398     }
00399 
00400     ai0 = *pai;
00401 
00402     /* NULL hostname, or numeric hostname */
00403     for (ex = explore; ex->e_af >= 0; ex++) {
00404         *pai = ai0;
00405 
00406         if (!MATCH_FAMILY(pai->ai_family, ex->e_af, WILD_AF(ex)))
00407             continue;
00408         if (!MATCH(pai->ai_socktype, ex->e_socktype, WILD_SOCKTYPE(ex)))
00409             continue;
00410         if (!MATCH(pai->ai_protocol, ex->e_protocol, WILD_PROTOCOL(ex)))
00411             continue;
00412 
00413         if (pai->ai_family == PF_UNSPEC)
00414             pai->ai_family = ex->e_af;
00415         if (pai->ai_socktype == ANY && ex->e_socktype != ANY)
00416             pai->ai_socktype = ex->e_socktype;
00417         if (pai->ai_protocol == ANY && ex->e_protocol != ANY)
00418             pai->ai_protocol = ex->e_protocol;
00419 
00420         if (hostname == NULL)
00421             error = explore_null(pai, hostname, servname, &cur->ai_next);
00422         else
00423             error = explore_numeric_scope(pai, hostname, servname, &cur->ai_next);
00424 
00425         if (error)
00426             goto free;
00427 
00428         while (cur && cur->ai_next)
00429             cur = cur->ai_next;
00430     }
00431 
00432     /*
00433      * XXX
00434      * If numreic representation of AF1 can be interpreted as FQDN
00435      * representation of AF2, we need to think again about the code below.
00436      */
00437     if (sentinel.ai_next)
00438         goto good;
00439 
00440     if (pai->ai_flags & AI_NUMERICHOST)
00441         ERR(EAI_NONAME);
00442     if (hostname == NULL)
00443         ERR(EAI_NONAME);
00444 
00445     /*
00446      * hostname as alphabetical name.
00447      * we would like to prefer AF_INET6 than AF_INET, so we'll make a

00450     for (afd = afdl; afd->a_af; afd++) {
00451         *pai = ai0;
00452 
00453         if (!MATCH_FAMILY(pai->ai_family, afd->a_af, 1))
00454             continue;
00455 
00456         for (ex = explore; ex->e_af >= 0; ex++) {
00457             *pai = ai0;
00458 
00459             if (pai->ai_family == PF_UNSPEC)
00460                 pai->ai_family = afd->a_af;
00461 
00462             if (!MATCH_FAMILY(pai->ai_family, ex->e_af, WILD_AF(ex)))
00463                 continue;
00464             if (!MATCH(pai->ai_socktype, ex->e_socktype,
00465                     WILD_SOCKTYPE(ex))) {
00466                 continue;
00467             }
00468             if (!MATCH(pai->ai_protocol, ex->e_protocol,
00469                     WILD_PROTOCOL(ex))) {
00470                 continue;
00471             }
00472 
00473             if (pai->ai_family == PF_UNSPEC)
00474                 pai->ai_family = ex->e_af;
00475             if (pai->ai_socktype == ANY && ex->e_socktype != ANY)
00476                 pai->ai_socktype = ex->e_socktype;
00477             if (pai->ai_protocol == ANY && ex->e_protocol != ANY)
00478                 pai->ai_protocol = ex->e_protocol;

00481                 &cur->ai_next);
00482 
00483             while (cur && cur->ai_next)
00484                 cur = cur->ai_next;

00514 explore_fqdn(pai, hostname, servname, res)
00515     const struct addrinfo *pai;
00516     const char *hostname;
00517     const char *servname;
00518     struct addrinfo **res;
00519 {
00520     struct hostent *hp;
00521     int h_error;
00522     int af;
00523     char **aplist = NULL, *apbuf = NULL;
00524     char *ap;
00525     struct addrinfo sentinel, *cur;
00526     int i;
00527 #ifndef USE_GETIPNODEBY
00528     int naddrs;
00529 #endif
00530     const struct afd *afd;
00531     int error;
00532 
00533     *res = NULL;
00534     sentinel.ai_next = NULL;
00535     cur = &sentinel;
00536 
00537     /*
00538      * Do not filter unsupported AFs here.  We need to honor content of
00539      * databases (/etc/hosts, DNS and others).  Otherwise we cannot
00540      * replace gethostbyname() by getaddrinfo().
00541      */
00542 
00543     /*
00544      * if the servname does not match socktype/protocol, ignore it.
00545      */
00546     if (get_portmatch(pai, servname) != 0)
00547         return 0;
00548 
00549     afd = find_afd(pai->ai_family);
00550 
00551     /*
00552      * post-RFC2553: should look at (pai->ai_flags & AI_ADDRCONFIG)
00553      * rather than hardcoding it.  we may need to add AI_ADDRCONFIG
00554      * handling code by ourselves in case we don't have getipnodebyname().
00555      */
00556 #ifdef USE_GETIPNODEBY
00557     hp = getipnodebyname(hostname, pai->ai_family, AI_ADDRCONFIG, &h_error);
00558 #else
00559 #ifdef HAVE_GETHOSTBYNAME2
00560     hp = gethostbyname2(hostname, pai->ai_family);
00561 #else
00562     if (pai->ai_family != AF_INET)
00563         return 0;
00564     hp = gethostbyname(hostname);
00565 #ifdef HAVE_H_ERRNO
00566     h_error = h_errno;
00567 #else
00568     h_error = EINVAL;
00569 #endif
00570 #endif /*HAVE_GETHOSTBYNAME2*/
00571 #endif /*USE_GETIPNODEBY*/
00572 
00573     if (hp == NULL) {
00574         switch (h_error) {
00575         case HOST_NOT_FOUND:
00576         case NO_DATA:
00577             error = EAI_NODATA;
00578             break;
00579         case TRY_AGAIN:
00580             error = EAI_AGAIN;
00581             break;
00582         case NO_RECOVERY:
00583         case NETDB_INTERNAL:
00584         default:
00585             error = EAI_FAIL;
00586             break;
00587         }
00588     } else if ((hp->h_name == NULL) || (hp->h_name[0] == 0)
00589             || (hp->h_addr_list[0] == NULL)) {
00590 #ifdef USE_GETIPNODEBY
00591         freehostent(hp);
00592 #endif
00593         hp = NULL;
00594         error = EAI_FAIL;
00595     }
00596 
00597     if (hp == NULL)
00598         goto free;
00599 
00600 #ifdef USE_GETIPNODEBY
00601     aplist = hp->h_addr_list;
00602 #else
00603     /*
00604      * hp will be overwritten if we use gethostbyname2().
00605      * always deep copy for simplification.
00606      */
00607     for (naddrs = 0; hp->h_addr_list[naddrs] != NULL; naddrs++)
00608         ;
00609     naddrs++;
00610     aplist = (char **)malloc(sizeof(aplist[0]) * naddrs);
00611     apbuf = (char *)malloc(hp->h_length * naddrs);
00612     if (aplist == NULL || apbuf == NULL) {
00613         error = EAI_MEMORY;
00614         goto free;
00615     }
00616     memset(aplist, 0, sizeof(aplist[0]) * naddrs);
00617     for (i = 0; i < naddrs; i++) {
00618         if (hp->h_addr_list[i] == NULL) {
00619             aplist[i] = NULL;
00620             continue;
00621         }
00622         memcpy(&apbuf[i * hp->h_length], hp->h_addr_list[i],
00623             hp->h_length);
00624         aplist[i] = &apbuf[i * hp->h_length];
00625     }
00626 #endif
00627 
00628     for (i = 0; aplist[i] != NULL; i++) {
00629         af = hp->h_addrtype;
00630         ap = aplist[i];
00631 #ifdef AF_INET6
00632         if (af == AF_INET6
00633          && IN6_IS_ADDR_V4MAPPED((struct in6_addr *)ap)) {
