00292 void e1000_rar_set_generic(struct e1000_hw *hw, u8 *addr, u32 index)
00293 {
00294     u32 rar_low, rar_high;
00295 
00296     DEBUGFUNC("e1000_rar_set_generic");
00297 
00298     /*
00299      * HW expects these in little endian so we reverse the byte order
00300      * from network order (big endian) to little endian
00301      */
00302     rar_low = ((u32) addr[0] |
00303                ((u32) addr[1] << 8) |
00304                ((u32) addr[2] << 16) | ((u32) addr[3] << 24));
00305 
00306     rar_high = ((u32) addr[4] | ((u32) addr[5] << 8));
00307 
00308     /* If MAC address zero, no need to set the AV bit */
00309     if (rar_low || rar_high)
00310         rar_high |= E1000_RAH_AV;
00311 
00312     /*
00313      * Some bridges will combine consecutive 32-bit writes into
00314      * a single burst write, which will malfunction on some parts.
00315      * The flushes avoid this.
00316      */
00317     E1000_WRITE_REG(hw, E1000_RAL(index), rar_low);
00318     E1000_WRITE_FLUSH(hw);
00319     E1000_WRITE_REG(hw, E1000_RAH(index), rar_high);
00320     E1000_WRITE_FLUSH(hw);
00321 }

00368 u32 e1000_hash_mc_addr_generic(struct e1000_hw *hw, u8 *mc_addr)
00369 {
00370     u32 hash_value, hash_mask;
00371     u8 bit_shift = 0;
00372 
00373     DEBUGFUNC("e1000_hash_mc_addr_generic");
00374 
00375     /* Register count multiplied by bits per register */
00376     hash_mask = (hw->mac.mta_reg_count * 32) - 1;
00377 
00378     /*
00379      * For a mc_filter_type of 0, bit_shift is the number of left-shifts
00380      * where 0xFF would still fall within the hash mask.
00381      */
00382     while (hash_mask >> bit_shift != 0xFF)
00383         bit_shift++;
00384 
00385     /*
00386      * The portion of the address that is used for the hash table
00387      * is determined by the mc_filter_type setting.
00388      * The algorithm is such that there is a total of 8 bits of shifting.
00389      * The bit_shift for a mc_filter_type of 0 represents the number of
00390      * left-shifts where the MSB of mc_addr[5] would still fall within
00391      * the hash_mask.  Case 0 does this exactly.  Since there are a total
00392      * of 8 bits of shifting, then mc_addr[4] will shift right the
00393      * remaining number of bits. Thus 8 - bit_shift.  The rest of the
00394      * cases are a variation of this algorithm...essentially raising the
00395      * number of bits to shift mc_addr[5] left, while still keeping the
00396      * 8-bit shifting total.
00397      *
00398      * For example, given the following Destination MAC Address and an
00399      * mta register count of 128 (thus a 4096-bit vector and 0xFFF mask),
00400      * we can see that the bit_shift for case 0 is 4.  These are the hash
00401      * values resulting from each mc_filter_type...
00402      * [0] [1] [2] [3] [4] [5]
00403      * 01  AA  00  12  34  56
00404      * LSB                 MSB
00405      *
00406      * case 0: hash_value = ((0x34 >> 4) | (0x56 << 4)) & 0xFFF = 0x563
00407      * case 1: hash_value = ((0x34 >> 3) | (0x56 << 5)) & 0xFFF = 0xAC6
00408      * case 2: hash_value = ((0x34 >> 2) | (0x56 << 6)) & 0xFFF = 0x163
00409      * case 3: hash_value = ((0x34 >> 0) | (0x56 << 8)) & 0xFFF = 0x634
00410      */
00411     switch (hw->mac.mc_filter_type) {
00412     default:
00413     case 0:
00414         break;
00415     case 1:
00416         bit_shift += 1;
00417         break;
00418     case 2:
00419         bit_shift += 2;
00420         break;
00421     case 3:
00422         bit_shift += 4;
00423         break;
00424     }
00425 
00426     hash_value = hash_mask & (((mc_addr[4] >> (8 - bit_shift)) |
00427                               (((u16) mc_addr[5]) << bit_shift)));
00428 
00429     return hash_value;
00430 }

00565 s32 e1000_check_for_fiber_link_generic(struct e1000_hw *hw)
00566 {
00567     struct e1000_mac_info *mac = &hw->mac;
00568     u32 rxcw;
00569     u32 ctrl;
00570     u32 status;
00571     s32 ret_val = E1000_SUCCESS;
00572 
00573     DEBUGFUNC("e1000_check_for_fiber_link_generic");
00574 
00575     ctrl = E1000_READ_REG(hw, E1000_CTRL);
00576     status = E1000_READ_REG(hw, E1000_STATUS);
00577     rxcw = E1000_READ_REG(hw, E1000_RXCW);
00578 
00579     /*
00580      * If we don't have link (auto-negotiation failed or link partner
00581      * cannot auto-negotiate), the cable is plugged in (we have signal),
00582      * and our link partner is not trying to auto-negotiate with us (we
00583      * are receiving idles or data), we need to force link up. We also
00584      * need to give auto-negotiation time to complete, in case the cable
00585      * was just plugged in. The autoneg_failed flag does this.
00586      */
00587     /* (ctrl & E1000_CTRL_SWDPIN1) == 1 == have signal */
00588     if ((ctrl & E1000_CTRL_SWDPIN1) && (!(status & E1000_STATUS_LU)) &&
00589         (!(rxcw & E1000_RXCW_C))) {
00590         if (mac->autoneg_failed == 0) {
00591             mac->autoneg_failed = 1;
00592             goto out;
00593         }
00594         DEBUGOUT("NOT Rx'ing /C/, disable AutoNeg and force link.\n");
00595 
00596         /* Disable auto-negotiation in the TXCW register */
00597         E1000_WRITE_REG(hw, E1000_TXCW, (mac->txcw & ~E1000_TXCW_ANE));
00598 
00599         /* Force link-up and also force full-duplex. */
00600         ctrl = E1000_READ_REG(hw, E1000_CTRL);
00601         ctrl |= (E1000_CTRL_SLU | E1000_CTRL_FD);
00602         E1000_WRITE_REG(hw, E1000_CTRL, ctrl);
00603 
00604         /* Configure Flow Control after forcing link up. */
00605         ret_val = e1000_config_fc_after_link_up_generic(hw);
00606         if (ret_val) {
00607             DEBUGOUT("Error configuring flow control\n");
00608             goto out;
00609         }
00610     } else if ((ctrl & E1000_CTRL_SLU) && (rxcw & E1000_RXCW_C)) {
00611         /*
00612          * If we are forcing link and we are receiving /C/ ordered
00613          * sets, re-enable auto-negotiation in the TXCW register
00614          * and disable forced link in the Device Control register
00615          * in an attempt to auto-negotiate with our link partner.
00616          */
00617         DEBUGOUT("Rx'ing /C/, enable AutoNeg and stop forcing link.\n");
00618         E1000_WRITE_REG(hw, E1000_TXCW, mac->txcw);
00619         E1000_WRITE_REG(hw, E1000_CTRL, (ctrl & ~E1000_CTRL_SLU));
00620 
00621         mac->serdes_has_link = true;
00622     }
00623 
00624 out:

00703             }
00704         } else {

00706             DEBUGOUT("SERDES: Link down - force failed.\n");
00707         }

00715             rxcw = E1000_READ_REG(hw, E1000_RXCW);
00716             if (rxcw & E1000_RXCW_SYNCH) {

00719                     DEBUGOUT("SERDES: Link up - autoneg "
00720                        "completed sucessfully.\n");
00721                 } else {
00722                     mac->serdes_has_link = false;
00723                     DEBUGOUT("SERDES: Link down - invalid"
00724                        "codewords detected in autoneg.\n");
00725                 }
00726             } else {
00727                 mac->serdes_has_link = false;
00728                 DEBUGOUT("SERDES: Link down - no sync.\n");
00729             }
00730         } else {
00731             mac->serdes_has_link = false;
00732             DEBUGOUT("SERDES: Link down - autoneg failed\n");
00733         }
00734     }
00735 
00736 out:
00737     return ret_val;

00942 static s32 e1000_commit_fc_settings_generic(struct e1000_hw *hw)
00943 {
00944     struct e1000_mac_info *mac = &hw->mac;
00945     u32 txcw;
00946     s32 ret_val = E1000_SUCCESS;
00947 
00948     DEBUGFUNC("e1000_commit_fc_settings_generic");
00949 
00950     /*
00951      * Check for a software override of the flow control settings, and
00952      * setup the device accordingly.  If auto-negotiation is enabled, then
00953      * software will have to set the "PAUSE" bits to the correct value in
00954      * the Transmit Config Word Register (TXCW) and re-start auto-
00955      * negotiation.  However, if auto-negotiation is disabled, then
00956      * software will have to manually configure the two flow control enable
00957      * bits in the CTRL register.
00958      *
00959      * The possible values of the "fc" parameter are:
00960      *      0:  Flow control is completely disabled
00961      *      1:  Rx flow control is enabled (we can receive pause frames,
00962      *          but not send pause frames).
00963      *      2:  Tx flow control is enabled (we can send pause frames but we
00964      *          do not support receiving pause frames).
00965      *      3:  Both Rx and Tx flow control (symmetric) are enabled.
00966      */
00967     switch (hw->fc.current_mode) {
00968     case e1000_fc_none:
00969         /* Flow control completely disabled by a software over-ride. */
00970         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD);
00971         break;
00972     case e1000_fc_rx_pause:
00973         /*
00974          * Rx Flow control is enabled and Tx Flow control is disabled
00975          * by a software over-ride. Since there really isn't a way to
00976          * advertise that we are capable of Rx Pause ONLY, we will
00977          * advertise that we support both symmetric and asymmetric Rx
00978          * PAUSE.  Later, we will disable the adapter's ability to send
00979          * PAUSE frames.
00980          */
00981         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_PAUSE_MASK);
00982         break;
00983     case e1000_fc_tx_pause:
00984         /*
00985          * Tx Flow control is enabled, and Rx Flow control is disabled,
00986          * by a software over-ride.
00987          */
00988         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_ASM_DIR);
00989         break;
00990     case e1000_fc_full:
00991         /*
00992          * Flow control (both Rx and Tx) is enabled by a software
00993          * over-ride.
00994          */
00995         txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_PAUSE_MASK);
00996         break;
00997     default:
00998         DEBUGOUT("Flow control param set incorrectly\n");
00999         ret_val = -E1000_ERR_CONFIG;
01000         goto out;
01001         break;
01002     }
01003 
01004     E1000_WRITE_REG(hw, E1000_TXCW, txcw);
01005     mac->txcw = txcw;
01006 
01007 out:
01008     return ret_val;
01009 }

01102 s32 e1000_force_mac_fc_generic(struct e1000_hw *hw)

01110

01116      *
01117      * The "Case" statement below enables/disable flow control
01118      * according to the "hw->fc.current_mode" parameter.

01121      *      0:  Flow control is completely disabled

01144         ctrl |= (E1000_CTRL_TFCE | E1000_CTRL_RFCE);

01168 s32 e1000_config_fc_after_link_up_generic(struct e1000_hw *hw)
01169 {
01170     struct e1000_mac_info *mac = &hw->mac;
01171     s32 ret_val = E1000_SUCCESS;
01172     u16 mii_status_reg, mii_nway_adv_reg, mii_nway_lp_ability_reg;
01173     u16 speed, duplex;
01174 
01175     DEBUGFUNC("e1000_config_fc_after_link_up_generic");
01176 
01177     /*
01178      * Check for the case where we have fiber media and auto-neg failed
01179      * so we had to force link.  In this case, we need to force the
01180      * configuration of the MAC to match the "fc" parameter.
01181      */
01182     if (mac->autoneg_failed) {
01183         if (hw->phy.media_type == e1000_media_type_fiber ||
01184             hw->phy.media_type == e1000_media_type_internal_serdes)
01185             ret_val = e1000_force_mac_fc_generic(hw);
01186     } else {
01187         if (hw->phy.media_type == e1000_media_type_copper)
01188             ret_val = e1000_force_mac_fc_generic(hw);
01189     }
01190 
01191     if (ret_val) {
01192         DEBUGOUT("Error forcing flow control settings\n");
01193         goto out;
01194     }
01195 
01196     /*
01197      * Check for the case where we have copper media and auto-neg is
01198      * enabled.  In this case, we need to check and see if Auto-Neg
01199      * has completed, and if so, how the PHY and link partner has
01200      * flow control configured.
01201      */
01202     if ((hw->phy.media_type == e1000_media_type_copper) && mac->autoneg) {
01203         /*
01204          * Read the MII Status Register and check to see if AutoNeg
01205          * has completed.  We read this twice because this reg has
01206          * some "sticky" (latched) bits.
01207          */
01208         ret_val = hw->phy.ops.read_reg(hw, PHY_STATUS, &mii_status_reg);
01209         if (ret_val)
01210             goto out;
01211         ret_val = hw->phy.ops.read_reg(hw, PHY_STATUS, &mii_status_reg);
01212         if (ret_val)
01213             goto out;
01214 
01215         if (!(mii_status_reg & MII_SR_AUTONEG_COMPLETE)) {
01216             DEBUGOUT("Copper PHY and Auto Neg "
01217                      "has not completed.\n");
01218             goto out;
01219         }
01220 
01221         /*
01222          * The AutoNeg process has completed, so we now need to
01223          * read both the Auto Negotiation Advertisement
01224          * Register (Address 4) and the Auto_Negotiation Base
01225          * Page Ability Register (Address 5) to determine how
01226          * flow control was negotiated.
01227          */
01228         ret_val = hw->phy.ops.read_reg(hw, PHY_AUTONEG_ADV,
01229                                      &mii_nway_adv_reg);
01230         if (ret_val)
01231             goto out;
01232         ret_val = hw->phy.ops.read_reg(hw, PHY_LP_ABILITY,
01233                                      &mii_nway_lp_ability_reg);
01234         if (ret_val)
01235             goto out;
01236 
01237         /*
01238          * Two bits in the Auto Negotiation Advertisement Register
01239          * (Address 4) and two bits in the Auto Negotiation Base
01240          * Page Ability Register (Address 5) determine flow control
01241          * for both the PHY and the link partner.  The following
01242          * table, taken out of the IEEE 802.3ab/D6.0 dated March 25,
01243          * 1999, describes these PAUSE resolution bits and how flow
01244          * control is determined based upon these settings.
01245          * NOTE:  DC = Don't Care
01246          *
01247          *   LOCAL DEVICE  |   LINK PARTNER
01248          * PAUSE | ASM_DIR | PAUSE | ASM_DIR | NIC Resolution
01249          *-------|---------|-------|---------|--------------------
01250          *   0   |    0    |  DC   |   DC    | e1000_fc_none
01251          *   0   |    1    |   0   |   DC    | e1000_fc_none
01252          *   0   |    1    |   1   |    0    | e1000_fc_none
01253          *   0   |    1    |   1   |    1    | e1000_fc_tx_pause
01254          *   1   |    0    |   0   |   DC    | e1000_fc_none
01255          *   1   |   DC    |   1   |   DC    | e1000_fc_full
01256          *   1   |    1    |   0   |    0    | e1000_fc_none
01257          *   1   |    1    |   0   |    1    | e1000_fc_rx_pause
01258          *
01259          * Are both PAUSE bits set to 1?  If so, this implies
01260          * Symmetric Flow Control is enabled at both ends.  The
01261          * ASM_DIR bits are irrelevant per the spec.
01262          *
01263          * For Symmetric Flow Control:
01264          *
01265          *   LOCAL DEVICE  |   LINK PARTNER
01266          * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
01267          *-------|---------|-------|---------|--------------------
01268          *   1   |   DC    |   1   |   DC    | E1000_fc_full
01269          *
01270          */
01271         if ((mii_nway_adv_reg & NWAY_AR_PAUSE) &&
01272             (mii_nway_lp_ability_reg & NWAY_LPAR_PAUSE)) {
01273             /*
01274              * Now we need to check if the user selected Rx ONLY
01275              * of pause frames.  In this case, we had to advertise
01276              * FULL flow control because we could not advertise Rx
01277              * ONLY. Hence, we must now check to see if we need to
01278              * turn OFF the TRANSMISSION of PAUSE frames.
01279              */
01280             if (hw->fc.requested_mode == e1000_fc_full) {
01281                 hw->fc.current_mode = e1000_fc_full;
01282                 DEBUGOUT("Flow Control = FULL.\r\n");
01283             } else {
01284                 hw->fc.current_mode = e1000_fc_rx_pause;
01285                 DEBUGOUT("Flow Control = "
01286                          "Rx PAUSE frames only.\r\n");
01287             }
01288         }
01289         /*
01290          * For receiving PAUSE frames ONLY.
01291          *
01292          *   LOCAL DEVICE  |   LINK PARTNER
01293          * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
01294          *-------|---------|-------|---------|--------------------
01295          *   0   |    1    |   1   |    1    | e1000_fc_tx_pause
01296          */
01297         else if (!(mii_nway_adv_reg & NWAY_AR_PAUSE) &&
01298                   (mii_nway_adv_reg & NWAY_AR_ASM_DIR) &&
01299                   (mii_nway_lp_ability_reg & NWAY_LPAR_PAUSE) &&
01300                   (mii_nway_lp_ability_reg & NWAY_LPAR_ASM_DIR)) {
01301             hw->fc.current_mode = e1000_fc_tx_pause;
01302             DEBUGOUT("Flow Control = Tx PAUSE frames only.\r\n");
01303         }
01304         /*
01305          * For transmitting PAUSE frames ONLY.
01306          *
01307          *   LOCAL DEVICE  |   LINK PARTNER
