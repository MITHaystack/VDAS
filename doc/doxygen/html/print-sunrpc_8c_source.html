00084 void
00085 sunrpcrequest_print(register const u_char *bp, register u_int length,
00086             register const u_char *bp2)
00087 {
00088     register const struct sunrpc_msg *rp;
00089     register const struct ip *ip;
00090 #ifdef INET6
00091     register const struct ip6_hdr *ip6;
00092 #endif
00093     u_int32_t x;
00094     char srcid[20], dstid[20];  /*fits 32bit*/
00095 
00096     rp = (struct sunrpc_msg *)bp;
00097 
00098     if (!nflag) {
00099         snprintf(srcid, sizeof(srcid), "0x%x",
00100             EXTRACT_32BITS(&rp->rm_xid));
00101         strlcpy(dstid, "sunrpc", sizeof(dstid));
00102     } else {
00103         snprintf(srcid, sizeof(srcid), "0x%x",
00104             EXTRACT_32BITS(&rp->rm_xid));
00105         snprintf(dstid, sizeof(dstid), "0x%x", SUNRPC_PMAPPORT);
00106     }
00107 
00108     switch (IP_V((struct ip *)bp2)) {
00109     case 4:
00110         ip = (struct ip *)bp2;
00111         printf("%s.%s > %s.%s: %d",
00112             ipaddr_string(&ip->ip_src), srcid,
00113             ipaddr_string(&ip->ip_dst), dstid, length);
00114         break;
00115 #ifdef INET6
00116     case 6:
00117         ip6 = (struct ip6_hdr *)bp2;
00118         printf("%s.%s > %s.%s: %d",
00119             ip6addr_string(&ip6->ip6_src), srcid,
00120             ip6addr_string(&ip6->ip6_dst), dstid, length);
00121         break;
00122 #endif
00123     default:
00124         printf("%s.%s > %s.%s: %d", "?", srcid, "?", dstid, length);
00125         break;
00126     }
00127 
00128     printf(" %s", tok2str(proc2str, " proc #%u",
00129         EXTRACT_32BITS(&rp->rm_call.cb_proc)));
00130     x = EXTRACT_32BITS(&rp->rm_call.cb_rpcvers);
00131     if (x != 2)
00132         printf(" [rpcver %u]", x);
00133 
00134     switch (EXTRACT_32BITS(&rp->rm_call.cb_proc)) {
00135 
00136     case SUNRPC_PMAPPROC_SET:
00137     case SUNRPC_PMAPPROC_UNSET:
00138     case SUNRPC_PMAPPROC_GETPORT:
00139     case SUNRPC_PMAPPROC_CALLIT:
00140         x = EXTRACT_32BITS(&rp->rm_call.cb_prog);
00141         if (!nflag)
00142             printf(" %s", progstr(x));
00143         else
00144             printf(" %u", x);
00145         printf(".%u", EXTRACT_32BITS(&rp->rm_call.cb_vers));
00146         break;
00147     }
00148 }
