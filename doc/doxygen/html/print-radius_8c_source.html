00441 /* Print an attribute string */
00442 /* value pointed by 'data'   */
00443 /* and 'length' size.        */
00444 /*****************************/
00445 /* Returns nothing.          */
00446 /*****************************/
00447 static void
00448 print_attr_string(register u_char *data, u_int length, u_short attr_code )
00449 {
00450    register u_int i;
00451 
00452    TCHECK2(data[0],length);
00453 
00454    switch(attr_code)
00455    {
00456       case TUNNEL_PASS:
00457            if (length < 3)
00458            {
00459               printf(" [|radius]");
00460               return;
00461            }
00462            if (*data && (*data <=0x1F) )
00463               printf("Tag %u, ",*data);
00464            data++;
00465            length--;
00466            printf("Salt %u ",EXTRACT_16BITS(data) );
00467            data+=2;
00468            length-=2;
00469         break;
00470       case TUNNEL_CLIENT_END:
00471       case TUNNEL_SERVER_END:
00472       case TUNNEL_PRIV_GROUP:
00473       case TUNNEL_ASSIGN_ID:
00474       case TUNNEL_CLIENT_AUTH:
00475       case TUNNEL_SERVER_AUTH:
00476            if (*data <= 0x1F)
00477            {
00478               if (length < 1)
00479               {
00480                  printf(" [|radius]");
00481                  return;
00482               }
00483               printf("Tag %u",*data);
00484               data++;
00485               length--;
00486            }
00487         break;
00488    }
00489 
00490    for (i=0; *data && i < length ; i++, data++)
00491        printf("%c",(*data < 32 || *data > 128) ? '.' : *data );
00492 
00493    return;
00494 
00495    trunc:
00496       printf(" [|radius]");
00497 }

00503 static void
00504 print_vendor_attr(register u_char *data, u_int length, u_short attr_code _U_)
00505 {
00506     u_int idx;
00507     u_int vendor_id;
00508     u_int vendor_type;
00509     u_int vendor_length;
00510 
00511     if (length < 4)
00512         goto trunc;
00513     TCHECK2(*data, 4);
00514     vendor_id = EXTRACT_32BITS(data);
00515     data+=4;
00516     length-=4;
00517 
00518     printf("Vendor: %s (%u)",
00519            tok2str(smi_values,"Unknown",vendor_id),
00520            vendor_id);
00521 
00522     while (length >= 2) {
00523     TCHECK2(*data, 2);
00524 
00525         vendor_type = *(data);
00526         vendor_length = *(data+1);
00527 
00528         if (vendor_length < 2)
00529         {
00530             printf("\n\t    Vendor Attribute: %u, Length: %u (bogus, must be >= 2)",
00531                    vendor_type,
00532                    vendor_length);
00533             return;
00534         }
00535         if (vendor_length > length)
00536         {
00537             printf("\n\t    Vendor Attribute: %u, Length: %u (bogus, goes past end of vendor-specific attribute)",
00538                    vendor_type,
00539                    vendor_length);
00540             return;
00541         }
00542         data+=2;
00543         vendor_length-=2;
00544         length-=2;
00545     TCHECK2(*data, vendor_length);
00546 
00547         printf("\n\t    Vendor Attribute: %u, Length: %u, Value: ",
00548                vendor_type,
00549                vendor_length);
00550         for (idx = 0; idx < vendor_length ; idx++, data++)
00551             printf("%c",(*data < 32 || *data > 128) ? '.' : *data );
00552         length-=vendor_length;
00553     }
00554     return;
00555 
00556    trunc:
00557      printf(" [|radius]");
00558 }
