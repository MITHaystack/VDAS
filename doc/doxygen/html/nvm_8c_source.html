00285 static s32 e1000_ready_nvm_eeprom(struct e1000_hw *hw)
00286 {
00287     struct e1000_nvm_info *nvm = &hw->nvm;
00288     u32 eecd = er32(EECD);
00289     s32 ret_val = E1000_SUCCESS;
00290     u8 spi_stat_reg;
00291 
00292     if (nvm->type == e1000_nvm_eeprom_spi) {
00293         u16 timeout = NVM_MAX_RETRY_SPI;
00294 
00295         /* Clear SK and CS */
00296         eecd &= ~(E1000_EECD_CS | E1000_EECD_SK);
00297         ew32(EECD, eecd);
00298         udelay(1);
00299 
00300         /*
00301          * Read "Status Register" repeatedly until the LSB is cleared.
00302          * The EEPROM will signal that the command has been completed
00303          * by clearing bit 0 of the internal status register.  If it's
00304          * not cleared within 'timeout', then error out.
00305          */
00306         while (timeout) {
00307             e1000_shift_out_eec_bits(hw, NVM_RDSR_OPCODE_SPI,
00308                                      hw->nvm.opcode_bits);
00309             spi_stat_reg = (u8)e1000_shift_in_eec_bits(hw, 8);
00310             if (!(spi_stat_reg & NVM_STATUS_RDY_SPI))
00311                 break;
00312 
00313             udelay(5);
00314             e1000_standby_nvm(hw);
00315             timeout--;
00316         }
00317 
00318         if (!timeout) {
00319             e_dbg("SPI NVM Status error\n");
00320             ret_val = -E1000_ERR_NVM;
00321             goto out;
00322         }
00323     }
00324 
00325 out:
00326     return ret_val;
00327 }

00384 s32 e1000e_write_nvm_spi(struct e1000_hw *hw, u16 offset, u16 words, u16 *data)
00385 {
00386     struct e1000_nvm_info *nvm = &hw->nvm;
00387     s32 ret_val;
00388     u16 widx = 0;
00389 
00390     /*
00391      * A check for invalid values:  offset too large, too many words,
00392      * and not enough words.
00393      */
00394     if ((offset >= nvm->word_size) || (words > (nvm->word_size - offset)) ||
00395         (words == 0)) {
00396         e_dbg("nvm parameter(s) out of bounds\n");
00397         ret_val = -E1000_ERR_NVM;
00398         goto out;
00399     }
00400 
00401     ret_val = nvm->ops.acquire(hw);
00402     if (ret_val)
00403         goto out;
00404 
00405     while (widx < words) {
00406         u8 write_opcode = NVM_WRITE_OPCODE_SPI;
00407 
00408         ret_val = e1000_ready_nvm_eeprom(hw);
00409         if (ret_val)
00410             goto release;
00411 
00412         e1000_standby_nvm(hw);
00413 
00414         /* Send the WRITE ENABLE command (8 bit opcode) */
00415         e1000_shift_out_eec_bits(hw, NVM_WREN_OPCODE_SPI,
00416                                  nvm->opcode_bits);
00417 
00418         e1000_standby_nvm(hw);
00419 
00420         /*
00421          * Some SPI eeproms use the 8th address bit embedded in the
00422          * opcode
00423          */
00424         if ((nvm->address_bits == 8) && (offset >= 128))
00425             write_opcode |= NVM_A8_OPCODE_SPI;
00426 
00427         /* Send the Write command (8-bit opcode + addr) */
00428         e1000_shift_out_eec_bits(hw, write_opcode, nvm->opcode_bits);
00429         e1000_shift_out_eec_bits(hw, (u16)((offset + widx) * 2),
00430                                  nvm->address_bits);
00431 
00432         /* Loop to allow for up to whole page write of eeprom */
00433         while (widx < words) {
00434             u16 word_out = data[widx];
00435             word_out = (word_out >> 8) | (word_out << 8);
00436             e1000_shift_out_eec_bits(hw, word_out, 16);
00437             widx++;
00438 
00439             if ((((offset + widx) * 2) % nvm->page_size) == 0) {
00440                 e1000_standby_nvm(hw);
00441                 break;
00442             }
00443         }
00444     }
00445 
00446     msleep(10);
00447 release:
00448     nvm->ops.release(hw);
00449 
00450 out:
00451     return ret_val;
00452 }

00468     u16 pba_ptr;

00483

00486         e_dbg("NVM Read Error\n");
00487         goto out;
00488     }
00489 
00490     /*
00491      * if nvm_data is not ptr guard the PBA must be in legacy format which
00492      * means pba_ptr is actually our second data word for the PBA number
00493      * and we can decode it into an ascii string
00494      */

00496         e_dbg("NVM PBA number is not stored as string\n");
00497

00500             e_dbg("PBA string buffer too small\n");
00501             return E1000_ERR_NO_SPACE;

00503

00505         pba_num[0] = (nvm_data >> 12) & 0xF;
00506         pba_num[1] = (nvm_data >> 8) & 0xF;
00507         pba_num[2] = (nvm_data >> 4) & 0xF;
00508         pba_num[3] = nvm_data & 0xF;

00510         pba_num[5] = (pba_ptr >> 8) & 0xF;
00511         pba_num[6] = '-';
00512         pba_num[7] = 0;
00513         pba_num[8] = (pba_ptr >> 4) & 0xF;
00514         pba_num[9] = pba_ptr & 0xF;
00515 
00516         /* put a null character on the end of our string */
00517         pba_num[10] = '\0';
00518 
00519         /* switch all the data but the '-' to hex char */
00520         for (offset = 0; offset < 10; offset++) {
00521             if (pba_num[offset] < 0xA)
00522                 pba_num[offset] += '0';
00523             else if (pba_num[offset] < 0x10)
00524                 pba_num[offset] += 'A' - 0xA;
00525         }
00526 
00527         goto out;
00528     }
00529 
00530     ret_val = e1000_read_nvm(hw, pba_ptr, 1, &length);
00531     if (ret_val) {
00532         e_dbg("NVM Read Error\n");
00533         goto out;
00534     }
00535

00537         e_dbg("NVM PBA number section invalid length\n");
00538         ret_val = E1000_ERR_NVM_PBA_SECTION;
00539         goto out;
00540     }
00541     /* check if pba_num buffer is big enough */
00542     if (pba_num_size < (((u32)length * 2) - 1)) {
00543         e_dbg("PBA string buffer too small\n");
00544         ret_val = E1000_ERR_NO_SPACE;
00545         goto out;
00546     }
00547 
00548     /* trim pba length from start of string */
00549     pba_ptr++;
00550     length--;
00551 
00552     for (offset = 0; offset < length; offset++) {
00553         ret_val = e1000_read_nvm(hw, pba_ptr + offset, 1, &nvm_data);
00554         if (ret_val) {
00555             e_dbg("NVM Read Error\n");
00556             goto out;

00558         pba_num[offset * 2] = (u8)(nvm_data >> 8);
00559         pba_num[(offset * 2) + 1] = (u8)(nvm_data & 0xFF);
00560     }
00561     pba_num[offset * 2] = '\0';
