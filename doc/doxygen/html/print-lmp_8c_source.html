00363 void
00364 lmp_print(register const u_char *pptr, register u_int len) {
00365 
00366     const struct lmp_common_header *lmp_com_header;
00367     const struct lmp_object_header *lmp_obj_header;
00368     const u_char *tptr,*obj_tptr;
00369     int tlen,lmp_obj_len,lmp_obj_ctype,obj_tlen;
00370     int hexdump;
00371     int offset,subobj_type,subobj_len,total_subobj_len;
00372     int link_type;
00373 
00374     union { /* int to float conversion buffer */
00375         float f; 
00376         u_int32_t i;
00377     } bw;
00378 
00379     tptr=pptr;
00380     lmp_com_header = (const struct lmp_common_header *)pptr;
00381     TCHECK(*lmp_com_header);
00382 
00383     /*
00384      * Sanity checking of the header.
00385      */
00386     if (LMP_EXTRACT_VERSION(lmp_com_header->version_res[0]) != LMP_VERSION) {
00387     printf("LMP version %u packet not supported",
00388                LMP_EXTRACT_VERSION(lmp_com_header->version_res[0]));
00389     return;
00390     }
00391 
00392     /* in non-verbose mode just lets print the basic Message Type*/
00393     if (vflag < 1) {
00394         printf("LMPv%u %s Message, length: %u",
00395                LMP_EXTRACT_VERSION(lmp_com_header->version_res[0]),
00396                tok2str(lmp_msg_type_values, "unknown (%u)",lmp_com_header->msg_type),
00397                len);
00398         return;
00399     }
00400 
00401     /* ok they seem to want to know everything - lets fully decode it */
00402 
00403     tlen=EXTRACT_16BITS(lmp_com_header->length);
00404 
00405     printf("\n\tLMPv%u, msg-type: %s, Flags: [%s], length: %u",
00406            LMP_EXTRACT_VERSION(lmp_com_header->version_res[0]),
00407            tok2str(lmp_msg_type_values, "unknown, type: %u",lmp_com_header->msg_type),
00408            bittok2str(lmp_header_flag_values,"none",lmp_com_header->flags),
00409            tlen);
00410 
00411     tptr+=sizeof(const struct lmp_common_header);
00412     tlen-=sizeof(const struct lmp_common_header);
00413 
00414     while(tlen>0) {
00415         /* did we capture enough for fully decoding the object header ? */
00416         if (!TTEST2(*tptr, sizeof(struct lmp_object_header)))
00417             goto trunc;
00418 
00419         lmp_obj_header = (const struct lmp_object_header *)tptr;
00420         lmp_obj_len=EXTRACT_16BITS(lmp_obj_header->length);
00421         lmp_obj_ctype=(lmp_obj_header->ctype)&0x7f;
00422 
00423         if(lmp_obj_len % 4 || lmp_obj_len < 4)
00424             return;
00425 
00426         printf("\n\t  %s Object (%u), Class-Type: %s (%u) Flags: [%snegotiable], length: %u",
00427                tok2str(lmp_obj_values,
00428                        "Unknown",
00429                        lmp_obj_header->class_num),
00430                lmp_obj_header->class_num,
00431                tok2str(lmp_ctype_values,
00432                        "Unknown",
00433                        ((lmp_obj_header->class_num)<<8)+lmp_obj_ctype),
00434                lmp_obj_ctype,
00435                (lmp_obj_header->ctype)&0x80 ? "" : "non-",
00436                lmp_obj_len);
00437 
00438         obj_tptr=tptr+sizeof(struct lmp_object_header);
00439         obj_tlen=lmp_obj_len-sizeof(struct lmp_object_header);
00440 
00441         /* did we capture enough for fully decoding the object ? */
00442         if (!TTEST2(*tptr, lmp_obj_len))
00443             goto trunc;
00444         hexdump=FALSE;
00445 
00446         switch(lmp_obj_header->class_num) {
00447 
00448         case LMP_OBJ_CC_ID:
00449             switch(lmp_obj_ctype) {
00450             case LMP_CTYPE_LOC:
00451             case LMP_CTYPE_RMT:
00452                 printf("\n\t    Control Channel ID: %u (0x%08x)",
00453                        EXTRACT_32BITS(obj_tptr),
00454                        EXTRACT_32BITS(obj_tptr));
00455                 break;
00456 
00457             default:
00458                 hexdump=TRUE;
00459             }
00460             break;
00461 
00462         case LMP_OBJ_LINK_ID:
00463         case LMP_OBJ_INTERFACE_ID:
00464             switch(lmp_obj_ctype) {
00465             case LMP_CTYPE_IPV4_LOC:
00466             case LMP_CTYPE_IPV4_RMT:
00467                 printf("\n\t    IPv4 Link ID: %s (0x%08x)",
00468                        ipaddr_string(obj_tptr),
00469                        EXTRACT_32BITS(obj_tptr));
00470                 break;
00471 #ifdef INET6
00472             case LMP_CTYPE_IPV6_LOC:
00473             case LMP_CTYPE_IPV6_RMT:
00474                 printf("\n\t    IPv6 Link ID: %s (0x%08x)",
00475                        ip6addr_string(obj_tptr),
00476                        EXTRACT_32BITS(obj_tptr));
00477                 break;
00478 #endif
00479             case LMP_CTYPE_UNMD_LOC:
00480             case LMP_CTYPE_UNMD_RMT:
00481                 printf("\n\t    Link ID: %u (0x%08x)",
00482                        EXTRACT_32BITS(obj_tptr),
00483                        EXTRACT_32BITS(obj_tptr));
00484                 break;
00485             default:
00486                 hexdump=TRUE;
00487             }
00488             break;
00489
