00040 /*
00041  * Structure of a header for Apple's IP-over-IEEE 1384 BPF header.
00042  */
00043 #define FIREWIRE_EUI64_LEN  8
00044 struct firewire_header {
00045     u_char  firewire_dhost[FIREWIRE_EUI64_LEN];
00046     u_char  firewire_shost[FIREWIRE_EUI64_LEN];
00047     u_short firewire_type;
00048 };
00049 
00050 /*
00051  * Length of that header; note that some compilers may pad
00052  * "struct firewire_header" to a multiple of 4 bytes, for example, so
00053  * "sizeof (struct firewire_header)" may not give the right answer.
00054  */
00055 #define FIREWIRE_HDRLEN     18
00056 
00057 static inline void
00058 ap1394_hdr_print(register const u_char *bp, u_int length)
00059 {
00060     register const struct firewire_header *fp;
00061     u_int16_t firewire_type;
00062 
00063     fp = (const struct firewire_header *)bp;
00064 
00065     (void)printf("%s > %s",
00066              linkaddr_string(fp->firewire_dhost, LINKADDR_IEEE1394, FIREWIRE_EUI64_LEN),
00067              linkaddr_string(fp->firewire_shost, LINKADDR_IEEE1394, FIREWIRE_EUI64_LEN));
00068 
00069     firewire_type = EXTRACT_16BITS(&fp->firewire_type);
00070     if (!qflag) {
00071         (void)printf(", ethertype %s (0x%04x)",
00072                    tok2str(ethertype_values,"Unknown", firewire_type),
00073                                firewire_type);
00074         } else {
00075                 (void)printf(", %s", tok2str(ethertype_values,"Unknown Ethertype (0x%04x)", firewire_type));
00076         }
00077 
00078     (void)printf(", length %u: ", length);
00079 }

00081 /*
00082  * This is the top level routine of the printer.  'p' points
00083  * to the ether header of the packet, 'h->ts' is the timestamp,
00084  * 'h->len' is the length of the packet off the wire, and 'h->caplen'
00085  * is the number of bytes actually captured.
00086  */
00087 u_int
00088 ap1394_if_print(const struct pcap_pkthdr *h, const u_char *p)
00089 {
00090     u_int length = h->len;
00091     u_int caplen = h->caplen;
00092     struct firewire_header *fp;
00093     u_short ether_type;
00094 
00095     if (caplen < FIREWIRE_HDRLEN) {
00096         printf("[|ap1394]");
00097         return FIREWIRE_HDRLEN;
00098     }
00099 
00100     if (eflag)
00101         ap1394_hdr_print(p, length);
00102 
00103     length -= FIREWIRE_HDRLEN;
00104     caplen -= FIREWIRE_HDRLEN;
00105     fp = (struct firewire_header *)p;
00106     p += FIREWIRE_HDRLEN;
00107 
00108     ether_type = EXTRACT_16BITS(&fp->firewire_type);
00109     if (ethertype_print(ether_type, p, length, caplen) == 0) {
00110         /* ether_type not known, print raw packet */
00111         if (!eflag)
00112             ap1394_hdr_print((u_char *)fp, length + FIREWIRE_HDRLEN);
00113 
00114         if (!suppress_default_print)
00115             default_print(p, caplen);
00116     } 
00117 
00118     return FIREWIRE_HDRLEN;
00119 }
