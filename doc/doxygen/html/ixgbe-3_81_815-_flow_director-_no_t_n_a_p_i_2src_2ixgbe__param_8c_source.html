00319 struct ixgbe_option {
00320     enum { enable_option, range_option, list_option } type;
00321     const char *name;
00322     const char *err;
00323     int def;
00324     union {
00325         struct { /* range_option info */
00326             int min;
00327             int max;
00328         } r;
00329         struct { /* list_option info */
00330             int nr;
00331             const struct ixgbe_opt_list {
00332                 int i;
00333                 char *str;
00334             } *p;
00335         } l;
00336     } arg;
00337 };

00339 static int __devinit ixgbe_validate_option(unsigned int *value,
00340                                            struct ixgbe_option *opt)
00341 {
00342     if (*value == OPTION_UNSET) {
00343         *value = opt->def;
00344         return 0;
00345     }
00346 
00347     switch (opt->type) {
00348     case enable_option:
00349         switch (*value) {
00350         case OPTION_ENABLED:
00351             printk(KERN_INFO "ixgbe: %s Enabled\n", opt->name);
00352             return 0;
00353         case OPTION_DISABLED:
00354             printk(KERN_INFO "ixgbe: %s Disabled\n", opt->name);
00355             return 0;
00356         }
00357         break;
00358     case range_option:
00359         if (*value >= opt->arg.r.min && *value <= opt->arg.r.max) {
00360             printk(KERN_INFO "ixgbe: %s set to %d\n", opt->name, *value);
00361             return 0;
00362         }
00363         break;
00364     case list_option: {
00365         int i;
00366         const struct ixgbe_opt_list *ent;
00367 
00368         for (i = 0; i < opt->arg.l.nr; i++) {
00369             ent = &opt->arg.l.p[i];
00370             if (*value == ent->i) {
00371                 if (ent->str[0] != '\0')
00372                     printk(KERN_INFO "%s\n", ent->str);
00373                 return 0;
00374             }
00375         }
00376     }
00377         break;
00378     default:
00379         BUG();
00380     }
00381 
00382     printk(KERN_INFO "ixgbe: Invalid %s specified (%d),  %s\n",
00383            opt->name, *value, opt->err);
00384     *value = opt->def;
00385     return -1;
00386 }

00399 void __devinit ixgbe_check_options(struct ixgbe_adapter *adapter)
00400 {
00401     int bd = adapter->bd_number;
00402     u32 *aflags = &adapter->flags;
00403     struct ixgbe_ring_feature *feature = adapter->ring_feature;
00404 
00405     if (bd >= IXGBE_MAX_NIC) {
00406         printk(KERN_NOTICE
00407                "Warning: no configuration for board #%d\n", bd);
00408         printk(KERN_NOTICE "Using defaults for all values\n");
00409 #ifndef module_param_array
00410         bd = IXGBE_MAX_NIC;
00411 #endif
00412     }
00413 
00414     { /* Interrupt Mode */
00415         unsigned int int_mode;
00416         static struct ixgbe_option opt = {
00417             .type = range_option,
00418             .name = "Interrupt Mode",
00419             .err =
00420               "using default of "__MODULE_STRING(IXGBE_DEFAULT_INT),
00421             .def = IXGBE_DEFAULT_INT,
00422             .arg = { .r = { .min = IXGBE_INT_LEGACY,
00423                     .max = IXGBE_INT_MSIX}}
00424         };
00425 
00426 #ifdef module_param_array
00427         if (num_IntMode > bd || num_InterruptType > bd) {
00428 #endif
00429             int_mode = IntMode[bd];
00430             if (int_mode == OPTION_UNSET)
00431                 int_mode = InterruptType[bd];
00432             ixgbe_validate_option(&int_mode, &opt);
00433             switch (int_mode) {
00434             case IXGBE_INT_MSIX:
00435                 if (!(*aflags & IXGBE_FLAG_MSIX_CAPABLE))
00436                     printk(KERN_INFO
00437                            "Ignoring MSI-X setting; "
00438                            "support unavailable\n");
00439                 break;
00440             case IXGBE_INT_MSI:
00441                 if (!(*aflags & IXGBE_FLAG_MSI_CAPABLE)) {
00442                     printk(KERN_INFO
00443                            "Ignoring MSI setting; "
00444                            "support unavailable\n");
00445                 } else {
00446                     *aflags &= ~IXGBE_FLAG_MSIX_CAPABLE;
00447                     *aflags &= ~IXGBE_FLAG_DCB_CAPABLE;
00448                 }
00449                 break;
00450             case IXGBE_INT_LEGACY:
00451             default:
00452                 *aflags &= ~IXGBE_FLAG_MSIX_CAPABLE;
00453                 *aflags &= ~IXGBE_FLAG_MSI_CAPABLE;
00454                 *aflags &= ~IXGBE_FLAG_DCB_CAPABLE;
00455                 break;
00456             }
00457 #ifdef module_param_array
00458         } else {
00459             /* default settings */
00460             if (opt.def == IXGBE_INT_MSIX &&
00461                 *aflags & IXGBE_FLAG_MSIX_CAPABLE) {
00462                 *aflags |= IXGBE_FLAG_MSIX_CAPABLE;
00463                 *aflags |= IXGBE_FLAG_MSI_CAPABLE;
00464             } else if (opt.def == IXGBE_INT_MSI &&
00465                 *aflags & IXGBE_FLAG_MSI_CAPABLE) {
00466                 *aflags &= ~IXGBE_FLAG_MSIX_CAPABLE;
00467                 *aflags |= IXGBE_FLAG_MSI_CAPABLE;
00468                 *aflags &= ~IXGBE_FLAG_DCB_CAPABLE;
00469             } else {
00470                 *aflags &= ~IXGBE_FLAG_MSIX_CAPABLE;
00471                 *aflags &= ~IXGBE_FLAG_MSI_CAPABLE;
00472                 *aflags &= ~IXGBE_FLAG_DCB_CAPABLE;
00473             }
00474         }
00475 #endif
00476     }
00477     { /* Multiple Queue Support */
00478         static struct ixgbe_option opt = {
00479             .type = enable_option,
00480             .name = "Multiple Queue Support",
00481             .err  = "defaulting to Enabled",
00482             .def  = OPTION_ENABLED
00483         };
00484 
00485 #ifdef module_param_array
00486         if (num_MQ > bd) {
00487 #endif
00488             unsigned int mq = MQ[bd];
00489             ixgbe_validate_option(&mq, &opt);
00490             if (mq)
00491                 *aflags |= IXGBE_FLAG_MQ_CAPABLE;
00492             else
00493                 *aflags &= ~IXGBE_FLAG_MQ_CAPABLE;
00494 #ifdef module_param_array
00495         } else {
00496             if (opt.def == OPTION_ENABLED)
00497                 *aflags |= IXGBE_FLAG_MQ_CAPABLE;
00498             else
00499                 *aflags &= ~IXGBE_FLAG_MQ_CAPABLE;
00500         }
00501 #endif
00502         /* Check Interoperability */
00503         if ((*aflags & IXGBE_FLAG_MQ_CAPABLE) &&
00504             !(*aflags & IXGBE_FLAG_MSIX_CAPABLE)) {
00505             DPRINTK(PROBE, INFO,
00506                     "Multiple queues are not supported while MSI-X "
00507                     "is disabled.  Disabling Multiple Queues.\n");
00508             *aflags &= ~IXGBE_FLAG_MQ_CAPABLE;
00509         }
00510     }
00511 #if defined(CONFIG_DCA) || defined(CONFIG_DCA_MODULE)
00512     { /* Direct Cache Access (DCA) */
00513         static struct ixgbe_option opt = {
00514             .type = range_option,
00515             .name = "Direct Cache Access (DCA)",
00516             .err  = "defaulting to Enabled",
00517             .def  = IXGBE_MAX_DCA,
00518             .arg  = { .r = { .min = OPTION_DISABLED,
00519                      .max = IXGBE_MAX_DCA}}
00520         };
00521         unsigned int dca = opt.def;
00522 
00523 #ifdef module_param_array
00524         if (num_DCA > bd) {
00525 #endif
00526             dca = DCA[bd];
00527             ixgbe_validate_option(&dca, &opt);
00528             if (!dca)
00529                 *aflags &= ~IXGBE_FLAG_DCA_CAPABLE;
00530 
00531             /* Check Interoperability */
00532             if (!(*aflags & IXGBE_FLAG_DCA_CAPABLE)) {
00533                 DPRINTK(PROBE, INFO, "DCA is disabled\n");
00534                 *aflags &= ~IXGBE_FLAG_DCA_ENABLED;
00535             }
00536 
00537             if (dca == IXGBE_MAX_DCA) {
00538                 DPRINTK(PROBE, INFO,
00539                         "DCA enabled for rx data\n");
00540                 adapter->flags |= IXGBE_FLAG_DCA_ENABLED_DATA;
00541             }
00542 #ifdef module_param_array
00543         } else {
00544             /* make sure to clear the capability flag if the
00545              * option is disabled by default above */
00546             if (opt.def == OPTION_DISABLED)
00547                 *aflags &= ~IXGBE_FLAG_DCA_CAPABLE;
00548         }
00549 #endif
00550         if (dca == IXGBE_MAX_DCA)
00551             adapter->flags |= IXGBE_FLAG_DCA_ENABLED_DATA;
00552     }
00553 #endif /* CONFIG_DCA or CONFIG_DCA_MODULE */
00554     { /* Receive-Side Scaling (RSS) */
00555         static struct ixgbe_option opt = {
00556             .type = range_option,
00557             .name = "Receive-Side Scaling (RSS)",
00558             .err  = "using default.",
00559             .def  = OPTION_ENABLED,
00560             .arg  = { .r = { .min = OPTION_DISABLED,
00561                      .max = IXGBE_MAX_RSS_INDICES}}
00562         };
00563         unsigned int rss = RSS[bd];
00564 
00565 #ifdef module_param_array
00566         if (num_RSS > bd) {
00567 #endif
00568             if (rss != OPTION_ENABLED)
00569                 ixgbe_validate_option(&rss, &opt);
00570             /*
00571              * we cannot use an else since validate option may
00572              * have changed the state of RSS
00573              */
00574             if (rss == OPTION_ENABLED) {
00575                 /*
00576                  * Base it off num_online_cpus() with
00577                  * a hardware limit cap.
00578                  */
00579                 rss = min(IXGBE_MAX_RSS_INDICES,
00580                           (int)num_online_cpus());
00581             }
00582             feature[RING_F_RSS].indices = rss;
00583             if (rss)
00584                 *aflags |= IXGBE_FLAG_RSS_ENABLED;
00585             else
00586                 *aflags &= ~IXGBE_FLAG_RSS_ENABLED;
00587 #ifdef module_param_array
00588         } else {
00589             if (opt.def == OPTION_DISABLED) {
00590                 *aflags &= ~IXGBE_FLAG_RSS_ENABLED;
00591             } else {
00592                 rss = min(IXGBE_MAX_RSS_INDICES,
00593                           (int)num_online_cpus());
00594                 feature[RING_F_RSS].indices = rss;
00595                 if (rss)
00596                     *aflags |= IXGBE_FLAG_RSS_ENABLED;
00597                 else
00598                     *aflags &= ~IXGBE_FLAG_RSS_ENABLED;
00599             }
00600         }
00601 #endif
00602         /* Check Interoperability */
