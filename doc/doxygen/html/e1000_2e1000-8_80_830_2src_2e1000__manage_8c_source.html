00132     u32 *buffer = (u32 *)&hw->mng_cookie;
00133     u32 offset;
00134     s32 ret_val, hdr_csum, csum;

00136

00138 
00139     hw->mac.tx_pkt_filtering = true;
00140 
00141     /* No manageability, no filtering */
00142     if (!hw->mac.ops.check_mng_mode(hw)) {
00143         hw->mac.tx_pkt_filtering = false;
00144         goto out;
00145     }
00146 
00147     /*
00148      * If we can't read from the host interface for whatever
00149      * reason, disable filtering.
00150      */
00151     ret_val = hw->mac.ops.mng_enable_host_if(hw);
00152     if (ret_val != E1000_SUCCESS) {
00153         hw->mac.tx_pkt_filtering = false;
00154         goto out;
00155     }
00156 
00157     /* Read in the header.  Length and offset are in dwords. */
00158     len    = E1000_MNG_DHCP_COOKIE_LENGTH >> 2;
00159     offset = E1000_MNG_DHCP_COOKIE_OFFSET >> 2;
00160     for (i = 0; i < len; i++)
00161         *(buffer + i) = E1000_READ_REG_ARRAY_DWORD(hw, E1000_HOST_IF,
00162                                                    offset + i);
00163     hdr_csum = hdr->checksum;
00164     hdr->checksum = 0;
00165     csum = e1000_calculate_checksum((u8 *)hdr,
00166                                     E1000_MNG_DHCP_COOKIE_LENGTH);
00167     /*
00168      * If either the checksums or signature don't match, then
00169      * the cookie area isn't considered valid, in which case we
00170      * take the safe route of assuming Tx filtering is enabled.
00171      */
00172     if ((hdr_csum != csum) || (hdr->signature != E1000_IAMT_SIGNATURE)) {
00173         hw->mac.tx_pkt_filtering = true;
00174         goto out;
00175     }
00176 
00177     /* Cookie area is valid, make the final check for filtering. */
00178     if (!(hdr->status & E1000_MNG_DHCP_COOKIE_STATUS_PARSING)) {
00179         hw->mac.tx_pkt_filtering = false;
00180         goto out;
00181     }
00182
