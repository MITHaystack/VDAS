00395                 pcap_datalink_val_to_description(dlts[n_dlts]));

00504 MakeFilename(char *buffer, char *orig_name, int cnt, int max_chars)
00505 {
00506         char *filename = malloc(NAME_MAX + 1);
00507 
00508         /* Process with strftime if Gflag is set. */
00509         if (Gflag != 0) {
00510           struct tm *local_tm;
00511 
00512           /* Convert Gflag_time to a usable format */
00513           if ((local_tm = localtime(&Gflag_time)) == NULL) {
00514                   error("MakeTimedFilename: localtime");
00515           }
00516 
00517           /* There's no good way to detect an error in strftime since a return
00518            * value of 0 isn't necessarily failure.
00519            */
00520           strftime(filename, NAME_MAX, orig_name, local_tm);
00521         } else {
00522           strncpy(filename, orig_name, NAME_MAX);
00523         }
00524 
00525     if (cnt == 0 && max_chars == 0)
00526         strncpy(buffer, filename, NAME_MAX + 1);
00527     else
00528         if (snprintf(buffer, NAME_MAX + 1, "%s%0*d", filename, max_chars, cnt) > NAME_MAX)
00529                   /* Report an error if the filename is too large */
00530                   error("too many output files or filename is too long (> %d)", NAME_MAX);
00531         free(filename);
