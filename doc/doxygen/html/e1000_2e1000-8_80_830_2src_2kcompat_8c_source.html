00039     int i=0;

00042         i = i*10 + *((*s)++) - '0';

00054 static char * number(char * buf, char * end, long long num, int base, int size, int precision, int type)
00055 {
00056     char c,sign,tmp[66];
00057     const char *digits;
00058     const char small_digits[] = "0123456789abcdefghijklmnopqrstuvwxyz";
00059     const char large_digits[] = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
00060     int i;
00061 
00062     digits = (type & _kc_LARGE) ? large_digits : small_digits;
00063     if (type & _kc_LEFT)
00064         type &= ~_kc_ZEROPAD;
00065     if (base < 2 || base > 36)
00066         return 0;
00067     c = (type & _kc_ZEROPAD) ? '0' : ' ';
00068     sign = 0;
00069     if (type & _kc_SIGN) {
00070         if (num < 0) {
00071             sign = '-';
00072             num = -num;
00073             size--;
00074         } else if (type & _kc_PLUS) {
00075             sign = '+';
00076             size--;
00077         } else if (type & _kc_SPACE) {
00078             sign = ' ';
00079             size--;
00080         }
00081     }
00082     if (type & _kc_SPECIAL) {
00083         if (base == 16)
00084             size -= 2;
00085         else if (base == 8)
00086             size--;
00087     }
00088     i = 0;
00089     if (num == 0)
00090         tmp[i++]='0';
00091     else while (num != 0)
00092         tmp[i++] = digits[do_div(num,base)];
00093     if (i > precision)
00094         precision = i;
00095     size -= precision;
00096     if (!(type&(_kc_ZEROPAD+_kc_LEFT))) {
00097         while(size-->0) {
00098             if (buf <= end)
00099                 *buf = ' ';
00100             ++buf;
00101         }
00102     }
00103     if (sign) {
00104         if (buf <= end)
00105             *buf = sign;
00106         ++buf;
00107     }
00108     if (type & _kc_SPECIAL) {
00109         if (base==8) {
00110             if (buf <= end)
00111                 *buf = '0';
00112             ++buf;
00113         } else if (base==16) {
00114             if (buf <= end)
00115                 *buf = '0';
00116             ++buf;
00117             if (buf <= end)
00118                 *buf = digits[33];
00119             ++buf;
00120         }
00121     }
00122     if (!(type & _kc_LEFT)) {
00123         while (size-- > 0) {
00124             if (buf <= end)
00125                 *buf = c;
00126             ++buf;
00127         }
00128     }
00129     while (i < precision--) {
00130         if (buf <= end)
00131             *buf = '0';
00132         ++buf;
00133     }
00134     while (i-- > 0) {
00135         if (buf <= end)
00136             *buf = tmp[i];
00137         ++buf;
00138     }
00139     while (size-- > 0) {
00140         if (buf <= end)
00141             *buf = ' ';
00142         ++buf;
00143     }
00144     return buf;
00145 }

00150     unsigned long long num;
00151     int i, base;
00152     char *str, *end, c;
00153     const char *s;
00154 
00155     int flags;      /* flags to number() */
00156 
00157     int field_width;    /* width of output field */
00158     int precision;      /* min. # of digits for integers; max
00159                    number of chars for from string */
00160     int qualifier;      /* 'h', 'l', or 'L' for integer fields */
00161                 /* 'z' support added 23/7/1999 S.H.    */
00162                 /* 'z' changed to 'Z' --davidm 1/25/99 */
00163 
00164     str = buf;
00165     end = buf + size - 1;
00166 
00167     if (end < buf - 1) {
00168         end = ((void *) -1);
00169         size = end - buf + 1;
00170     }
00171 
00172     for (; *fmt ; ++fmt) {
00173         if (*fmt != '%') {
00174             if (str <= end)
00175                 *str = *fmt;
00176             ++str;
00177             continue;
00178         }
00179 
00180         /* process flags */
00181         flags = 0;
00182         repeat:
00183             ++fmt;      /* this also skips first '%' */
00184             switch (*fmt) {
00185                 case '-': flags |= _kc_LEFT; goto repeat;
00186                 case '+': flags |= _kc_PLUS; goto repeat;
00187                 case ' ': flags |= _kc_SPACE; goto repeat;
00188                 case '#': flags |= _kc_SPECIAL; goto repeat;
00189                 case '0': flags |= _kc_ZEROPAD; goto repeat;
00190             }
00191 
00192         /* get field width */
00193         field_width = -1;
00194         if (isdigit(*fmt))
00195             field_width = skip_atoi(&fmt);
00196         else if (*fmt == '*') {
00197             ++fmt;
00198             /* it's the next argument */
00199             field_width = va_arg(args, int);
00200             if (field_width < 0) {
00201                 field_width = -field_width;
00202                 flags |= _kc_LEFT;
00203             }
00204         }
00205 
00206         /* get the precision */
00207         precision = -1;
00208         if (*fmt == '.') {
00209             ++fmt;  
00210             if (isdigit(*fmt))
00211                 precision = skip_atoi(&fmt);
00212             else if (*fmt == '*') {
00213                 ++fmt;
00214                 /* it's the next argument */
00215                 precision = va_arg(args, int);
00216             }
00217             if (precision < 0)
00218                 precision = 0;
00219         }
00220 
00221         /* get the conversion qualifier */
00222         qualifier = -1;
00223         if (*fmt == 'h' || *fmt == 'l' || *fmt == 'L' || *fmt =='Z') {
00224             qualifier = *fmt;
00225             ++fmt;
00226         }
00227 
00228         /* default base */
00229         base = 10;
00230 
00231         switch (*fmt) {
00232             case 'c':
00233                 if (!(flags & _kc_LEFT)) {
00234                     while (--field_width > 0) {
00235                         if (str <= end)
00236                             *str = ' ';
00237                         ++str;
00238                     }
00239                 }
00240                 c = (unsigned char) va_arg(args, int);
00241                 if (str <= end)
00242                     *str = c;
00243                 ++str;
00244                 while (--field_width > 0) {
00245                     if (str <= end)
00246                         *str = ' ';
00247                     ++str;
00248                 }
00249                 continue;
00250 
00251             case 's':
00252                 s = va_arg(args, char *);
00253                 if (!s)
00254                     s = "<NULL>";
00255 
00256                 len = strnlen(s, precision);
00257 
00258                 if (!(flags & _kc_LEFT)) {
00259                     while (len < field_width--) {
00260                         if (str <= end)
00261                             *str = ' ';
00262                         ++str;
00263                     }
00264                 }
00265                 for (i = 0; i < len; ++i) {
00266                     if (str <= end)
00267                         *str = *s;
00268                     ++str; ++s;
00269                 }
00270                 while (len < field_width--) {
00271                     if (str <= end)
00272                         *str = ' ';
00273                     ++str;
00274                 }
00275                 continue;
00276 
00277             case 'p':
00278                 if (field_width == -1) {
00279                     field_width = 2*sizeof(void *);
00280                     flags |= _kc_ZEROPAD;
00281                 }
00282                 str = number(str, end,
00283                         (unsigned long) va_arg(args, void *),
00284                         16, field_width, precision, flags);
00285                 continue;
00286 
00287 
00288             case 'n':
00289                 /* FIXME:
00290                 * What does C99 say about the overflow case here? */
00291                 if (qualifier == 'l') {
00292                     long * ip = va_arg(args, long *);
00293                     *ip = (str - buf);
00294                 } else if (qualifier == 'Z') {
00295                     size_t * ip = va_arg(args, size_t *);
00296                     *ip = (str - buf);
00297                 } else {
00298                     int * ip = va_arg(args, int *);
00299                     *ip = (str - buf);
00300                 }
00301                 continue;
00302 
00303             case '%':
00304                 if (str <= end)
00305                     *str = '%';
00306                 ++str;
00307                 continue;
00308 
00309                 /* integer number formats - set up the flags and "break" */
00310             case 'o':
00311                 base = 8;
00312                 break;
00313 
00314             case 'X':
00315                 flags |= _kc_LARGE;
00316             case 'x':
00317                 base = 16;
00318                 break;
00319 
00320             case 'd':
00321             case 'i':
00322                 flags |= _kc_SIGN;
00323             case 'u':
00324                 break;
00325 
00326             default:
00327                 if (str <= end)
00328                     *str = '%';
00329                 ++str;
00330                 if (*fmt) {
00331                     if (str <= end)
00332                         *str = *fmt;
00333                     ++str;
00334                 } else {
00335                     --fmt;
00336                 }
00337                 continue;
00338         }
00339         if (qualifier == 'L')
00340             num = va_arg(args, long long);
00341         else if (qualifier == 'l') {
00342             num = va_arg(args, unsigned long);
00343             if (flags & _kc_SIGN)
00344                 num = (signed long) num;
00345         } else if (qualifier == 'Z') {
00346             num = va_arg(args, size_t);
00347         } else if (qualifier == 'h') {
00348             num = (unsigned short) va_arg(args, int);
00349             if (flags & _kc_SIGN)
00350                 num = (signed short) num;
00351         } else {
00352             num = va_arg(args, unsigned int);
00353             if (flags & _kc_SIGN)
00354                 num = (signed int) num;
00355         }
00356         str = number(str, end, num, base,
00357                 field_width, precision, flags);
00358     }
00359     if (str <= end)
00360         *str = '\0';
00361     else if (size > 0)
00362         /* don't write out a null byte if the buf size is zero */
00363         *end = '\0';
00364     /* the trailing null byte doesn't count towards the total
00365     * ++str;
00366     */
00367     return str-buf;
00368 }

00565  * find_next_bit.c: fallback find next bit implementation

00577 unsigned long find_next_bit(const unsigned long *addr, unsigned long size,
00578                             unsigned long offset)
00579 {
00580     const unsigned long *p = addr + BITOP_WORD(offset);
00581     unsigned long result = offset & ~(BITS_PER_LONG-1);
00582     unsigned long tmp;
00583 
00584     if (offset >= size)
00585         return size;
00586     size -= result;
00587     offset %= BITS_PER_LONG;
00588     if (offset) {
00589         tmp = *(p++);
00590         tmp &= (~0UL << offset);
00591         if (size < BITS_PER_LONG)
00592             goto found_first;
00593         if (tmp)
00594             goto found_middle;
00595         size -= BITS_PER_LONG;
00596         result += BITS_PER_LONG;
00597     }
00598     while (size & ~(BITS_PER_LONG-1)) {
00599         if ((tmp = *(p++)))
00600             goto found_middle;
00601         result += BITS_PER_LONG;
00602         size -= BITS_PER_LONG;
00603     }
00604     if (!size)
00605         return result;
00606     tmp = *p;
00607 
00608 found_first:
00609     tmp &= (~0UL >> (BITS_PER_LONG - size));
00610     if (tmp == 0UL)     /* Are any bits set? */
00611         return result + size;   /* Nope. */
00612 found_middle:
00613     return result + ffs(tmp);
00614 }
