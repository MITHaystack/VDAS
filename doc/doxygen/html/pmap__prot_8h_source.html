00035 /*
00036  * pmap_prot.h
00037  * Protocol for the local binder service, or pmap.
00038  *
00039  * Copyright (C) 1984, Sun Microsystems, Inc.
00040  *
00041  * The following procedures are supported by the protocol:
00042  *
00043  * PMAPPROC_NULL() returns ()
00044  *  takes nothing, returns nothing
00045  *
00046  * PMAPPROC_SET(struct pmap) returns (bool_t)
00047  *  TRUE is success, FALSE is failure.  Registers the tuple
00048  *  [prog, vers, prot, port].
00049  *
00050  * PMAPPROC_UNSET(struct pmap) returns (bool_t)
00051  *  TRUE is success, FALSE is failure.  Un-registers pair
00052  *  [prog, vers].  prot and port are ignored.
00053  *
00054  * PMAPPROC_GETPORT(struct pmap) returns (long unsigned).
00055  *  0 is failure.  Otherwise returns the port number where the pair
00056  *  [prog, vers] is registered.  It may lie!
00057  *
00058  * PMAPPROC_DUMP() RETURNS (struct pmaplist *)
00059  *
00060  * PMAPPROC_CALLIT(unsigned, unsigned, unsigned, string<>)
00061  *  RETURNS (port, string<>);
00062  * usage: encapsulatedresults = PMAPPROC_CALLIT(prog, vers, proc, encapsulatedargs);
00063  *  Calls the procedure on the local machine.  If it is not registered,
00064  *  this procedure is quite; ie it does not return error information!!!
00065  *  This procedure only is supported on rpc/udp and calls via
00066  *  rpc/udp.  This routine only passes null authentication parameters.
00067  *  This file has no interface to xdr routines for PMAPPROC_CALLIT.
00068  *
00069  * The service supports remote procedure calls on udp/ip or tcp/ip socket 111.
00070  */
