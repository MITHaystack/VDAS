00388 static s32 e1000_acquire_swfw_sync_80003es2lan(struct e1000_hw *hw, u16 mask)
00389 {
00390     u32 swfw_sync;
00391     u32 swmask = mask;
00392     u32 fwmask = mask << 16;
00393     s32 ret_val = E1000_SUCCESS;
00394     s32 i = 0, timeout = 50;
00395 
00396     while (i < timeout) {
00397         if (e1000e_get_hw_semaphore(hw)) {
00398             ret_val = -E1000_ERR_SWFW_SYNC;
00399             goto out;
00400         }
00401 
00402         swfw_sync = er32(SW_FW_SYNC);
00403         if (!(swfw_sync & (fwmask | swmask)))
00404             break;
00405 
00406         /*
00407          * Firmware currently using resource (fwmask)
00408          * or other software thread using resource (swmask)
00409          */
00410         e1000e_put_hw_semaphore(hw);
00411         mdelay(5);
00412         i++;
00413     }
00414 
00415     if (i == timeout) {
00416         e_dbg("Driver can't access resource, SW_FW_SYNC timeout.\n");
00417         ret_val = -E1000_ERR_SWFW_SYNC;
00418         goto out;
00419     }
00420 
00421     swfw_sync |= swmask;
00422     ew32(SW_FW_SYNC, swfw_sync);
00423 
00424     e1000e_put_hw_semaphore(hw);
00425 
00426 out:
00427     return ret_val;
00428 }
