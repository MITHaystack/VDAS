00131     u_int flags, const char *description, char *errbuf)
00132 {
00133     pcap_t *p;
00134     pcap_if_t *curdev, *prevdev, *nextdev;
00135     int this_instance;
00136 
00137     /*
00138      * Is there already an entry in the list for this interface?
00139      */
00140     for (curdev = *alldevs; curdev != NULL; curdev = curdev->next) {
00141         if (strcmp(name, curdev->name) == 0)
00142             break;  /* yes, we found it */
00143     }
00144 
00145     if (curdev == NULL) {
00146         /*
00147          * No, we didn't find it.
00148          *
00149          * Can we open this interface for live capture?
00150          *
00151          * We do this check so that interfaces that are
00152          * supplied by the interface enumeration mechanism
00153          * we're using but that don't support packet capture
00154          * aren't included in the list.  Loopback interfaces
00155          * on Solaris are an example of this; we don't just
00156          * omit loopback interfaces on all platforms because
00157          * you *can* capture on loopback interfaces on some
00158          * OSes.
00159          *
00160          * On OS X, we don't do this check if the device
00161          * name begins with "wlt"; at least some versions
00162          * of OS X offer monitor mode capturing by having
00163          * a separate "monitor mode" device for each wireless
00164          * adapter, rather than by implementing the ioctls
00165          * that {Free,Net,Open,DragonFly}BSD provide.
00166          * Opening that device puts the adapter into monitor
00167          * mode, which, at least for some adapters, causes
00168          * them to deassociate from the network with which
00169          * they're associated.
00170          *
00171          * Instead, we try to open the corresponding "en"
00172          * device (so that we don't end up with, for users
00173          * without sufficient privilege to open capture
00174          * devices, a list of adapters that only includes
00175          * the wlt devices).
00176          */
00177 #ifdef __APPLE__
00178         if (strncmp(name, "wlt", 3) == 0) {
00179             char *en_name;
00180             size_t en_name_len;
00181 
00182             /*
00183              * Try to allocate a buffer for the "en"
00184              * device's name.
00185              */
00186             en_name_len = strlen(name) - 1;
00187             en_name = malloc(en_name_len + 1);
00188             if (en_name == NULL) {
00189                 (void)snprintf(errbuf, PCAP_ERRBUF_SIZE,
00190                     "malloc: %s", pcap_strerror(errno));
00191                 return (-1);
00192             }
00193             strcpy(en_name, "en");
00194             strcat(en_name, name + 3);
00195             p = pcap_open_live(en_name, 68, 0, 0, errbuf);
00196             free(en_name);
00197         } else
00198 #endif /* __APPLE */
00199         p = pcap_open_live(name, 68, 0, 0, errbuf);
00200         if (p == NULL) {
00201             /*
00202              * No.  Don't bother including it.
00203              * Don't treat this as an error, though.
00204              */
00205             *curdev_ret = NULL;
00206             return (0);
00207         }
00208         pcap_close(p);
00209 
00210         /*
00211          * Yes, we can open it.
00212          * Allocate a new entry.
00213          */
00214         curdev = malloc(sizeof(pcap_if_t));
00215         if (curdev == NULL) {
00216             (void)snprintf(errbuf, PCAP_ERRBUF_SIZE,
00217                 "malloc: %s", pcap_strerror(errno));
00218             return (-1);
00219         }
00220 
00221         /*
00222          * Fill in the entry.
00223          */
00224         curdev->next = NULL;
00225         curdev->name = strdup(name);
00226         if (curdev->name == NULL) {
00227             (void)snprintf(errbuf, PCAP_ERRBUF_SIZE,
00228                 "malloc: %s", pcap_strerror(errno));
00229             free(curdev);
00230             return (-1);
00231         }
00232         if (description != NULL) {
00233             /*
00234              * We have a description for this interface.
00235              */
00236             curdev->description = strdup(description);
00237             if (curdev->description == NULL) {
00238                 (void)snprintf(errbuf, PCAP_ERRBUF_SIZE,
00239                     "malloc: %s", pcap_strerror(errno));
00240                 free(curdev->name);
00241                 free(curdev);
00242                 return (-1);
00243             }
00244         } else {
00245             /*
00246              * We don't.
00247              */
00248             curdev->description = NULL;
00249         }
00250         curdev->addresses = NULL;   /* list starts out as empty */
00251         curdev->flags = 0;
00252         if (ISLOOPBACK(name, flags))
00253             curdev->flags |= PCAP_IF_LOOPBACK;
00254 
00255         /*
00256          * Add it to the list, in the appropriate location.
00257          * First, get the instance number of this interface.
00258          */
00259         this_instance = get_instance(name);
00260 
00261         /*
00262          * Now look for the last interface with an instance number
00263          * less than or equal to the new interface's instance
00264          * number - except that non-loopback interfaces are
00265          * arbitrarily treated as having interface numbers less
00266          * than those of loopback interfaces, so the loopback
00267          * interfaces are put at the end of the list.
00268          *
00269          * We start with "prevdev" being NULL, meaning we're before
00270          * the first element in the list.
00271          */
00272         prevdev = NULL;
00273         for (;;) {
00274             /*
00275              * Get the interface after this one.
00276              */
00277             if (prevdev == NULL) {
00278                 /*
00279                  * The next element is the first element.
00280                  */
00281                 nextdev = *alldevs;
00282             } else
00283                 nextdev = prevdev->next;
00284 
00285             /*
00286              * Are we at the end of the list?
00287              */
00288             if (nextdev == NULL) {
00289                 /*
00290                  * Yes - we have to put the new entry
00291                  * after "prevdev".
00292                  */
00293                 break;
00294             }
00295 
00296             /*
00297              * Is the new interface a non-loopback interface
00298              * and the next interface a loopback interface?
00299              */
00300             if (!(curdev->flags & PCAP_IF_LOOPBACK) &&
00301                 (nextdev->flags & PCAP_IF_LOOPBACK)) {
00302                 /*
00303                  * Yes, we should put the new entry
00304                  * before "nextdev", i.e. after "prevdev".
00305                  */
00306                 break;
00307             }
00308 
00309             /*
00310              * Is the new interface's instance number less
00311              * than the next interface's instance number,
00312              * and is it the case that the new interface is a
00313              * non-loopback interface or the next interface is
00314              * a loopback interface?
00315              *
00316              * (The goal of both loopback tests is to make
00317              * sure that we never put a loopback interface
00318              * before any non-loopback interface and that we
00319              * always put a non-loopback interface before all
00320              * loopback interfaces.)
00321              */
00322             if (this_instance < get_instance(nextdev->name) &&
00323                 (!(curdev->flags & PCAP_IF_LOOPBACK) ||
00324                    (nextdev->flags & PCAP_IF_LOOPBACK))) {
00325                 /*
00326                  * Yes - we should put the new entry
00327                  * before "nextdev", i.e. after "prevdev".
00328                  */
00329                 break;
00330             }
00331 
00332             prevdev = nextdev;
00333         }
00334 
00335         /*
00336          * Insert before "nextdev".
00337          */
00338         curdev->next = nextdev;
00339 
00340         /*
00341          * Insert after "prevdev" - unless "prevdev" is null,
00342          * in which case this is the first interface.
00343          */
00344         if (prevdev == NULL) {
00345             /*
00346              * This is the first interface.  Pass back a
00347              * pointer to it, and put "curdev" before
00348              * "nextdev".
00349              */
00350             *alldevs = curdev;
00351         } else
00352             prevdev->next = curdev;
00353     }
00354 
00355     *curdev_ret = curdev;
00356     return (0);
00357 }
