00549 {

00551     struct e1000_hw *hw = &adapter->hw;
00552     u16 *eeprom_buff;
00553     int first_word;

00557

00562 
00563     first_word = eeprom->offset >> 1;
00564     last_word = (eeprom->offset + eeprom->len - 1) >> 1;

00567             (last_word - first_word + 1), GFP_KERNEL);

00572         ret_val = e1000_read_nvm(hw, first_word,

00580                 break;
00581         }
00582     }
00583 
00584     if (ret_val) {

00588     } else {
00589         /* Device's eeprom is always little-endian, word addressable */
00590         for (i = 0; i < last_word - first_word + 1; i++)
00591             le16_to_cpus(&eeprom_buff[i]);
00592     }
00593 
00594     memcpy(bytes, (u8 *)eeprom_buff + (eeprom->offset & 1), eeprom->len);
00595     kfree(eeprom_buff);
00596 
00597     return ret_val;

00600 static int e1000_set_eeprom(struct net_device *netdev,
00601                 struct ethtool_eeprom *eeprom, u8 *bytes)
00602 {
00603     struct e1000_adapter *adapter = netdev_priv(netdev);
00604     struct e1000_hw *hw = &adapter->hw;
00605     u16 *eeprom_buff;
00606     void *ptr;
00607     int max_len;
00608     int first_word;
00609     int last_word;
00610     int ret_val = 0;
00611     u16 i;
00612 
00613     if (eeprom->len == 0)
00614         return -EOPNOTSUPP;
00615 
00616     if (eeprom->magic != (adapter->pdev->vendor | (adapter->pdev->device << 16)))
00617         return -EFAULT;
00618 
00619     max_len = hw->nvm.word_size * 2;
00620 
00621     first_word = eeprom->offset >> 1;
00622     last_word = (eeprom->offset + eeprom->len - 1) >> 1;
00623     eeprom_buff = kmalloc(max_len, GFP_KERNEL);
00624     if (!eeprom_buff)
00625         return -ENOMEM;
00626 
00627     ptr = (void *)eeprom_buff;
00628 
00629     if (eeprom->offset & 1) {
00630         /* need read/modify/write of first changed EEPROM word */
00631         /* only the second byte of the word is being modified */
00632         ret_val = e1000_read_nvm(hw, first_word, 1, &eeprom_buff[0]);
00633         ptr++;
00634     }
00635     if (((eeprom->offset + eeprom->len) & 1) && (ret_val == 0))
00636         /* need read/modify/write of last changed EEPROM word */
00637         /* only the first byte of the word is being modified */
00638         ret_val = e1000_read_nvm(hw, last_word, 1,
00639                   &eeprom_buff[last_word - first_word]);
00640 
00641     if (ret_val)
00642         goto out;
00643 
00644     /* Device's eeprom is always little-endian, word addressable */
00645     for (i = 0; i < last_word - first_word + 1; i++)
00646         le16_to_cpus(&eeprom_buff[i]);
00647 
00648     memcpy(ptr, bytes, eeprom->len);
00649 
00650     for (i = 0; i < last_word - first_word + 1; i++)
00651         eeprom_buff[i] = cpu_to_le16(eeprom_buff[i]);
00652 
00653     ret_val = e1000_write_nvm(hw, first_word,
00654                   last_word - first_word + 1, eeprom_buff);
00655 
00656     if (ret_val)
00657         goto out;
00658 
00659     /*
00660      * Update the checksum over the first part of the EEPROM if needed
00661      * and flush shadow RAM for applicable controllers
00662      */
00663     if ((first_word <= NVM_CHECKSUM_REG) ||
00664         (hw->mac.type == e1000_82583) ||
00665         (hw->mac.type == e1000_82574) ||
00666         (hw->mac.type == e1000_82573))
00667         ret_val = e1000e_update_nvm_checksum(hw);
00668 
00669 out:
00670     kfree(eeprom_buff);
00671     return ret_val;
00672 }

01323 static int e1000_integrated_phy_loopback(struct e1000_adapter *adapter)
01324 {
01325     struct e1000_hw *hw = &adapter->hw;
01326     u32 ctrl_reg = 0;
01327     u16 phy_reg = 0;
01328     s32 ret_val = 0;
01329 
01330     hw->mac.autoneg = 0;
01331 
01332     if (hw->phy.type == e1000_phy_ife) {
01333         /* force 100, set loopback */
01334         e1e_wphy(hw, PHY_CONTROL, 0x6100);
01335 
01336         /* Now set up the MAC to the same speed/duplex as the PHY. */
01337         ctrl_reg = er32(CTRL);
01338         ctrl_reg &= ~E1000_CTRL_SPD_SEL; /* Clear the speed sel bits */
01339         ctrl_reg |= (E1000_CTRL_FRCSPD | /* Set the Force Speed Bit */
01340                  E1000_CTRL_FRCDPX | /* Set the Force Duplex Bit */
01341                  E1000_CTRL_SPD_100 |/* Force Speed to 100 */
01342                  E1000_CTRL_FD);     /* Force Duplex to FULL */
01343 
01344         ew32(CTRL, ctrl_reg);
01345         udelay(500);
01346 
01347         return 0;
01348     }
01349 
01350     /* Specific PHY configuration for loopback */
01351     switch (hw->phy.type) {
01352     case e1000_phy_m88:
01353         /* Auto-MDI/MDIX Off */
01354         e1e_wphy(hw, M88E1000_PHY_SPEC_CTRL, 0x0808);
01355         /* reset to update Auto-MDI/MDIX */
01356         e1e_wphy(hw, PHY_CONTROL, 0x9140);
01357         /* autoneg off */
01358         e1e_wphy(hw, PHY_CONTROL, 0x8140);
01359         break;
01360     case e1000_phy_gg82563:
01361         e1e_wphy(hw, GG82563_PHY_KMRN_MODE_CTRL, 0x1CC);
01362         break;
01363     case e1000_phy_bm:
01364         /* Set Default MAC Interface speed to 1GB */
01365         e1e_rphy(hw, PHY_REG(2, 21), &phy_reg);
01366         phy_reg &= ~0x0007;
01367         phy_reg |= 0x006;
01368         e1e_wphy(hw, PHY_REG(2, 21), phy_reg);
01369         /* Assert SW reset for above settings to take effect */
01370         e1000e_commit_phy(hw);
01371         mdelay(1);

01373         e1e_rphy(hw, PHY_REG(769, 16), &phy_reg);
01374         e1e_wphy(hw, PHY_REG(769, 16), phy_reg | 0x000C);
01375         /* Set Link Up (in force link) */
01376         e1e_rphy(hw, PHY_REG(776, 16), &phy_reg);
01377         e1e_wphy(hw, PHY_REG(776, 16), phy_reg | 0x0040);
01378         /* Force Link */
01379         e1e_rphy(hw, PHY_REG(769, 16), &phy_reg);
01380         e1e_wphy(hw, PHY_REG(769, 16), phy_reg | 0x0040);

01382         e1e_rphy(hw, PHY_REG(769, 20), &phy_reg);
01383         e1e_wphy(hw, PHY_REG(769, 20), phy_reg | 0x0400);

01390             e_err("Cannot setup 1Gbps loopback.\n");
01391             return ret_val;
01392         }

01394         hw->phy.ops.release(hw);
01395         break;
01396     case e1000_phy_82579:
01397         /* Disable PHY energy detect power down */
01398         e1e_rphy(hw, PHY_REG(0, 21), &phy_reg);
01399         e1e_wphy(hw, PHY_REG(0, 21), phy_reg & ~(1 << 3));
01400         /* Disable full chip energy detect */
01401         e1e_rphy(hw, PHY_REG(776, 18), &phy_reg);
01402         e1e_wphy(hw, PHY_REG(776, 18), phy_reg | 1);

01406     default:
01407         break;
01408     }
01409 
01410     /* force 1000, set loopback */
01411     e1e_wphy(hw, PHY_CONTROL, 0x4140);
01412     mdelay(250);
01413 
01414     /* Now set up the MAC to the same speed/duplex as the PHY. */
01415     ctrl_reg = er32(CTRL);
01416     ctrl_reg &= ~E1000_CTRL_SPD_SEL; /* Clear the speed sel bits */
01417     ctrl_reg |= (E1000_CTRL_FRCSPD | /* Set the Force Speed Bit */
01418              E1000_CTRL_FRCDPX | /* Set the Force Duplex Bit */
01419              E1000_CTRL_SPD_1000 |/* Force Speed to 1000 */
01420              E1000_CTRL_FD);     /* Force Duplex to FULL */
01421 
01422     if (adapter->flags & FLAG_IS_ICH)
01423         ctrl_reg |= E1000_CTRL_SLU; /* Set Link Up */
01424

01426         hw->phy.type == e1000_phy_m88) {

01428     } else {
01429         /*
01430          * Set the ILOS bit on the fiber Nic if half duplex link is
01431          * detected.
01432          */
01433         if ((er32(STATUS) & E1000_STATUS_FD) == 0)

01435     }
01436 
01437     ew32(CTRL, ctrl_reg);
01438 
01439     /*
01440      * Disable the receiver on the PHY so when a cable is plugged in, the
01441      * PHY does not begin to autoneg when a cable is reconnected to the NIC.
01442      */
01443     if (hw->phy.type == e1000_phy_m88)
01444         e1000_phy_disable_receiver(adapter);
01445 
01446     udelay(500);
01447 
01448     return 0;
01449 }

01620 static int e1000_run_loopback_test(struct e1000_adapter *adapter)
01621 {
01622     struct e1000_ring *tx_ring = &adapter->test_tx_ring;
01623     struct e1000_ring *rx_ring = &adapter->test_rx_ring;
01624     struct pci_dev *pdev = adapter->pdev;
01625     struct e1000_hw *hw = &adapter->hw;
01626     int i, j, k, l;
01627     int lc;
01628     int good_cnt;
01629     int ret_val = 0;
01630     unsigned long time;
01631 
01632     ew32(RDT(0), rx_ring->count - 1);
01633 
01634     /*
01635      * Calculate the loop count based on the largest descriptor ring
01636      * The idea is to wrap the largest ring a number of times using 64
01637      * send/receive pairs during each loop
01638      */
01639 
01640     if (rx_ring->count <= tx_ring->count)
01641         lc = ((tx_ring->count / 64) * 2) + 1;
01642     else
01643         lc = ((rx_ring->count / 64) * 2) + 1;
01644 
01645     k = 0;
01646     l = 0;
01647     for (j = 0; j <= lc; j++) { /* loop count loop */
01648         for (i = 0; i < 64; i++) { /* send the packets */
01649             e1000_create_lbtest_frame(tx_ring->buffer_info[k].skb,
01650                           1024);
01651             dma_sync_single_for_device(pci_dev_to_dev(pdev),
01652                     tx_ring->buffer_info[k].dma,
01653                     tx_ring->buffer_info[k].length,
01654                     DMA_TO_DEVICE);
01655             k++;
01656             if (k == tx_ring->count)
01657                 k = 0;
01658         }
01659         ew32(TDT(0), k);
01660         msleep(200);
01661         time = jiffies; /* set the start time for the receive */
01662         good_cnt = 0;
01663         do { /* receive the sent packets */
01664             dma_sync_single_for_cpu(pci_dev_to_dev(pdev),
01665                     rx_ring->buffer_info[l].dma, 2048,
01666                     DMA_FROM_DEVICE);
01667 
01668             ret_val = e1000_check_lbtest_frame(
01669                     rx_ring->buffer_info[l].skb, 1024);
01670             if (!ret_val)
01671                 good_cnt++;
01672             l++;
01673             if (l == rx_ring->count)
01674                 l = 0;
01675             /*
01676              * time + 20 msecs (200 msecs on 2.4) is more than
01677              * enough time to complete the receives, if it's
01678              * exceeded, break and error off
01679              */
01680         } while ((good_cnt < 64) && !time_after(jiffies, time + 20));
01681         if (good_cnt != 64) {
01682             ret_val = 13; /* ret_val is the same as mis-compare */
01683             break;
01684         }
01685         if (jiffies >= (time + 20)) {
01686             ret_val = 14; /* error code for time out error */
01687             break;
