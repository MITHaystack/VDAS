00158 s32 ixgbe_setup_sfp_modules_82599(struct ixgbe_hw *hw)
00159 {
00160     s32 ret_val = 0;
00161     u32 reg_anlp1 = 0;
00162     u32 i = 0;
00163     u16 list_offset, data_offset, data_value;
00164 
00165     if (hw->phy.sfp_type != ixgbe_sfp_type_unknown) {
00166         ixgbe_init_mac_link_ops_82599(hw);
00167 
00168         hw->phy.ops.reset = NULL;
00169 
00170         ret_val = ixgbe_get_sfp_init_sequence_offsets(hw, &list_offset,
00171                                                       &data_offset);
00172         if (ret_val != 0)
00173             goto setup_sfp_out;
00174 
00175         /* PHY config will finish before releasing the semaphore */
00176         ret_val = ixgbe_acquire_swfw_sync(hw, IXGBE_GSSR_MAC_CSR_SM);
00177         if (ret_val != 0) {
00178             ret_val = IXGBE_ERR_SWFW_SYNC;
00179             goto setup_sfp_out;
00180         }
00181 
00182         hw->eeprom.ops.read(hw, ++data_offset, &data_value);
00183         while (data_value != 0xffff) {
00184             IXGBE_WRITE_REG(hw, IXGBE_CORECTL, data_value);
00185             IXGBE_WRITE_FLUSH(hw);
00186             hw->eeprom.ops.read(hw, ++data_offset, &data_value);
00187         }
00188 
00189         /* Release the semaphore */
00190         ixgbe_release_swfw_sync(hw, IXGBE_GSSR_MAC_CSR_SM);
00191         /* Delay obtaining semaphore again to allow FW access */
00192         msleep(hw->eeprom.semaphore_delay);
00193 
00194         /* Now restart DSP by setting Restart_AN and clearing LMS */
00195         IXGBE_WRITE_REG(hw, IXGBE_AUTOC, ((IXGBE_READ_REG(hw,
00196                         IXGBE_AUTOC) & ~IXGBE_AUTOC_LMS_MASK) |
00197                         IXGBE_AUTOC_AN_RESTART));
00198 
00199         /* Wait for AN to leave state 0 */
00200         for (i = 0; i < 10; i++) {
00201             msleep(4);
00202             reg_anlp1 = IXGBE_READ_REG(hw, IXGBE_ANLP1);
00203             if (reg_anlp1 & IXGBE_ANLP1_AN_STATE_MASK)
00204                 break;
00205         }
00206         if (!(reg_anlp1 & IXGBE_ANLP1_AN_STATE_MASK)) {
00207             hw_dbg(hw, "sfp module setup not complete\n");
00208             ret_val = IXGBE_ERR_SFP_SETUP_NOT_COMPLETE;
00209             goto setup_sfp_out;
00210         }
00211 
00212         /* Restart DSP by setting Restart_AN and return to SFI mode */
00213         IXGBE_WRITE_REG(hw, IXGBE_AUTOC, (IXGBE_READ_REG(hw,
00214                         IXGBE_AUTOC) | IXGBE_AUTOC_LMS_10G_SERIAL |
00215                         IXGBE_AUTOC_AN_RESTART));
00216     }
00217 
00218 setup_sfp_out:
00219     return ret_val;
00220 }

00301     s32 status = 0;
00302     u32 autoc = 0;
00303 
00304     /* Check if 1G SFP module. */
00305     if (hw->phy.sfp_type == ixgbe_sfp_type_1g_cu_core0 ||
00306         hw->phy.sfp_type == ixgbe_sfp_type_1g_cu_core1) {
00307         *speed = IXGBE_LINK_SPEED_1GB_FULL;
00308         *negotiation = true;

00310     }
00311 
00312     /*
00313      * Determine link capabilities based on the stored value of AUTOC,

00318         autoc = hw->mac.orig_autoc;

00320         autoc = IXGBE_READ_REG(hw, IXGBE_AUTOC);

00323     case IXGBE_AUTOC_LMS_1G_LINK_NO_AN:
00324         *speed = IXGBE_LINK_SPEED_1GB_FULL;
00325         *negotiation = false;

00329         *speed = IXGBE_LINK_SPEED_10GB_FULL;

00334         *speed = IXGBE_LINK_SPEED_1GB_FULL;
00335         *negotiation = true;
00336         break;
00337

00339         *speed = IXGBE_LINK_SPEED_10GB_FULL;

00563     u32 i = 0;
00564     bool link_up = false;
00565     bool negotiation;
00566 
00567     /* Mask off requested but non-supported speeds */
00568     status = ixgbe_get_link_capabilities(hw, &link_speed, &negotiation);
00569     if (status != 0)
00570         return status;
00571 
00572     speed &= link_speed;
00573 
00574     /*
00575      * Try each speed one by one, highest priority first.  We do this in
00576      * software because 10gb fiber doesn't support speed autonegotiation.
00577      */
00578     if (speed & IXGBE_LINK_SPEED_10GB_FULL) {
00579         speedcnt++;
00580         highest_link_speed = IXGBE_LINK_SPEED_10GB_FULL;
00581 
00582         /* If we already have link at this speed, just jump out */
00583         status = ixgbe_check_link(hw, &link_speed, &link_up, false);
00584         if (status != 0)
00585             return status;
00586 
00587         if ((link_speed == IXGBE_LINK_SPEED_10GB_FULL) && link_up)
00588             goto out;
00589 
00590         /* Set the module link speed */
00591         esdp_reg |= (IXGBE_ESDP_SDP5_DIR | IXGBE_ESDP_SDP5);
00592         IXGBE_WRITE_REG(hw, IXGBE_ESDP, esdp_reg);
00593         IXGBE_WRITE_FLUSH(hw);
00594 
00595         /* Allow module to change analog characteristics (1G->10G) */
00596         msleep(40);
00597 
00598         status = ixgbe_setup_mac_link_82599(hw,
00599                         IXGBE_LINK_SPEED_10GB_FULL,
00600                         autoneg,
00601                         autoneg_wait_to_complete);
00602         if (status != 0)
00603             return status;
00604 
00605         /* Flap the tx laser if it has not already been done */
00606         ixgbe_flap_tx_laser(hw);
00607 
00608         /*
00609          * Wait for the controller to acquire link.  Per IEEE 802.3ap,
00610          * Section 73.10.2, we may have to wait up to 500ms if KR is
00611          * attempted.  82599 uses the same timing for 10g SFI.
00612          */
00613         for (i = 0; i < 5; i++) {
00614             /* Wait for the link partner to also set speed */
00615             msleep(100);
00616

00618             status = ixgbe_check_link(hw, &link_speed,
00619                                       &link_up, false);
00620             if (status != 0)
00621                 return status;
00622 
00623             if (link_up)
00624                 goto out;
00625         }
00626     }
00627

00629         speedcnt++;

00631             highest_link_speed = IXGBE_LINK_SPEED_1GB_FULL;
00632 
00633         /* If we already have link at this speed, just jump out */
00634         status = ixgbe_check_link(hw, &link_speed, &link_up, false);
00635         if (status != 0)
00636             return status;
00637 
00638         if ((link_speed == IXGBE_LINK_SPEED_1GB_FULL) && link_up)
00639             goto out;
00640 
00641         /* Set the module link speed */
00642         esdp_reg &= ~IXGBE_ESDP_SDP5;
00643         esdp_reg |= IXGBE_ESDP_SDP5_DIR;
00644         IXGBE_WRITE_REG(hw, IXGBE_ESDP, esdp_reg);
00645         IXGBE_WRITE_FLUSH(hw);
00646 
00647         /* Allow module to change analog characteristics (10G->1G) */
00648         msleep(40);
00649 
00650         status = ixgbe_setup_mac_link_82599(hw,
00651                             IXGBE_LINK_SPEED_1GB_FULL,
00652                             autoneg,
00653                             autoneg_wait_to_complete);
00654         if (status != 0)
00655             return status;
00656 
00657         /* Flap the tx laser if it has not already been done */
00658         ixgbe_flap_tx_laser(hw);
00659 
00660         /* Wait for the link partner to also set speed */
00661         msleep(100);
00662 
00663         /* If we have link, just jump out */
00664         status = ixgbe_check_link(hw, &link_speed, &link_up, false);
00665         if (status != 0)
00666             return status;
00667 
00668         if (link_up)
00669             goto out;
00670     }
00671 
00672     /*
00673      * We didn't get link.  Configure back to the highest speed we tried,
00674      * (if there was more than one).  We call ourselves back with just the
00675      * single highest speed that the user requested.
00676      */
00677     if (speedcnt > 1)
00678         status = ixgbe_setup_mac_link_multispeed_fiber(hw,
00679                 highest_link_speed, autoneg, autoneg_wait_to_complete);
00680 
00681 out:
00682     /* Set autoneg_advertised value based on input link speed */
00683     hw->phy.autoneg_advertised = 0;
00684 
00685     if (speed & IXGBE_LINK_SPEED_10GB_FULL)
00686         hw->phy.autoneg_advertised |= IXGBE_LINK_SPEED_10GB_FULL;
00687 
00688     if (speed & IXGBE_LINK_SPEED_1GB_FULL)
00689         hw->phy.autoneg_advertised |= IXGBE_LINK_SPEED_1GB_FULL;
00690 
00691     return status;
00692 }

00703 s32 ixgbe_setup_mac_link_smartspeed(struct ixgbe_hw *hw,
00704                      ixgbe_link_speed speed, bool autoneg,
00705                      bool autoneg_wait_to_complete)
00706 {
00707     s32 status = 0;
00708     ixgbe_link_speed link_speed = IXGBE_LINK_SPEED_UNKNOWN;
00709     s32 i, j;
00710     bool link_up = false;
00711     u32 autoc_reg = IXGBE_READ_REG(hw, IXGBE_AUTOC);
00712 
00713      /* Set autoneg_advertised value based on input link speed */
00714     hw->phy.autoneg_advertised = 0;
00715 
00716     if (speed & IXGBE_LINK_SPEED_10GB_FULL)
00717         hw->phy.autoneg_advertised |= IXGBE_LINK_SPEED_10GB_FULL;
00718 
00719     if (speed & IXGBE_LINK_SPEED_1GB_FULL)
00720         hw->phy.autoneg_advertised |= IXGBE_LINK_SPEED_1GB_FULL;
00721 
00722     if (speed & IXGBE_LINK_SPEED_100_FULL)
00723         hw->phy.autoneg_advertised |= IXGBE_LINK_SPEED_100_FULL;
00724 
00725     /*
00726      * Implement Intel SmartSpeed algorithm.  SmartSpeed will reduce the
00727      * autoneg advertisement if link is unable to be established at the
00728      * highest negotiated rate.  This can sometimes happen due to integrity
00729      * issues with the physical media connection.
00730      */
00731 
00732     /* First, try to get link with full advertisement */
00733     hw->phy.smart_speed_active = false;
00734     for (j = 0; j < IXGBE_SMARTSPEED_MAX_RETRIES; j++) {
00735         status = ixgbe_setup_mac_link_82599(hw, speed, autoneg,
00736                             autoneg_wait_to_complete);
00737         if (status != 0)
00738             goto out;
00739 
00740         /*
00741          * Wait for the controller to acquire link.  Per IEEE 802.3ap,
00742          * Section 73.10.2, we may have to wait up to 500ms if KR is
00743          * attempted, or 200ms if KX/KX4/BX/BX4 is attempted, per
00744          * Table 9 in the AN MAS.
00745          */
00746         for (i = 0; i < 5; i++) {
00747             msleep(100);
00748 
00749             /* If we have link, just jump out */
00750             status = ixgbe_check_link(hw, &link_speed, &link_up,
00751                           false);
00752             if (status != 0)
00753                 goto out;
00754 
00755             if (link_up)
00756                 goto out;
00757         }
00758     }
00759 
00760     /*
00761      * We didn't get link.  If we advertised KR plus one of KX4/KX
00762      * (or BX4/BX), then disable KR and try again.
00763      */
00764     if (((autoc_reg & IXGBE_AUTOC_KR_SUPP) == 0) ||
00765         ((autoc_reg & IXGBE_AUTOC_KX4_KX_SUPP_MASK) == 0))
00766         goto out;
00767 
00768     /* Turn SmartSpeed on to disable KR support */
00769     hw->phy.smart_speed_active = true;
00770     status = ixgbe_setup_mac_link_82599(hw, speed, autoneg,
00771                         autoneg_wait_to_complete);
00772     if (status != 0)
00773         goto out;
00774 
00775     /*
00776      * Wait for the controller to acquire link.  600ms will allow for
00777      * the AN link_fail_inhibit_timer as well for multiple cycles of
00778      * parallel detect, both 10g and 1g. This allows for the maximum
00779      * connect attempts as defined in the AN MAS table 73-7.
00780      */
00781     for (i = 0; i < 6; i++) {
00782         msleep(100);
00783 
00784         /* If we have link, just jump out */
00785         status = ixgbe_check_link(hw, &link_speed, &link_up, false);
00786         if (status != 0)
00787             goto out;
00788 
00789         if (link_up)
00790             goto out;
00791     }
00792 
00793     /* We didn't get link.  Turn SmartSpeed back off. */
00794     hw->phy.smart_speed_active = false;
00795     status = ixgbe_setup_mac_link_82599(hw, speed, autoneg,
00796                         autoneg_wait_to_complete);
00797 
00798 out:
00799     if (link_up && (link_speed == IXGBE_LINK_SPEED_1GB_FULL))
00800         hw_dbg(hw, "Smartspeed has downgraded the link speed "
00801         "from the maximum advertised\n");
00802     return status;
00803 }

00815                                      ixgbe_link_speed speed, bool autoneg,
00816                                      bool autoneg_wait_to_complete)

00832     if (status != 0)
00833         goto out;
00834 
00835     speed &= link_capabilities;
00836 
00837     if (speed == IXGBE_LINK_SPEED_UNKNOWN) {
00838         status = IXGBE_ERR_LINK_SETUP;
00839         goto out;
00840     }
00841 
00842     /* Use stored value (EEPROM defaults) of AUTOC to find KR/KX4 support*/
00843     if (hw->mac.orig_link_settings_stored)

00846         orig_autoc = autoc;

00877                 autoc |= IXGBE_AUTOC_LMS_1G_AN;

00879                 autoc |= IXGBE_AUTOC_LMS_1G_LINK_NO_AN;

00891                 link_mode == IXGBE_AUTOC_LMS_KX4_KX_KR_1G_AN ||
00892                 link_mode == IXGBE_AUTOC_LMS_KX4_KX_KR_SGMII) {
00893                 links_reg = 0; /*Just in case Autoneg time=0*/

00895                     links_reg =
00896                            IXGBE_READ_REG(hw, IXGBE_LINKS);
00897                     if (links_reg & IXGBE_LINKS_KX_AN_COMP)
00898                         break;
00899                     msleep(100);
00900                 }
00901                 if (!(links_reg & IXGBE_LINKS_KX_AN_COMP)) {
00902                     status =
00903                         IXGBE_ERR_AUTONEG_NOT_COMPLETE;
00904                     hw_dbg(hw, "Autoneg did not complete.\n");

00906             }
00907         }

00909         /* Add delay to filter out noises during initial link setup */
00910         msleep(50);
00911     }

00913 out:

00950 s32 ixgbe_reset_hw_82599(struct ixgbe_hw *hw)
00951 {
00952     s32 status = 0;
00953     u32 ctrl;
00954     u32 i;
00955     u32 autoc;
00956     u32 autoc2;
00957 
00958     /* Call adapter stop to disable tx/rx and clear interrupts */
00959     hw->mac.ops.stop_adapter(hw);
00960 
00961     /* PHY ops must be identified and initialized prior to reset */
00962 
00963     /* Identify PHY and related function pointers */
00964     status = hw->phy.ops.init(hw);
00965 
00966     if (status == IXGBE_ERR_SFP_NOT_SUPPORTED)
00967         goto reset_hw_out;
00968 
00969     /* Setup SFP module if there is one present. */
00970     if (hw->phy.sfp_setup_needed) {
00971         status = hw->mac.ops.setup_sfp(hw);
00972         hw->phy.sfp_setup_needed = false;
00973     }
00974 
00975     if (status == IXGBE_ERR_SFP_NOT_SUPPORTED)
00976         goto reset_hw_out;
00977 
00978     /* Reset PHY */
00979     if (hw->phy.reset_disable == false && hw->phy.ops.reset != NULL)
00980         hw->phy.ops.reset(hw);
00981 
00982     /*
00983      * Prevent the PCI-E bus from from hanging by disabling PCI-E master
00984      * access and verify no pending requests before reset
00985      */
00986     ixgbe_disable_pcie_master(hw);
00987 
00988 mac_reset_top:
00989     /*
00990      * Issue global reset to the MAC.  This needs to be a SW reset.
00991      * If link reset is used, it might reset the MAC when mng is using it
00992      */
00993     ctrl = IXGBE_READ_REG(hw, IXGBE_CTRL);
00994     IXGBE_WRITE_REG(hw, IXGBE_CTRL, (ctrl | IXGBE_CTRL_RST));
00995     IXGBE_WRITE_FLUSH(hw);
00996 
00997     /* Poll for reset bit to self-clear indicating reset is complete */
00998     for (i = 0; i < 10; i++) {
00999         udelay(1);
01000         ctrl = IXGBE_READ_REG(hw, IXGBE_CTRL);
01001         if (!(ctrl & IXGBE_CTRL_RST))
01002             break;
01003     }
01004     if (ctrl & IXGBE_CTRL_RST) {
01005         status = IXGBE_ERR_RESET_FAILED;
01006         hw_dbg(hw, "Reset polling failed to complete.\n");
01007     }
01008 
01009     /*
01010      * Double resets are required for recovery from certain error
01011      * conditions.  Between resets, it is necessary to stall to allow time
01012      * for any pending HW events to complete.  We use 1usec since that is
01013      * what is needed for ixgbe_disable_pcie_master().  The second reset
01014      * then clears out any effects of those events.
01015      */
01016     if (hw->mac.flags & IXGBE_FLAGS_DOUBLE_RESET_REQUIRED) {
01017         hw->mac.flags &= ~IXGBE_FLAGS_DOUBLE_RESET_REQUIRED;
01018         udelay(1);
01019         goto mac_reset_top;
01020     }
01021 
01022     msleep(50);
01023 
01024     /*
01025      * Store the original AUTOC/AUTOC2 values if they have not been
01026      * stored off yet.  Otherwise restore the stored original
01027      * values since the reset operation sets back to defaults.
01028      */
01029     autoc = IXGBE_READ_REG(hw, IXGBE_AUTOC);
01030     autoc2 = IXGBE_READ_REG(hw, IXGBE_AUTOC2);
01031     if (hw->mac.orig_link_settings_stored == false) {
01032         hw->mac.orig_autoc = autoc;
01033         hw->mac.orig_autoc2 = autoc2;
01034         hw->mac.orig_link_settings_stored = true;
01035     } else {
01036         if (autoc != hw->mac.orig_autoc)
01037             IXGBE_WRITE_REG(hw, IXGBE_AUTOC, (hw->mac.orig_autoc |
01038                     IXGBE_AUTOC_AN_RESTART));
01039 
01040         if ((autoc2 & IXGBE_AUTOC2_UPPER_MASK) !=
01041             (hw->mac.orig_autoc2 & IXGBE_AUTOC2_UPPER_MASK)) {
01042             autoc2 &= ~IXGBE_AUTOC2_UPPER_MASK;
01043             autoc2 |= (hw->mac.orig_autoc2 &
01044                        IXGBE_AUTOC2_UPPER_MASK);
01045             IXGBE_WRITE_REG(hw, IXGBE_AUTOC2, autoc2);
01046         }
01047     }
01048 
01049     /* Store the permanent mac address */
01050     hw->mac.ops.get_mac_addr(hw, hw->mac.perm_addr);
01051 
01052     /*
01053      * Store MAC address from RAR0, clear receive address registers, and
01054      * clear the multicast table.  Also reset num_rar_entries to 128,
01055      * since we modify this value when programming the SAN MAC address.
01056      */
01057     hw->mac.num_rar_entries = 128;
01058     hw->mac.ops.init_rx_addrs(hw);
01059 
01060     /* Store the permanent SAN mac address */
01061     hw->mac.ops.get_san_mac_addr(hw, hw->mac.san_addr);
01062 
01063     /* Add the SAN MAC address to the RAR only if it's a valid address */
01064     if (ixgbe_validate_mac_addr(hw->mac.san_addr) == 0) {
01065         hw->mac.ops.set_rar(hw, hw->mac.num_rar_entries - 1,
01066                             hw->mac.san_addr, 0, IXGBE_RAH_AV);
01067 
01068         /* Reserve the last RAR for the SAN MAC address */
01069         hw->mac.num_rar_entries--;
01070     }
01071 
01072     /* Store the alternative WWNN/WWPN prefix */
01073     hw->mac.ops.get_wwn_prefix(hw, &hw->mac.wwnn_prefix,
01074                                    &hw->mac.wwpn_prefix);
01075 
01076 reset_hw_out:
01077     return status;
01078 }

01084 s32 ixgbe_reinit_fdir_tables_82599(struct ixgbe_hw *hw)
01085 {
01086     int i;
01087     u32 fdirctrl = IXGBE_READ_REG(hw, IXGBE_FDIRCTRL);
01088     fdirctrl &= ~IXGBE_FDIRCTRL_INIT_DONE;
01089 
01090     /*
01091      * Before starting reinitialization process,
01092      * FDIRCMD.CMD must be zero.
01093      */
01094     for (i = 0; i < IXGBE_FDIRCMD_CMD_POLL; i++) {
01095         if (!(IXGBE_READ_REG(hw, IXGBE_FDIRCMD) &
01096               IXGBE_FDIRCMD_CMD_MASK))
01097             break;
01098         udelay(10);
01099     }
01100     if (i >= IXGBE_FDIRCMD_CMD_POLL) {
01101         hw_dbg(hw, "Flow Director previous command isn't complete, "
01102                  "aborting table re-initialization. \n");
01103         return IXGBE_ERR_FDIR_REINIT_FAILED;
01104     }
01105 
01106     IXGBE_WRITE_REG(hw, IXGBE_FDIRFREE, 0);
01107     IXGBE_WRITE_FLUSH(hw);
01108     /*
01109      * 82599 adapters flow director init flow cannot be restarted,
01110      * Workaround 82599 silicon errata by performing the following steps
01111      * before re-writing the FDIRCTRL control register with the same value.
01112      * - write 1 to bit 8 of FDIRCMD register &
01113      * - write 0 to bit 8 of FDIRCMD register
01114      */
01115     IXGBE_WRITE_REG(hw, IXGBE_FDIRCMD,
01116                     (IXGBE_READ_REG(hw, IXGBE_FDIRCMD) |
01117                      IXGBE_FDIRCMD_CLEARHT));
01118     IXGBE_WRITE_FLUSH(hw);
01119     IXGBE_WRITE_REG(hw, IXGBE_FDIRCMD,
01120                     (IXGBE_READ_REG(hw, IXGBE_FDIRCMD) &
01121                      ~IXGBE_FDIRCMD_CLEARHT));
01122     IXGBE_WRITE_FLUSH(hw);
01123     /*
01124      * Clear FDIR Hash register to clear any leftover hashes
01125      * waiting to be programmed.
01126      */
01127     IXGBE_WRITE_REG(hw, IXGBE_FDIRHASH, 0x00);
01128     IXGBE_WRITE_FLUSH(hw);
01129 
01130     IXGBE_WRITE_REG(hw, IXGBE_FDIRCTRL, fdirctrl);
01131     IXGBE_WRITE_FLUSH(hw);
01132 
01133     /* Poll init-done after we write FDIRCTRL register */
01134     for (i = 0; i < IXGBE_FDIR_INIT_DONE_POLL; i++) {
01135         if (IXGBE_READ_REG(hw, IXGBE_FDIRCTRL) &
01136                            IXGBE_FDIRCTRL_INIT_DONE)
01137             break;
01138         udelay(10);
01139     }
01140     if (i >= IXGBE_FDIR_INIT_DONE_POLL) {
01141         hw_dbg(hw, "Flow Director Signature poll time exceeded!\n");
01142         return IXGBE_ERR_FDIR_REINIT_FAILED;
01143     }

01343 u16 ixgbe_atr_compute_hash_82599(union ixgbe_atr_input *atr_input,
01344                  u32 key)
01345 {
01346     /*
01347      * The algorithm is as follows:
01348      *    Hash[15:0] = Sum { S[n] x K[n+16] }, n = 0...350
01349      *    where Sum {A[n]}, n = 0...n is bitwise XOR of A[0], A[1]...A[n]
01350      *    and A[n] x B[n] is bitwise AND between same length strings
01351      *
01352      *    K[n] is 16 bits, defined as:
01353      *       for n modulo 32 >= 15, K[n] = K[n % 32 : (n % 32) - 15]
01354      *       for n modulo 32 < 15, K[n] =
01355      *             K[(n % 32:0) | (31:31 - (14 - (n % 32)))]
01356      *
01357      *    S[n] is 16 bits, defined as:
01358      *       for n >= 15, S[n] = S[n:n - 15]
01359      *       for n < 15, S[n] = S[(n:0) | (350:350 - (14 - n))]
01360      *
01361      *    To simplify for programming, the algorithm is implemented
01362      *    in software this way:
01363      *
01364      *    key[31:0], hi_hash_dword[31:0], lo_hash_dword[31:0], hash[15:0]
01365      *
01366      *    for (i = 0; i < 352; i+=32)
01367      *        hi_hash_dword[31:0] ^= Stream[(i+31):i];
01368      *
01369      *    lo_hash_dword[15:0]  ^= Stream[15:0];
01370      *    lo_hash_dword[15:0]  ^= hi_hash_dword[31:16];
01371      *    lo_hash_dword[31:16] ^= hi_hash_dword[15:0];
01372      *
01373      *    hi_hash_dword[31:0]  ^= Stream[351:320];
01374      *
01375      *    if(key[0])
01376      *        hash[15:0] ^= Stream[15:0];
01377      *
01378      *    for (i = 0; i < 16; i++) {
01379      *        if (key[i])
01380      *            hash[15:0] ^= lo_hash_dword[(i+15):i];
01381      *        if (key[i + 16])
01382      *            hash[15:0] ^= hi_hash_dword[(i+15):i];
01383      *    }
01384      *
01385      */
01386     __be32 common_hash_dword = 0;
01387     u32 hi_hash_dword, lo_hash_dword;
01388     u16 hash_result = 0;
01389     u8  i;
01390 
01391     /*
01392      * the hi_hash_dword starts with vlan_id, the lo_hash_dword starts
01393      * and ends with it, the vlan at the end is added via the word swapped
01394      * xor with the hi_hash_dword a few lines down.
01395      */
01396     hi_hash_dword = IXGBE_NTOHL(atr_input->dword_stream[0]) & 0x0000FFFF;
01397     lo_hash_dword = hi_hash_dword;
01398 
01399     /* generate common hash dword */
01400     for (i = 1; i < 11; i++)
01401         common_hash_dword ^= (u32)atr_input->dword_stream[i];
01402     hi_hash_dword ^= IXGBE_NTOHL(common_hash_dword);
01403 
01404     /* low dword is word swapped version of common with vlan added */
01405     lo_hash_dword ^= (hi_hash_dword >> 16) | (hi_hash_dword << 16);
01406 
01407     /* hi dword is common dword with l4type and vm_pool shifted */
01408     hi_hash_dword ^= IXGBE_NTOHL(atr_input->dword_stream[10]) << 16;
01409 
01410     /*
01411      * Process all 32 bits of the 2 keys 2 bits at a time
01412      *
01413      * Bit flip vlan from hash result if hash key has bit 0 set, the
01414      * reason for doing this is because the hash generation shouldn't
01415      * start until bit 1 in the stream so we need to cancel out a vlan
01416      * if it was added starting at bit 0.
01417      */
01418     if (key & 0x0001) {
01419         hash_result ^= IXGBE_NTOHL(atr_input->dword_stream[0]) &
01420                    0x0FFFF;
01421         hash_result ^= lo_hash_dword;
01422     }
01423     if (key & 0x00010000)
01424         hash_result ^= hi_hash_dword;
01425 
01426     /* process the remaining bits in the key */
01427     for (i = 1; i < 16; i++) {
01428         if (key & (0x0001 << i))
01429             hash_result ^= lo_hash_dword >> i;
01430         if (key & (0x00010000 << i))
01431             hash_result ^= hi_hash_dword >> i;
01432     }
01433 
01434     return hash_result;
01435 }

01905 /*
01906  * These two macros are meant to address the fact that we have registers
01907  * that are either all or in part big-endian.  As a result on big-endian
01908  * systems we will end up byte swapping the value to little-endian before
01909  * it is byte swapped again and written to the hardware in the original
01910  * big-endian format.
01911  */
01912 #define IXGBE_STORE_AS_BE32(_value) \
01913     (((u32)(_value) >> 24) | (((u32)(_value) & 0x00FF0000) >> 8) | \
01914      (((u32)(_value) & 0x0000FF00) << 8) | ((u32)(_value) << 24))
01915 
01916 #define IXGBE_WRITE_REG_BE32(a, reg, value) \
01917     IXGBE_WRITE_REG((a), (reg), IXGBE_STORE_AS_BE32(IXGBE_NTOHL(value)))
01918 
01919 #define IXGBE_STORE_AS_BE16(_value) \
01920     (((u16)(_value) >> 8) | ((u16)(_value) << 8))

01934 s32 ixgbe_fdir_add_perfect_filter_82599(struct ixgbe_hw *hw,
01935                                       union ixgbe_atr_input *input,
01936                                       struct ixgbe_atr_input_masks *input_masks,
01937                                       u16 soft_id, u8 queue)
01938 {
01939     u32 fdircmd = 0;
01940     u32 fdirhash;
01941     u32 fdirport, fdirtcpm;
01942     u32 fdirvlan;
01943     /* start with VLAN, flex bytes, VM pool, and IPv6 destination masked */
01944     u32 fdirm = IXGBE_FDIRM_VLANID | IXGBE_FDIRM_VLANP | IXGBE_FDIRM_FLEX |
01945             IXGBE_FDIRM_POOL | IXGBE_FDIRM_DIPv6;
01946 
01947     /*
01948      * Check flow_type formatting, and bail out before we touch the hardware
01949      * if there's a configuration issue
01950      */
01951     switch (input->formatted.flow_type) {
01952     case IXGBE_ATR_FLOW_TYPE_IPV4:
01953         /* use the L4 protocol mask for raw IPv4/IPv6 traffic */
01954         fdirm |= IXGBE_FDIRM_L4P;
01955     case IXGBE_ATR_FLOW_TYPE_SCTPV4:
01956         if (input_masks->dst_port_mask || input_masks->src_port_mask) {
01957             hw_dbg(hw, " Error on src/dst port mask\n");
01958             return IXGBE_ERR_CONFIG;
01959         }
01960     case IXGBE_ATR_FLOW_TYPE_TCPV4:
01961     case IXGBE_ATR_FLOW_TYPE_UDPV4:
01962         break;
01963     default:
01964         hw_dbg(hw, " Error on flow type input\n");
01965         return IXGBE_ERR_CONFIG;
01966     }
01967 
01968     /*
01969      * Program the relevant mask registers.  If src/dst_port or src/dst_addr
01970      * are zero, then assume a full mask for that field.  Also assume that
01971      * a VLAN of 0 is unspecified, so mask that out as well.  L4type
01972      * cannot be masked out in this implementation.
01973      *
01974      * This also assumes IPv4 only.  IPv6 masking isn't supported at this
01975      * point in time.
01976      */
01977 
01978     /* Program FDIRM */
01979     switch (IXGBE_NTOHS(input_masks->vlan_id_mask) & 0xEFFF) {
01980     case 0xEFFF:
01981         /* Unmask VLAN ID - bit 0 and fall through to unmask prio */
01982         fdirm &= ~IXGBE_FDIRM_VLANID;
01983     case 0xE000:
01984         /* Unmask VLAN prio - bit 1 */
01985         fdirm &= ~IXGBE_FDIRM_VLANP;
01986         break;
01987     case 0x0FFF:
01988         /* Unmask VLAN ID - bit 0 */
01989         fdirm &= ~IXGBE_FDIRM_VLANID;
01990         break;
01991     case 0x0000:
01992         /* do nothing, vlans already masked */
01993         break;
01994     default:
01995         hw_dbg(hw, " Error on VLAN mask\n");
01996         return IXGBE_ERR_CONFIG;
01997     }
01998 
01999     if (input_masks->flex_mask & 0xFFFF) {
02000         if ((input_masks->flex_mask & 0xFFFF) != 0xFFFF) {
02001             hw_dbg(hw, " Error on flexible byte mask\n");
02002             return IXGBE_ERR_CONFIG;
02003         }
02004         /* Unmask Flex Bytes - bit 4 */
02005         fdirm &= ~IXGBE_FDIRM_FLEX;
02006     }
02007 
02008     /* Now mask VM pool and destination IPv6 - bits 5 and 2 */
02009     IXGBE_WRITE_REG(hw, IXGBE_FDIRM, fdirm);
02010 
02011     /* store the TCP/UDP port masks, bit reversed from port layout */
02012     fdirtcpm = ixgbe_get_fdirtcpm_82599(input_masks);
02013 
02014     /* write both the same so that UDP and TCP use the same mask */
02015     IXGBE_WRITE_REG(hw, IXGBE_FDIRTCPM, ~fdirtcpm);
02016     IXGBE_WRITE_REG(hw, IXGBE_FDIRUDPM, ~fdirtcpm);
02017 
02018     /* store source and destination IP masks (big-enian) */
02019     IXGBE_WRITE_REG_BE32(hw, IXGBE_FDIRSIP4M,
02020                  ~input_masks->src_ip_mask[0]);
02021     IXGBE_WRITE_REG_BE32(hw, IXGBE_FDIRDIP4M,
02022                  ~input_masks->dst_ip_mask[0]);
02023 
02024     /* Apply masks to input data */
02025     // printk("input->formatted.vlan_id=%04X / %04X\n", input->formatted.vlan_id, input_masks->vlan_id_mask);
02026     input->formatted.vlan_id &= input_masks->vlan_id_mask;
02027     // printk("input->formatted.vlan_id=%04X\n", input->formatted.vlan_id);
02028     input->formatted.flex_bytes &= input_masks->flex_mask;
02029     input->formatted.src_port &= input_masks->src_port_mask;
02030     input->formatted.dst_port &= input_masks->dst_port_mask;
02031     input->formatted.src_ip[0] &= input_masks->src_ip_mask[0];
02032     input->formatted.dst_ip[0] &= input_masks->dst_ip_mask[0];
02033

02035     fdirvlan =
02036         IXGBE_STORE_AS_BE16(IXGBE_NTOHS(input->formatted.flex_bytes));
02037     fdirvlan <<= IXGBE_FDIRVLAN_FLEX_SHIFT;
02038     fdirvlan |= IXGBE_NTOHS(input->formatted.vlan_id);
02039     IXGBE_WRITE_REG(hw, IXGBE_FDIRVLAN, fdirvlan);
02040 
02041     /* record source and destination port (little-endian)*/
02042     fdirport = IXGBE_NTOHS(input->formatted.dst_port);
02043     fdirport <<= IXGBE_FDIRPORT_DESTINATION_SHIFT;
02044     fdirport |= IXGBE_NTOHS(input->formatted.src_port);
02045     IXGBE_WRITE_REG(hw, IXGBE_FDIRPORT, fdirport);
02046 
02047     /* record the first 32 bits of the destination address (big-endian) */
02048     IXGBE_WRITE_REG_BE32(hw, IXGBE_FDIRIPDA, input->formatted.dst_ip[0]);
02049 
02050     /* record the source address (big-endian) */
02051     IXGBE_WRITE_REG_BE32(hw, IXGBE_FDIRIPSA, input->formatted.src_ip[0]);
02052 
02053     /* configure FDIRHASH register */
02054     fdirhash = ixgbe_atr_compute_sig_hash_82599(input);
02055 
02056     /* we only want the bucket hash so drop the upper 16 bits */
02057     fdirhash &= IXGBE_ATR_HASH_MASK;
02058     fdirhash |= soft_id << IXGBE_FDIRHASH_SIG_SW_INDEX_SHIFT;
02059     IXGBE_WRITE_REG(hw, IXGBE_FDIRHASH, fdirhash);
02060 
02061     fdircmd |= IXGBE_FDIRCMD_CMD_ADD_FLOW;
02062     fdircmd |= IXGBE_FDIRCMD_FILTER_UPDATE;
02063     fdircmd |= IXGBE_FDIRCMD_LAST;
02064     fdircmd |= IXGBE_FDIRCMD_QUEUE_EN;
02065     fdircmd |= input->formatted.flow_type << IXGBE_FDIRCMD_FLOW_TYPE_SHIFT;
02066     fdircmd |= queue << IXGBE_FDIRCMD_RX_QUEUE_SHIFT;
02067 
02068     IXGBE_WRITE_REG(hw, IXGBE_FDIRCMD, fdircmd);
