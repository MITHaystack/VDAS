00264 static s32 e1000_reset_hw_82540(struct e1000_hw *hw)
00265 {
00266     u32 ctrl, manc;
00267     s32 ret_val = E1000_SUCCESS;
00268 
00269     DEBUGFUNC("e1000_reset_hw_82540");
00270 
00271     DEBUGOUT("Masking off all interrupts\n");
00272     E1000_WRITE_REG(hw, E1000_IMC, 0xFFFFFFFF);
00273 
00274     E1000_WRITE_REG(hw, E1000_RCTL, 0);
00275     E1000_WRITE_REG(hw, E1000_TCTL, E1000_TCTL_PSP);
00276     E1000_WRITE_FLUSH(hw);
00277 
00278     /*
00279      * Delay to allow any outstanding PCI transactions to complete
00280      * before resetting the device.
00281      */
00282     msec_delay(10);
00283 
00284     ctrl = E1000_READ_REG(hw, E1000_CTRL);
00285 
00286     DEBUGOUT("Issuing a global reset to 82540/82545/82546 MAC\n");
00287     switch (hw->mac.type) {
00288     case e1000_82545_rev_3:
00289     case e1000_82546_rev_3:
00290         E1000_WRITE_REG(hw, E1000_CTRL_DUP, ctrl | E1000_CTRL_RST);
00291         break;
00292     default:
00293         /*
00294          * These controllers can't ack the 64-bit write when
00295          * issuing the reset, so we use IO-mapping as a
00296          * workaround to issue the reset.
00297          */
00298         E1000_WRITE_REG_IO(hw, E1000_CTRL, ctrl | E1000_CTRL_RST);
00299         break;
00300     }
00301 
00302     /* Wait for EEPROM reload */
00303     msec_delay(5);
00304 
00305     /* Disable HW ARPs on ASF enabled adapters */
00306     manc = E1000_READ_REG(hw, E1000_MANC);
00307     manc &= ~E1000_MANC_ARP_EN;
00308     E1000_WRITE_REG(hw, E1000_MANC, manc);
00309 
00310     E1000_WRITE_REG(hw, E1000_IMC, 0xffffffff);
00311     E1000_READ_REG(hw, E1000_ICR);
00312 
00313     return ret_val;
00314 }

00322 static s32 e1000_init_hw_82540(struct e1000_hw *hw)
00323 {
00324     struct e1000_mac_info *mac = &hw->mac;
00325     u32 txdctl, ctrl_ext;
00326     s32 ret_val = E1000_SUCCESS;
00327     u16 i;
00328 
00329     DEBUGFUNC("e1000_init_hw_82540");
00330 
00331     /* Initialize identification LED */
00332     ret_val = mac->ops.id_led_init(hw);
00333     if (ret_val) {
00334         DEBUGOUT("Error initializing identification LED\n");
00335         /* This is not fatal and we should not stop init due to this */
00336     }
00337 
00338     /* Disabling VLAN filtering */
00339     DEBUGOUT("Initializing the IEEE VLAN\n");
00340     if (mac->type < e1000_82545_rev_3)
00341         E1000_WRITE_REG(hw, E1000_VET, 0);
00342 
00343     mac->ops.clear_vfta(hw);
00344 
00345     /* Setup the receive address. */
00346     e1000_init_rx_addrs_generic(hw, mac->rar_entry_count);
00347 
00348     /* Zero out the Multicast HASH table */
00349     DEBUGOUT("Zeroing the MTA\n");
00350     for (i = 0; i < mac->mta_reg_count; i++) {
00351         E1000_WRITE_REG_ARRAY(hw, E1000_MTA, i, 0);
00352         /*
00353          * Avoid back to back register writes by adding the register
00354          * read (flush).  This is to protect against some strange
00355          * bridge configurations that may issue Memory Write Block
00356          * (MWB) to our register space.  The *_rev_3 hardware at
00357          * least doesn't respond correctly to every other dword in an
00358          * MWB to our register space.
00359          */
00360         E1000_WRITE_FLUSH(hw);
00361     }
00362 
00363     if (mac->type < e1000_82545_rev_3)
00364         e1000_pcix_mmrbc_workaround_generic(hw);
00365 
00366     /* Setup link and flow control */
00367     ret_val = mac->ops.setup_link(hw);
00368 
00369     txdctl = E1000_READ_REG(hw, E1000_TXDCTL(0));
00370     txdctl = (txdctl & ~E1000_TXDCTL_WTHRESH) |
00371              E1000_TXDCTL_FULL_TX_DESC_WB;
00372     E1000_WRITE_REG(hw, E1000_TXDCTL(0), txdctl);
00373 
00374     /*
00375      * Clear all of the statistics registers (clear on read).  It is
00376      * important that we do this after we have tried to establish link
00377      * because the symbol error count will increment wildly if there
00378      * is no link.
00379      */
00380     e1000_clear_hw_cntrs_82540(hw);
00381 
00382     if ((hw->device_id == E1000_DEV_ID_82546GB_QUAD_COPPER) ||
00383         (hw->device_id == E1000_DEV_ID_82546GB_QUAD_COPPER_KSP3)) {
00384         ctrl_ext = E1000_READ_REG(hw, E1000_CTRL_EXT);
00385         /*
00386          * Relaxed ordering must be disabled to avoid a parity
00387          * error crash in a PCI slot.
00388          */
00389         ctrl_ext |= E1000_CTRL_EXT_RO_DIS;
00390         E1000_WRITE_REG(hw, E1000_CTRL_EXT, ctrl_ext);
00391     }
00392 
00393     return ret_val;
00394 }
