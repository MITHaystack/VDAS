00452 static void
00453 l2tp_avp_print(const u_char *dat, int length)
00454 {
00455     u_int len;
00456     const u_int16_t *ptr = (u_int16_t *)dat;
00457     u_int16_t attr_type;
00458     int hidden = FALSE;
00459 
00460     if (length <= 0) {
00461         return;
00462     }
00463 
00464     printf(" ");
00465 
00466     TCHECK(*ptr);   /* Flags & Length */
00467     len = EXTRACT_16BITS(ptr) & L2TP_AVP_HDR_LEN_MASK;
00468 
00469     /* If it is not long enough to contain the header, we'll give up. */
00470     if (len < 6)
00471         goto trunc;
00472 
00473     /* If it goes past the end of the remaining length of the packet,
00474        we'll give up. */
00475     if (len > (u_int)length)
00476         goto trunc;
00477 
00478     /* If it goes past the end of the remaining length of the captured
00479        data, we'll give up. */
00480     TCHECK2(*ptr, len);
00481     /* After this point, no need to worry about truncation */
00482 
00483     if (EXTRACT_16BITS(ptr) & L2TP_AVP_HDR_FLAG_MANDATORY) {
00484         printf("*");
00485     }
00486     if (EXTRACT_16BITS(ptr) & L2TP_AVP_HDR_FLAG_HIDDEN) {
00487         hidden = TRUE;
00488         printf("?");
00489     }
00490     ptr++;
00491 
00492     if (EXTRACT_16BITS(ptr)) {
00493         /* Vendor Specific Attribute */
00494             printf("VENDOR%04x:", EXTRACT_16BITS(ptr)); ptr++;
00495         printf("ATTR%04x", EXTRACT_16BITS(ptr)); ptr++;
00496         printf("(");
00497         print_octets((u_char *)ptr, len-6);
