00080 void NetReader::run() {
00081   LOG4CXX_INFO(logger, "NetReader Running...");
00082 
00083   Timer run_timer;
00084   Timer command_timer;
00085   
00086   try {
00087     // Main processing loop.
00088     while (_running) {
00089 
00090       // State machine.
00091       switch (_state) {
00092 
00093       case READ_FROM_NETWORK:
00094     if (command_timer.elapsed() > _command_interval) {
00095       command_timer.restart();
00096       continue;
00097     }
00098     handle_read_from_network();
00099     break;
00100 
00101       case IDLE:
00102     if (command_timer.elapsed() > _command_interval) {
00103       command_timer.restart();
00104       continue;
00105     }
00106     handle_idle();
00107     break;
00108 
00109       case STOP:
00110     handle_stop();
00111     break;
00112 
00113       default:
00114     LOG4CXX_ERROR(logger, "Unknown state.");
00115     break;
00116       }
00117     }
00118 
00119     LOG4CXX_DEBUG(logger, "elapsed run time: " << run_timer.elapsed());
00120   } catch(std::exception &ex) {
00121     LOG4CXX_ERROR(logger, "error: " << ex.what());
00122   }
00123 }

00179   int bytes_read = 0;
00180   static int remainder_len = 0;
00181   static boost::uint8_t remainder_buf[9000];
00182   static bool first_write = true;
00183   static boost::uint8_t net_buf[9000];

00185 
00186 #define DYNAMIC_BUFFER
00187 #ifdef DYNAMIC_BUFFER
00188   int bytes_left = _buffer_size;
00189   boost::uint8_t* file_buf = _fw->malloc_buffer();
00190   if (!file_buf) {
00191     LOG4CXX_ERROR(logger, "File buffer's full: timedout");
00192     return;
00193   }
00194   assert(file_buf);
00195 #else
00196   const int BUFFER_SIZE = 1048576;
00197   int bytes_left = BUFFER_SIZE;
00198   static boost::uint8_t file_buf[BUFFER_SIZE];
00199 #endif
00200   boost::uint64_t dropped_packets = 0;
00201   boost::uint64_t num_packets = 0;
00202   boost::uint64_t num_bytes = 0;
00203 
00204   if (first_write) {
00205     memcpy(remainder_buf, PCAP_HEADER, PCAP_HEADER_LENGTH);
00206     remainder_len = PCAP_HEADER_LENGTH;
00207     first_write = false;
00208   }
00209 
00210   // Copy partial packet.
00211   if (remainder_len > 0) {
00212     memcpy(file_buf, remainder_buf, remainder_len);
00213     bytes_read += remainder_len;
00214     bytes_left -= remainder_len;
00215     remainder_len = 0;
00216   }
00217 
00218   // Fill the new file_buf;
00219   while (bytes_left > 0) {
00220     if (_ring->get_next_packet(&hdr, &net_buf[PCAP_PACKET_HEADER_LENGTH],
00221                    _snaplen) > 0) {
00222       // Successful read.
00223 
00224       // Extract offsets etc. from pfring structures.
00225       struct pfring_extended_pkthdr& pep(hdr.extended_hdr);
00226       struct pkt_parsing_info& ppi(pep.parsed_pkt);
00227       struct pkt_offset po(ppi.offset);
00228       const boost::uint16_t eth_offset(po.eth_offset);
00229       const boost::uint16_t l3_offset(po.l3_offset);
00230       const boost::uint16_t l4_offset(po.l4_offset);
00231       const boost::uint16_t payload_offset(po.payload_offset);
00232 
00233       // Get payload information.
00234       // FIXME payload_ptr = &_net_buf[payload_offset];
00235       payload_ptr = net_buf;
00236       // FIXME payload_length = hdr.caplen - payload_offset;
00237       payload_length = hdr.len + PCAP_PACKET_HEADER_LENGTH;
00238 
00239       pph.ts_sec = hdr.ts.tv_sec;
00240       pph.ts_usec = hdr.ts.tv_usec;
00241       pph.incl_len = hdr.len;
00242       pph.orig_len = hdr.len;
00243       memcpy(net_buf, &pph, PCAP_PACKET_HEADER_LENGTH);
00244 
00245       // Check for validity of capture.
00246       if (hdr.caplen != _snaplen) {
00247       // || hdr.caplen != hdr.len
00248       // || payload_length != _payload_length) {
00249     LOG4CXX_DEBUG(logger, "Short capture(caplen/snaplen"
00250               "len/payload_length/PAYLOAD_LENGTH) -> ("
00251               << hdr.caplen << "/" << _snaplen << "/"
00252               << hdr.len << "/" << payload_length << "/"
00253               << _payload_length << ")");
00254     continue;
00255       }
00256 
00257       // Update stats.
00258       num_packets++;
00259       num_bytes += hdr.len;
00260     } else {
00261       LOG4CXX_ERROR(logger, "Error while calling get_next_packet(): "
00262             << strerror(errno));
00263       continue;
00264     }
00265 
00266     // Accumulate or flush data.
00267     if (bytes_left < payload_length) {
00268       // Copy fragment into buffer.
00269       memcpy(&file_buf[bytes_read], payload_ptr, bytes_left);
00270       bytes_read += bytes_left;
00271       
00272       // Copy remainder into remainder_buf.
00273       remainder_len = payload_length - bytes_left;
00274       memcpy(remainder_buf, payload_ptr + bytes_left, remainder_len);
00275 
00276 #define BUFFERED_WRITE      
00277 #ifdef BUFFERED_WRITE
00278       if (!_fw->write(file_buf))
00279     ++dropped_packets;
00280 #else
00281       _fw->write_unbuffered(file_buf, BUFFER_SIZE);
00282 #endif
00283       // Update stats.
00284       _sw->update(num_packets, num_bytes, dropped_packets, 0);
00285       bytes_left = 0;
00286       break;
00287     } else if (bytes_left == payload_length) {
00288       // Copy captured payload to file buffer.
00289       memcpy(&file_buf[bytes_read], payload_ptr, payload_length);
00290       bytes_read += payload_length;
00291       bytes_left -= payload_length;
00292 #ifdef BUFFERED_WRITE
00293       if (!_fw->write(file_buf))
00294     ++dropped_packets;
00295 #else
00296       _fw->write_unbuffered(file_buf, BUFFER_SIZE);
00297 #endif
00298     } else {
00299       // Copy captured payload to file buffer.
00300       memcpy(&file_buf[bytes_read], payload_ptr, payload_length);
00301       bytes_read += payload_length;
00302       bytes_left -= payload_length;
00303     }
00304   }
