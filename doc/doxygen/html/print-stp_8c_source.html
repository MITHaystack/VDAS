00138 /*
00139  * MSTP packet format
00140  * Ref. IEEE 802.1Q 2003 Ed. Section 14
00141  *
00142  * MSTP BPDU
00143  *
00144  * 2 -  bytes Protocol Id
00145  * 1 -  byte  Protocol Ver. 
00146  * 1 -  byte  BPDU tye
00147  * 1 -  byte  Flags
00148  * 8 -  bytes CIST Root Identifier
00149  * 4 -  bytes CIST External Path Cost
00150  * 8 -  bytes CIST Regional Root Identifier
00151  * 2 -  bytes CIST Port Identifier
00152  * 2 -  bytes Message Age
00153  * 2 -  bytes Max age
00154  * 2 -  bytes Hello Time
00155  * 2 -  bytes Forward delay
00156  * 1 -  byte  Version 1 length. Must be 0
00157  * 2 -  bytes Version 3 length
00158  * 1 -  byte  Config Identifier
00159  * 32 - bytes Config Name
00160  * 2 -  bytes Revision level
00161  * 16 - bytes Config Digest [MD5]
00162  * 4 -  bytes CIST Internal Root Path Cost
00163  * 8 -  bytes CIST Bridge Identifier
00164  * 1 -  byte  CIST Remaining Hops
00165  * 16 - bytes MSTI information [Max 64 MSTI, each 16 bytes]
00166  *
00167  * MSTI Payload
00168  *
00169  * 1 - byte  MSTI flag
00170  * 8 - bytes MSTI Regional Root Identifier
00171  * 4 - bytes MSTI Regional Path Cost
00172  * 1 - byte  MSTI Bridge Priority
00173  * 1 - byte  MSTI Port Priority
00174  * 1 - byte  MSTI Remaining Hops
00175  */
00176 
00177 #define MST_BPDU_MSTI_LENGTH            16
00178 #define MST_BPDU_CONFIG_INFO_LENGTH     64
00179 
00180 /* Offsets of fields from the begginning for the packet */
00181 #define MST_BPDU_VER3_LEN_OFFSET        36
00182 #define MST_BPDU_CONFIG_NAME_OFFSET     39
00183 #define MST_BPDU_CONFIG_DIGEST_OFFSET       73
00184 #define MST_BPDU_CIST_INT_PATH_COST_OFFSET  89
00185 #define MST_BPDU_CIST_BRIDGE_ID_OFFSET      93
00186 #define MST_BPDU_CIST_REMAIN_HOPS_OFFSET    101
00187 #define MST_BPDU_MSTI_OFFSET            102
00188 /* Offsets within  an MSTI */
00189 #define MST_BPDU_MSTI_ROOT_PRIO_OFFSET      1
00190 #define MST_BPDU_MSTI_ROOT_PATH_COST_OFFSET 9
00191 #define MST_BPDU_MSTI_BRIDGE_PRIO_OFFSET    13
00192 #define MST_BPDU_MSTI_PORT_PRIO_OFFSET      14
00193 #define MST_BPDU_MSTI_REMAIN_HOPS_OFFSET    15

00195 static void
00196 stp_print_mstp_bpdu(const struct stp_bpdu_ *stp_bpdu, u_int length)
00197 {
00198     const u_char    *ptr;
00199     u_int16_t       v3len;
00200     u_int16_t       len;
00201     u_int16_t       msti;
00202     u_int16_t       offset;
00203 
00204     ptr = (const u_char *)stp_bpdu;
00205     printf(", CIST Flags [%s]",
00206            bittok2str(stp_bpdu_flag_values, "none", stp_bpdu->flags));
00207 
00208     /*
00209      * in non-verbose mode just print the flags. We dont read that much
00210      * of the packet (DEFAULT_SNAPLEN) to print out cist bridge-id
00211      */
00212     if (!vflag) {
00213         return;
00214     }
00215 
00216     printf(", CIST bridge-id %s.%04x, length %u",
00217            stp_print_bridge_id(ptr + MST_BPDU_CIST_BRIDGE_ID_OFFSET),
00218            EXTRACT_16BITS(&stp_bpdu->port_id), length);
00219 
00220 
00221     printf("\n\tmessage-age %.2fs, max-age %.2fs"
00222            ", hello-time %.2fs, forwarding-delay %.2fs",
00223            (float)EXTRACT_16BITS(&stp_bpdu->message_age) / STP_TIME_BASE,
00224            (float)EXTRACT_16BITS(&stp_bpdu->max_age) / STP_TIME_BASE,
00225            (float)EXTRACT_16BITS(&stp_bpdu->hello_time) / STP_TIME_BASE,
00226            (float)EXTRACT_16BITS(&stp_bpdu->forward_delay) / STP_TIME_BASE);
00227 
00228     printf("\n\tCIST root-id %s, ext-pathcost %u int-pathcost %u",
00229            stp_print_bridge_id((const u_char *)&stp_bpdu->root_id),
00230            EXTRACT_32BITS(&stp_bpdu->root_path_cost),
00231            EXTRACT_32BITS(ptr + MST_BPDU_CIST_INT_PATH_COST_OFFSET));
00232 
00233     printf(", port-role %s",
00234            tok2str(rstp_obj_port_role_values, "Unknown",
00235                    RSTP_EXTRACT_PORT_ROLE(stp_bpdu->flags)));
00236 
00237     printf("\n\tCIST regional-root-id %s",
00238            stp_print_bridge_id((const u_char *)&stp_bpdu->bridge_id));
00239 
00240     printf("\n\tMSTP Configuration Name %s, revision %u, digest %08x%08x%08x%08x",
00241            ptr + MST_BPDU_CONFIG_NAME_OFFSET,
00242        EXTRACT_16BITS(ptr + MST_BPDU_CONFIG_NAME_OFFSET + 32),
00243        EXTRACT_32BITS(ptr + MST_BPDU_CONFIG_DIGEST_OFFSET),
00244        EXTRACT_32BITS(ptr + MST_BPDU_CONFIG_DIGEST_OFFSET + 4),
00245        EXTRACT_32BITS(ptr + MST_BPDU_CONFIG_DIGEST_OFFSET + 8),
00246        EXTRACT_32BITS(ptr + MST_BPDU_CONFIG_DIGEST_OFFSET + 12));
00247 
00248     printf("\n\tCIST remaining-hops %d", ptr[MST_BPDU_CIST_REMAIN_HOPS_OFFSET]);
00249 
00250     /* Dump all MSTI's */
00251     v3len = EXTRACT_16BITS(ptr + MST_BPDU_VER3_LEN_OFFSET);
00252     if (v3len > MST_BPDU_CONFIG_INFO_LENGTH) {
00253         len = v3len - MST_BPDU_CONFIG_INFO_LENGTH;
00254         offset = MST_BPDU_MSTI_OFFSET;
00255         while (len >= MST_BPDU_MSTI_LENGTH) {
00256             msti = EXTRACT_16BITS(ptr + offset +
00257                                   MST_BPDU_MSTI_ROOT_PRIO_OFFSET);
00258             msti = msti & 0x0FFF;
00259 
00260             printf("\n\tMSTI %d, Flags [%s], port-role %s", 
00261                    msti, bittok2str(stp_bpdu_flag_values, "none", ptr[offset]),
00262                    tok2str(rstp_obj_port_role_values, "Unknown",
00263                            RSTP_EXTRACT_PORT_ROLE(ptr[offset])));
00264             printf("\n\t\tMSTI regional-root-id %s, pathcost %u",
00265                    stp_print_bridge_id(ptr + offset +
00266                                        MST_BPDU_MSTI_ROOT_PRIO_OFFSET),
00267                    EXTRACT_32BITS(ptr + offset +
00268                                   MST_BPDU_MSTI_ROOT_PATH_COST_OFFSET));
00269             printf("\n\t\tMSTI bridge-prio %d, port-prio %d, hops %d",
00270                    ptr[offset + MST_BPDU_MSTI_BRIDGE_PRIO_OFFSET] >> 4,
00271                    ptr[offset + MST_BPDU_MSTI_PORT_PRIO_OFFSET] >> 4,
00272                    ptr[offset + MST_BPDU_MSTI_REMAIN_HOPS_OFFSET]);
00273 
00274             len -= MST_BPDU_MSTI_LENGTH;
00275             offset += MST_BPDU_MSTI_LENGTH;
00276         }
00277     }
00278 }

00280 /*
00281  * Print 802.1d / 802.1w / 802.1q (mstp) packets.
00282  */
00283 void
00284 stp_print(const u_char *p, u_int length)
00285 {
00286     const struct stp_bpdu_ *stp_bpdu;
00287     u_int16_t              mstp_len;
00288     
00289     stp_bpdu = (struct stp_bpdu_*)p;
00290 
00291     /* Minimum STP Frame size. */
00292     if (length < 4)
00293         goto trunc;
00294         
00295     if (EXTRACT_16BITS(&stp_bpdu->protocol_id)) {
00296         printf("unknown STP version, length %u", length);
00297         return;
00298     }
00299 
00300     printf("STP %s", tok2str(stp_proto_values, "Unknown STP protocol (0x%02x)",
00301                          stp_bpdu->protocol_version));
00302 
00303     switch (stp_bpdu->protocol_version) {
00304     case STP_PROTO_REGULAR:
00305     case STP_PROTO_RAPID:
00306     case STP_PROTO_MSTP:
00307         break;
00308     default:
00309         return;
00310     }
00311 
00312     printf(", %s", tok2str(stp_bpdu_type_values, "Unknown BPDU Type (0x%02x)",
00313                            stp_bpdu->bpdu_type));
00314 
00315     switch (stp_bpdu->bpdu_type) {
00316     case STP_BPDU_TYPE_CONFIG:
00317         if (length < sizeof(struct stp_bpdu_) - 1) {
00318             goto trunc;
00319         }
00320         stp_print_config_bpdu(stp_bpdu, length);
00321         break;
00322 
00323     case STP_BPDU_TYPE_RSTP:
00324         if (stp_bpdu->protocol_version == STP_PROTO_RAPID) {
00325             if (length < sizeof(struct stp_bpdu_)) {
00326                 goto trunc;
00327             }
00328             stp_print_config_bpdu(stp_bpdu, length);
00329         } else if (stp_bpdu->protocol_version == STP_PROTO_MSTP) {
00330             if (length < STP_BPDU_MSTP_MIN_LEN) {
00331                 goto trunc;
00332             }
00333             if (stp_bpdu->v1_length != 0) {
00334                 /* FIX ME: Emit a message here ? */
00335                 goto trunc;
00336             }
00337             /* Validate v3 length */
00338             mstp_len = EXTRACT_16BITS(p + MST_BPDU_VER3_LEN_OFFSET);
00339             mstp_len += 2;  /* length encoding itself is 2 bytes */
00340             if (length < (sizeof(struct stp_bpdu_) + mstp_len)) {
00341                 goto trunc;
00342             }
00343             stp_print_mstp_bpdu(stp_bpdu, length);
00344         }
00345         break;
00346 
00347     case STP_BPDU_TYPE_TOPO_CHANGE:
00348         /* always empty message - just break out */
00349         break;
00350 
00351     default:
00352         break;
00353     }
00354 
00355     return;
00356  trunc:
00357     printf("[|stp %d]", length);
00358 }
