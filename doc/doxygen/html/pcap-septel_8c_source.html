00062 /*
00063  *  Read at most max_packets from the capture queue and call the callback
00064  *  for each of them. Returns the number of packets handled, -1 if an
00065  *  error occured, or -2 if we were told to break out of the loop.
00066  */
00067 static int septel_read(pcap_t *p, int cnt, pcap_handler callback, u_char *user) {
00068 
00069   HDR *h;
00070   MSG *m;
00071   int processed = 0 ;
00072   int t = 0 ;
00073 
00074   /* identifier for the message queue of the module(upe) from which we are capturing
00075    * packets.These IDs are defined in system.txt . By default it is set to 0x2d
00076    * so change it to 0xdd for technical reason and therefore the module id for upe becomes:
00077    * LOCAL        0xdd           * upe - Example user part task */
00078   unsigned int id = 0xdd;
00079 
00080   /* process the packets */
00081   do  {
00082 
00083     unsigned short packet_len = 0;
00084     int caplen = 0;
00085     int counter = 0;
00086     struct pcap_pkthdr   pcap_header;
00087     u_char *dp ;
00088 
00089     /*
00090      * Has "pcap_breakloop()" been called?
00091      */
00092 loop:
00093     if (p->break_loop) {
00094       /*
00095        * Yes - clear the flag that indicates that
00096        * it has, and return -2 to indicate that
00097        * we were told to break out of the loop.
00098        */
00099       p->break_loop = 0;
00100       return -2;
00101     }
00102 
00103     /*repeat until a packet is read
00104      *a NULL message means :
00105      * when no packet is in queue or all packets in queue already read */
00106     do  {
00107       /* receive packet in non-blocking mode
00108        * GCT_grab is defined in the septel library software */
00109       h = GCT_grab(id);
00110 
00111       m = (MSG*)h;
00112       /* a couter is added here to avoid an infinite loop
00113        * that will cause our capture program GUI to freeze while waiting
00114        * for a packet*/
00115       counter++ ;
00116 
00117     }
00118     while  ((m == NULL)&& (counter< 100)) ;
00119 
00120     if (m != NULL) {
00121 
00122       t = h->type ;
00123 
00124       /* catch only messages with type = 0xcf00 or 0x8f01 corrsponding to ss7 messages*/
00125       /* XXX = why not use API_MSG_TX_REQ for 0xcf00 and API_MSG_RX_IND
00126        * for 0x8f01? */
00127       if ((t != 0xcf00) && (t != 0x8f01)) {
00128         relm(h);
00129         goto loop ;
00130       }
00131 
00132       /* XXX - is API_MSG_RX_IND for an MTP2 or MTP3 message? */
00133       dp = get_param(m);/* get pointer to MSG parameter area (m->param) */
00134       packet_len = m->len;
00135       caplen =  p->snapshot ;
00136 
00137 
00138       if (caplen > packet_len) {
00139 
00140         caplen = packet_len;
00141       }
00142       /* Run the packet filter if there is one. */
00143       if ((p->fcode.bf_insns == NULL) || bpf_filter(p->fcode.bf_insns, dp, packet_len, caplen)) {
00144 
00145 
00146         /*  get a time stamp , consisting of :
00147          *
00148          *  pcap_header.ts.tv_sec:
00149          *  ----------------------
00150          *   a UNIX format time-in-seconds when he packet was captured,
00151          *   i.e. the number of seconds since Epoch time (January 1,1970, 00:00:00 GMT)
00152          *
00153          *  pcap_header.ts.tv_usec :
00154          *  ------------------------
00155          *   the number of microseconds since that second
00156          *   when the packet was captured
00157          */
00158 
00159         (void)gettimeofday(&pcap_header.ts, NULL);
00160 
00161         /* Fill in our own header data */
00162         pcap_header.caplen = caplen;
00163         pcap_header.len = packet_len;
00164 
00165         /* Count the packet. */
00166         p->md.stat.ps_recv++;
00167 
00168         /* Call the user supplied callback function */
00169         callback(user, &pcap_header, dp);
00170 
00171         processed++ ;
00172 
00173       }
00174       /* after being processed the packet must be
00175        *released in order to receive another one */
00176       relm(h);
00177     }else
00178       processed++;
00179 
00180   }
00181   while (processed < cnt) ;
00182 
00183   return processed ;
00184 }

00195 /*
00196  *  Activate a handle for a live capture from the given Septel device.  Always pass a NULL device
00197  *  The promisc flag is ignored because Septel cards have built-in tracing.
00198  *  The timeout is also ignored as it is not supported in hardware.
00199  *
00200  *  See also pcap(3).
00201  */
00202 static pcap_t *septel_activate(pcap_t* handle) {
00203   /* Initialize some components of the pcap structure. */  
00204   handle->linktype = DLT_MTP2;
00205   
00206   handle->bufsize = 0;
00207 
00208   /*
00209    * "select()" and "poll()" don't work on Septel queues
00210    */
00211   handle->selectable_fd = -1;
00212 
00213   handle->read_op = septel_read;
00214   handle->inject_op = septel_inject;
00215   handle->setfilter_op = septel_setfilter;
00216   handle->set_datalink_op = NULL; /* can't change data link type */
00217   handle->getnonblock_op = pcap_getnonblock_fd;
00218   handle->setnonblock_op = septel_setnonblock;
00219   handle->stats_op = septel_stats;
00220 
00221   return 0;
00222 }
