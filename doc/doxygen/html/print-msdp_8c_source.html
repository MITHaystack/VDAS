00038 void
00039 msdp_print(const unsigned char *sp, u_int length)
00040 {
00041     unsigned int type, len;
00042 
00043     TCHECK2(*sp, 3);
00044     /* See if we think we're at the beginning of a compound packet */
00045     type = *sp;
00046     len = EXTRACT_16BITS(sp + 1);
00047     if (len > 1500 || len < 3 || type == 0 || type > MSDP_TYPE_MAX)
00048         goto trunc; /* not really truncated, but still not decodable */
00049     (void)printf(" msdp:");
00050     while (length > 0) {
00051         TCHECK2(*sp, 3);
00052         type = *sp;
00053         len = EXTRACT_16BITS(sp + 1);
00054         if (len > 1400 || vflag)
00055             printf(" [len %u]", len);
00056         if (len < 3)
00057             goto trunc;
00058         sp += 3;
00059         length -= 3;
00060         switch (type) {
00061         case 1: /* IPv4 Source-Active */
00062         case 3: /* IPv4 Source-Active Response */
00063             if (type == 1)
00064                 (void)printf(" SA");
00065             else
00066                 (void)printf(" SA-Response");
00067             TCHECK(*sp);
00068             (void)printf(" %u entries", *sp);
00069             if ((u_int)((*sp * 12) + 8) < len) {
00070                 (void)printf(" [w/data]");
00071                 if (vflag > 1) {
00072                     (void)printf(" ");
00073                     ip_print(gndo, sp + *sp * 12 + 8 - 3,
00074                              len - (*sp * 12 + 8));
00075                 }
00076             }
00077             break;
00078         case 2:
00079             (void)printf(" SA-Request");
00080             TCHECK2(*sp, 5);
00081             (void)printf(" for %s", ipaddr_string(sp + 1));
00082             break;
00083         case 4:
00084             (void)printf(" Keepalive");
00085             if (len != 3)
00086                 (void)printf("[len=%d] ", len);
00087             break;
00088         case 5:
00089             (void)printf(" Notification");
00090             break;
00091         default:
00092             (void)printf(" [type=%d len=%d]", type, len);
00093             break;
00094         }
00095         sp += (len - 3);
00096         length -= (len - 3);
00097     }
00098     return;
00099 trunc:
00100     (void)printf(" [|msdp]");
00101 }
