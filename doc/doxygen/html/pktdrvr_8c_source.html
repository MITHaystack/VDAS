00307 LOCAL __inline BOOL PktInterrupt (void)
00308 {
00309   BOOL okay;
00310 
00311 #if (DOSX & PHARLAP)
00312   _dx_real_int ((UINT)pktInfo.intr, &reg);
00313   okay = ((reg.flags & 1) == 0);  /* OK if carry clear */
00314 
00315 #elif (DOSX & DJGPP)
00316   __dpmi_int ((int)pktInfo.intr, &reg);
00317   okay = ((reg.x.flags & 1) == 0);
00318 
00319 #elif (DOSX & DOS4GW)
00320   union  REGS  r;
00321   struct SREGS s;
00322 
00323   memset (&r, 0, sizeof(r));
00324   segread (&s);
00325   r.w.ax  = 0x300;
00326   r.x.ebx = pktInfo.intr;
00327   r.w.cx  = 0;
00328   s.es    = FP_SEG (&reg);
00329   r.x.edi = FP_OFF (&reg);
00330   reg.r_flags = 0;
00331   reg.r_ss = reg.r_sp = 0;     /* DPMI host provides stack */
00332 
00333   int386x (0x31, &r, &r, &s);
00334   okay = (!r.w.cflag);
00335 
00336 #else
00337   reg.r_flags = 0;
00338   intr (pktInfo.intr, (struct REGPACK*)&reg);
00339   okay = ((reg.r_flags & 1) == 0);
00340 #endif
00341 
00342   if (okay)
00343        pktInfo.error = NULL;
00344   else pktInfo.error = PktGetErrorStr (reg.r_dx >> 8);
00345   return (okay);
00346 }

00350 /*
00351  * Search for packet driver at interrupt 60h through 80h. If ASCIIZ
00352  * string "PKT DRVR" found at offset 3 in the interrupt handler, return
00353  * interrupt number, else return zero in pktInfo.intr
00354  */
00355 PUBLIC BOOL PktSearchDriver (void)
00356 {
00357   BYTE intr  = 0x20;
00358   BOOL found = FALSE;
00359 
00360   while (!found && intr < 0xFF)
00361   {
00362     static char str[12];                 /* 3 + strlen("PKT DRVR") */
00363     static char pktStr[9] = "PKT DRVR";  /* ASCIIZ string at ofs 3 */
00364     DWORD  rp;                           /* in interrupt  routine  */
00365 
00366 #if (DOSX & PHARLAP)
00367     _dx_rmiv_get (intr, &rp);
00368     ReadRealMem (&str, (REALPTR)rp, sizeof(str));
00369 
00370 #elif (DOSX & DJGPP)
00371     __dpmi_raddr realAdr;
00372     __dpmi_get_real_mode_interrupt_vector (intr, &realAdr);
00373     rp = (realAdr.segment << 4) + realAdr.offset16;
00374     dosmemget (rp, sizeof(str), &str);
00375 
00376 #elif (DOSX & DOS4GW)
00377     rp = dpmi_get_real_vector (intr);
00378     memcpy (&str, (void*)rp, sizeof(str));
00379 
00380 #else
00381     _fmemcpy (&str, getvect(intr), sizeof(str));
00382 #endif
00383 
00384     found = memcmp (&str[3],&pktStr,sizeof(pktStr)) == 0;
00385     intr++;
00386   }
00387   pktInfo.intr = (found ? intr-1 : 0);
00388   return (found);
00389 }

01319 /*
01320  * Description:
01321  *     This routine allocates conventional memory for the specified block
01322  *     of code (which must be within the first 64K of the protected mode
01323  *     program segment) and copies the code to it.
01324  *
01325  *     The caller should free up the conventional memory block when it
01326  *     is done with the conventional memory.
01327  *
01328  *     NOTE THIS ROUTINE REQUIRES 386|DOS-EXTENDER 3.0 OR LATER.
01329  *
01330  * Calling arguments:
01331  *     start_offs      start of real mode code in program segment
01332  *     end_offs        1 byte past end of real mode code in program segment
01333  *     real_basep      returned;  real mode ptr to use as a base for the
01334  *                        real mode code (eg, to get the real mode FAR
01335  *                        addr of a function foo(), take
01336  *                        real_basep + (ULONG) foo).
01337  *                        This pointer is constructed such that
01338  *                        offsets within the real mode segment are
01339  *                        the same as the link-time offsets in the
01340  *                        protected mode program segment
01341  *     prot_basep      returned;  prot mode ptr to use as a base for getting
01342  *                        to the conventional memory, also constructed
01343  *                        so that adding the prot mode offset of a
01344  *                        function or variable to the base gets you a
01345  *                        ptr to the function or variable in the
01346  *                        conventional memory block.
01347  *     rmem_adrp       returned;  real mode para addr of allocated
01348  *                        conventional memory block, to be used to free
01349  *                        up the conventional memory when done.  DO NOT
01350  *                        USE THIS TO CONSTRUCT A REAL MODE PTR, USE
01351  *                        REAL_BASEP INSTEAD SO THAT OFFSETS WORK OUT
01352  *                        CORRECTLY.
01353  *
01354  * Returned values:
01355  *     0      if error
01356  *     1      if success
01357  */
01358 int RealCopy (ULONG    start_offs,
01359               ULONG    end_offs,
01360               REALPTR *real_basep,
01361               FARPTR  *prot_basep,
01362               USHORT  *rmem_adrp)
01363 {
01364   ULONG   rm_base;    /* base real mode para addr for accessing */
01365                       /* allocated conventional memory          */
01366   UCHAR  *source;     /* source pointer for copy                */
01367   FARPTR  destin;     /* destination pointer for copy           */
01368   ULONG   len;        /* number of bytes to copy                */
01369   ULONG   temp;
01370   USHORT  stemp;
01371 
01372   /* First check for valid inputs
01373    */
01374   if (start_offs >= end_offs || end_offs > 0x10000)
01375      return (FALSE);
01376 
01377   /* Round start_offs down to a paragraph (16-byte) boundary so we can set up
01378    * the real mode pointer easily. Round up end_offs to make sure we allocate
01379    * enough paragraphs
01380    */
01381   start_offs &= ~15;
01382   end_offs = (15 + (end_offs << 4)) >> 4;
01383 
01384   /* Allocate the conventional memory for our real mode code.  Remember to
01385    * round byte count UP to 16-byte paragraph size.  We alloc it
01386    * above the DOS data buffer so both the DOS data buffer and the appl
01387    * conventional mem block can still be resized.
01388    *
01389    * First just try to alloc it;  if we can't get it, shrink the appl mem
01390    * block down to the minimum, try to alloc the memory again, then grow the
01391    * appl mem block back to the maximum.  (Don't try to shrink the DOS data
01392    * buffer to free conventional memory;  it wouldn't be good for this routine
01393    * to have the possible side effect of making file I/O run slower.)
01394    */
01395   len = ((end_offs - start_offs) + 15) >> 4;
01396   if (_dx_real_above(len, rmem_adrp, &stemp) != _DOSE_NONE)
01397   {
01398     if (_dx_cmem_usage(0, 0, &temp, &temp) != _DOSE_NONE)
01399        return (FALSE);
01400 
01401     if (_dx_real_above(len, rmem_adrp, &stemp) != _DOSE_NONE)
01402        *rmem_adrp = 0;
01403 
01404     if (_dx_cmem_usage(0, 1, &temp, &temp) != _DOSE_NONE)
01405     {
01406       if (*rmem_adrp != 0)
01407          _dx_real_free (*rmem_adrp);
01408       return (FALSE);
01409     }
01410 
01411     if (*rmem_adrp == 0)
01412        return (FALSE);
01413   }
01414 
01415   /* Construct real mode & protected mode pointers to access the allocated
01416    * memory.  Note we know start_offs is aligned on a paragraph (16-byte)
01417    * boundary, because we rounded it down.
01418    *
01419    * We make the offsets come out rights by backing off the real mode selector
01420    * by start_offs.
01421    */
01422   rm_base = ((ULONG) *rmem_adrp) - (start_offs >> 4);
01423   RP_SET (*real_basep, 0, rm_base);
01424   FP_SET (*prot_basep, rm_base << 4, SS_DOSMEM);
01425 
01426   /* Copy the real mode code/data to the allocated memory
01427    */
01428   source = (UCHAR *) start_offs;
01429   destin = *prot_basep;
01430   FP_SET (destin, FP_OFF(*prot_basep) + start_offs, FP_SEL(*prot_basep));
01431   len = end_offs - start_offs;
01432   WriteFarMem (destin, source, len);
01433 
01434   return (TRUE);
01435 }
