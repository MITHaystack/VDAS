00045 int32_t gmt2local(time_t t) {
00046   int dt, dir;
00047   struct tm *gmt, *loc;
00048   struct tm sgmt;
00049 
00050   if (t == 0)
00051     t = time(NULL);
00052   gmt = &sgmt;
00053   *gmt = *gmtime(&t);
00054   loc = localtime(&t);
00055   dt = (loc->tm_hour - gmt->tm_hour) * 60 * 60 +
00056         (loc->tm_min - gmt->tm_min) * 60;
00057 
00058   /*
00059    * If the year or julian day is different, we span 00:00 GMT
00060    * and must add or subtract a day. Check the year first to
00061    * avoid problems when the julian day wraps.
00062    */
00063   dir = loc->tm_year - gmt->tm_year;
00064   if (dir == 0)
00065     dir = loc->tm_yday - gmt->tm_yday;
00066   dt += dir * 24 * 60 * 60;
00067 
00068   return (dt);
00069 }
